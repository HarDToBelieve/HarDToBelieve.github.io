44struct SystemMallocStats {
45 malloc_statistics_t defaultMallocZoneStats;
46 malloc_statistics_t dispatchContinuationMallocZoneStats;
47 malloc_statistics_t purgeableMallocZoneStats;
48};

50SystemMallocStats WebMemorySampler::sampleSystemMalloc() const
51{
52 static const char* defaultMallocZoneName = "DefaultMallocZone";
53 static const char* dispatchContinuationMallocZoneName = "DispatchContinuations";
54 static const char* purgeableMallocZoneName = "DefaultPurgeableMallocZone";
55 SystemMallocStats mallocStats;
56 vm_address_t* zones;
57 unsigned count;
58
59 // Zero out the structures in case a zone is missing
60 malloc_statistics_t stats;
61 stats.blocks_in_use = 0;
62 stats.size_in_use = 0;
63 stats.max_size_in_use = 0;
64 stats.size_allocated = 0;
65 mallocStats.defaultMallocZoneStats = stats;
66 mallocStats.dispatchContinuationMallocZoneStats = stats;
67 mallocStats.purgeableMallocZoneStats = stats;
68
69 malloc_get_all_zones(mach_task_self(), 0, &zones, &count);
70 for (unsigned i = 0; i < count; i++) {
71 if (const char* name = malloc_get_zone_name(reinterpret_cast<malloc_zone_t*>(zones[i]))) {
72 stats.blocks_in_use = 0;
73 stats.size_in_use = 0;
74 stats.max_size_in_use = 0;
75 stats.size_allocated = 0;
76 malloc_zone_statistics(reinterpret_cast<malloc_zone_t*>(zones[i]), &stats);
77 if (!strcmp(name, defaultMallocZoneName))
78 mallocStats.defaultMallocZoneStats = stats;
79 else if (!strcmp(name, dispatchContinuationMallocZoneName))
80 mallocStats.dispatchContinuationMallocZoneStats = stats;
81 else if (!strcmp(name, purgeableMallocZoneName))
82 mallocStats.purgeableMallocZoneStats = stats;
83 }
84 }
85 return mallocStats;
86}

88size_t WebMemorySampler::sampleProcessCommittedBytes() const
89{
90 task_basic_info_64 taskInfo;
91 mach_msg_type_number_t count = TASK_BASIC_INFO_64_COUNT;
92 task_info(mach_task_self(), TASK_BASIC_INFO_64, reinterpret_cast<task_info_t>(&taskInfo), &count);
93 return taskInfo.resident_size;
94}

96String WebMemorySampler::processName() const
97{
98 NSString *appName = [[NSBundle mainBundle] bundleIdentifier];
99 if (!appName)
100 appName = [[NSProcessInfo processInfo] processName];
101 return String(appName);
102}

104WebMemoryStatistics WebMemorySampler::sampleWebKit() const
105{
106 size_t totalBytesInUse = 0, totalBytesCommitted = 0;
107
108 WebMemoryStatistics webKitMemoryStats;
109
110 auto fastMallocStatistics = WTF::fastMallocStatistics();
111 size_t fastMallocBytesInUse = fastMallocStatistics.committedVMBytes - fastMallocStatistics.freeListBytes;
112 size_t fastMallocBytesCommitted = fastMallocStatistics.committedVMBytes;
113 totalBytesInUse += fastMallocBytesInUse;
114 totalBytesCommitted += fastMallocBytesCommitted;
115
116 JSC::JSLockHolder lock(WebCore::commonVM());
117 size_t jscHeapBytesInUse = WebCore::commonVM().heap.size();
118 size_t jscHeapBytesCommitted = WebCore::commonVM().heap.capacity();
119 totalBytesInUse += jscHeapBytesInUse;
120 totalBytesCommitted += jscHeapBytesCommitted;
121
122 JSC::GlobalMemoryStatistics globalMemoryStats = JSC::globalMemoryStatistics();
123 totalBytesInUse += globalMemoryStats.stackBytes + globalMemoryStats.JITBytes;
124 totalBytesCommitted += globalMemoryStats.stackBytes + globalMemoryStats.JITBytes;
125
126 SystemMallocStats systemStats = sampleSystemMalloc();
127
128 size_t defaultMallocZoneBytesInUse = systemStats.defaultMallocZoneStats.size_in_use;
129 size_t dispatchContinuationMallocZoneBytesInUse = systemStats.dispatchContinuationMallocZoneStats.size_in_use;
130 size_t purgeableMallocZoneBytesInUse = systemStats.purgeableMallocZoneStats.size_in_use;
131 size_t defaultMallocZoneBytesCommitted = systemStats.defaultMallocZoneStats.size_allocated;
132 size_t dispatchContinuationMallocZoneBytesCommitted = systemStats.dispatchContinuationMallocZoneStats.size_allocated;
133 size_t purgeableMallocZoneBytesCommitted = systemStats.purgeableMallocZoneStats.size_allocated;
134 totalBytesInUse += defaultMallocZoneBytesInUse + dispatchContinuationMallocZoneBytesInUse + purgeableMallocZoneBytesInUse;
135 totalBytesCommitted += defaultMallocZoneBytesCommitted + dispatchContinuationMallocZoneBytesCommitted + purgeableMallocZoneBytesCommitted;
136
137 size_t residentSize = sampleProcessCommittedBytes();
138
139 WallTime now = WallTime::now();
140
141 webKitMemoryStats.keys.append("Timestamp"_s);
142 webKitMemoryStats.values.append(now.secondsSinceEpoch().seconds());
143 webKitMemoryStats.keys.append("Total Bytes of Memory In Use"_s);
144 webKitMemoryStats.values.append(totalBytesInUse);
145 webKitMemoryStats.keys.append("Fast Malloc Zone Bytes"_s);
146 webKitMemoryStats.values.append(fastMallocBytesInUse);
147 webKitMemoryStats.keys.append("Default Malloc Zone Bytes"_s);
148 webKitMemoryStats.values.append(defaultMallocZoneBytesInUse);
149 webKitMemoryStats.keys.append("Dispatch Continuation Malloc Zone Bytes"_s);
150 webKitMemoryStats.values.append(dispatchContinuationMallocZoneBytesInUse);
151 webKitMemoryStats.keys.append("Purgeable Malloc Zone Bytes"_s);
152 webKitMemoryStats.values.append(purgeableMallocZoneBytesInUse);
153 webKitMemoryStats.keys.append("JavaScript Heap Bytes"_s);
154 webKitMemoryStats.values.append(jscHeapBytesInUse);
155 webKitMemoryStats.keys.append("Total Bytes of Committed Memory"_s);
156 webKitMemoryStats.values.append(totalBytesCommitted);
157 webKitMemoryStats.keys.append("Fast Malloc Zone Bytes"_s);
158 webKitMemoryStats.values.append(fastMallocBytesCommitted);
159 webKitMemoryStats.keys.append("Default Malloc Zone Bytes"_s);
160 webKitMemoryStats.values.append(defaultMallocZoneBytesCommitted);
161 webKitMemoryStats.keys.append("Dispatch Continuation Malloc Zone Bytes"_s);
162 webKitMemoryStats.values.append(dispatchContinuationMallocZoneBytesCommitted);
163 webKitMemoryStats.keys.append("Purgeable Malloc Zone Bytes"_s);
164 webKitMemoryStats.values.append(purgeableMallocZoneBytesCommitted);
165 webKitMemoryStats.keys.append("JavaScript Heap Bytes"_s);
166 webKitMemoryStats.values.append(jscHeapBytesCommitted);
167 webKitMemoryStats.keys.append("JavaScript Stack Bytes"_s);
168 webKitMemoryStats.values.append(globalMemoryStats.stackBytes);
169 webKitMemoryStats.keys.append("JavaScript JIT Bytes"_s);
170 webKitMemoryStats.values.append(globalMemoryStats.JITBytes);
171 webKitMemoryStats.keys.append("Resident Size"_s);
172 webKitMemoryStats.values.append(residentSize);
173
174 return webKitMemoryStats;
175}

177void WebMemorySampler::sendMemoryPressureEvent()
178{
179 // Free memory that could be released if we needed more.
180 // We want to track memory that cannot.
181 notify_post("org.WebKit.lowMemory");
182}
