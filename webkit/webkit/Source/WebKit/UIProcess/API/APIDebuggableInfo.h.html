34class DebuggableInfo final : public ObjectImpl<Object::Type::DebuggableInfo> {
35public:
36 static Ref<DebuggableInfo> create(const WebKit::DebuggableInfoData&);
37 DebuggableInfo() = default;
38 virtual ~DebuggableInfo();
39
40 Inspector::DebuggableType debuggableType() const { return m_data.debuggableType; }
41 void setDebuggableType(Inspector::DebuggableType debuggableType) { m_data.debuggableType = debuggableType; }
42
43 const WTF::String& targetPlatformName() const { return m_data.targetPlatformName; }
44 void setTargetPlatformName(const WTF::String& targetPlatformName) { m_data.targetPlatformName = targetPlatformName; }
45
46 const WTF::String& targetBuildVersion() const { return m_data.targetBuildVersion; }
47 void setTargetBuildVersion(const WTF::String& targetBuildVersion) { m_data.targetBuildVersion = targetBuildVersion; }
48
49 const WTF::String& targetProductVersion() const { return m_data.targetProductVersion; }
50 void setTargetProductVersion(const WTF::String& targetProductVersion) { m_data.targetProductVersion = targetProductVersion; }
51
52 bool targetIsSimulator() const { return m_data.targetIsSimulator; }
53 void setTargetIsSimulator(bool targetIsSimulator) { m_data.targetIsSimulator = targetIsSimulator; }
54
55 const WebKit::DebuggableInfoData& debuggableInfoData() const { return m_data; }
56
57private:
58 explicit DebuggableInfo(const WebKit::DebuggableInfoData&);
59
60 WebKit::DebuggableInfoData m_data;
61};
