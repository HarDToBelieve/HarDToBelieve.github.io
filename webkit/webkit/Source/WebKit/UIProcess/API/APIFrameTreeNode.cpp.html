35
36Ref<FrameHandle> FrameTreeNode::handle() const
37{
38 return FrameHandle::create(m_data.info.frameID ? *m_data.info.frameID : WebCore::FrameIdentifier { });
39}
40
41RefPtr<FrameHandle> FrameTreeNode::parentFrameHandle() const
42{
43 return m_data.info.parentFrameID ? RefPtr<FrameHandle>(FrameHandle::create(*m_data.info.parentFrameID)) : nullptr;
44}
