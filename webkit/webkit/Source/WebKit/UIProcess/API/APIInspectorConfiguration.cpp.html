34Ref<InspectorConfiguration> InspectorConfiguration::create()
35{
36 return adoptRef(*new InspectorConfiguration);
37}

39void InspectorConfiguration::addURLSchemeHandler(Ref<WebKit::WebURLSchemeHandler>&& urlSchemeHandler, const WTF::String& urlScheme)
40{
41 m_customURLSchemes.append(std::make_pair(WTFMove(urlSchemeHandler), urlScheme));
42}
43
44WebKit::WebProcessPool* InspectorConfiguration::processPool()
45{
46 return m_processPool.get();
47}
48
49void InspectorConfiguration::setProcessPool(WebKit::WebProcessPool* processPool)
50{
51 m_processPool = processPool;
52}
