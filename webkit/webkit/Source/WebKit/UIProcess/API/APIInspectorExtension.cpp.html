38InspectorExtension::InspectorExtension(const WTF::String& identifier, WebKit::WebInspectorUIExtensionControllerProxy& extensionControllerProxy)
39 : m_identifier(identifier)
40 , m_extensionControllerProxy(makeWeakPtr(extensionControllerProxy))
41{
42}
43
44Ref<InspectorExtension> InspectorExtension::create(const WTF::String& identifier, WebKit::WebInspectorUIExtensionControllerProxy& extensionControllerProxy)
45{
46 return adoptRef(*new InspectorExtension(identifier, extensionControllerProxy));
47}
48
49void InspectorExtension::setClient(UniqueRef<InspectorExtensionClient>&& client)
50{
51 m_client = client.moveToUniquePtr();
52}
53
54void InspectorExtension::createTab(const WTF::String& tabName, const WTF::URL& tabIconURL, const WTF::URL& sourceURL, WTF::CompletionHandler<void(Expected<Inspector::ExtensionTabID, Inspector::ExtensionError>)>&& completionHandler)
55{
56 if (!m_extensionControllerProxy) {
57 completionHandler(makeUnexpected(Inspector::ExtensionError::ContextDestroyed));
58 return;
59 }
60
61 m_extensionControllerProxy->createTabForExtension(m_identifier, tabName, tabIconURL, sourceURL, WTFMove(completionHandler));
62}
63
64void InspectorExtension::evaluateScript(const WTF::String& scriptSource, const std::optional<WTF::URL>& frameURL, const std::optional<WTF::URL>& contextSecurityOrigin, const std::optional<bool>& useContentScriptContext, WTF::CompletionHandler<void(Inspector::ExtensionEvaluationResult)>&& completionHandler)
65{
66 if (!m_extensionControllerProxy) {
67 completionHandler(makeUnexpected(Inspector::ExtensionError::ContextDestroyed));
68 return;
69 }
70
71 m_extensionControllerProxy->evaluateScriptForExtension(m_identifier, scriptSource, frameURL, contextSecurityOrigin, useContentScriptContext, WTFMove(completionHandler));
72}

74void InspectorExtension::reloadIgnoringCache(const std::optional<bool>& ignoreCache, const std::optional<WTF::String>& userAgent, const std::optional<WTF::String>& injectedScript, WTF::CompletionHandler<void(Inspector::ExtensionEvaluationResult)>&& completionHandler)
75{
76 if (!m_extensionControllerProxy) {
77 completionHandler(makeUnexpected(Inspector::ExtensionError::ContextDestroyed));
78 return;
79 }
80
81 m_extensionControllerProxy->reloadForExtension(m_identifier, ignoreCache, userAgent, injectedScript, WTFMove(completionHandler));
82}
