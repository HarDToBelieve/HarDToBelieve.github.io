60Navigation::Navigation(WebNavigationState& state, WebCore::ResourceRequest&& request, WebBackForwardListItem* fromItem)
61 : m_navigationID(state.generateNavigationID())
62 , m_originalRequest(WTFMove(request))
63 , m_currentRequest(m_originalRequest)
64 , m_fromItem(fromItem)
65 , m_clientNavigationActivity(navigationActivityTimeout)
66{
67 m_redirectChain.append(m_originalRequest.url());
68}

70Navigation::Navigation(WebNavigationState& state, WebBackForwardListItem& targetItem, WebBackForwardListItem* fromItem, FrameLoadType backForwardFrameLoadType)
71 : m_navigationID(state.generateNavigationID())
72 , m_originalRequest(targetItem.url())
73 , m_currentRequest(m_originalRequest)
74 , m_targetItem(&targetItem)
75 , m_fromItem(fromItem)
76 , m_backForwardFrameLoadType(backForwardFrameLoadType)
77 , m_clientNavigationActivity(navigationActivityTimeout)

99void Navigation::setCurrentRequest(ResourceRequest&& request, ProcessIdentifier processIdentifier)
100{
101 m_currentRequest = WTFMove(request);
102 m_currentRequestProcessIdentifier = processIdentifier;
103}
104
105void Navigation::appendRedirectionURL(const WTF::URL& url)
106{
107 if (m_redirectChain.isEmpty() || m_redirectChain.last() != url)
108 m_redirectChain.append(url);
109}
