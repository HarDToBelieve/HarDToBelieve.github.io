26#ifndef APINavigationData_h
27#define APINavigationData_h
28
29#include "APIObject.h"
30#include "WebNavigationDataStore.h"
31
32namespace API {
33
34class NavigationData : public ObjectImpl<Object::Type::NavigationData> {
35public:
36 static Ref<NavigationData> create(const WebKit::WebNavigationDataStore& store)
37 {
38 return adoptRef(*new NavigationData(store));
39 }
40
41 virtual ~NavigationData();
42
43 WTF::String title() const { return m_store.title; }
44 WTF::String url() const { return m_store.url; }
45 const WebCore::ResourceRequest& originalRequest() const { return m_store.originalRequest; }
46 const WebCore::ResourceResponse& response() const { return m_store.response; }
47
48private:
49 explicit NavigationData(const WebKit::WebNavigationDataStore&);
50
51 WebKit::WebNavigationDataStore m_store;
52};
53
54} // namespace API
55
56#endif // APINavigationData_h
