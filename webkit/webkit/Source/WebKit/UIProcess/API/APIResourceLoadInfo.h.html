26#pragma once
27
28#include "APIObject.h"
29#include "ResourceLoadInfo.h"
30
31namespace API {
32
33class ResourceLoadInfo final : public ObjectImpl<Object::Type::ResourceLoadInfo> {
34public:
35 static Ref<ResourceLoadInfo> create(WebKit::ResourceLoadInfo&& info)
36 {
37 return adoptRef(*new ResourceLoadInfo(WTFMove(info)));
38 }
39
40 explicit ResourceLoadInfo(WebKit::ResourceLoadInfo&& info)
41 : m_info(WTFMove(info)) { }
42
43 WebKit::NetworkResourceLoadIdentifier resourceLoadID() const { return m_info.resourceLoadID; }
44 std::optional<WebCore::FrameIdentifier> frameID() const { return m_info.frameID; }
45 std::optional<WebCore::FrameIdentifier> parentFrameID() const { return m_info.parentFrameID; }
46 const WTF::URL& originalURL() const { return m_info.originalURL; }
47 const WTF::String& originalHTTPMethod() const { return m_info.originalHTTPMethod; }
48 WallTime eventTimestamp() const { return m_info.eventTimestamp; }
49 bool loadedFromCache() const { return m_info.loadedFromCache; }
50 WebKit::ResourceLoadInfo::Type resourceLoadType() const { return m_info.type; }
51
52private:
53 const WebKit::ResourceLoadInfo m_info;
54};
55
56} // namespace API
