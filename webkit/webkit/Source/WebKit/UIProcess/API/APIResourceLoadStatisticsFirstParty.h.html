27
28#include "APIObject.h"
29#include "WebResourceLoadStatisticsStore.h"
30#include <wtf/text/WTFString.h>
31
32namespace API {
33
34class ResourceLoadStatisticsFirstParty final : public ObjectImpl<Object::Type::ResourceLoadStatisticsFirstParty> {
35public:
36 static Ref<ResourceLoadStatisticsFirstParty> create(const WebKit::WebResourceLoadStatisticsStore::ThirdPartyDataForSpecificFirstParty& firstPartyData)
37 {
38 RELEASE_ASSERT(RunLoop::isMain());
39 return adoptRef(*new ResourceLoadStatisticsFirstParty(firstPartyData));
40 }
41
42 ~ResourceLoadStatisticsFirstParty()
43 {
44 RELEASE_ASSERT(RunLoop::isMain());
45 }
46
47 const WTF::String& firstPartyDomain() const { return m_firstPartyData.firstPartyDomain.string(); }
48 bool storageAccess() const { return m_firstPartyData.storageAccessGranted; }
49 double timeLastUpdated() const { return m_firstPartyData.timeLastUpdated.value(); }
50
51private:
52 explicit ResourceLoadStatisticsFirstParty(const WebKit::WebResourceLoadStatisticsStore::ThirdPartyDataForSpecificFirstParty& firstPartyData)
53 : m_firstPartyData(firstPartyData)
54 {
55 }
56
57 const WebKit::WebResourceLoadStatisticsStore::ThirdPartyDataForSpecificFirstParty m_firstPartyData;
58};
59
60} // namespace API
