27
28#include "APIObject.h"
29#include "WebResourceLoadStatisticsStore.h"
30#include <wtf/text/WTFString.h>
31
32namespace API {
33
34class ResourceLoadStatisticsThirdParty final : public ObjectImpl<Object::Type::ResourceLoadStatisticsThirdParty> {
35public:
36 static Ref<ResourceLoadStatisticsThirdParty> create(WebKit::WebResourceLoadStatisticsStore::ThirdPartyData&& thirdPartyData)
37 {
38 RELEASE_ASSERT(RunLoop::isMain());
39 return adoptRef(*new ResourceLoadStatisticsThirdParty(WTFMove(thirdPartyData)));
40 }
41
42 ~ResourceLoadStatisticsThirdParty()
43 {
44 RELEASE_ASSERT(RunLoop::isMain());
45 }
46
47 const WTF::String& thirdPartyDomain() const { return m_thirdPartyData.thirdPartyDomain.string(); }
48 const Vector<WebKit::WebResourceLoadStatisticsStore::ThirdPartyDataForSpecificFirstParty>& underFirstParties() const { return m_thirdPartyData.underFirstParties; }
49
50private:
51 explicit ResourceLoadStatisticsThirdParty(WebKit::WebResourceLoadStatisticsStore::ThirdPartyData&& thirdPartyData)
52 : m_thirdPartyData(WTFMove(thirdPartyData))
53 {
54 }
55
56 const WebKit::WebResourceLoadStatisticsStore::ThirdPartyData m_thirdPartyData;
57};
58
59} // namespace API
