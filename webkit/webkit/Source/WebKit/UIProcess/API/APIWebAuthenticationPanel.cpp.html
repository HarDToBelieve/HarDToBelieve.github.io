40Ref<WebAuthenticationPanel> WebAuthenticationPanel::create(const AuthenticatorManager& manager, const WTF::String& rpId, const TransportSet& transports, ClientDataType type, const WTF::String& userName)
41{
42 return adoptRef(*new WebAuthenticationPanel(manager, rpId, transports, type, userName));
43}
44
45WebAuthenticationPanel::WebAuthenticationPanel()
46 : m_manager(makeUnique<AuthenticatorManager>())
47 , m_client(makeUniqueRef<WebAuthenticationPanelClient>())
48{
49 m_manager->enableNativeSupport();
50}
51
52WebAuthenticationPanel::WebAuthenticationPanel(const AuthenticatorManager& manager, const WTF::String& rpId, const TransportSet& transports, ClientDataType type, const WTF::String& userName)
53 : m_client(makeUniqueRef<WebAuthenticationPanelClient>())
54 , m_weakManager(makeWeakPtr(manager))
55 , m_rpId(rpId)
56 , m_clientDataType(type)
57 , m_userName(userName)
58{
59 m_transports = Vector<AuthenticatorTransport>();
60 m_transports.reserveInitialCapacity(AuthenticatorManager::maxTransportNumber);
61 if (transports.contains(AuthenticatorTransport::Usb))
62 m_transports.uncheckedAppend(AuthenticatorTransport::Usb);
63 if (transports.contains(AuthenticatorTransport::Nfc))
64 m_transports.uncheckedAppend(AuthenticatorTransport::Nfc);
65 if (transports.contains(AuthenticatorTransport::Internal))
66 m_transports.uncheckedAppend(AuthenticatorTransport::Internal);
67}
68
69WebAuthenticationPanel::~WebAuthenticationPanel() = default;
70
71void WebAuthenticationPanel::handleRequest(WebAuthenticationRequestData&& request, Callback&& callback)
72{
73 ASSERT(m_manager);
74 request.weakPanel = makeWeakPtr(*this);
75 m_manager->handleRequest(WTFMove(request), WTFMove(callback));
76}
77
78void WebAuthenticationPanel::cancel() const
79{
80 if (m_weakManager) {
81 m_weakManager->cancelRequest(*this);
82 return;
83 }
84
85 m_manager->cancel();
86}
87
88void WebAuthenticationPanel::setMockConfiguration(WebCore::MockWebAuthenticationConfiguration&& configuration)
89{
90 ASSERT(m_manager);
91
92 if (!m_manager->isMock()) {
93 m_manager = makeUnique<MockAuthenticatorManager>(WTFMove(configuration));
94 m_manager->enableNativeSupport();
95 return;
96 }
97 static_cast<MockAuthenticatorManager*>(m_manager.get())->setTestConfiguration(WTFMove(configuration));
98}
99
100void WebAuthenticationPanel::setClient(UniqueRef<WebAuthenticationPanelClient>&& client)
101{
102 m_client = WTFMove(client);
103}
