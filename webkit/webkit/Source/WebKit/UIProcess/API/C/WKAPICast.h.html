233inline WKProcessTerminationReason toAPI(ProcessTerminationReason reason)
234{
235 switch (reason) {
236 case ProcessTerminationReason::ExceededMemoryLimit:
237 return kWKProcessTerminationReasonExceededMemoryLimit;
238 case ProcessTerminationReason::ExceededCPULimit:
239 return kWKProcessTerminationReasonExceededCPULimit;
240 case ProcessTerminationReason::NavigationSwap:
241 // We probably shouldn't bother coming up with a new C-API type for process-swapping.
242 // "Requested by client" seems like the best match for existing types.
243 FALLTHROUGH;
244 case ProcessTerminationReason::RequestedByClient:
245 return kWKProcessTerminationReasonRequestedByClient;
246 case ProcessTerminationReason::RequestedByNetworkProcess:
247 case ProcessTerminationReason::RequestedByGPUProcess:
248 case ProcessTerminationReason::Crash:
249 return kWKProcessTerminationReasonCrash;
250 }
251
252 return kWKProcessTerminationReasonCrash;
253}
