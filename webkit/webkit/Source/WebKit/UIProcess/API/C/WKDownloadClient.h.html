31typedef bool (*WKDownloadWillPerformHTTPRedirection)(WKDownloadRef download, WKURLResponseRef response, WKURLRequestRef newRequest, const void* clientInfo);
32typedef void (*WKDownloadDidReceiveAuthenticationChallenge)(WKDownloadRef download, WKAuthenticationChallengeRef challenge, const void* clientInfo);
33typedef WKStringRef (*WKDownloadDecideDestinationWithResponse)(WKDownloadRef download, WKURLResponseRef response, WKStringRef suggestedFilename, const void* clientInfo);
34typedef void (*WKDownloadDidWriteData)(WKDownloadRef download, long long bytesWritten, long long totalBytesWritten, long long totalBytesExpectedToWrite, const void* clientInfo);
35typedef void (*WKDownloadDidFinish)(WKDownloadRef download, const void* clientInfo);

50 WKDownloadDidWriteData didWriteData;
51 WKDownloadDidFinish didFinish;
52 WKDownloadDidFailWithError didFailWithError;
53} WKDownloadClientV0;
