74void WKDownloadSetClient(WKDownloadRef download, WKDownloadClientBase* client)
75{
76 class DownloadClient : public API::Client<WKDownloadClientBase>, public API::DownloadClient {
77 public:
78 explicit DownloadClient(const WKDownloadClientBase* client)
79 {
80 initialize(client);
81 }
82
83 private:
84
85 void didReceiveAuthenticationChallenge(WebKit::DownloadProxy& download, WebKit::AuthenticationChallengeProxy& challenge) override
86 {
87 if (!m_client.didReceiveAuthenticationChallenge) {
88 challenge.listener().completeChallenge(WebKit::AuthenticationChallengeDisposition::RejectProtectionSpaceAndContinue);
89 return;
90 }
91 m_client.didReceiveAuthenticationChallenge(toAPI(download), toAPI(challenge), m_client.base.clientInfo);
92 }
93
94 void didReceiveData(WebKit::DownloadProxy& download, uint64_t bytesWritten, uint64_t totalBytesWritten, uint64_t totalBytesExpectedToWrite) override
95 {
96 if (!m_client.didWriteData)
97 return;
98 m_client.didWriteData(toAPI(download), bytesWritten, totalBytesWritten, totalBytesExpectedToWrite, m_client.base.clientInfo);
99
100 }
101
102 void decideDestinationWithSuggestedFilename(WebKit::DownloadProxy& download, const WebCore::ResourceResponse& response, const WTF::String& suggestedFilename, CompletionHandler<void(WebKit::AllowOverwrite, WTF::String)>&& completionHandler) override
103 {
104 if (!m_client.decideDestinationWithResponse) {
105 completionHandler(WebKit::AllowOverwrite::No, { });
106 return;
107 }
108 API::String* destination = toImpl(m_client.decideDestinationWithResponse(toAPI(download), toAPI(response), toAPI(suggestedFilename.impl()), m_client.base.clientInfo));
109 if (!destination) {
110 completionHandler(WebKit::AllowOverwrite::No, { });
111 return;
112 }
113 completionHandler(WebKit::AllowOverwrite::No, destination->string());
114 }
115
116 void didFinish(WebKit::DownloadProxy& download) override
117 {
118 if (!m_client.didFinish)
119 return;
120 m_client.didFinish(toAPI(download), m_client.base.clientInfo);
121 }
122
123 void didFail(WebKit::DownloadProxy& download, const WebCore::ResourceError& error, API::Data* resumeData) override
124 {
125 if (!m_client.didFailWithError)
126 return;
127 m_client.didFailWithError(toAPI(download), toAPI(error), toAPI(resumeData), m_client.base.clientInfo);
128 }
129
130 void processDidCrash(WebKit::DownloadProxy& download) override
131 {
132 didFail(download, WebCore::ResourceError { WebCore::errorDomainWebKitInternal, 0, download.request().url(), "Network process crashed during download" }, nullptr);
133 }
134
135 void willSendRequest(WebKit::DownloadProxy& download, WebCore::ResourceRequest&& request, const WebCore::ResourceResponse& response, CompletionHandler<void(WebCore::ResourceRequest&&)>&& completionHandler) override
136 {
137 if (!m_client.willPerformHTTPRedirection) {
138 completionHandler(WTFMove(request));
139 return;
140 }
141 if (!m_client.willPerformHTTPRedirection(toAPI(download), toAPI(response), toAPI(request), m_client.base.clientInfo)) {
142 completionHandler({ });
143 return;
144 }
145 completionHandler(WTFMove(request));
146 }
147 };
148
149 toImpl(download)->setClient(adoptRef(*new DownloadClient(client)));
150}
