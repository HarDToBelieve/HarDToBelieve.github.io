33using namespace WebKit;
34
35WKTypeID WKNavigationDataGetTypeID()
36{
37 return toAPI(API::NavigationData::APIType);
38}
39
40WKStringRef WKNavigationDataCopyTitle(WKNavigationDataRef navigationDataRef)
41{
42 return toCopiedAPI(toImpl(navigationDataRef)->title());
43}
44
45WKURLRef WKNavigationDataCopyURL(WKNavigationDataRef navigationDataRef)
46{
47 // This returns the URL of the original request for backwards-compatibility purposes.
48 return toCopiedURLAPI(toImpl(navigationDataRef)->originalRequest().url());
49}
50
51WKURLRef WKNavigationDataCopyNavigationDestinationURL(WKNavigationDataRef navigationDataRef)
52{
53 return toCopiedURLAPI(toImpl(navigationDataRef)->url());
54}
55
56WKURLRequestRef WKNavigationDataCopyOriginalRequest(WKNavigationDataRef navigationData)
57{
58 return toAPI(&API::URLRequest::create(toImpl(navigationData)->originalRequest()).leakRef());
59}
