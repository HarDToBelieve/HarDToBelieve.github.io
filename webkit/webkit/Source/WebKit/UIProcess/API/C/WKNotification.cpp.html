67WKStringRef WKNotificationCopyDir(WKNotificationRef notification)
68{
69 switch (toImpl(notification)->dir()) {
70 case WebCore::NotificationDirection::Auto:
71 return WKStringCreateWithUTF8CString("auto");
72 case WebCore::NotificationDirection::Ltr:
73 return WKStringCreateWithUTF8CString("ltr");
74 case WebCore::NotificationDirection::Rtl:
75 return WKStringCreateWithUTF8CString("rtl");
76 }
77
78 RELEASE_ASSERT_NOT_REACHED();
79}
