29#include <WebKit/WKBase.h>

33extern "C" {

36typedef void (*WKPageLogDiagnosticMessageCallback)(WKPageRef page, WKStringRef message, WKStringRef description, const void* clientInfo);
37typedef void (*WKPageLogDiagnosticMessageWithResultCallback)(WKPageRef page, WKStringRef message, WKStringRef description, WKDiagnosticLoggingResultType result, const void* clientInfo);
38typedef void (*WKPageLogDiagnosticMessageWithValueCallback)(WKPageRef page, WKStringRef message, WKStringRef description, WKStringRef value, const void* clientInfo);
39typedef void (*WKPageLogDiagnosticMessageWithEnhancedPrivacyCallback)(WKPageRef page, WKStringRef message, WKStringRef description, const void* clientInfo);
40
41typedef struct WKPageDiagnosticLoggingClientBase {
42 int version;
43 const void * clientInfo;
44} WKPageDiagnosticLoggingClientBase;
45
46typedef struct WKPageDiagnosticLoggingClientV0 {
47 WKPageDiagnosticLoggingClientBase base;
48
49 // Version 0.
50 WKPageLogDiagnosticMessageCallback logDiagnosticMessage;
51 WKPageLogDiagnosticMessageWithResultCallback logDiagnosticMessageWithResult;
52 WKPageLogDiagnosticMessageWithValueCallback logDiagnosticMessageWithValue;
53} WKPageDiagnosticLoggingClientV0;
54
55typedef struct WKPageDiagnosticLoggingClientV1 {
56 WKPageDiagnosticLoggingClientBase base;
57
58 // Version 0.
59 WKPageLogDiagnosticMessageCallback logDiagnosticMessage;
60 WKPageLogDiagnosticMessageWithResultCallback logDiagnosticMessageWithResult;
61 WKPageLogDiagnosticMessageWithValueCallback logDiagnosticMessageWithValue;
62
63 // Version 1.
64 WKPageLogDiagnosticMessageWithEnhancedPrivacyCallback logDiagnosticMessageWithEnhancedPrivacy;
65} WKPageDiagnosticLoggingClientV1;
