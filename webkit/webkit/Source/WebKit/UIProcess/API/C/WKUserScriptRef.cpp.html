32using namespace WebKit;
33
34WKTypeID WKUserScriptGetTypeID()
35{
36 return toAPI(API::UserScript::APIType);
37}
38
39WKUserScriptRef WKUserScriptCreateWithSource(WKStringRef sourceRef, _WKUserScriptInjectionTime injectionTime, bool forMainFrameOnly)
40{
41 return toAPI(&API::UserScript::create(WebCore::UserScript { toWTFString(sourceRef), API::UserScript::generateUniqueURL(), { }, { }, toUserScriptInjectionTime(injectionTime), forMainFrameOnly ? WebCore::UserContentInjectedFrames::InjectInTopFrameOnly : WebCore::UserContentInjectedFrames::InjectInAllFrames, WebCore::WaitForNotificationBeforeInjecting::No }, API::ContentWorld::pageContentWorld()).leakRef());
42}
43
44WKStringRef WKUserScriptCopySource(WKUserScriptRef userScriptRef)
45{
46 return toCopiedAPI(toImpl(userScriptRef)->userScript().source());
47}
48
49_WKUserScriptInjectionTime WKUserScriptGetInjectionTime(WKUserScriptRef userScriptRef)
50{
51 return toWKUserScriptInjectionTime(toImpl(userScriptRef)->userScript().injectionTime());
52}
53
54bool WKUserScriptGetMainFrameOnly(WKUserScriptRef userScriptRef)
55{
56 return toImpl(userScriptRef)->userScript().injectedFrames() == WebCore::UserContentInjectedFrames::InjectInTopFrameOnly;
57}
