83void WKWebsiteDataStoreRemoveITPDataForDomain(WKWebsiteDataStoreRef dataStoreRef, WKStringRef host, void* context, WKWebsiteDataStoreRemoveITPDataForDomainFunction callback)
84{
85#if ENABLE(RESOURCE_LOAD_STATISTICS)
86 WebKit::WebsiteDataRecord dataRecord;
87 dataRecord.types.add(WebKit::WebsiteDataType::ResourceLoadStatistics);
88 dataRecord.addResourceLoadStatisticsRegistrableDomain(WebCore::RegistrableDomain::uncheckedCreateFromHost(WebKit::toImpl(host)->string()));
89 Vector<WebKit::WebsiteDataRecord> dataRecords = { WTFMove(dataRecord) };
90
91 OptionSet<WebKit::WebsiteDataType> dataTypes = WebKit::WebsiteDataType::ResourceLoadStatistics;
92 WebKit::toImpl(dataStoreRef)->removeData(dataTypes, dataRecords, [context, callback] {
93 callback(context);
94 });
95#else
96 callback(context);
97#endif
98}
