59WKDictionaryRef WKContextCopyPlugInInfoForBundleIdentifier(WKContextRef contextRef, WKStringRef plugInBundleIdentifierRef)
60{
61#if ENABLE(NETSCAPE_PLUGIN_API)
62 WebKit::PluginModuleInfo plugin = WebKit::toImpl(contextRef)->pluginInfoStore().findPluginWithBundleIdentifier(WebKit::toWTFString(plugInBundleIdentifierRef));
63 if (plugin.path.isNull())
64 return 0;
65
66 auto dictionary = createPluginInformationDictionary(plugin);
67 return WebKit::toAPI(&dictionary.leakRef());
68#else
69 return 0;
70#endif
71}

73void WKContextGetInfoForInstalledPlugIns(WKContextRef contextRef, WKContextGetInfoForInstalledPlugInsBlock block)
74{
75#if ENABLE(NETSCAPE_PLUGIN_API)
76 Vector<WebKit::PluginModuleInfo> plugins = WebKit::toImpl(contextRef)->pluginInfoStore().plugins();
77
78 Vector<RefPtr<API::Object>> pluginInfoDictionaries;
79 pluginInfoDictionaries.reserveInitialCapacity(plugins.size());
80
81 for (const auto& plugin: plugins)
82 pluginInfoDictionaries.uncheckedAppend(createPluginInformationDictionary(plugin));
83
84 RefPtr<API::Array> array = API::Array::create(WTFMove(pluginInfoDictionaries));
85
86 WebKit::toImpl(contextRef)->ref();
87 RunLoop::main().dispatch([block = makeBlockPtr(block), array = WTFMove(array), contextRef] {
88 block(WebKit::toAPI(array.get()), 0);
89
90 WebKit::toImpl(contextRef)->deref();
91 });
92#endif
93}
