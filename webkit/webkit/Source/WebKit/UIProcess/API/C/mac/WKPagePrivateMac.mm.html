42@interface WKObservablePageState : NSObject <_WKObservablePageState> {
43 RefPtr<WebKit::WebPageProxy> _page;
44 std::unique_ptr<WebKit::PageLoadStateObserver> _observer;
45}
46
47@end
48
49@implementation WKObservablePageState
50
51- (id)initWithPage:(RefPtr<WebKit::WebPageProxy>&&)page
52{
53 if (!(self = [super init]))
54 return nil;
55
56 _page = WTFMove(page);
57 _observer = makeUnique<WebKit::PageLoadStateObserver>(self, @"URL");
58 _page->pageLoadState().addObserver(*_observer);
59
60 return self;
61}
62
63- (void)dealloc
64{
65 _observer->clearObject();
66
67 ensureOnMainRunLoop([page = WTFMove(_page), observer = WTFMove(_observer)] {
68 page->pageLoadState().removeObserver(*observer);
69 });
70
71 [super dealloc];
72}
73
74- (BOOL)isLoading
75{
76 return _page->pageLoadState().isLoading();
77}
78
79- (NSString *)title
80{
81 return _page->pageLoadState().title();
82}
83
84- (NSURL *)URL
85{
86 return [NSURL _web_URLWithWTFString:_page->pageLoadState().activeURL()];
87}
88
89- (BOOL)hasOnlySecureContent
90{
91 return _page->pageLoadState().hasOnlySecureContent();
92}
93
94- (BOOL)_webProcessIsResponsive
95{
96 return _page->process().isResponsive();
97}
98
99- (double)estimatedProgress
100{
101 return _page->estimatedProgress();
102}
103
104- (NSURL *)unreachableURL
105{
106 return [NSURL _web_URLWithWTFString:_page->pageLoadState().unreachableURL()];
107}
108
109- (SecTrustRef)serverTrust
110{
111#if HAVE(SEC_TRUST_SERIALIZATION)
112 auto certificateInfo = _page->pageLoadState().certificateInfo();
113 if (!certificateInfo)
114 return nil;
115
116 return certificateInfo->certificateInfo().trust();
117#else
118 return nil;
119#endif
120}
121
122@end

124id <_WKObservablePageState> WKPageCreateObservableState(WKPageRef pageRef)
125{
126 return [[WKObservablePageState alloc] initWithPage:WebKit::toImpl(pageRef)];
127}

138bool WKPageIsURLKnownHSTSHost(WKPageRef page, WKURLRef url)
139{
140 WebKit::WebPageProxy* webPageProxy = WebKit::toImpl(page);
141
142 return webPageProxy->process().processPool().isURLKnownHSTSHost(WebKit::toImpl(url)->string());
143}

145WKNavigation *WKPageLoadURLRequestReturningNavigation(WKPageRef pageRef, WKURLRequestRef urlRequestRef)
146{
147 auto resourceRequest = WebKit::toImpl(urlRequestRef)->resourceRequest();
148 return WebKit::wrapper(WebKit::toImpl(pageRef)->loadRequest(WTFMove(resourceRequest)));
149}
150
151WKNavigation *WKPageLoadFileReturningNavigation(WKPageRef pageRef, WKURLRef fileURL, WKURLRef resourceDirectoryURL)
152{
153 return WebKit::wrapper(WebKit::toImpl(pageRef)->loadFile(WebKit::toWTFString(fileURL), WebKit::toWTFString(resourceDirectoryURL)));
154}

173}

176{
