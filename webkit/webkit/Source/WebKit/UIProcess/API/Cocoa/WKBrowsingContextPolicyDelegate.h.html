42typedef NS_ENUM(NSUInteger, WKPolicyDecision) {
43 WKPolicyDecisionCancel,
44 WKPolicyDecisionAllow,
45 WKPolicyDecisionBecomeDownload
46};
47
48typedef void (^WKPolicyDecisionHandler)(WKPolicyDecision);
49
50WK_CLASS_DEPRECATED_WITH_REPLACEMENT("WKNavigationDelegate", macos(10.10, 10.14.4), ios(8.0, 12.2))
51@protocol WKBrowsingContextPolicyDelegate <NSObject>
52@optional
53
54- (void)browsingContextController:(WKBrowsingContextController *)sender decidePolicyForNavigationAction:(NSDictionary *)actionInformation decisionHandler:(WKPolicyDecisionHandler)decisionHandler;
55- (void)browsingContextController:(WKBrowsingContextController *)sender decidePolicyForNewWindowAction:(NSDictionary *)actionInformation decisionHandler:(WKPolicyDecisionHandler)decisionHandler;
56- (void)browsingContextController:(WKBrowsingContextController *)sender decidePolicyForResponseAction:(NSDictionary *)actionInformation decisionHandler:(WKPolicyDecisionHandler)decisionHandler;
57
58@end
