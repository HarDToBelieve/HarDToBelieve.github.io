44- (void)dealloc
45{
46 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKConnection.class, self))
47 return;
48
49 self._connection.~WebConnection();
50
51 [super dealloc];
52}

54static void didReceiveMessage(WKConnectionRef, WKStringRef messageName, WKTypeRef messageBody, const void* clientInfo)
55{
56 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
57 auto connection = (__bridge WKConnection *)clientInfo;
58 ALLOW_DEPRECATED_DECLARATIONS_END
59 auto delegate = connection->_delegate.get();
60
61 if ([delegate respondsToSelector:@selector(connection:didReceiveMessageWithName:body:)]) {
62 RetainPtr<CFStringRef> nsMessageName = adoptCF(WKStringCopyCFString(kCFAllocatorDefault, messageName));
63 RetainPtr<id> nsMessageBody = static_cast<WebKit::ObjCObjectGraph*>(WebKit::toImpl(messageBody))->rootObject();
64 [delegate connection:connection didReceiveMessageWithName:(__bridge NSString *)nsMessageName.get() body:nsMessageBody.get()];
65 }
66}
67
68static void didClose(WKConnectionRef, const void* clientInfo)
69{
70 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
71 auto connection = (__bridge WKConnection *)clientInfo;
72 ALLOW_DEPRECATED_DECLARATIONS_END
73 auto delegate = connection->_delegate.get();
74
75 if ([delegate respondsToSelector:@selector(connectionDidClose:)])
76 [delegate connectionDidClose:connection];
77}
78
79ALLOW_DEPRECATED_DECLARATIONS_BEGIN
80static void setUpClient(WKConnection *wrapper, WebKit::WebConnection& connection)
81ALLOW_DEPRECATED_DECLARATIONS_END
82{
83 WKConnectionClientV0 client;
84 memset(&client, 0, sizeof(client));
85
86 client.base.version = 0;
87 client.base.clientInfo = (__bridge void*)wrapper;
88 client.didReceiveMessage = didReceiveMessage;
89 client.didClose = didClose;
90
91 connection.initializeConnectionClient(&client.base);
92}
93
94- (id <WKConnectionDelegate>)delegate
95{
96 return _delegate.getAutoreleased();
97}
98
99- (void)setDelegate:(id <WKConnectionDelegate>)delegate
100{
101 _delegate = delegate;
102 if (delegate)
103 setUpClient(self, self._connection);
104 else
105 self._connection.initializeConnectionClient(nullptr);
106}
107
108- (void)sendMessageWithName:(NSString *)messageName body:(id)messageBody
109{
110 auto wkMessageBody = WebKit::ObjCObjectGraph::create(messageBody);
111 self._connection.postMessage(messageName, wkMessageBody.ptr());
112}
