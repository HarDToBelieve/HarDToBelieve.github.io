41class DownloadClient final : public API::DownloadClient {
42public:
43 explicit DownloadClient(id <WKDownloadDelegate> delegate)
44 : m_delegate(delegate)
45 , m_respondsToWillPerformHTTPRedirection([delegate respondsToSelector:@selector(download:willPerformHTTPRedirection:newRequest:decisionHandler:)])
46 , m_respondsToDidReceiveAuthenticationChallenge([delegate respondsToSelector:@selector(download:didReceiveAuthenticationChallenge:completionHandler:)])
47 , m_respondsToDidFinish([m_delegate respondsToSelector:@selector(downloadDidFinish:)])
48 , m_respondsToDidFailWithError([delegate respondsToSelector:@selector(download:didFailWithError:resumeData:)])
49 {
50 ASSERT([delegate respondsToSelector:@selector(download:decideDestinationUsingResponse:suggestedFilename:completionHandler:)]);
51 }
52
53private:
54 void willSendRequest(WebKit::DownloadProxy& download, WebCore::ResourceRequest&& request, const WebCore::ResourceResponse& response, CompletionHandler<void(WebCore::ResourceRequest&&)>&& completionHandler) final
55 {
56 if (!m_delegate || !m_respondsToWillPerformHTTPRedirection)
57 return completionHandler(WTFMove(request));
58
59 RetainPtr<NSURLRequest> nsRequest = request.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody);
60 [m_delegate download:wrapper(download) willPerformHTTPRedirection:static_cast<NSHTTPURLResponse *>(response.nsURLResponse()) newRequest:nsRequest.get() decisionHandler:makeBlockPtr([request = WTFMove(request), completionHandler = WTFMove(completionHandler), checker = WebKit::CompletionHandlerCallChecker::create(m_delegate.get().get(), @selector(download:willPerformHTTPRedirection:newRequest:decisionHandler:))](WKDownloadRedirectPolicy policy) mutable {
61 if (checker->completionHandlerHasBeenCalled())
62 return;
63 checker->didCallCompletionHandler();
64 switch (policy) {
65 case WKDownloadRedirectPolicyCancel:
66 return completionHandler({ });
67 case WKDownloadRedirectPolicyAllow:
68 return completionHandler(WTFMove(request));
69 default:
70 [NSException raise:NSInvalidArgumentException format:@"Invalid WKDownloadRedirectPolicy (%ld)", (long)policy];
71 }
72 }).get()];
73 }
74
75 void didReceiveAuthenticationChallenge(WebKit::DownloadProxy& download, WebKit::AuthenticationChallengeProxy& challenge) final
76 {
77 if (!m_delegate || !m_respondsToDidReceiveAuthenticationChallenge)
78 return challenge.listener().completeChallenge(WebKit::AuthenticationChallengeDisposition::RejectProtectionSpaceAndContinue);
79
80 [m_delegate download:wrapper(download) didReceiveAuthenticationChallenge:wrapper(challenge) completionHandler:makeBlockPtr([challenge = makeRef(challenge), checker = WebKit::CompletionHandlerCallChecker::create(m_delegate.get().get(), @selector(download:didReceiveAuthenticationChallenge:completionHandler:))] (NSURLSessionAuthChallengeDisposition disposition, NSURLCredential *credential) mutable {
81 if (checker->completionHandlerHasBeenCalled())
82 return;
83 checker->didCallCompletionHandler();
84 switch (disposition) {
85 case NSURLSessionAuthChallengeUseCredential:
86 challenge->listener().completeChallenge(WebKit::AuthenticationChallengeDisposition::UseCredential, credential ? WebCore::Credential(credential) : WebCore::Credential());
87 break;
88 case NSURLSessionAuthChallengePerformDefaultHandling:
89 challenge->listener().completeChallenge(WebKit::AuthenticationChallengeDisposition::PerformDefaultHandling);
90 break;
91 case NSURLSessionAuthChallengeCancelAuthenticationChallenge:
92 challenge->listener().completeChallenge(WebKit::AuthenticationChallengeDisposition::Cancel);
93 break;
94 case NSURLSessionAuthChallengeRejectProtectionSpace:
95 challenge->listener().completeChallenge(WebKit::AuthenticationChallengeDisposition::RejectProtectionSpaceAndContinue);
96 break;
97 default:
98 [NSException raise:NSInvalidArgumentException format:@"Invalid NSURLSessionAuthChallengeDisposition (%ld)", (long)disposition];
99 }
100 }).get()];
101 }
102
103 void decideDestinationWithSuggestedFilename(WebKit::DownloadProxy& download, const WebCore::ResourceResponse& response, const WTF::String& suggestedFilename, CompletionHandler<void(WebKit::AllowOverwrite, WTF::String)>&& completionHandler) final
104 {
105 if (!m_delegate)
106 return completionHandler(WebKit::AllowOverwrite::No, { });
107
108 [m_delegate download:wrapper(download) decideDestinationUsingResponse:response.nsURLResponse() suggestedFilename:suggestedFilename completionHandler:makeBlockPtr([download = makeRef(download), completionHandler = WTFMove(completionHandler), checker = WebKit::CompletionHandlerCallChecker::create(m_delegate.get().get(), @selector(download:decideDestinationUsingResponse:suggestedFilename:completionHandler:))] (NSURL *destination) mutable {
109 if (checker->completionHandlerHasBeenCalled())
110 return;
111 checker->didCallCompletionHandler();
112
113 if (!destination)
114 return completionHandler(WebKit::AllowOverwrite::No, { });
115
116 if (!destination.isFileURL) {
117 completionHandler(WebKit::AllowOverwrite::No, { });
118 [NSException raise:NSInvalidArgumentException format:@"destination must be a file URL"];
119 return;
120 }
121
122 NSFileManager *fileManager = [NSFileManager defaultManager];
123 if (![fileManager fileExistsAtPath:[destination URLByDeletingLastPathComponent].path]) {
124 RunLoop::main().dispatch([download] {
125 download->didFail([NSError errorWithDomain:NSURLErrorDomain code:NSURLErrorCannotCreateFile userInfo:nil], { });
126 });
127 completionHandler(WebKit::AllowOverwrite::No, { });
128 return;
129 }
130 if ([fileManager fileExistsAtPath:destination.path]) {
131 RunLoop::main().dispatch([download] {
132 download->didFail([NSError errorWithDomain:NSURLErrorDomain code:NSURLErrorCannotCreateFile userInfo:nil], { });
133 });
134 completionHandler(WebKit::AllowOverwrite::No, { });
135 return;
136 }
137
138 wrapper(download.get()).progress.fileURL = destination;
139
140 completionHandler(WebKit::AllowOverwrite::No, destination.path);
141 }).get()];
142 }
143
144 void didReceiveData(WebKit::DownloadProxy& download, uint64_t, uint64_t totalBytesWritten, uint64_t totalBytesExpectedToWrite) final
145 {
146 NSProgress *progress = wrapper(download).progress;
147 progress.totalUnitCount = totalBytesExpectedToWrite;
148 progress.completedUnitCount = totalBytesWritten;
149 }
150
151 void didFinish(WebKit::DownloadProxy& download) final
152 {
153 if (!m_delegate || !m_respondsToDidFinish)
154 return;
155
156 [m_delegate downloadDidFinish:wrapper(download)];
157 }
158
159 void didFail(WebKit::DownloadProxy& download, const WebCore::ResourceError& error, API::Data* resumeData) final
160 {
161 if (!m_delegate || !m_respondsToDidFailWithError)
162 return;
163
164 [m_delegate download:wrapper(download) didFailWithError:error.nsError() resumeData:wrapper(resumeData)];
165 }
166
167 void processDidCrash(WebKit::DownloadProxy& download) final
168 {
169 if (!m_delegate || !m_respondsToDidFailWithError)
170 return;
171
172 [m_delegate download:wrapper(download) didFailWithError:[NSError errorWithDomain:NSURLErrorDomain code:NSURLErrorNetworkConnectionLost userInfo:nil] resumeData:nil];
173 }
174
175 WeakObjCPtr<id <WKDownloadDelegate> > m_delegate;
176
177 bool m_respondsToWillPerformHTTPRedirection : 1;
178 bool m_respondsToDidReceiveAuthenticationChallenge : 1;
179 bool m_respondsToDidFinish : 1;
180 bool m_respondsToDidFailWithError : 1;
181};

183@implementation WKDownload
184
185- (void)cancel:(void (^)(NSData *resumeData))completionHandler
186{
187 _download->cancel([completionHandler = makeBlockPtr(completionHandler)] (auto* data) {
188 if (completionHandler)
189 completionHandler(wrapper(data));
190 });
191}
192
193- (NSURLRequest *)originalRequest
194{
195 return _download->request().nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody);
196}
197
198- (WKWebView *)webView
199{
200 auto page = _download->originatingPage();
201 return page ? page->cocoaView().autorelease() : nil;
202}
203
204- (id <WKDownloadDelegate>)delegate
205{
206 return _delegate.get().get();
207}
208
209- (void)setDelegate:(id <WKDownloadDelegate>)delegate
210{
211 _delegate = delegate;
212 _download->setClient(adoptRef(*new DownloadClient(delegate)));
213}
214
215#pragma mark NSProgressReporting protocol implementation
216
217- (NSProgress *)progress
218{
219 NSProgress* downloadProgress = _download->progress();
220 if (!downloadProgress) {
221 constexpr auto indeterminateUnitCount = -1;
222 downloadProgress = [NSProgress progressWithTotalUnitCount:indeterminateUnitCount];
223
224 downloadProgress.kind = NSProgressKindFile;
225 downloadProgress.fileOperationKind = NSProgressFileOperationKindDownloading;
226
227 downloadProgress.cancellable = YES;
228 downloadProgress.cancellationHandler = makeBlockPtr([weakSelf = WeakObjCPtr<WKDownload> { self }] () mutable {
229 ensureOnMainRunLoop([weakSelf = WTFMove(weakSelf)] {
230 [weakSelf cancel:nil];
231 });
232 }).get();
233
234 _download->setProgress(downloadProgress);
235 }
236 return downloadProgress;
237}
238
239- (void)dealloc
240{
241 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKDownload.class, self))
242 return;
243 _download->~DownloadProxy();
244 [super dealloc];
245}

247#pragma mark WKObject protocol implementation
248
249- (API::Object&)_apiObject
250{
251 return *_download;
252}
