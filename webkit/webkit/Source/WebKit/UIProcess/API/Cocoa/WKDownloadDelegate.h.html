26#import <Foundation/Foundation.h>
27#import <WebKit/WKFoundation.h>
28
29@class WKDownload;
30
31/* @enum WKDownloadRedirectPolicy
32 @abstract The policy to pass back to the decision handler from the download:willPerformHTTPRedirection:newRequest:decisionHandler: method.
33 @constant WKDownloadRedirectPolicyCancel Cancel the redirect.
34 @constant WKDownloadRedirectPolicyAllow Allow the redirect.
35 */
36typedef NS_ENUM(NSInteger, WKDownloadRedirectPolicy) {
37 WKDownloadRedirectPolicyCancel,
38 WKDownloadRedirectPolicyAllow,
39} NS_SWIFT_NAME(WKDownload.RedirectPolicy) WK_API_AVAILABLE(macos(12.0), ios(15.0));
40
41NS_ASSUME_NONNULL_BEGIN
42
43@protocol WKDownloadDelegate <NSObject>
44
45@required
46
47/* @abstract Invoked when the download needs a location to write the downloaded bytes.
48 @param download The download for which we need a file to which to write.
49 @param response The server response if this download was the result of an HTTP request,
50 or a synthesized response for blob downloads.
51 @param suggestedFilename The suggested filename.
52 @param completionHandler The completion handler you must invoke with
53 either a file URL to begin the download or nil to cancel the download.
54 @discussion suggestedFilename will often be the same as response.suggestedFilename,
55 but web content can specify the suggested download filename. If the destination file
56 URL is non-null, it must be a file that does not exist in a directory that does exist
57 and can be written to.
58 */
59- (void)download:(WKDownload *)download decideDestinationUsingResponse:(NSURLResponse *)response suggestedFilename:(NSString *)suggestedFilename completionHandler:(void (^)(NSURL * _Nullable destination))completionHandler;
60
61@optional
62
63/* @abstract Invoked when the download has received an HTTP redirection response.
64 @param download The download that received the redirect.
65 @param response The redirection response.
66 @param newRequest The new request that will be sent.
67 @param completionHandler The completion handler you must invoke to indicate whether or not
68 to proceed with the redirection.
69 @discussion If you do not implement this method, all server suggested redirects will be taken.
70 */
71- (void)download:(WKDownload *)download willPerformHTTPRedirection:(NSHTTPURLResponse *)response newRequest:(NSURLRequest *)request decisionHandler:(void (^)(WKDownloadRedirectPolicy))decisionHandler WK_SWIFT_ASYNC_NAME(download(_:decidedPolicyForHTTPRedirection:newRequest:)) WK_SWIFT_ASYNC(4);
72
73/* @abstract Invoked when the download needs to respond to an authentication challenge.
74 @param download The download that received the authentication challenge.
75 @param challenge The authentication challenge.
76 @param completionHandler The completion handler you must invoke to respond to the challenge. The
77 disposition argument is one of the constants of the enumerated type
78 NSURLSessionAuthChallengeDisposition. When disposition is NSURLSessionAuthChallengeUseCredential,
79 the credential argument is the credential to use, or nil to indicate continuing without a
80 credential.
81 @discussion If you do not implement this method, the web view will respond to the authentication challenge with the NSURLSessionAuthChallengeRejectProtectionSpace disposition.
82 */
83- (void)download:(WKDownload *)download didReceiveAuthenticationChallenge:(NSURLAuthenticationChallenge *)challenge completionHandler:(void (^)(NSURLSessionAuthChallengeDisposition, NSURLCredential * _Nullable))completionHandler WK_SWIFT_ASYNC_NAME(download(_:respondTo:));
84
85/* @abstract Invoked when the download has finished successfully.
86 @param download The download that finished.
87 */
88- (void)downloadDidFinish:(WKDownload *)download;
89
90/* @abstract Invoked when the download has failed.
91 @param download The download that has failed.
92 @param error The error indicating the failure reason.
93 @param resumeData This data can be passed to WKWebView resumeDownloadFromResumeData: to attempt to resume this download.
94 */
95- (void)download:(WKDownload *)download didFailWithError:(NSError *)error resumeData:(nullable NSData *)resumeData;
96
97@end
98
99NS_ASSUME_NONNULL_END
