35/*! @enum WKErrorCode
36 @abstract Constants used by NSError to indicate errors in the WebKit domain.
37 @constant WKErrorUnknown Indicates that an unknown error occurred.
38 @constant WKErrorWebContentProcessTerminated Indicates that the Web Content process was terminated.
39 @constant WKErrorWebViewInvalidated Indicates that the WKWebView was invalidated.
40 @constant WKErrorJavaScriptExceptionOccurred Indicates that a JavaScript exception occurred.
41 @constant WKErrorJavaScriptResultTypeIsUnsupported Indicates that the result of JavaScript execution could not be returned.
42 @constant WKErrorContentRuleListStoreCompileFailed Indicates that compiling a WKUserContentRuleList failed.
43 @constant WKErrorContentRuleListStoreLookUpFailed Indicates that looking up a WKUserContentRuleList failed.
44 @constant WKErrorContentRuleListStoreRemoveFailed Indicates that removing a WKUserContentRuleList failed.
45 @constant WKErrorContentRuleListStoreVersionMismatch Indicates that the WKUserContentRuleList version did not match the latest.
46 @constant WKErrorAttributedStringContentFailedToLoad Indicates that the attributed string content failed to load.
47 @constant WKErrorAttributedStringContentLoadTimedOut Indicates that loading attributed string content timed out.
48 @constant WKErrorNavigationAppBoundDomain Indicates that a navigation failed due to an app-bound domain restriction.
49 @constant WKErrorJavaScriptAppBoundDomain Indicates that JavaScript execution failed due to an app-bound domain restriction.
50 */
51typedef NS_ENUM(NSInteger, WKErrorCode) {
52 WKErrorUnknown = 1,
53 WKErrorWebContentProcessTerminated,
54 WKErrorWebViewInvalidated,
55 WKErrorJavaScriptExceptionOccurred,
56 WKErrorJavaScriptResultTypeIsUnsupported WK_API_AVAILABLE(macos(10.11), ios(9.0)),
57 WKErrorContentRuleListStoreCompileFailed WK_API_AVAILABLE(macos(10.13), ios(11.0)),
58 WKErrorContentRuleListStoreLookUpFailed WK_API_AVAILABLE(macos(10.13), ios(11.0)),
59 WKErrorContentRuleListStoreRemoveFailed WK_API_AVAILABLE(macos(10.13), ios(11.0)),
60 WKErrorContentRuleListStoreVersionMismatch WK_API_AVAILABLE(macos(10.13), ios(11.0)),
61 WKErrorAttributedStringContentFailedToLoad WK_API_AVAILABLE(macos(10.15), ios(13.0)),
62 WKErrorAttributedStringContentLoadTimedOut WK_API_AVAILABLE(macos(10.15), ios(13.0)),
63 WKErrorJavaScriptInvalidFrameTarget WK_API_AVAILABLE(macos(11.0), ios(14.0)),
64 WKErrorNavigationAppBoundDomain WK_API_AVAILABLE(macos(11.0), ios(14.0)),
65 WKErrorJavaScriptAppBoundDomain WK_API_AVAILABLE(macos(11.0), ios(14.0)),
66} WK_API_AVAILABLE(macos(10.10), ios(8.0));
