26#import "config.h"
27#import "WKFrameInfoInternal.h"
28
29#import "WKSecurityOriginInternal.h"
30#import "WKWebViewInternal.h"
31#import "_WKFrameHandleInternal.h"
32#import <WebCore/WebCoreObjCExtras.h>
33
34@implementation WKFrameInfo
35
36- (void)dealloc
37{
38 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKFrameInfo.class, self))
39 return;
40
41 _frameInfo->~FrameInfo();
42
43 [super dealloc];
44}
45
46- (NSString *)description
47{
48 return [NSString stringWithFormat:@"<%@: %p; webView = %p; isMainFrame = %s; request = %@>", NSStringFromClass(self.class), self, self.webView, self.mainFrame ? "YES" : "NO", self.request];
49}
50
51- (BOOL)isMainFrame
52{
53 return _frameInfo->isMainFrame();
54}
55
56- (NSURLRequest *)request
57{
58 return _frameInfo->request().nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody);
59}
60
61- (WKSecurityOrigin *)securityOrigin
62{
63 auto& data = _frameInfo->securityOrigin();
64 auto apiOrigin = API::SecurityOrigin::create(data.protocol, data.host, data.port);
65 return retainPtr(wrapper(apiOrigin.get())).autorelease();
66}
67
68- (WKWebView *)webView
69{
70 auto page = _frameInfo->page();
71 return page ? page->cocoaView().autorelease() : nil;
72}
73
74- (id)copyWithZone:(NSZone *)zone
75{
76 return [self retain];
77}
78
79#pragma mark WKObject protocol implementation
80
81- (API::Object&)_apiObject
82{
83 return *_frameInfo;
84}
85
86@end
87
88@implementation WKFrameInfo (WKPrivate)
89
90- (_WKFrameHandle *)_handle
91{
92 return retainPtr(wrapper(_frameInfo->handle())).autorelease();
93}
94
95- (_WKFrameHandle *)_parentFrameHandle
96{
97 return retainPtr(wrapper(_frameInfo->parentFrameHandle())).autorelease();
98}
99
100@end
