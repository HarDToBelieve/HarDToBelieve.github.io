41static NSArray<NSHTTPCookie *> *coreCookiesToNSCookies(const Vector<WebCore::Cookie>& coreCookies)
42{
43 return createNSArray(coreCookies, [] (auto& cookie) -> NSHTTPCookie * {
44 return cookie;
45 }).autorelease();
46}

48class WKHTTPCookieStoreObserver : public API::HTTPCookieStore::Observer {
49 WTF_MAKE_FAST_ALLOCATED;
50public:
51 explicit WKHTTPCookieStoreObserver(id<WKHTTPCookieStoreObserver> observer)
52 : m_observer(observer)
53 {
54 }
55
56private:
57 void cookiesDidChange(API::HTTPCookieStore& cookieStore) final
58 {
59 [m_observer cookiesDidChangeInCookieStore:wrapper(cookieStore)];
60 }
61
62 WeakObjCPtr<id<WKHTTPCookieStoreObserver>> m_observer;
63};

65@implementation WKHTTPCookieStore {
66 HashMap<CFTypeRef, std::unique_ptr<WKHTTPCookieStoreObserver>> _observers;
67}
68
69- (void)dealloc
70{
71 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKHTTPCookieStore.class, self))
72 return;
73
74 for (auto& observer : _observers.values())
75 _cookieStore->unregisterObserver(*observer);
76
77 _cookieStore->API::HTTPCookieStore::~HTTPCookieStore();
78
79 [super dealloc];
80}

82- (void)getAllCookies:(void (^)(NSArray<NSHTTPCookie *> *))completionHandler
83{
84 _cookieStore->cookies([handler = adoptNS([completionHandler copy])](const Vector<WebCore::Cookie>& cookies) {
85 auto rawHandler = (void (^)(NSArray<NSHTTPCookie *> *))handler.get();
86 rawHandler(coreCookiesToNSCookies(cookies));
87 });
88}
89
90- (void)setCookie:(NSHTTPCookie *)cookie completionHandler:(void (^)(void))completionHandler
91{
92 _cookieStore->setCookies({ cookie }, [handler = adoptNS([completionHandler copy])]() {
93 auto rawHandler = (void (^)())handler.get();
94 if (rawHandler)
95 rawHandler();
96 });
97
98}
99
100- (void)deleteCookie:(NSHTTPCookie *)cookie completionHandler:(void (^)(void))completionHandler
101{
102 _cookieStore->deleteCookie(cookie, [handler = adoptNS([completionHandler copy])]() {
103 auto rawHandler = (void (^)())handler.get();
104 if (rawHandler)
105 rawHandler();
106 });
107}
108
109- (void)addObserver:(id<WKHTTPCookieStoreObserver>)observer
110{
111 auto result = _observers.add((__bridge CFTypeRef)observer, nullptr);
112 if (!result.isNewEntry)
113 return;
114
115 result.iterator->value = makeUnique<WKHTTPCookieStoreObserver>(observer);
116 _cookieStore->registerObserver(*result.iterator->value);
117}
118
119- (void)removeObserver:(id<WKHTTPCookieStoreObserver>)observer
120{
121 auto result = _observers.take((__bridge CFTypeRef)observer);
122 if (!result)
123 return;
124
125 _cookieStore->unregisterObserver(*result);
126}
