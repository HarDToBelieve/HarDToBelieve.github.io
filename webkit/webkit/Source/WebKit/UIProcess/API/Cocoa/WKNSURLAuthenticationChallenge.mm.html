40- (NSObject *)_web_createTarget
41{
42 WebKit::AuthenticationChallengeProxy& challenge = *reinterpret_cast<WebKit::AuthenticationChallengeProxy*>(&self._apiObject);
43
44 static dispatch_once_t token;
45 static NeverDestroyed<RetainPtr<WKNSURLAuthenticationChallengeSender>> sender;
46 dispatch_once(&token, ^{
47 sender.get() = adoptNS([[WKNSURLAuthenticationChallengeSender alloc] init]);
48 });
49
50 return [[NSURLAuthenticationChallenge alloc] initWithAuthenticationChallenge:mac(challenge.core()) sender:sender.get().get()];
51}
52
53- (WebKit::AuthenticationChallengeProxy&)_web_authenticationChallengeProxy
54{
55 return *reinterpret_cast<WebKit::AuthenticationChallengeProxy*>(&self._apiObject);
56}

62static void checkChallenge(NSURLAuthenticationChallenge *challenge)
63{
64 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
65 if ([challenge class] != [WKNSURLAuthenticationChallenge class])
66 [NSException raise:NSInvalidArgumentException format:@"The challenge was not sent by the receiver."];
67 ALLOW_DEPRECATED_DECLARATIONS_END
68}
69
70- (void)cancelAuthenticationChallenge:(NSURLAuthenticationChallenge *)challenge
71{
72 checkChallenge(challenge);
73 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
74 WebKit::AuthenticationChallengeProxy& webChallenge = ((WKNSURLAuthenticationChallenge *)challenge)._web_authenticationChallengeProxy;
75 ALLOW_DEPRECATED_DECLARATIONS_END
76 webChallenge.listener().completeChallenge(WebKit::AuthenticationChallengeDisposition::Cancel);
77}
