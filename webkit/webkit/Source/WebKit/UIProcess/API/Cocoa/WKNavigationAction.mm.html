42@implementation WKNavigationAction
43
44static WKNavigationType toWKNavigationType(WebCore::NavigationType navigationType)
45{
46 switch (navigationType) {
47 case WebCore::NavigationType::LinkClicked:
48 return WKNavigationTypeLinkActivated;
49 case WebCore::NavigationType::FormSubmitted:
50 return WKNavigationTypeFormSubmitted;
51 case WebCore::NavigationType::BackForward:
52 return WKNavigationTypeBackForward;
53 case WebCore::NavigationType::Reload:
54 return WKNavigationTypeReload;
55 case WebCore::NavigationType::FormResubmitted:
56 return WKNavigationTypeFormResubmitted;
57 case WebCore::NavigationType::Other:
58 return WKNavigationTypeOther;
59 }
60
61 ASSERT_NOT_REACHED();
62 return WKNavigationTypeOther;
63}
64
65#if PLATFORM(IOS_FAMILY)
66static WKSyntheticClickType toWKSyntheticClickType(WebKit::WebMouseEvent::SyntheticClickType syntheticClickType)
67{
68 switch (syntheticClickType) {
69 case WebKit::WebMouseEvent::NoTap:
70 return WKSyntheticClickTypeNoTap;
71 case WebKit::WebMouseEvent::OneFingerTap:
72 return WKSyntheticClickTypeOneFingerTap;
73 case WebKit::WebMouseEvent::TwoFingerTap:
74 return WKSyntheticClickTypeTwoFingerTap;
75 }
76 ASSERT_NOT_REACHED();
77 return WKSyntheticClickTypeNoTap;
78}
79#endif

81- (void)dealloc
82{
83 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKNavigationAction.class, self))
84 return;
85
86 _navigationAction->~NavigationAction();
87
88 [super dealloc];
89}

91- (NSString *)description
92{
93 return [NSString stringWithFormat:@"<%@: %p; navigationType = %ld; syntheticClickType = %ld; position x = %.2f y = %.2f request = %@; sourceFrame = %@; targetFrame = %@>", NSStringFromClass(self.class), self,
94 (long)self.navigationType,
95#if PLATFORM(IOS_FAMILY)
96 (long)self._syntheticClickType, self._clickLocationInRootViewCoordinates.x, self._clickLocationInRootViewCoordinates.y,
97#else
98 0L, 0.0, 0.0,
99#endif
100 self.request, self.sourceFrame, self.targetFrame];
101}
