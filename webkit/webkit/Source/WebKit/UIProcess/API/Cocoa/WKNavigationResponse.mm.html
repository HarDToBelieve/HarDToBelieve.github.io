26#import "config.h"
27#import "WKNavigationResponseInternal.h"
28
29#import "WKFrameInfoInternal.h"
30#import <WebCore/WebCoreObjCExtras.h>
31
32@implementation WKNavigationResponse
33
34- (void)dealloc
35{
36 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKNavigationResponse.class, self))
37 return;
38
39 _navigationResponse->~NavigationResponse();
40
41 [super dealloc];
42}
43
44- (NSString *)description
45{
46 return [NSString stringWithFormat:@"<%@: %p; response = %@>", NSStringFromClass(self.class), self, self.response];
47}
48
49- (BOOL)isForMainFrame
50{
51 return _navigationResponse->frame().isMainFrame();
52}
53
54- (NSURLResponse *)response
55{
56 return _navigationResponse->response().nsURLResponse();
57}
58
59- (BOOL)canShowMIMEType
60{
61 return _navigationResponse->canShowMIMEType();
62}
63
64#pragma mark WKObject protocol implementation
65
66- (API::Object&)_apiObject
67{
68 return *_navigationResponse;
69}
70
71@end
72
73@implementation WKNavigationResponse (WKPrivate)
74
75- (WKFrameInfo *)_frame
76{
77 return wrapper(_navigationResponse->frame());
78}
79
80- (NSURLRequest *)_request
81{
82 return _navigationResponse->request().nsURLRequest(WebCore::HTTPBodyUpdatePolicy::DoNotUpdateHTTPBody);
83}
84
85- (NSString *)_downloadAttribute
86{
87 const String& attribute = _navigationResponse->downloadAttribute();
88 return attribute.isNull() ? nil : (NSString *)attribute;
89}
90
91@end
