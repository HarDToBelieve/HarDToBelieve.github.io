65static void didCreateConnection(WKContextRef, WKConnectionRef connectionRef, const void* clientInfo)
66{
67 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
68 auto processGroup = (__bridge WKProcessGroup *)clientInfo;
69 ALLOW_DEPRECATED_DECLARATIONS_END
70 auto delegate = processGroup->_delegate.get();
71
72 if ([delegate respondsToSelector:@selector(processGroup:didCreateConnectionToWebProcessPlugIn:)])
73 [delegate processGroup:processGroup didCreateConnectionToWebProcessPlugIn:wrapper(*WebKit::toImpl(connectionRef))];
74}

77static void setUpConnectionClient(WKProcessGroup *processGroup, WKContextRef contextRef)
78ALLOW_DEPRECATED_DECLARATIONS_END
79{
80 WKContextConnectionClientV0 connectionClient;
81 memset(&connectionClient, 0, sizeof(connectionClient));
82
83 connectionClient.base.version = 0;
84 connectionClient.base.clientInfo = (__bridge CFTypeRef)processGroup;
85 connectionClient.didCreateConnection = didCreateConnection;
86
87 WKContextSetConnectionClient(contextRef, &connectionClient.base);
88}

90static WKTypeRef getInjectedBundleInitializationUserData(WKContextRef, const void* clientInfo)
91{
92 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
93 auto processGroup = (__bridge WKProcessGroup *)clientInfo;
94 ALLOW_DEPRECATED_DECLARATIONS_END
95 auto delegate = processGroup->_delegate.get();
96
97 if ([delegate respondsToSelector:@selector(processGroupWillCreateConnectionToWebProcessPlugIn:)]) {
98 RetainPtr<id> initializationUserData = [delegate processGroupWillCreateConnectionToWebProcessPlugIn:processGroup];
99
100 return toAPI(&WebKit::ObjCObjectGraph::create(initializationUserData.get()).leakRef());
101 }
102
103 return 0;
104}

193- (id)init
194{
195 return [self initWithInjectedBundleURL:nil];

198- (id)initWithInjectedBundleURL:(NSURL *)bundleURL
199{
200 self = [super init];
201 if (!self)
202 return nil;
203
204 auto configuration = API::ProcessPoolConfiguration::create();
205 configuration->setInjectedBundlePath(bundleURL ? String(bundleURL.path) : String());
206
207 _processPool = WebKit::WebProcessPool::create(configuration);
208
209 setUpConnectionClient(self, toAPI(_processPool.get()));
210 setUpInjectedBundleClient(self, toAPI(_processPool.get()));
211 setUpHistoryClient(self, toAPI(_processPool.get()));
212
213 return self;
214}

216static Vector<WTF::String> toStringVector(NSSet *input)
217{
218 Vector<WTF::String> vector;
219
220 NSUInteger size = input.count;
221 if (!size)
222 return vector;
223
224 vector.reserveInitialCapacity(size);
225 for (id classObj : input) {
226 if (auto* string = NSStringFromClass(classObj))
227 vector.uncheckedAppend(string);
228 }
229 return vector;
230}

232- (id)initWithInjectedBundleURL:(NSURL *)bundleURL andCustomClassesForParameterCoder:(NSSet *)classesForCoder
233{
234 self = [super init];
235 if (!self)
236 return nil;
237
238 auto configuration = API::ProcessPoolConfiguration::create();
239 configuration->setInjectedBundlePath(bundleURL ? String(bundleURL.path) : String());
240 configuration->setCustomClassesForParameterCoder(toStringVector(classesForCoder));
241
242 _processPool = WebKit::WebProcessPool::create(configuration);
243
244 setUpConnectionClient(self, toAPI(_processPool.get()));
245 setUpInjectedBundleClient(self, toAPI(_processPool.get()));
246 setUpHistoryClient(self, toAPI(_processPool.get()));
247
248 return self;
249}

251- (id <WKProcessGroupDelegate>)delegate
252{
253 return _delegate.getAutoreleased();
254}
255
256- (void)setDelegate:(id <WKProcessGroupDelegate>)delegate
257{
258 _delegate = delegate;
259
260 // If the client can observe when the connection to the WebProcess injected bundle is established, then we cannot
261 // safely delay the launch of the WebProcess until something is loaded in the web view.
262 if ([delegate respondsToSelector:@selector(processGroup:didCreateConnectionToWebProcessPlugIn:)])
263 _processPool->disableDelayedWebProcessLaunch();
264}
