32@class WKScriptMessage;
33@class WKUserContentController;
34
35/*! A class conforming to the WKScriptMessageHandlerWithReply protocol provides a
36method for receiving messages from JavaScript running in a webpage and replying to them asynchronously.
37*/
38@protocol WKScriptMessageHandlerWithReply <NSObject>
39
40/*! @abstract Invoked when a script message is received from a webpage.
41 @param userContentController The user content controller invoking the delegate method.
42 @param message The script message received.
43 @param replyHandler A block to be called with the result of processing the message.
44 @discussion When the JavaScript running in your application's web content called window.webkit.messageHandlers.<name>.postMessage(<messageBody>),
45 a JavaScript Promise object was returned. The values passed to the replyHandler are used to resolve that Promise.
46
47 Passing a non-nil NSString value to the second parameter of the replyHandler signals an error.
48 No matter what value you pass to the first parameter of the replyHandler,
49 the Promise will be rejected with a JavaScript error object whose message property is set to that errorMessage string.
50
51 If the second parameter to the replyHandler is nil, the first argument will be serialized into its JavaScript equivalent and the Promise will be fulfilled with the resulting value.
52 If the first argument is nil then the Promise will be resolved with a JavaScript value of "undefined"
53
54 Allowed non-nil result types are:
55 NSNumber, NSNull, NSString, NSDate, NSArray, and NSDictionary.
56 Any NSArray or NSDictionary containers can only contain objects of those types.
57
58 The replyHandler can be called at most once.
59 If the replyHandler is deallocated before it is called, the Promise will be rejected with a JavaScript Error object
60 with an appropriate message indicating the handler was never called.
61
62 Example:
63
64 With a WKScriptMessageHandlerWithReply object installed with the name "testHandler", consider the following JavaScript:
65 @textblock
66 var promise = window.webkit.messageHandlers.testHandler.postMessage("Fulfill me with 42");
67 await p;
68 return p;
69 @/textblock
70
71 And consider the following WKScriptMessageHandlerWithReply implementation:
72 @textblock
73 - (void)userContentController:(WKUserContentController *)userContentController didReceiveScriptMessage:(WKScriptMessage *)message replyHandler:(void (^)(id, NSString *))replyHandler
74 {
75 if ([message.body isEqual:@"Fulfill me with 42"])
76 replyHandler(@42, nil);
77 else
78 replyHandler(nil, @"Unexpected message received");
79 }
80 @/textblock
81
82 In this example:
83 - The JavaScript code sends a message to your application code with the body @"Fulfill me with 42"
84 - JavaScript execution is suspended while waiting for the resulting promise to resolve.
85 - Your message handler is invoked with that message and a block to call with the reply when ready.
86 - Your message handler sends the value @42 as a reply.
87 - The JavaScript promise is fulfilled with the value 42.
88 - JavaScript execution continues and the value 42 is returned.
89 */
90- (void)userContentController:(WKUserContentController *)userContentController didReceiveScriptMessage:(WKScriptMessage *)message replyHandler:(void (^)(id _Nullable reply, NSString *_Nullable errorMessage))replyHandler WK_SWIFT_ASYNC(3) WK_API_AVAILABLE(macos(11.0), ios(14.0));
91
92@end
