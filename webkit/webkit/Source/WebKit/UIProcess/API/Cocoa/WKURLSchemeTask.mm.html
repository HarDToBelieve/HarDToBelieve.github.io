40static WebKit::WebURLSchemeTask::ExceptionType getExceptionTypeFromMainRunLoop(Function<WebKit::WebURLSchemeTask::ExceptionType ()>&& function)
41{
42 WebKit::WebURLSchemeTask::ExceptionType exceptionType;
43 callOnMainRunLoopAndWait([function = WTFMove(function), &exceptionType] {
44 exceptionType = function();
45 });
46
47 return exceptionType;
48}
49
50static void raiseExceptionIfNecessary(WebKit::WebURLSchemeTask::ExceptionType exceptionType)
51{
52 switch (exceptionType) {
53 case WebKit::WebURLSchemeTask::ExceptionType::None:
54 return;
55 case WebKit::WebURLSchemeTask::ExceptionType::TaskAlreadyStopped:
56 [NSException raise:NSInternalInconsistencyException format:@"This task has already been stopped"];
57 break;
58 case WebKit::WebURLSchemeTask::ExceptionType::CompleteAlreadyCalled:
59 [NSException raise:NSInternalInconsistencyException format:@"[WKURLSchemeTask taskDidCompleteWithError:] has already been called for this task"];
60 break;
61 case WebKit::WebURLSchemeTask::ExceptionType::DataAlreadySent:
62 [NSException raise:NSInternalInconsistencyException format:@"[WKURLSchemeTask taskDidReceiveData:] has already been called for this task"];
63 break;
64 case WebKit::WebURLSchemeTask::ExceptionType::NoResponseSent:
65 [NSException raise:NSInternalInconsistencyException format:@"No response has been sent for this task"];
66 break;
67 case WebKit::WebURLSchemeTask::ExceptionType::RedirectAfterResponse:
68 [NSException raise:NSInternalInconsistencyException format:@"No redirects are allowed after the response"];
69 break;
70 case WebKit::WebURLSchemeTask::ExceptionType::WaitingForRedirectCompletionHandler:
71 [NSException raise:NSInternalInconsistencyException format:@"No callbacks are allowed while waiting for the redirection completion handler to be invoked"];
72 break;
73 }
74}

83- (void)dealloc
84{

91- (NSURLRequest *)request

96- (BOOL)_requestOnlyIfCached
97{
98 return _urlSchemeTask->nsRequest().cachePolicy == NSURLRequestReturnCacheDataDontLoad;
99}
100
101- (void)_willPerformRedirection:(NSURLResponse *)response newRequest:(NSURLRequest *)request completionHandler:(void (^)(NSURLRequest *))completionHandler
102{
103 auto function = [protectedSelf = retainPtr(self), self, protectedResponse = retainPtr(response), response, protectedRequest = retainPtr(request), request, handler = makeBlockPtr(completionHandler)] () mutable {
104 return _urlSchemeTask->willPerformRedirection(response, request, [handler = WTFMove(handler)] (WebCore::ResourceRequest&& actualNewRequest) {
105 handler.get()(actualNewRequest.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::UpdateHTTPBody));
106 });
107 };
108
109 auto result = getExceptionTypeFromMainRunLoop(WTFMove(function));
110 raiseExceptionIfNecessary(result);
111}
112
113- (void)didReceiveResponse:(NSURLResponse *)response

116 return _urlSchemeTask->didReceiveResponse(response);
117 };
118

120 raiseExceptionIfNecessary(result);
121}
