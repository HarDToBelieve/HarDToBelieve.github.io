26#import <WebKit/WKURLSchemeTask.h>
27
28@class WKFrameInfo;
29
30WK_API_AVAILABLE(macos(10.13), ios(11.0))
31@protocol WKURLSchemeTaskPrivate <WKURLSchemeTask>
32
33/*! @abstract Indicate the task was redirected.
34 @param response The response causing a redirect to a new URL.
35 @param newRequest The new proposed request to handle the redirection.
36 @param completionHandler A completion handler called with the request you should use to commit to the redirection.
37 @discussion When performing a load of a URL over a network, the server might decide a different URL should be loaded instead.
38 A common example is an HTTP redirect.
39 When this happens, you should notify WebKit by sending the servers response and a proposed new request to the WKURLSchemeTask.
40 WebKit might decide that changes need to be make to the proposed request.
41 This is communicated through the completionHandler which tells you the request you should make to commit to the redirection.
42
43 An exception will be thrown if you make any other callbacks to the WKURLSchemeTask while this completionHandler is pending, other than didFailWithError:.
44 An exception will be thrown if your app has been told to stop loading this task via the registered WKURLSchemeHandler object.
45 */
46- (void)_willPerformRedirection:(NSURLResponse *)response newRequest:(NSURLRequest *)request completionHandler:(void (^)(NSURLRequest *))completionHandler WK_API_AVAILABLE(macos(12.0), ios(15.0));
47- (void)_didPerformRedirection:(NSURLResponse *)response newRequest:(NSURLRequest *)request;
48
49@property (nonatomic, readonly) BOOL _requestOnlyIfCached WK_API_AVAILABLE(macos(10.15), ios(13.0));
50@property (nonatomic, readonly) WKFrameInfo *_frame WK_API_AVAILABLE(macos(11.0), ios(14.0));
51
52@end
