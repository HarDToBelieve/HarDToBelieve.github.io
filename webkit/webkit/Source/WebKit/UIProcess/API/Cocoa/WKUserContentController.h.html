38/*! A WKUserContentController object provides a way for JavaScript to post
39 messages to a web view.
40 The user content controller associated with a web view is specified by its
41 web view configuration.
42 */
43WK_CLASS_AVAILABLE(macos(10.10), ios(8.0))
44@interface WKUserContentController : NSObject <NSSecureCoding>
45
46/*! @abstract The user scripts associated with this user content
47 controller.
48*/
49@property (nonatomic, readonly, copy) NSArray<WKUserScript *> *userScripts;
50
51/*! @abstract Adds a user script.
52 @param userScript The user script to add.
53*/
54- (void)addUserScript:(WKUserScript *)userScript;
55
56/*! @abstract Removes all associated user scripts.
57*/
58- (void)removeAllUserScripts;
59
60/*! @abstract Adds a script message handler.
61 @param scriptMessageHandler The script message handler to add.
62 @param contentWorld The WKContentWorld in which to add the script message handler.
63 @param name The name of the message handler.
64 @discussion Adding a script message handler adds a function
65 window.webkit.messageHandlers.<name>.postMessage(<messageBody>) to all frames, available in the given WKContentWorld.
66
67 The name argument must be a non-empty string.
68
69 Each WKContentWorld can have any number of script message handlers, but only one per unique name.
70
71 Once any script message handler has been added to a WKContentWorld for a given name, it is an error to add another
72 script message handler to that WKContentWorld for that same name without first removing the previous script message handler.
73
74 The above restriction applies to any type of script message handler - WKScriptMessageHandler and WKScriptMessageHandlerWithReply
75 objects will conflict with each other if you try to add them to the same WKContentWorld with the same name.
76 */
77- (void)addScriptMessageHandler:(id <WKScriptMessageHandler>)scriptMessageHandler contentWorld:(WKContentWorld *)world name:(NSString *)name WK_API_AVAILABLE(macos(11.0), ios(14.0));
78
79/*! @abstract Adds a script message handler.
80 @param scriptMessageHandlerWithReply The script message handler to add.
81 @param contentWorld The WKContentWorld in which to add the script message handler.
82 @param name The name of the message handler.
83 @discussion Adding a script message handler adds a function
84 window.webkit.messageHandlers.<name>.postMessage(<messageBody>) to all frames, available in the given WKContentWorld.
85
86 The name argument must be a non-empty string.
87
88 Each WKContentWorld can have any number of script message handlers, but only one per unique name.
89
90 Once any script message handler has been added to a WKContentWorld for a given name, it is an error to add another
91 script message handler to that WKContentWorld for that same name without first removing the previous script message handler.
92
93 The above restriction applies to any type of script message handler - WKScriptMessageHandlerWithReply and WKScriptMessageHandler
94 objects will conflict with each other if you try to add them to the same WKContentWorld with the same name.
95
96 Refer to the WKScriptMessageHandlerWithReply documentation for examples of how it is more flexible than WKScriptMessageHandler.
97 */
98- (void)addScriptMessageHandlerWithReply:(id <WKScriptMessageHandlerWithReply>)scriptMessageHandlerWithReply contentWorld:(WKContentWorld *)contentWorld name:(NSString *)name WK_API_AVAILABLE(macos(11.0), ios(14.0));
99
100/*! @abstract Adds a script message handler to the main world used by page content itself.
101 @param scriptMessageHandler The script message handler to add.
102 @param name The name of the message handler.
103 @discussion Calling this method is equivalent to calling addScriptMessageHandler:contentWorld:name:
104 with [WKContentWorld pageWorld] as the contentWorld argument.
105 */
106- (void)addScriptMessageHandler:(id <WKScriptMessageHandler>)scriptMessageHandler name:(NSString *)name;

108/*! @abstract Removes a script message handler.

112- (void)removeScriptMessageHandlerForName:(NSString *)name contentWorld:(WKContentWorld *)contentWorld WK_API_AVAILABLE(macos(11.0), ios(14.0));

119- (void)removeScriptMessageHandlerForName:(NSString *)name;
