145 void didPostMessage(WebKit::WebPageProxy& page, WebKit::FrameInfoData&& frameInfoData, API::ContentWorld& world, WebCore::SerializedScriptValue& serializedScriptValue) final
146 {
147 @autoreleasepool {
148 auto webView = page.cocoaView();
149 if (!webView)
150 return;
151 RetainPtr<WKFrameInfo> frameInfo = wrapper(API::FrameInfo::create(WTFMove(frameInfoData), &page));
152 id body = API::SerializedScriptValue::deserialize(serializedScriptValue, 0);
153 auto message = adoptNS([[WKScriptMessage alloc] _initWithBody:body webView:webView.get() frameInfo:frameInfo.get() name:m_name.get() world:wrapper(world)]);
154
155 [(id<WKScriptMessageHandler>)m_handler.get() userContentController:m_controller.get() didReceiveScriptMessage:message.get()];
156 }
157 }
158
159 bool supportsAsyncReply() final
160 {
161 return m_supportsAsyncReply;
162 }
163
164 void didPostMessageWithAsyncReply(WebKit::WebPageProxy& page, WebKit::FrameInfoData&& frameInfoData, API::ContentWorld& world, WebCore::SerializedScriptValue& serializedScriptValue, Function<void(API::SerializedScriptValue*, const String&)>&& replyHandler) final
165 {
166 ASSERT(m_supportsAsyncReply);
167
168 auto webView = page.cocoaView();
169 if (!webView) {
170 replyHandler(nullptr, "The WKWebView was deallocated before the message was delivered"_s);
171 return;
172 }
173
174 auto finalizer = [](Function<void(API::SerializedScriptValue*, const String&)>& function) {
175 function(nullptr, "WKWebView API client did not respond to this postMessage"_s);
176 };
177 __block auto handler = CompletionHandlerWithFinalizer<void(API::SerializedScriptValue*, const String&)>(WTFMove(replyHandler), WTFMove(finalizer));
178
179 @autoreleasepool {
180 RetainPtr<WKFrameInfo> frameInfo = wrapper(API::FrameInfo::create(WTFMove(frameInfoData), &page));
181 id body = API::SerializedScriptValue::deserialize(serializedScriptValue, 0);
182 auto message = adoptNS([[WKScriptMessage alloc] _initWithBody:body webView:webView.get() frameInfo:frameInfo.get() name:m_name.get() world:wrapper(world)]);
183
184 [(id<WKScriptMessageHandlerWithReply>)m_handler.get() userContentController:m_controller.get() didReceiveScriptMessage:message.get() replyHandler:^(id result, NSString *errorMessage) {
185 if (!handler)
186 [NSException raise:NSInternalInconsistencyException format:@"replyHandler passed to userContentController:didReceiveScriptMessage:replyHandler: should not be called twice"];
187
188 if (errorMessage) {
189 handler(nullptr, errorMessage);
190 return;
191 }
192
193 auto serializedValue = API::SerializedScriptValue::createFromNSObject(result);
194 if (!serializedValue) {
195 handler(nullptr, "The result value passed back from the WKWebView API client was unable to be serialized"_s);
196 return;
197 }
198
199 handler(serializedValue.get(), { });
200 }];
201 }
202 }
203
204private:
205 RetainPtr<WKUserContentController> m_controller;
206 RetainPtr<id> m_handler;
207 RetainPtr<NSString> m_name;
208 bool m_supportsAsyncReply;
209};
