58template<typename T> class LazyInitialized {
59public:
60 typedef typename WTF::GetPtrHelper<T>::PtrType PtrType;
61
62 template<typename F>
63 PtrType get(F&& f)
64 {
65 if (!m_isInitialized) {
66 m_value = f();
67 m_isInitialized = true;
68 }
69
70 return m_value.get();
71 }
72
73 void set(PtrType t)
74 {
75 m_value = t;
76 m_isInitialized = true;
77 }
78
79 void set(T&& t)
80 {
81 m_value = WTFMove(t);
82 m_isInitialized = true;
83 }
84
85 PtrType peek()
86 {
87 return m_value.get();
88 }
89
90private:
91 bool m_isInitialized = false;
92 T m_value;
93};

563- (void)setURLSchemeHandler:(id <WKURLSchemeHandler>)urlSchemeHandler forURLScheme:(NSString *)urlScheme
564{
565 if ([WKWebView handlesURLScheme:urlScheme])
566 [NSException raise:NSInvalidArgumentException format:@"'%@' is a URL scheme that WKWebView handles natively", urlScheme];
567
568 auto canonicalScheme = WTF::URLParser::maybeCanonicalizeScheme(String(urlScheme));
569 if (!canonicalScheme)
570 [NSException raise:NSInvalidArgumentException format:@"'%@' is not a valid URL scheme", urlScheme];
571
572 if (_pageConfiguration->urlSchemeHandlerForURLScheme(*canonicalScheme))
573 [NSException raise:NSInvalidArgumentException format:@"URL scheme '%@' already has a registered URL scheme handler", urlScheme];
574
575 _pageConfiguration->setURLSchemeHandlerForURLScheme(WebKit::WebURLSchemeHandlerCocoa::create(urlSchemeHandler), *canonicalScheme);
576}
577
578- (id <WKURLSchemeHandler>)urlSchemeHandlerForURLScheme:(NSString *)urlScheme
579{
580 auto canonicalScheme = WTF::URLParser::maybeCanonicalizeScheme(String(urlScheme));
581 if (!canonicalScheme)
582 return nil;
583
584 auto handler = _pageConfiguration->urlSchemeHandlerForURLScheme(*canonicalScheme);
585 return handler ? static_cast<WebKit::WebURLSchemeHandlerCocoa*>(handler.get())->apiHandler() : nil;
586}
