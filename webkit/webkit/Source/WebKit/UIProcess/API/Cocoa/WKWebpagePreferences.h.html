26#import <Foundation/Foundation.h>
27#import <WebKit/WKFoundation.h>
28
29/*! @enum WKContentMode
30 @abstract A content mode represents the type of content to load, as well as
31 additional layout and rendering adaptations that are applied as a result of
32 loading the content
33 @constant WKContentModeRecommended The recommended content mode for the current platform
34 @constant WKContentModeMobile Represents content targeting mobile browsers
35 @constant WKContentModeDesktop Represents content targeting desktop browsers
36 @discussion WKContentModeRecommended behaves like WKContentModeMobile on iPhone and iPad mini
37 and WKContentModeDesktop on other iPad models as well as Mac.
38 */
39typedef NS_ENUM(NSInteger, WKContentMode) {
40 WKContentModeRecommended,
41 WKContentModeMobile,
42 WKContentModeDesktop
43} WK_API_AVAILABLE(ios(13.0));
44
45/*! A WKWebpagePreferences object is a collection of properties that
46 determine the preferences to use when loading and rendering a page.
47 @discussion Contains properties used to determine webpage preferences.
48 */
49WK_CLASS_AVAILABLE(macos(10.15), ios(13.0))
50@interface WKWebpagePreferences : NSObject
51
52/*! @abstract A WKContentMode indicating the content mode to prefer
53 when loading and rendering a webpage.
54 @discussion The default value is WKContentModeRecommended. The stated
55 preference is ignored on subframe navigation
56 */
57@property (nonatomic) WKContentMode preferredContentMode WK_API_AVAILABLE(ios(13.0));
58
59/* @abstract A Boolean value indicating whether JavaScript from web content is enabled
60 @discussion If this value is set to NO then JavaScript referenced by the web content will not execute.
61 This includes JavaScript found in inline <script> elements, referenced by external JavaScript resources,
62 "javascript:" URLs, and all other forms.
63
64 Even if this value is set to NO your application can still execute JavaScript using:
65 - [WKWebView evaluteJavaScript:completionHandler:]
66 - [WKWebView evaluteJavaScript:inContentWorld:completionHandler:]
67 - [WKWebView callAsyncJavaScript:arguments:inContentWorld:completionHandler:]
68 - WKUserScripts
69
70 The default value is YES.
71*/
72@property (nonatomic) BOOL allowsContentJavaScript WK_API_AVAILABLE(macos(11.0), ios(14.0));
73
74@end
