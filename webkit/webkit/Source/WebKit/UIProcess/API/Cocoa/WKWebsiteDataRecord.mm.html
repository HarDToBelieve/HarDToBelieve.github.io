71static NSString *dataTypesToString(NSSet *dataTypes)
72{
73 auto array = adoptNS([[NSMutableArray alloc] init]);
74
75 if ([dataTypes containsObject:WKWebsiteDataTypeDiskCache])
76 [array addObject:@"Disk Cache"];
77 if ([dataTypes containsObject:WKWebsiteDataTypeFetchCache])
78 [array addObject:@"Fetch Cache"];
79 if ([dataTypes containsObject:WKWebsiteDataTypeMemoryCache])
80 [array addObject:@"Memory Cache"];
81 if ([dataTypes containsObject:WKWebsiteDataTypeOfflineWebApplicationCache])
82 [array addObject:@"Offline Web Application Cache"];
83 if ([dataTypes containsObject:WKWebsiteDataTypeCookies])
84 [array addObject:@"Cookies"];
85 if ([dataTypes containsObject:WKWebsiteDataTypeSessionStorage])
86 [array addObject:@"Session Storage"];
87 if ([dataTypes containsObject:WKWebsiteDataTypeLocalStorage])
88 [array addObject:@"Local Storage"];
89 if ([dataTypes containsObject:WKWebsiteDataTypeWebSQLDatabases])
90 [array addObject:@"Web SQL"];
91 if ([dataTypes containsObject:WKWebsiteDataTypeIndexedDBDatabases])
92 [array addObject:@"IndexedDB"];
93 if ([dataTypes containsObject:WKWebsiteDataTypeServiceWorkerRegistrations])
94 [array addObject:@"Service Worker Registrations"];
95 if ([dataTypes containsObject:_WKWebsiteDataTypeHSTSCache])
96 [array addObject:@"HSTS Cache"];
97 if ([dataTypes containsObject:_WKWebsiteDataTypeMediaKeys])
98 [array addObject:@"Media Keys"];
99 if ([dataTypes containsObject:_WKWebsiteDataTypeSearchFieldRecentSearches])
100 [array addObject:@"Search Field Recent Searches"];
101#if PLATFORM(MAC)
102 if ([dataTypes containsObject:_WKWebsiteDataTypePlugInData])
103 [array addObject:@"Plug-in Data"];
104#endif
105 if ([dataTypes containsObject:_WKWebsiteDataTypeResourceLoadStatistics])
106 [array addObject:@"Resource Load Statistics"];
107 if ([dataTypes containsObject:_WKWebsiteDataTypeCredentials])
108 [array addObject:@"Credentials"];
109 if ([dataTypes containsObject:_WKWebsiteDataTypeAdClickAttributions] || [dataTypes containsObject:_WKWebsiteDataTypePrivateClickMeasurements])
110 [array addObject:@"Private Click Measurements"];
111 if ([dataTypes containsObject:_WKWebsiteDataTypeAlternativeServices])
112 [array addObject:@"Alternative Services"];
113
114 return [array componentsJoinedByString:@", "];
115}

117- (NSString *)description
118{
119 auto result = adoptNS([[NSMutableString alloc] initWithFormat:@"<%@: %p; displayName = %@; dataTypes = { %@ }", NSStringFromClass(self.class), self, self.displayName, dataTypesToString(self.dataTypes)]);
120
121 if (auto* dataSize = self._dataSize)
122 [result appendFormat:@"; _dataSize = { %llu bytes }", dataSize.totalSize];
123
124 [result appendString:@">"];
125 return result.autorelease();
126}

149- (_WKWebsiteDataSize *)_dataSize
150{
151 auto& size = _websiteDataRecord->websiteDataRecord().size;
152
153 if (!size)
154 return nil;
155
156 return adoptNS([[_WKWebsiteDataSize alloc] initWithSize:*size]).autorelease();
157}
