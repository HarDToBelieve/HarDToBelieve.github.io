26#import <WebKit/WKFoundation.h>
27
28#import <WebKit/WKWebsiteDataRecord.h>
29
30NS_ASSUME_NONNULL_BEGIN
31
32@class WKHTTPCookieStore;
33
34/*! A WKWebsiteDataStore represents various types of data that a website might
35 make use of. This includes cookies, disk and memory caches, and persistent data such as WebSQL,
36 IndexedDB databases, and local storage.
37 */
38WK_CLASS_AVAILABLE(macos(10.11), ios(9.0))
39@interface WKWebsiteDataStore : NSObject <NSSecureCoding>
40
41/* @abstract Returns the default data store. */
42+ (WKWebsiteDataStore *)defaultDataStore;
43
44/** @abstract Returns a new non-persistent data store.
45 @discussion If a WKWebView is associated with a non-persistent data store, no data will
46 be written to the file system. This is useful for implementing "private browsing" in a web view.
47*/
48+ (WKWebsiteDataStore *)nonPersistentDataStore;
49
50- (instancetype)new NS_UNAVAILABLE;
51- (instancetype)init NS_UNAVAILABLE;
52
53/*! @abstract Whether the data store is persistent or not. */
54@property (nonatomic, readonly, getter=isPersistent) BOOL persistent;
55
56/*! @abstract Returns a set of all available website data types. */
57+ (NSSet<NSString *> *)allWebsiteDataTypes;
58
59/*! @abstract Fetches data records containing the given website data types.
60 @param dataTypes The website data types to fetch records for.
61 @param completionHandler A block to invoke when the data records have been fetched.
62*/
63- (void)fetchDataRecordsOfTypes:(NSSet<NSString *> *)dataTypes completionHandler:(void (^)(NSArray<WKWebsiteDataRecord *> *))completionHandler WK_SWIFT_ASYNC_NAME(dataRecords(ofTypes:));
64
65/*! @abstract Removes website data of the given types for the given data records.
66 @param dataTypes The website data types that should be removed.
67 @param dataRecords The website data records to delete website data for.
68 @param completionHandler A block to invoke when the website data for the records has been removed.
69*/
70- (void)removeDataOfTypes:(NSSet<NSString *> *)dataTypes forDataRecords:(NSArray<WKWebsiteDataRecord *> *)dataRecords completionHandler:(void (^)(void))completionHandler;
71
72/*! @abstract Removes all website data of the given types that has been modified since the given date.
73 @param dataTypes The website data types that should be removed.
74 @param date A date. All website data modified after this date will be removed.
75 @param completionHandler A block to invoke when the website data has been removed.
76*/
77- (void)removeDataOfTypes:(NSSet<NSString *> *)dataTypes modifiedSince:(NSDate *)date completionHandler:(void (^)(void))completionHandler;
78
79/*! @abstract Returns the cookie store representing HTTP cookies in this website data store. */
80@property (nonatomic, readonly) WKHTTPCookieStore *httpCookieStore WK_API_AVAILABLE(macos(10.13), ios(11.0));
81
82@end
83
84NS_ASSUME_NONNULL_END
