48+ (BOOL)supportsSecureCoding
49{
50 return YES;
51}
52
53- (instancetype)initWithCoder:(NSCoder *)aDecoder
54{
55 NSString *name = [aDecoder decodeObjectOfClass:[NSString class] forKey:@"name"];
56 NSString *shortName = [aDecoder decodeObjectOfClass:[NSString class] forKey:@"short_name"];
57 NSString *description = [aDecoder decodeObjectOfClass:[NSString class] forKey:@"description"];
58 NSURL *scopeURL = [aDecoder decodeObjectOfClass:[NSURL class] forKey:@"scope"];
59 NSInteger display = [aDecoder decodeIntegerForKey:@"display"];
60 NSURL *startURL = [aDecoder decodeObjectOfClass:[NSURL class] forKey:@"start_url"];
61 CocoaColor *themeColor = [aDecoder decodeObjectOfClass:[CocoaColor class] forKey:@"theme_color"];
62
63 WebCore::ApplicationManifest coreApplicationManifest {
64 WTF::String(name),
65 WTF::String(shortName),
66 WTF::String(description),
67 URL(scopeURL),
68 static_cast<WebCore::ApplicationManifest::Display>(display),
69 URL(startURL),
70 WebCore::Color(themeColor.CGColor),
71 };
72
73 API::Object::constructInWrapper<API::ApplicationManifest>(self, WTFMove(coreApplicationManifest));
74
75 return self;
76}
77
78- (void)dealloc
79{
80 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(_WKApplicationManifest.class, self))
81 return;
82
83 _applicationManifest->~ApplicationManifest();
84
85 [super dealloc];
86}
87
88- (void)encodeWithCoder:(NSCoder *)aCoder
89{
90 [aCoder encodeObject:self.name forKey:@"name"];
91 [aCoder encodeObject:self.shortName forKey:@"short_name"];
92 [aCoder encodeObject:self.applicationDescription forKey:@"description"];
93 [aCoder encodeObject:self.scope forKey:@"scope"];
94 [aCoder encodeInteger:static_cast<NSInteger>(_applicationManifest->applicationManifest().display) forKey:@"display"];
95 [aCoder encodeObject:self.startURL forKey:@"start_url"];
96 [aCoder encodeObject:self.themeColor forKey:@"theme_color"];
97}
98
99+ (_WKApplicationManifest *)applicationManifestFromJSON:(NSString *)json manifestURL:(NSURL *)manifestURL documentURL:(NSURL *)documentURL
100{
101 auto manifest = WebCore::ApplicationManifestParser::parse(WTF::String(json), URL(manifestURL), URL(documentURL));
102 return wrapper(API::ApplicationManifest::create(manifest));
103}

106{
107 return *_applicationManifest;
108}

110static NSString *nullableNSString(const WTF::String& string)
111{
112 return string.isNull() ? nil : (NSString *)string;
113}

145- (_WKApplicationManifestDisplayMode)displayMode
146{
147 switch (_applicationManifest->applicationManifest().display) {
148 case WebCore::ApplicationManifest::Display::Browser:
149 return _WKApplicationManifestDisplayModeBrowser;
150 case WebCore::ApplicationManifest::Display::MinimalUI:
151 return _WKApplicationManifestDisplayModeMinimalUI;
152 case WebCore::ApplicationManifest::Display::Standalone:
153 return _WKApplicationManifestDisplayModeStandalone;
154 case WebCore::ApplicationManifest::Display::Fullscreen:
155 return _WKApplicationManifestDisplayModeFullScreen;
156 }
157
158 ASSERT_NOT_REACHED();
159}
