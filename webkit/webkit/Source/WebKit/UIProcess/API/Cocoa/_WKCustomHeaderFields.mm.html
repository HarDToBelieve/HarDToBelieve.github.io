27#import "_WKCustomHeaderFields.h"
28
29#import "_WKCustomHeaderFieldsInternal.h"
30#import <WebCore/WebCoreObjCExtras.h>
31#import <wtf/BlockPtr.h>
32#import <wtf/cocoa/VectorCocoa.h>
33
34@implementation _WKCustomHeaderFields
35
36- (instancetype)init
37{
38 if (!(self = [super init]))
39 return nil;
40
41 API::Object::constructInWrapper<API::CustomHeaderFields>(self);
42 return self;
43}
44
45- (void)dealloc
46{
47 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(_WKCustomHeaderFields.class, self))
48 return;
49 _fields->API::CustomHeaderFields::~CustomHeaderFields();
50 [super dealloc];
51}
52
53- (NSDictionary<NSString *, NSString *> *)fields
54{
55 auto& vector = _fields->fields();
56 NSMutableDictionary<NSString *, NSString *> *dictionary = [NSMutableDictionary dictionaryWithCapacity:vector.size()];
57 for (auto& field : vector)
58 [dictionary setObject:field.value() forKey:field.name()];
59 return dictionary;
60}
61
62- (void)setFields:(NSDictionary<NSString *, NSString *> *)fields
63{
64 Vector<WebCore::HTTPHeaderField> vector;
65 vector.reserveInitialCapacity(fields.count);
66 [fields enumerateKeysAndObjectsUsingBlock:makeBlockPtr([&](id key, id value, BOOL* stop) {
67 if (auto field = WebCore::HTTPHeaderField::create((NSString *)key, (NSString *)value); field && startsWithLettersIgnoringASCIICase(field->name(), "x-"))
68 vector.uncheckedAppend(WTFMove(*field));
69 }).get()];
70 _fields->setFields(WTFMove(vector));
71}
72
73- (NSArray<NSString *> *)thirdPartyDomains
74{
75 return createNSArray(_fields->thirdPartyDomains()).autorelease();
76}
77
78- (void)setThirdPartyDomains:(NSArray<NSString *> *)thirdPartyDomains
79{
80 _fields->setThirdPartyDomains(makeVector<String>(thirdPartyDomains));
81}
82
83- (API::Object&)_apiObject
84{
85 return *_fields;
86}
87
88@end
