50class InspectorClient final : public API::InspectorClient {
51public:
52 explicit InspectorClient(id <_WKInspectorDelegate> delegate)
53 : m_delegate(delegate)
54 , m_respondsToInspectorOpenURLExternally([delegate respondsToSelector:@selector(inspector:openURLExternally:)])
55 , m_respondsToInspectorFrontendLoaded([delegate respondsToSelector:@selector(inspectorFrontendLoaded:)])
56 {
57 }
58
59 ~InspectorClient() = default;
60
61private:
62 // API::InspectorClient
63 void openURLExternally(WebKit::WebInspectorUIProxy& inspector, const WTF::String& url) final
64 {
65 if (!m_delegate || !m_respondsToInspectorOpenURLExternally)
66 return;
67
68 [m_delegate inspector:wrapper(inspector) openURLExternally:[NSURL URLWithString:url]];
69 }
70
71 void frontendLoaded(WebKit::WebInspectorUIProxy& inspector) final
72 {
73 if (!m_delegate || !m_respondsToInspectorFrontendLoaded)
74 return;
75
76 [m_delegate inspectorFrontendLoaded:wrapper(inspector)];
77 }
78
79 WeakObjCPtr<id <_WKInspectorDelegate> > m_delegate;
80
81 bool m_respondsToInspectorOpenURLExternally : 1;
82 bool m_respondsToInspectorFrontendLoaded : 1;
83};

229- (void)registerExtensionWithID:(NSString *)extensionID displayName:(NSString *)displayName completionHandler:(void(^)(NSError *, _WKInspectorExtension *))completionHandler
230{
231#if ENABLE(INSPECTOR_EXTENSIONS)
232 // It is an error to call this method prior to creating a frontend (i.e., with -connect or -show).
233 if (!_inspector->extensionController()) {
234 completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:@{ NSLocalizedFailureReasonErrorKey: Inspector::extensionErrorToString(Inspector::ExtensionError::InvalidRequest)}], nil);
235 return;
236 }
237
238 _inspector->extensionController()->registerExtension(extensionID, displayName, [protectedSelf = retainPtr(self), capturedBlock = makeBlockPtr(completionHandler)] (Expected<RefPtr<API::InspectorExtension>, Inspector::ExtensionError> result) mutable {
239 if (!result) {
240 capturedBlock([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:@{ NSLocalizedFailureReasonErrorKey: Inspector::extensionErrorToString(result.error())}], nil);
241 return;
242 }
243
244 capturedBlock(nil, wrapper(result.value()));
245 });
246#else
247 completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:nil], nil);
248#endif
249}
250
251- (void)unregisterExtension:(_WKInspectorExtension *)extension completionHandler:(void(^)(NSError *))completionHandler
252{
253#if ENABLE(INSPECTOR_EXTENSIONS)
254 // It is an error to call this method prior to creating a frontend (i.e., with -connect or -show).
255 if (!_inspector->extensionController()) {
256 completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:@{ NSLocalizedFailureReasonErrorKey: Inspector::extensionErrorToString(Inspector::ExtensionError::InvalidRequest)}]);
257 return;
258 }
259
260 _inspector->extensionController()->unregisterExtension(extension.extensionID, [protectedSelf = retainPtr(self), capturedBlock = makeBlockPtr(completionHandler)] (Expected<bool, Inspector::ExtensionError> result) mutable {
261 if (!result) {
262 capturedBlock([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:@{ NSLocalizedFailureReasonErrorKey: Inspector::extensionErrorToString(result.error())}]);
263 return;
264 }
265
266 capturedBlock(nil);
267 });
268#else
269 completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:nil]);
270#endif
271}
