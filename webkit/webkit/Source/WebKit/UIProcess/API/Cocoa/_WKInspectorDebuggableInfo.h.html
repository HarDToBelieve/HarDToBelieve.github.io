26#import <WebKit/WKFoundation.h>
27
28/*! A _WKInspectorDebuggableInfo object contains information about a debuggable target.
29 @discussion An instance of this class is a transient, data-only object;
30 it does not uniquely identify a debuggable across multiple method calls.
31 */
32
33NS_ASSUME_NONNULL_BEGIN
34
35typedef NS_ENUM(NSInteger, _WKInspectorDebuggableType) {
36 _WKInspectorDebuggableTypeITML,
37 _WKInspectorDebuggableTypeJavaScript,
38 _WKInspectorDebuggableTypeServiceWorker,
39 _WKInspectorDebuggableTypePage,
40 _WKInspectorDebuggableTypeWebPage,
41} WK_API_AVAILABLE(macos(10.15.4), ios(13.4));
42
43WK_CLASS_AVAILABLE(macos(10.15.4), ios(13.4))
44@interface _WKInspectorDebuggableInfo : NSObject <NSCopying>
45@property (nonatomic) _WKInspectorDebuggableType debuggableType;
46@property (nonatomic, copy) NSString *targetPlatformName;
47@property (nonatomic, copy) NSString *targetBuildVersion;
48@property (nonatomic, copy) NSString *targetProductVersion;
49@property (nonatomic) BOOL targetIsSimulator;
50@end
51
52NS_ASSUME_NONNULL_END
