26#pragma once
27
28#import <WebKit/WKFoundation.h>
29
30#if TARGET_OS_OSX
31
32#import <Foundation/Foundation.h>
33
34NS_ASSUME_NONNULL_BEGIN
35
36@protocol _WKInspectorExtensionDelegate;
37
38WK_CLASS_AVAILABLE(macos(12.0))
39@interface _WKInspectorExtension : NSObject
40
41- (instancetype)new NS_UNAVAILABLE;
42- (instancetype)init NS_UNAVAILABLE;
43
44/**
45 * @abstract Creates a new tab in the Web Inspector interface for this extension.
46 * @param tabName A localized display name for the tab.
47 * @param tabIconURL The location of an image resource to use for display in the created tab's title.
48 * @param sourceURL The location of the main resource to load in the new tab's iframe browsing context.
49 * @param completionHandler The completion handler to be called when creating a tab succeeds or fails.
50 * @discussion The inspectorTabIdentifier provided to the completion handler can be used to identify tabs
51 * that are passed as a parameter to _WKInspectorExtensionDelegate methods.
52 */
53- (void)createTabWithName:(NSString *)tabName tabIconURL:(NSURL *)tabIconURL sourceURL:(NSURL *)sourceURL completionHandler:(void(^)(NSError * _Nullable, NSString * _Nullable inspectorTabIdentifier))completionHandler;
54
55/**
56 * @abstract Evaluates JavaScript in the context of the inspected page on behalf of the _WKInspectorExtension.
57 * @param scriptSource The JavaScript code to be evaluated.
58 * @param frameURL URL for the frame in which to evaluate the script. If nil is passed, the main frame will be used.
59 * @param contextSecurityOrigin If specified, evaluate the script in the content script context of a different security origin.
60 * @param usesContentScriptContext If YES, evaluate the script in the context of the extension's content scripts.
61 * @param completionHandler A block to invoke when the operation completes or fails.
62 * @discussion The completionHandler is passed an NSJSONSerialization-compatible NSObject representing the evaluation result, or an error.
63 * scriptSource is treated as a top-level evaluation. By default, the script is evaluated in the inspected page's script context.
64 * The inspected page ultimately controls its execution context and the result of this evaluation. Thus, the result shall be treated as untrusted input.
65 */
66- (void)evaluateScript:(NSString *)scriptSource frameURL:(NSURL *)frameURL contextSecurityOrigin:(NSURL *)contextSecurityOrigin useContentScriptContext:(BOOL)useContentScriptContext completionHandler:(void(^)(NSError * _Nullable, NSDictionary * _Nullable result))completionHandler;
67
68/**
69 * @abstract Reloads the inspected page on behalf of the _WKInspectorExtension.
70 * @param ignoreCache If YES, reloads the page while ignoring the cache.
71 * @param userAgent If specified, overrides the user agent to be sent in the `User-Agent` header and returned by calls to `navigator.userAgent` made by scripts running in the page. This only affects the next navigation.
72 * @param injectedScript If specified, injects the given JavaScript expression into all frames on the page before any other scripts.
73 * @param completionHandler A block to invoke when the operation completes or fails.
74 */
75- (void)reloadIgnoringCache:(BOOL)ignoreCache userAgent:(NSString *)userAgent injectedScript:(NSString *)injectedScript completionHandler:(void(^)(NSError * _Nullable))completionHandler;
76
77@property (readonly, nonatomic) NSString *extensionID;
78
79/**
80 * @abstract Allows the client to receive extension lifecycle events that
81 * arise from within Web Inspector.
82 */
83@property (nonatomic, weak) id <_WKInspectorExtensionDelegate> delegate WK_API_AVAILABLE(macos(12.0));
84
85@end
86
87NS_ASSUME_NONNULL_END
