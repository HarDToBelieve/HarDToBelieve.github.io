26#import "config.h"
27#import "_WKLinkIconParametersInternal.h"
28
29#import <WebCore/LinkIcon.h>
30
31@implementation _WKLinkIconParameters {
32 RetainPtr<NSURL> _url;
33 WKLinkIconType _iconType;
34 RetainPtr<NSString> _mimeType;
35 RetainPtr<NSNumber> _size;
36 RetainPtr<NSMutableDictionary> _attributes;
37}
38
39- (instancetype)_initWithLinkIcon:(const WebCore::LinkIcon&)linkIcon
40{
41 if (!(self = [super init]))
42 return nil;
43
44 _url = (NSURL *)linkIcon.url;
45 _mimeType = (NSString *)linkIcon.mimeType;
46
47 if (linkIcon.size)
48 _size = adoptNS([[NSNumber alloc] initWithUnsignedInt:linkIcon.size.value()]);
49
50 switch (linkIcon.type) {
51 case WebCore::LinkIconType::Favicon:
52 _iconType = WKLinkIconTypeFavicon;
53 break;
54 case WebCore::LinkIconType::TouchIcon:
55 _iconType = WKLinkIconTypeTouchIcon;
56 break;
57 case WebCore::LinkIconType::TouchPrecomposedIcon:
58 _iconType = WKLinkIconTypeTouchPrecomposedIcon;
59 break;
60 }
61
62 _attributes = adoptNS([[NSMutableDictionary alloc] initWithCapacity:linkIcon.attributes.size()]);
63 for (auto& attributePair : linkIcon.attributes)
64 _attributes.get()[(NSString *)attributePair.first] = attributePair.second;
65
66 return self;
67}
68
69- (NSURL *)url
70{
71 return _url.get();
72}
73
74- (NSString *)mimeType
75{
76 return _mimeType.get();
77}
78
79- (NSNumber *)size
80{
81 return _size.get();
82}
83
84- (WKLinkIconType)iconType
85{
86 return _iconType;
87}
88
89- (NSDictionary *)attributes
90{
91 return _attributes.get();
92}
93
94@end
