70- (NSSet<Class> *)customClassesForParameterCoder
71{
72 auto classes = _processPoolConfiguration->customClassesForParameterCoder();
73 if (classes.isEmpty())
74 return [NSSet set];
75
76 auto result = adoptNS([[NSMutableSet alloc] initWithCapacity:classes.size()]);
77 for (const auto& value : classes)
78 [result addObject: objc_lookUpClass(value.utf8().data())];
79
80 return result.autorelease();
81}
82
83- (void)setCustomClassesForParameterCoder:(NSSet<Class> *)classesForCoder
84{
85 Vector<WTF::String> classes;
86 classes.reserveInitialCapacity(classesForCoder.count);
87 for (id classObj : classesForCoder) {
88 if (auto* string = NSStringFromClass(classObj))
89 classes.uncheckedAppend(string);
90 }
91
92 _processPoolConfiguration->setCustomClassesForParameterCoder(WTFMove(classes));
93}

154- (NSArray<NSURL *> *)additionalReadAccessAllowedURLs
155{
156 auto paths = _processPoolConfiguration->additionalReadAccessAllowedPaths();
157 if (paths.isEmpty())
158 return @[ ];
159
160 return createNSArray(paths, [] (auto& path) {
161 return [NSURL fileURLWithFileSystemRepresentation:path.data() isDirectory:NO relativeToURL:nil];
162 }).autorelease();
163}
164
165- (void)setAdditionalReadAccessAllowedURLs:(NSArray<NSURL *> *)additionalReadAccessAllowedURLs
166{
167 Vector<CString> paths;
168 paths.reserveInitialCapacity(additionalReadAccessAllowedURLs.count);
169 for (NSURL *url in additionalReadAccessAllowedURLs) {
170 if (!url.isFileURL)
171 [NSException raise:NSInvalidArgumentException format:@"%@ is not a file URL", url];
172
173 paths.uncheckedAppend(url.fileSystemRepresentation);
174 }
175
176 _processPoolConfiguration->setAdditionalReadAccessAllowedPaths(WTFMove(paths));
177}
