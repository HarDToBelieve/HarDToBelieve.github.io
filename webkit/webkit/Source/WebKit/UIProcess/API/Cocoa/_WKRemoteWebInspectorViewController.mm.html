59 _WKRemoteWebInspectorUIProxyClient(_WKRemoteWebInspectorViewController *controller)
60 : m_controller(controller)
61 {
62 }
63
64 virtual ~_WKRemoteWebInspectorUIProxyClient()
65 {
66 }
67
68 void sendMessageToBackend(const String& message) override
69 {
70 [m_controller sendMessageToBackend:message];
71 }
72
73 void closeFromFrontend() override
74 {
75 [m_controller closeFromFrontend];
76 }
77
78 Ref<API::InspectorConfiguration> configurationForRemoteInspector(RemoteWebInspectorUIProxy& inspector) override
79 {
80 return static_cast<API::InspectorConfiguration&>([m_controller.configuration _apiObject]);
81 }

85};

89@implementation _WKRemoteWebInspectorViewController {
90 std::unique_ptr<WebKit::_WKRemoteWebInspectorUIProxyClient> m_remoteInspectorClient;
91 _WKInspectorConfiguration *_configuration;
92}
93
94- (instancetype)initWithConfiguration:(_WKInspectorConfiguration *)configuration
95{
96 if (!(self = [super init]))
97 return nil;
98
99 _configuration = [configuration copy];
100
101 m_remoteInspectorProxy = WebKit::RemoteWebInspectorUIProxy::create();
102 m_remoteInspectorClient = makeUnique<WebKit::_WKRemoteWebInspectorUIProxyClient>(self);
103 m_remoteInspectorProxy->setClient(m_remoteInspectorClient.get());
104
105 return self;
106}
107
108- (NSWindow *)window
109{
110 return m_remoteInspectorProxy->window();
111}
112
113- (WKWebView *)webView
114{
115 return m_remoteInspectorProxy->webView();
116}
117
118- (void)loadForDebuggable:(_WKInspectorDebuggableInfo *)debuggableInfo backendCommandsURL:(NSURL *)backendCommandsURL
119{
120 m_remoteInspectorProxy->load(static_cast<API::DebuggableInfo&>([debuggableInfo _apiObject]), backendCommandsURL.absoluteString);
121}
122
123- (void)close
124{
125 m_remoteInspectorProxy->closeFromBackend();
126}
127
128- (void)show
129{
130 m_remoteInspectorProxy->show();
131}
132
133- (void)sendMessageToFrontend:(NSString *)message
134{
135 m_remoteInspectorProxy->sendMessageToFrontend(message);
136}

138// MARK: RemoteWebInspectorUIProxyClient methods
139
140- (void)sendMessageToBackend:(NSString *)message
141{
142 if (_delegate && [_delegate respondsToSelector:@selector(inspectorViewController:sendMessageToBackend:)])
143 [_delegate inspectorViewController:self sendMessageToBackend:message];
144}
145
146- (void)closeFromFrontend
147{
148 if (_delegate && [_delegate respondsToSelector:@selector(inspectorViewControllerInspectorDidClose:)])
149 [_delegate inspectorViewControllerInspectorDidClose:self];
150}

167- (void)registerExtensionWithID:(NSString *)extensionID displayName:(NSString *)displayName completionHandler:(void(^)(NSError *, _WKInspectorExtension * _Nullable))completionHandler
168{
169#if ENABLE(INSPECTOR_EXTENSIONS)
170 // If this method is called prior to creating a frontend with -loadForDebuggable:backendCommandsURL:, it will not succeed.
171 if (!m_remoteInspectorProxy->extensionController()) {
172 completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:@{ NSLocalizedFailureReasonErrorKey: Inspector::extensionErrorToString(Inspector::ExtensionError::InvalidRequest) }], nil);
173 return;
174 }
175
176 m_remoteInspectorProxy->extensionController()->registerExtension(extensionID, displayName, [protectedExtensionID = retainPtr(extensionID), protectedSelf = retainPtr(self), capturedBlock = makeBlockPtr(completionHandler)] (Expected<RefPtr<API::InspectorExtension>, Inspector::ExtensionError> result) mutable {
177 if (!result) {
178 capturedBlock([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:@{ NSLocalizedFailureReasonErrorKey: Inspector::extensionErrorToString(result.error()) }], nil);
179 return;
180 }
181
182 capturedBlock(nil, wrapper(result.value()));
183 });
184#else
185 completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:nil], nil);
186#endif
187}
188
189- (void)unregisterExtension:(_WKInspectorExtension *)extension completionHandler:(void(^)(NSError *))completionHandler
190{
191#if ENABLE(INSPECTOR_EXTENSIONS)
192 // If this method is called prior to creating a frontend with -loadForDebuggable:backendCommandsURL:, it will not succeed.
193 if (!m_remoteInspectorProxy->extensionController()) {
194 completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:@{ NSLocalizedFailureReasonErrorKey: Inspector::extensionErrorToString(Inspector::ExtensionError::InvalidRequest) }]);
195 return;
196 }
197 m_remoteInspectorProxy->extensionController()->unregisterExtension(extension.extensionID, [protectedSelf = retainPtr(self), capturedBlock = makeBlockPtr(completionHandler)] (Expected<bool, Inspector::ExtensionError> result) mutable {
198 if (!result) {
199 capturedBlock([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:@{ NSLocalizedFailureReasonErrorKey: Inspector::extensionErrorToString(result.error()) }]);
200 return;
201 }
202
203 capturedBlock(nil);
204 });
205#else
206 completionHandler([NSError errorWithDomain:WKErrorDomain code:WKErrorUnknown userInfo:nil]);
207#endif
208}
