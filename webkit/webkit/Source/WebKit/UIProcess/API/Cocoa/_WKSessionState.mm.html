26#import "config.h"
27#import "_WKSessionStateInternal.h"
28
29#import "SessionStateCoding.h"
30
31@implementation _WKSessionState
32
33- (instancetype)initWithData:(NSData *)data
34{
35 if (!(self = [super init]))
36 return nil;
37
38 if (!WebKit::decodeSessionState(data, _sessionState)) {
39 [self release];
40 return nil;
41 }
42
43 return self;
44}
45
46- (instancetype)_initWithSessionState:(WebKit::SessionState)sessionState
47{
48 if (!(self = [super init]))
49 return nil;
50
51 _sessionState = WTFMove(sessionState);
52
53 return self;
54}
55
56- (NSData *)data
57{
58 return WebKit::encodeSessionState(_sessionState).autorelease();
59}
60
61@end
