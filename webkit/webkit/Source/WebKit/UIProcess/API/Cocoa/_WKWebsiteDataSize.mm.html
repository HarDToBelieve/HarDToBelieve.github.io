26#import "config.h"
27#import "_WKWebsiteDataSizeInternal.h"
28
29#import "WKWebsiteDataRecordInternal.h"
30
31@implementation _WKWebsiteDataSize
32
33- (instancetype)initWithSize:(const WebKit::WebsiteDataRecord::Size&)size
34{
35 if (!(self = [super init]))
36 return nil;
37
38 _size = size;
39
40 return self;
41}
42
43- (unsigned long long)totalSize
44{
45 return _size.totalSize;
46}
47
48- (unsigned long long)sizeOfDataTypes:(NSSet *)dataTypes
49{
50 unsigned long long size = 0;
51
52 for (NSString *dataType in dataTypes) {
53 if (auto websiteDataType = WebKit::toWebsiteDataType(dataType))
54 size += _size.typeSizes.get(static_cast<unsigned>(*websiteDataType));
55 }
56
57 return size;
58}
59
60@end
