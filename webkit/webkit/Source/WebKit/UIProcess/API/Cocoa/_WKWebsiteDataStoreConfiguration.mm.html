32static void checkURLArgument(NSURL *url)
33{
34 if (url && ![url isFileURL])
35 [NSException raise:NSInvalidArgumentException format:@"%@ is not a file URL", url];
36}

40- (instancetype)init
41{
42 self = [super init];
43 if (!self)
44 return nil;
45

62- (void)dealloc
63{

80- (void)_setWebStorageDirectory:(NSURL *)url
81{
82 if (!_configuration->isPersistent())
83 [NSException raise:NSInvalidArgumentException format:@"Cannot set _webStorageDirectory on a non-persistent _WKWebsiteDataStoreConfiguration."];
84 checkURLArgument(url);
