350static inline CGFloat floorToDevicePixel(CGFloat input, float deviceScaleFactor)
351{
352 return CGFloor(input * deviceScaleFactor) / deviceScaleFactor;
353}

355static inline bool pointsEqualInDevicePixels(CGPoint a, CGPoint b, float deviceScaleFactor)
356{
357 return fabs(a.x * deviceScaleFactor - b.x * deviceScaleFactor) < std::numeric_limits<float>::epsilon()
358 && fabs(a.y * deviceScaleFactor - b.y * deviceScaleFactor) < std::numeric_limits<float>::epsilon();
359}

382- (void)_setHasCustomContentView:(BOOL)pageHasCustomContentView loadedMIMEType:(const WTF::String&)mimeType
383{
384 Class representationClass = nil;
385 if (pageHasCustomContentView)
386 representationClass = [[_configuration _contentProviderRegistry] providerForMIMEType:mimeType];
387
388 if (pageHasCustomContentView && representationClass) {
389 [_customContentView removeFromSuperview];
390 [_customContentFixedOverlayView removeFromSuperview];
391
392 _customContentView = adoptNS([[representationClass alloc] web_initWithFrame:self.bounds webView:self mimeType:mimeType]);
393 _customContentFixedOverlayView = adoptNS([[UIView alloc] initWithFrame:self.bounds]);
394 [_customContentFixedOverlayView layer].name = @"CustomContentFixedOverlay";
395 [_customContentFixedOverlayView setUserInteractionEnabled:NO];
396
397 [[_contentView unscaledView] removeFromSuperview];
398 [_contentView removeFromSuperview];
399 [_scrollView addSubview:_customContentView.get()];
400 [self addSubview:_customContentFixedOverlayView.get()];
401
402 [_customContentView web_setMinimumSize:self.bounds.size];
403 [_customContentView web_setFixedOverlayView:_customContentFixedOverlayView.get()];
404
405 _scrollViewBackgroundColor = WebCore::Color();
406 [_scrollView setContentOffset:[self _initialContentOffsetForScrollView]];
407 [_scrollView _setScrollEnabledInternal:YES];
408
409 [self _setAvoidsUnsafeArea:NO];
410 } else if (_customContentView) {
411 [_customContentView removeFromSuperview];
412 _customContentView = nullptr;
413
414 [_customContentFixedOverlayView removeFromSuperview];
415 _customContentFixedOverlayView = nullptr;
416
417 [_scrollView addSubview:_contentView.get()];
418 [_scrollView addSubview:[_contentView unscaledView]];
419 [_scrollView setContentSize:roundScrollViewContentSize(*_page, [_contentView frame].size)];
420
421 [_customContentFixedOverlayView setFrame:self.bounds];
422 [self addSubview:_customContentFixedOverlayView.get()];
423 }
424
425 if (self.isFirstResponder) {
426 UIView *currentContentView = self._currentContentView;
427 if (currentContentView == _contentView ? [_contentView canBecomeFirstResponderForWebView] : currentContentView.canBecomeFirstResponder)
428 [currentContentView becomeFirstResponder];
429 }

495static WebCore::Color scrollViewBackgroundColor(WKWebView *webView)
496{
497 if (!webView.opaque)
498 return WebCore::Color::transparentBlack;
499
500#if HAVE(OS_DARK_MODE_SUPPORT)
501 WebCore::LocalCurrentTraitCollection localTraitCollection(webView.traitCollection);
502#endif
503
504 WebCore::Color color = baseScrollViewBackgroundColor(webView);
505
506 if (!color.isValid() && webView->_contentView)
507 color = [webView->_contentView backgroundColor].CGColor;
508
509 if (!color.isValid()) {
510#if HAVE(OS_DARK_MODE_SUPPORT)
511 color = UIColor.systemBackgroundColor.CGColor;
512#else
513 color = WebCore::Color::white;
514#endif
515 }
516
517 CGFloat zoomScale = contentZoomScale(webView);
518 CGFloat minimumZoomScale = [webView->_scrollView minimumZoomScale];
519 if (zoomScale < minimumZoomScale) {
520 CGFloat slope = 12;
521 CGFloat opacity = std::max<CGFloat>(1 - slope * (minimumZoomScale - zoomScale), 0);
522 color = color.colorWithAlpha(opacity);
523 }
524
525 return color;
526}

587- (UIEdgeInsets)_contentInsetsFromSystemMinimumLayoutMargins
588{
589 if (auto controller = [UIViewController _viewControllerForFullScreenPresentationFromView:self]) {
590 auto margins = controller.systemMinimumLayoutMargins;
591 auto insets = UIEdgeInsetsMake(margins.top, margins.leading, margins.bottom, margins.trailing);
592 if (_page && _page->userInterfaceLayoutDirection() == WebCore::UserInterfaceLayoutDirection::RTL)
593 std::swap(insets.left, insets.right);
594
595 if (auto view = controller.viewIfLoaded) {
596 auto adjustInsetEdge = [](CGFloat& insetEdge, CGFloat distanceFromEdge) {
597 insetEdge -= std::max<CGFloat>(0, distanceFromEdge);
598 insetEdge = std::max<CGFloat>(0, insetEdge);
599 };
600
601 auto viewBounds = view.bounds;
602 auto webViewBoundsInView = [self convertRect:self.bounds toView:view];
603 adjustInsetEdge(insets.top, CGRectGetMinY(webViewBoundsInView) - CGRectGetMinY(viewBounds));
604 adjustInsetEdge(insets.left, CGRectGetMinX(webViewBoundsInView) - CGRectGetMinX(viewBounds));
605 adjustInsetEdge(insets.bottom, CGRectGetMaxY(viewBounds) - CGRectGetMaxY(webViewBoundsInView));
606 adjustInsetEdge(insets.right, CGRectGetMaxX(viewBounds) - CGRectGetMaxX(webViewBoundsInView));
607 }
608 return insets;
609 }
610 return UIEdgeInsetsZero;
611}

1246- (void)_zoomToFocusRect:(const WebCore::FloatRect&)focusedElementRectInDocumentCoordinates selectionRect:(const WebCore::FloatRect&)selectionRectInDocumentCoordinates

1271 minimumDistanceFromKeyboardToTriggerScroll = 50;
1272 } else {
1273 visibleSize.height = heightVisibleBelowFormAssistant;
1274 visibleOffsetFromTop = CGRectGetMaxY(intersectionBetweenScrollViewAndFormAssistant) - CGRectGetMinY(visibleScrollViewBoundsInWebViewCoordinates);

1278 // Zoom around the element's bounding frame. We use a "standard" size to determine the proper frame.
1279 double currentScale = contentZoomScale(self);
1280 double scale = currentScale;
1281 if (allowScaling) {
1282#if PLATFORM(WATCHOS)
1283 const CGFloat minimumMarginForZoomingToEntireFocusRectInWebViewCoordinates = 10;

1295#endif

1303 focusedElementRectInNewScale.moveBy([_contentView frame].origin);
1304

1317

1319 if (selectionRectIsNotNull && CGRectContainsRect(currentlyVisibleRegionInWebViewCoordinates, [self convertRect:selectionRectInDocumentCoordinates fromView:_contentView.get()]))
1320 return;
1321 }

1323 // We want to center the focused element within the viewport, with as much spacing on all sides as
1324 // we can get based on the visible area after zooming. The spacing in either dimension is half the
1325 // difference between the size of the DOM node and the size of the visible frame.
1326 // If the element is too wide to be horizontally centered or too tall to be vertically centered, we
1327 // instead scroll such that the left edge or top edge of the element is within the left half or top
1328 // half of the viewport, respectively.

1330 CGFloat verticalSpaceInWebViewCoordinates = (visibleSize.height - focusedElementRectInNewScale.height()) / 2.0;

1343 else {

1355 }

1361 documentBoundsInNewScale.moveBy([_contentView frame].origin);

1374 minimumAllowableVerticalOffsetInWebViewCoordinates = CGRectGetMaxY(selectionRectInNewScale) + caretOffsetFromWindowEdge - visibleSize.height - visibleOffsetFromTop;
1375 maximumAllowableHorizontalOffsetInWebViewCoordinates = std::min<CGFloat>(maximumAllowableHorizontalOffsetInWebViewCoordinates, CGRectGetMinX(selectionRectInNewScale) - caretOffsetFromWindowEdge);
1376 maximumAllowableVerticalOffsetInWebViewCoordinates = std::min<CGFloat>(maximumAllowableVerticalOffsetInWebViewCoordinates, CGRectGetMinY(selectionRectInNewScale) - caretOffsetFromWindowEdge - visibleOffsetFromTop);
1377 }
1378
1379 // Constrain the left edge in document coordinates so that:
1380 // - it isn't so small that the scrollVisibleRect isn't visible on the screen
1381 // - it isn't so great that the document's right edge is less than the right edge of the screen
1382 topLeft.x = clampTo<CGFloat>(topLeft.x, minimumAllowableHorizontalOffsetInWebViewCoordinates, maximumAllowableHorizontalOffsetInWebViewCoordinates);
1383
1384 // Constrain the top edge in document coordinates so that:
1385 // - it isn't so small that the scrollVisibleRect isn't visible on the screen
1386 // - it isn't so great that the document's bottom edge is higher than the top of the form assistant
1387 topLeft.y = clampTo<CGFloat>(topLeft.y, minimumAllowableVerticalOffsetInWebViewCoordinates, maximumAllowableVerticalOffsetInWebViewCoordinates);
1388

1390 // This looks unintuitive, but is necessary in order to precisely center the focused element in the visible area.
1391 // The top left position already accounts for top and left obscured insets - i.e., a topLeft of (0, 0) corresponds
1392 // to the top- and left-most point below (and to the right of) the top inset area and left inset areas, respectively.
1393 // However, when telling WKScrollView to scroll to a given center position, this center position is computed relative
1394 // to the coordinate space of the scroll view. Thus, to compute our center position from the top left position, we
1395 // need to first move the top left position up and to the left, and then add half the width and height of the content
1396 // area (including obscured insets).
1397 topLeft.x -= _obscuredInsets.left;
1398 topLeft.y -= _obscuredInsets.top;
1399 }
1400
1401 WebCore::FloatPoint newCenter = CGPointMake(topLeft.x + CGRectGetWidth(self.bounds) / 2, topLeft.y + CGRectGetHeight(self.bounds) / 2);
1402

1407

1409 newCenter.scale(1 / scale);
1410 [_scrollView _zoomToCenter:newCenter scale:scale duration:formControlZoomAnimationDuration force:YES];
1411}
