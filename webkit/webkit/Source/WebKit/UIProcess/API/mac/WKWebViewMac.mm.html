46_WKOverlayScrollbarStyle toAPIScrollbarStyle(std::optional<WebCore::ScrollbarOverlayStyle> coreScrollbarStyle)
47{
48 if (!coreScrollbarStyle)
49 return _WKOverlayScrollbarStyleAutomatic;
50
51 switch (coreScrollbarStyle.value()) {
52 case WebCore::ScrollbarOverlayStyleDark:
53 return _WKOverlayScrollbarStyleDark;
54 case WebCore::ScrollbarOverlayStyleLight:
55 return _WKOverlayScrollbarStyleLight;
56 case WebCore::ScrollbarOverlayStyleDefault:
57 return _WKOverlayScrollbarStyleDefault;
58 }
59 ASSERT_NOT_REACHED();
60 return _WKOverlayScrollbarStyleAutomatic;
61}
62
63std::optional<WebCore::ScrollbarOverlayStyle> toCoreScrollbarStyle(_WKOverlayScrollbarStyle scrollbarStyle)
64{
65 switch (scrollbarStyle) {
66 case _WKOverlayScrollbarStyleDark:
67 return WebCore::ScrollbarOverlayStyleDark;
68 case _WKOverlayScrollbarStyleLight:
69 return WebCore::ScrollbarOverlayStyleLight;
70 case _WKOverlayScrollbarStyleDefault:
71 return WebCore::ScrollbarOverlayStyleDefault;
72 case _WKOverlayScrollbarStyleAutomatic:
73 break;
74 }
75 return std::nullopt;
76}
