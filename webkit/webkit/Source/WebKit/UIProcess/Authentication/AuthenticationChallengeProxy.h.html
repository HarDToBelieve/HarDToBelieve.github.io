26#pragma once
27
28#include "APIObject.h"
29#include <WebCore/AuthenticationChallenge.h>
30#include <wtf/WeakPtr.h>
31
32namespace IPC {
33class Connection;
34}
35
36namespace WebKit {
37
38class AuthenticationDecisionListener;
39class SecKeyProxyStore;
40class WebCredential;
41class WebProtectionSpace;
42
43class AuthenticationChallengeProxy : public API::ObjectImpl<API::Object::Type::AuthenticationChallenge> {
44public:
45 static Ref<AuthenticationChallengeProxy> create(WebCore::AuthenticationChallenge&& authenticationChallenge, uint64_t challengeID, Ref<IPC::Connection>&& connection, WeakPtr<SecKeyProxyStore>&& secKeyProxyStore)
46 {
47 return adoptRef(*new AuthenticationChallengeProxy(WTFMove(authenticationChallenge), challengeID, WTFMove(connection), WTFMove(secKeyProxyStore)));
48 }
49
50 WebCredential* proposedCredential() const;
51 WebProtectionSpace* protectionSpace() const;
52
53 AuthenticationDecisionListener& listener() const { return m_listener.get(); }
54 const WebCore::AuthenticationChallenge& core() { return m_coreAuthenticationChallenge; }
55
56private:
57 AuthenticationChallengeProxy(WebCore::AuthenticationChallenge&&, uint64_t challengeID, Ref<IPC::Connection>&&, WeakPtr<SecKeyProxyStore>&&);
58
59#if HAVE(SEC_KEY_PROXY)
60 static void sendClientCertificateCredentialOverXpc(IPC::Connection&, SecKeyProxyStore&, uint64_t challengeID, const WebCore::Credential&);
61#endif
62
63 WebCore::AuthenticationChallenge m_coreAuthenticationChallenge;
64 mutable RefPtr<WebCredential> m_webCredential;
65 mutable RefPtr<WebProtectionSpace> m_webProtectionSpace;
66 Ref<AuthenticationDecisionListener> m_listener;
67};

69} // namespace WebKit
