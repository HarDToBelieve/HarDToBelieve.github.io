26#ifndef WebCredential_h
27#define WebCredential_h
28
29#include "APIObject.h"
30#include "APIString.h"
31#include <WebCore/Credential.h>
32
33namespace WebKit {
34
35class WebCertificateInfo;
36
37class WebCredential : public API::ObjectImpl<API::Object::Type::Credential> {
38public:
39 ~WebCredential();
40
41 static Ref<WebCredential> create(const WebCore::Credential& credential)
42 {
43 return adoptRef(*new WebCredential(credential));
44 }
45
46 static Ref<WebCredential> create(WebCertificateInfo* certificateInfo)
47 {
48 return adoptRef(*new WebCredential(certificateInfo));
49 }
50
51 const WebCore::Credential& credential();
52
53private:
54 explicit WebCredential(const WebCore::Credential&);
55 explicit WebCredential(WebCertificateInfo*);
56
57 WebCore::Credential m_coreCredential;
58};
59
60} // namespace WebKit
61
62#endif // WebCredential_h
