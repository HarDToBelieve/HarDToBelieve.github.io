26#ifndef WebProtectionSpace_h
27#define WebProtectionSpace_h
28
29#include "APIObject.h"
30#include <WebCore/ProtectionSpace.h>
31
32namespace WebKit {
33
34class WebProtectionSpace : public API::ObjectImpl<API::Object::Type::ProtectionSpace> {
35public:
36 static Ref<WebProtectionSpace> create(const WebCore::ProtectionSpace& protectionSpace)
37 {
38 return adoptRef(*new WebProtectionSpace(protectionSpace));
39 }
40
41 const String& protocol() const;
42 const String& host() const;
43 int port() const;
44 const String& realm() const;
45 bool isProxy() const;
46 WebCore::ProtectionSpaceServerType serverType() const;
47 bool receivesCredentialSecurely() const;
48 WebCore::ProtectionSpaceAuthenticationScheme authenticationScheme() const;
49
50 const WebCore::ProtectionSpace& protectionSpace() const { return m_coreProtectionSpace; }
51
52private:
53 explicit WebProtectionSpace(const WebCore::ProtectionSpace&);
54
55 WebCore::ProtectionSpace m_coreProtectionSpace;
56};
57
58} // namespace WebKit
59
60#endif // WebProtectionSpace_h
