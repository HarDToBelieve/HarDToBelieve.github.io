39static RetainPtr<SecCertificateRef> leafCertificate(const CertificateInfo& certificateInfo)
40{
41#if HAVE(SEC_TRUST_SERIALIZATION)
42
43#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
44 if (certificateInfo.type() == CertificateInfo::Type::Trust)
45 return checked_cf_cast<SecCertificateRef>(CFArrayGetValueAtIndex(adoptCF(SecTrustCopyCertificateChain(certificateInfo.trust())).get(), 0));
46#else
47 if (certificateInfo.type() == CertificateInfo::Type::Trust)
48 return SecTrustGetCertificateAtIndex(certificateInfo.trust(), 0);
49#endif // HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
50
51#endif // HAVE(SEC_TRUST_SERIALIZATION)
52 ASSERT(certificateInfo.type() == CertificateInfo::Type::CertificateChain);
53 ASSERT(CFArrayGetCount(certificateInfo.certificateChain()));
54 return checked_cf_cast<SecCertificateRef>(CFArrayGetValueAtIndex(certificateInfo.certificateChain(), 0));
55}
56
57static NSArray *chain(const CertificateInfo& certificateInfo)
58{
59#if HAVE(SEC_TRUST_SERIALIZATION)
60 if (certificateInfo.type() == CertificateInfo::Type::Trust) {
61 CFIndex count = SecTrustGetCertificateCount(certificateInfo.trust());
62 if (count < 2)
63 return nil;
64
65#if HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
66 return (__bridge NSArray *)adoptCF(SecTrustCopyCertificateChain(certificateInfo.trust())).autorelease();
67#else
68 NSMutableArray *array = [NSMutableArray array];
69 for (CFIndex i = 1; i < count; ++i)
70 [array addObject:(id)SecTrustGetCertificateAtIndex(certificateInfo.trust(), i)];
71
72 return array;
73#endif // HAVE(SEC_TRUST_COPY_CERTIFICATE_CHAIN)
74 }
75#endif // HAVE(SEC_TRUST_SERIALIZATION)
76 ASSERT(certificateInfo.type() == CertificateInfo::Type::CertificateChain);
77 CFIndex chainCount = CFArrayGetCount(certificateInfo.certificateChain());
78 return chainCount > 1 ? [(__bridge NSArray *)certificateInfo.certificateChain() subarrayWithRange:NSMakeRange(1, chainCount - 1)] : nil;
79}

81WebCredential::WebCredential(WebCertificateInfo* certificateInfo)
82{
83 if (!certificateInfo || certificateInfo->certificateInfo().isEmpty())
84 return;
85
86 // The passed-in certificate chain includes the identity certificate at index 0, and additional certificates starting at index 1.
87 SecIdentityRef identity;
88 OSStatus result = SecIdentityCreateWithCertificate(NULL, leafCertificate(certificateInfo->certificateInfo()).get(), &identity);
89 if (result != errSecSuccess) {
90 LOG_ERROR("Unable to create SecIdentityRef with certificate - %i", result);
91 return;
92 }
93
94 m_coreCredential = Credential([NSURLCredential credentialWithIdentity:identity certificates:chain(certificateInfo->certificateInfo()) persistence:NSURLCredentialPersistenceNone]);
95}
