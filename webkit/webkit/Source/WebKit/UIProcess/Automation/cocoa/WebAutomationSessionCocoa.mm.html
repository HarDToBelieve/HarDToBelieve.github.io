42std::optional<String> WebAutomationSession::platformGetBase64EncodedPNGData(const ShareableBitmap::Handle& imageDataHandle)
43{
44 auto bitmap = ShareableBitmap::create(imageDataHandle, SharedMemory::Protection::ReadOnly);
45 if (!bitmap)
46 return std::nullopt;
47
48 RetainPtr<CGImageRef> cgImage = bitmap->makeCGImage();
49 RetainPtr<NSMutableData> imageData = adoptNS([[NSMutableData alloc] init]);
50ALLOW_DEPRECATED_DECLARATIONS_BEGIN
51 RetainPtr<CGImageDestinationRef> destination = adoptCF(CGImageDestinationCreateWithData((CFMutableDataRef)imageData.get(), kUTTypePNG, 1, 0));
52ALLOW_DEPRECATED_DECLARATIONS_END
53 if (!destination)
54 return std::nullopt;
55
56 CGImageDestinationAddImage(destination.get(), cgImage.get(), 0);
57 CGImageDestinationFinalize(destination.get());
58
59 return String([imageData base64EncodedStringWithOptions:0]);
60}

62std::optional<String> WebAutomationSession::platformGenerateLocalFilePathForRemoteFile(const String& remoteFilePath, const String& base64EncodedFileContents)
63{
64 RetainPtr<NSData> fileContents = adoptNS([[NSData alloc] initWithBase64EncodedString:base64EncodedFileContents options:0]);
65 if (!fileContents) {
66 LOG_ERROR("WebAutomationSession: unable to decode base64-encoded file contents.");
67 return std::nullopt;
68 }
69
70 NSString *temporaryDirectory = FileSystem::createTemporaryDirectory(@"WebDriver");
71 NSURL *remoteFile = [NSURL fileURLWithPath:remoteFilePath isDirectory:NO];
72 NSString *localFilePath = [temporaryDirectory stringByAppendingPathComponent:remoteFile.lastPathComponent];
73
74 NSError *fileWriteError;
75 [fileContents.get() writeToFile:localFilePath options:NSDataWritingAtomic error:&fileWriteError];
76 if (fileWriteError) {
77 LOG_ERROR("WebAutomationSession: Error writing image data to temporary file: %@", fileWriteError);
78 return std::nullopt;
79 }
80
81 return String(localFilePath);
82}

84std::optional<unichar> WebAutomationSession::charCodeForVirtualKey(Inspector::Protocol::Automation::VirtualKey key) const
85{
86 switch (key) {
87 case Inspector::Protocol::Automation::VirtualKey::Shift:
88 case Inspector::Protocol::Automation::VirtualKey::Control:
89 case Inspector::Protocol::Automation::VirtualKey::Alternate:
90 case Inspector::Protocol::Automation::VirtualKey::Meta:
91 case Inspector::Protocol::Automation::VirtualKey::Command:
92 return std::nullopt;
93 case Inspector::Protocol::Automation::VirtualKey::Help:
94 return NSHelpFunctionKey;
95 case Inspector::Protocol::Automation::VirtualKey::Backspace:
96 return NSBackspaceCharacter;
97 case Inspector::Protocol::Automation::VirtualKey::Tab:
98 return NSTabCharacter;
99 case Inspector::Protocol::Automation::VirtualKey::Clear:
100 return NSClearLineFunctionKey;
101 case Inspector::Protocol::Automation::VirtualKey::Enter:
102 return NSEnterCharacter;
103 case Inspector::Protocol::Automation::VirtualKey::Pause:
104 return NSPauseFunctionKey;
105 case Inspector::Protocol::Automation::VirtualKey::Cancel:
106 // The 'cancel' key does not exist on Apple keyboards and has no keycode.
107 // According to the internet its functionality is similar to 'Escape'.
108 case Inspector::Protocol::Automation::VirtualKey::Escape:
109 return 0x1B;
110 case Inspector::Protocol::Automation::VirtualKey::PageUp:
111 return NSPageUpFunctionKey;
112 case Inspector::Protocol::Automation::VirtualKey::PageDown:
113 return NSPageDownFunctionKey;
114 case Inspector::Protocol::Automation::VirtualKey::End:
115 return NSEndFunctionKey;
116 case Inspector::Protocol::Automation::VirtualKey::Home:
117 return NSHomeFunctionKey;
118 case Inspector::Protocol::Automation::VirtualKey::LeftArrow:
119 return NSLeftArrowFunctionKey;
120 case Inspector::Protocol::Automation::VirtualKey::UpArrow:
121 return NSUpArrowFunctionKey;
122 case Inspector::Protocol::Automation::VirtualKey::RightArrow:
123 return NSRightArrowFunctionKey;
124 case Inspector::Protocol::Automation::VirtualKey::DownArrow:
125 return NSDownArrowFunctionKey;
126 case Inspector::Protocol::Automation::VirtualKey::Insert:
127 return NSInsertFunctionKey;
128 case Inspector::Protocol::Automation::VirtualKey::Delete:
129 return NSDeleteFunctionKey;
130 case Inspector::Protocol::Automation::VirtualKey::Space:
131 return ' ';
132 case Inspector::Protocol::Automation::VirtualKey::Semicolon:
133 return ';';
134 case Inspector::Protocol::Automation::VirtualKey::Equals:
135 return '=';
136 case Inspector::Protocol::Automation::VirtualKey::Return:
137 return NSCarriageReturnCharacter;
138 case Inspector::Protocol::Automation::VirtualKey::NumberPad0:
139 return '0';
140 case Inspector::Protocol::Automation::VirtualKey::NumberPad1:
141 return '1';
142 case Inspector::Protocol::Automation::VirtualKey::NumberPad2:
143 return '2';
144 case Inspector::Protocol::Automation::VirtualKey::NumberPad3:
145 return '3';
146 case Inspector::Protocol::Automation::VirtualKey::NumberPad4:
147 return '4';
148 case Inspector::Protocol::Automation::VirtualKey::NumberPad5:
149 return '5';
150 case Inspector::Protocol::Automation::VirtualKey::NumberPad6:
151 return '6';
152 case Inspector::Protocol::Automation::VirtualKey::NumberPad7:
153 return '7';
154 case Inspector::Protocol::Automation::VirtualKey::NumberPad8:
155 return '8';
156 case Inspector::Protocol::Automation::VirtualKey::NumberPad9:
157 return '9';
158 case Inspector::Protocol::Automation::VirtualKey::NumberPadMultiply:
159 return '*';
160 case Inspector::Protocol::Automation::VirtualKey::NumberPadAdd:
161 return '+';
162 case Inspector::Protocol::Automation::VirtualKey::NumberPadSubtract:
163 return '-';
164 case Inspector::Protocol::Automation::VirtualKey::NumberPadSeparator:
165 // The 'Separator' key is only present on a few international keyboards.
166 // It is usually mapped to the same character as Decimal ('.' or ',').
167 case Inspector::Protocol::Automation::VirtualKey::NumberPadDecimal:
168 return '.';
169 case Inspector::Protocol::Automation::VirtualKey::NumberPadDivide:
170 return '/';
171 case Inspector::Protocol::Automation::VirtualKey::Function1:
172 return NSF1FunctionKey;
173 case Inspector::Protocol::Automation::VirtualKey::Function2:
174 return NSF2FunctionKey;
175 case Inspector::Protocol::Automation::VirtualKey::Function3:
176 return NSF3FunctionKey;
177 case Inspector::Protocol::Automation::VirtualKey::Function4:
178 return NSF4FunctionKey;
179 case Inspector::Protocol::Automation::VirtualKey::Function5:
180 return NSF5FunctionKey;
181 case Inspector::Protocol::Automation::VirtualKey::Function6:
182 return NSF6FunctionKey;
183 case Inspector::Protocol::Automation::VirtualKey::Function7:
184 return NSF7FunctionKey;
185 case Inspector::Protocol::Automation::VirtualKey::Function8:
186 return NSF8FunctionKey;
187 case Inspector::Protocol::Automation::VirtualKey::Function9:
188 return NSF9FunctionKey;
189 case Inspector::Protocol::Automation::VirtualKey::Function10:
190 return NSF10FunctionKey;
191 case Inspector::Protocol::Automation::VirtualKey::Function11:
192 return NSF11FunctionKey;
193 case Inspector::Protocol::Automation::VirtualKey::Function12:
194 return NSF12FunctionKey;
195 default:
196 return std::nullopt;
197 }
198}

200std::optional<unichar> WebAutomationSession::charCodeIgnoringModifiersForVirtualKey(Inspector::Protocol::Automation::VirtualKey key) const
201{
202 switch (key) {
203 case Inspector::Protocol::Automation::VirtualKey::NumberPadMultiply:
204 return '8';
205 case Inspector::Protocol::Automation::VirtualKey::NumberPadAdd:
206 return '=';
207 default:
208 return charCodeForVirtualKey(key);
209 }
