47void WebAutomationSession::inspectBrowsingContext(const Inspector::Protocol::Automation::BrowsingContextHandle& handle, std::optional<bool>&& enableAutoCapturing, Ref<InspectBrowsingContextCallback>&& callback)
48{
49 WebPageProxy* page = webPageProxyForHandle(handle);
50 if (!page)
51 ASYNC_FAIL_WITH_PREDEFINED_ERROR(WindowNotFound);
52
53 if (auto callback = m_pendingInspectorCallbacksPerPage.take(page->identifier()))
54 callback->sendFailure(STRING_FOR_PREDEFINED_ERROR_NAME(Timeout));
55 m_pendingInspectorCallbacksPerPage.set(page->identifier(), WTFMove(callback));
56
57 // Don't bring the inspector to front since this may be done automatically.
58 // We just want it loaded so it can pause if a breakpoint is hit during a command.
59 if (page->inspector()) {
60 page->inspector()->show();
61
62 // Start collecting profile information immediately so the entire session is captured.
63 if (enableAutoCapturing && *enableAutoCapturing)
64 page->inspector()->togglePageProfiling();
65 }
66}

71static const void *synthesizedAutomationEventAssociatedObjectKey = &synthesizedAutomationEventAssociatedObjectKey;

73void WebAutomationSession::sendSynthesizedEventsToPage(WebPageProxy& page, NSArray *eventsToSend)
74{
75 NSWindow *window = page.platformWindow();
76 [window makeKeyAndOrderFront:nil];
77 page.makeFirstResponder();
78
79 for (NSEvent *event in eventsToSend) {
80 LOG(Automation, "Sending event[%p] to window[%p]: %@", event, window, event);
81
82 // Take focus back in case the Inspector became focused while the prior command or
83 // NSEvent was delivered to the window.
84 [window makeKeyAndOrderFront:nil];
85 page.makeFirstResponder();
86
87 markEventAsSynthesizedForAutomation(event);
88 [window sendEvent:event];
89 }
90}

92void WebAutomationSession::markEventAsSynthesizedForAutomation(NSEvent *event)
93{
94 objc_setAssociatedObject(event, &synthesizedAutomationEventAssociatedObjectKey, m_sessionIdentifier, OBJC_ASSOCIATION_RETAIN_NONATOMIC);
95}

97bool WebAutomationSession::wasEventSynthesizedForAutomation(NSEvent *event)
98{
99 NSString *senderSessionIdentifier = objc_getAssociatedObject(event, &synthesizedAutomationEventAssociatedObjectKey);
100 if ([senderSessionIdentifier isEqualToString:m_sessionIdentifier])
101 return true;
102
103 switch (event.type) {
104 case NSEventTypeLeftMouseDown:
105 case NSEventTypeLeftMouseDragged:
106 case NSEventTypeLeftMouseUp:
107 case NSEventTypeMouseMoved:
108 case NSEventTypeOtherMouseDown:
109 case NSEventTypeOtherMouseDragged:
110 case NSEventTypeOtherMouseUp:
111 case NSEventTypeRightMouseDown:
112 case NSEventTypeRightMouseDragged:
113 case NSEventTypeRightMouseUp:
114 // Use this as a backup for checking mouse events, which are frequently copied
115 // and/or faked by AppKit, causing them to lose their associated object tag.
116 return event.eventNumber == synthesizedMouseEventMagicEventNumber;
117 default:
118 break;
119 }
120
121 return false;
122}

139void WebAutomationSession::platformSimulateMouseInteraction(WebPageProxy& page, MouseInteraction interaction, MouseButton button, const WebCore::IntPoint& locationInViewport, OptionSet<WebEvent::Modifier> keyModifiers, const String& pointerType)
140{
141 UNUSED_PARAM(pointerType);
142
143 IntRect windowRect;
144
145 IntPoint locationInView = locationInViewport + IntPoint(0, page.topContentInset());
146 page.rootViewToWindow(IntRect(locationInView, IntSize()), windowRect);
147 IntPoint locationInWindow = windowRect.location();
148
149 NSEventModifierFlags modifiers = 0;
150 if (keyModifiers.contains(WebEvent::Modifier::MetaKey))
151 modifiers |= NSEventModifierFlagCommand;
152 if (keyModifiers.contains(WebEvent::Modifier::AltKey))
153 modifiers |= NSEventModifierFlagOption;
154 if (keyModifiers.contains(WebEvent::Modifier::ControlKey))
155 modifiers |= NSEventModifierFlagControl;
156 if (keyModifiers.contains(WebEvent::Modifier::ShiftKey))
157 modifiers |= NSEventModifierFlagShift;
158 if (keyModifiers.contains(WebEvent::Modifier::CapsLockKey))
159 modifiers |= NSEventModifierFlagCapsLock;
160
161 NSTimeInterval timestamp = [NSDate timeIntervalSinceReferenceDate];
162 NSWindow *window = page.platformWindow();
163 NSInteger windowNumber = window.windowNumber;
164
165 NSEventType downEventType = (NSEventType)0;
166 NSEventType dragEventType = (NSEventType)0;
167 NSEventType upEventType = (NSEventType)0;
168 switch (automationMouseButtonToPlatformMouseButton(button)) {
169 case WebMouseEvent::NoButton:
170 downEventType = upEventType = dragEventType = NSEventTypeMouseMoved;
171 break;
172 case WebMouseEvent::LeftButton:
173 downEventType = NSEventTypeLeftMouseDown;
174 dragEventType = NSEventTypeLeftMouseDragged;
175 upEventType = NSEventTypeLeftMouseUp;
176 break;
177 case WebMouseEvent::MiddleButton:
178 downEventType = NSEventTypeOtherMouseDown;
179 dragEventType = NSEventTypeLeftMouseDragged;
180 upEventType = NSEventTypeOtherMouseUp;
181 break;
182 case WebMouseEvent::RightButton:
183 downEventType = NSEventTypeRightMouseDown;
184 upEventType = NSEventTypeRightMouseUp;
185 break;
186 }
187
188 auto eventsToBeSent = adoptNS([[NSMutableArray alloc] init]);
189
190 NSInteger eventNumber = synthesizedMouseEventMagicEventNumber;
191
192 switch (interaction) {
193 case MouseInteraction::Move:
194 ASSERT(dragEventType);
195 [eventsToBeSent addObject:[NSEvent mouseEventWithType:dragEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:0 pressure:0.0f]];
196 break;
197 case MouseInteraction::Down:
198 ASSERT(downEventType);
199
200 // Hard-code the click count to one, since clients don't expect successive simulated
201 // down/up events to be potentially counted as a double click event.
202 [eventsToBeSent addObject:[NSEvent mouseEventWithType:downEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:1 pressure:WebCore::ForceAtClick]];
203 break;
204 case MouseInteraction::Up:
205 ASSERT(upEventType);
206
207 // Hard-code the click count to one, since clients don't expect successive simulated
208 // down/up events to be potentially counted as a double click event.
209 [eventsToBeSent addObject:[NSEvent mouseEventWithType:upEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:1 pressure:0.0f]];
210 break;
211 case MouseInteraction::SingleClick:
212 ASSERT(upEventType);
213 ASSERT(downEventType);
214
215 // Send separate down and up events. WebCore will see this as a single-click event.
216 [eventsToBeSent addObject:[NSEvent mouseEventWithType:downEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:1 pressure:WebCore::ForceAtClick]];
217 [eventsToBeSent addObject:[NSEvent mouseEventWithType:upEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:1 pressure:0.0f]];
218 break;
219 case MouseInteraction::DoubleClick:
220 ASSERT(upEventType);
221 ASSERT(downEventType);
222
223 // Send multiple down and up events with proper click count.
224 // WebCore will see this as a single-click event then double-click event.
225 [eventsToBeSent addObject:[NSEvent mouseEventWithType:downEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:1 pressure:WebCore::ForceAtClick]];
226 [eventsToBeSent addObject:[NSEvent mouseEventWithType:upEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:1 pressure:0.0f]];
227 [eventsToBeSent addObject:[NSEvent mouseEventWithType:downEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:2 pressure:WebCore::ForceAtClick]];
228 [eventsToBeSent addObject:[NSEvent mouseEventWithType:upEventType location:locationInWindow modifierFlags:modifiers timestamp:timestamp windowNumber:windowNumber context:nil eventNumber:eventNumber clickCount:2 pressure:0.0f]];
229 }
230
231 sendSynthesizedEventsToPage(page, eventsToBeSent.get());
232}

234OptionSet<WebEvent::Modifier> WebAutomationSession::platformWebModifiersFromRaw(unsigned modifiers)
235{
236 OptionSet<WebEvent::Modifier> webModifiers;
237
238 if (modifiers & NSEventModifierFlagCommand)
239 webModifiers.add(WebEvent::Modifier::MetaKey);
240 if (modifiers & NSEventModifierFlagOption)
241 webModifiers.add(WebEvent::Modifier::AltKey);
242 if (modifiers & NSEventModifierFlagControl)
243 webModifiers.add(WebEvent::Modifier::ControlKey);
244 if (modifiers & NSEventModifierFlagShift)
245 webModifiers.add(WebEvent::Modifier::ShiftKey);
246 if (modifiers & NSEventModifierFlagCapsLock)
247 webModifiers.add(WebEvent::Modifier::CapsLockKey);
248
249 return webModifiers;
250}

628void WebAutomationSession::platformSimulateKeyboardInteraction(WebPageProxy& page, KeyboardInteraction interaction, WTF::Variant<VirtualKey, CharKey>&& key)
629{
630 // FIXME: this function and the Automation protocol enum should probably adopt key names
631 // from W3C UIEvents standard. For more details: https://w3c.github.io/uievents-code/
632
633 bool isStickyModifier = false;
634 NSEventModifierFlags changedModifiers = 0;
635 int keyCode = 0;
636 std::optional<unichar> charCode;
637 std::optional<unichar> charCodeIgnoringModifiers;
638
639 WTF::switchOn(key,
640 [&] (VirtualKey virtualKey) {
641 isStickyModifier = virtualKeyHasStickyModifier(virtualKey);
642 changedModifiers = eventModifierFlagsForVirtualKey(virtualKey);
643 keyCode = keyCodeForVirtualKey(virtualKey);
644 charCode = charCodeForVirtualKey(virtualKey);
645 charCodeIgnoringModifiers = charCodeIgnoringModifiersForVirtualKey(virtualKey);
646 },
647 [&] (CharKey charKey) {
648 keyCode = keyCodeForCharKey(charKey);
649 charCode = (unichar)charKey;
650 charCodeIgnoringModifiers = (unichar)charKey;
651 }
652 );
653
654 // FIXME: consider using AppKit SPI to normalize 'characters', i.e., changing * to Shift-8,
655 // and passing that in to charactersIgnoringModifiers. We could hardcode this for ASCII if needed.
656 NSString *characters = charCode ? [NSString stringWithCharacters:&charCode.value() length:1] : nil;
657 NSString *unmodifiedCharacters = charCodeIgnoringModifiers ? [NSString stringWithCharacters:&charCodeIgnoringModifiers.value() length:1] : nil;
658
659 auto eventsToBeSent = adoptNS([[NSMutableArray alloc] init]);
660
661 // FIXME: this timestamp is not even close to matching native events. Find out how to get closer.
662 NSTimeInterval timestamp = [NSDate timeIntervalSinceReferenceDate];
663 NSWindow *window = page.platformWindow();
664 NSInteger windowNumber = window.windowNumber;
665 NSPoint eventPosition = NSMakePoint(0, window.frame.size.height);
666
667 switch (interaction) {
668 case KeyboardInteraction::KeyPress: {
669 NSEventType eventType = isStickyModifier ? NSEventTypeFlagsChanged : NSEventTypeKeyDown;
670 m_currentModifiers |= changedModifiers;
671 [eventsToBeSent addObject:[NSEvent keyEventWithType:eventType location:eventPosition modifierFlags:m_currentModifiers timestamp:timestamp windowNumber:windowNumber context:nil characters:characters charactersIgnoringModifiers:unmodifiedCharacters isARepeat:NO keyCode:keyCode]];
672 break;
673 }
674 case KeyboardInteraction::KeyRelease: {
675 NSEventType eventType = isStickyModifier ? NSEventTypeFlagsChanged : NSEventTypeKeyUp;
676 m_currentModifiers &= ~changedModifiers;
677
678 // When using a physical keyboard, if command is held down, releasing a non-modifier key doesn't send a KeyUp event.
679 bool commandKeyHeldDown = m_currentModifiers & NSEventModifierFlagCommand;
680 if (characters && commandKeyHeldDown)
681 break;
682
683 [eventsToBeSent addObject:[NSEvent keyEventWithType:eventType location:eventPosition modifierFlags:m_currentModifiers timestamp:timestamp windowNumber:windowNumber context:nil characters:characters charactersIgnoringModifiers:unmodifiedCharacters isARepeat:NO keyCode:keyCode]];
684 break;
685 }
686 case KeyboardInteraction::InsertByKey: {
687 // Sticky modifiers should either be 'KeyPress' or 'KeyRelease'.
688 ASSERT(!isStickyModifier);
689 if (isStickyModifier)
690 return;
691
692 m_currentModifiers |= changedModifiers;
693 [eventsToBeSent addObject:[NSEvent keyEventWithType:NSEventTypeKeyDown location:eventPosition modifierFlags:m_currentModifiers timestamp:timestamp windowNumber:windowNumber context:nil characters:characters charactersIgnoringModifiers:unmodifiedCharacters isARepeat:NO keyCode:keyCode]];
694 [eventsToBeSent addObject:[NSEvent keyEventWithType:NSEventTypeKeyUp location:eventPosition modifierFlags:m_currentModifiers timestamp:timestamp windowNumber:windowNumber context:nil characters:characters charactersIgnoringModifiers:unmodifiedCharacters isARepeat:NO keyCode:keyCode]];
695 break;
696 }
697 }
698
699 sendSynthesizedEventsToPage(page, eventsToBeSent.get());
700}

702static BOOL characterIsProducedUsingShift(NSString *characters)
703{
704 ASSERT(characters.length == 1);
705
706 // Per the WebDriver specification, keystrokes are assumed to be produced a standard 108-key US keyboard layout.
707 // If this is no longer the case, then we'll need to use system keyboard SPI to decompose modifiers from keystrokes.
708 unichar c = [characters characterAtIndex:0];
709 if (c > 128)
710 return NO;
711
712 if (c >= 'A' && c <= 'Z')
713 return YES;
714
715 switch (c) {
716 case '~':
717 case '!':
718 case '@':
719 case '#':
720 case '$':
721 case '%':
722 case '^':
723 case '&':
724 case '*':
725 case '(':
726 case ')':
727 case '_':
728 case '+':
729 case '{':
730 case '}':
731 case '|':
732 case ':':
733 case '"':
734 case '<':
735 case '>':
736 case '?':
737 return YES;
738 default:
739 return NO;
740 }
741}

743void WebAutomationSession::platformSimulateKeySequence(WebPageProxy& page, const String& keySequence)
744{
745 auto eventsToBeSent = adoptNS([[NSMutableArray alloc] init]);
746
747 // Split the text into combining character sequences and send each separately.
748 // This has no similarity to how keyboards work when inputting complex text.
749 // This command is more similar to the 'insertText:' editing command, except
750 // that this emits keyup/keydown/keypress events for roughly each character.
751 // This API should move more towards that direction in the future.
752 NSString *text = keySequence;
753
754 NSTimeInterval timestamp = [NSDate timeIntervalSinceReferenceDate];
755 NSWindow *window = page.platformWindow();
756 NSInteger windowNumber = window.windowNumber;
757 NSPoint eventPosition = NSMakePoint(0, window.frame.size.height);
758
759 [text enumerateSubstringsInRange:NSMakeRange(0, text.length) options:NSStringEnumerationByComposedCharacterSequences usingBlock:^(NSString *substring, NSRange substringRange, NSRange enclosingRange, BOOL *stop) {
760
761 // For ASCII characters that are produced using Shift on a US-108 key keyboard layout,
762 // WebDriver expects these to be delivered as [shift-down, key-down, key-up, shift-up]
763 // keystrokes unless the shift key is already pressed down from an earlier interaction.
764 BOOL shiftAlreadyPressed = m_currentModifiers & NSEventModifierFlagShift;
765 BOOL shouldPressShift = !shiftAlreadyPressed && substringRange.length == 1 && characterIsProducedUsingShift(substring);
766 NSEventModifierFlags modifiersForKeystroke = shouldPressShift ? m_currentModifiers | NSEventModifierFlagShift : m_currentModifiers;
767
768 if (shouldPressShift)
769 [eventsToBeSent addObject:[NSEvent keyEventWithType:NSEventTypeFlagsChanged location:eventPosition modifierFlags:modifiersForKeystroke timestamp:timestamp windowNumber:windowNumber context:nil characters:@"" charactersIgnoringModifiers:@"" isARepeat:NO keyCode:kVK_Shift]];
770
771 [eventsToBeSent addObject:[NSEvent keyEventWithType:NSEventTypeKeyDown location:eventPosition modifierFlags:modifiersForKeystroke timestamp:timestamp windowNumber:windowNumber context:nil characters:substring charactersIgnoringModifiers:substring isARepeat:NO keyCode:0]];
772 [eventsToBeSent addObject:[NSEvent keyEventWithType:NSEventTypeKeyUp location:eventPosition modifierFlags:modifiersForKeystroke timestamp:timestamp windowNumber:windowNumber context:nil characters:substring charactersIgnoringModifiers:substring isARepeat:NO keyCode:0]];
773
774 if (shouldPressShift)
775 [eventsToBeSent addObject:[NSEvent keyEventWithType:NSEventTypeFlagsChanged location:eventPosition modifierFlags:m_currentModifiers timestamp:timestamp windowNumber:windowNumber context:nil characters:@"" charactersIgnoringModifiers:@"" isARepeat:NO keyCode:kVK_Shift]];
776 }];
777
778 sendSynthesizedEventsToPage(page, eventsToBeSent.get());
779}
