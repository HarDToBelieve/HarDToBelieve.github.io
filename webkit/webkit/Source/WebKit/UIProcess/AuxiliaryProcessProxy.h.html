185template<typename T>
186bool AuxiliaryProcessProxy::send(T&& message, uint64_t destinationID, OptionSet<IPC::SendOption> sendOptions)
187{
188 COMPILE_ASSERT(!T::isSync, AsyncMessageExpected);
189
190 auto encoder = makeUniqueRef<IPC::Encoder>(T::name(), destinationID);
191 encoder.get() << message.arguments();
192
193 return sendMessage(WTFMove(encoder), sendOptions);
194}

196template<typename U>
197AuxiliaryProcessProxy::SendSyncResult AuxiliaryProcessProxy::sendSync(U&& message, typename U::Reply&& reply, uint64_t destinationID, IPC::Timeout timeout, OptionSet<IPC::SendSyncOption> sendSyncOptions)
198{
199 COMPILE_ASSERT(U::isSync, SyncMessageExpected);
200
201 if (!m_connection)
202 return { };
203
204 TraceScope scope(SyncMessageStart, SyncMessageEnd);
205
206 return connection()->sendSync(std::forward<U>(message), WTFMove(reply), destinationID, timeout, sendSyncOptions);
207}

209template<typename T, typename C>
210uint64_t AuxiliaryProcessProxy::sendWithAsyncReply(T&& message, C&& completionHandler, uint64_t destinationID, OptionSet<IPC::SendOption> sendOptions, ShouldStartProcessThrottlerActivity shouldStartProcessThrottlerActivity)
211{
212 COMPILE_ASSERT(!T::isSync, AsyncMessageExpected);
213
214 auto encoder = makeUniqueRef<IPC::Encoder>(T::name(), destinationID);
215 uint64_t listenerID = IPC::nextAsyncReplyHandlerID();
216 encoder.get() << listenerID;
217 encoder.get() << message.arguments();
218 sendMessage(WTFMove(encoder), sendOptions, {{ [completionHandler = WTFMove(completionHandler)] (IPC::Decoder* decoder) mutable {
219 if (decoder && decoder->isValid())
220 T::callReply(*decoder, WTFMove(completionHandler));
221 else
222 T::cancelReply(WTFMove(completionHandler));
223 }, listenerID }}, shouldStartProcessThrottlerActivity);
224 return listenerID;
225}
