39AutomationSessionClient::AutomationSessionClient(id <_WKAutomationSessionDelegate> delegate)
40 : m_delegate(delegate)
41{
42 m_delegateMethods.didDisconnectFromRemote = [delegate respondsToSelector:@selector(_automationSessionDidDisconnectFromRemote:)];
43
44 m_delegateMethods.requestNewWebViewWithOptions = [delegate respondsToSelector:@selector(_automationSession:requestNewWebViewWithOptions:completionHandler:)];
45 m_delegateMethods.requestSwitchToWebView = [delegate respondsToSelector:@selector(_automationSession:requestSwitchToWebView:completionHandler:)];
46 m_delegateMethods.requestHideWindowOfWebView = [delegate respondsToSelector:@selector(_automationSession:requestHideWindowOfWebView:completionHandler:)];
47 m_delegateMethods.requestRestoreWindowOfWebView = [delegate respondsToSelector:@selector(_automationSession:requestRestoreWindowOfWebView:completionHandler:)];
48 m_delegateMethods.requestMaximizeWindowOfWebView = [delegate respondsToSelector:@selector(_automationSession:requestMaximizeWindowOfWebView:completionHandler:)];
49 m_delegateMethods.isShowingJavaScriptDialogForWebView = [delegate respondsToSelector:@selector(_automationSession:isShowingJavaScriptDialogForWebView:)];
50 m_delegateMethods.dismissCurrentJavaScriptDialogForWebView = [delegate respondsToSelector:@selector(_automationSession:dismissCurrentJavaScriptDialogForWebView:)];
51 m_delegateMethods.acceptCurrentJavaScriptDialogForWebView = [delegate respondsToSelector:@selector(_automationSession:acceptCurrentJavaScriptDialogForWebView:)];
52 m_delegateMethods.messageOfCurrentJavaScriptDialogForWebView = [delegate respondsToSelector:@selector(_automationSession:messageOfCurrentJavaScriptDialogForWebView:)];
53 m_delegateMethods.setUserInputForCurrentJavaScriptPromptForWebView = [delegate respondsToSelector:@selector(_automationSession:setUserInput:forCurrentJavaScriptDialogForWebView:)];
54 m_delegateMethods.typeOfCurrentJavaScriptDialogForWebView = [delegate respondsToSelector:@selector(_automationSession:typeOfCurrentJavaScriptDialogForWebView:)];
55 m_delegateMethods.currentPresentationForWebView = [delegate respondsToSelector:@selector(_automationSession:currentPresentationForWebView:)];
56}

58void AutomationSessionClient::didDisconnectFromRemote(WebAutomationSession& session)
59{
60 if (m_delegateMethods.didDisconnectFromRemote)
61 [m_delegate.get() _automationSessionDidDisconnectFromRemote:wrapper(session)];
62}

64static inline _WKAutomationSessionBrowsingContextOptions toAPI(API::AutomationSessionBrowsingContextOptions options)
65{
66 uint16_t wkOptions = 0;
67
68 if (options & API::AutomationSessionBrowsingContextOptionsPreferNewTab)
69 wkOptions |= _WKAutomationSessionBrowsingContextOptionsPreferNewTab;
70
71 return static_cast<_WKAutomationSessionBrowsingContextOptions>(wkOptions);
72}

74void AutomationSessionClient::requestNewPageWithOptions(WebAutomationSession& session, API::AutomationSessionBrowsingContextOptions options, CompletionHandler<void(WebKit::WebPageProxy*)>&& completionHandler)
75{
76 if (m_delegateMethods.requestNewWebViewWithOptions) {
77 [m_delegate.get() _automationSession:wrapper(session) requestNewWebViewWithOptions:toAPI(options) completionHandler:makeBlockPtr([completionHandler = WTFMove(completionHandler)](WKWebView *webView) mutable {
78 completionHandler(webView->_page.get());
79 }).get()];
80 } else
81 completionHandler(nullptr);
82}

88 else
89 completionHandler();
90}

150static std::optional<API::AutomationSessionClient::JavaScriptDialogType> toImpl(_WKAutomationSessionJavaScriptDialogType type)
151{
152 switch (type) {
153 case _WKAutomationSessionJavaScriptDialogTypeNone:
154 return std::nullopt;
155 case _WKAutomationSessionJavaScriptDialogTypePrompt:
156 return API::AutomationSessionClient::JavaScriptDialogType::Prompt;
157 case _WKAutomationSessionJavaScriptDialogTypeConfirm:
158 return API::AutomationSessionClient::JavaScriptDialogType::Confirm;
159 case _WKAutomationSessionJavaScriptDialogTypeAlert:
160 return API::AutomationSessionClient::JavaScriptDialogType::Alert;
161 }
162}

164static API::AutomationSessionClient::BrowsingContextPresentation toImpl(_WKAutomationSessionBrowsingContextPresentation presentation)
165{
166 switch (presentation) {
167 case _WKAutomationSessionBrowsingContextPresentationTab:
168 return API::AutomationSessionClient::BrowsingContextPresentation::Tab;
169 case _WKAutomationSessionBrowsingContextPresentationWindow:
170 return API::AutomationSessionClient::BrowsingContextPresentation::Window;
171 }
172}

174std::optional<API::AutomationSessionClient::JavaScriptDialogType> AutomationSessionClient::typeOfCurrentJavaScriptDialogOnPage(WebAutomationSession& session, WebPageProxy& page)
175{
176 if (auto webView = page.cocoaView(); webView && m_delegateMethods.typeOfCurrentJavaScriptDialogForWebView)
177 return toImpl([m_delegate.get() _automationSession:wrapper(session) typeOfCurrentJavaScriptDialogForWebView:webView.get()]);
178
179 return API::AutomationSessionClient::JavaScriptDialogType::Prompt;
