35DiagnosticLoggingClient::DiagnosticLoggingClient(WKWebView *webView)
36 : m_webView(webView)
37{
38}
39
40RetainPtr<id <_WKDiagnosticLoggingDelegate>> DiagnosticLoggingClient::delegate()
41{
42 return m_delegate.get();
43}
44
45void DiagnosticLoggingClient::setDelegate(id <_WKDiagnosticLoggingDelegate> delegate)
46{
47 m_delegate = delegate;
48
49 m_delegateMethods.webviewLogDiagnosticMessage = [delegate respondsToSelector:@selector(_webView:logDiagnosticMessage:description:)];
50 m_delegateMethods.webviewLogDiagnosticMessageWithResult = [delegate respondsToSelector:@selector(_webView:logDiagnosticMessageWithResult:description:result:)];
51 m_delegateMethods.webviewLogDiagnosticMessageWithValue = [delegate respondsToSelector:@selector(_webView:logDiagnosticMessageWithValue:description:value:)];
52 m_delegateMethods.webviewLogDiagnosticMessageWithEnhancedPrivacy = [delegate respondsToSelector:@selector(_webView:logDiagnosticMessageWithEnhancedPrivacy:description:)];
53 m_delegateMethods.webviewLogDiagnosticMessageWithValueDictionary = [delegate respondsToSelector:@selector(_webView:logDiagnosticMessage:description:valueDictionary:)];
54 m_delegateMethods.webviewLogDiagnosticMessageWithDomain = [delegate respondsToSelector:@selector(_webView:logDiagnosticMessageWithDomain:domain:)];
55}
56
57void DiagnosticLoggingClient::logDiagnosticMessage(WebKit::WebPageProxy*, const WTF::String& message, const WTF::String& description)
58{
59 if (m_delegateMethods.webviewLogDiagnosticMessage)
60 [m_delegate.get() _webView:m_webView logDiagnosticMessage:message description:description];
61}
62
63static _WKDiagnosticLoggingResultType toWKDiagnosticLoggingResultType(WebCore::DiagnosticLoggingResultType result)
64{
65 switch (result) {
66 case WebCore::DiagnosticLoggingResultPass:
67 return _WKDiagnosticLoggingResultPass;
68 case WebCore::DiagnosticLoggingResultFail:
69 return _WKDiagnosticLoggingResultFail;
70 case WebCore::DiagnosticLoggingResultNoop:
71 return _WKDiagnosticLoggingResultNoop;
72 }
73}
74
75static _WKDiagnosticLoggingDomain toWKDiagnosticLoggingDomain(WebCore::DiagnosticLoggingDomain domain)
76{
77 switch (domain) {
78 case WebCore::DiagnosticLoggingDomain::Media:
79 return _WKDiagnosticLoggingDomainMedia;
80 }
81}
