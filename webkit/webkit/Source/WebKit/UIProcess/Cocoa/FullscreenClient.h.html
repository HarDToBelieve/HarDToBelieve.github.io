26#pragma once
27
28#import "WKFoundation.h"
29
30#import "APIFullscreenClient.h"
31#import <wtf/RetainPtr.h>
32#import <wtf/WeakObjCPtr.h>
33
34#if PLATFORM(MAC)
35@class NSView;
36using WKFullscreenClientView = NSView;
37#else
38@class WKWebView;
39using WKFullscreenClientView = WKWebView;
40#endif
41
42@protocol _WKFullscreenDelegate;
43
44namespace WebKit {
45
46class FullscreenClient : public API::FullscreenClient {
47public:
48 explicit FullscreenClient(WKFullscreenClientView *);
49 ~FullscreenClient() { };
50
51 bool isType(API::FullscreenClient::Type target) const override { return target == API::FullscreenClient::WebKitType; };
52
53 RetainPtr<id<_WKFullscreenDelegate>> delegate();
54 void setDelegate(id<_WKFullscreenDelegate>);
55
56 void willEnterFullscreen(WebPageProxy*) override;
57 void didEnterFullscreen(WebPageProxy*) override;
58 void willExitFullscreen(WebPageProxy*) override;
59 void didExitFullscreen(WebPageProxy*) override;
60
61private:
62 WKFullscreenClientView *m_webView;
63 WeakObjCPtr<id <_WKFullscreenDelegate> > m_delegate;
64
65 struct {
66#if PLATFORM(MAC)
67 bool webViewWillEnterFullscreen : 1;
68 bool webViewDidEnterFullscreen : 1;
69 bool webViewWillExitFullscreen : 1;
70 bool webViewDidExitFullscreen : 1;
71#else
72 bool webViewWillEnterElementFullscreen : 1;
73 bool webViewDidEnterElementFullscreen : 1;
74 bool webViewWillExitElementFullscreen : 1;
75 bool webViewDidExitElementFullscreen : 1;
76#endif
77 } m_delegateMethods;
78};
79
80} // namespace WebKit
