29namespace WebKit {

31_WKMediaCaptureStateDeprecated toWKMediaCaptureStateDeprecated(WebCore::MediaProducer::MediaStateFlags state)
32{
33 _WKMediaCaptureStateDeprecated mediaCaptureState = _WKMediaCaptureStateDeprecatedNone;
34 if (state & WebCore::MediaProducer::MediaState::HasActiveAudioCaptureDevice)
35 mediaCaptureState |= _WKMediaCaptureStateDeprecatedActiveMicrophone;
36 if (state & WebCore::MediaProducer::MediaState::HasActiveVideoCaptureDevice)
37 mediaCaptureState |= _WKMediaCaptureStateDeprecatedActiveCamera;
38 if (state & WebCore::MediaProducer::MediaState::HasMutedAudioCaptureDevice)
39 mediaCaptureState |= _WKMediaCaptureStateDeprecatedMutedMicrophone;
40 if (state & WebCore::MediaProducer::MediaState::HasMutedVideoCaptureDevice)
41 mediaCaptureState |= _WKMediaCaptureStateDeprecatedMutedCamera;
42
43 return mediaCaptureState;
44}
45
46_WKMediaMutedState toWKMediaMutedState(WebCore::MediaProducer::MutedStateFlags state)
47{
48 _WKMediaMutedState mediaMutedState = _WKMediaNoneMuted;
49 if (state & WebCore::MediaProducer::MutedState::AudioIsMuted)
50 mediaMutedState |= _WKMediaAudioMuted;
51 if (state & WebCore::MediaProducer::AudioAndVideoCaptureIsMuted)
52 mediaMutedState |= _WKMediaCaptureDevicesMuted;
53 if (state & WebCore::MediaProducer::MutedState::ScreenCaptureIsMuted)
54 mediaMutedState |= _WKMediaScreenCaptureMuted;
55 return mediaMutedState;
56}
