103NavigationState::NavigationState(WKWebView *webView)
104 : m_webView(webView)
105 , m_navigationDelegateMethods()

118NavigationState::~NavigationState()
119{
120 ASSERT(navigationStates().get(m_webView->_page.get()) == this);

453 delete localCompletionHandler;
454 });
455 }];
456 return;
457 }

463void NavigationState::NavigationClient::decidePolicyForNavigationAction(WebPageProxy& webPageProxy, Ref<API::NavigationAction>&& navigationAction, Ref<WebFramePolicyListenerProxy>&& listener, API::Object* userInfo)
464{
465 bool subframeNavigation = navigationAction->targetFrame() && !navigationAction->targetFrame()->isMainFrame();
466
467 RefPtr<API::WebsitePolicies> defaultWebsitePolicies;
468 if (auto* policies = webPageProxy.configuration().defaultWebsitePolicies())
469 defaultWebsitePolicies = policies->copy();
470
471 if (!m_navigationState || (!m_navigationState->m_navigationDelegateMethods.webViewDecidePolicyForNavigationActionDecisionHandler
472 && !m_navigationState->m_navigationDelegateMethods.webViewDecidePolicyForNavigationActionWithPreferencesUserInfoDecisionHandler
473 && !m_navigationState->m_navigationDelegateMethods.webViewDecidePolicyForNavigationActionWithPreferencesDecisionHandler)) {
474 auto completionHandler = [webPage = makeRef(webPageProxy), listener = WTFMove(listener), navigationAction, defaultWebsitePolicies] (bool interceptedNavigation) {
475 if (interceptedNavigation) {
476 listener->ignore();
477 return;
478 }
479
480 if (!navigationAction->targetFrame()) {
481 listener->use(defaultWebsitePolicies.get());
482 return;
483 }
484
485 NSURLRequest *nsURLRequest = wrapper(API::URLRequest::create(navigationAction->request()));
486 if ([NSURLConnection canHandleRequest:nsURLRequest]
487 || webPage->urlSchemeHandlerForScheme([nsURLRequest URL].scheme)
488 || [nsURLRequest.URL.scheme isEqualToString:@"blob"]) {
489 if (navigationAction->shouldPerformDownload())
490 listener->download();
491 else
492 listener->use(defaultWebsitePolicies.get());
493 return;
494 }
495
496#if PLATFORM(MAC)
497 // A file URL shouldn't fall through to here, but if it did,
498 // it would be a security risk to open it.
499 if (![[nsURLRequest URL] isFileURL])
500 [[NSWorkspace sharedWorkspace] openURL:[nsURLRequest URL]];
501#endif
502 listener->ignore();
503 };
504 tryInterceptNavigation(WTFMove(navigationAction), webPageProxy, WTFMove(completionHandler));
505 return;
506 }
507
508 auto navigationDelegate = m_navigationState->m_navigationDelegate.get();
509 if (!navigationDelegate)
510 return;
511
512 bool delegateHasWebpagePreferences = m_navigationState->m_navigationDelegateMethods.webViewDecidePolicyForNavigationActionWithPreferencesDecisionHandler
513 || m_navigationState->m_navigationDelegateMethods.webViewDecidePolicyForNavigationActionWithPreferencesUserInfoDecisionHandler;
514
515 auto selectorForCompletionHandlerChecker = ([&] () -> SEL {
516 if (delegateHasWebpagePreferences)
517 return m_navigationState->m_navigationDelegateMethods.webViewDecidePolicyForNavigationActionWithPreferencesDecisionHandler ? @selector(webView:decidePolicyForNavigationAction:preferences:decisionHandler:) : @selector(_webView:decidePolicyForNavigationAction:preferences:userInfo:decisionHandler:);
518 return @selector(webView:decidePolicyForNavigationAction:decisionHandler:);
519 })();
520
521 auto checker = CompletionHandlerCallChecker::create(navigationDelegate.get(), selectorForCompletionHandlerChecker);
522 auto decisionHandlerWithPreferencesOrPolicies = [localListener = WTFMove(listener), navigationAction, checker = WTFMove(checker), webPageProxy = makeRef(webPageProxy), subframeNavigation, defaultWebsitePolicies] (WKNavigationActionPolicy actionPolicy, WKWebpagePreferences *preferences) mutable {
523 if (checker->completionHandlerHasBeenCalled())
524 return;
525 checker->didCallCompletionHandler();
526
