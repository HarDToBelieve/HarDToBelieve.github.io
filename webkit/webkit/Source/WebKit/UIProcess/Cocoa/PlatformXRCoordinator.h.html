26#pragma once
27
28#if ENABLE(WEBXR)
29
30#include "XRDeviceIdentifier.h"
31#include "XRDeviceInfo.h"
32#include <WebCore/PlatformXR.h>
33#include <wtf/Function.h>
34
35namespace WebKit {
36
37class WebPageProxy;
38
39class PlatformXRCoordinator {
40public:
41 virtual ~PlatformXRCoordinator() = default;
42
43 // FIXME: Temporary and will be fixed later.
44 static PlatformXR::LayerHandle defaultLayerHandle() { return 1; }
45
46 using DeviceInfoCallback = Function<void(std::optional<XRDeviceInfo>)>;
47 virtual void getPrimaryDeviceInfo(DeviceInfoCallback&&) = 0;
48
49 // Session creation/termination.
50 using OnSessionEndCallback = Function<void(XRDeviceIdentifier)>;
51 virtual void startSession(WebPageProxy&, OnSessionEndCallback&&) = 0;
52 virtual void endSessionIfExists(WebPageProxy&) = 0;
53
54 // Session display loop.
55 virtual void scheduleAnimationFrame(WebPageProxy&, PlatformXR::Device::RequestFrameCallback&&) = 0;
56 virtual void submitFrame(WebPageProxy&) { }
57};
58
59} // namespace WebKit
60
61#endif // ENABLE(WEBXR)
