39PlatformXRSystem::PlatformXRSystem(WebPageProxy& page)
40 : m_page(page)
41{
42 m_page.process().addMessageReceiver(Messages::PlatformXRSystem::messageReceiverName(), m_page.webPageID(), *this);
43}
44
45PlatformXRSystem::~PlatformXRSystem()
46{
47 m_page.process().removeMessageReceiver(Messages::PlatformXRSystem::messageReceiverName(), m_page.webPageID());
48}
49
50void PlatformXRSystem::invalidate()
51{
52 if (xrCoordinator())
53 xrCoordinator()->endSessionIfExists(m_page);
54}
55
56void PlatformXRSystem::enumerateImmersiveXRDevices(CompletionHandler<void(Vector<XRDeviceInfo>&&)>&& completionHandler)
57{
58 auto* xrCoordinator = PlatformXRSystem::xrCoordinator();
59 if (!xrCoordinator) {
60 completionHandler({ });
61 return;
62 }
63
64 xrCoordinator->getPrimaryDeviceInfo([completionHandler = WTFMove(completionHandler)](std::optional<XRDeviceInfo> deviceInfo) mutable {
65 RunLoop::main().dispatch([completionHandler = WTFMove(completionHandler), deviceInfo = WTFMove(deviceInfo)]() mutable {
66 if (!deviceInfo) {
67 completionHandler({ });
68 return;
69 }
70
71 completionHandler({ deviceInfo.value() });
72 });
73 });
74}
75
76void PlatformXRSystem::initializeTrackingAndRendering()
77{
78 auto* xrCoordinator = PlatformXRSystem::xrCoordinator();
79 if (!xrCoordinator)
80 return;
81
82 xrCoordinator->startSession(m_page, [weakThis = makeWeakPtr(*this)](XRDeviceIdentifier deviceIdentifier) {
83 RunLoop::main().dispatch([weakThis, deviceIdentifier]() mutable {
84 if (weakThis)
85 weakThis->m_page.send(Messages::PlatformXRSystemProxy::SessionDidEnd(deviceIdentifier));
86 });
87 });
88}
89
90void PlatformXRSystem::shutDownTrackingAndRendering()
91{
92 if (auto* xrCoordinator = PlatformXRSystem::xrCoordinator())
93 xrCoordinator->endSessionIfExists(m_page);
94}
95
96void PlatformXRSystem::requestFrame(CompletionHandler<void(PlatformXR::Device::FrameData&&)>&& completionHandler)
97{
98 if (auto* xrCoordinator = PlatformXRSystem::xrCoordinator())
99 xrCoordinator->scheduleAnimationFrame(m_page, WTFMove(completionHandler));
100}
101
102void PlatformXRSystem::submitFrame()
103{
104 if (auto* xrCoordinator = PlatformXRSystem::xrCoordinator())
105 xrCoordinator->submitFrame(m_page);
106}
