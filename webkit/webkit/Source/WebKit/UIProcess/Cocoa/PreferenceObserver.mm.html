50- (void)findPreferenceChangesAndNotifyForKeys:(NSDictionary<NSString *, id> *)oldValues toValuesForKeys:(NSDictionary<NSString *, id> *)newValues
51{
52 if (!m_observer)
53 return;
54
55 for (NSString *key in oldValues) {
56 id oldValue = oldValues[key];
57 id newValue = newValues[key];
58
59 if ([oldValue isEqual:newValue])
60 continue;
61
62 if (newValue && ![[newValue class] supportsSecureCoding])
63 continue;
64
65 NSString *encodedString = nil;
66
67 if (newValue) {
68 NSError *e = nil;
69 auto data = retainPtr([NSKeyedArchiver archivedDataWithRootObject:newValue requiringSecureCoding:YES error:&e]);
70 ASSERT(!e);
71 encodedString = [data base64EncodedStringWithOptions:0];
72 }
73
74 auto globalValue = adoptCF(CFPreferencesCopyValue((__bridge CFStringRef)key, kCFPreferencesAnyApplication, kCFPreferencesCurrentUser, kCFPreferencesAnyHost));
75 auto domainValue = adoptCF(CFPreferencesCopyValue((__bridge CFStringRef)key, (__bridge CFStringRef)m_suiteName.get(), kCFPreferencesCurrentUser, kCFPreferencesAnyHost));
76
77 auto preferenceValuesAreEqual = [] (id a, id b) {
78 return a == b || [a isEqual:b];
79 };
80
81 if (preferenceValuesAreEqual((__bridge id)globalValue.get(), newValue))
82 [m_observer preferenceDidChange:nil key:key encodedValue:encodedString];
83
84 if (preferenceValuesAreEqual((__bridge id)domainValue.get(), newValue))
85 [m_observer preferenceDidChange:m_suiteName.get() key:key encodedValue:encodedString];
86 }
87}

89- (void)_notifyObserversOfChangeFromValuesForKeys:(NSDictionary<NSString *, id> *)oldValues toValuesForKeys:(NSDictionary<NSString *, id> *)newValues
90{
91 [super _notifyObserversOfChangeFromValuesForKeys:oldValues toValuesForKeys:newValues];
92
93 if (!isMainRunLoop()) {
94 [self findPreferenceChangesAndNotifyForKeys:oldValues toValuesForKeys:newValues];
95 return;
96 }
97
98 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), [self, protectedSelf = retainPtr(self), oldValues = retainPtr(oldValues), newValues = retainPtr(newValues)] {
99 [self findPreferenceChangesAndNotifyForKeys:oldValues.get() toValuesForKeys:newValues.get()];
100 });
101}

126- (instancetype)init
127{
128 if (!(self = [super init]))
129 return nil;
130
131 std::initializer_list<NSString*> domains = {
132#if PLATFORM(IOS_FAMILY)
133 @"com.apple.Accessibility",
134 @"com.apple.AdLib",
135 @"com.apple.SpeakSelection",
136 @"com.apple.UIKit",
137 @"com.apple.WebUI",
138 @"com.apple.avfaudio",
139 @"com.apple.itunesstored",
140 @"com.apple.mediaremote",
141 @"com.apple.preferences.sounds",
142 @"com.apple.voiceservices",
143#else
144 @"com.apple.CoreGraphics",
145 @"com.apple.HIToolbox",
146 @"com.apple.ServicesMenu.Services",
147 @"com.apple.ViewBridge",
148 @"com.apple.avfoundation",
149 @"com.apple.avfoundation.videoperformancehud",
150 @"com.apple.driver.AppleBluetoothMultitouch.mouse",
151 @"com.apple.driver.AppleBluetoothMultitouch.trackpad",
152 @"com.apple.mediaaccessibility",
153 @"com.apple.speech.voice.prefs",
154 @"com.apple.universalaccess",
155#endif
156 };
157
158 for (auto domain : domains) {
159 auto userDefaults = adoptNS([[WKUserDefaults alloc] initWithSuiteName:domain]);
160 if (!userDefaults) {
161 WTFLogAlways("Could not init user defaults instance for domain %s", String(domain).utf8().data());
162 continue;
163 }
164 userDefaults.get()->m_observer = self;
165 // Start observing a dummy key in order to make the preference daemon become aware of our NSUserDefaults instance.
166 // This is to make sure we receive KVO notifications. We cannot use normal KVO techniques here, since we are looking
167 // for _any_ changes in a preference domain. For normal KVO techniques to work, we need to provide the specific
168 // key(s) we want to observe, but that set of keys is unknown to us.
169 [userDefaults.get() addObserver:userDefaults.get() forKeyPath:@"testkey" options:NSKeyValueObservingOptionNew context:nil];
170 m_userDefaults.append(userDefaults);
171 }
172 return self;
173}

175- (void)preferenceDidChange:(NSString *)domain key:(NSString *)key encodedValue:(NSString *)encodedValue
176{
177#if ENABLE(CFPREFS_DIRECT_MODE)
178 RunLoop::main().dispatch([domain = retainPtr(domain), key = retainPtr(key), encodedValue = retainPtr(encodedValue)] {
179 std::optional<String> encodedString;
180 if (encodedValue)
181 encodedString = String(encodedValue.get());
182
183 for (auto& processPool : WebKit::WebProcessPool::allProcessPools())
184 processPool->notifyPreferencesChanged(domain.get(), key.get(), encodedString);
185 });
186#endif
187}
