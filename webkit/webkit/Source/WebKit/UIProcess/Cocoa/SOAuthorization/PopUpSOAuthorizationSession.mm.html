40@interface WKSOSecretDelegate : NSObject <WKNavigationDelegate, WKUIDelegate> {
41@private
42 WeakPtr<WebKit::PopUpSOAuthorizationSession> _session;
43 BOOL _isFirstNavigation;
44}
45
46- (instancetype)initWithSession:(WebKit::PopUpSOAuthorizationSession *)session;
47
48@end
49
50@implementation WKSOSecretDelegate
51
52- (instancetype)initWithSession:(WebKit::PopUpSOAuthorizationSession *)session
53{
54 if ((self = [super init])) {
55 _session = makeWeakPtr(session);
56 _isFirstNavigation = YES;
57 }
58 return self;
59}
60
61// WKUIDelegate
62- (void)webViewDidClose:(WKWebView *)webView
63{
64 if (!_session)
65 return;
66 _session->close(webView);
67}
68
69// WKNavigationDelegate
70- (void)webView:(WKWebView *)webView decidePolicyForNavigationAction:(WKNavigationAction *)navigationAction decisionHandler:(void (^)(WKNavigationActionPolicy))decisionHandler
71{
72 // FIXME<rdar://problem/48787839>: We should restrict the load to only substitute data.
73 // Use the following heuristic as a workaround right now.
74 // Ignore the first load in the secret window, which navigates to the authentication URL.
75 if (_isFirstNavigation) {
76 _isFirstNavigation = NO;
77 decisionHandler(WKNavigationActionPolicyCancel);
78 return;
79 }
80 decisionHandler(_WKNavigationActionPolicyAllowWithoutTryingAppLink);
81}
82
83- (void)webView:(WKWebView *)webView didFinishNavigation:(WKNavigation *)navigation
84{
85 if (!_session)
86 return;
87 _session->close(webView);
88}
89
90@end

96Ref<SOAuthorizationSession> PopUpSOAuthorizationSession::create(SOAuthorization *soAuthorization, WebPageProxy& page, Ref<API::NavigationAction>&& navigationAction, NewPageCallback&& newPageCallback, UIClientCallback&& uiClientCallback)

101PopUpSOAuthorizationSession::PopUpSOAuthorizationSession(SOAuthorization *soAuthorization, WebPageProxy& page, Ref<API::NavigationAction>&& navigationAction, NewPageCallback&& newPageCallback, UIClientCallback&& uiClientCallback)
102 : SOAuthorizationSession(soAuthorization, WTFMove(navigationAction), page, InitiatingAction::PopUp)
103 , m_newPageCallback(WTFMove(newPageCallback))
104 , m_uiClientCallback(WTFMove(uiClientCallback))

106}

108PopUpSOAuthorizationSession::~PopUpSOAuthorizationSession()
109{
110 ASSERT(state() != State::Waiting);
111 if (m_newPageCallback)
112 m_newPageCallback(nullptr);
113}

115void PopUpSOAuthorizationSession::shouldStartInternal()
116{
117 AUTHORIZATIONSESSION_RELEASE_LOG("shouldStartInternal: m_page=%p", page());
118 ASSERT(page() && page()->isInWindow());
119 start();
120}
121
122void PopUpSOAuthorizationSession::fallBackToWebPathInternal()
123{
124 AUTHORIZATIONSESSION_RELEASE_LOG("fallBackToWebPathInternal");
125 m_uiClientCallback(releaseNavigationAction(), WTFMove(m_newPageCallback));
126}
127
128void PopUpSOAuthorizationSession::abortInternal()
129{
130 AUTHORIZATIONSESSION_RELEASE_LOG("abortInternal: m_page=%p", page());
131 if (!page()) {
132 m_newPageCallback(nullptr);
133 return;
134 }
135
136 initSecretWebView();
137 if (!m_secretWebView) {
138 m_newPageCallback(nullptr);
139 return;
140 }
141
142 m_newPageCallback(m_secretWebView->_page.get());
143 [m_secretWebView evaluateJavaScript: @"window.close()" completionHandler:nil];
144}
145
146void PopUpSOAuthorizationSession::completeInternal(const WebCore::ResourceResponse& response, NSData *data)
147{
148 AUTHORIZATIONSESSION_RELEASE_LOG("completeInternal: httpState=%d", response.httpStatusCode());
149 if (response.httpStatusCode() != 200 || !page()) {
150 fallBackToWebPathInternal();
151 return;
152 }
153
154 initSecretWebView();
155 if (!m_secretWebView) {
156 fallBackToWebPathInternal();
157 return;
158 }
159
160 m_newPageCallback(m_secretWebView->_page.get());
161 [m_secretWebView loadData:data MIMEType:@"text/html" characterEncodingName:@"UTF-8" baseURL:response.url()];
162}
163
164void PopUpSOAuthorizationSession::close(WKWebView *webView)
165{
166 AUTHORIZATIONSESSION_RELEASE_LOG("close");
167 if (!m_secretWebView)
168 return;
169 if (state() != State::Completed || webView != m_secretWebView.get()) {
170 ASSERT_NOT_REACHED();
171 return;
172 }
173 m_secretWebView = nullptr;
174 WTFLogAlways("SecretWebView is cleaned.");
175}
176
177void PopUpSOAuthorizationSession::initSecretWebView()
178{
179 AUTHORIZATIONSESSION_RELEASE_LOG("initSecretWebView");
180 ASSERT(page());
181 auto initiatorWebView = page()->cocoaView();
182 if (!initiatorWebView)
183 return;
184 auto configuration = adoptNS([[initiatorWebView configuration] copy]);
185 [configuration _setRelatedWebView:initiatorWebView.get()];
186 m_secretWebView = adoptNS([[WKWebView alloc] initWithFrame:CGRectZero configuration:configuration.get()]);
187
188 m_secretDelegate = adoptNS([[WKSOSecretDelegate alloc] initWithSession:this]);
189 [m_secretWebView setUIDelegate:m_secretDelegate.get()];
190 [m_secretWebView setNavigationDelegate:m_secretDelegate.get()];
191
192 m_secretWebView->_page->setShouldSuppressSOAuthorizationInAllNavigationPolicyDecision();
193 WTFLogAlways("SecretWebView is created.");
194}
