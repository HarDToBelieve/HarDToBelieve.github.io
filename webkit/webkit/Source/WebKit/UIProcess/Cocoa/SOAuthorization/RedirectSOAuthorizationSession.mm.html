63void RedirectSOAuthorizationSession::completeInternal(const ResourceResponse& response, NSData *data)
64{
65 AUTHORIZATIONSESSION_RELEASE_LOG("completeInternal: httpState=%d, navigationAction=%p", response.httpStatusCode(), navigationAction());
66
67 auto* navigationAction = this->navigationAction();
68 ASSERT(navigationAction);
69 auto* page = this->page();
70 if ((response.httpStatusCode() != 302 && response.httpStatusCode() != 200) || !page) {
71 AUTHORIZATIONSESSION_RELEASE_LOG("completeInternal: httpState=%d page=%d, so falling back to web path.", response.httpStatusCode(), !!page);
72 fallBackToWebPathInternal();
73 return;
74 }
75 invokeCallback(true);
76 if (response.httpStatusCode() == 302) {
77#if PLATFORM(IOS)
78 // MobileSafari has a WBSURLSpoofingMitigator, which will not display the provisional URL for navigations without user gestures.
79 // For slow loads that are initiated from the MobileSafari Favorites screen, the aforementioned behavior will create a period
80 // after authentication completion where the new request to the application site loads with a blank URL and blank page. To
81 // workaround this issue, we load an html page that does a client side redirection to the application site on behalf of the
82 // request URL, instead of directly loading a new request. The html page should be super fast to load and therefore will not
83 // show an empty URL or a blank page. These changes ensure a relevant URL bar and useful page content during the load.
84 if (!navigationAction->isProcessingUserGesture()) {
85 page->setShouldSuppressSOAuthorizationInNextNavigationPolicyDecision();
86 auto html = makeString("<script>location = '", response.httpHeaderFields().get(HTTPHeaderName::Location), "'</script>").utf8();
87 auto data = IPC::DataReference(html.dataAsUInt8Ptr(), html.length());
88 page->loadData(data, "text/html"_s, "UTF-8"_s, navigationAction->request().url().string(), nullptr, navigationAction->shouldOpenExternalURLsPolicy());
89 return;
90 }
91#endif
92 page->loadRequest(ResourceRequest(response.httpHeaderFields().get(HTTPHeaderName::Location)));
93 }
94 if (response.httpStatusCode() == 200) {
95 page->setShouldSuppressSOAuthorizationInNextNavigationPolicyDecision();
96 page->loadData(IPC::DataReference(static_cast<const uint8_t*>(data.bytes), data.length), "text/html"_s, "UTF-8"_s, response.url().string(), nullptr, navigationAction->shouldOpenExternalURLsPolicy());
97 }
98}
