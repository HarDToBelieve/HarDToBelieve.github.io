48SOAuthorizationCoordinator::SOAuthorizationCoordinator()
49{
50#if PLATFORM(MAC)
51 // In the case of base system, which doesn't have AppSSO.framework.
52 if (!PAL::getSOAuthorizationClass())
53 return;
54#endif
55 m_soAuthorization = adoptNS([PAL::allocSOAuthorizationInstance() init]);
56 m_soAuthorizationDelegate = adoptNS([[WKSOAuthorizationDelegate alloc] init]);
57 m_soAuthorization.get().delegate = m_soAuthorizationDelegate.get();
58 [NSURLSession _disableAppSSO];
59}

61bool SOAuthorizationCoordinator::canAuthorize(const URL& url) const
62{
63 return m_soAuthorization && [PAL::getSOAuthorizationClass() canPerformAuthorizationWithURL:url responseCode:0];
64}

66void SOAuthorizationCoordinator::tryAuthorize(Ref<API::NavigationAction>&& navigationAction, WebPageProxy& page, Function<void(bool)>&& completionHandler)
67{
68 AUTHORIZATIONCOORDINATOR_RELEASE_LOG("tryAuthorize");
69 if (!canAuthorize(navigationAction->request().url())) {
70 AUTHORIZATIONCOORDINATOR_RELEASE_LOG("tryAuthorize: Cannot authorize the requested URL.");
71 completionHandler(false);
72 return;
73 }
74
75 // SubFrameSOAuthorizationSession should only be allowed for Apple first parties.
76 auto* targetFrame = navigationAction->targetFrame();
77 bool subframeNavigation = targetFrame && !targetFrame->isMainFrame();
78 if (subframeNavigation && (!page.mainFrame() || ![AKAuthorizationController isURLFromAppleOwnedDomain:page.mainFrame()->url()])) {
79 AUTHORIZATIONCOORDINATOR_RELEASE_LOG("tryAuthorize: Attempting to perform subframe navigation for non-Apple authorization URL.");
80 completionHandler(false);
81 return;
82 }
83
84 auto session = subframeNavigation ? SubFrameSOAuthorizationSession::create(m_soAuthorization.get(), WTFMove(navigationAction), page, WTFMove(completionHandler), targetFrame->handle()->frameID()) : RedirectSOAuthorizationSession::create(m_soAuthorization.get(), WTFMove(navigationAction), page, WTFMove(completionHandler));
85 [m_soAuthorizationDelegate setSession:WTFMove(session)];
86}
87
88void SOAuthorizationCoordinator::tryAuthorize(Ref<API::NavigationAction>&& navigationAction, WebPageProxy& page, NewPageCallback&& newPageCallback, UIClientCallback&& uiClientCallback)
89{
90 AUTHORIZATIONCOORDINATOR_RELEASE_LOG("tryAuthorize (2)");
91 bool subframeNavigation = navigationAction->sourceFrame() && !navigationAction->sourceFrame()->isMainFrame();
92 if (subframeNavigation || !navigationAction->isProcessingUserGesture() || !canAuthorize(navigationAction->request().url())) {
93 AUTHORIZATIONCOORDINATOR_RELEASE_LOG("tryAuthorize (2): Attempting to perform invalid auth.");
94 uiClientCallback(WTFMove(navigationAction), WTFMove(newPageCallback));
95 return;
96 }
97
98 auto session = PopUpSOAuthorizationSession::create(m_soAuthorization.get(), page, WTFMove(navigationAction), WTFMove(newPageCallback), WTFMove(uiClientCallback));
99 [m_soAuthorizationDelegate setSession:WTFMove(session)];
100}
