45const char* soAuthorizationPostDidStartMessageToParent = "<script>parent.postMessage('SOAuthorizationDidStart', '*');</script>";
46const char* soAuthorizationPostDidCancelMessageToParent = "<script>parent.postMessage('SOAuthorizationDidCancel', '*');</script>";

50Ref<SOAuthorizationSession> SubFrameSOAuthorizationSession::create(SOAuthorization *soAuthorization, Ref<API::NavigationAction>&& navigationAction, WebPageProxy& page, Callback&& completionHandler, FrameIdentifier frameID)
51{
52 return adoptRef(*new SubFrameSOAuthorizationSession(soAuthorization, WTFMove(navigationAction), page, WTFMove(completionHandler), frameID));
53}

55SubFrameSOAuthorizationSession::SubFrameSOAuthorizationSession(SOAuthorization *soAuthorization, Ref<API::NavigationAction>&& navigationAction, WebPageProxy& page, Callback&& completionHandler, FrameIdentifier frameID)
56 : NavigationSOAuthorizationSession(soAuthorization, WTFMove(navigationAction), page, InitiatingAction::SubFrame, WTFMove(completionHandler))
57 , m_frameID(frameID)
58{
59 if (auto* frame = page.process().webFrame(m_frameID))
60 frame->frameLoadState().addObserver(*this);
61}
62
63SubFrameSOAuthorizationSession::~SubFrameSOAuthorizationSession()
64{
65 auto* page = this->page();
66 if (!page)
67 return;
68 if (auto* frame = page->process().webFrame(m_frameID))
69 frame->frameLoadState().removeObserver(*this);
70}

72void SubFrameSOAuthorizationSession::fallBackToWebPathInternal()
73{
74 AUTHORIZATIONSESSION_RELEASE_LOG("fallBackToWebPathInternal: navigationAction=%p", navigationAction());
75 ASSERT(navigationAction());
76 appendRequestToLoad(URL(navigationAction()->request().url()), Vector { reinterpret_cast<const uint8_t*>(soAuthorizationPostDidCancelMessageToParent), strlen(soAuthorizationPostDidCancelMessageToParent) });
77 appendRequestToLoad(URL(navigationAction()->request().url()), String(navigationAction()->request().httpReferrer()));
78}
79
80void SubFrameSOAuthorizationSession::abortInternal()
81{
82 AUTHORIZATIONSESSION_RELEASE_LOG("abortInternal");
83 fallBackToWebPathInternal();
84}
85
86void SubFrameSOAuthorizationSession::completeInternal(const WebCore::ResourceResponse& response, NSData *data)
87{
88 AUTHORIZATIONSESSION_RELEASE_LOG("completeInternal: httpState=%d", response.httpStatusCode());
89 if (response.httpStatusCode() != 200) {
90 fallBackToWebPathInternal();
91 return;
92 }
93 appendRequestToLoad(URL(response.url()), Vector { reinterpret_cast<const uint8_t*>(data.bytes), data.length });
94}

96void SubFrameSOAuthorizationSession::beforeStart()
97{
98 AUTHORIZATIONSESSION_RELEASE_LOG("beforeStart");
99 // Cancelled the current load before loading the data to post SOAuthorizationDidStart to the parent frame.
100 invokeCallback(true);
101 ASSERT(navigationAction());
102 appendRequestToLoad(URL(navigationAction()->request().url()), Vector { reinterpret_cast<const uint8_t*>(soAuthorizationPostDidStartMessageToParent), strlen(soAuthorizationPostDidStartMessageToParent) });
103}
104
105void SubFrameSOAuthorizationSession::didFinishLoad()
106{
107 AUTHORIZATIONSESSION_RELEASE_LOG("didFinishLoad");
108 auto* page = this->page();
109 if (!page)
110 return;
111 auto* frame = page->process().webFrame(m_frameID);
112 ASSERT(frame);
113 if (m_requestsToLoad.isEmpty() || m_requestsToLoad.first().first != frame->url())
114 return;
115 m_requestsToLoad.takeFirst();
116 loadRequestToFrame();
117}
118
119void SubFrameSOAuthorizationSession::appendRequestToLoad(URL&& url, Supplement&& supplement)
120{
121 m_requestsToLoad.append({ WTFMove(url), WTFMove(supplement) });
122 if (m_requestsToLoad.size() == 1)
123 loadRequestToFrame();
124}
125
126void SubFrameSOAuthorizationSession::loadRequestToFrame()
127{
128 AUTHORIZATIONSESSION_RELEASE_LOG("loadRequestToFrame");
129 auto* page = this->page();
130 if (!page || m_requestsToLoad.isEmpty())
131 return;
132
133 if (auto* frame = page->process().webFrame(m_frameID)) {
134 page->setShouldSuppressSOAuthorizationInNextNavigationPolicyDecision();
135 auto& url = m_requestsToLoad.first().first;
136 WTF::switchOn(m_requestsToLoad.first().second, [&](const Vector<uint8_t>& data) {
137 frame->loadData(IPC::DataReference(data), "text/html", "UTF-8", url);
138 }, [&](const String& referrer) {
139 frame->loadURL(url, referrer);
140 });
141 }
142}
