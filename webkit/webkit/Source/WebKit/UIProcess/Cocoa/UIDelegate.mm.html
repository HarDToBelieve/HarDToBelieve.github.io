416void UIDelegate::UIClient::decidePolicyForGeolocationPermissionRequest(WebKit::WebPageProxy& page, WebKit::WebFrameProxy& frame, const FrameInfoData& frameInfo, Function<void(bool)>& completionHandler)
417{
418 if (!m_uiDelegate || (!m_uiDelegate->m_delegateMethods.webViewRequestGeolocationPermissionForFrameDecisionHandler && !m_uiDelegate->m_delegateMethods.webViewRequestGeolocationPermissionForOriginDecisionHandler))
419 return;
420
421 auto delegate = m_uiDelegate->m_delegate.get();
422 if (!delegate)
423 return;
424
425 if (m_uiDelegate->m_delegateMethods.webViewRequestGeolocationPermissionForOriginDecisionHandler) {
426 auto securityOrigin = WebCore::SecurityOrigin::createFromString(page.pageLoadState().activeURL());
427 auto checker = CompletionHandlerCallChecker::create(delegate.get(), @selector(_webView:requestGeolocationPermissionForOrigin:initiatedByFrame:decisionHandler:));
428 auto decisionHandler = makeBlockPtr([completionHandler = std::exchange(completionHandler, nullptr), securityOrigin = securityOrigin->data(), checker = WTFMove(checker), page = makeWeakPtr(page)] (WKPermissionDecision decision) mutable {
429 if (checker->completionHandlerHasBeenCalled())
430 return;
431 checker->didCallCompletionHandler();
432 if (!page) {
433 completionHandler(false);
434 return;
435 }
436 switch (decision) {
437 case WKPermissionDecisionPrompt:
438 alertForPermission(*page, MediaPermissionReason::Geolocation, securityOrigin, WTFMove(completionHandler));
439 break;
440 case WKPermissionDecisionGrant:
441 completionHandler(true);
442 break;
443 case WKPermissionDecisionDeny:
444 completionHandler(false);
445 break;
