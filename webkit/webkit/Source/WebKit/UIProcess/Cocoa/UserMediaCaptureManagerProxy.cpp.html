54class UserMediaCaptureManagerProxy::SourceProxy
55 : private RealtimeMediaSource::Observer
56 , private RealtimeMediaSource::AudioSampleObserver
57 , private RealtimeMediaSource::VideoSampleObserver {
58 WTF_MAKE_FAST_ALLOCATED;
59public:
60 SourceProxy(RealtimeMediaSourceIdentifier id, Ref<IPC::Connection>&& connection, Ref<RealtimeMediaSource>&& source)
61 : m_id(id)
62 , m_connection(WTFMove(connection))
63 , m_source(WTFMove(source))
64 {
65 m_source->addObserver(*this);
66 switch (m_source->type()) {
67 case RealtimeMediaSource::Type::Audio:
68 m_source->addAudioSampleObserver(*this);
69 break;
70 case RealtimeMediaSource::Type::Video:
71 m_source->addVideoSampleObserver(*this);
72 break;
73 case RealtimeMediaSource::Type::None:
74 ASSERT_NOT_REACHED();
75 }
76 }
77
78 ~SourceProxy()
79 {
80 // Make sure the rendering thread is stopped before we proceed with the destruction.
81 stop();
82
83 if (m_ringBuffer)
84 static_cast<SharedRingBufferStorage&>(m_ringBuffer->storage()).invalidate();
85
86 switch (m_source->type()) {
87 case RealtimeMediaSource::Type::Audio:
88 m_source->removeAudioSampleObserver(*this);
89 break;
90 case RealtimeMediaSource::Type::Video:
91 m_source->removeVideoSampleObserver(*this);
92 break;
93 case RealtimeMediaSource::Type::None:
94 ASSERT_NOT_REACHED();
95 }
96 m_source->removeObserver(*this);
97 }

99 RealtimeMediaSource& source() { return m_source; }
100 CAAudioStreamDescription& description() { return m_description; }
101 int64_t numberOfFrames() { return m_numberOfFrames; }

103 void audioUnitWillStart() final
104 {
105 // FIXME: WebProcess might want to set the category/bufferSize itself, in which case we should remove that code.
106 auto bufferSize = AudioSession::sharedSession().sampleRate() / 50;
107 if (AudioSession::sharedSession().preferredBufferSize() > bufferSize)
108 AudioSession::sharedSession().setPreferredBufferSize(bufferSize);
109 AudioSession::sharedSession().setCategory(AudioSession::CategoryType::PlayAndRecord, RouteSharingPolicy::Default);
110 }

150 // May get called on a background thread.
151 void audioSamplesAvailable(const MediaTime& time, const PlatformAudioData& audioData, const AudioStreamDescription& description, size_t numberOfFrames) final {
152 if (m_description != description || m_shouldReset) {
153 DisableMallocRestrictionsForCurrentThreadScope scope;
154
155 m_shouldReset = false;
156 m_writeOffset = 0;
157 m_remainingFrameCount = 0;
158 m_startTime = time;
159 m_captureSemaphore = makeUnique<IPC::Semaphore>();
160 ASSERT(description.platformDescription().type == PlatformDescription::CAAudioStreamBasicType);
161 m_description = *WTF::get<const AudioStreamBasicDescription*>(description.platformDescription().description);
162
163 m_frameChunkSize = std::max(WebCore::AudioUtilities::renderQuantumSize, AudioSession::sharedSession().preferredBufferSize());
164
165 // Allocate a ring buffer large enough to contain 2 seconds of audio.
166 m_numberOfFrames = m_description.sampleRate() * 2;
167 m_ringBuffer.reset();
168 m_ringBuffer = makeUnique<CARingBuffer>(makeUniqueRef<SharedRingBufferStorage>(std::bind(&SourceProxy::storageChanged, this, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3)));
169 m_ringBuffer->allocate(m_description.streamDescription(), m_numberOfFrames);
170 }
171
172 ASSERT(is<WebAudioBufferList>(audioData));
173 m_ringBuffer->store(downcast<WebAudioBufferList>(audioData).list(), numberOfFrames, m_writeOffset);
174 m_writeOffset += numberOfFrames;
175
176 size_t framesToSend = numberOfFrames + m_remainingFrameCount;
177 size_t signalCount = framesToSend / m_frameChunkSize;
178 m_remainingFrameCount = framesToSend - (signalCount * m_frameChunkSize);
179 for (unsigned i = 0; i < signalCount; ++i)
180 m_captureSemaphore->signal();
181 }

183 void videoSampleAvailable(MediaSample& sample) final
184 {
185 std::unique_ptr<RemoteVideoSample> remoteSample;
186 if (m_shouldApplyRotation && sample.videoRotation() != MediaSample::VideoRotation::None) {
187 auto pixelBuffer = rotatePixelBuffer(sample);
188 remoteSample = RemoteVideoSample::create(pixelBuffer.get(), sample.presentationTime());
189 } else
190 remoteSample = RemoteVideoSample::create(sample);
191 if (remoteSample)
192 m_connection->send(Messages::RemoteCaptureSampleManager::VideoSampleAvailable(m_id, WTFMove(*remoteSample)), 0);
193 }
194
195 RetainPtr<CVPixelBufferRef> rotatePixelBuffer(MediaSample& sample)
196 {
197 if (!m_rotationSession)
198 m_rotationSession = makeUnique<ImageRotationSessionVT>();
199
200 ImageRotationSessionVT::RotationProperties rotation;
201 switch (sample.videoRotation()) {
202 case MediaSample::VideoRotation::None:
203 ASSERT_NOT_REACHED();
204 rotation.angle = 0;
205 break;
206 case MediaSample::VideoRotation::UpsideDown:
207 rotation.angle = 180;
208 break;
209 case MediaSample::VideoRotation::Right:
210 rotation.angle = 90;
211 break;
212 case MediaSample::VideoRotation::Left:
213 rotation.angle = 270;
214 break;
215 }
216 return m_rotationSession->rotate(sample, rotation, ImageRotationSessionVT::IsCGImageCompatible::No);
217 }

219 void storageChanged(SharedMemory* storage, const WebCore::CAAudioStreamDescription& format, size_t frameCount)
220 {
221 SharedMemory::Handle handle;
222 if (storage)
223 storage->createHandle(handle, SharedMemory::Protection::ReadOnly);
224
225 // FIXME: Send the actual data size with IPCHandle.
226#if OS(DARWIN) || OS(WINDOWS)
227 uint64_t dataSize = handle.size();
228#else
229 uint64_t dataSize = 0;
230#endif
231 m_connection->send(Messages::RemoteCaptureSampleManager::AudioStorageChanged(m_id, SharedMemory::IPCHandle { WTFMove(handle), dataSize }, format, frameCount, *m_captureSemaphore, m_startTime, m_frameChunkSize), 0);
232 }

234 bool preventSourceFromStopping()
235 {
236 // Do not allow the source to stop if we are still using it.
237 return !m_isEnded;
238 }

269void UserMediaCaptureManagerProxy::createMediaSourceForCaptureDeviceWithConstraints(RealtimeMediaSourceIdentifier id, const CaptureDevice& device, String&& hashSalt, const MediaConstraints& mediaConstraints, CreateSourceCallback&& completionHandler)
270{
271 if (!m_connectionProxy->willStartCapture(device.type()))
272 return completionHandler(false, "Request is not allowed"_s, RealtimeMediaSourceSettings { }, { }, { }, { }, 0);
273
274 auto* constraints = mediaConstraints.isValid ? &mediaConstraints : nullptr;
275
276 CaptureSourceOrError sourceOrError;
277 switch (device.type()) {
278 case WebCore::CaptureDevice::DeviceType::Microphone:
279 sourceOrError = RealtimeMediaSourceCenter::singleton().audioCaptureFactory().createAudioCaptureSource(device, WTFMove(hashSalt), constraints);
280 break;
281 case WebCore::CaptureDevice::DeviceType::Camera:
282 sourceOrError = RealtimeMediaSourceCenter::singleton().videoCaptureFactory().createVideoCaptureSource(device, WTFMove(hashSalt), constraints);
283 if (sourceOrError)
284 sourceOrError.captureSource->monitorOrientation(m_orientationNotifier);
285 break;
286 case WebCore::CaptureDevice::DeviceType::Screen:
287 case WebCore::CaptureDevice::DeviceType::Window:
288 sourceOrError = RealtimeMediaSourceCenter::singleton().displayCaptureFactory().createDisplayCaptureSource(device, constraints);
289 break;
290 case WebCore::CaptureDevice::DeviceType::Speaker:
291 case WebCore::CaptureDevice::DeviceType::Unknown:
292 ASSERT_NOT_REACHED();
293 break;
294 }
295
296 bool succeeded = !!sourceOrError;
297 String invalidConstraints;
298 RealtimeMediaSourceSettings settings;
299 RealtimeMediaSourceCapabilities capabilities;
300 Vector<VideoPresetData> presets;
301 IntSize size;
302 double frameRate = 0;
303
304 if (sourceOrError) {
305 auto source = sourceOrError.source();
306#if !RELEASE_LOG_DISABLED
307 source->setLogger(m_connectionProxy->logger(), LoggerHelper::uniqueLogIdentifier());
308#endif
309 settings = source->settings();
310 capabilities = source->capabilities();
311 if (source->isVideoSource()) {
312 auto& videoSource = static_cast<RealtimeVideoSource&>(source.get());
313 presets = videoSource.presetsData();
314 size = videoSource.size();
315 frameRate = videoSource.frameRate();
316 }
317
318 ASSERT(!m_proxies.contains(id));
319 m_proxies.add(id, makeUnique<SourceProxy>(id, m_connectionProxy->connection(), WTFMove(source)));
320 } else
321 invalidConstraints = WTFMove(sourceOrError.errorMessage);
322
323
324 completionHandler(succeeded, invalidConstraints, WTFMove(settings), WTFMove(capabilities), WTFMove(presets), size, frameRate);
325}

352void UserMediaCaptureManagerProxy::applyConstraints(RealtimeMediaSourceIdentifier id, const WebCore::MediaConstraints& constraints)
353{
354 auto* proxy = m_proxies.get(id);
355 if (!proxy) {
356 m_connectionProxy->connection().send(Messages::UserMediaCaptureManager::ApplyConstraintsFailed(id, { }, "Unknown source"_s), 0);
357 return;
358 }
359
360 auto& source = proxy->source();
361 auto result = source.applyConstraints(constraints);
362 if (!result)
363 m_connectionProxy->connection().send(Messages::UserMediaCaptureManager::ApplyConstraintsSucceeded(id, source.settings()), 0);
364 else
365 m_connectionProxy->connection().send(Messages::UserMediaCaptureManager::ApplyConstraintsFailed(id, result->badConstraint, result->message), 0);
366}

368void UserMediaCaptureManagerProxy::clone(RealtimeMediaSourceIdentifier clonedID, RealtimeMediaSourceIdentifier newSourceID)
369{
370 MESSAGE_CHECK(m_proxies.contains(clonedID));
371 MESSAGE_CHECK(!m_proxies.contains(newSourceID));
372 if (auto* proxy = m_proxies.get(clonedID))
373 m_proxies.add(newSourceID, makeUnique<SourceProxy>(newSourceID, m_connectionProxy->connection(), proxy->source().clone()));
374}
