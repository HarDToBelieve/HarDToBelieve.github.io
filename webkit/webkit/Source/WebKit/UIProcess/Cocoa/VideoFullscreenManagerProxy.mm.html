349{

353 auto contextMap = WTFMove(m_contextMap);
354 m_clientCounts.clear();

417VideoFullscreenManagerProxy::ModelInterfaceTuple VideoFullscreenManagerProxy::createModelAndInterface(PlaybackSessionContextIdentifier contextId)
418{
419 auto& playbackSessionModel = m_playbackSessionManagerProxy->ensureModel(contextId);
420 Ref<VideoFullscreenModelContext> model = VideoFullscreenModelContext::create(*this, playbackSessionModel, contextId);
421 auto& playbackSessionInterface = m_playbackSessionManagerProxy->ensureInterface(contextId);
422 Ref<PlatformVideoFullscreenInterface> interface = PlatformVideoFullscreenInterface::create(playbackSessionInterface);
423 m_playbackSessionManagerProxy->addClientForContext(contextId);
424
425 interface->setVideoFullscreenModel(&model.get());
426 interface->setVideoFullscreenChangeObserver(&model.get());
427
428 return std::make_tuple(WTFMove(model), WTFMove(interface));
429}

431VideoFullscreenManagerProxy::ModelInterfaceTuple& VideoFullscreenManagerProxy::ensureModelAndInterface(PlaybackSessionContextIdentifier contextId)
432{
433 auto addResult = m_contextMap.add(contextId, ModelInterfaceTuple());

435 addResult.iterator->value = createModelAndInterface(contextId);
436 return addResult.iterator->value;
437}

469
470void VideoFullscreenManagerProxy::removeClientForContext(PlaybackSessionContextIdentifier contextId)
471{
472 if (!m_clientCounts.contains(contextId))
473 return;
474
475 int clientCount = m_clientCounts.get(contextId);
476 ASSERT(clientCount > 0);
477 clientCount--;
478
479 if (clientCount <= 0) {
480 ensureInterface(contextId).setVideoFullscreenModel(nullptr);
481 m_playbackSessionManagerProxy->removeClientForContext(contextId);
482 m_clientCounts.remove(contextId);
483 m_contextMap.remove(contextId);
484 return;
485 }
486
487 m_clientCounts.set(contextId, clientCount);
488}

490void VideoFullscreenManagerProxy::forEachSession(Function<void(VideoFullscreenModelContext&, PlatformVideoFullscreenInterface&)>&& callback)
491{
492 if (m_contextMap.isEmpty())
493 return;
494
495 Vector<ModelInterfaceTuple> values;
496 values.reserveInitialCapacity(m_contextMap.size());
497 for (auto& value : m_contextMap.values())
498 values.uncheckedAppend(value);
499
500 for (auto& value : values) {
501 RefPtr<VideoFullscreenModelContext> model;
502 RefPtr<PlatformVideoFullscreenInterface> interface;
503 std::tie(model, interface) = value;
504
505 ASSERT(model);
506 ASSERT(interface);
507 if (!model || !interface)
508 continue;
509
510 callback(*model, *interface);
511 }
512}

523void VideoFullscreenManagerProxy::setupFullscreenWithID(PlaybackSessionContextIdentifier contextId, WebKit::LayerHostingContextID videoLayerID, const WebCore::FloatRect& initialRect, const WebCore::FloatSize& videoDimensions, float hostingDeviceScaleFactor, HTMLMediaElementEnums::VideoFullscreenMode videoFullscreenMode, bool allowsPictureInPicture, bool standby, bool blocksReturnToFullscreenFromPictureInPicture)
524{
525 MESSAGE_CHECK(videoLayerID);
526
527 auto& [model, interface] = ensureModelAndInterface(contextId);
528 ensureClientForContext(contextId);
529
530 if (m_mockVideoPresentationModeEnabled) {
531 if (!videoDimensions.isEmpty())
532 m_mockPictureInPictureWindowSize.setHeight(DefaultMockPictureInPictureWindowWidth / videoDimensions.aspectRatio());
533#if PLATFORM(IOS_FAMILY)
534 requestVideoContentLayer(contextId);
535#else
536 didSetupFullscreen(contextId);
537#endif
538 return;
539 }
540
541 RetainPtr<WKLayerHostView> view = static_cast<WKLayerHostView*>(model->layerHostView());
542 if (!view) {
543 view = adoptNS([[WKLayerHostView alloc] init]);
544#if PLATFORM(MAC)
545 [view setWantsLayer:YES];
546#endif
547 model->setLayerHostView(view);
548 }
549 [view setContextID:videoLayerID];
550 if (hostingDeviceScaleFactor != 1) {
551 // Invert the scale transform added in the WebProcess to fix <rdar://problem/18316542>.
552 float inverseScale = 1 / hostingDeviceScaleFactor;
553 [[view layer] setSublayerTransform:CATransform3DMakeScale(inverseScale, inverseScale, 1)];
554 }
555
556#if PLATFORM(IOS_FAMILY)
557 auto* rootNode = downcast<RemoteLayerTreeDrawingAreaProxy>(*m_page->drawingArea()).remoteLayerTreeHost().rootNode();
558 UIView *parentView = rootNode ? rootNode->uiView() : nil;
559 interface->setupFullscreen(*model->layerHostView(), initialRect, videoDimensions, parentView, videoFullscreenMode, allowsPictureInPicture, standby, blocksReturnToFullscreenFromPictureInPicture);
560#else
561 UNUSED_PARAM(videoDimensions);
562 UNUSED_PARAM(blocksReturnToFullscreenFromPictureInPicture);
563 IntRect initialWindowRect;
564 m_page->rootViewToWindow(enclosingIntRect(initialRect), initialWindowRect);
565 interface->setupFullscreen(*model->layerHostView(), initialWindowRect, m_page->platformWindow(), videoFullscreenMode, allowsPictureInPicture);
566#endif
567}

595void VideoFullscreenManagerProxy::enterFullscreen(PlaybackSessionContextIdentifier contextId)
596{
597 if (m_mockVideoPresentationModeEnabled) {
598 didEnterFullscreen(contextId, m_mockPictureInPictureWindowSize);
599 return;
600 }
601
602 auto& interface = ensureInterface(contextId);
603 interface.enterFullscreen();
604
605 // Only one context can be in a given full screen mode at a time:
606 for (auto& contextPair : m_contextMap) {
607 auto& otherContextId = contextPair.key;
608 if (contextId == otherContextId)
609 continue;
610
611 auto& otherInterface = std::get<1>(contextPair.value);
612 if (otherInterface->hasMode(interface.mode()))
613 otherInterface->requestHideAndExitFullscreen();
614 }
615}

617void VideoFullscreenManagerProxy::exitFullscreen(PlaybackSessionContextIdentifier contextId, WebCore::FloatRect finalRect, CompletionHandler<void(bool)>&& completionHandler)
618{
619 ASSERT(m_contextMap.contains(contextId));
620 if (!m_contextMap.contains(contextId)) {
621 completionHandler(false);
622 return;
623 }
624
625#if !PLATFORM(IOS_FAMILY)
626 IntRect finalWindowRect;
627 m_page->rootViewToWindow(enclosingIntRect(finalRect), finalWindowRect);
628#else
629 if (hasMode(WebCore::HTMLMediaElementEnums::VideoFullscreenModeStandard))
630 m_page->fullscreenMayReturnToInline();
631#endif
632
633 if (m_mockVideoPresentationModeEnabled) {
634#if PLATFORM(IOS_FAMILY)
635 returnVideoContentLayer(contextId);
636#else
637 didExitFullscreen(contextId);
638#endif
639 completionHandler(true);
640 return;
641 }
642
643#if PLATFORM(IOS_FAMILY)
644 completionHandler(ensureInterface(contextId).exitFullscreen(finalRect));
645#else
646 completionHandler(ensureInterface(contextId).exitFullscreen(finalWindowRect, m_page->platformWindow()));
647#endif
648}

752void VideoFullscreenManagerProxy::requestCloseAllMediaPresentations(PlaybackSessionContextIdentifier contextId, bool finishedWithMedia, CompletionHandler<void()>&& completionHandler)
