154- (void)presentWithRequestData:(const WebCore::ContactsRequestData&)requestData completionHandler:(WTF::CompletionHandler<void(std::optional<Vector<WebCore::ContactInfo>>&&)>&&)completionHandler
155{
156 _properties = requestData.properties;
157 _completionHandler = WTFMove(completionHandler);
158
159 if (requestData.multiple)
160 _contactPickerDelegate = adoptNS([[WKCNContactPickerMultiSelectDelegate alloc] initWithContactPickerDelegate:self]);
161 else
162 _contactPickerDelegate = adoptNS([[WKCNContactPickerSingleSelectDelegate alloc] initWithContactPickerDelegate:self]);
163
164#if HAVE(CNCONTACTPICKERVIEWCONTROLLER)
165 _contactPickerViewController = adoptNS([allocCNContactPickerViewControllerInstance() init]);
166 [_contactPickerViewController setDelegate:_contactPickerDelegate.get()];
167 [_contactPickerViewController setPrompt:requestData.url];
168
169 UIViewController *presentationViewController = [UIViewController _viewControllerForFullScreenPresentationFromView:_webView.get().get()];
170 [presentationViewController presentViewController:_contactPickerViewController.get() animated:YES completion:[weakSelf = WeakObjCPtr<WKContactPicker>(self)] {
171 auto strongSelf = weakSelf.get();
172 if (!strongSelf)
173 return;
174
175 if ([[strongSelf delegate] respondsToSelector:@selector(contactPickerDidPresent:)])
176 [[strongSelf delegate] contactPickerDidPresent:strongSelf.get()];
177 }];
178#endif
179}

191- (void)contactPicker:(CNContactPickerViewController *)picker didSelectContact:(CNContact *)contact
192{
193 Vector<WebCore::ContactInfo> info = { [self _contactInfoFromCNContact:contact] };
194 [self _contactPickerDidDismissWithContactInfo:WTFMove(info)];
195}

197- (void)contactPicker:(CNContactPickerViewController *)picker didSelectContacts:(NSArray<CNContact*> *)contacts
198{
199 Vector<WebCore::ContactInfo> info;
200 info.reserveInitialCapacity(contacts.count);
201 for (CNContact *contact in contacts)
202 info.uncheckedAppend([self _contactInfoFromCNContact:contact]);
203 [self _contactPickerDidDismissWithContactInfo:WTFMove(info)];
204}

206#endif

208- (void)_contactPickerDidDismissWithContactInfo:(Vector<WebCore::ContactInfo>&&)info
209{
210 _completionHandler(WTFMove(info));
211
212 if ([_delegate respondsToSelector:@selector(contactPickerDidDismiss:)])
213 [_delegate contactPickerDidDismiss:self];
214}

216- (WebCore::ContactInfo)_contactInfoFromCNContact:(CNContact *)contact
217{
218 WebCore::ContactInfo contactInfo;
219
220 if (_properties.contains(WebCore::ContactProperty::Name)) {
221 NSString *contactName = [getCNContactFormatterClass() stringFromContact:contact style:CNContactFormatterStyleFullName];
222 contactInfo.name = { contactName };
223 }
224
225 if (_properties.contains(WebCore::ContactProperty::Email)) {
226 for (CNLabeledValue<NSString *> *emailAddress in contact.emailAddresses)
227 contactInfo.email.append(emailAddress.value);
228 }
229
230 if (_properties.contains(WebCore::ContactProperty::Tel)) {
231 for (CNLabeledValue<CNPhoneNumber *> *phoneNumber in contact.phoneNumbers)
232 contactInfo.tel.append(phoneNumber.value.stringValue);
233 }
234
235 return contactInfo;
236}

253- (NSArray<CNContact*> *)_contactsFromJSContacts:(NSArray *)jsContacts
254{
255 if (!jsContacts)
256 return [NSArray array];
257
258 NSMutableArray<CNContact*> *contacts = [NSMutableArray arrayWithCapacity:jsContacts.count];
259
260 NSPredicate *stringValuePredicate = [NSPredicate predicateWithFormat:@"self isKindOfClass: %@", [NSString class]];
261
262 for (id jsContact in jsContacts) {
263 if (![jsContact isKindOfClass:[NSDictionary class]])
264 continue;
265
266 auto contact = adoptNS([allocCNMutableContactInstance() init]);
267
268 id names = [(NSDictionary *)jsContact objectForKey:@"name"];
269 if ([names isKindOfClass:[NSArray class]]) {
270 for (NSString *name in [names filteredArrayUsingPredicate:stringValuePredicate]) {
271 [contact setGivenName:name];
272 break;
273 }
274 }
275
276 id emails = [(NSDictionary *)jsContact objectForKey:@"email"];
277 if ([emails isKindOfClass:[NSArray class]]) {
278 NSMutableArray<CNLabeledValue<NSString*>*> *emailAddresses = [NSMutableArray array];
279 for (NSString *email in [emails filteredArrayUsingPredicate:stringValuePredicate]) {
280 CNLabeledValue<NSString*> *labeledValue = [getCNLabeledValueClass() labeledValueWithLabel:nil value:email];
281 [emailAddresses addObject:labeledValue];
282 }
283 [contact setEmailAddresses:emailAddresses];
284 }
285
286 id phoneNumbers = [(NSDictionary *)jsContact objectForKey:@"tel"];
287 if ([phoneNumbers isKindOfClass:[NSArray class]]) {
288 NSMutableArray<CNLabeledValue<CNPhoneNumber*>*> *numbers = [NSMutableArray array];
289 for (NSString *phoneNumber in [phoneNumbers filteredArrayUsingPredicate:stringValuePredicate]) {
290 CNPhoneNumber *cnPhoneNumber = [getCNPhoneNumberClass() phoneNumberWithStringValue:phoneNumber];
291 CNLabeledValue<CNPhoneNumber*> *labeledValue = [getCNLabeledValueClass() labeledValueWithLabel:nil value:cnPhoneNumber];
292 [numbers addObject:labeledValue];
293 }
294 [contact setPhoneNumbers:numbers];
295 }
296
297 [contacts addObject:contact.get()];
298 }
299
300 return contacts;
301}
