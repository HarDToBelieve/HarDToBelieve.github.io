93static void appendFilesAsShareableURLs(RetainPtr<NSMutableArray>&& shareDataArray, const Vector<WebCore::RawFile>& files, NSURL* temporaryDirectory, CompletionHandler<void(RetainPtr<NSMutableArray>&&)>&& completionHandler)
94{
95 struct FileWriteTask {
96 String fileName;
97 RetainPtr<NSData> fileData;
98 };
99 Vector<FileWriteTask> fileWriteTasks;
100 for (auto& file : files)
101 fileWriteTasks.append({ file.fileName.isolatedCopy(), file.fileData->createNSData() });
102
103 auto queue = WorkQueue::create("com.apple.WebKit.WKShareSheet.ShareableFileWriter");
104 queue->dispatch([shareDataArray = WTFMove(shareDataArray), fileWriteTasks = WTFMove(fileWriteTasks), temporaryDirectory = retainPtr(temporaryDirectory), completionHandler = WTFMove(completionHandler)]() mutable {
105 for (auto& fileWriteTask : fileWriteTasks) {
106 NSURL *fileURL = [WKShareSheet writeFileToShareableURL:WebCore::ResourceResponseBase::sanitizeSuggestedFilename(fileWriteTask.fileName) data:fileWriteTask.fileData.get() temporaryDirectory:temporaryDirectory.get()];
107 if (!fileURL) {
108 shareDataArray = nullptr;
109 break;
110 }
111 [shareDataArray addObject:fileURL];
112 }
113 RunLoop::main().dispatch([completionHandler = WTFMove(completionHandler), shareDataArray = WTFMove(shareDataArray)]() mutable {
114 completionHandler(WTFMove(shareDataArray));
115 });
116 });
117}

119- (void)presentWithParameters:(const WebCore::ShareDataWithParsedURL &)data inRect:(std::optional<WebCore::FloatRect>)rect completionHandler:(WTF::CompletionHandler<void(bool)>&&)completionHandler
120{
121 auto shareDataArray = adoptNS([[NSMutableArray alloc] init]);
122
123 if (!data.shareData.text.isEmpty())
124 [shareDataArray addObject:(NSString *)data.shareData.text];
125
126 if (data.url) {
127 NSURL *url = (NSURL *)data.url.value();
128#if PLATFORM(IOS_FAMILY)
129 if (!data.shareData.title.isEmpty())
130 url._title = data.shareData.title;
131#endif
132 [shareDataArray addObject:url];
133 }
134
135 if (!data.shareData.title.isEmpty() && ![shareDataArray count])
136 [shareDataArray addObject:(NSString *)data.shareData.title];
137
138 _completionHandler = WTFMove(completionHandler);
139
140 if (auto resolution = [_webView _resolutionForShareSheetImmediateCompletionForTesting]) {
141 _didShareSuccessfully = *resolution;
142 [self dismiss];
143 return;
144 }
145
146 if (!data.files.isEmpty()) {
147 _temporaryFileShareDirectory = [WKShareSheet createTemporarySharingDirectory];
148 appendFilesAsShareableURLs(WTFMove(shareDataArray), data.files, _temporaryFileShareDirectory.get(), [retainedSelf = retainPtr(self), rect = WTFMove(rect)](RetainPtr<NSMutableArray>&& shareDataArray) mutable {
149 if (!shareDataArray) {
150 [retainedSelf dismiss];
151 return;
152 }
153 [retainedSelf presentWithShareDataArray:shareDataArray.get() inRect:rect];
154 });
155 return;
156 }
157
158 [self presentWithShareDataArray:shareDataArray.get() inRect:rect];
159}

161- (void)presentWithShareDataArray:(NSArray *)sharingItems inRect:(std::optional<WebCore::FloatRect>)rect
162{
163 WKWebView *webView = _webView.getAutoreleased();
164
165#if PLATFORM(MAC)
166 _sharingServicePicker = adoptNS([[NSSharingServicePicker alloc] initWithItems:sharingItems]);
167 _sharingServicePicker.get().delegate = self;
168
169 // WKShareSheet can be released under NSSharingServicePicker delegate callbacks.
170 RetainPtr<WKShareSheet> protector(self);
171 NSRect presentationRect;
172
173 if (rect)
174 presentationRect = *rect;
175 else {
176 NSPoint location = [NSEvent mouseLocation];
177 NSRect mouseLocationRect = NSMakeRect(location.x, location.y, 1.0, 1.0);
178 NSRect mouseLocationInWindow = [webView.window convertRectFromScreen:mouseLocationRect];
179 presentationRect = [webView convertRect:mouseLocationInWindow fromView:nil];
180 }
181 [_sharingServicePicker showRelativeToRect:presentationRect ofView:webView preferredEdge:NSMinYEdge];
182#else
183 _shareSheetViewController = adoptNS([[UIActivityViewController alloc] initWithActivityItems:sharingItems applicationActivities:nil]);
184 [_shareSheetViewController setCompletionWithItemsHandler:^(NSString *, BOOL completed, NSArray *, NSError *) {
185 _didShareSuccessfully |= completed;
186
187 // Make sure that we're actually not presented anymore (-completionWithItemsHandler can be called multiple times
188 // before the share sheet is actually dismissed), and if so, clean up.
189 if (![_shareSheetViewController presentingViewController])
190 [self dismiss];
191 }];
192
193 UIPopoverPresentationController *popoverController = [_shareSheetViewController popoverPresentationController];
194 if (rect) {
195 popoverController.sourceView = webView;
196 popoverController.sourceRect = *rect;
197 } else
198 popoverController._centersPopoverIfSourceViewNotSet = YES;
199
200 if ([_delegate respondsToSelector:@selector(shareSheet:willShowActivityItems:)])
201 [_delegate shareSheet:self willShowActivityItems:sharingItems];
202
203 _presentationViewController = [UIViewController _viewControllerForFullScreenPresentationFromView:webView];
204 [_presentationViewController presentViewController:_shareSheetViewController.get() animated:YES completion:nil];
205#endif
206}

237- (void)dismiss
238{
239 auto completionHandler = WTFMove(_completionHandler);
240 if (completionHandler)
241 completionHandler(_didShareSuccessfully);
242
243 if (_didShareSuccessfully) {
244 // <rdar://problem/63030288>: didShareItems callback for NSSharingServiceDelegate currently is called
245 // before the temporary files are copied, so we can't delete them here. UIActivityViewController doesn't
246 // have this problem, so we can delete immediately for iOS.
247#if PLATFORM(IOS_FAMILY)
248 [[NSFileManager defaultManager] removeItemAtURL:_temporaryFileShareDirectory.get() error:nil];
249#endif
250 } else
251 [[NSFileManager defaultManager] removeItemAtURL:_temporaryFileShareDirectory.get() error:nil];
252
253 _temporaryFileShareDirectory = nullptr;
254
255 auto dispatchDidDismiss = ^{
256 if ([_delegate respondsToSelector:@selector(shareSheetDidDismiss:)])
257 [_delegate shareSheetDidDismiss:self];
258 };
259
260#if PLATFORM(MAC)
261 [_sharingServicePicker hide];
262 dispatchDidDismiss();
263#else
264 if (!_presentationViewController)
265 return;
266
267 UIViewController *currentPresentedViewController = [_presentationViewController presentedViewController];
268 if (currentPresentedViewController != _shareSheetViewController) {
269 dispatchDidDismiss();
270 return;
271 }
272
273 [currentPresentedViewController dismissViewControllerAnimated:YES completion:^{
274 dispatchDidDismiss();
275 _presentationViewController = nil;
276 }];
277#endif
278}

280#if PLATFORM(MAC)
281+ (BOOL)setQuarantineInformationForFilePath:(NSURL *)fileURL
282{
283 auto quarantineProperties = @{
284 (__bridge NSString *)kLSQuarantineTypeKey: (__bridge NSString *)kLSQuarantineTypeWebDownload,
285 (__bridge NSString *)kLSQuarantineAgentBundleIdentifierKey: WebCore::applicationBundleIdentifier()
286 };
287
288 if (![fileURL setResourceValue:quarantineProperties forKey:NSURLQuarantinePropertiesKey error:nil])
289 return NO;
290
291 // Whether the file was downloaded by sandboxed WebProcess or not, LSSetItemAttribute resets the flags to 0 (advisory QTN_FLAG_DOWNLOAD,
292 // which can be then removed by WebProcess). Replace the flags with sandbox quarantine ones, which cannot be removed by sandboxed processes.
293 return [WKShareSheet applyQuarantineSandboxAndDownloadFlagsToFileAtPath:fileURL];
294}
295
296+ (BOOL)applyQuarantineSandboxAndDownloadFlagsToFileAtPath:(NSURL *)fileURL
297{
298 qtn_file_t fq = qtn_file_alloc();
299 auto scopeExit = WTF::makeScopeExit([&] {
300 qtn_file_free(fq);
301 });
302
303 int quarantineError = qtn_file_init_with_path(fq, fileURL.fileSystemRepresentation);
304 if (quarantineError)
305 return NO;
306
307 quarantineError = qtn_file_set_flags(fq, QTN_FLAG_SANDBOX | QTN_FLAG_DOWNLOAD);
308 if (quarantineError)
309 return NO;
310
311 quarantineError = qtn_file_apply_to_path(fq, fileURL.fileSystemRepresentation);
312
313 return YES;
314}
315#endif

317+ (NSURL *)createTemporarySharingDirectory
318{
319 NSString *temporaryDirectory = FileSystem::createTemporaryDirectory(@"WKFileShare");
320
321 if (![temporaryDirectory length])
322 return nil;
323
324 return [NSURL fileURLWithPath:temporaryDirectory isDirectory:YES];
325}
326
327+ (NSURL *)createRandomSharingDirectoryForFile:(NSURL *)temporaryDirectory
328{
329 NSString *randomDirectory = createCanonicalUUIDString();
330 if (![randomDirectory length] || !temporaryDirectory)
331 return nil;
332 NSURL *dataPath = [temporaryDirectory URLByAppendingPathComponent:randomDirectory];
333
334 if (![[NSFileManager defaultManager] createDirectoryAtURL:dataPath withIntermediateDirectories:NO attributes:nil error:nil])
335 return nil;
336 return dataPath;
337}

339+ (NSURL *)writeFileToShareableURL:(NSString *)fileName data:(NSData *)fileData temporaryDirectory:(NSURL *)temporaryDirectory
340{
341 ASSERT(!RunLoop::isMain());
342 if (!temporaryDirectory || ![fileName length] || !fileData)
343 return nil;
344
345 NSURL *temporaryDirectoryForFile = [WKShareSheet createRandomSharingDirectoryForFile:temporaryDirectory];
346 if (!temporaryDirectoryForFile)
347 return nil;
348
349 NSURL *fileURL = [temporaryDirectoryForFile URLByAppendingPathComponent:fileName];
350
351 if (![fileData writeToURL:fileURL options:NSDataWritingAtomic error:nil])
352 return nil;
353#if PLATFORM(MAC)
354 if (![WKShareSheet setQuarantineInformationForFilePath:fileURL])
355 return nil;
356#endif
357 return fileURL;
358}
