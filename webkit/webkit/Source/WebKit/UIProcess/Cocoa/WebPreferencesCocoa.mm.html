39static inline NSString *makeKey(const String& identifier, const String& keyPrefix, const String& key)
40{
41 ASSERT(!identifier.isEmpty());
42 return String(identifier + keyPrefix + key);
43}
44
45bool WebPreferences::platformGetStringUserValueForKey(const String& key, String& userValue)
46{
47 if (!m_identifier)
48 return false;
49
50 id object = [[NSUserDefaults standardUserDefaults] objectForKey:makeKey(m_identifier, m_keyPrefix, key)];
51 if (!object)
52 return false;
53 if (![object isKindOfClass:[NSString class]])
54 return false;
55
56 userValue = (NSString *)object;
57 return true;
58}

105static id debugUserDefaultsValue(const String& identifier, const String& keyPrefix, const String& globalDebugKeyPrefix, const String& key)
106{
107 NSUserDefaults *standardUserDefaults = [NSUserDefaults standardUserDefaults];
108 id object = nil;
109
110 if (!identifier.isEmpty())
111 object = [standardUserDefaults objectForKey:makeKey(identifier, keyPrefix, key)];
112
113 if (!object) {
114 // Allow debug preferences to be set globally, using the debug key prefix.
115 object = [standardUserDefaults objectForKey:[globalDebugKeyPrefix stringByAppendingString:key]];
116 }
117
118 return object;
119}
120
121static void setDebugBoolValueIfInUserDefaults(const String& identifier, const String& keyPrefix, const String& globalDebugKeyPrefix, const String& key, WebPreferencesStore& store)
122{
123 id object = debugUserDefaultsValue(identifier, keyPrefix, globalDebugKeyPrefix, key);
124 if (!object)
125 return;
126 if (![object respondsToSelector:@selector(boolValue)])
127 return;
128
129 store.setBoolValueForKey(key, [object boolValue]);
130}

132static void setDebugUInt32ValueIfInUserDefaults(const String& identifier, const String& keyPrefix, const String& globalDebugKeyPrefix, const String& key, WebPreferencesStore& store)

143void WebPreferences::platformInitializeStore()
144{
145 @autoreleasepool {
146#if ENABLE(MEDIA_STREAM)
147 // NOTE: This is set here, and does not setting the default using the 'defaultValue' mechanism, because the
148 // 'defaultValue' must be the same in both the UIProcess and WebProcess, which may not be true for audio
149 // and video capture state as the WebProcess is not entitled to use the camera or microphone by default.
150 // If other preferences need to dynamically set the initial value based on host app state, we should extended
151 // the declarative format rather than adding more special cases here.
152 m_store.setBoolValueForKey(WebPreferencesKey::mediaDevicesEnabledKey(), UserMediaPermissionRequestManagerProxy::permittedToCaptureAudio() || UserMediaPermissionRequestManagerProxy::permittedToCaptureVideo());
153#endif
154
155#define INITIALIZE_DEBUG_PREFERENCE_FROM_NSUSERDEFAULTS(KeyUpper, KeyLower, TypeName, Type, DefaultValue, HumanReadableName, HumanReadableDescription) \
156 setDebug##TypeName##ValueIfInUserDefaults(m_identifier, m_keyPrefix, m_globalDebugKeyPrefix, WebPreferencesKey::KeyLower##Key(), m_store);
157
158 FOR_EACH_WEBKIT_DEBUG_PREFERENCE(INITIALIZE_DEBUG_PREFERENCE_FROM_NSUSERDEFAULTS)
159
160#undef INITIALIZE_DEBUG_PREFERENCE_FROM_NSUSERDEFAULTS
161
162 if (!m_identifier)
163 return;
164
165#define INITIALIZE_PREFERENCE_FROM_NSUSERDEFAULTS(KeyUpper, KeyLower, TypeName, Type, DefaultValue, HumanReadableName, HumanReadableDescription) \
166 Type user##KeyUpper##Value; \
167 if (platformGet##TypeName##UserValueForKey(WebPreferencesKey::KeyLower##Key(), user##KeyUpper##Value)) \
168 m_store.set##TypeName##ValueForKey(WebPreferencesKey::KeyLower##Key(), user##KeyUpper##Value);
169
170 FOR_EACH_WEBKIT_PREFERENCE(INITIALIZE_PREFERENCE_FROM_NSUSERDEFAULTS)
171
172#undef INITIALIZE_PREFERENCE_FROM_NSUSERDEFAULTS
173 }
174}

176void WebPreferences::platformUpdateStringValueForKey(const String& key, const String& value)
177{
178 if (!m_identifier)
179 return;
180
181 [[NSUserDefaults standardUserDefaults] setObject:nsStringFromWebCoreString(value) forKey:makeKey(m_identifier, m_keyPrefix, key)];
182}
