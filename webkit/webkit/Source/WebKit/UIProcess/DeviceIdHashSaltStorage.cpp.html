43static constexpr unsigned deviceIdHashSaltStorageVersion { 1 };
44static constexpr unsigned hashSaltSize { 48 };
45static constexpr unsigned randomDataSize { hashSaltSize / 16 };

47Ref<DeviceIdHashSaltStorage> DeviceIdHashSaltStorage::create(const String& deviceIdHashSaltStorageDirectory)
48{
49 auto deviceIdHashSaltStorage = adoptRef(*new DeviceIdHashSaltStorage(deviceIdHashSaltStorageDirectory));
50 return deviceIdHashSaltStorage;
51}

53void DeviceIdHashSaltStorage::completePendingHandler(CompletionHandler<void(HashSet<SecurityOriginData>&&)>&& completionHandler)
54{
55 ASSERT(RunLoop::isMain());
56
57 HashSet<SecurityOriginData> origins;
58
59 for (auto& hashSaltForOrigin : m_deviceIdHashSaltForOrigins) {
60 origins.add(hashSaltForOrigin.value->documentOrigin);
61 origins.add(hashSaltForOrigin.value->parentOrigin);
62 }
63
64 RunLoop::main().dispatch([origins = WTFMove(origins), completionHandler = WTFMove(completionHandler)]() mutable {
65 completionHandler(WTFMove(origins));
66 });
67}

69DeviceIdHashSaltStorage::DeviceIdHashSaltStorage(const String& deviceIdHashSaltStorageDirectory)
70 : m_queue(WorkQueue::create("com.apple.WebKit.DeviceIdHashSaltStorage"))
71 , m_deviceIdHashSaltStorageDirectory(!deviceIdHashSaltStorageDirectory.isEmpty() ? FileSystem::pathByAppendingComponent(deviceIdHashSaltStorageDirectory, String::number(deviceIdHashSaltStorageVersion)) : String())
72{
73 if (m_deviceIdHashSaltStorageDirectory.isEmpty()) {
74 m_isLoaded = true;
75 return;
76 }
77
78 loadStorageFromDisk([this, protectedThis = makeRef(*this)] (auto&& deviceIdHashSaltForOrigins) {
79 ASSERT(RunLoop::isMain());
80 m_deviceIdHashSaltForOrigins = WTFMove(deviceIdHashSaltForOrigins);
81 m_isLoaded = true;
82
83 auto pendingCompletionHandlers = WTFMove(m_pendingCompletionHandlers);
84 for (auto& completionHandler : pendingCompletionHandlers)
85 completionHandler();
86 });
87}

89DeviceIdHashSaltStorage::~DeviceIdHashSaltStorage()
90{
91 auto pendingCompletionHandlers = WTFMove(m_pendingCompletionHandlers);
92 for (auto& completionHandler : pendingCompletionHandlers)
93 completionHandler();
94}

96static std::optional<SecurityOriginData> getSecurityOriginData(const char* name, KeyedDecoder* decoder)
97{
98 String origin;
99
100 if (!decoder->decodeString(name, origin))
101 return std::nullopt;
102
103 auto securityOriginData = SecurityOriginData::fromDatabaseIdentifier(origin);
104 if (!securityOriginData)
105 return std::nullopt;
106
107 return securityOriginData;
108}

110void DeviceIdHashSaltStorage::loadStorageFromDisk(CompletionHandler<void(HashMap<String, std::unique_ptr<HashSaltForOrigin>>&&)>&& completionHandler)
111{
112 m_queue->dispatch([this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)]() mutable {
113 ASSERT(!RunLoop::isMain());
114
115 FileSystem::makeAllDirectories(m_deviceIdHashSaltStorageDirectory);
116
117 HashMap<String, std::unique_ptr<HashSaltForOrigin>> deviceIdHashSaltForOrigins;
118 for (auto& origin : FileSystem::listDirectory(m_deviceIdHashSaltStorageDirectory)) {
119 auto originPath = FileSystem::pathByAppendingComponent(m_deviceIdHashSaltStorageDirectory, origin);
120 auto deviceIdHashSalt = URL::fileURLWithFileSystemPath(originPath).lastPathComponent().toString();
121
122 if (hashSaltSize != deviceIdHashSalt.length()) {
123 RELEASE_LOG_ERROR(DiskPersistency, "DeviceIdHashSaltStorage: The length of the hash salt (%d) is different to the length of the hash salts defined in WebKit (%d)", deviceIdHashSalt.length(), hashSaltSize);
124 continue;
125 }
126
127 if (!FileSystem::fileSize(originPath)) {
128 RELEASE_LOG_ERROR(DiskPersistency, "DeviceIdHashSaltStorage: Impossible to get the file size of: '%s'", originPath.utf8().data());
129 continue;
130 }
131
132 auto decoder = createForFile(originPath);
133
134 if (!decoder) {
135 RELEASE_LOG_ERROR(DiskPersistency, "DeviceIdHashSaltStorage: Impossible to access the file to restore the hash salt: '%s'", originPath.utf8().data());
136 continue;
137 }
138
139 auto hashSaltForOrigin = getDataFromDecoder(decoder.get(), WTFMove(deviceIdHashSalt));
140
141 auto origins = makeString(hashSaltForOrigin->documentOrigin.toString(), hashSaltForOrigin->parentOrigin.toString());
142 auto deviceIdHashSaltForOrigin = deviceIdHashSaltForOrigins.ensure(origins, [hashSaltForOrigin = WTFMove(hashSaltForOrigin)] () mutable {
143 return WTFMove(hashSaltForOrigin);
144 });
145
146 if (!deviceIdHashSaltForOrigin.isNewEntry)
147 RELEASE_LOG_ERROR(DiskPersistency, "DeviceIdHashSaltStorage: There are two files with different hash salts for the same origin: '%s'", originPath.utf8().data());
148 }
149
150 RunLoop::main().dispatch([deviceIdHashSaltForOrigins = WTFMove(deviceIdHashSaltForOrigins), completionHandler = WTFMove(completionHandler)]() mutable {
151 completionHandler(WTFMove(deviceIdHashSaltForOrigins));
152 });
153 });
154}

156std::unique_ptr<DeviceIdHashSaltStorage::HashSaltForOrigin> DeviceIdHashSaltStorage::getDataFromDecoder(KeyedDecoder* decoder, String&& deviceIdHashSalt) const
157{
158 auto securityOriginData = getSecurityOriginData("origin", decoder);
159 if (!securityOriginData) {
160 RELEASE_LOG_ERROR(DiskPersistency, "DeviceIdHashSaltStorage: The security origin data in the file is not correct: '%s'", deviceIdHashSalt.utf8().data());
161 return nullptr;
162 }
163
164 auto parentSecurityOriginData = getSecurityOriginData("parentOrigin", decoder);
165 if (!parentSecurityOriginData) {
166 RELEASE_LOG_ERROR(DiskPersistency, "DeviceIdHashSaltStorage: The parent security origin data in the file is not correct: '%s'", deviceIdHashSalt.utf8().data());
167 return nullptr;
168 }
169
170 double lastTimeUsed;
171 if (!decoder->decodeDouble("lastTimeUsed", lastTimeUsed)) {
172 RELEASE_LOG_ERROR(DiskPersistency, "DeviceIdHashSaltStorage: The last time used was not correctly restored for: '%s'", deviceIdHashSalt.utf8().data());
173 return nullptr;
174 }
175
176 auto hashSaltForOrigin = makeUnique<HashSaltForOrigin>(WTFMove(securityOriginData.value()), WTFMove(parentSecurityOriginData.value()), WTFMove(deviceIdHashSalt));
177
178 hashSaltForOrigin->lastTimeUsed = WallTime::fromRawSeconds(lastTimeUsed);
179
180 return hashSaltForOrigin;
181}

183std::unique_ptr<KeyedEncoder> DeviceIdHashSaltStorage::createEncoderFromData(const HashSaltForOrigin& hashSaltForOrigin) const
184{
185 auto encoder = KeyedEncoder::encoder();
186 encoder->encodeString("origin", hashSaltForOrigin.documentOrigin.databaseIdentifier());
187 encoder->encodeString("parentOrigin", hashSaltForOrigin.parentOrigin.databaseIdentifier());
188 encoder->encodeDouble("lastTimeUsed", hashSaltForOrigin.lastTimeUsed.secondsSinceEpoch().value());
189 return encoder;
190}

192void DeviceIdHashSaltStorage::storeHashSaltToDisk(const HashSaltForOrigin& hashSaltForOrigin)
193{
194 if (m_deviceIdHashSaltStorageDirectory.isEmpty())
195 return;
196
197 m_queue->dispatch([this, protectedThis = makeRef(*this), hashSaltForOrigin = hashSaltForOrigin.isolatedCopy()]() mutable {
198 auto encoder = createEncoderFromData(hashSaltForOrigin);
199 writeToDisk(WTFMove(encoder), FileSystem::pathByAppendingComponent(m_deviceIdHashSaltStorageDirectory, hashSaltForOrigin.deviceIdHashSalt));
200 });
201}

203void DeviceIdHashSaltStorage::completeDeviceIdHashSaltForOriginCall(SecurityOriginData&& documentOrigin, SecurityOriginData&& parentOrigin, CompletionHandler<void(String&&)>&& completionHandler)
204{
205 auto origins = makeString(documentOrigin.toString(), parentOrigin.toString());
206 auto& deviceIdHashSalt = m_deviceIdHashSaltForOrigins.ensure(origins, [documentOrigin = WTFMove(documentOrigin), parentOrigin = WTFMove(parentOrigin)] () mutable {
207 uint64_t randomData[randomDataSize];
208 cryptographicallyRandomValues(reinterpret_cast<unsigned char*>(randomData), sizeof(randomData));
209
210 StringBuilder builder;
211 builder.reserveCapacity(hashSaltSize);
212 for (unsigned i = 0; i < randomDataSize; i++)
213 builder.append(hex(randomData[i]));
214
215 String deviceIdHashSalt = builder.toString();
216
217 auto newHashSaltForOrigin = makeUnique<HashSaltForOrigin>(WTFMove(documentOrigin), WTFMove(parentOrigin), WTFMove(deviceIdHashSalt));
218
219 return newHashSaltForOrigin;
220 }).iterator->value;
221
222 deviceIdHashSalt->lastTimeUsed = WallTime::now();
223
224 storeHashSaltToDisk(*deviceIdHashSalt.get());
225
226 completionHandler(String(deviceIdHashSalt->deviceIdHashSalt));
227}

229void DeviceIdHashSaltStorage::deviceIdHashSaltForOrigin(const SecurityOrigin& documentOrigin, const SecurityOrigin& parentOrigin, CompletionHandler<void(String&&)>&& completionHandler)
230{
231 ASSERT(RunLoop::isMain());
232
233 if (!m_isLoaded) {
234 m_pendingCompletionHandlers.append([this, documentOrigin = documentOrigin.data().isolatedCopy(), parentOrigin = parentOrigin.data().isolatedCopy(), completionHandler = WTFMove(completionHandler)]() mutable {
235 completeDeviceIdHashSaltForOriginCall(WTFMove(documentOrigin), WTFMove(parentOrigin), WTFMove(completionHandler));
236 });
237 return;
238 }
239
240 completeDeviceIdHashSaltForOriginCall(SecurityOriginData(documentOrigin.data()), SecurityOriginData(parentOrigin.data()), WTFMove(completionHandler));
241}

243void DeviceIdHashSaltStorage::getDeviceIdHashSaltOrigins(CompletionHandler<void(HashSet<SecurityOriginData>&&)>&& completionHandler)
244{
245 ASSERT(RunLoop::isMain());
246
247 if (!m_isLoaded) {
248 m_pendingCompletionHandlers.append([this, completionHandler = WTFMove(completionHandler)]() mutable {
249 completePendingHandler(WTFMove(completionHandler));
250 });
251 return;
252 }
253
254 completePendingHandler(WTFMove(completionHandler));

257void DeviceIdHashSaltStorage::deleteHashSaltFromDisk(const HashSaltForOrigin& hashSaltForOrigin)
258{
259 m_queue->dispatch([this, protectedThis = makeRef(*this), deviceIdHashSalt = hashSaltForOrigin.deviceIdHashSalt.isolatedCopy()]() mutable {
260 ASSERT(!RunLoop::isMain());
261
262 String fileFullPath = FileSystem::pathByAppendingComponent(m_deviceIdHashSaltStorageDirectory, deviceIdHashSalt.utf8().data());
263 FileSystem::deleteFile(fileFullPath);
264 });
265}

267void DeviceIdHashSaltStorage::deleteDeviceIdHashSaltForOrigins(const Vector<SecurityOriginData>& origins, CompletionHandler<void()>&& completionHandler)
268{
269 ASSERT(RunLoop::isMain());
270
271 m_deviceIdHashSaltForOrigins.removeIf([this, &origins](auto& keyAndValue) {
272 bool needsRemoval = origins.contains(keyAndValue.value->documentOrigin) || origins.contains(keyAndValue.value->parentOrigin);
273 if (m_deviceIdHashSaltStorageDirectory.isEmpty())
274 return needsRemoval;
275 if (needsRemoval)
276 this->deleteHashSaltFromDisk(*keyAndValue.value.get());
277 return needsRemoval;
278 });
279
280 RunLoop::main().dispatch(WTFMove(completionHandler));
281}
282
283void DeviceIdHashSaltStorage::deleteDeviceIdHashSaltOriginsModifiedSince(WallTime time, CompletionHandler<void()>&& completionHandler)
284{
285 ASSERT(RunLoop::isMain());
286
287 m_deviceIdHashSaltForOrigins.removeIf([this, time](auto& keyAndValue) {
288 bool needsRemoval = keyAndValue.value->lastTimeUsed > time;
289 if (m_deviceIdHashSaltStorageDirectory.isEmpty())
290 return needsRemoval;
291 if (needsRemoval)
292 this->deleteHashSaltFromDisk(*keyAndValue.value.get());
293 return needsRemoval;
294 });
295
296 RunLoop::main().dispatch(WTFMove(completionHandler));
297}
