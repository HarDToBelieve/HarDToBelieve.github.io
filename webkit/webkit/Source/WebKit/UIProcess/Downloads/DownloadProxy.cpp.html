51DownloadProxy::DownloadProxy(DownloadProxyMap& downloadProxyMap, WebsiteDataStore& dataStore, API::DownloadClient& client, const ResourceRequest& resourceRequest, const FrameInfoData& frameInfoData, WebPageProxy* originatingPage)
52 : m_downloadProxyMap(downloadProxyMap)
53 , m_dataStore(&dataStore)
54 , m_client(client)
55 , m_downloadID(DownloadID::generate())
56 , m_request(resourceRequest)
57 , m_originatingPage(makeWeakPtr(originatingPage))
58 , m_frameInfo(API::FrameInfo::create(FrameInfoData { frameInfoData }, originatingPage))
59{
60}

62DownloadProxy::~DownloadProxy()
63{
64 if (m_didStartCallback)
65 m_didStartCallback(nullptr);
66}

68static RefPtr<API::Data> createData(const IPC::DataReference& data)
69{
70 if (data.isEmpty())
71 return nullptr;
72 return API::Data::create(data.data(), data.size());
73}

75void DownloadProxy::cancel(CompletionHandler<void(API::Data*)>&& completionHandler)
76{
77 if (m_dataStore) {
78 m_dataStore->networkProcess().sendWithAsyncReply(Messages::NetworkProcess::CancelDownload(m_downloadID), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)] (const IPC::DataReference& resumeData) mutable {
79 m_legacyResumeData = createData(resumeData);
80 completionHandler(m_legacyResumeData.get());
81 m_downloadProxyMap.downloadFinished(*this);
82 });
83 } else
84 completionHandler(nullptr);
85}

103#if PLATFORM(COCOA)
104void DownloadProxy::publishProgress(const URL& URL)
105{
106 if (!m_dataStore)
107 return;
108
109 SandboxExtension::Handle handle;
110 bool createdSandboxExtension = SandboxExtension::createHandle(URL.fileSystemPath(), SandboxExtension::Type::ReadWrite, handle);
111 ASSERT_UNUSED(createdSandboxExtension, createdSandboxExtension);
112 m_dataStore->networkProcess().send(Messages::NetworkProcess::PublishDownloadProgress(m_downloadID, URL, handle), 0);
113}
114#endif // PLATFORM(COCOA)

116void DownloadProxy::didStart(const ResourceRequest& request, const String& suggestedFilename)
117{
118 m_request = request;
119 m_suggestedFilename = suggestedFilename;
120
121 if (m_redirectChain.isEmpty() || m_redirectChain.last() != request.url())
122 m_redirectChain.append(request.url());
123
124 if (m_didStartCallback)
125 m_didStartCallback(this);
126 m_client->legacyDidStart(*this);
127}

129void DownloadProxy::didReceiveAuthenticationChallenge(AuthenticationChallenge&& authenticationChallenge, uint64_t challengeID)
130{
131 if (!m_dataStore)
132 return;
133
134 auto authenticationChallengeProxy = AuthenticationChallengeProxy::create(WTFMove(authenticationChallenge), challengeID, makeRef(*m_dataStore->networkProcess().connection()), nullptr);
135 m_client->didReceiveAuthenticationChallenge(*this, authenticationChallengeProxy.get());
136}

138void DownloadProxy::willSendRequest(ResourceRequest&& proposedRequest, const ResourceResponse& redirectResponse)
139{
140 m_client->willSendRequest(*this, WTFMove(proposedRequest), redirectResponse, [this, protectedThis = makeRef(*this)](ResourceRequest&& newRequest) {
141 m_redirectChain.append(newRequest.url());
142 if (!protectedThis->m_dataStore)
143 return;
144 auto& networkProcessProxy = protectedThis->m_dataStore->networkProcess();
145 networkProcessProxy.send(Messages::NetworkProcess::ContinueWillSendRequest(protectedThis->m_downloadID, newRequest), 0);
146 });
147}

154void DownloadProxy::decideDestinationWithSuggestedFilename(const WebCore::ResourceResponse& response, String&& suggestedFilename, CompletionHandler<void(String, SandboxExtension::Handle, AllowOverwrite)>&& completionHandler)
155{
156 // As per https://html.spec.whatwg.org/#as-a-download (step 2), the filename from the Content-Disposition header
157 // should override the suggested filename from the download attribute.
158 if (response.isAttachmentWithFilename() || (suggestedFilename.isEmpty() && m_suggestedFilename.isEmpty()))
159 suggestedFilename = response.suggestedFilename();
160 else if (!m_suggestedFilename.isEmpty())
161 suggestedFilename = m_suggestedFilename;
162 suggestedFilename = MIMETypeRegistry::appendFileExtensionIfNecessary(suggestedFilename, response.mimeType());
163
164 m_client->decideDestinationWithSuggestedFilename(*this, response, ResourceResponseBase::sanitizeSuggestedFilename(suggestedFilename), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)] (AllowOverwrite allowOverwrite, String destination) mutable {
165 SandboxExtension::Handle sandboxExtensionHandle;
166 if (!destination.isNull())
167 SandboxExtension::createHandle(destination, SandboxExtension::Type::ReadWrite, sandboxExtensionHandle);
168
169 setDestinationFilename(destination);
170 completionHandler(destination, WTFMove(sandboxExtensionHandle), allowOverwrite);
171 });
172}

179void DownloadProxy::didFinish()
180{
181 m_client->didFinish(*this);
182
183 // This can cause the DownloadProxy object to be deleted.
184 m_downloadProxyMap.downloadFinished(*this);
185}

187void DownloadProxy::didFail(const ResourceError& error, const IPC::DataReference& resumeData)
188{
189 m_legacyResumeData = createData(resumeData);
190
191 m_client->didFail(*this, error, m_legacyResumeData.get());
192
193 // This can cause the DownloadProxy object to be deleted.
194 m_downloadProxyMap.downloadFinished(*this);
195}
