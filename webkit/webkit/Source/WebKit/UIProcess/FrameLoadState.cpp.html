29namespace WebKit {
30
31FrameLoadState::~FrameLoadState()
32{
33}
34
35void FrameLoadState::addObserver(Observer& observer)
36{
37 auto result = m_observers.add(observer);
38 ASSERT_UNUSED(result, result.isNewEntry);
39}
40
41void FrameLoadState::removeObserver(Observer& observer)
42{
43 auto result = m_observers.remove(observer);
44 ASSERT_UNUSED(result, result);
45}
46
47void FrameLoadState::didStartProvisionalLoad(const URL& url)
48{
49 ASSERT(m_provisionalURL.isEmpty());
50
51 m_state = State::Provisional;
52 m_provisionalURL = url;
53}
54
55void FrameLoadState::didExplicitOpen(const URL& url)
56{
57 m_url = url;
58 m_provisionalURL = { };
59}
60
61void FrameLoadState::didReceiveServerRedirectForProvisionalLoad(const URL& url)
62{
63 ASSERT(m_state == State::Provisional);
64
65 m_provisionalURL = url;
66}
67
68void FrameLoadState::didFailProvisionalLoad()
69{
70 ASSERT(m_state == State::Provisional);
71
72 m_state = State::Finished;
73 m_provisionalURL = { };
74 m_unreachableURL = m_lastUnreachableURL;
75}
76
77void FrameLoadState::didCommitLoad()
78{
79 ASSERT(m_state == State::Provisional);
80
81 m_state = State::Committed;
82 m_url = m_provisionalURL;
83 m_provisionalURL = { };
84}
85
86void FrameLoadState::didFinishLoad()
87{
88 ASSERT(m_state == State::Committed);
89 ASSERT(m_provisionalURL.isEmpty());
90
91 m_state = State::Finished;
92
93 Vector<Observer*> observersCopy;
94 for (auto& observer : m_observers)
95 observersCopy.append(&observer);
96 for (auto* observer : observersCopy)
97 observer->didFinishLoad();
98}
99
100void FrameLoadState::didFailLoad()
101{
102 ASSERT(m_state == State::Committed);
103 ASSERT(m_provisionalURL.isEmpty());
104
105 m_state = State::Finished;
106}
107
108void FrameLoadState::didSameDocumentNotification(const URL& url)
109{
110 m_url = url;
111}
112
113void FrameLoadState::setUnreachableURL(const URL& unreachableURL)
114{
115 m_lastUnreachableURL = m_unreachableURL;
116 m_unreachableURL = unreachableURL;
117}
