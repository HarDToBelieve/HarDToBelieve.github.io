26#pragma once
27
28#include <wtf/URL.h>
29#include <wtf/WeakHashSet.h>
30
31namespace WebKit {
32
33class FrameLoadState {
34public:
35 ~FrameLoadState();
36
37 enum class State {
38 Provisional,
39 Committed,
40 Finished
41 };
42
43 class Observer : public CanMakeWeakPtr<Observer> {
44 public:
45 virtual ~Observer() = default;
46
47 virtual void didFinishLoad() = 0;
48 };
49
50 void addObserver(Observer&);
51 void removeObserver(Observer&);
52
53 void didStartProvisionalLoad(const URL&);
54 void didExplicitOpen(const URL&);
55 void didReceiveServerRedirectForProvisionalLoad(const URL&);
56 void didFailProvisionalLoad();
57
58 void didCommitLoad();
59 void didFinishLoad();
60 void didFailLoad();
61
62 void didSameDocumentNotification(const URL&);
63
64 State state() const { return m_state; }
65 const URL& url() const { return m_url; }
66 void setURL(const URL& url) { m_url = url; }
67 const URL& provisionalURL() const { return m_provisionalURL; }
68
69 void setUnreachableURL(const URL&);
70 const URL& unreachableURL() const { return m_unreachableURL; }
71
72private:
73 State m_state { State::Finished };
74 URL m_url;
75 URL m_provisionalURL;
76 URL m_unreachableURL;
77 URL m_lastUnreachableURL;
78 WeakHashSet<Observer> m_observers;
79};
80
81} // namespace WebKit
