42class UIGamepadProvider final : public WebCore::GamepadProviderClient {

46 void processPoolStartedUsingGamepads(WebProcessPool&);
47 void processPoolStoppedUsingGamepads(WebProcessPool&);
48
49 void viewBecameActive(WebPageProxy&);

51

64 UIGamepadProvider();
65 ~UIGamepadProvider() final;

67 void startMonitoringGamepads();
68 void stopMonitoringGamepads();

70 void platformSetDefaultGamepadProvider();

75 void platformGamepadConnected(WebCore::PlatformGamepad&, WebCore::EventMakesGamepadsVisible) final;

77 void platformGamepadInputActivity(WebCore::EventMakesGamepadsVisible) final;
78
79 void scheduleGamepadStateSync();
80 void gamepadSyncTimerFired();
81
82 HashSet<WebProcessPool*> m_processPoolsUsingGamepads;

84 Vector<std::unique_ptr<UIGamepad>> m_gamepads;

86 RunLoop::Timer<UIGamepadProvider> m_gamepadSyncTimer;
