36GeolocationPermissionRequestManagerProxy::GeolocationPermissionRequestManagerProxy(WebPageProxy& page)
37 : m_page(page)
38{
39}

41void GeolocationPermissionRequestManagerProxy::invalidateRequests()
42{
43 for (auto& request : m_pendingRequests.values())
44 request->invalidate();
45
46 m_pendingRequests.clear();
47}
48
49Ref<GeolocationPermissionRequestProxy> GeolocationPermissionRequestManagerProxy::createRequest(GeolocationIdentifier geolocationID)
50{
51 auto request = GeolocationPermissionRequestProxy::create(this, geolocationID);
52 m_pendingRequests.add(geolocationID, request.ptr());
53 return request;
54}
55
56void GeolocationPermissionRequestManagerProxy::didReceiveGeolocationPermissionDecision(GeolocationIdentifier geolocationID, bool allowed)
57{
58 if (!m_page.hasRunningProcess())
59 return;
60
61 auto it = m_pendingRequests.find(geolocationID);
62 if (it == m_pendingRequests.end())
63 return;
64
65#if ENABLE(GEOLOCATION)
66 String authorizationToken = allowed ? createCanonicalUUIDString() : String();
67 if (!authorizationToken.isNull())
68 m_validAuthorizationTokens.add(authorizationToken);
69 m_page.send(Messages::WebPage::DidReceiveGeolocationPermissionDecision(geolocationID, authorizationToken));
70#else
71 UNUSED_PARAM(allowed);
72#endif
73
74 m_pendingRequests.remove(it);
75}
76
77bool GeolocationPermissionRequestManagerProxy::isValidAuthorizationToken(const String& authorizationToken) const
78{
79 return !authorizationToken.isNull() && m_validAuthorizationTokens.contains(authorizationToken);
80}
81
82void GeolocationPermissionRequestManagerProxy::revokeAuthorizationToken(const String& authorizationToken)
83{
84 ASSERT(isValidAuthorizationToken(authorizationToken));
85 if (!isValidAuthorizationToken(authorizationToken))
86 return;
87 m_validAuthorizationTokens.remove(authorizationToken);
88}
