33namespace WebKit {
34using namespace WebCore;
35
36static const Seconds samplingInterval { 10_min };
37
38HighPerformanceGraphicsUsageSampler::HighPerformanceGraphicsUsageSampler(WebProcessPool& webProcessPool)
39 : m_webProcessPool(webProcessPool)
40 , m_timer(RunLoop::main(), this, &HighPerformanceGraphicsUsageSampler::timerFired)
41{
42 m_timer.startRepeating(samplingInterval);
43}
44
45void HighPerformanceGraphicsUsageSampler::timerFired()
46{
47 bool isUsingHighPerformanceWebGL = false;
48 bool isUsingHighPerformanceWebGLInVisibleView = false;
49
50 WebPageProxy* firstPage = nullptr;
51 for (auto& webProcess : m_webProcessPool.processes()) {
52 for (auto& page : webProcess->pages()) {
53 if (!firstPage)
54 firstPage = page;
55
56 if (page->isUsingHighPerformanceWebGL()) {
57 isUsingHighPerformanceWebGL = true;
58 if (page->isViewVisible()) {
59 isUsingHighPerformanceWebGLInVisibleView = true;
60 break;
61 }
62 }
63 }
64 }
65
66 if (!firstPage)
67 return;
68
69 String state = DiagnosticLoggingKeys::inactiveKey();
70 if (isUsingHighPerformanceWebGLInVisibleView)
71 state = DiagnosticLoggingKeys::activeInForegroundTabKey();
72 else if (isUsingHighPerformanceWebGL)
73 state = DiagnosticLoggingKeys::activeInBackgroundTabOnlyKey();
74
75 firstPage->logDiagnosticMessage(DiagnosticLoggingKeys::webGLStateKey(), state, ShouldSample::No);
76}
