43InspectorBrowserAgent::InspectorBrowserAgent(WebPageAgentContext& context)
44 : InspectorAgentBase("Browser"_s, context)
45 , m_frontendDispatcher(makeUnique<Inspector::BrowserFrontendDispatcher>(context.frontendRouter))
46 , m_backendDispatcher(Inspector::BrowserBackendDispatcher::create(context.backendDispatcher, this))
47 , m_inspectedPage(context.inspectedPage)
48{
49}

51InspectorBrowserAgent::~InspectorBrowserAgent() = default;
52
53bool InspectorBrowserAgent::enabled() const
54{
55 return m_inspectedPage.inspectorController().enabledBrowserAgent() == this;
56}
57
58void InspectorBrowserAgent::didCreateFrontendAndBackend(Inspector::FrontendRouter*, Inspector::BackendDispatcher*)
59{
60}
61
62void InspectorBrowserAgent::willDestroyFrontendAndBackend(Inspector::DisconnectReason)
63{
64 disable();
65}
66
67Inspector::Protocol::ErrorStringOr<void> InspectorBrowserAgent::enable()
68{
69 if (enabled())
70 return makeUnexpected("Browser domain already enabled"_s);
71
72 m_inspectedPage.inspectorController().setEnabledBrowserAgent(this);
73
74 return { };
75}
76
77Inspector::Protocol::ErrorStringOr<void> InspectorBrowserAgent::disable()
78{
79 if (!enabled())
80 return makeUnexpected("Browser domain already disabled"_s);
81
82 m_inspectedPage.inspectorController().setEnabledBrowserAgent(nullptr);
83
84 return { };
85}
86
87void InspectorBrowserAgent::extensionsEnabled(HashMap<String, String>&& extensionIDToName)
88{
89 ASSERT(enabled());
90
91 auto extensionsPayload = JSON::ArrayOf<Inspector::Protocol::Browser::Extension>::create();
92 for (auto& [id, name] : extensionIDToName) {
93 auto extensionPayload = Inspector::Protocol::Browser::Extension::create()
94 .setExtensionId(id)
95 .setName(name)
96 .release();
97 extensionsPayload->addItem(WTFMove(extensionPayload));
98 }
99 m_frontendDispatcher->extensionsEnabled(WTFMove(extensionsPayload));
100}
101
102void InspectorBrowserAgent::extensionsDisabled(HashSet<String>&& extensionIDs)
103{
104 ASSERT(enabled());
105
106 auto extensionIdsPayload = JSON::ArrayOf<String>::create();
107 for (auto& extensionId : extensionIDs)
108 extensionIdsPayload->addItem(extensionId);
109 m_frontendDispatcher->extensionsDisabled(WTFMove(extensionIdsPayload));
110}
