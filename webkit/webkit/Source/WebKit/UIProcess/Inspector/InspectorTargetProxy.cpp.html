40std::unique_ptr<InspectorTargetProxy> InspectorTargetProxy::create(WebPageProxy& page, const String& targetId, Inspector::InspectorTargetType type)
41{
42 return makeUnique<InspectorTargetProxy>(page, targetId, type);
43}
44
45std::unique_ptr<InspectorTargetProxy> InspectorTargetProxy::create(ProvisionalPageProxy& provisionalPage, const String& targetId, Inspector::InspectorTargetType type)
46{
47 auto target = InspectorTargetProxy::create(provisionalPage.page(), targetId, type);
48 target->m_provisionalPage = makeWeakPtr(provisionalPage);
49 return target;
50}
51
52InspectorTargetProxy::InspectorTargetProxy(WebPageProxy& page, const String& targetId, Inspector::InspectorTargetType type)
53 : m_page(page)
54 , m_identifier(targetId)
55 , m_type(type)
56{
57}
58
59void InspectorTargetProxy::connect(Inspector::FrontendChannel::ConnectionType connectionType)
60{
61 if (m_provisionalPage) {
62 m_provisionalPage->send(Messages::WebPage::ConnectInspector(identifier(), connectionType));
63 return;
64 }
65
66 if (m_page.hasRunningProcess())
67 m_page.send(Messages::WebPage::ConnectInspector(identifier(), connectionType));
68}
69
70void InspectorTargetProxy::disconnect()
71{
72 if (isPaused())
73 resume();
74
75 if (m_provisionalPage) {
76 m_provisionalPage->send(Messages::WebPage::DisconnectInspector(identifier()));
77 return;
78 }
79
80 if (m_page.hasRunningProcess())
81 m_page.send(Messages::WebPage::DisconnectInspector(identifier()));
82}
83
84void InspectorTargetProxy::sendMessageToTargetBackend(const String& message)
85{
86 if (m_provisionalPage) {
87 m_provisionalPage->send(Messages::WebPage::SendMessageToTargetBackend(identifier(), message));
88 return;
89 }
90
91 if (m_page.hasRunningProcess())
92 m_page.send(Messages::WebPage::SendMessageToTargetBackend(identifier(), message));
93}
94
95void InspectorTargetProxy::didCommitProvisionalTarget()
96{
97 m_provisionalPage = nullptr;
98}
99
100bool InspectorTargetProxy::isProvisional() const
101{
102 return !!m_provisionalPage;
103}
