26#pragma once
27
28#include <JavaScriptCore/InspectorTarget.h>
29#include <wtf/Noncopyable.h>
30#include <wtf/WeakPtr.h>
31
32namespace WebKit {
33
34class ProvisionalPageProxy;
35class WebPageProxy;
36
37// NOTE: This UIProcess side InspectorTarget doesn't care about the frontend channel, since
38// any target -> frontend messages will be routed to the WebPageProxy with a targetId.
39
40class InspectorTargetProxy final : public Inspector::InspectorTarget {
41 WTF_MAKE_FAST_ALLOCATED;
42 WTF_MAKE_NONCOPYABLE(InspectorTargetProxy);
43public:
44 static std::unique_ptr<InspectorTargetProxy> create(WebPageProxy&, const String& targetId, Inspector::InspectorTargetType);
45 static std::unique_ptr<InspectorTargetProxy> create(ProvisionalPageProxy&, const String& targetId, Inspector::InspectorTargetType);
46 InspectorTargetProxy(WebPageProxy&, const String& targetId, Inspector::InspectorTargetType);
47 ~InspectorTargetProxy() = default;
48
49 Inspector::InspectorTargetType type() const final { return m_type; }
50 String identifier() const final { return m_identifier; }
51
52 void didCommitProvisionalTarget();
53 bool isProvisional() const override;
54
55 void connect(Inspector::FrontendChannel::ConnectionType) override;
56 void disconnect() override;
57 void sendMessageToTargetBackend(const String&) override;
58
59private:
60 WebPageProxy& m_page;
61 String m_identifier;
62 Inspector::InspectorTargetType m_type;
63 WeakPtr<ProvisionalPageProxy> m_provisionalPage;
64};
65
66} // namespace WebKit
