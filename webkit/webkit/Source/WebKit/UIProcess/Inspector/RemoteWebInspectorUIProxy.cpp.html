70void RemoteWebInspectorUIProxy::load(Ref<API::DebuggableInfo>&& debuggableInfo, const String& backendCommandsURL)
71{
72 m_debuggableInfo = WTFMove(debuggableInfo);
73 m_backendCommandsURL = backendCommandsURL;
74
75 createFrontendPageAndWindow();
76
77 m_inspectorPage->send(Messages::RemoteWebInspectorUI::Initialize(m_debuggableInfo->debuggableInfoData(), backendCommandsURL));
78 m_inspectorPage->loadRequest(URL(URL(), WebInspectorUIProxy::inspectorPageURL()));
79}

109void RemoteWebInspectorUIProxy::frontendDidClose()
110{
111 Ref<RemoteWebInspectorUIProxy> protect(*this);
112
113 if (m_client)
114 m_client->closeFromFrontend();
115
116 closeFrontendPageAndWindow();
117}

119void RemoteWebInspectorUIProxy::reopen()
120{
121 ASSERT(!m_backendCommandsURL.isEmpty());
122
123 closeFrontendPageAndWindow();
124 load(m_debuggableInfo.copyRef(), m_backendCommandsURL);
125}

178void RemoteWebInspectorUIProxy::createFrontendPageAndWindow()
179{
180 if (m_inspectorPage)
181 return;
182
183 m_inspectorPage = platformCreateFrontendPageAndWindow();
184
185 trackInspectorPage(m_inspectorPage, nullptr);
186
187 m_inspectorPage->process().addMessageReceiver(Messages::RemoteWebInspectorUIProxy::messageReceiverName(), m_inspectorPage->webPageID(), *this);
188
189#if ENABLE(INSPECTOR_EXTENSIONS)
190 m_extensionController = WebInspectorUIExtensionControllerProxy::create(*m_inspectorPage);
191#endif
192}
193
194void RemoteWebInspectorUIProxy::closeFrontendPageAndWindow()
195{
196 if (!m_inspectorPage)
197 return;
198
199 m_inspectorPage->process().removeMessageReceiver(Messages::RemoteWebInspectorUIProxy::messageReceiverName(), m_inspectorPage->webPageID());
200
201 untrackInspectorPage(m_inspectorPage);
202
203#if ENABLE(INSPECTOR_EXTENSIONS)
204 // This extension controller may be kept alive by the IPC dispatcher beyond the point
205 // when m_inspectorPage is cleared below. Notify the controller so it can clean up before then.
206 m_extensionController->inspectorFrontendWillClose();
207 m_extensionController = nullptr;
208#endif
209
210 m_inspectorPage = nullptr;
211
212 platformCloseFrontendPageAndWindow();
213}
