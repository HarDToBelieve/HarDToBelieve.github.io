39WebInspectorUIExtensionControllerProxy::WebInspectorUIExtensionControllerProxy(WebPageProxy& inspectorPage)
40 : m_inspectorPage(makeWeakPtr(inspectorPage))
41{
42 m_inspectorPage->process().addMessageReceiver(Messages::WebInspectorUIExtensionControllerProxy::messageReceiverName(), m_inspectorPage->webPageID(), *this);
43}
44
45WebInspectorUIExtensionControllerProxy::~WebInspectorUIExtensionControllerProxy()
46{
47 auto callbacks = std::exchange(m_frontendLoadedCallbackQueue, { });
48 for (auto& callback : callbacks)
49 callback();
50
51 // At this point, we should have already been notified that the frontend has closed.
52 ASSERT(!m_inspectorPage);
53}
54
55Ref<WebInspectorUIExtensionControllerProxy> WebInspectorUIExtensionControllerProxy::create(WebPageProxy& inspectorPage)
56{
57 return adoptRef(*new WebInspectorUIExtensionControllerProxy(inspectorPage));
58}
59
60void WebInspectorUIExtensionControllerProxy::whenFrontendHasLoaded(Function<void()>&& callback)
61{
62 if (m_frontendLoaded && m_inspectorPage) {
63 callback();
64 return;
65 }
66
67 m_frontendLoadedCallbackQueue.append(WTFMove(callback));
68}
69
70void WebInspectorUIExtensionControllerProxy::inspectorFrontendLoaded()
71{
72 ASSERT(m_inspectorPage);
73
74 m_frontendLoaded = true;
75
76 auto callbacks = std::exchange(m_frontendLoadedCallbackQueue, { });
77 for (auto& callback : callbacks)
78 callback();
79}
80
81void WebInspectorUIExtensionControllerProxy::inspectorFrontendWillClose()
82{
83 if (!m_inspectorPage)
84 return;
85
86 m_inspectorPage->process().removeMessageReceiver(Messages::WebInspectorUIExtensionControllerProxy::messageReceiverName(), m_inspectorPage->webPageID());
87 m_inspectorPage = nullptr;
88}

92void WebInspectorUIExtensionControllerProxy::registerExtension(const Inspector::ExtensionID& extensionID, const String& displayName, WTF::CompletionHandler<void(Expected<RefPtr<API::InspectorExtension>, Inspector::ExtensionError>)>&& completionHandler)
93{
94 whenFrontendHasLoaded([weakThis = makeWeakPtr(this), extensionID, displayName, completionHandler = WTFMove(completionHandler)] () mutable {
95 if (!weakThis || !weakThis->m_inspectorPage) {
96 completionHandler(makeUnexpected(Inspector::ExtensionError::InvalidRequest));
97 return;
98 }
99
100 weakThis->m_inspectorPage->sendWithAsyncReply(Messages::WebInspectorUIExtensionController::RegisterExtension { extensionID, displayName }, [strongThis = makeRef(*weakThis.get()), extensionID, completionHandler = WTFMove(completionHandler)](Expected<bool, Inspector::ExtensionError> result) mutable {
101 if (!result) {
102 completionHandler(makeUnexpected(Inspector::ExtensionError::RegistrationFailed));
103 return;
104 }
105
106 RefPtr<API::InspectorExtension> extensionAPIObject = API::InspectorExtension::create(extensionID, strongThis.get());
107 strongThis->m_extensionAPIObjectMap.set(extensionID, extensionAPIObject.copyRef());
108
109 completionHandler(WTFMove(extensionAPIObject));
110 });
111 });
112}

114void WebInspectorUIExtensionControllerProxy::unregisterExtension(const Inspector::ExtensionID& extensionID, WTF::CompletionHandler<void(Expected<bool, Inspector::ExtensionError>)>&& completionHandler)
115{
116 whenFrontendHasLoaded([weakThis = makeWeakPtr(this), extensionID, completionHandler = WTFMove(completionHandler)] () mutable {
117 if (!weakThis || !weakThis->m_inspectorPage) {
118 completionHandler(makeUnexpected(Inspector::ExtensionError::InvalidRequest));
119 return;
120 }
121
122 weakThis->m_inspectorPage->sendWithAsyncReply(Messages::WebInspectorUIExtensionController::UnregisterExtension { extensionID }, [strongThis = makeRef(*weakThis.get()), extensionID, completionHandler = WTFMove(completionHandler)](Expected<bool, Inspector::ExtensionError> result) mutable {
123 if (!result) {
124 completionHandler(makeUnexpected(Inspector::ExtensionError::InvalidRequest));

127
128 strongThis->m_extensionAPIObjectMap.take(extensionID);

130 completionHandler(WTFMove(result));
131 });
132 });

147void WebInspectorUIExtensionControllerProxy::evaluateScriptForExtension(const Inspector::ExtensionID& extensionID, const String& scriptSource, const std::optional<WTF::URL>& frameURL, const std::optional<WTF::URL>& contextSecurityOrigin, const std::optional<bool>& useContentScriptContext, WTF::CompletionHandler<void(Inspector::ExtensionEvaluationResult)>&& completionHandler)
148{
149 whenFrontendHasLoaded([weakThis = makeWeakPtr(this), extensionID, scriptSource, frameURL, contextSecurityOrigin, useContentScriptContext, completionHandler = WTFMove(completionHandler)] () mutable {
150 if (!weakThis || !weakThis->m_inspectorPage) {
151 completionHandler(makeUnexpected(Inspector::ExtensionError::ContextDestroyed));
152 return;
153 }
154
155 weakThis->m_inspectorPage->sendWithAsyncReply(Messages::WebInspectorUIExtensionController::EvaluateScriptForExtension {extensionID, scriptSource, frameURL, contextSecurityOrigin, useContentScriptContext}, [completionHandler = WTFMove(completionHandler)](const IPC::DataReference& dataReference, const std::optional<WebCore::ExceptionDetails>& details, const std::optional<Inspector::ExtensionError> error) mutable {
156 if (error) {
157 completionHandler(makeUnexpected(error.value()));
158 return;
159 }
160
161 if (details) {
162 Expected<RefPtr<API::SerializedScriptValue>, WebCore::ExceptionDetails> returnedValue = makeUnexpected(details.value());
163 return completionHandler({ returnedValue });
164 }
165
166 completionHandler({ { API::SerializedScriptValue::adopt(Vector { dataReference.data(), dataReference.size() }).ptr() } });
167 });
168 });
169}

192void WebInspectorUIExtensionControllerProxy::didShowExtensionTab(const Inspector::ExtensionID& extensionID, const Inspector::ExtensionTabID& extensionTabID)
193{
194 auto it = m_extensionAPIObjectMap.find(extensionID);
195 if (it == m_extensionAPIObjectMap.end())
196 return;
197
198 RefPtr<API::InspectorExtension> extension = it->value;
199 auto extensionClient = extension->client();
200 if (!extensionClient)
201 return;
202
203 extensionClient->didShowExtensionTab(extensionTabID);
204}
