30#include "InspectorExtensionTypes.h"

32#include <wtf/Expected.h>
33#include <wtf/Forward.h>
34#include <wtf/URL.h>
35#include <wtf/WeakPtr.h>
36
37namespace API {
38class InspectorExtension;
39}
40
41namespace WebKit {
42
43class WebPageProxy;
44
45class WebInspectorUIExtensionControllerProxy final
46 : public RefCounted<WebInspectorUIExtensionControllerProxy>
47 , public IPC::MessageReceiver {
48 WTF_MAKE_FAST_ALLOCATED;
49 WTF_MAKE_NONCOPYABLE(WebInspectorUIExtensionControllerProxy);
50public:
51 static Ref<WebInspectorUIExtensionControllerProxy> create(WebPageProxy& inspectorPage);
52 virtual ~WebInspectorUIExtensionControllerProxy();
53
54 // Implemented in generated WebInspectorUIExtensionControllerProxyMessageReceiver.cpp
55 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
56
57 // API.
58 void registerExtension(const Inspector::ExtensionID&, const String& displayName, WTF::CompletionHandler<void(Expected<RefPtr<API::InspectorExtension>, Inspector::ExtensionError>)>&&);
59 void unregisterExtension(const Inspector::ExtensionID&, WTF::CompletionHandler<void(Expected<bool, Inspector::ExtensionError>)>&&);
60 void createTabForExtension(const Inspector::ExtensionID&, const String& tabName, const URL& tabIconURL, const URL& sourceURL, WTF::CompletionHandler<void(Expected<Inspector::ExtensionTabID, Inspector::ExtensionError>)>&&);
61 void evaluateScriptForExtension(const Inspector::ExtensionID&, const String& scriptSource, const std::optional<URL>& frameURL, const std::optional<URL>& contextSecurityOrigin, const std::optional<bool>& useContentScriptContext, WTF::CompletionHandler<void(Inspector::ExtensionEvaluationResult)>&&);
62 void reloadForExtension(const Inspector::ExtensionID&, const std::optional<bool>& ignoreCache, const std::optional<String>& userAgent, const std::optional<String>& injectedScript, WTF::CompletionHandler<void(Inspector::ExtensionEvaluationResult)>&&);
63
64 // WebInspectorUIExtensionControllerProxy IPC messages.
65 void didShowExtensionTab(const Inspector::ExtensionID&, const Inspector::ExtensionTabID&);
66 void didHideExtensionTab(const Inspector::ExtensionID&, const Inspector::ExtensionTabID&);
67
68 // Notifications.
69 void inspectorFrontendLoaded();
70 void inspectorFrontendWillClose();
71
72private:
73 explicit WebInspectorUIExtensionControllerProxy(WebPageProxy& inspectorPage);
74
75 void whenFrontendHasLoaded(Function<void()>&&);
76
77 WeakPtr<WebPageProxy> m_inspectorPage;
78 HashMap<Inspector::ExtensionID, RefPtr<API::InspectorExtension>> m_extensionAPIObjectMap;
79
80 // Used to queue actions such as registering extensions that happen early on.
81 // There's no point sending these before the frontend is fully loaded.
82 Vector<Function<void()>> m_frontendLoadedCallbackQueue;
83
84 bool m_frontendLoaded { false };
85};
86
87} // namespace WebKit

89#endif // ENABLE(INSPECTOR_EXTENSIONS)
