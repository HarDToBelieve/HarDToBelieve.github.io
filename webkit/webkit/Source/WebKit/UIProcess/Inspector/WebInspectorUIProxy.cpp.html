127void WebInspectorUIProxy::connect()
128{
129 if (!m_inspectedPage)
130 return;
131
132 if (!m_inspectedPage->preferences().developerExtrasEnabled())
133 return;
134
135 if (m_showMessageSent)
136 return;
137
138 m_showMessageSent = true;
139 m_ignoreFirstBringToFront = true;
140
141 createFrontendPage();
142
143 m_inspectedPage->send(Messages::WebInspectorInterruptDispatcher::NotifyNeedDebuggerBreak(), 0);
144 m_inspectedPage->send(Messages::WebInspector::Show());
145}

148{

158
159 // Don't ignore the first bringToFront so it opens the Inspector.
160 m_ignoreFirstBringToFront = false;
161}

287void WebInspectorUIProxy::attach(AttachmentSide side)
288{
289 ASSERT(m_inspectorPage);
290 if (!m_inspectedPage || !m_inspectorPage || (!m_isAttached && !platformCanAttach(m_canAttach)))
291 return;
292
293 m_isAttached = true;
294 m_attachmentSide = side;
295
296 inspectorPagePreferences().setInspectorAttachmentSide(static_cast<uint32_t>(side));
297
298 if (m_isVisible)
299 inspectorPagePreferences().setInspectorStartsAttached(true);
300
301 m_inspectedPage->send(Messages::WebInspector::SetAttached(true));
302
303 switch (m_attachmentSide) {
304 case AttachmentSide::Bottom:
305 m_inspectorPage->send(Messages::WebInspectorUI::AttachedBottom());
306 break;
307
308 case AttachmentSide::Right:
309 m_inspectorPage->send(Messages::WebInspectorUI::AttachedRight());
310 break;
311
312 case AttachmentSide::Left:
313 m_inspectorPage->send(Messages::WebInspectorUI::AttachedLeft());
314 break;
315 }
316
317 platformAttach();
318}

320void WebInspectorUIProxy::detach()
321{
322 if (!m_inspectedPage || !m_inspectorPage || (!m_isAttached && m_isVisible))
323 return;
324
325 m_isAttached = false;
326
327 if (m_isVisible)
328 inspectorPagePreferences().setInspectorStartsAttached(false);
329
330 m_inspectedPage->send(Messages::WebInspector::SetAttached(false));
331 m_inspectorPage->send(Messages::WebInspectorUI::Detached());
332
333 platformDetach();
334}

391bool WebInspectorUIProxy::isMainOrTestInspectorPage(const URL& url)
392{
393 // Use URL so we can compare the paths and protocols.
394 URL mainPageURL(URL(), WebInspectorUIProxy::inspectorPageURL());
395 if (url.protocol() == mainPageURL.protocol() && decodeURLEscapeSequences(url.path()) == decodeURLEscapeSequences(mainPageURL.path()))
396 return true;
397
398 // We might not have a Test URL in Production builds.
399 String testPageURLString = WebInspectorUIProxy::inspectorTestPageURL();
400 if (testPageURLString.isNull())
401 return false;
402
403 URL testPageURL(URL(), testPageURLString);
404 return url.protocol() == testPageURL.protocol() && decodeURLEscapeSequences(url.path()) == decodeURLEscapeSequences(testPageURL.path());
405}

407void WebInspectorUIProxy::createFrontendPage()
408{
409 if (m_inspectorPage)
410 return;
411
412 m_inspectorPage = platformCreateFrontendPage();
413 ASSERT(m_inspectorPage);
414 if (!m_inspectorPage)
415 return;
416
417 trackInspectorPage(m_inspectorPage, m_inspectedPage);
418
419 // Make sure the inspected page has a running WebProcess so we can inspect it.
420 m_inspectedPage->launchInitialProcessIfNecessary();
421
422 m_inspectorPage->process().addMessageReceiver(Messages::WebInspectorUIProxy::messageReceiverName(), m_inspectedPage->identifier(), *this);
423
424#if ENABLE(INSPECTOR_EXTENSIONS)
425 m_extensionController = WebInspectorUIExtensionControllerProxy::create(*m_inspectorPage);
426#endif
427}

429void WebInspectorUIProxy::openLocalInspectorFrontend(bool canAttach, bool underTest)
430{
431 if (!m_inspectedPage)
432 return;
433
434 if (!m_inspectedPage->preferences().developerExtrasEnabled())
435 return;
436
437 if (m_inspectedPage->inspectorController().hasLocalFrontend()) {
438 show();
439 return;
440 }
441
442 m_underTest = underTest;
443 createFrontendPage();
444
445 ASSERT(m_inspectorPage);
446 if (!m_inspectorPage)
447 return;
448
449 m_inspectorPage->send(Messages::WebInspectorUI::EstablishConnection(m_inspectedPage->identifier(), infoForLocalDebuggable(), m_underTest, inspectionLevel()));
450
451 ASSERT(!m_isActiveFrontend);
452 m_isActiveFrontend = true;
453 m_inspectedPage->inspectorController().connectFrontend(*this);
454
455 if (!m_underTest) {
456 m_canAttach = platformCanAttach(canAttach);
457 m_isAttached = shouldOpenAttached();
458 m_attachmentSide = static_cast<AttachmentSide>(inspectorPagePreferences().inspectorAttachmentSide());
459
460 m_inspectedPage->send(Messages::WebInspector::SetAttached(m_isAttached));
461
462 if (m_isAttached) {
463 switch (m_attachmentSide) {
464 case AttachmentSide::Bottom:
465 m_inspectorPage->send(Messages::WebInspectorUI::AttachedBottom());
466 break;
467
468 case AttachmentSide::Right:
469 m_inspectorPage->send(Messages::WebInspectorUI::AttachedRight());
470 break;
471
472 case AttachmentSide::Left:
473 m_inspectorPage->send(Messages::WebInspectorUI::AttachedLeft());
474 break;
475 }
476 } else
477 m_inspectorPage->send(Messages::WebInspectorUI::Detached());
478
479 m_inspectorPage->send(Messages::WebInspectorUI::SetDockingUnavailable(!m_canAttach));
480 }
481
482 // Notify WebKit client when a local inspector attaches so that it may install delegates prior to the _WKInspector loading its frontend.
483 m_inspectedPage->uiClient().didAttachLocalInspector(*m_inspectedPage, *this);
484
485 // Bail out if the client closed the inspector from the delegate method.
486 if (!m_inspectorPage)
487 return;
488
489 m_inspectorPage->loadRequest(URL(URL(), m_underTest ? WebInspectorUIProxy::inspectorTestPageURL() : WebInspectorUIProxy::inspectorPageURL()));
490}

492void WebInspectorUIProxy::open()
493{
494 if (m_underTest)
495 return;
496
497 if (!m_inspectorPage)
498 return;
499
500#if PLATFORM(GTK)
501 SetForScope<bool> isOpening(m_isOpening, true);
502#endif
503
504 m_isVisible = true;
505 m_inspectorPage->send(Messages::WebInspectorUI::SetIsVisible(m_isVisible));
506
507 if (m_isAttached)
508 platformAttach();
509 else
510 platformCreateFrontendWindow();
511
512 platformBringToFront();
513}

520void WebInspectorUIProxy::closeFrontendPageAndWindow()
521{
522 if (!m_inspectorPage)
523 return;
524
525 // Guard against calls to close() made by the client while already closing.
526 if (m_closing)
527 return;
528
529 SetForScope<bool> reentrancyProtector(m_closing, true);
530
531 // Notify WebKit client when a local inspector closes so it can clear _WKInspectorDelegate and perform other cleanup.
532 m_inspectedPage->uiClient().willCloseLocalInspector(*m_inspectedPage, *this);
533
534 m_isVisible = false;
535 m_isProfilingPage = false;
536 m_showMessageSent = false;
537 m_ignoreFirstBringToFront = false;
538
539 untrackInspectorPage(m_inspectorPage);
540
541 m_inspectorPage->send(Messages::WebInspectorUI::SetIsVisible(m_isVisible));
542 m_inspectorPage->process().removeMessageReceiver(Messages::WebInspectorUIProxy::messageReceiverName(), m_inspectedPage->identifier());
543
544 if (m_isActiveFrontend) {
545 m_isActiveFrontend = false;
546 m_inspectedPage->inspectorController().disconnectFrontend(*this);
547 }
548
549 if (m_isAttached)
550 platformDetach();
551
552#if ENABLE(INSPECTOR_EXTENSIONS)
553 // This extension controller may be kept alive by the IPC dispatcher beyond the point
554 // when m_inspectorPage is cleared below. Notify the controller so it can clean up before then.
555 m_extensionController->inspectorFrontendWillClose();
556 m_extensionController = nullptr;
557#endif
558
559 // Null out m_inspectorPage after platformDetach(), so the views can be cleaned up correctly.
560 m_inspectorPage = nullptr;
561
562 m_isAttached = false;
563 m_canAttach = false;
564 m_underTest = false;
565
566 platformCloseFrontendPageAndWindow();
567}

594void WebInspectorUIProxy::bringToFront()
595{
596 // WebCore::InspectorFrontendClientLocal tells us to do this on load. We want to
597 // ignore it once if we only wanted to connect. This allows the Inspector to later
598 // request to be brought to the front when a breakpoint is hit or some other action.
599 if (m_ignoreFirstBringToFront) {
600 m_ignoreFirstBringToFront = false;
601 return;
602 }
603
604 if (m_isVisible)
605 platformBringToFront();
606 else
607 open();
608}

615void WebInspectorUIProxy::attachAvailabilityChanged(bool available)
616{
617 bool previousCanAttach = m_canAttach;
618
619 m_canAttach = m_isAttached || platformCanAttach(available);
620
621 if (previousCanAttach == m_canAttach)
622 return;
623
624 if (m_inspectorPage && !m_underTest)
625 m_inspectorPage->send(Messages::WebInspectorUI::SetDockingUnavailable(!m_canAttach));
626
627 platformAttachAvailabilityChanged(m_canAttach);
628}

673void WebInspectorUIProxy::setDeveloperPreferenceOverride(WebCore::InspectorClient::DeveloperPreference developerPreference, std::optional<bool> overrideValue)
674{
675 switch (developerPreference) {
676 case InspectorClient::DeveloperPreference::PrivateClickMeasurementDebugModeEnabled:
677 if (m_inspectedPage)
678 m_inspectedPage->websiteDataStore().setPrivateClickMeasurementDebugMode(overrideValue && overrideValue.value());
679 return;
680
681 case InspectorClient::DeveloperPreference::ITPDebugModeEnabled:
682 if (m_inspectedPage)
683 m_inspectedPage->websiteDataStore().setResourceLoadStatisticsDebugMode(overrideValue && overrideValue.value());
684 return;
685
686 case InspectorClient::DeveloperPreference::MockCaptureDevicesEnabled:
687#if ENABLE(MEDIA_STREAM)
688 if (m_inspectedPage)
689 m_inspectedPage->setMockCaptureDevicesEnabledOverride(overrideValue);
690#endif // ENABLE(MEDIA_STREAM)
691 return;
692 }
693
694 ASSERT_NOT_REACHED();
695}
