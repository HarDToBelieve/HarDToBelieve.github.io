39WebPageDebuggable::WebPageDebuggable(WebPageProxy& page)
40 : m_page(page)
41{
42}
43
44String WebPageDebuggable::name() const
45{
46 if (!m_nameOverride.isNull())
47 return m_nameOverride;
48
49 if (!m_page.mainFrame())
50 return String();
51
52 return m_page.mainFrame()->title();
53}
54
55String WebPageDebuggable::url() const
56{
57 if (!m_page.mainFrame())
58 return aboutBlankURL().string();
59
60 String url = m_page.mainFrame()->url().string();
61 if (url.isEmpty())
62 return aboutBlankURL().string();
63
64 return url;
65}
66
67bool WebPageDebuggable::hasLocalDebugger() const
68{
69 return m_page.inspectorController().hasLocalFrontend();
70}
71
72void WebPageDebuggable::connect(FrontendChannel& channel, bool isAutomaticConnection, bool immediatelyPause)
73{
74 m_page.inspectorController().connectFrontend(channel, isAutomaticConnection, immediatelyPause);
75}
76
77void WebPageDebuggable::disconnect(FrontendChannel& channel)
78{
79 m_page.inspectorController().disconnectFrontend(channel);
80}
81
82void WebPageDebuggable::dispatchMessageFromRemote(const String& message)
83{
84 m_page.inspectorController().dispatchMessageFromFrontend(message);
85}
86
87void WebPageDebuggable::setIndicating(bool indicating)
88{
89 m_page.inspectorController().setIndicating(indicating);
90}
91
92void WebPageDebuggable::setNameOverride(const String& name)
93{
94 m_nameOverride = name;
95 update();
96}
