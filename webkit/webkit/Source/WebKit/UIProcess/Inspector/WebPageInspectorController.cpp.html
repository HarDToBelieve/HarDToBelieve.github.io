47static String getTargetID(const ProvisionalPageProxy& provisionalPage)
48{
49 return WebPageInspectorTarget::toTargetID(provisionalPage.webPageID());

52WebPageInspectorController::WebPageInspectorController(WebPageProxy& inspectedPage)
53 : m_frontendRouter(FrontendRouter::create())
54 , m_backendDispatcher(BackendDispatcher::create(m_frontendRouter.copyRef()))
55 , m_inspectedPage(inspectedPage)
56{
57 auto targetAgent = makeUnique<InspectorTargetAgent>(m_frontendRouter.get(), m_backendDispatcher.get());
58 m_targetAgent = targetAgent.get();
59 m_agents.append(WTFMove(targetAgent));
60}

62void WebPageInspectorController::init()
63{
64 String pageTargetId = WebPageInspectorTarget::toTargetID(m_inspectedPage.webPageID());
65 createInspectorTarget(pageTargetId, Inspector::InspectorTargetType::Page);
66}

80void WebPageInspectorController::connectFrontend(Inspector::FrontendChannel& frontendChannel, bool, bool)
81{
82 createLazyAgents();
83
84 bool connectingFirstFrontend = !m_frontendRouter->hasFrontends();
85
86 m_frontendRouter->connectFrontend(frontendChannel);
87
88 if (connectingFirstFrontend)
89 m_agents.didCreateFrontendAndBackend(&m_frontendRouter.get(), &m_backendDispatcher.get());
90
91 m_inspectedPage.didChangeInspectorFrontendCount(m_frontendRouter->frontendCount());
92
93#if ENABLE(REMOTE_INSPECTOR)
94 if (hasLocalFrontend())
95 m_inspectedPage.remoteInspectorInformationDidChange();
96#endif
97}
98
99void WebPageInspectorController::disconnectFrontend(FrontendChannel& frontendChannel)
100{
101 m_frontendRouter->disconnectFrontend(frontendChannel);
102
103 bool disconnectingLastFrontend = !m_frontendRouter->hasFrontends();
104 if (disconnectingLastFrontend)
105 m_agents.willDestroyFrontendAndBackend(DisconnectReason::InspectorDestroyed);
106
107 m_inspectedPage.didChangeInspectorFrontendCount(m_frontendRouter->frontendCount());
108
109#if ENABLE(REMOTE_INSPECTOR)
110 if (disconnectingLastFrontend)
111 m_inspectedPage.remoteInspectorInformationDidChange();
112#endif
113}
114
115void WebPageInspectorController::disconnectAllFrontends()
116{
117 // FIXME: Handle a local inspector client.
118
119 if (!m_frontendRouter->hasFrontends())
120 return;
121
122 // Notify agents first, since they may need to use InspectorClient.
123 m_agents.willDestroyFrontendAndBackend(DisconnectReason::InspectedTargetDestroyed);
124
125 // Disconnect any remaining remote frontends.
126 m_frontendRouter->disconnectAllFrontends();
127
128 m_inspectedPage.didChangeInspectorFrontendCount(m_frontendRouter->frontendCount());
129
130#if ENABLE(REMOTE_INSPECTOR)
131 m_inspectedPage.remoteInspectorInformationDidChange();
132#endif
133}

154void WebPageInspectorController::createInspectorTarget(const String& targetId, Inspector::InspectorTargetType type)

156 addTarget(InspectorTargetProxy::create(m_inspectedPage, targetId, type));

159void WebPageInspectorController::destroyInspectorTarget(const String& targetId)
160{
161 auto it = m_targets.find(targetId);
162 if (it == m_targets.end())
163 return;
164 m_targetAgent->targetDestroyed(*it->value);
165 m_targets.remove(it);
166}

173bool WebPageInspectorController::shouldPauseLoading(const ProvisionalPageProxy& provisionalPage) const
174{
175 if (!m_frontendRouter->hasFrontends())
176 return false;
177
178 auto* target = m_targets.get(getTargetID(provisionalPage));
179 ASSERT(target);
180 return target->isPaused();
181}

183void WebPageInspectorController::setContinueLoadingCallback(const ProvisionalPageProxy& provisionalPage, WTF::Function<void()>&& callback)
184{
185 auto* target = m_targets.get(getTargetID(provisionalPage));
186 ASSERT(target);
187 target->setResumeCallback(WTFMove(callback));

200void WebPageInspectorController::didCommitProvisionalPage(WebCore::PageIdentifier oldWebPageID, WebCore::PageIdentifier newWebPageID)
201{
202 String oldID = WebPageInspectorTarget::toTargetID(oldWebPageID);
203 String newID = WebPageInspectorTarget::toTargetID(newWebPageID);
204 auto newTarget = m_targets.take(newID);
205 ASSERT(newTarget);
206 newTarget->didCommitProvisionalTarget();
207 m_targetAgent->didCommitProvisionalTarget(oldID, newID);
208
209 // We've disconnected from the old page and will not receive any message from it, so
210 // we destroy everything but the new target here.
211 // FIXME: <https://webkit.org/b/202937> do not destroy targets that belong to the committed page.
212 for (auto& target : m_targets.values())
213 m_targetAgent->targetDestroyed(*target);
214 m_targets.clear();
215 m_targets.set(newTarget->identifier(), WTFMove(newTarget));
216}

227void WebPageInspectorController::createLazyAgents()
228{
229 if (m_didCreateLazyAgents)
230 return;
231
232 m_didCreateLazyAgents = true;
233
234 auto webPageContext = webPageAgentContext();
235
236 m_agents.append(makeUnique<InspectorBrowserAgent>(webPageContext));
237}

239void WebPageInspectorController::addTarget(std::unique_ptr<InspectorTargetProxy>&& target)
240{
241 m_targetAgent->targetCreated(*target);
242 m_targets.set(target->identifier(), WTFMove(target));
243}

245void WebPageInspectorController::setEnabledBrowserAgent(InspectorBrowserAgent* agent)
246{
247 if (m_enabledBrowserAgent == agent)
248 return;
249
250 m_enabledBrowserAgent = agent;
251
252 if (m_enabledBrowserAgent)
253 m_inspectedPage.uiClient().didEnableInspectorBrowserDomain(m_inspectedPage);
254 else
255 m_inspectedPage.uiClient().didDisableInspectorBrowserDomain(m_inspectedPage);
256}
