65- (void)_createLayers:(NSUInteger)numLayers
66{
67 if ([_layers count] == numLayers)
68 return;
69
70 for (NSUInteger i = 0; i < numLayers; ++i) {
71 auto layer = adoptNS([[CAShapeLayer alloc] init]);
72 [_layers addObject:layer.get()];
73 [self.layer addSublayer:layer.get()];

77static bool findIntersectionOnLineBetweenPoints(const WebCore::FloatPoint& p1, const WebCore::FloatPoint& p2, const WebCore::FloatPoint& d1, const WebCore::FloatPoint& d2, WebCore::FloatPoint& intersection)
78{
79 // Do the lines intersect?
80 WebCore::FloatPoint temporaryIntersectionPoint;
81 if (!findIntersection(p1, p2, d1, d2, temporaryIntersectionPoint))
82 return false;
83
84 // Is the intersection between the two points on the line?
85 if (p1.x() >= p2.x()) {
86 if (temporaryIntersectionPoint.x() > p1.x() || temporaryIntersectionPoint.x() < p2.x())
87 return false;
88 } else {
89 if (temporaryIntersectionPoint.x() > p2.x() || temporaryIntersectionPoint.x() < p1.x())
90 return false;
91 }
92 if (p1.y() >= p2.y()) {
93 if (temporaryIntersectionPoint.y() > p1.y() || temporaryIntersectionPoint.y() < p2.y())
94 return false;
95 } else {
96 if (temporaryIntersectionPoint.y() > p2.y() || temporaryIntersectionPoint.y() < p1.y())
97 return false;
98 }
99
100 intersection = temporaryIntersectionPoint;
101 return true;
102}

104// This quad intersection works because the two quads are known to be at the same
105// rotation and clockwise-ness.
106static WebCore::FloatQuad quadIntersection(WebCore::FloatQuad bounds, WebCore::FloatQuad toClamp)
107{
108 // Resulting points.
109 WebCore::FloatPoint p1, p2, p3, p4;
110 bool containsPoint1 = false;
111 bool containsPoint2 = false;
112 bool containsPoint3 = false;
113 bool containsPoint4 = false;
114 bool intersectForPoint1 = false;
115 bool intersectForPoint2 = false;
116 bool intersectForPoint3 = false;
117 bool intersectForPoint4 = false;
118
119 // Top / bottom vertical clamping.
120 if (bounds.containsPoint(toClamp.p1())) {
121 containsPoint1 = true;
122 p1 = toClamp.p1();
123 } else if (!(intersectForPoint1 = findIntersectionOnLineBetweenPoints(bounds.p1(), bounds.p2(), toClamp.p1(), toClamp.p4(), p1)))
124 p1 = toClamp.p1();
125
126 if (bounds.containsPoint(toClamp.p2())) {
127 containsPoint2 = true;
128 p2 = toClamp.p2();
129 } else if (!(intersectForPoint2 = findIntersectionOnLineBetweenPoints(bounds.p1(), bounds.p2(), toClamp.p2(), toClamp.p3(), p2)))
130 p2 = toClamp.p2();
131
132 if (bounds.containsPoint(toClamp.p3())) {
133 containsPoint3 = true;
134 p3 = toClamp.p3();
135 } else if (!(intersectForPoint3 = findIntersectionOnLineBetweenPoints(bounds.p4(), bounds.p3(), toClamp.p2(), toClamp.p3(), p3)))
136 p3 = toClamp.p3();
137
138 if (bounds.containsPoint(toClamp.p4())) {
139 containsPoint4 = true;
140 p4 = toClamp.p4();
141 } else if (!(intersectForPoint4 = findIntersectionOnLineBetweenPoints(bounds.p4(), bounds.p3(), toClamp.p1(), toClamp.p4(), p4)))
142 p4 = toClamp.p4();
143
144 // If only one of the points intersected on either the top or bottom line then we
145 // can clamp the other point on that line to the corner of the bounds.
146 if (!containsPoint1 && intersectForPoint2 && !intersectForPoint1) {
147 containsPoint1 = true;
148 p1 = bounds.p1();
149 } else if (!containsPoint2 && intersectForPoint1 && !intersectForPoint2) {
150 containsPoint2 = true;
151 p2 = bounds.p2();
152 }
153 if (!containsPoint4 && intersectForPoint3 && !intersectForPoint4) {
154 containsPoint4 = true;
155 p4 = bounds.p4();
156 } else if (!containsPoint3 && intersectForPoint4 && !intersectForPoint3) {
157 containsPoint3 = true;
158 p3 = bounds.p3();
159 }
160
161 // Now we only need to perform horizontal clamping for unadjusted points.
162 if (!containsPoint2 && !intersectForPoint2)
163 findIntersectionOnLineBetweenPoints(bounds.p2(), bounds.p3(), p1, p2, p2);
164 if (!containsPoint3 && !intersectForPoint3)
165 findIntersectionOnLineBetweenPoints(bounds.p2(), bounds.p3(), p4, p3, p3);
166 if (!containsPoint1 && !intersectForPoint1)
167 findIntersectionOnLineBetweenPoints(bounds.p1(), bounds.p4(), p1, p2, p1);
168 if (!containsPoint4 && !intersectForPoint4)
169 findIntersectionOnLineBetweenPoints(bounds.p1(), bounds.p4(), p4, p3, p4);
170
171 return WebCore::FloatQuad(p1, p2, p3, p4);
172}

174static void layerPathWithHole(CAShapeLayer *layer, const WebCore::FloatQuad& outerQuad, const WebCore::FloatQuad& holeQuad)
175{
176 // Nothing to show.
177 if (outerQuad == holeQuad || holeQuad.containsQuad(outerQuad)) {
178 layer.path = NULL;
179 return;
180 }
181
182 // If there is a negative margin / padding then the outer box might not
183 // fully contain the hole box. In such cases we recalculate the hole to
184 // be the intersection of the two quads.
185 WebCore::FloatQuad innerHole;
186 if (outerQuad.containsQuad(holeQuad))
187 innerHole = holeQuad;
188 else
189 innerHole = quadIntersection(outerQuad, holeQuad);
190
191 // Clockwise inside rect (hole), Counter-Clockwise outside rect (fill).
192 auto path = adoptCF(CGPathCreateMutable());
193 CGPathMoveToPoint(path.get(), 0, innerHole.p1().x(), innerHole.p1().y());
194 CGPathAddLineToPoint(path.get(), 0, innerHole.p2().x(), innerHole.p2().y());
195 CGPathAddLineToPoint(path.get(), 0, innerHole.p3().x(), innerHole.p3().y());
196 CGPathAddLineToPoint(path.get(), 0, innerHole.p4().x(), innerHole.p4().y());
197 CGPathMoveToPoint(path.get(), 0, outerQuad.p1().x(), outerQuad.p1().y());
198 CGPathAddLineToPoint(path.get(), 0, outerQuad.p4().x(), outerQuad.p4().y());
199 CGPathAddLineToPoint(path.get(), 0, outerQuad.p3().x(), outerQuad.p3().y());
200 CGPathAddLineToPoint(path.get(), 0, outerQuad.p2().x(), outerQuad.p2().y());
201 layer.path = path.get();
202}

204static void layerPath(CAShapeLayer *layer, const WebCore::FloatQuad& outerQuad)
205{
206 auto path = adoptCF(CGPathCreateMutable());
207 CGPathMoveToPoint(path.get(), 0, outerQuad.p1().x(), outerQuad.p1().y());
208 CGPathAddLineToPoint(path.get(), 0, outerQuad.p4().x(), outerQuad.p4().y());
209 CGPathAddLineToPoint(path.get(), 0, outerQuad.p3().x(), outerQuad.p3().y());
210 CGPathAddLineToPoint(path.get(), 0, outerQuad.p2().x(), outerQuad.p2().y());
211 CGPathCloseSubpath(path.get());
212 layer.path = path.get();
213}

215- (void)_layoutForNodeHighlight:(const WebCore::InspectorOverlay::Highlight&)highlight offset:(unsigned)offset
216{
217 ASSERT([_layers count] >= offset + 4);
218 ASSERT(highlight.quads.size() >= offset + 4);
219 if ([_layers count] < offset + 4 || highlight.quads.size() < offset + 4)
220 return;
221
222 CAShapeLayer *marginLayer = [_layers objectAtIndex:offset];
223 CAShapeLayer *borderLayer = [_layers objectAtIndex:offset + 1];
224 CAShapeLayer *paddingLayer = [_layers objectAtIndex:offset + 2];
225 CAShapeLayer *contentLayer = [_layers objectAtIndex:offset + 3];
226
227 WebCore::FloatQuad marginQuad = highlight.quads[offset];
228 WebCore::FloatQuad borderQuad = highlight.quads[offset + 1];
229 WebCore::FloatQuad paddingQuad = highlight.quads[offset + 2];
230 WebCore::FloatQuad contentQuad = highlight.quads[offset + 3];
231
232 marginLayer.fillColor = cachedCGColor(highlight.marginColor);
233 borderLayer.fillColor = cachedCGColor(highlight.borderColor);
234 paddingLayer.fillColor = cachedCGColor(highlight.paddingColor);
235 contentLayer.fillColor = cachedCGColor(highlight.contentColor);
236
237 layerPathWithHole(marginLayer, marginQuad, borderQuad);
238 layerPathWithHole(borderLayer, borderQuad, paddingQuad);
239 layerPathWithHole(paddingLayer, paddingQuad, contentQuad);
240 layerPath(contentLayer, contentQuad);
241}

243- (void)_layoutForNodeListHighlight:(const WebCore::InspectorOverlay::Highlight&)highlight
244{
245 if (!highlight.quads.size())
246 return;
247
248 unsigned nodeCount = highlight.quads.size() / 4;
249 [self _createLayers:nodeCount * 4];
250
251 for (unsigned i = 0; i < nodeCount; ++i)
252 [self _layoutForNodeHighlight:highlight offset:i * 4];
253}

255- (void)_layoutForRectsHighlight:(const WebCore::InspectorOverlay::Highlight&)highlight
256{
257 NSUInteger numLayers = highlight.quads.size();
258 if (!numLayers)
259 return;
260
261 [self _createLayers:numLayers];
262
263 CGColorRef contentColor = cachedCGColor(highlight.contentColor);
264 for (NSUInteger i = 0; i < numLayers; ++i) {
265 CAShapeLayer *layer = [_layers objectAtIndex:i];
266 layer.fillColor = contentColor;
267 layerPath(layer, highlight.quads[i]);
268 }

280static CALayer * createLayoutHatchingLayer(WebCore::FloatQuad quad, WebCore::Color strokeColor)
281{
282 CAShapeLayer *layer = [CAShapeLayer layer];
283
284 constexpr auto hatchSpacing = 12;
285 auto hatchPath = adoptCF(CGPathCreateMutable());
286
287 WebCore::FloatLine topSide = { quad.p1(), quad.p2() };
288 WebCore::FloatLine leftSide = { quad.p1(), quad.p4() };
289
290 // The opposite axis' length is used to determine how far to draw a hatch line in both dimensions, which keeps the lines at a 45deg angle.
291 if (topSide.length() > leftSide.length()) {
292 WebCore::FloatLine bottomSide = { quad.p4(), quad.p3() };
293 // Move across the relative top of the quad, starting left of `0, 0` to ensure that the tail of the previous hatch line is drawn while scrolling.
294 for (float x = -leftSide.length(); x < topSide.length(); x += hatchSpacing) {
295 auto startPoint = topSide.pointAtAbsoluteDistance(x);
296 auto endPoint = bottomSide.pointAtAbsoluteDistance(x + leftSide.length());
297 CGPathMoveToPoint(hatchPath.get(), 0, startPoint.x(), startPoint.y());
298 CGPathAddLineToPoint(hatchPath.get(), 0, endPoint.x(), endPoint.y());
299 }
300 } else {
301 WebCore::FloatLine rightSide = { quad.p2(), quad.p3() };
302 // Move down the relative left side of the quad, starting above `0, 0` to ensure that the tail of the previous hatch line is drawn while scrolling.
303 for (float y = -topSide.length(); y < leftSide.length(); y += hatchSpacing) {
304 auto startPoint = leftSide.pointAtAbsoluteDistance(y);
305 auto endPoint = rightSide.pointAtAbsoluteDistance(y + topSide.length());
306 CGPathMoveToPoint(hatchPath.get(), 0, startPoint.x(), startPoint.y());
307 CGPathAddLineToPoint(hatchPath.get(), 0, endPoint.x(), endPoint.y());
308 }
309 }
310 layer.path = hatchPath.get();
311 layer.strokeColor = cachedCGColor(strokeColor);
312 layer.lineWidth = 0.5;
313 layer.lineDashPattern = @[[NSNumber numberWithInt:2], [NSNumber numberWithInt:2]];
314
315 CAShapeLayer *maskLayer = [CAShapeLayer layer];
316 layerPath(maskLayer, quad);
317 layer.mask = maskLayer;
318 return layer;
319}

321static CALayer * createLayoutLabelLayer(String label, WebCore::FloatPoint point, WebCore::InspectorOverlay::LabelArrowDirection arrowDirection, WebCore::InspectorOverlay::LabelArrowEdgePosition arrowEdgePosition, WebCore::Color backgroundColor, WebCore::Color strokeColor, double scale, float maximumWidth = 0)
322{
323 auto font = WebCore::InspectorOverlay::fontForLayoutLabel();
324
325 constexpr auto padding = 4;
326 constexpr auto arrowSize = 6;
327 float textHeight = font.fontMetrics().floatHeight();
328
329 float textWidth = font.width(WebCore::TextRun(label));
330 if (maximumWidth && textWidth + (padding * 2) > maximumWidth) {
331 label.append("..."_s);
332 while (textWidth + (padding * 2) > maximumWidth && label.length() >= 4) {
333 // Remove the fourth from last character (the character before the ellipsis) and remeasure.
334 label.remove(label.length() - 4);
335 textWidth = font.width(WebCore::TextRun(label));
336 }
337 }
338
339 // Note: Implementation Difference - The textPosition is the center of text, unlike WebCore::InspectorOverlay, where the textPosition is leftmost point on the baseline of the text.
340 WebCore::FloatPoint textPosition;
341 switch (arrowDirection) {
342 case WebCore::InspectorOverlay::LabelArrowDirection::Down:
343 switch (arrowEdgePosition) {
344 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Leading:
345 textPosition = WebCore::FloatPoint((textWidth / 2) + padding, -(textHeight / 2) - arrowSize - padding);
346 break;
347 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Middle:
348 textPosition = WebCore::FloatPoint(0, -(textHeight / 2) - arrowSize - padding);
349 break;
350 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Trailing:
351 textPosition = WebCore::FloatPoint(-(textWidth / 2) - padding, -(textHeight / 2) - arrowSize - padding);
352 break;
353 case WebCore::InspectorOverlay::LabelArrowEdgePosition::None:
354 break;
355 }
356 break;
357 case WebCore::InspectorOverlay::LabelArrowDirection::Up:
358 switch (arrowEdgePosition) {
359 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Leading:
360 textPosition = WebCore::FloatPoint((textWidth / 2) + padding, (textHeight / 2) + arrowSize + padding);
361 break;
362 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Middle:
363 textPosition = WebCore::FloatPoint(0, (textHeight / 2) + arrowSize + padding);
364 break;
365 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Trailing:
366 textPosition = WebCore::FloatPoint(-(textWidth / 2) - padding, (textHeight / 2) + arrowSize + padding);
367 break;
368 case WebCore::InspectorOverlay::LabelArrowEdgePosition::None:
369 break;
370 }
371 break;
372 case WebCore::InspectorOverlay::LabelArrowDirection::Right:
373 switch (arrowEdgePosition) {
374 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Leading:
375 textPosition = WebCore::FloatPoint(-(textWidth / 2) - arrowSize - padding, (textHeight / 2) + padding);
376 break;
377 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Middle:
378 textPosition = WebCore::FloatPoint(-(textWidth / 2) - arrowSize - padding, 0);
379 break;
380 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Trailing:
381 textPosition = WebCore::FloatPoint(-(textWidth / 2) - arrowSize - padding, -(textHeight / 2) - padding);
382 break;
383 case WebCore::InspectorOverlay::LabelArrowEdgePosition::None:
384 break;
385 }
386 break;
387 case WebCore::InspectorOverlay::LabelArrowDirection::Left:
388 switch (arrowEdgePosition) {
389 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Leading:
390 textPosition = WebCore::FloatPoint((textWidth / 2) + arrowSize + padding, (textHeight / 2) + padding);
391 break;
392 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Middle:
393 textPosition = WebCore::FloatPoint((textWidth / 2) + arrowSize + padding, 0);
394 break;
395 case WebCore::InspectorOverlay::LabelArrowEdgePosition::Trailing:
396 textPosition = WebCore::FloatPoint((textWidth / 2) + arrowSize + padding, -(textHeight / 2) - padding);
397 break;
398 case WebCore::InspectorOverlay::LabelArrowEdgePosition::None:
399 break;
400 }
401 break;
402 case WebCore::InspectorOverlay::LabelArrowDirection::None:
403 // Text position will remain (0, 0).
404 break;
405 }
406
407 CALayer *layer = [CALayer layer];
408
409#if USE(CG)
410 // WebCore::Path::PlatformPathPtr is only a CGPath* when `USE(CG)` is true.
411 auto labelPath = WebCore::InspectorOverlay::backgroundPathForLayoutLabel(textWidth + (padding * 2), textHeight + (padding * 2), arrowDirection, arrowEdgePosition, arrowSize);
412 CGPath* platformLabelPath = labelPath.ensurePlatformPath();
413
414 CAShapeLayer *labelPathLayer = [CAShapeLayer layer];
415 labelPathLayer.path = platformLabelPath;
416 labelPathLayer.fillColor = cachedCGColor(backgroundColor);
417 labelPathLayer.strokeColor = cachedCGColor(strokeColor);
418 labelPathLayer.position = CGPointMake(point.x(), point.y());
419 [layer addSublayer:labelPathLayer];
420#endif
421
422 CATextLayer *textLayer = [CATextLayer layer];
423 textLayer.frame = CGRectMake(0, 0, textWidth, textHeight);
424 textLayer.string = label;
425 textLayer.font = font.primaryFont().getCTFont();
426 textLayer.fontSize = 12;
427 textLayer.alignmentMode = kCAAlignmentLeft;
428 textLayer.foregroundColor = CGColorGetConstantColor(kCGColorBlack);
429 textLayer.position = CGPointMake(textPosition.x() + point.x(), textPosition.y() + point.y());
430 textLayer.contentsScale = [[UIScreen mainScreen] scale] * scale;
431 [layer addSublayer:textLayer];
432
433 return layer;
434}

436- (CALayer *)_createGridOverlayLayer:(const WebCore::InspectorOverlay::Highlight::GridHighlightOverlay&)overlay scale:(double)scale
437{
438 // Keep implementation roughly equivalent to `WebCore::InspectorOverlay::drawGridOverlay`.
439 CALayer *layer = [CALayer layer];
440
441 auto gridLinesPath = adoptCF(CGPathCreateMutable());
442 for (auto gridLine : overlay.gridLines) {
443 CGPathMoveToPoint(gridLinesPath.get(), 0, gridLine.start().x(), gridLine.start().y());
444 CGPathAddLineToPoint(gridLinesPath.get(), 0, gridLine.end().x(), gridLine.end().y());
445 }
446 CAShapeLayer *gridLinesLayer = [CAShapeLayer layer];
447 gridLinesLayer.path = gridLinesPath.get();
448 gridLinesLayer.lineWidth = 1;
449 gridLinesLayer.strokeColor = cachedCGColor(overlay.color);
450 [layer addSublayer:gridLinesLayer];
451
452 for (auto gapQuad : overlay.gaps)
453 [layer addSublayer:createLayoutHatchingLayer(gapQuad, overlay.color)];
454
455 for (auto area : overlay.areas) {
456 CAShapeLayer *areaLayer = [CAShapeLayer layer];
457 layerPath(areaLayer, area.quad);
458 areaLayer.lineWidth = 3;
459 areaLayer.fillColor = CGColorGetConstantColor(kCGColorClear);
460 areaLayer.strokeColor = cachedCGColor(overlay.color);
461 [layer addSublayer:areaLayer];
462 }
463
464 constexpr auto translucentLabelBackgroundColor = WebCore::Color::white.colorWithAlphaByte(230);
465
466 for (auto area : overlay.areas)
467 [layer addSublayer:createLayoutLabelLayer(area.name, area.quad.center(), WebCore::InspectorOverlay::LabelArrowDirection::None, WebCore::InspectorOverlay::LabelArrowEdgePosition::None, translucentLabelBackgroundColor, overlay.color, scale, area.quad.boundingBox().width())];
468
469 for (auto label : overlay.labels)
470 [layer addSublayer:createLayoutLabelLayer(label.text, label.location, label.arrowDirection, label.arrowEdgePosition, label.backgroundColor, overlay.color, scale)];
471
472 return layer;
473}

475- (void)update:(const WebCore::InspectorOverlay::Highlight&)highlight scale:(double)scale
476{
477 [self _removeAllLayers];
478
479 if (highlight.type == WebCore::InspectorOverlay::Highlight::Type::Node || highlight.type == WebCore::InspectorOverlay::Highlight::Type::NodeList)
480 [self _layoutForNodeListHighlight:highlight];
481 else if (highlight.type == WebCore::InspectorOverlay::Highlight::Type::Rects)
482 [self _layoutForRectsHighlight:highlight];
483
484 [self _createGridOverlayLayers:highlight scale:scale];
485}
