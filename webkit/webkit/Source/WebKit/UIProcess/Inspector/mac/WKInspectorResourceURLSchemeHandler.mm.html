47- (NSSet<NSString *> *)allowedURLSchemesForCSP
48{
49 return _allowedURLSchemesForCSP.get();

52- (void)setAllowedURLSchemesForCSP:(NSSet<NSString *> *)allowedURLSchemes
53{
54 _allowedURLSchemesForCSP = adoptNS([allowedURLSchemes copy]);
55}

57- (NSSet<NSURL *> *)mainResourceURLsForCSP
58{
59 if (!_mainResourceURLsForCSP)
60 _mainResourceURLsForCSP = adoptNS([[NSSet alloc] initWithObjects:[NSURL URLWithString:WebKit::WebInspectorUIProxy::inspectorPageURL()], [NSURL URLWithString:WebKit::WebInspectorUIProxy::inspectorTestPageURL()], nil]);
61
62 return _mainResourceURLsForCSP.get();
63}

65// MARK - WKURLSchemeHandler Protocol
66
67- (void)webView:(WKWebView *)webView startURLSchemeTask:(id <WKURLSchemeTask>)urlSchemeTask
68{
69 if (!_cachedBundle) {
70 _cachedBundle = [NSBundle bundleWithIdentifier:@"com.apple.WebInspectorUI"];
71
72 // It is an error if WebInspectorUI has not already been soft-linked by the time
73 // we load resources from it. And if soft-linking fails, we shouldn't start loads.
74 RELEASE_ASSERT(_cachedBundle);
75 }
76
77 if (!_fileLoadOperations)
78 _fileLoadOperations = adoptNS([[NSMapTable alloc] initWithKeyOptions:NSPointerFunctionsStrongMemory valueOptions:NSPointerFunctionsStrongMemory capacity:5]);
79
80 if (!_operationQueue) {
81 _operationQueue = adoptNS([[NSOperationQueue alloc] init]);
82 _operationQueue.get().underlyingQueue = dispatch_get_global_queue(QOS_CLASS_USER_INTERACTIVE, 0);
83 _operationQueue.get().qualityOfService = NSOperationQualityOfServiceUserInteractive;
84
85 // The default value (NSOperationQueueDefaultMaxConcurrentOperationCount) results in a large number of threads
86 // that can exceed the soft limit if two Web Inspector instances are being loaded simultaneously.
87 _operationQueue.get().maxConcurrentOperationCount = 4;
88 }
89
90 NSBlockOperation *operation = [NSBlockOperation blockOperationWithBlock:^{
91 dispatch_async(dispatch_get_main_queue(), ^{
92 [_fileLoadOperations removeObjectForKey:urlSchemeTask];
93 });
94
95 NSURL *requestURL = urlSchemeTask.request.URL;
96 NSURL *fileURLForRequest = [_cachedBundle URLForResource:requestURL.relativePath withExtension:@""];
97 if (!fileURLForRequest) {
98 LOG_ERROR("Unable to find Web Inspector resource: %@", requestURL.absoluteString);
99 [urlSchemeTask didFailWithError:[NSError errorWithDomain:NSCocoaErrorDomain code:NSURLErrorFileDoesNotExist userInfo:nil]];
100 return;
101 }
102
103 NSError *readError;
104 NSData *fileData = [NSData dataWithContentsOfURL:fileURLForRequest options:0 error:&readError];
105 if (!fileData) {
106 LOG_ERROR("Unable to read data for Web Inspector resource: %@", requestURL.absoluteString);
107 [urlSchemeTask didFailWithError:[NSError errorWithDomain:NSCocoaErrorDomain code:NSURLErrorResourceUnavailable userInfo:@{
108 NSUnderlyingErrorKey: readError,
109 }]];
110 return;
111 }
112
113 NSString *mimeType = WebCore::MIMETypeRegistry::mimeTypeForExtension(fileURLForRequest.pathExtension);
114 if (!mimeType)
115 mimeType = @"application/octet-stream";
116
117 RetainPtr<NSMutableDictionary> headerFields = adoptNS(@{
118 @"Access-Control-Allow-Origin": @"*",
119 @"Content-Length": [NSString stringWithFormat:@"%zu", (size_t)fileData.length],
120 @"Content-Type": mimeType,
121 }.mutableCopy);
122
123 // Allow fetches for resources that use a registered custom URL scheme.
124 if (_allowedURLSchemesForCSP && [self.mainResourceURLsForCSP containsObject:requestURL]) {
125 NSString *stringForCSPPolicy = [NSString stringWithFormat:@"connect-src * %@:", [_allowedURLSchemesForCSP.get().allObjects componentsJoinedByString:@": "]];
126 [headerFields setObject:stringForCSPPolicy forKey:@"Content-Security-Policy"];
127 }
128
129 RetainPtr<NSHTTPURLResponse> urlResponse = adoptNS([[NSHTTPURLResponse alloc] initWithURL:urlSchemeTask.request.URL statusCode:200 HTTPVersion:nil headerFields:headerFields.get()]);
130 [urlSchemeTask didReceiveResponse:urlResponse.get()];
131 [urlSchemeTask didReceiveData:fileData];
132 [urlSchemeTask didFinish];
133 }];
134
135 [_fileLoadOperations setObject:operation forKey:urlSchemeTask];
136 [_operationQueue addOperation:operation];
137}
138
139- (void)webView:(WKWebView *)webView stopURLSchemeTask:(id <WKURLSchemeTask>)urlSchemeTask
140{
141 // Ensure that all blocks with pending removals are dispatched before doing a map lookup.
142 dispatch_async(dispatch_get_main_queue(), ^{
143 if (NSOperation *operation = [_fileLoadOperations objectForKey:urlSchemeTask]) {
144 [operation cancel];
145 [_fileLoadOperations removeObjectForKey:urlSchemeTask];
146 }
147 });
148}
