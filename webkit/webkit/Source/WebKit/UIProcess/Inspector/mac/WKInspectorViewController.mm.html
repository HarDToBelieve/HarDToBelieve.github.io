62- (instancetype)initWithConfiguration:(_WKInspectorConfiguration *)configuration inspectedPage:(NakedPtr<WebKit::WebPageProxy>)inspectedPage
63{
64 if (!(self = [super init]))
65 return nil;
66
67 _configuration = adoptNS([configuration copy]);
68
69 // The (local) inspected page is nil if the controller is hosting a Remote Web Inspector view.
70 _inspectedPage = inspectedPage;
71
72 return self;
73}

92- (WKWebView *)webView
93{
94 // Construct lazily so the client can set the delegate before the WebView is created.
95 if (!_webView) {
96 NSRect initialFrame = NSMakeRect(0, 0, WebKit::WebInspectorUIProxy::initialWindowWidth, WebKit::WebInspectorUIProxy::initialWindowHeight);
97 _webView = adoptNS([[WKInspectorWKWebView alloc] initWithFrame:initialFrame configuration:self.webViewConfiguration]);
98 [_webView setUIDelegate:self];
99 [_webView setNavigationDelegate:self];
100 [_webView setInspectorWKWebViewDelegate:self];
101 [_webView _setAutomaticallyAdjustsContentInsets:NO];
102 [_webView _setUseSystemAppearance:YES];
103 [_webView setAutoresizingMask:NSViewWidthSizable | NSViewHeightSizable];
104 }
105
106 return _webView.get();
107}

114- (WKWebViewConfiguration *)webViewConfiguration
115{
116 RetainPtr<WKWebViewConfiguration> configuration = adoptNS([[WKWebViewConfiguration alloc] init]);
117 RetainPtr<WKInspectorResourceURLSchemeHandler> inspectorSchemeHandler = adoptNS([WKInspectorResourceURLSchemeHandler new]);
118 RetainPtr<NSMutableSet<NSString *>> allowedURLSchemes = adoptNS([[NSMutableSet alloc] initWithObjects:WKInspectorResourceScheme, nil]);
119 for (auto& pair : _configuration->_configuration->urlSchemeHandlers())
120 [allowedURLSchemes addObject:pair.second];
121
122 [inspectorSchemeHandler setAllowedURLSchemesForCSP:allowedURLSchemes.get()];
123 [configuration setURLSchemeHandler:inspectorSchemeHandler.get() forURLScheme:WKInspectorResourceScheme];
124
125 WKPreferences *preferences = configuration.get().preferences;
126 preferences._allowFileAccessFromFileURLs = YES;
127 [configuration _setAllowUniversalAccessFromFileURLs:YES];
128 [configuration _setAllowTopNavigationToDataURLs:YES];
129 preferences._storageBlockingPolicy = _WKStorageBlockingPolicyAllowAll;
130 preferences._javaScriptRuntimeFlags = 0;
131
132#ifndef NDEBUG
133 // Allow developers to inspect the Web Inspector in debug builds without changing settings.
134 preferences._developerExtrasEnabled = YES;
135 preferences._logsPageMessagesToSystemConsoleEnabled = YES;
136#endif
137
138 preferences._diagnosticLoggingEnabled = YES;
139
140 [_configuration applyToWebViewConfiguration:configuration.get()];
141
142 if (!!_delegate && [_delegate respondsToSelector:@selector(inspectorViewControllerInspectorIsUnderTest:)]) {
143 if ([_delegate inspectorViewControllerInspectorIsUnderTest:self]) {
144 preferences._hiddenPageDOMTimerThrottlingEnabled = NO;
145 preferences._pageVisibilityBasedProcessSuppressionEnabled = NO;
146 }
147 }
148
149 // WKInspectorConfiguration allows the client to specify a process pool to use.
150 // If not specified or the inspection level is >1, use the default strategy.
151 // This ensures that Inspector^2 cannot be affected by client (mis)configuration.
152 auto* customProcessPool = configuration.get().processPool;
153 auto inspectorLevel = WebKit::inspectorLevelForPage(_inspectedPage);
154 auto useDefaultProcessPool = inspectorLevel > 1 || !customProcessPool;
155 if (customProcessPool && !useDefaultProcessPool)
156 WebKit::prepareProcessPoolForInspector(*customProcessPool->_processPool.get());
157
158 if (useDefaultProcessPool)
159 [configuration setProcessPool:wrapper(WebKit::defaultInspectorProcessPool(inspectorLevel))];
160
161 // Ensure that a page group identifier is set. This is for computing inspection levels.
162 if (!configuration.get()._groupIdentifier)
163 [configuration _setGroupIdentifier:WebKit::defaultInspectorPageGroupIdentifierForPage(_inspectedPage)];
164
165 return configuration.autorelease();
166}

196- (void)webView:(WKWebView *)webView runOpenPanelWithParameters:(WKOpenPanelParameters *)parameters initiatedByFrame:(WKFrameInfo *)frame completionHandler:(void (^)(NSArray<NSURL *> *URLs))completionHandler
197{
198 NSOpenPanel *openPanel = [NSOpenPanel openPanel];
199 openPanel.allowsMultipleSelection = parameters.allowsMultipleSelection;
200
201 auto reportSelectedFiles = ^(NSInteger result) {
202 if (result == NSModalResponseOK)
203 completionHandler(openPanel.URLs);
204 else
205 completionHandler(nil);
206 };
207
208 if (_webView.get().window)
209 [openPanel beginSheetModalForWindow:_webView.get().window completionHandler:reportSelectedFiles];
210 else
211 reportSelectedFiles([openPanel runModal]);
212}

219- (NSMenu *)_webView:(WKWebView *)webView contextMenu:(NSMenu *)menu forElement:(_WKContextMenuElementInfo *)element
220{
221 for (NSInteger i = menu.numberOfItems - 1; i >= 0; --i) {
222 NSMenuItem *item = [menu itemAtIndex:i];
223 switch (item.tag) {
224 case kWKContextMenuItemTagOpenLinkInNewWindow:
225 case kWKContextMenuItemTagOpenImageInNewWindow:
226 case kWKContextMenuItemTagOpenFrameInNewWindow:
227 case kWKContextMenuItemTagOpenMediaInNewWindow:
228 case kWKContextMenuItemTagCopyImageUrlToClipboard:
229 case kWKContextMenuItemTagCopyImageToClipboard:
230 case kWKContextMenuItemTagDownloadLinkToDisk:
231 case kWKContextMenuItemTagDownloadImageToDisk:
232 [menu removeItemAtIndex:i];
233 break;
234 }
235 }
236
237 return menu;
238}

248- (void)webView:(WKWebView *)webView decidePolicyForNavigationAction:(WKNavigationAction *)navigationAction decisionHandler:(void (^)(WKNavigationActionPolicy))decisionHandler
249{
250 // Allow non-main frames to navigate anywhere.
251 if (!navigationAction.targetFrame.isMainFrame) {
252 decisionHandler(WKNavigationActionPolicyAllow);
253 return;
254 }
255
256 // Allow loading of the main inspector file.
257 if ([navigationAction.request.URL.scheme isEqualToString:WKInspectorResourceScheme]) {
258 decisionHandler(WKNavigationActionPolicyAllow);
259 return;
260 }
261
262 // Prevent everything else.
263 decisionHandler(WKNavigationActionPolicyCancel);
264
265 if (!!_delegate && [_delegate respondsToSelector:@selector(inspectorViewController:openURLExternally:)]) {
266 [_delegate inspectorViewController:self openURLExternally:navigationAction.request.URL];
267 return;
268 }
269
270 // Try to load the request in the inspected page if the delegate can't handle it.
271 if (_inspectedPage)
272 _inspectedPage->loadRequest(navigationAction.request);
273}

283- (void)inspectorWKWebViewReload:(WKInspectorWKWebView *)webView
284{
285 if (!_inspectedPage)
286 return;
287
288 OptionSet<WebCore::ReloadOption> reloadOptions;
289 if (WebCore::linkedOnOrAfter(WebCore::SDKVersion::FirstWithExpiredOnlyReloadBehavior))
290 reloadOptions.add(WebCore::ReloadOption::ExpiredOnly);
291
292 _inspectedPage->reload(reloadOptions);
293}
