26#import "config.h"
27#import "WKInspectorWKWebView.h"
28
29#if PLATFORM(MAC)
30
31#import "WKInspectorPrivateMac.h"
32#import <wtf/WeakObjCPtr.h>
33
34@implementation WKInspectorWKWebView {
35 WeakObjCPtr<id <WKInspectorWKWebViewDelegate>> _inspectorWKWebViewDelegate;
36}
37
38- (void)dealloc
39{
40 [[NSNotificationCenter defaultCenter] removeObserver:self name:NSWindowDidBecomeKeyNotification object:nil];
41
42 [super dealloc];
43}
44
45- (NSRect)_opaqueRectForWindowMoveWhenInTitlebar
46{
47 // This convinces AppKit to allow window moves when clicking anywhere in the titlebar (top 22pt)
48 // when this view's contents cover the entire window's contents, including the titlebar.
49 return NSZeroRect;
50}
51
52- (NSInteger)tag
53{
54 return WKInspectorViewTag;
55}
56
57- (id <WKInspectorWKWebViewDelegate>)inspectorWKWebViewDelegate
58{
59 return _inspectorWKWebViewDelegate.getAutoreleased();
60}
61
62- (void)setInspectorWKWebViewDelegate:(id <WKInspectorWKWebViewDelegate>)delegate
63{
64 if (!!_inspectorWKWebViewDelegate)
65 [[NSNotificationCenter defaultCenter] removeObserver:self name:NSWindowDidBecomeKeyNotification object:nil];
66
67 _inspectorWKWebViewDelegate = delegate;
68
69 if (!!_inspectorWKWebViewDelegate)
70 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(_handleWindowDidBecomeKey:) name:NSWindowDidBecomeKeyNotification object:nil];
71}
72
73- (IBAction)reload:(id)sender
74{
75 [self.inspectorWKWebViewDelegate inspectorWKWebViewReload:self];
76}
77
78- (IBAction)reloadFromOrigin:(id)sender
79{
80 [self.inspectorWKWebViewDelegate inspectorWKWebViewReloadFromOrigin:self];
81}
82
83- (void)viewWillMoveToWindow:(NSWindow *)newWindow
84{
85 [super viewWillMoveToWindow:newWindow];
86 [self.inspectorWKWebViewDelegate inspectorWKWebView:self willMoveToWindow:newWindow];
87}
88
89- (void)viewDidMoveToWindow
90{
91 [super viewDidMoveToWindow];
92 [self.inspectorWKWebViewDelegate inspectorWKWebViewDidMoveToWindow:self];
93}
94
95- (BOOL)becomeFirstResponder
96{
97 BOOL result = [super becomeFirstResponder];
98 [self.inspectorWKWebViewDelegate inspectorWKWebViewDidBecomeActive:self];
99 return result;
100}
101
102- (void)_handleWindowDidBecomeKey:(NSNotification *)notification
103{
104 if (notification.object == self.window)
105 [self.inspectorWKWebViewDelegate inspectorWKWebViewDidBecomeActive:self];
106}
107
108@end
109
110#endif
