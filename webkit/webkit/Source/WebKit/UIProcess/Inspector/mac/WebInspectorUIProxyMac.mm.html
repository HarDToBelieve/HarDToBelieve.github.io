139- (void)inspectedViewFrameDidChange:(NSNotification *)notification
140{
141 // Resizing the views while inside this notification can lead to bad results when entering
142 // or exiting full screen. To avoid that we need to perform the work after a delay. We only
143 // depend on this for enforcing the height constraints, so a small delay isn't terrible. Most
144 // of the time the views will already have the correct frames because of autoresizing masks.
145
146 dispatch_after(DISPATCH_TIME_NOW, dispatch_get_main_queue(), ^{
147 if (_inspectorProxy)
148 _inspectorProxy->inspectedViewFrameDidChange();
149 });
150}

152- (void)observeValueForKeyPath:(NSString *)keyPath ofObject:(id)object change:(NSDictionary<NSKeyValueChangeKey, id> *)change context:(void *)context
153{
154 if (context != kWindowContentLayoutObserverContext) {
155 [super observeValueForKeyPath:keyPath ofObject:object change:change context:context];
156 return;
157 }
158
159 NSWindow *window = object;
160 ASSERT([window isKindOfClass:[NSWindow class]]);
161 if (window.inLiveResize)
162 return;
163
164 dispatch_after(DISPATCH_TIME_NOW, dispatch_get_main_queue(), ^{
165 if (_inspectorProxy)
166 _inspectorProxy->inspectedViewFrameDidChange();
167 });
168}

217void WebInspectorUIProxy::attachmentViewDidChange(NSView *oldView, NSView *newView)
218{
219 [[NSNotificationCenter defaultCenter] removeObserver:m_objCAdapter.get() name:NSViewFrameDidChangeNotification object:oldView];
220 [[NSNotificationCenter defaultCenter] addObserver:m_objCAdapter.get() selector:@selector(inspectedViewFrameDidChange:) name:NSViewFrameDidChangeNotification object:newView];
221
222 if (m_isAttached)
223 attach(m_attachmentSide);
224}

226void WebInspectorUIProxy::attachmentWillMoveFromWindow(NSWindow *oldWindow)
227{
228 if (m_isObservingContentLayoutRect) {
229 m_isObservingContentLayoutRect = false;
230 [oldWindow removeObserver:m_objCAdapter.get() forKeyPath:@"contentLayoutRect" context:kWindowContentLayoutObserverContext];
231 }

234void WebInspectorUIProxy::attachmentDidMoveToWindow(NSWindow *newWindow)
235{
236 if (m_isAttached && !!newWindow) {
237 m_isObservingContentLayoutRect = true;
238 [newWindow addObserver:m_objCAdapter.get() forKeyPath:@"contentLayoutRect" options:0 context:kWindowContentLayoutObserverContext];
239 inspectedViewFrameDidChange();
240 }

243void WebInspectorUIProxy::updateInspectorWindowTitle() const
244{
245 if (!m_inspectorWindow)
246 return;
247
248 unsigned level = inspectionLevel();
249 if (level > 1) {
250 NSString *debugTitle = [NSString stringWithFormat:WEB_UI_STRING("Web Inspector [%d] — %@", "Web Inspector window title when inspecting Web Inspector"), level, (NSString *)m_urlString];
251 [m_inspectorWindow setTitle:debugTitle];
252 } else {
253 NSString *title = [NSString stringWithFormat:WEB_UI_STRING("Web Inspector — %@", "Web Inspector window title"), (NSString *)m_urlString];
254 [m_inspectorWindow setTitle:title];
255 }

258RetainPtr<NSWindow> WebInspectorUIProxy::createFrontendWindow(NSRect savedWindowFrame, InspectionTargetType targetType)
259{
260 NSRect windowFrame = !NSIsEmptyRect(savedWindowFrame) ? savedWindowFrame : NSMakeRect(0, 0, initialWindowWidth, initialWindowHeight);
261 auto window = adoptNS([[_WKInspectorWindow alloc] initWithContentRect:windowFrame styleMask:windowStyleMask backing:NSBackingStoreBuffered defer:NO]);
262 [window setMinSize:NSMakeSize(minimumWindowWidth, minimumWindowHeight)];
263 [window setReleasedWhenClosed:NO];
264 [window setCollectionBehavior:([window collectionBehavior] | NSWindowCollectionBehaviorFullScreenPrimary)];
265
266 bool forRemoteTarget = targetType == InspectionTargetType::Remote;
267 [window setForRemoteTarget:forRemoteTarget];
268
269 CGFloat approximatelyHalfScreenSize = ([window screen].frame.size.width / 2) - 4;
270 CGFloat minimumFullScreenWidth = std::max<CGFloat>(636, approximatelyHalfScreenSize);
271 [window setMinFullScreenContentSize:NSMakeSize(minimumFullScreenWidth, minimumWindowHeight)];
272 [window setCollectionBehavior:([window collectionBehavior] | NSWindowCollectionBehaviorFullScreenAllowsTiling)];
273
274 [window setTitlebarAppearsTransparent:YES];
275
276 // Center the window if the saved frame was empty.
277 if (NSIsEmptyRect(savedWindowFrame))
278 [window center];
279
280 return window;
281}

283WebPageProxy* WebInspectorUIProxy::platformCreateFrontendPage()
284{
285 ASSERT(inspectedPage());
286 ASSERT(!m_inspectorPage);
287
288 m_closeFrontendAfterInactivityTimer.stop();
289
290 if (m_inspectorViewController) {
291 ASSERT(m_objCAdapter);
292 return [m_inspectorViewController webView]->_page.get();
293 }
294
295 m_objCAdapter = adoptNS([[WKWebInspectorUIProxyObjCAdapter alloc] initWithWebInspectorUIProxy:this]);
296 NSView *inspectedView = inspectedPage()->inspectorAttachmentView();
297 [[NSNotificationCenter defaultCenter] addObserver:m_objCAdapter.get() selector:@selector(inspectedViewFrameDidChange:) name:NSViewFrameDidChangeNotification object:inspectedView];
298
299 auto configuration = inspectedPage()->uiClient().configurationForLocalInspector(*inspectedPage(), *this);
300 m_inspectorViewController = adoptNS([[WKInspectorViewController alloc] initWithConfiguration: WebKit::wrapper(configuration.get()) inspectedPage:inspectedPage()]);
301 [m_inspectorViewController.get() setDelegate:m_objCAdapter.get()];
302
303 WebPageProxy *inspectorPage = [m_inspectorViewController webView]->_page.get();
304 ASSERT(inspectorPage);
305
306 return inspectorPage;
307}

309void WebInspectorUIProxy::platformCreateFrontendWindow()
310{
311 ASSERT(!m_inspectorWindow);
312
313 NSString *savedWindowFrameString = inspectedPage()->pageGroup().preferences().inspectorWindowFrame();
314 NSRect savedWindowFrame = NSRectFromString(savedWindowFrameString);
315
316 m_inspectorWindow = WebInspectorUIProxy::createFrontendWindow(savedWindowFrame, InspectionTargetType::Local);
317 [m_inspectorWindow setDelegate:m_objCAdapter.get()];
318
319 WKWebView *inspectorView = [m_inspectorViewController webView];
320 NSView *contentView = [m_inspectorWindow contentView];
321 inspectorView.frame = [contentView bounds];
322 [contentView addSubview:inspectorView];
323
324 updateInspectorWindowTitle();
325 applyForcedAppearance();
326}

395{
396 // If the Web Inspector is no longer in the same window as the inspected view,
397 // then we need to reopen the Inspector to get it attached to the right window.
398 // This can happen when dragging tabs to another window in Safari.
399 if (m_isAttached && [m_inspectorViewController webView].window != inspectedPage()->platformWindow()) {

406 [[m_inspectorViewController webView].window makeFirstResponder:[m_inspectorViewController webView]];

420bool WebInspectorUIProxy::platformCanAttach(bool webProcessCanAttach)
421{
422 NSView *inspectedView = inspectedPage()->inspectorAttachmentView();
423 if ([WKInspectorViewController viewIsInspectorWebView:inspectedView])
424 return webProcessCanAttach;
425
426 if (inspectedView.hidden)
427 return false;
428
429 static const float minimumAttachedHeight = 250;
430 static const float maximumAttachedHeightRatio = 0.75;
431 static const float minimumAttachedWidth = 500;
432
433 NSRect inspectedViewFrame = inspectedView.frame;
434
435 float maximumAttachedHeight = NSHeight(inspectedViewFrame) * maximumAttachedHeightRatio;
436 return minimumAttachedHeight <= maximumAttachedHeight && minimumAttachedWidth <= NSWidth(inspectedViewFrame);
437}

460 ASSERT(!certificateInfo.isEmpty());

477#endif
478
479 // This must be called after the trust panel has been displayed, because the certificateView doesn't exist beforehand.
480 SFCertificateView *certificateView = [certificatePanel certificateView];
481 [certificateView setDisplayTrust:YES];
482 [certificateView setEditableTrust:NO];
483 [certificateView setDisplayDetails:YES];
484 [certificateView setDetailsDisclosed:YES];

487void WebInspectorUIProxy::platformSave(const String& suggestedURL, const String& content, bool base64Encoded, bool forceSaveDialog)
488{
489 ASSERT(!suggestedURL.isEmpty());
490
491 NSURL *platformURL = m_suggestedToActualURLMap.get(suggestedURL).get();
492 if (!platformURL) {
493 platformURL = [NSURL URLWithString:suggestedURL];
494 // The user must confirm new filenames before we can save to them.
495 forceSaveDialog = true;
496 }
497
498 ASSERT(platformURL);
499 if (!platformURL)
500 return;
501
502 // Necessary for the block below.
503 String suggestedURLCopy = suggestedURL;
504 String contentCopy = content;
505
506 auto saveToURL = ^(NSURL *actualURL) {
507 ASSERT(actualURL);
508
509 m_suggestedToActualURLMap.set(suggestedURLCopy, actualURL);
510
511 if (base64Encoded) {
512 auto decodedData = base64Decode(contentCopy, Base64DecodeOptions::ValidatePadding);
513 if (!decodedData)
514 return;
515 auto dataContent = adoptNS([[NSData alloc] initWithBytes:decodedData->data() length:decodedData->size()]);
516 [dataContent writeToURL:actualURL atomically:YES];
517 } else
518 [contentCopy writeToURL:actualURL atomically:YES encoding:NSUTF8StringEncoding error:NULL];
519
520 m_inspectorPage->send(Messages::WebInspectorUI::DidSave([actualURL absoluteString]));
521 };
522
523 if (!forceSaveDialog) {
524 saveToURL(platformURL);
525 return;
526 }
527
528 NSSavePanel *panel = [NSSavePanel savePanel];
529 panel.nameFieldStringValue = platformURL.lastPathComponent;
530
531 // If we have a file URL we've already saved this file to a path and
532 // can provide a good directory to show. Otherwise, use the system's
533 // default behavior for the initial directory to show in the dialog.
534 if (platformURL.isFileURL)
535 panel.directoryURL = [platformURL URLByDeletingLastPathComponent];
536
537 auto completionHandler = ^(NSInteger result) {
538 if (result == NSModalResponseCancel)
539 return;
540 ASSERT(result == NSModalResponseOK);
541 saveToURL(panel.URL);
542 };
543
544 NSWindow *window = m_inspectorWindow ? m_inspectorWindow.get() : [NSApp keyWindow];
545 if (window)
546 [panel beginSheetModalForWindow:window completionHandler:completionHandler];
547 else
548 completionHandler([panel runModal]);
549}

551void WebInspectorUIProxy::platformAppend(const String& suggestedURL, const String& content)
552{
553 ASSERT(!suggestedURL.isEmpty());
554
555 RetainPtr<NSURL> actualURL = m_suggestedToActualURLMap.get(suggestedURL);
556 // Do not append unless the user has already confirmed this filename in save().
557 if (!actualURL)
558 return;
559
560 NSFileHandle *handle = [NSFileHandle fileHandleForWritingToURL:actualURL.get() error:NULL];
561 [handle seekToEndOfFile];
562 [handle writeData:[content dataUsingEncoding:NSUTF8StringEncoding]];
563 [handle closeFile];
564
565 m_inspectorPage->send(Messages::WebInspectorUI::DidAppend([actualURL absoluteString]));
566}

593void WebInspectorUIProxy::inspectedViewFrameDidChange(CGFloat currentDimension)
594{
595 if (!m_isVisible)
596 return;
597
598 if (!m_isAttached) {
599 // Check if the attach availability changed. We need to do this here in case
600 // the attachment view is not the WKView.
601 attachAvailabilityChanged(platformCanAttach(canAttach()));
602 return;
603 }
604
605 if (!inspectedPage())
606 return;
607
608 NSView *inspectedView = inspectedPage()->inspectorAttachmentView();
609 WKWebView *inspectorView = [m_inspectorViewController webView];
610
611 NSRect inspectedViewFrame = inspectedView.frame;
612 NSRect oldInspectorViewFrame = inspectorView.frame;
613 NSRect newInspectorViewFrame = NSZeroRect;
614 NSRect parentBounds = inspectedView.superview.bounds;
615 CGFloat inspectedViewTop = NSMaxY(inspectedViewFrame);
616
617 switch (m_attachmentSide) {
618 case AttachmentSide::Bottom: {
619 if (!currentDimension)
620 currentDimension = NSHeight(oldInspectorViewFrame);
621
622 CGFloat parentHeight = NSHeight(parentBounds);
623 CGFloat inspectorHeight = InspectorFrontendClientLocal::constrainedAttachedWindowHeight(currentDimension, parentHeight);
624
625 // Preserve the top position of the inspected view so banners in Safari still work.
626 inspectedViewFrame = NSMakeRect(0, inspectorHeight, NSWidth(parentBounds), inspectedViewTop - inspectorHeight);
627 newInspectorViewFrame = NSMakeRect(0, 0, NSWidth(inspectedViewFrame), inspectorHeight);
628 break;
629 }
630
631 case AttachmentSide::Right: {
632 if (!currentDimension)
633 currentDimension = NSWidth(oldInspectorViewFrame);
634
635 CGFloat parentWidth = NSWidth(parentBounds);
636 CGFloat inspectorWidth = InspectorFrontendClientLocal::constrainedAttachedWindowWidth(currentDimension, parentWidth);
637
638 // Preserve the top position of the inspected view so banners in Safari still work. But don't use that
639 // top position for the inspector view since the banners only stretch as wide as the inspected view.
640 inspectedViewFrame = NSMakeRect(0, 0, parentWidth - inspectorWidth, inspectedViewTop);
641 newInspectorViewFrame = NSMakeRect(parentWidth - inspectorWidth, 0, inspectorWidth, NSHeight(parentBounds));
642
643 if (NSWindow *inspectorWindow = inspectorView.window) {
644 NSRect contentLayoutRect = [inspectedView.superview convertRect:inspectorWindow.contentLayoutRect fromView:nil];
645 newInspectorViewFrame = NSIntersectionRect(newInspectorViewFrame, contentLayoutRect);
646 }
647 break;
648 }
649
650 case AttachmentSide::Left: {
651 if (!currentDimension)
652 currentDimension = NSWidth(oldInspectorViewFrame);
653
654 CGFloat parentWidth = NSWidth(parentBounds);
655 CGFloat inspectorWidth = InspectorFrontendClientLocal::constrainedAttachedWindowWidth(currentDimension, parentWidth);
656
657 // Preserve the top position of the inspected view so banners in Safari still work. But don't use that
658 // top position for the inspector view since the banners only stretch as wide as the inspected view.
659 inspectedViewFrame = NSMakeRect(inspectorWidth, 0, parentWidth - inspectorWidth, inspectedViewTop);
660 newInspectorViewFrame = NSMakeRect(0, 0, inspectorWidth, NSHeight(parentBounds));
661
662 if (NSWindow *inspectorWindow = inspectorView.window) {
663 NSRect contentLayoutRect = [inspectedView.superview convertRect:inspectorWindow.contentLayoutRect fromView:nil];
664 newInspectorViewFrame = NSIntersectionRect(newInspectorViewFrame, contentLayoutRect);
665 }
666 break;
667 }
668 }
669
670 if (NSEqualRects(oldInspectorViewFrame, newInspectorViewFrame) && NSEqualRects([inspectedView frame], inspectedViewFrame))
671 return;
672
673 // Disable screen updates to make sure the layers for both views resize in sync.
674 [inspectorView.window disableScreenUpdatesUntilFlush];
675
676 [inspectorView setFrame:newInspectorViewFrame];
677 [inspectedView setFrame:inspectedViewFrame];
678}

694void WebInspectorUIProxy::platformAttach()
695{
696 NSView *inspectedView = inspectedPage()->inspectorAttachmentView();
697 WKWebView *inspectorView = [m_inspectorViewController webView];
698
699 if (m_inspectorWindow) {
700 [m_inspectorWindow setDelegate:nil];
701 [m_inspectorWindow close];
702 m_inspectorWindow = nil;
703 }
704
705 [inspectorView removeFromSuperview];
706
707 CGFloat currentDimension;
708 switch (m_attachmentSide) {
709 case AttachmentSide::Bottom:
710 [inspectorView setAutoresizingMask:NSViewWidthSizable | NSViewMaxYMargin];
711 currentDimension = inspectorPagePreferences().inspectorAttachedHeight();
712 break;
713 case AttachmentSide::Right:
714 [inspectorView setAutoresizingMask:NSViewHeightSizable | NSViewMinXMargin];
715 currentDimension = inspectorPagePreferences().inspectorAttachedWidth();
716 break;
717 case AttachmentSide::Left:
718 [inspectorView setAutoresizingMask:NSViewHeightSizable | NSViewMaxXMargin];
719 currentDimension = inspectorPagePreferences().inspectorAttachedWidth();
720 break;
721 }
722
723 inspectedViewFrameDidChange(currentDimension);
724
725 [inspectedView.superview addSubview:inspectorView positioned:NSWindowBelow relativeTo:inspectedView];
726 [inspectorView.window makeFirstResponder:inspectorView];
727}

729void WebInspectorUIProxy::platformDetach()
730{
731 NSView *inspectedView = inspectedPage()->inspectorAttachmentView();
732 WKWebView *inspectorView = [m_inspectorViewController webView];
733
734 [inspectorView removeFromSuperview];
735
736 [inspectorView setAutoresizingMask:NSViewWidthSizable | NSViewHeightSizable];
737
738 // Make sure that we size the inspected view's frame after detaching so that it takes up the space that the
739 // attached inspector used to. Preserve the top position of the inspected view so banners in Safari still work.
740
741 inspectedView.frame = NSMakeRect(0, 0, NSWidth(inspectedView.superview.bounds), NSMaxY(inspectedView.frame));
742
743 // Return early if we are not visible. This means the inspector was closed while attached
744 // and we should not create and show the inspector window.
745 if (!m_isVisible)
746 return;
747
748 open();
749}

813{

830 window.appearance = platformAppearance;

832 [m_inspectorViewController webView].appearance = platformAppearance;
