46namespace WebKit {
47
48#if PLATFORM(GTK) || PLATFORM(WPE)
49enum class SandboxPermission {
50 ReadOnly,
51 ReadWrite,
52};
53#endif
54
55class ProcessLauncher : public ThreadSafeRefCounted<ProcessLauncher>, public CanMakeWeakPtr<ProcessLauncher> {
56public:
57 class Client {
58 public:
59 virtual ~Client() { }
60
61 virtual void didFinishLaunching(ProcessLauncher*, IPC::Connection::Identifier) = 0;
62 virtual bool shouldConfigureJSCForTesting() const { return false; }
63 virtual bool isJITEnabled() const { return true; }
64#if PLATFORM(COCOA)
65 virtual RefPtr<XPCEventHandler> xpcEventHandler() const { return nullptr; }
66#endif
67 };
68
69 enum class ProcessType {
70 Web,
71#if ENABLE(NETSCAPE_PLUGIN_API)
72 Plugin,
73#endif
74 Network,
75#if ENABLE(GPU_PROCESS)
76 GPU,
77#endif
78#if ENABLE(WEB_AUTHN)
79 WebAuthn,
80#endif
81#if ENABLE(BUBBLEWRAP_SANDBOX)
82 DBusProxy,
83#endif
84 };
85
86 struct LaunchOptions {
87 ProcessType processType;
88 WebCore::ProcessIdentifier processIdentifier;
89 HashMap<String, String> extraInitializationData;
90 bool nonValidInjectedCodeAllowed { false };
91 bool shouldMakeProcessLaunchFailForTesting { false };
92 CString customWebContentServiceBundleIdentifier;
93
94#if PLATFORM(GTK) || PLATFORM(WPE)
95 HashMap<CString, SandboxPermission> extraWebProcessSandboxPaths;
96#if ENABLE(DEVELOPER_MODE)
97 String processCmdPrefix;
98#endif
99#endif
100
101#if PLATFORM(PLAYSTATION)
102 String processPath;
103 int32_t userId { -1 };
104#endif
105 };
106
107 static Ref<ProcessLauncher> create(Client* client, LaunchOptions&& launchOptions)
108 {
109 return adoptRef(*new ProcessLauncher(client, WTFMove(launchOptions)));
110 }
111
112 bool isLaunching() const { return m_isLaunching; }
113 ProcessID processIdentifier() const { return m_processIdentifier; }
114
115 void terminateProcess();
116 void invalidate();
117
118private:
119 ProcessLauncher(Client*, LaunchOptions&&);
120
121 void launchProcess();
122 void didFinishLaunchingProcess(ProcessID, IPC::Connection::Identifier);
123
124 void platformInvalidate();
125
126 Client* m_client;
127
128#if PLATFORM(COCOA)
129 OSObjectPtr<xpc_connection_t> m_xpcConnection;
130#endif
131
132#if PLATFORM(WIN)
133 WTF::Win32Handle m_hProcess;
134#endif
135
136 const LaunchOptions m_launchOptions;
137 bool m_isLaunching { true };
138 ProcessID m_processIdentifier { 0 };
139};
