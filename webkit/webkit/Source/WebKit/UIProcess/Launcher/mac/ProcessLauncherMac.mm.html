56static const char* serviceName(const ProcessLauncher::LaunchOptions& launchOptions)
57{
58 switch (launchOptions.processType) {
59 case ProcessLauncher::ProcessType::Web:
60 return launchOptions.nonValidInjectedCodeAllowed ? "com.apple.WebKit.WebContent.Development" : "com.apple.WebKit.WebContent";
61 case ProcessLauncher::ProcessType::Network:
62 return "com.apple.WebKit.Networking";
63#if ENABLE(GPU_PROCESS)
64 case ProcessLauncher::ProcessType::GPU:
65 return "com.apple.WebKit.GPU";
66#endif
67#if ENABLE(WEB_AUTHN)
68 case ProcessLauncher::ProcessType::WebAuthn:
69 return "com.apple.WebKit.WebAuthn";
70#endif
71#if ENABLE(NETSCAPE_PLUGIN_API)
72 case ProcessLauncher::ProcessType::Plugin:
73 return "com.apple.WebKit.Plugin.64";
74#endif
75 }
76}

78static bool shouldLeakBoost(const ProcessLauncher::LaunchOptions& launchOptions)
79{
80#if PLATFORM(IOS_FAMILY)
81 UNUSED_PARAM(launchOptions);
82#if HAVE(RUNNINGBOARD_WEBKIT_PRIORITY_SUPPORT)
83 // On iOS, we don't need to leak a boost message when RunningBoard process assertions give us the
84 // right priorities.
85 static const bool runningBoardHandlesPriorities = isFeatureFlagEnabled("RB_full_manage_WK_jetsam"_s);
86 return !runningBoardHandlesPriorities;
87#else
88 return true;
89#endif // HAVE(RUNNINGBOARD_WEBKIT_PRIORITY_SUPPORT)
90#else
91 // On Mac, leak a boost onto the NetworkProcess, GPUProcess, and WebAuthnProcess.
92#if ENABLE(GPU_PROCESS)
93 if (launchOptions.processType == ProcessLauncher::ProcessType::GPU)
94 return true;
95#endif
96#if ENABLE(WEB_AUTHN)
97 if (launchOptions.processType == ProcessLauncher::ProcessType::WebAuthn)
98 return true;
99#endif
100 return launchOptions.processType == ProcessLauncher::ProcessType::Network;
101#endif
102}

104static NSString *systemDirectoryPath()
105{
106 static NeverDestroyed<RetainPtr<NSString>> path = adoptNS([^{
107#if PLATFORM(IOS_FAMILY_SIMULATOR)
108 char *simulatorRoot = getenv("SIMULATOR_ROOT");
109 return simulatorRoot ? [NSString stringWithFormat:@"%s/System/", simulatorRoot] : @"/System/";
110#else
111 return @"/System/";
112#endif
113 }() copy]);
114
115 return path.get().get();
116}

118void ProcessLauncher::launchProcess()
119{
120 ASSERT(!m_xpcConnection);
121
122 const char* name;
123 if (!m_launchOptions.customWebContentServiceBundleIdentifier.isNull())
124 name = m_launchOptions.customWebContentServiceBundleIdentifier.data();
125 else
126 name = serviceName(m_launchOptions);
127
128 m_xpcConnection = adoptOSObject(xpc_connection_create(name, nullptr));
129
130 uuid_t uuid;
131 uuid_generate(uuid);
132 xpc_connection_set_oneshot_instance(m_xpcConnection.get(), uuid);
133
134 // Inherit UI process localization. It can be different from child process default localization:
135 // 1. When the application and system frameworks simply have different localized resources available, we should match the application.
136 // 1.1. An important case is WebKitTestRunner, where we should use English localizations for all system frameworks.
137 // 2. When AppleLanguages is passed as command line argument for UI process, or set in its preferences, we should respect it in child processes.
138 auto initializationMessage = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
139 _CFBundleSetupXPCBootstrap(initializationMessage.get());
140#if PLATFORM(IOS_FAMILY)
141 // Clients that set these environment variables explicitly do not have the values automatically forwarded by libxpc.
142 auto containerEnvironmentVariables = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
143 if (const char* environmentHOME = getenv("HOME"))
144 xpc_dictionary_set_string(containerEnvironmentVariables.get(), "HOME", environmentHOME);
145 if (const char* environmentCFFIXED_USER_HOME = getenv("CFFIXED_USER_HOME"))
146 xpc_dictionary_set_string(containerEnvironmentVariables.get(), "CFFIXED_USER_HOME", environmentCFFIXED_USER_HOME);
147 if (const char* environmentTMPDIR = getenv("TMPDIR"))
148 xpc_dictionary_set_string(containerEnvironmentVariables.get(), "TMPDIR", environmentTMPDIR);
149 xpc_dictionary_set_value(initializationMessage.get(), "ContainerEnvironmentVariables", containerEnvironmentVariables.get());
150#endif
151
152 auto languagesIterator = m_launchOptions.extraInitializationData.find("OverrideLanguages");
153 if (languagesIterator != m_launchOptions.extraInitializationData.end()) {
154 auto languages = adoptOSObject(xpc_array_create(nullptr, 0));
155 for (auto& language : languagesIterator->value.split(','))
156 xpc_array_set_string(languages.get(), XPC_ARRAY_APPEND, language.utf8().data());
157 xpc_dictionary_set_value(initializationMessage.get(), "OverrideLanguages", languages.get());
158 }
159
160#if PLATFORM(MAC)
161 xpc_dictionary_set_string(initializationMessage.get(), "WebKitBundleVersion", [[NSBundle bundleWithIdentifier:@"com.apple.WebKit"].infoDictionary[(__bridge NSString *)kCFBundleVersionKey] UTF8String]);
162#endif
163 xpc_connection_set_bootstrap(m_xpcConnection.get(), initializationMessage.get());
164
165 if (shouldLeakBoost(m_launchOptions)) {
166 auto preBootstrapMessage = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
167 xpc_dictionary_set_string(preBootstrapMessage.get(), "message-name", "pre-bootstrap");
168 xpc_connection_send_message(m_xpcConnection.get(), preBootstrapMessage.get());
169 }
170
171 // Create the listening port.
172 mach_port_t listeningPort = MACH_PORT_NULL;
173 auto kr = mach_port_allocate(mach_task_self(), MACH_PORT_RIGHT_RECEIVE, &listeningPort);
174 if (kr != KERN_SUCCESS) {
175 LOG_ERROR("Could not allocate mach port, error %x: %s", kr, mach_error_string(kr));
176 CRASH();
177 }
178
179 // Insert a send right so we can send to it.
180 mach_port_insert_right(mach_task_self(), listeningPort, listeningPort, MACH_MSG_TYPE_MAKE_SEND);
181
182 mach_port_t previousNotificationPort = MACH_PORT_NULL;
183 auto mc = mach_port_request_notification(mach_task_self(), listeningPort, MACH_NOTIFY_NO_SENDERS, 0, listeningPort, MACH_MSG_TYPE_MAKE_SEND_ONCE, &previousNotificationPort);
184 ASSERT(!previousNotificationPort);
185 ASSERT(mc == KERN_SUCCESS);
186 if (mc != KERN_SUCCESS) {
187 // If mach_port_request_notification fails, 'previousNotificationPort' will be uninitialized.
188 LOG_ERROR("mach_port_request_notification failed: (%x) %s", mc, mach_error_string(mc));
189 }
190
191 String clientIdentifier;
192#if PLATFORM(MAC)
193 clientIdentifier = codeSigningIdentifierForCurrentProcess();
194#endif
195 if (clientIdentifier.isNull())
196 clientIdentifier = [[NSBundle mainBundle] bundleIdentifier];
197
198 // FIXME: Switch to xpc_connection_set_bootstrap once it's available everywhere we need.
199 auto bootstrapMessage = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
200
201 if (m_client) {
202 if (m_client->shouldConfigureJSCForTesting())
203 xpc_dictionary_set_bool(bootstrapMessage.get(), "configure-jsc-for-testing", true);
204 if (!m_client->isJITEnabled())
205 xpc_dictionary_set_bool(bootstrapMessage.get(), "disable-jit", true);
206 }
207
208 xpc_dictionary_set_string(bootstrapMessage.get(), "message-name", "bootstrap");
209
210 xpc_dictionary_set_mach_send(bootstrapMessage.get(), "server-port", listeningPort);
211
212 xpc_dictionary_set_string(bootstrapMessage.get(), "client-identifier", !clientIdentifier.isEmpty() ? clientIdentifier.utf8().data() : *_NSGetProgname());
213 xpc_dictionary_set_string(bootstrapMessage.get(), "client-bundle-identifier", WebCore::applicationBundleIdentifier().utf8().data());
214 xpc_dictionary_set_string(bootstrapMessage.get(), "client-sdk-version", String::number(applicationSDKVersion()).utf8().data());
215 xpc_dictionary_set_string(bootstrapMessage.get(), "process-identifier", String::number(m_launchOptions.processIdentifier.toUInt64()).utf8().data());
216 xpc_dictionary_set_string(bootstrapMessage.get(), "ui-process-name", [[[NSProcessInfo processInfo] processName] UTF8String]);
217 xpc_dictionary_set_string(bootstrapMessage.get(), "service-name", name);
218
219 bool isWebKitDevelopmentBuild = ![[[[NSBundle bundleWithIdentifier:@"com.apple.WebKit"] bundlePath] stringByDeletingLastPathComponent] hasPrefix:systemDirectoryPath()];
220 if (isWebKitDevelopmentBuild) {
221 xpc_dictionary_set_fd(bootstrapMessage.get(), "stdout", STDOUT_FILENO);
222 xpc_dictionary_set_fd(bootstrapMessage.get(), "stderr", STDERR_FILENO);
223 }
224
225 auto extraInitializationData = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
226
227 for (const auto& keyValuePair : m_launchOptions.extraInitializationData)
228 xpc_dictionary_set_string(extraInitializationData.get(), keyValuePair.key.utf8().data(), keyValuePair.value.utf8().data());
229
230 xpc_dictionary_set_value(bootstrapMessage.get(), "extra-initialization-data", extraInitializationData.get());
231
232 auto errorHandlerImpl = [weakProcessLauncher = makeWeakPtr(*this), listeningPort] (xpc_object_t event) {
233 ASSERT(!event || xpc_get_type(event) == XPC_TYPE_ERROR);
234
235 auto processLauncher = weakProcessLauncher.get();
236 if (!processLauncher)
237 return;
238
239 if (!processLauncher->isLaunching())
240 return;
241
242#if ASSERT_ENABLED
243 mach_port_urefs_t sendRightCount = 0;
244 mach_port_get_refs(mach_task_self(), listeningPort, MACH_PORT_RIGHT_SEND, &sendRightCount);
245 ASSERT(sendRightCount >= 1);
246#endif
247
248 // We failed to launch. Release the send right.
249 deallocateSendRightSafely(listeningPort);
250
251 // And the receive right.
252 mach_port_mod_refs(mach_task_self(), listeningPort, MACH_PORT_RIGHT_RECEIVE, -1);
253
254 if (processLauncher->m_xpcConnection)
255 xpc_connection_cancel(processLauncher->m_xpcConnection.get());
256 processLauncher->m_xpcConnection = nullptr;
257
258 processLauncher->didFinishLaunchingProcess(0, IPC::Connection::Identifier());
259 };
260
261 auto eventHandler = [errorHandlerImpl = WTFMove(errorHandlerImpl), eventHandler = m_client->xpcEventHandler()] (xpc_object_t event) mutable {
262
263 if (!event || xpc_get_type(event) == XPC_TYPE_ERROR) {
264 RunLoop::main().dispatch([errorHandlerImpl = WTFMove(errorHandlerImpl), event = OSObjectPtr(event)] {
265 errorHandlerImpl(event.get());
266 });
267 return;
268 }
269
270 if (eventHandler) {
271 RunLoop::main().dispatch([eventHandler = eventHandler, event = OSObjectPtr(event)] {
272 eventHandler->handleXPCEvent(event.get());
273 });
274 }
275 };
276
277 xpc_connection_set_event_handler(m_xpcConnection.get(), eventHandler);
278
279 xpc_connection_resume(m_xpcConnection.get());
280
281 if (UNLIKELY(m_launchOptions.shouldMakeProcessLaunchFailForTesting)) {
282 eventHandler(nullptr);
283 return;
284 }
285
286 ref();
287 xpc_connection_send_message_with_reply(m_xpcConnection.get(), bootstrapMessage.get(), dispatch_get_main_queue(), ^(xpc_object_t reply) {
288 // Errors are handled in the event handler.
289 // It is possible for this block to be called after the error event handler, in which case we're no longer
290 // launching and we already took care of cleaning things up.
291 if (isLaunching() && xpc_get_type(reply) != XPC_TYPE_ERROR) {
292 ASSERT(xpc_get_type(reply) == XPC_TYPE_DICTIONARY);
293 ASSERT(!strcmp(xpc_dictionary_get_string(reply, "message-name"), "process-finished-launching"));
294
295#if ASSERT_ENABLED
296 mach_port_urefs_t sendRightCount = 0;
297 mach_port_get_refs(mach_task_self(), listeningPort, MACH_PORT_RIGHT_SEND, &sendRightCount);
298 ASSERT(sendRightCount >= 1);
299#endif
300
301 deallocateSendRightSafely(listeningPort);
302
303 if (!m_xpcConnection) {
304 // The process was terminated.
305 didFinishLaunchingProcess(0, IPC::Connection::Identifier());
306 return;
307 }
308
309 // The process has finished launching, grab the pid from the connection.
310 pid_t processIdentifier = xpc_connection_get_pid(m_xpcConnection.get());
311
312 didFinishLaunchingProcess(processIdentifier, IPC::Connection::Identifier(listeningPort, m_xpcConnection));
313 m_xpcConnection = nullptr;
314 }
315
316 deref();
317 });
318}

320void ProcessLauncher::terminateProcess()
321{
322 if (m_isLaunching) {
323 invalidate();
324 return;
325 }
326
327 if (!m_processIdentifier)
328 return;
329
330 kill(m_processIdentifier, SIGKILL);
331 m_processIdentifier = 0;
332}
333
334void ProcessLauncher::platformInvalidate()
335{
336 if (!m_xpcConnection)
337 return;
338
339 xpc_connection_cancel(m_xpcConnection.get());
340 terminateWithReason(m_xpcConnection.get(), WebKit::ReasonCode::Invalidation, "ProcessLauncher::platformInvalidate");
341 m_xpcConnection = nullptr;
342}

344void terminateWithReason(xpc_connection_t connection, ReasonCode, const char*)
345{
346 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
347 xpc_connection_kill(connection, SIGKILL);
348 ALLOW_DEPRECATED_DECLARATIONS_END
349}
