39namespace WebKit {
40
41class WebProcessProxy;
42
43class AudioSessionRoutingArbitratorProxy
44 : public IPC::MessageReceiver {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 AudioSessionRoutingArbitratorProxy(WebProcessProxy&);
48 virtual ~AudioSessionRoutingArbitratorProxy();
49
50 void processDidTerminate();
51 WebCore::AudioSession::CategoryType category() const { return m_category; }
52
53 static uint64_t destinationId() { return 1; }
54
55 using RoutingArbitrationError = WebCore::AudioSessionRoutingArbitrationClient::RoutingArbitrationError;
56 using DefaultRouteChanged = WebCore::AudioSessionRoutingArbitrationClient::DefaultRouteChanged;
57 using ArbitrationCallback = WebCore::AudioSessionRoutingArbitrationClient::ArbitrationCallback;
58
59 enum class ArbitrationStatus : uint8_t {
60 None,
61 Pending,
62 Active,
63 };
64
65 ArbitrationStatus arbitrationStatus() const { return m_arbitrationStatus; }
66 WallTime arbitrationUpdateTime() const { return m_arbitrationUpdateTime; }
67
68private:
69 // IPC::MessageReceiver
70 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) final;
71
72 // Messages
73 void beginRoutingArbitrationWithCategory(WebCore::AudioSession::CategoryType, ArbitrationCallback&&);
74 void endRoutingArbitration();
75
76 WebProcessProxy& m_process;
77 WebCore::AudioSession::CategoryType m_category { WebCore::AudioSession::CategoryType::None };
78 ArbitrationStatus m_arbitrationStatus { ArbitrationStatus::None };
79 WallTime m_arbitrationUpdateTime;
80
81#if HAVE(AVAUDIO_ROUTING_ARBITER)
82 UniqueRef<WebCore::SharedRoutingArbitrator::Token> m_token;
83#endif
84};
