63MediaKeySystemPermissionRequestManagerProxy::MediaKeySystemPermissionRequestManagerProxy(WebPageProxy& page)
64 : m_page(page)
65#if !RELEASE_LOG_DISABLED
66 , m_logIdentifier(LoggerHelper::uniqueLogIdentifier())
67#endif
68{
69}
70
71MediaKeySystemPermissionRequestManagerProxy::~MediaKeySystemPermissionRequestManagerProxy()
72{
73 invalidatePendingRequests();
74}
75
76void MediaKeySystemPermissionRequestManagerProxy::invalidatePendingRequests()
77{
78 for (auto& request : m_pendingRequests.values())
79 request->invalidate();
80
81 m_pendingRequests.clear();
82}
83
84void MediaKeySystemPermissionRequestManagerProxy::denyRequest(MediaKeySystemPermissionRequestProxy& request, const String& message)
85{
86 if (!m_page.hasRunningProcess())
87 return;
88
89 ALWAYS_LOG(LOGIDENTIFIER, request.mediaKeySystemID(), ", reason: ", message);
90
91#if ENABLE(ENCRYPTED_MEDIA)
92 m_page.send(Messages::WebPage::MediaKeySystemWasDenied(request.mediaKeySystemID(), message));
93#else
94 UNUSED_PARAM(message);
95#endif
96}
97
98void MediaKeySystemPermissionRequestManagerProxy::grantRequest(MediaKeySystemPermissionRequestProxy& request)
99{
100 if (!m_page.hasRunningProcess())
101 return;
102
103#if ENABLE(ENCRYPTED_MEDIA)
104 ALWAYS_LOG(LOGIDENTIFIER, request.mediaKeySystemID(), ", keySystem: ", request.keySystem());
105
106 m_page.sendWithAsyncReply(Messages::WebPage::MediaKeySystemWasGranted { request.mediaKeySystemID() }, [] { });
107#else
108 UNUSED_PARAM(request);
109#endif
110}
111
112Ref<MediaKeySystemPermissionRequestProxy> MediaKeySystemPermissionRequestManagerProxy::createRequestForFrame(uint64_t mediaKeySystemID, FrameIdentifier frameID, Ref<SecurityOrigin>&& topLevelDocumentOrigin, const String& keySystem)
113{
114 ALWAYS_LOG(LOGIDENTIFIER, mediaKeySystemID);
115
116 auto request = MediaKeySystemPermissionRequestProxy::create(*this, mediaKeySystemID, m_page.mainFrame()->frameID(), frameID, WTFMove(topLevelDocumentOrigin), keySystem);
117 m_pendingRequests.add(mediaKeySystemID, request.ptr());
118 return request;
119}
