34namespace WebKit {
35using namespace WebCore;
36
37MediaKeySystemPermissionRequestProxy::MediaKeySystemPermissionRequestProxy(MediaKeySystemPermissionRequestManagerProxy& manager, uint64_t mediaKeySystemID, FrameIdentifier mainFrameID, FrameIdentifier frameID, Ref<WebCore::SecurityOrigin>&& topLevelDocumentOrigin, const String& keySystem)
38 : m_manager(makeWeakPtr(manager))
39 , m_mediaKeySystemID(mediaKeySystemID)
40 , m_mainFrameID(mainFrameID)
41 , m_frameID(frameID)
42 , m_topLevelDocumentSecurityOrigin(WTFMove(topLevelDocumentOrigin))
43 , m_keySystem(keySystem)
44{
45}
46
47void MediaKeySystemPermissionRequestProxy::allow()
48{
49 ASSERT(m_manager);
50 if (!m_manager)
51 return;
52
53 m_manager->grantRequest(*this);
54 invalidate();
55}
56
57void MediaKeySystemPermissionRequestProxy::deny()
58{
59 if (!m_manager)
60 return;
61
62 m_manager->denyRequest(*this);
63 invalidate();
64}
65
66void MediaKeySystemPermissionRequestProxy::invalidate()
67{
68 m_manager = nullptr;
69}
70
71void MediaKeySystemPermissionRequestProxy::doDefaultAction()
72{
73#if PLATFORM(COCOA)
74 // Backward compatibility for platforms not yet supporting the MediaKeySystem permission request
75 // in their APIUIClient.
76 allow();
77#else
78 deny();
79#endif
80}
