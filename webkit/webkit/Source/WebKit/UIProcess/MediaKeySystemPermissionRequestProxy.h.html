34namespace WebCore {
35class SecurityOrigin;
36}
37
38namespace WebKit {
39
40class MediaKeySystemPermissionRequestManagerProxy;
41
42class MediaKeySystemPermissionRequestProxy : public RefCounted<MediaKeySystemPermissionRequestProxy> {
43public:
44 static Ref<MediaKeySystemPermissionRequestProxy> create(MediaKeySystemPermissionRequestManagerProxy& manager, uint64_t mediaKeySystemID, WebCore::FrameIdentifier mainFrameID, WebCore::FrameIdentifier frameID, Ref<WebCore::SecurityOrigin>&& topLevelDocumentOrigin, const String& keySystem)
45 {
46 return adoptRef(*new MediaKeySystemPermissionRequestProxy(manager, mediaKeySystemID, mainFrameID, frameID, WTFMove(topLevelDocumentOrigin), keySystem));
47 }
48
49 void allow();
50 void deny();
51
52 void invalidate();
53
54 uint64_t mediaKeySystemID() const { return m_mediaKeySystemID; }
55 WebCore::FrameIdentifier mainFrameID() const { return m_mainFrameID; }
56 WebCore::FrameIdentifier frameID() const { return m_frameID; }
57
58 WebCore::SecurityOrigin& topLevelDocumentSecurityOrigin() { return m_topLevelDocumentSecurityOrigin.get(); }
59 const WebCore::SecurityOrigin& topLevelDocumentSecurityOrigin() const { return m_topLevelDocumentSecurityOrigin.get(); }
60
61 const String& keySystem() const { return m_keySystem; }
62
63 void doDefaultAction();
64
65private:
66 MediaKeySystemPermissionRequestProxy(MediaKeySystemPermissionRequestManagerProxy&, uint64_t mediaKeySystemID, WebCore::FrameIdentifier mainFrameID, WebCore::FrameIdentifier, Ref<WebCore::SecurityOrigin>&& topLevelDocumentOrigin, const String& keySystem);
67
68 WeakPtr<MediaKeySystemPermissionRequestManagerProxy> m_manager;
69 uint64_t m_mediaKeySystemID;
70 WebCore::FrameIdentifier m_mainFrameID;
71 WebCore::FrameIdentifier m_frameID;
72 Ref<WebCore::SecurityOrigin> m_topLevelDocumentSecurityOrigin;
73 String m_keySystem;
74};
