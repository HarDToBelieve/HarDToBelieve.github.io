131void NetworkProcessProxy::sendCreationParametersToNewProcess()
132{
133 ASSERT(RunLoop::isMain());
134
135 // FIXME: This is a temporary workaround for apps using WebKit API on non-main threads.
136 // We should remove this once we enforce threading violation check on our APIs.
137 // https://bugs.webkit.org/show_bug.cgi?id=200246.
138 if (!RunLoop::isMain()) {
139 callOnMainRunLoopAndWait([this] {
140 sendCreationParametersToNewProcess();
141 });
142 }
143
144 NetworkProcessCreationParameters parameters;
145 parameters.urlSchemesRegisteredAsSecure = copyToVector(LegacyGlobalSettings::singleton().schemesToRegisterAsSecure());
146 parameters.urlSchemesRegisteredAsBypassingContentSecurityPolicy = copyToVector(LegacyGlobalSettings::singleton().schemesToRegisterAsBypassingContentSecurityPolicy());
147 parameters.urlSchemesRegisteredAsLocal = copyToVector(LegacyGlobalSettings::singleton().schemesToRegisterAsLocal());
148 parameters.urlSchemesRegisteredAsNoAccess = copyToVector(LegacyGlobalSettings::singleton().schemesToRegisterAsNoAccess());
149 parameters.cacheModel = LegacyGlobalSettings::singleton().cacheModel();
150 for (auto& scheme : WebProcessPool::urlSchemesWithCustomProtocolHandlers())
151 parameters.urlSchemesRegisteredForCustomProtocols.append(scheme);
152#if PLATFORM(IOS_FAMILY)
153 if (String cookieStorageDirectory = WebProcessPool::cookieStorageDirectory(); !cookieStorageDirectory.isEmpty())
154 SandboxExtension::createHandleForReadWriteDirectory(cookieStorageDirectory, parameters.cookieStorageDirectoryExtensionHandle);
155 if (String containerCachesDirectory = WebProcessPool::networkingCachesDirectory(); !containerCachesDirectory.isEmpty())
156 SandboxExtension::createHandleForReadWriteDirectory(containerCachesDirectory, parameters.containerCachesDirectoryExtensionHandle);
157 if (String parentBundleDirectory = WebProcessPool::parentBundleDirectory(); !parentBundleDirectory.isEmpty())
158 SandboxExtension::createHandle(parentBundleDirectory, SandboxExtension::Type::ReadOnly, parameters.parentBundleDirectoryExtensionHandle);
159 SandboxExtension::createHandleForTemporaryFile(emptyString(), SandboxExtension::Type::ReadWrite, parameters.tempDirectoryExtensionHandle);
160#endif
161
162#if !PLATFORM(GTK) && !PLATFORM(WPE) // GTK and WPE don't use defaultNetworkProcess
163 parameters.websiteDataStoreParameters = WebsiteDataStore::parametersFromEachWebsiteDataStore();
164 WebsiteDataStore::forEachWebsiteDataStore([this](auto& websiteDataStore) {
165 addSession(websiteDataStore, SendParametersToNetworkProcess::No);
166 });
167#endif
168
169 WebProcessPool::platformInitializeNetworkProcess(parameters);
170 send(Messages::NetworkProcess::InitializeNetworkProcess(parameters), 0);
171}

243void NetworkProcessProxy::getNetworkProcessConnection(WebProcessProxy& webProcessProxy, Messages::WebProcessProxy::GetNetworkProcessConnection::DelayedReply&& reply)
244{
245 RELEASE_LOG(ProcessSuspension, "%p - NetworkProcessProxy is taking a background assertion because a web process is requesting a connection", this);
246 startResponsivenessTimer(UseLazyStop::No);
247 sendWithAsyncReply(Messages::NetworkProcess::CreateNetworkConnectionToWebProcess { webProcessProxy.coreProcessIdentifier(), webProcessProxy.sessionID() }, [this, weakThis = makeWeakPtr(*this), reply = WTFMove(reply)](auto&& identifier, auto cookieAcceptPolicy) mutable {
248 if (!weakThis) {
249 RELEASE_LOG_ERROR(Process, "NetworkProcessProxy::getNetworkProcessConnection: NetworkProcessProxy deallocated during connection establishment");
250 return reply({ });
251 }
252
253 stopResponsivenessTimer();
254 if (!identifier) {
255 RELEASE_LOG_ERROR(Process, "NetworkProcessProxy::getNetworkProcessConnection: connection identifier is empty");
256 return reply({ });
257 }
258
259#if USE(UNIX_DOMAIN_SOCKETS) || OS(WINDOWS)
260 reply(NetworkProcessConnectionInfo { WTFMove(*identifier), cookieAcceptPolicy });
261 UNUSED_VARIABLE(this);
262#elif OS(DARWIN)
263 MESSAGE_CHECK(MACH_PORT_VALID(identifier->port()));
264 reply(NetworkProcessConnectionInfo { IPC::Attachment { identifier->port(), MACH_MSG_TYPE_MOVE_SEND }, cookieAcceptPolicy, connection()->getAuditToken() });
265#else
266 notImplemented();
267#endif
268 }, 0, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
269}

309void NetworkProcessProxy::networkProcessDidTerminate(TerminationReason reason)
310{
311 auto protectedThis = makeRef(*this);
312
313 if (m_downloadProxyMap)
314 m_downloadProxyMap->invalidate();
315#if ENABLE(LEGACY_CUSTOM_PROTOCOL_MANAGER)
316 m_customProtocolManagerProxy.invalidate();
317#endif
318
319 m_uploadActivity = std::nullopt;
320
321 if (defaultNetworkProcess() == this)
322 defaultNetworkProcess() = nullptr;
323
324 for (auto& processPool : WebProcessPool::allProcessPools())
325 processPool->networkProcessDidTerminate(*this, reason);
326 for (auto& websiteDataStore : copyToVectorOf<Ref<WebsiteDataStore>>(m_websiteDataStores))
327 websiteDataStore->networkProcessDidTerminate(*this);
328}

364void NetworkProcessProxy::processAuthenticationChallenge(PAL::SessionID sessionID, Ref<AuthenticationChallengeProxy>&& authenticationChallenge)
365{
366 auto* store = websiteDataStoreFromSessionID(sessionID);
367 if (!store || authenticationChallenge->core().protectionSpace().authenticationScheme() != ProtectionSpaceAuthenticationSchemeServerTrustEvaluationRequested) {
368 authenticationChallenge->listener().completeChallenge(AuthenticationChallengeDisposition::PerformDefaultHandling);
369 return;
370 }
371 store->client().didReceiveAuthenticationChallenge(WTFMove(authenticationChallenge));
372}

374void NetworkProcessProxy::didReceiveAuthenticationChallenge(PAL::SessionID sessionID, WebPageProxyIdentifier pageID, const std::optional<SecurityOriginData>& topOrigin, WebCore::AuthenticationChallenge&& coreChallenge, bool negotiatedLegacyTLS, uint64_t challengeID)
375{
376#if HAVE(SEC_KEY_PROXY)
377 WeakPtr<SecKeyProxyStore> secKeyProxyStore;
378 if (coreChallenge.protectionSpace().authenticationScheme() == ProtectionSpaceAuthenticationSchemeClientCertificateRequested) {
379 if (auto* store = websiteDataStoreFromSessionID(sessionID)) {
380 auto newSecKeyProxyStore = SecKeyProxyStore::create();
381 secKeyProxyStore = makeWeakPtr(newSecKeyProxyStore.get());
382 store->addSecKeyProxyStore(WTFMove(newSecKeyProxyStore));
383 }
384 }
385 auto authenticationChallenge = AuthenticationChallengeProxy::create(WTFMove(coreChallenge), challengeID, makeRef(*connection()), WTFMove(secKeyProxyStore));
386#else
387 auto authenticationChallenge = AuthenticationChallengeProxy::create(WTFMove(coreChallenge), challengeID, makeRef(*connection()), nullptr);
388#endif
389
390 WebPageProxy* page = nullptr;
391 if (pageID)
392 page = WebProcessProxy::webPage(pageID);
393
394 if (page) {
395 page->didReceiveAuthenticationChallengeProxy(WTFMove(authenticationChallenge), negotiatedLegacyTLS ? NegotiatedLegacyTLS::Yes : NegotiatedLegacyTLS::No);
396 return;
397 }
398
399 if (!topOrigin) {
400 authenticationChallenge->listener().completeChallenge(AuthenticationChallengeDisposition::RejectProtectionSpaceAndContinue);
401 return;
402 }
403
404 WebPageProxy::forMostVisibleWebPageIfAny(sessionID, *topOrigin, [this, weakThis = makeWeakPtr(this), sessionID, authenticationChallenge = WTFMove(authenticationChallenge), negotiatedLegacyTLS](auto* page) mutable {
405 if (!weakThis)
406 return;
407
408 if (page) {
409 page->didReceiveAuthenticationChallengeProxy(WTFMove(authenticationChallenge), negotiatedLegacyTLS ? NegotiatedLegacyTLS::Yes : NegotiatedLegacyTLS::No);
410 return;
411 }
412 processAuthenticationChallenge(sessionID, WTFMove(authenticationChallenge));
413 });
414}
