96void WebNotificationManagerProxy::show(WebPageProxy* webPage, const String& title, const String& body, const String& iconURL, const String& tag, const String& lang, WebCore::NotificationDirection dir, const String& originString, uint64_t pageNotificationID)
97{
98 uint64_t globalNotificationID = generateGlobalNotificationID();
99 auto notification = WebNotification::create(title, body, iconURL, tag, lang, dir, originString, globalNotificationID);
100 auto notificationIDPair = std::make_pair(webPage->identifier(), pageNotificationID);
101 m_globalNotificationMap.set(globalNotificationID, notificationIDPair);
102 m_notifications.set(notificationIDPair, std::make_pair(globalNotificationID, notification.copyRef()));
103 m_provider->show(*webPage, notification.get());
104}

106void WebNotificationManagerProxy::cancel(WebPageProxy* webPage, uint64_t pageNotificationID)
107{
108 if (WebNotification* notification = m_notifications.get(std::make_pair(webPage->identifier(), pageNotificationID)).second.get())
109 m_provider->cancel(*notification);
110}

112void WebNotificationManagerProxy::didDestroyNotification(WebPageProxy* webPage, uint64_t pageNotificationID)
113{
114 auto globalIDNotificationPair = m_notifications.take(std::make_pair(webPage->identifier(), pageNotificationID));
115 if (uint64_t globalNotificationID = globalIDNotificationPair.first) {
116 WebNotification* notification = globalIDNotificationPair.second.get();
117 m_globalNotificationMap.remove(globalNotificationID);
118 m_provider->didDestroyNotification(*notification);
119 }
120}

142void WebNotificationManagerProxy::clearNotifications(WebPageProxy* webPage, const Vector<uint64_t>& pageNotificationIDs, NotificationFilterFunction filterFunction)
143{
144 auto targetPageProxyID = webPage->identifier();
145
146 Vector<uint64_t> globalNotificationIDs;
147 globalNotificationIDs.reserveCapacity(m_globalNotificationMap.size());
148
149 for (auto it = m_notifications.begin(), end = m_notifications.end(); it != end; ++it) {
150 auto pageProxyID = it->key.first;
151 uint64_t pageNotificationID = it->key.second;
152 if (!filterFunction(pageProxyID, pageNotificationID, targetPageProxyID, pageNotificationIDs))
153 continue;
154
155 uint64_t globalNotificationID = it->value.first;
156 globalNotificationIDs.append(globalNotificationID);
157 }
158
159 for (auto it = globalNotificationIDs.begin(), end = globalNotificationIDs.end(); it != end; ++it) {
160 auto pageNotification = m_globalNotificationMap.take(*it);
161 m_notifications.remove(pageNotification);
162 }
163
164 m_provider->clearNotifications(globalNotificationIDs);
165}

167void WebNotificationManagerProxy::providerDidShowNotification(uint64_t globalNotificationID)
168{
169 auto it = m_globalNotificationMap.find(globalNotificationID);
170 if (it == m_globalNotificationMap.end())
171 return;
172
173 auto* webPage = WebProcessProxy::webPage(it->value.first);
174 if (!webPage)
175 return;
176
177 uint64_t pageNotificationID = it->value.second;
178 webPage->process().send(Messages::WebNotificationManager::DidShowNotification(pageNotificationID), 0);
179}

196void WebNotificationManagerProxy::providerDidCloseNotifications(API::Array* globalNotificationIDs)
197{
198 HashMap<WebPageProxy*, Vector<uint64_t>> pageNotificationIDs;
199
200 size_t size = globalNotificationIDs->size();
201 for (size_t i = 0; i < size; ++i) {
202 auto it = m_globalNotificationMap.find(globalNotificationIDs->at<API::UInt64>(i)->value());
203 if (it == m_globalNotificationMap.end())
204 continue;
205
206 if (WebPageProxy* webPage = WebProcessProxy::webPage(it->value.first)) {
207 auto pageIt = pageNotificationIDs.find(webPage);
208 if (pageIt == pageNotificationIDs.end()) {
209 Vector<uint64_t> newVector;
210 newVector.reserveInitialCapacity(size);
211 pageIt = pageNotificationIDs.add(webPage, WTFMove(newVector)).iterator;
212 }
213
214 uint64_t pageNotificationID = it->value.second;
215 pageIt->value.append(pageNotificationID);
216 }
217
218 m_notifications.remove(it->value);
219 m_globalNotificationMap.remove(it);
220 }
221
222 for (auto it = pageNotificationIDs.begin(), end = pageNotificationIDs.end(); it != end; ++it)
223 it->key->process().send(Messages::WebNotificationManager::DidCloseNotifications(it->value), 0);
224}

226void WebNotificationManagerProxy::providerDidUpdateNotificationPolicy(const API::SecurityOrigin* origin, bool allowed)
227{
228 if (!processPool())
229 return;
230
231 processPool()->sendToAllProcesses(Messages::WebNotificationManager::DidUpdateNotificationDecision(origin->securityOrigin().toString(), allowed));
232}
233
234void WebNotificationManagerProxy::providerDidRemoveNotificationPolicies(API::Array* origins)
235{
236 if (!processPool())
237 return;
238
239 size_t size = origins->size();
240 if (!size)
241 return;
242
243 Vector<String> originStrings;
244 originStrings.reserveInitialCapacity(size);
245
246 for (size_t i = 0; i < size; ++i)
247 originStrings.append(origins->at<API::SecurityOrigin>(i)->securityOrigin().toString());
248
249 processPool()->sendToAllProcesses(Messages::WebNotificationManager::DidRemoveNotificationDecisions(originStrings));
250}

252uint64_t WebNotificationManagerProxy::notificationLocalIDForTesting(WebNotification* notification)
253{
254 if (!notification)
255 return 0;
256
257 auto it = m_globalNotificationMap.find(notification->notificationID());
258 if (it == m_globalNotificationMap.end())
259 return 0;
260
261 return it->value.second;
262}
