100 class Token {
101 public:
102 Token(Transaction& transaction)
103#if ASSERT_ENABLED
104 : m_pageLoadState(*transaction.m_pageLoadState)
105#endif
106 {
107 transaction.m_pageLoadState->m_mayHaveUncommittedChanges = true;
108 }
109
110#if ASSERT_ENABLED
111 PageLoadState& m_pageLoadState;
112#endif
113 };

115 RefPtr<WebPageProxy> m_webPageProxy;
116 PageLoadState* m_pageLoadState;
117 };
