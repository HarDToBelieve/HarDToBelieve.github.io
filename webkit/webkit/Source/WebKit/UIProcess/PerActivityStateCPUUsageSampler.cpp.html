38static const Seconds loggingInterval { 60_min };
39
40PerActivityStateCPUUsageSampler::PerActivityStateCPUUsageSampler(WebProcessPool& processPool)
41 : m_processPool(processPool)
42 , m_loggingTimer(RunLoop::main(), this, &PerActivityStateCPUUsageSampler::loggingTimerFired)
43{
44 m_lastCPUTime = MonotonicTime::now();
45 m_loggingTimer.startRepeating(loggingInterval);
46}
47
48PerActivityStateCPUUsageSampler::~PerActivityStateCPUUsageSampler()
49{
50}
51
52void PerActivityStateCPUUsageSampler::reportWebContentCPUTime(Seconds cpuTime, ActivityStateForCPUSampling activityState)
53{
54 auto result = m_cpuTimeInActivityState.add(activityState, cpuTime);
55 if (!result.isNewEntry)
56 result.iterator->value += cpuTime;
57}
58
59static inline String loggingKeyForActivityState(ActivityStateForCPUSampling state)
60{
61 switch (state) {
62 case ActivityStateForCPUSampling::NonVisible:
63 return DiagnosticLoggingKeys::nonVisibleStateKey();
64 case ActivityStateForCPUSampling::VisibleNonActive:
65 return DiagnosticLoggingKeys::visibleNonActiveStateKey();
66 case ActivityStateForCPUSampling::VisibleAndActive:
67 return DiagnosticLoggingKeys::visibleAndActiveStateKey();
68 }
69}
70
71void PerActivityStateCPUUsageSampler::loggingTimerFired()
72{
73 auto* page = pageForLogging();
74 if (!page) {
75 m_cpuTimeInActivityState.clear();
76 return;
77 }
78
79 MonotonicTime currentCPUTime = MonotonicTime::now();
80 Seconds cpuTimeDelta = currentCPUTime - m_lastCPUTime;
81
82 for (auto& pair : m_cpuTimeInActivityState) {
83 double cpuUsage = pair.value.value() * 100. / cpuTimeDelta.value();
84 String activityStateKey = loggingKeyForActivityState(pair.key);
85 page->logDiagnosticMessageWithValue(DiagnosticLoggingKeys::cpuUsageKey(), activityStateKey, cpuUsage, 2, ShouldSample::No);
86 RELEASE_LOG(PerformanceLogging, "WebContent processes used %.1f%% CPU in %s state", cpuUsage, activityStateKey.utf8().data());
87 }
88
89 m_cpuTimeInActivityState.clear();
90 m_lastCPUTime = currentCPUTime;
91}
92
93WebPageProxy* PerActivityStateCPUUsageSampler::pageForLogging() const
94{
95 for (auto& webProcess : m_processPool.processes()) {
96 if (!webProcess->pageCount())
97 continue;
98 return *webProcess->pages().begin();
99 }
100 return nullptr;
101}
