54uint64_t PluginProcessManager::pluginProcessToken(const PluginModuleInfo& pluginModuleInfo, PluginProcessSandboxPolicy pluginProcessSandboxPolicy)
55{
56 // See if we know this token already.
57 for (size_t i = 0; i < m_pluginProcessTokens.size(); ++i) {
58 const PluginProcessAttributes& attributes = m_pluginProcessTokens[i].first;
59
60 if (attributes.moduleInfo.path == pluginModuleInfo.path
61 && attributes.sandboxPolicy == pluginProcessSandboxPolicy)
62 return m_pluginProcessTokens[i].second;
63 }
64
65 uint64_t token;
66 while (true) {
67 cryptographicallyRandomValues(&token, sizeof(token));
68
69 if (m_knownTokens.isValidValue(token) && !m_knownTokens.contains(token))
70 break;
71 }
72
73 PluginProcessAttributes attributes;
74 attributes.moduleInfo = pluginModuleInfo;
75 attributes.sandboxPolicy = pluginProcessSandboxPolicy;
76
77 m_pluginProcessTokens.append(std::make_pair(WTFMove(attributes), token));
78 m_knownTokens.add(token);
79
80 return token;
81}

83bool PluginProcessManager::getPluginProcessConnection(uint64_t pluginProcessToken, Messages::WebProcessProxy::GetPluginProcessConnection::DelayedReply&& reply)
84{
85 ASSERT(pluginProcessToken);
86
87 auto* pluginProcess = getOrCreatePluginProcess(pluginProcessToken);
88 ASSERT(pluginProcess);
89 if (!pluginProcess)
90 return false;
91
92 pluginProcess->getPluginProcessConnection(WTFMove(reply));
93 return true;
94}

96void PluginProcessManager::removePluginProcessProxy(PluginProcessProxy* pluginProcessProxy)
97{
98 size_t vectorIndex = m_pluginProcesses.find(pluginProcessProxy);
99 ASSERT(vectorIndex != notFound);
100
101 m_pluginProcesses.remove(vectorIndex);
102}

104void PluginProcessManager::fetchWebsiteData(const PluginModuleInfo& plugin, OptionSet<WebsiteDataFetchOption> fetchOptions, WTF::Function<void (Vector<String>)>&& completionHandler)
105{
106 auto token = pluginProcessToken(plugin, PluginProcessSandboxPolicy::Normal);
107 auto pluginProcess = fetchOptions.contains(WebsiteDataFetchOption::DoNotCreateProcesses) ? getPluginProcess(token) : getOrCreatePluginProcess(token);
108 if (!pluginProcess) {
109 completionHandler({ });
110 return;
111 }
112
113 pluginProcess->fetchWebsiteData(WTFMove(completionHandler));
114}

116void PluginProcessManager::deleteWebsiteData(const PluginModuleInfo& plugin, WallTime modifiedSince, WTF::Function<void ()>&& completionHandler)
117{
118 PluginProcessProxy* pluginProcess = getOrCreatePluginProcess(pluginProcessToken(plugin, PluginProcessSandboxPolicy::Normal));

128PluginProcessProxy* PluginProcessManager::getPluginProcess(uint64_t pluginProcessToken)
129{
130 for (const auto& pluginProcess : m_pluginProcesses) {
131 if (pluginProcess->pluginProcessToken() == pluginProcessToken)
132 return pluginProcess.get();
133 }
134
135 return nullptr;
136}

146PluginProcessProxy* PluginProcessManager::getOrCreatePluginProcess(uint64_t pluginProcessToken)
147{
148 if (auto existingProcess = getPluginProcess(pluginProcessToken))
149 return existingProcess;
150
151 for (auto& attributesAndToken : m_pluginProcessTokens) {
152 if (attributesAndToken.second == pluginProcessToken) {
153 auto pluginProcess = PluginProcessProxy::create(this, attributesAndToken.first, attributesAndToken.second);
154 PluginProcessProxy* pluginProcessPtr = pluginProcess.ptr();
155 m_pluginProcesses.append(WTFMove(pluginProcess));
156 return pluginProcessPtr;
157 }
158 }
159
160 return nullptr;
161}
