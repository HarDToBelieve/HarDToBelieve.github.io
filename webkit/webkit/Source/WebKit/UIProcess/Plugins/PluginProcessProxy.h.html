66class PluginProcessProxy final : public AuxiliaryProcessProxy, private ProcessThrottlerClient {
67public:
68 static Ref<PluginProcessProxy> create(PluginProcessManager*, const PluginProcessAttributes&, uint64_t pluginProcessToken);
69 ~PluginProcessProxy();
70
71 const PluginProcessAttributes& pluginProcessAttributes() const { return m_pluginProcessAttributes; }
72 uint64_t pluginProcessToken() const { return m_pluginProcessToken; }
73
74 // Asks the plug-in process to create a new connection to a web process. The connection identifier will be
75 // encoded in the given argument encoder and sent back to the connection of the given web process.
76 void getPluginProcessConnection(Messages::WebProcessProxy::GetPluginProcessConnectionDelayedReply&&);
77
78 void fetchWebsiteData(CompletionHandler<void (Vector<String>)>&&);
79 void deleteWebsiteData(WallTime modifiedSince, CompletionHandler<void ()>&&);
80 void deleteWebsiteDataForHostNames(const Vector<String>& hostNames, CompletionHandler<void ()>&&);
81
82#if OS(LINUX)
83 void sendMemoryPressureEvent(bool isCritical);
84#endif
85
86 bool isValid() const { return m_connection; }
87
88#if PLUGIN_ARCHITECTURE(UNIX)
89 static bool scanPlugin(const String& pluginPath, RawPluginMetaData& result);
90#endif
91
92 ProcessThrottler& throttler() final { return m_throttler; }
93
94private:
95 PluginProcessProxy(PluginProcessManager*, const PluginProcessAttributes&, uint64_t pluginProcessToken);
96
97 // AuxiliaryProcessProxy
98 ASCIILiteral processName() const final { return "Plugin"_s; }
99
100 void getLaunchOptions(ProcessLauncher::LaunchOptions&) override;
101 void platformGetLaunchOptionsWithAttributes(ProcessLauncher::LaunchOptions&, const PluginProcessAttributes&);
102 void processWillShutDown(IPC::Connection&) override;
103
104 void pluginProcessCrashedOrFailedToLaunch();
105
106 // ProcessThrottlerClient
107 void sendPrepareToSuspend(IsSuspensionImminent, CompletionHandler<void()>&& completionHandler) final { completionHandler(); }
108 void sendProcessDidResume() final { }
109 ASCIILiteral clientName() const final { return "PluginProcess"_s; }
110
111 // IPC::Connection::Client
112 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
113 bool didReceiveSyncMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&) override;
114
115 void didClose(IPC::Connection&) override;
116 void didReceiveInvalidMessage(IPC::Connection&, IPC::MessageName) override;
117
118 // ProcessLauncher::Client
119 void didFinishLaunching(ProcessLauncher*, IPC::Connection::Identifier) override;
120
121 // Message handlers
122 void didCreateWebProcessConnection(const IPC::Attachment&, bool supportsAsynchronousPluginInitialization);
123 void didGetSitesWithData(const Vector<String>& sites, uint64_t callbackID);
124 void didDeleteWebsiteData(uint64_t callbackID);
125 void didDeleteWebsiteDataForHostNames(uint64_t callbackID);
126
127#if PLATFORM(COCOA)
128 bool getPluginProcessSerialNumber(ProcessSerialNumber&);
129 void makePluginProcessTheFrontProcess();
130 void makeUIProcessTheFrontProcess();
131
132 void setFullscreenWindowIsShowing(bool);
133 void enterFullscreen();
134 void exitFullscreen();
135
136 void setModalWindowIsShowing(bool);
137 void beginModal();
138 void endModal();
139
140 void applicationDidBecomeActive();
141 void launchProcess(const String& launchPath, const Vector<String>& arguments, CompletionHandler<void(bool)>&&);
142 void launchApplicationAtURL(const String& urlString, const Vector<String>& arguments, CompletionHandler<void(bool)>&&);
143 void openURL(const String& url, CompletionHandler<void(bool result, int32_t status, String launchedURLString)>&&);
144 void openFile(const String& fullPath, CompletionHandler<void(bool)>&&);
145#endif
146
147 void platformInitializePluginProcess(PluginProcessCreationParameters& parameters);
148
149 ProcessThrottler m_throttler;
150
151 // The plug-in host process manager.
152 PluginProcessManager* m_pluginProcessManager;
153
154 PluginProcessAttributes m_pluginProcessAttributes;
155 uint64_t m_pluginProcessToken;
156
157 // The connection to the plug-in host process.
158 RefPtr<IPC::Connection> m_connection;
159
160 Deque<Messages::WebProcessProxy::GetPluginProcessConnectionDelayedReply> m_pendingConnectionReplies;
161
162 Vector<uint64_t> m_pendingFetchWebsiteDataRequests;
163 HashMap<uint64_t, CompletionHandler<void (Vector<String>)>> m_pendingFetchWebsiteDataCallbacks;
164
165 struct DeleteWebsiteDataRequest {
166 WallTime modifiedSince;
167 uint64_t callbackID;
168 };
169 Vector<DeleteWebsiteDataRequest> m_pendingDeleteWebsiteDataRequests;
170 HashMap<uint64_t, CompletionHandler<void ()>> m_pendingDeleteWebsiteDataCallbacks;
171
172 struct DeleteWebsiteDataForHostNamesRequest {
173 Vector<String> hostNames;
174 uint64_t callbackID;
175 };
176 Vector<DeleteWebsiteDataForHostNamesRequest> m_pendingDeleteWebsiteDataForHostNamesRequests;
177 HashMap<uint64_t, CompletionHandler<void ()>> m_pendingDeleteWebsiteDataForHostNamesCallbacks;
178
179 // If createPluginConnection is called while the process is still launching we'll keep count of it and send a bunch of requests
180 // when the process finishes launching.
181 unsigned m_numPendingConnectionRequests;
182
183#if PLATFORM(COCOA)
184 RetainPtr<NSObject> m_activationObserver;
185 RetainPtr<WKPlaceholderModalWindow> m_placeholderWindow;
186 bool m_modalWindowIsShowing;
187 bool m_fullscreenWindowIsShowing;
188 unsigned m_preFullscreenAppPresentationOptions;
189#endif
190};
