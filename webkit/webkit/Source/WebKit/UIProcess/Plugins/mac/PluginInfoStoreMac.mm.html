42Vector<String> PluginInfoStore::pluginsDirectories()
43{
44 Vector<String> pluginsDirectories;
45 pluginsDirectories.reserveInitialCapacity(2);
46
47 ASCIILiteral pluginPath { "/Library/Internet Plug-Ins"_s };
48
49 if (auto* pw = getpwuid(getuid()))
50 pluginsDirectories.uncheckedAppend(makeString(pw->pw_dir, pluginPath));
51 pluginsDirectories.uncheckedAppend(pluginPath);
52
53 return pluginsDirectories;
54}

56Vector<String> PluginInfoStore::pluginPathsInDirectory(const String& directory)
57{
58 Vector<String> pluginPaths;
59
60 RetainPtr<CFStringRef> directoryCFString = directory.createCFString();
61 NSArray *filenames = [[NSFileManager defaultManager] contentsOfDirectoryAtPath:(__bridge NSString *)directoryCFString.get() error:nil];
62 for (NSString *filename in filenames)
63 pluginPaths.append([(__bridge NSString *)directoryCFString.get() stringByAppendingPathComponent:filename]);
64
65 return pluginPaths;
66}

78static bool shouldBlockPlugin(const PluginModuleInfo& plugin)
79{
80 PluginModuleLoadPolicy loadPolicy = PluginInfoStore::defaultLoadPolicyForPlugin(plugin);
81 return loadPolicy == PluginModuleBlockedForSecurity || loadPolicy == PluginModuleBlockedForCompatibility;
82}
83
84bool PluginInfoStore::shouldUsePlugin(Vector<PluginModuleInfo>& alreadyLoadedPlugins, const PluginModuleInfo& plugin)
85{
86 for (size_t i = 0; i < alreadyLoadedPlugins.size(); ++i) {
87 const PluginModuleInfo& loadedPlugin = alreadyLoadedPlugins[i];
88
89 // If a plug-in with the same bundle identifier already exists, we don't want to load it.
90 // However, if the already existing plug-in is blocked we want to replace it with the new plug-in.
91 if (loadedPlugin.bundleIdentifier == plugin.bundleIdentifier) {
92 if (!shouldBlockPlugin(loadedPlugin))
93 return false;
94
95 alreadyLoadedPlugins.remove(i);
96 break;
97 }
98 }
99
100 if (plugin.bundleIdentifier == "com.apple.java.JavaAppletPlugin") {
101 LOG(Plugins, "Ignoring com.apple.java.JavaAppletPlugin");
102 return false;
103 }
104
105 if (currentProcessIsSandboxed() && !plugin.hasSandboxProfile) {
106 LOG(Plugins, "Ignoring unsandboxed plug-in %s", plugin.bundleIdentifier.utf8().data());
107 return false;
108 }
109
110 return true;
111}

113PluginModuleLoadPolicy PluginInfoStore::defaultLoadPolicyForPlugin(const PluginModuleInfo& plugin)
114{
115 switch (PluginBlocklist::loadPolicyForPluginVersion(plugin.bundleIdentifier, plugin.versionString)) {
116 case PluginBlocklist::LoadPolicy::LoadNormally:
117 return PluginModuleLoadNormally;
118 case PluginBlocklist::LoadPolicy::BlockedForSecurity:
119 return PluginModuleBlockedForSecurity;
120 case PluginBlocklist::LoadPolicy::BlockedForCompatibility:
121 return PluginModuleBlockedForCompatibility;
122 }
123}

125PluginModuleInfo PluginInfoStore::findPluginWithBundleIdentifier(const String& bundleIdentifier)
126{
127 loadPluginsIfNecessary();
128
129 for (size_t i = 0; i < m_plugins.size(); ++i) {
130 if (m_plugins[i].bundleIdentifier == bundleIdentifier)
131 return m_plugins[i];
132 }
133
134 return PluginModuleInfo();
135}
