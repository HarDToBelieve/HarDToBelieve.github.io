44namespace WebKit {
45
46enum class ProcessAssertionType {
47 Suspended,
48 Background,
49 UnboundedNetworking,
50 Foreground,
51 MediaPlayback,
52 FinishTaskUninterruptable,
53};
54
55class ProcessAssertion : public CanMakeWeakPtr<ProcessAssertion> {
56 WTF_MAKE_FAST_ALLOCATED;
57public:
58 ProcessAssertion(ProcessID, const String& reason, ProcessAssertionType);
59 virtual ~ProcessAssertion();
60
61 void setInvalidationHandler(Function<void()>&& handler) { m_invalidationHandler = WTFMove(handler); }
62
63 ProcessAssertionType type() const { return m_assertionType; }
64 ProcessID pid() const { return m_pid; }
65
66 bool isValid() const;
67
68#if PLATFORM(IOS_FAMILY)
69protected:
70 virtual void processAssertionWasInvalidated();
71#endif
72
73private:
74 const ProcessAssertionType m_assertionType;
75 const ProcessID m_pid;
76 const String m_reason;
77#if PLATFORM(IOS_FAMILY)
78 RetainPtr<RBSAssertion> m_rbsAssertion;
79 RetainPtr<WKRBSAssertionDelegate> m_delegate;
80#endif
81 Function<void()> m_invalidationHandler;
82};
83
84class ProcessAndUIAssertion final : public ProcessAssertion {
85public:
86 ProcessAndUIAssertion(ProcessID, const String& reason, ProcessAssertionType);
87 ~ProcessAndUIAssertion();
88
89 void uiAssertionWillExpireImminently();
90
91 void setUIAssertionExpirationHandler(Function<void()>&& handler) { m_uiAssertionExpirationHandler = WTFMove(handler); }
92
93private:
94#if PLATFORM(IOS_FAMILY)
95 void processAssertionWasInvalidated() final;
96#endif
97 void updateRunInBackgroundCount();
98
99 Function<void()> m_uiAssertionExpirationHandler;
100 bool m_isHoldingBackgroundTask { false };
101};

103} // namespace WebKit
