79void ProcessThrottler::invalidateAllActivities()
80{
81 PROCESSTHROTTLER_RELEASE_LOG("invalidateAllActivities: BEGIN (foregroundActivityCount: %u, backgroundActivityCount: %u)", m_foregroundActivities.size(), m_backgroundActivities.size());
82 while (!m_foregroundActivities.isEmpty())
83 (*m_foregroundActivities.begin())->invalidate();
84 while (!m_backgroundActivities.isEmpty())
85 (*m_backgroundActivities.begin())->invalidate();
86 PROCESSTHROTTLER_RELEASE_LOG("invalidateAllActivities: END");
87}

89ProcessAssertionType ProcessThrottler::expectedAssertionType()
90{
91 if (!m_foregroundActivities.isEmpty())
92 return ProcessAssertionType::Foreground;
93 if (!m_backgroundActivities.isEmpty())
94 return ProcessAssertionType::Background;
95 return ProcessAssertionType::Suspended;
96}

98void ProcessThrottler::updateAssertionTypeNow()
99{
100 setAssertionType(expectedAssertionType());

103String ProcessThrottler::assertionName(ProcessAssertionType type) const
104{
105 ASCIILiteral typeString = [type] {
106 switch (type) {
107 case ProcessAssertionType::Foreground:
108 return "Foreground"_s;
109 case ProcessAssertionType::Background:
110 return "Background"_s;
111 case ProcessAssertionType::Suspended:
112 return "Suspended"_s;
113 case ProcessAssertionType::UnboundedNetworking:
114 case ProcessAssertionType::MediaPlayback:
115 case ProcessAssertionType::FinishTaskUninterruptable:
116 ASSERT_NOT_REACHED(); // These other assertion types are not used by the ProcessThrottler.
117 break;
118 }
119 return "Unknown"_s;
120 }();
121
122 return makeString(m_process.clientName(), " ", typeString, " Assertion");
123}

125void ProcessThrottler::setAssertionType(ProcessAssertionType newType)
126{
127 if (m_assertion && m_assertion->isValid() && m_assertion->type() == newType)
128 return;
129
130 PROCESSTHROTTLER_RELEASE_LOG("setAssertionType: Updating process assertion type to %u (foregroundActivities=%u, backgroundActivities=%u)", newType, m_foregroundActivities.size(), m_backgroundActivities.size());
131
132 // Keep the previous assertion around until after the new one has been created so that we always hold
133 // a process assertion for the process.
134 auto previousAssertion = std::exchange(m_assertion, nullptr);
135 if (m_shouldTakeUIBackgroundAssertion) {
136 auto assertion = makeUnique<ProcessAndUIAssertion>(m_processIdentifier, assertionName(newType), newType);
137 assertion->setUIAssertionExpirationHandler([this] {
138 uiAssertionWillExpireImminently();
139 });
140 m_assertion = WTFMove(assertion);
141 } else
142 m_assertion = makeUnique<ProcessAssertion>(m_processIdentifier, assertionName(newType), newType);
143
144 m_assertion->setInvalidationHandler([this] {
145 assertionWasInvalidated();
146 });
147 m_process.didSetAssertionType(newType);
148}

150void ProcessThrottler::updateAssertionIfNeeded()
151{
152 if (!m_assertion)
153 return;
154
155 if (shouldBeRunnable()) {
156 if (m_assertion->type() == ProcessAssertionType::Suspended || m_pendingRequestToSuspendID) {
157 if (m_assertion->type() == ProcessAssertionType::Suspended)
158 PROCESSTHROTTLER_RELEASE_LOG("updateAssertionIfNeeded: sending ProcessDidResume IPC because the process was suspended");
159 else
160 PROCESSTHROTTLER_RELEASE_LOG("updateAssertionIfNeeded: sending ProcessDidResume IPC because the WebProcess is still processing request to suspend=%" PRIu64, *m_pendingRequestToSuspendID);
161 m_process.sendProcessDidResume();
162 clearPendingRequestToSuspend();
163 }
164 } else {
165 // If the process is currently runnable but will be suspended then first give it a chance to complete what it was doing
166 // and clean up - move it to the background and send it a message to notify. Schedule a timeout so it can't stay running
167 // in the background for too long.
168 if (m_assertion->type() != ProcessAssertionType::Suspended) {
169 m_prepareToSuspendTimeoutTimer.startOneShot(processSuspensionTimeout);
170 sendPrepareToSuspendIPC(IsSuspensionImminent::No);
171 return;
172 }
173 }
174
175 updateAssertionTypeNow();
176}

178void ProcessThrottler::didConnectToProcess(ProcessID pid)
179{
180 PROCESSTHROTTLER_RELEASE_LOG_WITH_PID("didConnectToProcess:", pid);
181 RELEASE_ASSERT(!m_assertion);
182
183 m_processIdentifier = pid;
184 setAssertionType(expectedAssertionType());
185 RELEASE_ASSERT(m_assertion);
186}

188void ProcessThrottler::prepareToSuspendTimeoutTimerFired()
189{
190 PROCESSTHROTTLER_RELEASE_LOG("prepareToSuspendTimeoutTimerFired: Updating process assertion to allow suspension");
191 RELEASE_ASSERT(m_pendingRequestToSuspendID);
192 updateAssertionTypeNow();
193}
194
195void ProcessThrottler::processReadyToSuspend()
196{
197 PROCESSTHROTTLER_RELEASE_LOG("processReadyToSuspend: Updating process assertion to allow suspension");
198
199 RELEASE_ASSERT(m_pendingRequestToSuspendID);
200 clearPendingRequestToSuspend();
201
202 if (m_assertion->type() != ProcessAssertionType::Suspended)
203 updateAssertionTypeNow();
204}
205
206void ProcessThrottler::clearPendingRequestToSuspend()
207{
208 m_prepareToSuspendTimeoutTimer.stop();
209 m_pendingRequestToSuspendID = std::nullopt;
210}

212void ProcessThrottler::sendPrepareToSuspendIPC(IsSuspensionImminent isSuspensionImminent)
213{
214 PROCESSTHROTTLER_RELEASE_LOG("sendPrepareToSuspendIPC: isSuspensionImminent=%d", isSuspensionImminent == IsSuspensionImminent::Yes);
215 if (m_pendingRequestToSuspendID) {
216 // Do not send a new PrepareToSuspend IPC for imminent suspension if we've already sent a non-imminent PrepareToSuspend IPC.
217 RELEASE_ASSERT(isSuspensionImminent == IsSuspensionImminent::Yes);
218 PROCESSTHROTTLER_RELEASE_LOG("sendPrepareToSuspendIPC: Not sending PrepareToSuspend() IPC because there is already one in flight (%" PRIu64 ")", *m_pendingRequestToSuspendID);
219 } else {
220 m_pendingRequestToSuspendID = generatePrepareToSuspendRequestID();
221 PROCESSTHROTTLER_RELEASE_LOG("sendPrepareToSuspendIPC: Sending PrepareToSuspend(%" PRIu64 ", isSuspensionImminent=%d) IPC", *m_pendingRequestToSuspendID, isSuspensionImminent == IsSuspensionImminent::Yes);
222 m_process.sendPrepareToSuspend(isSuspensionImminent, [this, weakThis = makeWeakPtr(*this), requestToSuspendID = *m_pendingRequestToSuspendID]() mutable {
223 if (weakThis && m_pendingRequestToSuspendID && *m_pendingRequestToSuspendID == requestToSuspendID)
224 processReadyToSuspend();
225 });
226 }
227
228 setAssertionType(isSuspensionImminent == IsSuspensionImminent::Yes ? ProcessAssertionType::Suspended : ProcessAssertionType::Background);
229}

231void ProcessThrottler::uiAssertionWillExpireImminently()
232{
233 PROCESSTHROTTLER_RELEASE_LOG("uiAssertionWillExpireImminently:");
234 sendPrepareToSuspendIPC(IsSuspensionImminent::Yes);
235 invalidateAllActivities();
236 m_prepareToSuspendTimeoutTimer.stop();
237}

245bool ProcessThrottler::isValidBackgroundActivity(const ProcessThrottler::ActivityVariant& activity)
246{
247 if (!WTF::holds_alternative<UniqueRef<ProcessThrottler::BackgroundActivity>>(activity))
248 return false;
249 return WTF::get<UniqueRef<ProcessThrottler::BackgroundActivity>>(activity)->isValid();
250}

259ProcessThrottler::TimedActivity::TimedActivity(Seconds timeout, ProcessThrottler::ActivityVariant&& activity)
260 : m_timer(RunLoop::main(), this, &TimedActivity::activityTimedOut)
261 , m_timeout(timeout)
262 , m_activity(WTFMove(activity))
263{
264 updateTimer();
265}
266
267auto ProcessThrottler::TimedActivity::operator=(ProcessThrottler::ActivityVariant&& activity) -> TimedActivity&
268{
269 m_activity = WTFMove(activity);
270 updateTimer();
271 return *this;
272}
273
274void ProcessThrottler::TimedActivity::activityTimedOut()
275{
276 RELEASE_LOG_ERROR(ProcessSuspension, "%p - TimedActivity::activityTimedOut:", this);
277 m_activity = nullptr;
278}
279
280void ProcessThrottler::TimedActivity::updateTimer()
281{
282 if (WTF::holds_alternative<std::nullptr_t>(m_activity))
283 m_timer.stop();
284 else
285 m_timer.startOneShot(m_timeout);
286}
