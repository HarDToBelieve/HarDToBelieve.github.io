54class ProcessThrottler : public CanMakeWeakPtr<ProcessThrottler> {
55public:
56 ProcessThrottler(ProcessThrottlerClient&, bool shouldTakeUIBackgroundAssertion);
57 ~ProcessThrottler();
58
59 enum class ActivityType { Background, Foreground };
60 template<ActivityType type> class Activity {
61 WTF_MAKE_FAST_ALLOCATED;
62 public:
63 Activity(ProcessThrottler& throttler, ASCIILiteral name)
64 : m_throttler(&throttler)
65 , m_name(name)
66 {
67 throttler.addActivity(*this);
68 if (!isQuietActivity()) {
69 PROCESSTHROTTLER_ACTIVITY_RELEASE_LOG("Activity: Starting %" PUBLIC_LOG_STRING " activity / '%" PUBLIC_LOG_STRING "'",
70 type == ActivityType::Foreground ? "foreground" : "background", m_name.characters());
71 }
72 }
73
74 ~Activity()
75 {
76 if (isValid())
77 invalidate();
78 }
79
80 bool isValid() const { return !!m_throttler; }
81
82 private:
83 friend class ProcessThrottler;
84
85 bool isQuietActivity() const { return !m_name.characters(); }
86
87 void invalidate()
88 {
89 ASSERT(isValid());
90 if (!isQuietActivity()) {
91 PROCESSTHROTTLER_ACTIVITY_RELEASE_LOG("invalidate: Ending %" PUBLIC_LOG_STRING " activity / '%" PUBLIC_LOG_STRING "'",
92 type == ActivityType::Foreground ? "foreground" : "background", m_name.characters());
93 }
94 m_throttler->removeActivity(*this);
95 m_throttler = nullptr;
96 }
97
98 ProcessThrottler* m_throttler { nullptr };
99 ASCIILiteral m_name;
100 };
