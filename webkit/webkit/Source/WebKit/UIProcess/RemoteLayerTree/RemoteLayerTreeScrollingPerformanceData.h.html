31namespace WebKit {
32
33class RemoteLayerTreeDrawingAreaProxy;
34
35class RemoteLayerTreeScrollingPerformanceData {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 RemoteLayerTreeScrollingPerformanceData(RemoteLayerTreeDrawingAreaProxy&);
39 ~RemoteLayerTreeScrollingPerformanceData();
40
41 void didCommitLayerTree(const WebCore::FloatRect& visibleRect);
42 void didScroll(const WebCore::FloatRect& visibleRect);
43
44 NSArray *data(); // Array of [ time, event type, unfilled pixel count ]
45
46private:
47 unsigned blankPixelCount(const WebCore::FloatRect& visibleRect) const;
48
49 struct BlankPixelCount {
50 enum EventType { Filled, Exposed };
51
52 double startTime;
53 double endTime;
54 EventType eventType;
55 unsigned blankPixelCount;
56
57 BlankPixelCount(double start, double end, EventType type, unsigned count)
58 : startTime(start)
59 , endTime(end)
60 , eventType(type)
61 , blankPixelCount(count)
62 { }
63
64 bool canCoalesce(BlankPixelCount::EventType, unsigned blankPixelCount) const;
65 };
66
67 void appendBlankPixelCount(BlankPixelCount::EventType, unsigned blankPixelCount);
68
69 RemoteLayerTreeDrawingAreaProxy& m_drawingArea;
70 Vector<BlankPixelCount> m_blankPixelCounts;
71};
