88void RemoteScrollingTree::scrollingTreeNodeDidScroll(ScrollingTreeScrollingNode& node, ScrollingLayerPositionAction scrollingLayerPositionAction)
89{
90 std::optional<FloatPoint> layoutViewportOrigin;
91 if (is<ScrollingTreeFrameScrollingNode>(node))
92 layoutViewportOrigin = downcast<ScrollingTreeFrameScrollingNode>(node).layoutViewport().location();
93
94 m_scrollingCoordinatorProxy.scrollingTreeNodeDidScroll(node.scrollingNodeID(), node.currentScrollPosition(), layoutViewportOrigin, scrollingLayerPositionAction);
95}

102Ref<ScrollingTreeNode> RemoteScrollingTree::createScrollingTreeNode(ScrollingNodeType nodeType, ScrollingNodeID nodeID)
103{
104 switch (nodeType) {
105 case ScrollingNodeType::MainFrame:
106 case ScrollingNodeType::Subframe:
107#if PLATFORM(IOS_FAMILY)
108 return ScrollingTreeFrameScrollingNodeRemoteIOS::create(*this, nodeType, nodeID);
109#else
110 return ScrollingTreeFrameScrollingNodeRemoteMac::create(*this, nodeType, nodeID);
111#endif
112 case ScrollingNodeType::FrameHosting:
113 return ScrollingTreeFrameHostingNode::create(*this, nodeID);
114 case ScrollingNodeType::Overflow:
115#if PLATFORM(IOS_FAMILY)
116 return ScrollingTreeOverflowScrollingNodeIOS::create(*this, nodeID);
117#else
118 return ScrollingTreeOverflowScrollingNodeRemoteMac::create(*this, nodeID);
119#endif
120 case ScrollingNodeType::OverflowProxy:
121 return ScrollingTreeOverflowScrollProxyNode::create(*this, nodeID);
122 case ScrollingNodeType::Fixed:
123 return ScrollingTreeFixedNode::create(*this, nodeID);
124 case ScrollingNodeType::Sticky:
125 return ScrollingTreeStickyNode::create(*this, nodeID);
126 case ScrollingNodeType::Positioned:
127 return ScrollingTreePositionedNode::create(*this, nodeID);
128 }
129 ASSERT_NOT_REACHED();
130 return ScrollingTreeFixedNode::create(*this, nodeID);
131}
