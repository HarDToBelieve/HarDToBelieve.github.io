50- (id)initWithScrollerPair:(WebKit::ScrollerPairMac*)scrollerPair

52 self = [super init];

54 return nil;

60- (void)invalidate
61{
62 _scrollerPair = nullptr;

65- (NSRect)contentAreaRectForScrollerImpPair:(NSScrollerImpPair *)scrollerImpPair
66{
67 UNUSED_PARAM(scrollerImpPair);
68 if (!_scrollerPair)
69 return NSZeroRect;
70
71 auto size = _scrollerPair->visibleSize();
72 return NSMakeRect(0, 0, size.width(), size.height());
73}

75- (BOOL)inLiveResizeForScrollerImpPair:(NSScrollerImpPair *)scrollerImpPair

81- (NSPoint)mouseLocationInContentAreaForScrollerImpPair:(NSScrollerImpPair *)scrollerImpPair
82{
83 UNUSED_PARAM(scrollerImpPair);
84 if (!_scrollerPair)
85 return NSZeroPoint;
86
87 return _scrollerPair->lastKnownMousePosition();
88}
89
90- (NSPoint)scrollerImpPair:(NSScrollerImpPair *)scrollerImpPair convertContentPoint:(NSPoint)pointInContentArea toScrollerImp:(NSScrollerImp *)scrollerImp
91{
92 UNUSED_PARAM(scrollerImpPair);
93
94 if (!_scrollerPair || !scrollerImp)
95 return NSZeroPoint;
96
97 WebKit::ScrollerMac* scroller = nullptr;
98 if ([scrollerImp isHorizontal])
99 scroller = &_scrollerPair->horizontalScroller();
100 else
101 scroller = &_scrollerPair->verticalScroller();
102
103 ASSERT(scrollerImp == scroller->scrollerImp());
104
105 return scroller->convertFromContent(WebCore::IntPoint(pointInContentArea));
106}

123ScrollerPairMac::ScrollerPairMac(WebCore::ScrollingTreeScrollingNode& node)
124 : m_scrollingNode(node)
125 , m_verticalScroller(*this, ScrollerMac::Orientation::Vertical)
126 , m_horizontalScroller(*this, ScrollerMac::Orientation::Horizontal)
127{
128 m_scrollerImpPairDelegate = adoptNS([[WKScrollerImpPairDelegate alloc] initWithScrollerPair:this]);
129
130 m_scrollerImpPair = adoptNS([[NSScrollerImpPair alloc] init]);
131 [m_scrollerImpPair.get() setDelegate:m_scrollerImpPairDelegate.get()];
132 [m_scrollerImpPair setScrollerStyle:WebCore::ScrollerStyle::recommendedScrollerStyle()];
133
134 m_verticalScroller.attach();
135 m_horizontalScroller.attach();
136}

144bool ScrollerPairMac::handleWheelEvent(const WebCore::PlatformWheelEvent& event)
145{
146 switch (event.phase()) {
147 case WebCore::PlatformWheelEventPhase::Began:
148 [m_scrollerImpPair beginScrollGesture];
149 break;
150 case WebCore::PlatformWheelEventPhase::Ended:
151 case WebCore::PlatformWheelEventPhase::Cancelled:
152 [m_scrollerImpPair endScrollGesture];
153 break;
154 case WebCore::PlatformWheelEventPhase::MayBegin:
155 [m_scrollerImpPair beginScrollGesture];
156 [m_scrollerImpPair contentAreaScrolled];
157 break;
158 default:
159 break;
160 }
161 // FIXME: this needs to return whether the event was handled.
162 return true;
163}

165bool ScrollerPairMac::handleMouseEvent(const WebCore::PlatformMouseEvent& event)
166{
167 if (event.type() != WebCore::PlatformEvent::MouseMoved)
168 return false;
169
170 m_lastKnownMousePosition = event.position();
171 [m_scrollerImpPair mouseMovedInContentArea];
172 // FIXME: this needs to return whether the event was handled.
173 return true;
174}

176void ScrollerPairMac::updateValues()
177{
178 auto position = m_scrollingNode.currentScrollPosition();
179
180 if (position != m_lastScrollPosition) {
181 if (m_lastScrollPosition) {
182 auto delta = position - *m_lastScrollPosition;
183 [m_scrollerImpPair contentAreaScrolledInDirection:NSMakePoint(delta.width(), delta.height())];
184 }
185 m_lastScrollPosition = position;
186 }
187
188 m_verticalScroller.updateValues();
189 m_horizontalScroller.updateValues();
190}

202ScrollerPairMac::Values ScrollerPairMac::valuesForOrientation(ScrollerMac::Orientation orientation)
203{
204 float position;
205 float totalSize;
206 float visibleSize;
207 if (orientation == ScrollerMac:: Orientation::Vertical) {
208 position = m_scrollingNode.currentScrollPosition().y();
209 totalSize = m_scrollingNode.totalContentsSize().height();
210 visibleSize = m_scrollingNode.scrollableAreaSize().height();
211 } else {
212 position = m_scrollingNode.currentScrollPosition().x();
213 totalSize = m_scrollingNode.totalContentsSize().width();
214 visibleSize = m_scrollingNode.scrollableAreaSize().width();
215 }
216
217 float value;
218 float overhang;
219 WebCore::ScrollableArea::computeScrollbarValueAndOverhang(position, totalSize, visibleSize, value, overhang);
220
221 float proportion = totalSize ? (visibleSize - overhang) / totalSize : 1;
222
223 return { value, proportion };
224}
