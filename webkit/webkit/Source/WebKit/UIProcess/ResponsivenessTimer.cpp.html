29namespace WebKit {
30
31static const Seconds responsivenessTimeout { 3_s };
32
33ResponsivenessTimer::ResponsivenessTimer(ResponsivenessTimer::Client& client)
34 : m_client(client)
35 , m_timer(RunLoop::main(), this, &ResponsivenessTimer::timerFired)
36{
37}
38
39ResponsivenessTimer::~ResponsivenessTimer() = default;
40
41void ResponsivenessTimer::invalidate()
42{
43 m_timer.stop();
44 m_restartFireTime = MonotonicTime();
45 m_waitingForTimer = false;
46 m_useLazyStop = false;
47}
48
49void ResponsivenessTimer::timerFired()
50{
51 if (!m_waitingForTimer)
52 return;
53
54 if (m_restartFireTime) {
55 MonotonicTime now = MonotonicTime::now();
56 MonotonicTime restartFireTime = m_restartFireTime;
57 m_restartFireTime = MonotonicTime();
58
59 if (restartFireTime > now) {
60 m_timer.startOneShot(restartFireTime - now);
61 return;
62 }
63 }
64
65 m_waitingForTimer = false;
66 m_useLazyStop = false;
67
68 if (!m_isResponsive)
69 return;
70
71 auto protectedClient = makeRef(m_client);
72
73 if (!mayBecomeUnresponsive()) {
74 m_waitingForTimer = true;
75 m_timer.startOneShot(responsivenessTimeout);
76 return;
77 }
78
79 m_client.willChangeIsResponsive();
80 m_isResponsive = false;
81 m_client.didChangeIsResponsive();
82
83 m_client.didBecomeUnresponsive();
84}
85
86void ResponsivenessTimer::start()
87{
88 if (m_waitingForTimer)
89 return;
90
91 m_waitingForTimer = true;
92 m_useLazyStop = false;
93
94 if (m_timer.isActive()) {
95 // The timer is still active from a lazy stop.
96 // Instead of restarting the timer, we schedule a new delay after this one finishes.
97 //
98 // In most cases, stop is called before we get to schedule the second timer, saving us
99 // the scheduling of the timer entirely.
100 m_restartFireTime = MonotonicTime::now() + responsivenessTimeout;
101 } else {
102 m_restartFireTime = MonotonicTime();
103 m_timer.startOneShot(responsivenessTimeout);
104 }
105}
106
107bool ResponsivenessTimer::mayBecomeUnresponsive() const
108{
109#if !defined(NDEBUG) || ASAN_ENABLED
110 return false;
111#else
112 static bool isLibgmallocEnabled = [] {
113 char* variable = getenv("DYLD_INSERT_LIBRARIES");
114 if (!variable)
115 return false;
116 if (!strstr(variable, "libgmalloc"))
117 return false;
118 return true;
119 }();
120 if (isLibgmallocEnabled)
121 return false;
122
123 return m_client.mayBecomeUnresponsive();
124#endif
125}
126
127void ResponsivenessTimer::startWithLazyStop()
128{
129 if (!m_waitingForTimer) {
130 start();
131 m_useLazyStop = true;
132 }
133}
134
135void ResponsivenessTimer::stop()
136{
137 if (!m_isResponsive) {
138 auto protectedClient = makeRef(m_client);
139
140 // We got a life sign from the web process.
141 m_client.willChangeIsResponsive();
142 m_isResponsive = true;
143 m_client.didChangeIsResponsive();
144
145 m_client.didBecomeResponsive();
146 }
147
148 m_waitingForTimer = false;
149
150 if (m_useLazyStop)
151 m_useLazyStop = false;
152 else
153 m_timer.stop();
154}
155
156void ResponsivenessTimer::processTerminated()
157{
158 invalidate();
159}
