26#pragma once
27
28#include <wtf/URL.h>
29#include <wtf/RefCounted.h>
30#include <wtf/RetainPtr.h>
31#include <wtf/text/WTFString.h>
32
33OBJC_CLASS NSAttributedString;
34OBJC_CLASS NSURL;
35OBJC_CLASS SSBServiceLookupResult;
36
37namespace WebKit {
38
39class SafeBrowsingWarning : public RefCounted<SafeBrowsingWarning> {
40public:
41#if HAVE(SAFE_BROWSING)
42 static Ref<SafeBrowsingWarning> create(const URL& url, bool forMainFrameNavigation, SSBServiceLookupResult *result)
43 {
44 return adoptRef(*new SafeBrowsingWarning(url, forMainFrameNavigation, result));
45 }
46#endif
47#if PLATFORM(COCOA)
48 static Ref<SafeBrowsingWarning> create(URL&& url, String&& title, String&& warning, RetainPtr<NSAttributedString>&& details)
49 {
50 return adoptRef(*new SafeBrowsingWarning(WTFMove(url), WTFMove(title), WTFMove(warning), WTFMove(details)));
51 }
52#endif
53
54 const URL& url() const { return m_url; }
55 const String& title() const { return m_title; }
56 const String& warning() const { return m_warning; }
57 bool forMainFrameNavigation() const { return m_forMainFrameNavigation; }
58#if PLATFORM(COCOA)
59 RetainPtr<NSAttributedString> details() const { return m_details; }
60#endif
61
62 static NSURL *visitUnsafeWebsiteSentinel();
63 static NSURL *confirmMalwareSentinel();
64
65private:
66#if HAVE(SAFE_BROWSING)
67 SafeBrowsingWarning(const URL&, bool, SSBServiceLookupResult *);
68#endif
69#if PLATFORM(COCOA)
70 SafeBrowsingWarning(URL&&, String&&, String&&, RetainPtr<NSAttributedString>&&);
71#endif
72
73 URL m_url;
74 String m_title;
75 String m_warning;
76 bool m_forMainFrameNavigation { false };
77#if PLATFORM(COCOA)
78 RetainPtr<NSAttributedString> m_details;
79#endif
80};
81
82} // namespace WebKit
