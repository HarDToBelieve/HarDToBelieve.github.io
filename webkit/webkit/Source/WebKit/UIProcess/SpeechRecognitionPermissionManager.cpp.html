35static SpeechRecognitionPermissionManager::CheckResult computeMicrophoneAccess()
36{
37#if HAVE(AVCAPTUREDEVICE)
38 auto result = checkAVCaptureAccessForType(MediaPermissionType::Audio);
39 if (result != MediaPermissionResult::Unknown)
40 return result == MediaPermissionResult::Granted ? SpeechRecognitionPermissionManager::CheckResult::Granted : SpeechRecognitionPermissionManager::CheckResult::Denied;
41
42 if (!checkUsageDescriptionStringForType(MediaPermissionType::Audio))
43 return SpeechRecognitionPermissionManager::CheckResult::Denied;
44
45 return SpeechRecognitionPermissionManager::CheckResult::Unknown;
46#else
47 return SpeechRecognitionPermissionManager::CheckResult::Granted;
48#endif
49}
50
51static SpeechRecognitionPermissionManager::CheckResult computeSpeechRecognitionServiceAccess()
52{
53#if HAVE(SPEECHRECOGNIZER)
54 auto result = checkSpeechRecognitionServiceAccess();
55 if (result != MediaPermissionResult::Unknown)
56 return result == MediaPermissionResult::Granted ? SpeechRecognitionPermissionManager::CheckResult::Granted : SpeechRecognitionPermissionManager::CheckResult::Denied;
57
58 if (!checkUsageDescriptionStringForSpeechRecognition())
59 return SpeechRecognitionPermissionManager::CheckResult::Denied;
60
61 return SpeechRecognitionPermissionManager::CheckResult::Unknown;
62#else
63 return SpeechRecognitionPermissionManager::CheckResult::Granted;
64#endif
65}

72SpeechRecognitionPermissionManager::~SpeechRecognitionPermissionManager()
73{
74 for (auto& request : m_requests)

78void SpeechRecognitionPermissionManager::request(WebCore::SpeechRecognitionRequest& request, SpeechRecognitionPermissionRequestCallback&& completiontHandler)
79{
80 m_requests.append(SpeechRecognitionPermissionRequest::create(request, WTFMove(completiontHandler)));
81 if (m_requests.size() == 1)
82 startNextRequest();
83}
84
85void SpeechRecognitionPermissionManager::startNextRequest()
86{
87 while (!m_requests.isEmpty() && !m_requests.first()->request())
88 m_requests.removeFirst();
89
90 if (m_requests.isEmpty())
91 return;
92
93 startProcessingRequest();
94}
95
96void SpeechRecognitionPermissionManager::startProcessingRequest()
97{
98#if PLATFORM(COOCA)
99 if (!checkSandboxRequirementForType(MediaPermissionType::Audio)) {
100 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::NotAllowed, "Sandbox check has failed"_s });
101 return;
102 }
103#endif
104
105 m_page.syncIfMockDevicesEnabledChanged();
106 if (m_page.preferences().mockCaptureDevicesEnabled()) {
107 m_microphoneCheck = CheckResult::Granted;
108 m_speechRecognitionServiceCheck = CheckResult::Granted;
109 } else {
110 // TCC status may have changed between requests.
111 m_microphoneCheck = computeMicrophoneAccess();
112 if (m_microphoneCheck == CheckResult::Denied) {
113 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::NotAllowed, "Microphone permission check has failed"_s });
114 return;
115 }
116
117 m_speechRecognitionServiceCheck = computeSpeechRecognitionServiceAccess();
118 if (m_speechRecognitionServiceCheck == CheckResult::Denied) {
119 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::ServiceNotAllowed, "Speech recognition service permission check has failed"_s });
120 return;
121 }
122
123#if HAVE(SPEECHRECOGNIZER)
124 if (!checkSpeechRecognitionServiceAvailability(m_requests.first()->request()->lang())) {
125 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::ServiceNotAllowed, "Speech recognition service is not available"_s });
126 return;
127 }
128#endif
129 }
130
131 if (m_userPermissionCheck == CheckResult::Denied) {
132 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::NotAllowed, "User permission check has failed"_s });
133 return;
134 }
135
136 continueProcessingRequest();
137}

139void SpeechRecognitionPermissionManager::continueProcessingRequest()
140{
141 ASSERT(!m_requests.isEmpty());
142 auto recognitionRequest = m_requests.first()->request();
143 if (!recognitionRequest) {
144 completeCurrentRequest();
145 return;
146 }
147
148 if (m_speechRecognitionServiceCheck == CheckResult::Unknown) {
149 requestSpeechRecognitionServiceAccess();
150 return;
151 }
152 ASSERT(m_speechRecognitionServiceCheck == CheckResult::Granted);
153
154 if (m_microphoneCheck == CheckResult::Unknown) {
155 requestMicrophoneAccess();
156 return;
157 }
158 ASSERT(m_microphoneCheck == CheckResult::Granted);
159
160 if (m_userPermissionCheck == CheckResult::Unknown) {
161 requestUserPermission(*recognitionRequest);
162 return;
163 }
164 ASSERT(m_userPermissionCheck == CheckResult::Granted);
165
166 if (!m_page.isViewVisible()) {
167 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::NotAllowed, "Page is not visible to user"_s });
168 return;
169 }
170
171 completeCurrentRequest();
172}

174void SpeechRecognitionPermissionManager::completeCurrentRequest(std::optional<WebCore::SpeechRecognitionError>&& error)
175{
176 ASSERT(!m_requests.isEmpty());
177 auto currentRequest = m_requests.takeFirst();
178 currentRequest->complete(WTFMove(error));
179
180 startNextRequest();
181}

183void SpeechRecognitionPermissionManager::requestSpeechRecognitionServiceAccess()
184{
185 ASSERT(m_speechRecognitionServiceCheck == CheckResult::Unknown);
186
187#if HAVE(SPEECHRECOGNIZER)
188 requestSpeechRecognitionAccess([this, weakThis = makeWeakPtr(this)](bool authorized) mutable {
189 if (!weakThis)
190 return;
191
192 m_speechRecognitionServiceCheck = authorized ? CheckResult::Granted : CheckResult::Denied;
193 if (m_speechRecognitionServiceCheck == CheckResult::Denied) {
194 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::ServiceNotAllowed, "Speech recognition service permission check has failed"_s });
195 return;
196 }
197
198 continueProcessingRequest();
199 });
200#endif
201}

203void SpeechRecognitionPermissionManager::requestMicrophoneAccess()
204{
205 ASSERT(m_microphoneCheck == CheckResult::Unknown);
206
207#if HAVE(AVCAPTUREDEVICE)
208 requestAVCaptureAccessForType(MediaPermissionType::Audio, [this, weakThis = makeWeakPtr(this)](bool authorized) {
209 if (!weakThis)
210 return;
211
212 m_microphoneCheck = authorized ? CheckResult::Granted : CheckResult::Denied;
213 if (m_microphoneCheck == CheckResult::Denied) {
214 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::NotAllowed, "Microphone permission check has failed"_s });
215 return;
216 }
217
218 continueProcessingRequest();
219 });
220#endif
221}

223void SpeechRecognitionPermissionManager::requestUserPermission(WebCore::SpeechRecognitionRequest& recognitionRequest)
224{
225 auto clientOrigin = recognitionRequest.clientOrigin();
226 auto requestingOrigin = clientOrigin.clientOrigin.securityOrigin();
227 auto topOrigin = clientOrigin.topOrigin.securityOrigin();
228 auto decisionHandler = [this, weakThis = makeWeakPtr(*this)](bool granted) {
229 if (!weakThis)
230 return;
231
232 m_userPermissionCheck = granted ? CheckResult::Granted : CheckResult::Denied;
233 if (m_userPermissionCheck == CheckResult::Denied) {
234 completeCurrentRequest(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::NotAllowed, "User permission check has failed"_s });
235 return;
236 }
237
238 continueProcessingRequest();
239 };
240 m_page.requestUserMediaPermissionForSpeechRecognition(recognitionRequest.frameIdentifier(), requestingOrigin, topOrigin, WTFMove(decisionHandler));
241}

244{

248 completionHandler(false);
249#endif
