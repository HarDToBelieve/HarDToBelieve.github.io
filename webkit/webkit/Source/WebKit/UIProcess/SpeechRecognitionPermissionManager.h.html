33
34class WebPageProxy;
35
36class SpeechRecognitionPermissionManager : public CanMakeWeakPtr<SpeechRecognitionPermissionManager> {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 enum class CheckResult { Denied, Granted, Unknown };
40 explicit SpeechRecognitionPermissionManager(WebPageProxy&);
41 ~SpeechRecognitionPermissionManager();
42 void request(WebCore::SpeechRecognitionRequest&, SpeechRecognitionPermissionRequestCallback&&);
43
44 void decideByDefaultAction(const WebCore::SecurityOriginData&, CompletionHandler<void(bool)>&&);
45 WebPageProxy& page() { return m_page; }
46
47private:
48 void startNextRequest();
49 void startProcessingRequest();
50 void continueProcessingRequest();
51 void completeCurrentRequest(std::optional<WebCore::SpeechRecognitionError>&& = std::nullopt);
52 void requestMicrophoneAccess();
53 void requestSpeechRecognitionServiceAccess();
54 void requestUserPermission(WebCore::SpeechRecognitionRequest& request);
55
56 WebPageProxy& m_page;
57 Deque<Ref<SpeechRecognitionPermissionRequest>> m_requests;
58 CheckResult m_microphoneCheck { CheckResult::Unknown };
59 CheckResult m_speechRecognitionServiceCheck { CheckResult::Unknown };
60 CheckResult m_userPermissionCheck { CheckResult::Unknown };
61};
