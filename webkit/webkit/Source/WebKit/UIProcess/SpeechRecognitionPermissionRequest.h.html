35namespace WebKit {
36
37using SpeechRecognitionPermissionRequestCallback = CompletionHandler<void(std::optional<WebCore::SpeechRecognitionError>&&)>;
38
39class SpeechRecognitionPermissionRequest : public RefCounted<SpeechRecognitionPermissionRequest> {
40public:
41 static Ref<SpeechRecognitionPermissionRequest> create(WebCore::SpeechRecognitionRequest& request, SpeechRecognitionPermissionRequestCallback&& completionHandler)
42 {
43 return adoptRef(*new SpeechRecognitionPermissionRequest(request, WTFMove(completionHandler)));
44 }
45
46 ~SpeechRecognitionPermissionRequest()
47 {
48 if (m_completionHandler)
49 m_completionHandler(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::NotAllowed, "Request is cancelled"_s });
50 }
51
52 void complete(std::optional<WebCore::SpeechRecognitionError>&& error)
53 {
54 auto completionHandler = std::exchange(m_completionHandler, { });
55 completionHandler(WTFMove(error));
56 }
57
58 WebCore::SpeechRecognitionRequest* request() { return m_request.get(); }
59
60private:
61 SpeechRecognitionPermissionRequest(WebCore::SpeechRecognitionRequest& request, SpeechRecognitionPermissionRequestCallback&& completionHandler)
62 : m_request(makeWeakPtr(request))
63 , m_completionHandler(WTFMove(completionHandler))
64 { }
65
66
67 WeakPtr<WebCore::SpeechRecognitionRequest> m_request;
68 SpeechRecognitionPermissionRequestCallback m_completionHandler;
69};
70
71class SpeechRecognitionPermissionCallback : public API::ObjectImpl<API::Object::Type::SpeechRecognitionPermissionCallback> {
72public:
73 static Ref<SpeechRecognitionPermissionCallback> create(CompletionHandler<void(bool)>&& completionHandler)
74 {
75 return adoptRef(*new SpeechRecognitionPermissionCallback(WTFMove(completionHandler)));
76 }
77
78 void complete(bool granted) { m_completionHandler(granted); }
79
80private:
81 SpeechRecognitionPermissionCallback(CompletionHandler<void(bool)>&& completionHandler)
82 : m_completionHandler(WTFMove(completionHandler))
83 { }
84
85 CompletionHandler<void(bool)> m_completionHandler;
86};
