42Ref<WebCore::RealtimeMediaSource> SpeechRecognitionRemoteRealtimeMediaSource::create(SpeechRecognitionRemoteRealtimeMediaSourceManager& manager, const WebCore::CaptureDevice& captureDevice)
43{
44 return adoptRef(*new SpeechRecognitionRemoteRealtimeMediaSource(WebCore::RealtimeMediaSourceIdentifier::generate(), manager, captureDevice));
45}
46
47SpeechRecognitionRemoteRealtimeMediaSource::SpeechRecognitionRemoteRealtimeMediaSource(WebCore::RealtimeMediaSourceIdentifier identifier, SpeechRecognitionRemoteRealtimeMediaSourceManager& manager, const WebCore::CaptureDevice& captureDevice)
48 : WebCore::RealtimeMediaSource(WebCore::RealtimeMediaSource::Type::Audio, String { captureDevice.label() }, String { captureDevice.persistentId() })
49 , m_identifier(identifier)
50 , m_manager(makeWeakPtr(manager))
51#if PLATFORM(COCOA)
52 , m_ringBuffer(makeUnique<WebCore::CARingBuffer>())
53#endif
54{
55 m_manager->addSource(*this, captureDevice);
56}
57
58SpeechRecognitionRemoteRealtimeMediaSource::~SpeechRecognitionRemoteRealtimeMediaSource()
59{
60 if (m_manager)
61 m_manager->removeSource(*this);
62}
63
64void SpeechRecognitionRemoteRealtimeMediaSource::startProducingData()
65{
66 if (m_manager)
67 m_manager->send(Messages::SpeechRecognitionRealtimeMediaSourceManager::Start { m_identifier });
68}
69
70void SpeechRecognitionRemoteRealtimeMediaSource::stopProducingData()
71{
72 if (m_manager)
73 m_manager->send(Messages::SpeechRecognitionRealtimeMediaSourceManager::Stop { m_identifier });
74}

78void SpeechRecognitionRemoteRealtimeMediaSource::setStorage(const SharedMemory::Handle& handle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames)
79{
80 m_description = description;
81
82 m_ringBuffer = WebCore::CARingBuffer::adoptStorage(makeUniqueRef<ReadOnlySharedRingBufferStorage>(handle), description, numberOfFrames).moveToUniquePtr();
83 m_buffer = makeUnique<WebCore::WebAudioBufferList>(description);
84}
85
86#endif
87
88void SpeechRecognitionRemoteRealtimeMediaSource::remoteAudioSamplesAvailable(MediaTime time, uint64_t numberOfFrames)
89{
90#if PLATFORM(COCOA)
91 if (!m_buffer) {
92 LOG_ERROR("Buffer for remote source is null");
93 captureFailed();
94 return;
95 }
96
97 m_buffer->setSampleCount(numberOfFrames);
98 m_ringBuffer->fetch(m_buffer->list(), numberOfFrames, time.timeValue());
99 audioSamplesAvailable(time, *m_buffer, m_description, numberOfFrames);
100#else
101 UNUSED_PARAM(time);
102 UNUSED_PARAM(numberOfFrames);
103#endif
104}

106void SpeechRecognitionRemoteRealtimeMediaSource::remoteCaptureFailed()

109}

111void SpeechRecognitionRemoteRealtimeMediaSource::remoteSourceStopped()

113 stop();
