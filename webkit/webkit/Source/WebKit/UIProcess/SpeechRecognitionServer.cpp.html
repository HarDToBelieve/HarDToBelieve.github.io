54void SpeechRecognitionServer::start(WebCore::SpeechRecognitionConnectionClientIdentifier clientIdentifier, String&& lang, bool continuous, bool interimResults, uint64_t maxAlternatives, WebCore::ClientOrigin&& origin, WebCore::FrameIdentifier frameIdentifier)
55{
56 MESSAGE_CHECK(clientIdentifier);
57 ASSERT(!m_requests.contains(clientIdentifier));
58 auto requestInfo = WebCore::SpeechRecognitionRequestInfo { clientIdentifier, WTFMove(lang), continuous, interimResults, maxAlternatives, WTFMove(origin), frameIdentifier };
59 auto& newRequest = m_requests.add(clientIdentifier, makeUnique<WebCore::SpeechRecognitionRequest>(WTFMove(requestInfo))).iterator->value;
60
61 requestPermissionForRequest(*newRequest);
62}
63
64void SpeechRecognitionServer::requestPermissionForRequest(WebCore::SpeechRecognitionRequest& request)
65{
66 m_permissionChecker(request, [this, weakThis = makeWeakPtr(this), weakRequest = makeWeakPtr(request)](auto error) mutable {
67 if (!weakThis || !weakRequest)
68 return;
69
70 auto identifier = weakRequest->clientIdentifier();
71 auto request = m_requests.take(identifier);
72 if (error) {
73 sendUpdate(identifier, WebCore::SpeechRecognitionUpdateType::Error, WTFMove(error));
74 return;
75 }
76
77 ASSERT(request);
78 handleRequest(makeUniqueRefFromNonNullUniquePtr(WTFMove(request)));
79 });
80}
81
82void SpeechRecognitionServer::handleRequest(UniqueRef<WebCore::SpeechRecognitionRequest>&& request)
83{
84 if (m_recognizer) {
85 m_recognizer->abort(WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::Aborted, "Another request is started"_s });
86 m_recognizer->prepareForDestruction();
87 }
88
89 auto clientIdentifier = request->clientIdentifier();
90 m_recognizer = makeUnique<WebCore::SpeechRecognizer>([this, weakThis = makeWeakPtr(this)](auto& update) {
91 if (!weakThis)
92 return;
93
94 sendUpdate(update);
95
96 if (update.type() == WebCore::SpeechRecognitionUpdateType::Error)
97 m_recognizer->abort();
98 else if (update.type() == WebCore::SpeechRecognitionUpdateType::End)
99 m_recognizer->setInactive();
100 }, WTFMove(request));
101
102#if ENABLE(MEDIA_STREAM)
103 auto sourceOrError = m_realtimeMediaSourceCreateFunction();
104 if (!sourceOrError) {
105 sendUpdate(WebCore::SpeechRecognitionUpdate::createError(clientIdentifier, WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::AudioCapture, sourceOrError.errorMessage }));
106 return;
107 }
108
109 WebProcessProxy::muteCaptureInPagesExcept(m_identifier);
110 bool mockDeviceCapturesEnabled = m_checkIfMockSpeechRecognitionEnabled();
111 m_recognizer->start(sourceOrError.source(), mockDeviceCapturesEnabled);
112#else
113 sendUpdate(clientIdentifier, WebCore::SpeechRecognitionUpdateType::Error, WebCore::SpeechRecognitionError { WebCore::SpeechRecognitionErrorType::AudioCapture, "Audio capture is not implemented"_s });
114#endif
115}

117void SpeechRecognitionServer::stop(WebCore::SpeechRecognitionConnectionClientIdentifier clientIdentifier)
118{
119 MESSAGE_CHECK(clientIdentifier);
120
121 if (m_requests.remove(clientIdentifier)) {
122 sendUpdate(clientIdentifier, WebCore::SpeechRecognitionUpdateType::End);
123 return;
124 }
125
126 if (m_recognizer && m_recognizer->clientIdentifier() == clientIdentifier)
127 m_recognizer->stop();
128}
129
130void SpeechRecognitionServer::abort(WebCore::SpeechRecognitionConnectionClientIdentifier clientIdentifier)
131{
132 MESSAGE_CHECK(clientIdentifier);
133 if (m_requests.remove(clientIdentifier)) {
134 sendUpdate(clientIdentifier, WebCore::SpeechRecognitionUpdateType::End);
135 return;
136 }
137
138 if (m_recognizer && m_recognizer->clientIdentifier() == clientIdentifier)
139 m_recognizer->abort();
140}
141
142void SpeechRecognitionServer::invalidate(WebCore::SpeechRecognitionConnectionClientIdentifier clientIdentifier)
143{
144 MESSAGE_CHECK(clientIdentifier);
145 if (m_recognizer && m_recognizer->clientIdentifier() == clientIdentifier)
146 m_recognizer->abort();
147}

149void SpeechRecognitionServer::sendUpdate(WebCore::SpeechRecognitionConnectionClientIdentifier clientIdentifier, WebCore::SpeechRecognitionUpdateType type, std::optional<WebCore::SpeechRecognitionError> error, std::optional<Vector<WebCore::SpeechRecognitionResultData>> result)
150{
151 auto update = WebCore::SpeechRecognitionUpdate::create(clientIdentifier, type);
152 if (type == WebCore::SpeechRecognitionUpdateType::Error)
153 update = WebCore::SpeechRecognitionUpdate::createError(clientIdentifier, *error);
154 if (type == WebCore::SpeechRecognitionUpdateType::Result)
155 update = WebCore::SpeechRecognitionUpdate::createResult(clientIdentifier, *result);
156 sendUpdate(update);
157}
158
159void SpeechRecognitionServer::sendUpdate(const WebCore::SpeechRecognitionUpdate& update)
160{
161 send(Messages::WebSpeechRecognitionConnection::DidReceiveUpdate(update));
162}

174void SpeechRecognitionServer::mute()
175{
176 if (!m_recognizer)
177 return;
178
179 if (auto* source = m_recognizer->source())
180 source->mute();
181}
