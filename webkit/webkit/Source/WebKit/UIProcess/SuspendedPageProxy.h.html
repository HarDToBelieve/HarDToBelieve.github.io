55class SuspendedPageProxy final: public IPC::MessageReceiver, public IPC::MessageSender {
56 WTF_MAKE_FAST_ALLOCATED;
57public:
58 SuspendedPageProxy(WebPageProxy&, Ref<WebProcessProxy>&&, WebCore::FrameIdentifier mainFrameID, ShouldDelayClosingUntilFirstLayerFlush);

60
61 static RefPtr<WebProcessProxy> findReusableSuspendedPageProcess(WebProcessPool&, const WebCore::RegistrableDomain&, WebsiteDataStore&);
62
63 WebPageProxy& page() const { return m_page; }
64 WebCore::PageIdentifier webPageID() const { return m_webPageID; }
65 WebProcessProxy& process() const { return m_process.get(); }
66 WebCore::FrameIdentifier mainFrameID() const { return m_mainFrameID; }
67
68 WebBackForwardCache& backForwardCache() const;
69
70 bool pageIsClosedOrClosing() const;
71
72 void waitUntilReadyToUnsuspend(CompletionHandler<void(SuspendedPageProxy*)>&&);
73 void unsuspend();
74
75 void pageDidFirstLayerFlush();
76 void closeWithoutFlashing();

78#if HAVE(VISIBILITY_PROPAGATION_VIEW)
79 LayerHostingContextID contextIDForVisibilityPropagationInWebProcess() const { return m_contextIDForVisibilityPropagationInWebProcess; }

83#endif

85#if !LOG_DISABLED
86 const char* loggingString() const;

91 void didProcessRequestToSuspend(SuspensionState);

101 IPC::Connection* messageSenderConnection() const final;
102 uint64_t messageSenderDestinationID() const final;
103 bool sendMessage(UniqueRef<IPC::Encoder>&&, OptionSet<IPC::SendOption>, std::optional<std::pair<CompletionHandler<void(IPC::Decoder*)>, uint64_t>>&&) final;
104
105 WebPageProxy& m_page;
106 WebCore::PageIdentifier m_webPageID;
107 Ref<WebProcessProxy> m_process;
108 WebCore::FrameIdentifier m_mainFrameID;
109 bool m_isClosed { false };

111 bool m_shouldCloseWhenEnteringAcceleratedCompositingMode { false };

113 SuspensionState m_suspensionState { SuspensionState::Suspending };

122 LayerHostingContextID m_contextIDForVisibilityPropagationInGPUProcess { 0 };
123#endif
124#endif
