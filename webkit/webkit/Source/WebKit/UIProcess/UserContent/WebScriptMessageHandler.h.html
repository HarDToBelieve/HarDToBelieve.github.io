51class WebScriptMessageHandler : public RefCounted<WebScriptMessageHandler>, public Identified<WebScriptMessageHandler> {
52public:
53 class Client {
54 public:
55 virtual ~Client() { }
56 virtual void didPostMessage(WebPageProxy&, FrameInfoData&&, API::ContentWorld&, WebCore::SerializedScriptValue&) = 0;
57 virtual bool supportsAsyncReply() = 0;
58 virtual void didPostMessageWithAsyncReply(WebPageProxy&, FrameInfoData&&, API::ContentWorld&, WebCore::SerializedScriptValue&, WTF::Function<void(API::SerializedScriptValue*, const String&)>&&) = 0;
59 };
60
61 static Ref<WebScriptMessageHandler> create(std::unique_ptr<Client>, const String& name, API::ContentWorld&);
62 virtual ~WebScriptMessageHandler();
63
64 String name() const { return m_name; }
65
66 API::ContentWorld& world() { return m_world.get(); }
67
68 Client& client() const { return *m_client; }
69
70private:
71 WebScriptMessageHandler(std::unique_ptr<Client>, const String&, API::ContentWorld&);
72
73 std::unique_ptr<Client> m_client;
74 String m_name;
75 Ref<API::ContentWorld> m_world;
76};
