34namespace WebCore {
35class SecurityOrigin;
36}
37
38namespace WebKit {
39
40class UserMediaPermissionCheckProxy : public API::ObjectImpl<API::Object::Type::UserMediaPermissionCheck> {
41public:
42 enum class PermissionInfo { Error, Unknown, Granted };
43 using CompletionHandler = WTF::CompletionHandler<void(PermissionInfo)>;
44
45 static Ref<UserMediaPermissionCheckProxy> create(WebCore::FrameIdentifier frameID, CompletionHandler&& handler, Ref<WebCore::SecurityOrigin>&& userMediaDocumentOrigin, Ref<WebCore::SecurityOrigin>&& topLevelDocumentOrigin)
46 {
47 return adoptRef(*new UserMediaPermissionCheckProxy(frameID, WTFMove(handler), WTFMove(userMediaDocumentOrigin), WTFMove(topLevelDocumentOrigin)));
48 }
49
50 void deny() { setUserMediaAccessInfo(false); }
51 void setUserMediaAccessInfo(bool);
52 void invalidate() { complete(PermissionInfo::Error); }
53
54 WebCore::FrameIdentifier frameID() const { return m_frameID; }
55 WebCore::SecurityOrigin& userMediaDocumentSecurityOrigin() { return m_userMediaDocumentSecurityOrigin.get(); }
56 WebCore::SecurityOrigin& topLevelDocumentSecurityOrigin() { return m_topLevelDocumentSecurityOrigin.get(); }
57
58private:
59 UserMediaPermissionCheckProxy(WebCore::FrameIdentifier, CompletionHandler&&, Ref<WebCore::SecurityOrigin>&& userMediaDocumentOrigin, Ref<WebCore::SecurityOrigin>&& topLevelDocumentOrigin);
60 ~UserMediaPermissionCheckProxy();
61
62 void complete(PermissionInfo);
63
64 WebCore::FrameIdentifier m_frameID;
65 CompletionHandler m_completionHandler;
66 Ref<WebCore::SecurityOrigin> m_userMediaDocumentSecurityOrigin;
67 Ref<WebCore::SecurityOrigin> m_topLevelDocumentSecurityOrigin;
68};
