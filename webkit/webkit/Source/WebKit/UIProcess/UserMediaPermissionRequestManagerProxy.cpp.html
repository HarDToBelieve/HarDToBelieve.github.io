112void UserMediaPermissionRequestManagerProxy::invalidatePendingRequests()
113{
114 if (m_currentUserMediaRequest) {
115 m_currentUserMediaRequest->invalidate();
116 m_currentUserMediaRequest = nullptr;
117 }
118
119 auto pendingUserMediaRequests = WTFMove(m_pendingUserMediaRequests);
120 for (auto& request : pendingUserMediaRequests)
121 request->invalidate();
122
123 auto pregrantedRequests = WTFMove(m_pregrantedRequests);
124 for (auto& request : pregrantedRequests)
125 request->invalidate();
126
127 m_pendingDeviceRequests.clear();
128
129#if ENABLE(MEDIA_STREAM)
130 RealtimeMediaSourceCenter::singleton().audioCaptureFactory().removeExtensiveObserver(*this);
131#endif
132}

182static uint64_t toWebCore(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason reason)
183{
184 switch (reason) {
185 case UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::NoConstraints:
186 return static_cast<uint64_t>(UserMediaRequest::MediaAccessDenialReason::NoConstraints);
187 case UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::UserMediaDisabled:
188 return static_cast<uint64_t>(UserMediaRequest::MediaAccessDenialReason::UserMediaDisabled);
189 case UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::NoCaptureDevices:
190 return static_cast<uint64_t>(UserMediaRequest::MediaAccessDenialReason::NoCaptureDevices);
191 case UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::InvalidConstraint:
192 return static_cast<uint64_t>(UserMediaRequest::MediaAccessDenialReason::InvalidConstraint);
193 case UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::HardwareError:
194 return static_cast<uint64_t>(UserMediaRequest::MediaAccessDenialReason::HardwareError);
195 case UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::PermissionDenied:
196 return static_cast<uint64_t>(UserMediaRequest::MediaAccessDenialReason::PermissionDenied);
197 case UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::OtherFailure:
198 return static_cast<uint64_t>(UserMediaRequest::MediaAccessDenialReason::OtherFailure);
199 }
200
201 ASSERT_NOT_REACHED();
202 return static_cast<uint64_t>(UserMediaRequest::MediaAccessDenialReason::OtherFailure);
203}
204#endif

206void UserMediaPermissionRequestManagerProxy::denyRequest(UserMediaPermissionRequestProxy& request, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason reason, const String& invalidConstraint)
207{
208 if (!m_page.hasRunningProcess())
209 return;
210
211 ALWAYS_LOG(LOGIDENTIFIER, request.userMediaID(), ", reason: ", reason);
212
213 if (reason == UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::PermissionDenied)
214 m_deniedRequests.append(DeniedRequest { request.mainFrameID(), request.userMediaDocumentSecurityOrigin(), request.topLevelDocumentSecurityOrigin(), request.requiresAudioCapture(), request.requiresVideoCapture(), request.requiresDisplayCapture() });
215
216 if (auto callback = request.decisionCompletionHandler()) {
217 callback(false);
218 return;
219 }
220
221#if ENABLE(MEDIA_STREAM)
222 if (m_pregrantedRequests.isEmpty() && request.userRequest().audioConstraints.isValid)
223 RealtimeMediaSourceCenter::singleton().audioCaptureFactory().removeExtensiveObserver(*this);
224
225 m_page.send(Messages::WebPage::UserMediaAccessWasDenied(request.userMediaID(), toWebCore(reason), invalidConstraint));
226#else
227 UNUSED_PARAM(reason);
228 UNUSED_PARAM(invalidConstraint);
229#endif
230
231 processNextUserMediaRequestIfNeeded();
232}

234void UserMediaPermissionRequestManagerProxy::grantRequest(UserMediaPermissionRequestProxy& request)
235{
236 if (!m_page.hasRunningProcess())
237 return;
238
239#if ENABLE(MEDIA_STREAM)
240 ALWAYS_LOG(LOGIDENTIFIER, request.userMediaID(), ", video: ", request.videoDevice().label(), ", audio: ", request.audioDevice().label());
241
242 if (auto callback = request.decisionCompletionHandler()) {
243 m_page.willStartCapture(request, [callback = WTFMove(callback)]() mutable {
244 callback(true);
245 });
246 m_grantedRequests.append(makeRef(request));
247 return;
248 }
249
250 auto& userMediaDocumentSecurityOrigin = request.userMediaDocumentSecurityOrigin();
251 auto& topLevelDocumentSecurityOrigin = request.topLevelDocumentSecurityOrigin();
252 m_page.websiteDataStore().deviceIdHashSaltStorage().deviceIdHashSaltForOrigin(userMediaDocumentSecurityOrigin, topLevelDocumentSecurityOrigin, [this, weakThis = makeWeakPtr(*this), request = makeRef(request)](String&&) mutable {
253 if (!weakThis)
254 return;
255 finishGrantingRequest(request);
256 });
257#else
258 UNUSED_PARAM(request);
259#endif
260}

271void UserMediaPermissionRequestManagerProxy::finishGrantingRequest(UserMediaPermissionRequestProxy& request)
272{
273 ALWAYS_LOG(LOGIDENTIFIER, request.userMediaID());
274 updateStoredRequests(request);
275
276 if (!UserMediaProcessManager::singleton().willCreateMediaStream(*this, request.hasAudioDevice(), request.hasVideoDevice())) {
277 denyRequest(request, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::OtherFailure, "Unable to extend sandbox.");
278 return;
279 }
280
281 m_page.willStartCapture(request, [this, weakThis = makeWeakPtr(this), strongRequest = makeRef(request)]() mutable {
282 if (!weakThis)
283 return;
284
285 // FIXME: m_hasFilteredDeviceList will trigger ondevicechange events for various documents from different origins.
286 if (m_hasFilteredDeviceList)
287 captureDevicesChanged(PermissionInfo::Granted);
288 m_hasFilteredDeviceList = false;
289
290 ++m_hasPendingCapture;
291
292 SandboxExtension::Handle handle;
293#if PLATFORM(COCOA)
294 if (!m_hasCreatedSandboxExtensionForTCCD && doesPageNeedTCCD(m_page)) {
295 SandboxExtension::createHandleForMachLookup("com.apple.tccd"_s, m_page.process().connection()->getAuditToken(), handle);
296 m_hasCreatedSandboxExtensionForTCCD = true;
297 }
298#endif
299
300 auto& request = strongRequest.get();
301 m_page.sendWithAsyncReply(Messages::WebPage::UserMediaAccessWasGranted { request.userMediaID(), request.audioDevice(), request.videoDevice(), request.deviceIdentifierHashSalt(), handle }, [this, weakThis = WTFMove(weakThis)] {
302 if (!weakThis)
303 return;
304 if (!--m_hasPendingCapture)
305 UserMediaProcessManager::singleton().revokeSandboxExtensionsIfNeeded(page().process());
306 });
307
308 processNextUserMediaRequestIfNeeded();
309 });
310}

312void UserMediaPermissionRequestManagerProxy::resetAccess(std::optional<FrameIdentifier> frameID)
313{
314 if (frameID) {
315 ALWAYS_LOG(LOGIDENTIFIER, frameID ? frameID->loggingString() : String { });
316 m_grantedRequests.removeAllMatching([frameID](const auto& grantedRequest) {
317 return grantedRequest->mainFrameID() == frameID;
318 });
319 } else
320 m_grantedRequests.clear();
321 m_pregrantedRequests.clear();
322 m_deniedRequests.clear();
323 m_hasFilteredDeviceList = false;
324#if ENABLE(MEDIA_STREAM)
325 RealtimeMediaSourceCenter::singleton().audioCaptureFactory().removeExtensiveObserver(*this);
326#endif
327}

329const UserMediaPermissionRequestProxy* UserMediaPermissionRequestManagerProxy::searchForGrantedRequest(FrameIdentifier frameID, const SecurityOrigin& userMediaDocumentOrigin, const SecurityOrigin& topLevelDocumentOrigin, bool needsAudio, bool needsVideo) const
330{
331 if (m_page.isMediaStreamCaptureMuted())
332 return nullptr;
333
334 bool checkForAudio = needsAudio;
335 bool checkForVideo = needsVideo;
336 for (const auto& grantedRequest : m_grantedRequests) {
337 if (grantedRequest->requiresDisplayCapture())
338 continue;
339 if (!grantedRequest->userMediaDocumentSecurityOrigin().isSameSchemeHostPort(userMediaDocumentOrigin))
340 continue;
341 if (!grantedRequest->topLevelDocumentSecurityOrigin().isSameSchemeHostPort(topLevelDocumentOrigin))
342 continue;
343 if (grantedRequest->frameID() != frameID)
344 continue;
345
346 if (grantedRequest->requiresVideoCapture())
347 checkForVideo = false;
348
349 if (grantedRequest->requiresAudioCapture())
350 checkForAudio = false;
351
352 if (checkForVideo || checkForAudio)
353 continue;
354
355 return grantedRequest.ptr();
356 }
357 return nullptr;
358}

360static bool isMatchingDeniedRequest(const UserMediaPermissionRequestProxy& request, const UserMediaPermissionRequestManagerProxy::DeniedRequest& deniedRequest)
361{
362 return deniedRequest.mainFrameID == request.mainFrameID()
363 && deniedRequest.userMediaDocumentOrigin->isSameSchemeHostPort(request.userMediaDocumentSecurityOrigin())
364 && deniedRequest.topLevelDocumentOrigin->isSameSchemeHostPort(request.topLevelDocumentSecurityOrigin());
365}

367bool UserMediaPermissionRequestManagerProxy::wasRequestDenied(const UserMediaPermissionRequestProxy& request, bool needsAudio, bool needsVideo, bool needsScreenCapture)
368{
369 for (const auto& deniedRequest : m_deniedRequests) {
370 if (!isMatchingDeniedRequest(request, deniedRequest))
371 continue;
372
373 if (deniedRequest.isScreenCaptureDenied && needsScreenCapture)
374 return true;
375
376 // In case we asked for both audio and video, maybe the callback would have returned true for just audio or just video.
377 if (deniedRequest.isAudioDenied && deniedRequest.isVideoDenied) {
378 if (needsAudio && needsVideo)
379 return true;
380 continue;
381 }
382
383 if (deniedRequest.isAudioDenied && needsAudio)
384 return true;
385 if (deniedRequest.isVideoDenied && needsVideo)
386 return true;
387 }
388 return false;
389}

391void UserMediaPermissionRequestManagerProxy::updateStoredRequests(UserMediaPermissionRequestProxy& request)
392{
393 if (request.requestType() == MediaStreamRequest::Type::UserMedia)
394 m_grantedRequests.append(makeRef(request));
395
396 m_deniedRequests.removeAllMatching([&request](auto& deniedRequest) {
397 if (!isMatchingDeniedRequest(request, deniedRequest))
398 return false;
399
400 if (deniedRequest.isAudioDenied && request.requiresAudioCapture())
401 return true;
402 if (deniedRequest.isVideoDenied && request.requiresVideoCapture())
403 return true;
404 if (deniedRequest.isScreenCaptureDenied && request.requiresDisplayCapture())
405 return true;
406
407 return false;
408 });
409}

412void UserMediaPermissionRequestManagerProxy::rejectionTimerFired()
413{
414 denyRequest(m_pendingRejections.takeFirst(), UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::PermissionDenied, emptyString());
415 if (!m_pendingRejections.isEmpty())
416 scheduleNextRejection();
417}

419void UserMediaPermissionRequestManagerProxy::scheduleNextRejection()
420{
421 const double mimimumDelayBeforeReplying = .25;
422 if (!m_rejectionTimer.isActive())
423 m_rejectionTimer.startOneShot(Seconds(mimimumDelayBeforeReplying + randomNumber()));
424}

426#if ENABLE(MEDIA_STREAM)
427UserMediaPermissionRequestManagerProxy::RequestAction UserMediaPermissionRequestManagerProxy::getRequestAction(const UserMediaPermissionRequestProxy& request)
428{
429 bool requestingScreenCapture = request.requestType() == MediaStreamRequest::Type::DisplayMedia;
430 bool requestingCamera = !requestingScreenCapture && request.hasVideoDevice();
431 bool requestingMicrophone = request.hasAudioDevice();
432
433 ASSERT(!(requestingScreenCapture && !request.hasVideoDevice()));
434 ASSERT(!(requestingScreenCapture && requestingMicrophone));
435
436 if (!request.isUserGesturePriviledged() && wasRequestDenied(request, requestingMicrophone, requestingCamera, requestingScreenCapture))
437 return RequestAction::Deny;
438
439 if (request.requestType() == MediaStreamRequest::Type::DisplayMedia)
440 return RequestAction::Prompt;
441
442 return searchForGrantedRequest(request.frameID(), request.userMediaDocumentSecurityOrigin(), request.topLevelDocumentSecurityOrigin(), requestingMicrophone, requestingCamera) ? RequestAction::Grant : RequestAction::Prompt;
443}
444#endif

446void UserMediaPermissionRequestManagerProxy::requestUserMediaPermissionForFrame(uint64_t userMediaID, FrameIdentifier frameID, Ref<SecurityOrigin>&& userMediaDocumentOrigin, Ref<SecurityOrigin>&& topLevelDocumentOrigin, MediaStreamRequest&& userRequest)
447{
448#if ENABLE(MEDIA_STREAM)
449 if (!m_page.hasRunningProcess())
450 return;
451
452 ALWAYS_LOG(LOGIDENTIFIER, userMediaID);
453
454 auto request = UserMediaPermissionRequestProxy::create(*this, userMediaID, m_page.mainFrame()->frameID(), frameID, WTFMove(userMediaDocumentOrigin), WTFMove(topLevelDocumentOrigin), { }, { }, WTFMove(userRequest));
455 if (m_currentUserMediaRequest) {
456 m_pendingUserMediaRequests.append(WTFMove(request));
457 return;
458 }
459
460 if (!UserMediaProcessManager::singleton().captureEnabled()) {
461 ALWAYS_LOG(LOGIDENTIFIER, "capture disabled");
462 m_pendingRejections.append(WTFMove(request));
463 scheduleNextRejection();
464 return;
465 }
466
467 startProcessingUserMediaPermissionRequest(WTFMove(request));
468#else
469 UNUSED_PARAM(userMediaID);
470 UNUSED_PARAM(frameID);
471 UNUSED_PARAM(userMediaDocumentOrigin);
472 UNUSED_PARAM(topLevelDocumentOrigin);
473 UNUSED_PARAM(userRequest);
474#endif
475}

488#if ENABLE(MEDIA_STREAM)
489void UserMediaPermissionRequestManagerProxy::startProcessingUserMediaPermissionRequest(Ref<UserMediaPermissionRequestProxy>&& request)
490{
491 if (request->userRequest().audioConstraints.isValid)
492 RealtimeMediaSourceCenter::singleton().audioCaptureFactory().addExtensiveObserver(*this);
493
494 m_currentUserMediaRequest = WTFMove(request);
495
496 auto& userMediaDocumentSecurityOrigin = m_currentUserMediaRequest->userMediaDocumentSecurityOrigin();
497 auto& topLevelDocumentSecurityOrigin = m_currentUserMediaRequest->topLevelDocumentSecurityOrigin();
498 getUserMediaPermissionInfo(m_currentUserMediaRequest->frameID(), userMediaDocumentSecurityOrigin, topLevelDocumentSecurityOrigin, [this, request = m_currentUserMediaRequest](auto permissionInfo) mutable {
499 if (!request->isPending())
500 return;
501
502 switch (permissionInfo) {
503 case PermissionInfo::Error:
504 this->denyRequest(*m_currentUserMediaRequest, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::OtherFailure);
505 return;
506 case PermissionInfo::Unknown:
507 break;
508 case PermissionInfo::Granted:
509 m_currentUserMediaRequest->setHasPersistentAccess();
510 break;
511 }
512 this->processUserMediaPermissionRequest();
513 });
514}

516void UserMediaPermissionRequestManagerProxy::processUserMediaPermissionRequest()
517{
518 ALWAYS_LOG(LOGIDENTIFIER, m_currentUserMediaRequest->userMediaID(), ", persistent access: ", m_currentUserMediaRequest->hasPersistentAccess());
519
520 auto& userMediaDocumentSecurityOrigin = m_currentUserMediaRequest->userMediaDocumentSecurityOrigin();
521 auto& topLevelDocumentSecurityOrigin = m_currentUserMediaRequest->topLevelDocumentSecurityOrigin();
522 m_page.websiteDataStore().deviceIdHashSaltStorage().deviceIdHashSaltForOrigin(userMediaDocumentSecurityOrigin, topLevelDocumentSecurityOrigin, [this, request = m_currentUserMediaRequest] (String&& deviceIDHashSalt) mutable {
523 if (!request->isPending())
524 return;
525
526 RealtimeMediaSourceCenter::InvalidConstraintsHandler invalidHandler = [this, request](const String& invalidConstraint) {
527 if (!request->isPending())
528 return;
529
530 if (!m_page.hasRunningProcess())
531 return;
532
533 processUserMediaPermissionInvalidRequest(invalidConstraint);
534 };
535
536 auto validHandler = [this, request](Vector<CaptureDevice>&& audioDevices, Vector<CaptureDevice>&& videoDevices, String&& deviceIdentifierHashSalt) mutable {
537 if (!request->isPending())
538 return;
539
540 if (!m_page.hasRunningProcess() || !m_page.mainFrame())
541 return;
542
543 processUserMediaPermissionValidRequest(WTFMove(audioDevices), WTFMove(videoDevices), WTFMove(deviceIdentifierHashSalt));
544 };
545
546 syncWithWebCorePrefs();
547
548 platformValidateUserMediaRequestConstraints(WTFMove(validHandler), WTFMove(invalidHandler), WTFMove(deviceIDHashSalt));
549 });
550}

559void UserMediaPermissionRequestManagerProxy::processUserMediaPermissionInvalidRequest(const String& invalidConstraint)
560{
561 ALWAYS_LOG(LOGIDENTIFIER, m_currentUserMediaRequest->userMediaID());
562 bool filterConstraint = !m_currentUserMediaRequest->hasPersistentAccess() && !wasGrantedVideoOrAudioAccess(m_currentUserMediaRequest->frameID(), m_currentUserMediaRequest->userMediaDocumentSecurityOrigin(), m_currentUserMediaRequest->topLevelDocumentSecurityOrigin());
563
564 denyRequest(*m_currentUserMediaRequest, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::InvalidConstraint, filterConstraint ? String { } : invalidConstraint);
565}
566
567void UserMediaPermissionRequestManagerProxy::processUserMediaPermissionValidRequest(Vector<CaptureDevice>&& audioDevices, Vector<CaptureDevice>&& videoDevices, String&& deviceIdentifierHashSalt)
568{
569 ALWAYS_LOG(LOGIDENTIFIER, m_currentUserMediaRequest->userMediaID(), ", video: ", videoDevices.size(), " audio: ", audioDevices.size());
570 if (videoDevices.isEmpty() && audioDevices.isEmpty()) {
571 denyRequest(*m_currentUserMediaRequest, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::NoConstraints, emptyString());
572 return;
573 }
574
575 m_currentUserMediaRequest->setDeviceIdentifierHashSalt(WTFMove(deviceIdentifierHashSalt));
576 m_currentUserMediaRequest->setEligibleVideoDeviceUIDs(WTFMove(videoDevices));
577 m_currentUserMediaRequest->setEligibleAudioDeviceUIDs(WTFMove(audioDevices));
578
579 auto action = getRequestAction(*m_currentUserMediaRequest);
580 ALWAYS_LOG(LOGIDENTIFIER, m_currentUserMediaRequest->userMediaID(), ", action: ", action);
581
582 if (action == RequestAction::Deny) {
583 denyRequest(*m_currentUserMediaRequest, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::PermissionDenied, emptyString());
584 return;
585 }
586
587 if (action == RequestAction::Grant) {
588 ASSERT(m_currentUserMediaRequest->requestType() != MediaStreamRequest::Type::DisplayMedia);
589
590 if (m_page.isViewVisible())
591 grantRequest(*m_currentUserMediaRequest);
592 else
593 m_pregrantedRequests.append(m_currentUserMediaRequest.releaseNonNull());
594
595 return;
596 }
597
598 if (m_page.isControlledByAutomation()) {
599 if (WebAutomationSession* automationSession = m_page.process().processPool().automationSession()) {
600 ALWAYS_LOG(LOGIDENTIFIER, m_currentUserMediaRequest->userMediaID(), ", page controlled by automation");
601 if (automationSession->shouldAllowGetUserMediaForPage(m_page))
602 grantRequest(*m_currentUserMediaRequest);
603 else
604 denyRequest(*m_currentUserMediaRequest, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::PermissionDenied);
605 return;
606 }
607 }
608
609 if (m_page.preferences().mockCaptureDevicesEnabled() && !m_page.preferences().mockCaptureDevicesPromptEnabled()) {
610 ALWAYS_LOG(LOGIDENTIFIER, m_currentUserMediaRequest->userMediaID(), ", mock devices don't require prompt");
611 grantRequest(*m_currentUserMediaRequest);
612 return;
613 }
614
615 requestSystemValidation(m_page, *m_currentUserMediaRequest, [weakThis = makeWeakPtr(this)](bool isOK) {
616 if (!weakThis)
617 return;
618
619 if (!isOK) {
620 weakThis->denyRequest(*weakThis->m_currentUserMediaRequest, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::PermissionDenied);
621 return;
622 }
623 weakThis->decidePolicyForUserMediaPermissionRequest();
624 });
625}
626
627void UserMediaPermissionRequestManagerProxy::decidePolicyForUserMediaPermissionRequest()
628{
629 if (!m_currentUserMediaRequest)
630 return;
631
632 // If page navigated, there is no need to call the page client for authorization.
633 auto* webFrame = m_page.process().webFrame(m_currentUserMediaRequest->frameID());
634
635 if (!webFrame || !SecurityOrigin::createFromString(m_page.pageLoadState().activeURL())->isSameSchemeHostPort(m_currentUserMediaRequest->topLevelDocumentSecurityOrigin())) {
636 denyRequest(*m_currentUserMediaRequest, UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::NoConstraints, emptyString());
637 return;
638 }
639
640 // FIXME: Remove webFrame, userMediaOrigin and topLevelOrigin from this uiClient API call.
641 auto userMediaOrigin = API::SecurityOrigin::create(m_currentUserMediaRequest->userMediaDocumentSecurityOrigin());
642 auto topLevelOrigin = API::SecurityOrigin::create(m_currentUserMediaRequest->topLevelDocumentSecurityOrigin());
643 m_page.uiClient().decidePolicyForUserMediaPermissionRequest(m_page, *webFrame, WTFMove(userMediaOrigin), WTFMove(topLevelOrigin), *m_currentUserMediaRequest);
644}

705bool UserMediaPermissionRequestManagerProxy::wasGrantedVideoOrAudioAccess(FrameIdentifier frameID, const SecurityOrigin& userMediaDocumentOrigin, const SecurityOrigin& topLevelDocumentOrigin)
706{
707 for (const auto& grantedRequest : m_grantedRequests) {
708 if (grantedRequest->requiresDisplayCapture())
709 continue;
710 if (!grantedRequest->userMediaDocumentSecurityOrigin().isSameSchemeHostPort(userMediaDocumentOrigin))
711 continue;
712 if (!grantedRequest->topLevelDocumentSecurityOrigin().isSameSchemeHostPort(topLevelDocumentOrigin))
713 continue;
714 if (grantedRequest->frameID() != frameID)
715 continue;
716
717 if (grantedRequest->requiresVideoCapture() || grantedRequest->requiresAudioCapture())
718 return true;
719 }
720
721 return false;
722}

738void UserMediaPermissionRequestManagerProxy::computeFilteredDeviceList(bool revealIdsAndLabels, CompletionHandler<void(Vector<CaptureDevice>&&)>&& completion)
739{
740 static const unsigned defaultMaximumCameraCount = 1;
741 static const unsigned defaultMaximumMicrophoneCount = 1;
742
743 platformGetMediaStreamDevices([this, weakThis = makeWeakPtr(this), revealIdsAndLabels, completion = WTFMove(completion)](auto&& devices) mutable {
744
745 if (!weakThis) {
746 completion({ });
747 return;
748 }
749
750 unsigned cameraCount = 0;
751 unsigned microphoneCount = 0;
752
753 Vector<CaptureDevice> filteredDevices;
754 for (const auto& device : devices) {
755 if (!device.enabled() || (device.type() != WebCore::CaptureDevice::DeviceType::Camera && device.type() != WebCore::CaptureDevice::DeviceType::Microphone && device.type() != WebCore::CaptureDevice::DeviceType::Speaker))
756 continue;
757
758 if (!revealIdsAndLabels) {
759 if (device.type() == WebCore::CaptureDevice::DeviceType::Camera && ++cameraCount > defaultMaximumCameraCount)
760 continue;
761 if (device.type() == WebCore::CaptureDevice::DeviceType::Microphone && ++microphoneCount > defaultMaximumMicrophoneCount)
762 continue;
763 if (device.type() != WebCore::CaptureDevice::DeviceType::Camera && device.type() != WebCore::CaptureDevice::DeviceType::Microphone)
764 continue;
765 } else {
766 // We only expose speakers tied to a microphone for the moment.
767 if (device.type() == WebCore::CaptureDevice::DeviceType::Speaker && !haveMicrophoneDevice(devices, device.groupId()))
768 continue;
769 }
770
771 filteredDevices.append(revealIdsAndLabels ? device : CaptureDevice({ }, device.type(), { }, { }));
772 }
773
774 m_hasFilteredDeviceList = !revealIdsAndLabels;
775 ALWAYS_LOG(LOGIDENTIFIER, filteredDevices.size(), " devices revealed");
776
777 completion(WTFMove(filteredDevices));
778 });
779}

782void UserMediaPermissionRequestManagerProxy::enumerateMediaDevicesForFrame(FrameIdentifier frameID, Ref<SecurityOrigin>&& userMediaDocumentOrigin, Ref<SecurityOrigin>&& topLevelDocumentOrigin, CompletionHandler<void(const Vector<CaptureDevice>&, const String&)>&& completionHandler)
783{
784#if ENABLE(MEDIA_STREAM)
785 ALWAYS_LOG(LOGIDENTIFIER);
786
787 auto callback = [this, frameID, userMediaDocumentOrigin, topLevelDocumentOrigin, completionHandler = WTFMove(completionHandler)](PermissionInfo permissionInfo) mutable {
788 auto callCompletionHandler = makeScopeExit([&completionHandler] {
789 completionHandler({ }, { });
790 });
791
792 bool originHasPersistentAccess;
793 switch (permissionInfo) {
794 case PermissionInfo::Error:
795 return;
796 case PermissionInfo::Unknown:
797 originHasPersistentAccess = false;
798 break;
799 case PermissionInfo::Granted:
800 originHasPersistentAccess = true;
801 break;
802 }
803
804 if (!m_page.hasRunningProcess())
805 return;
806
807 auto requestID = generateRequestID();
808 m_pendingDeviceRequests.add(requestID);
809
810 auto& requestOrigin = userMediaDocumentOrigin.get();
811 auto& topOrigin = topLevelDocumentOrigin.get();
812
813 callCompletionHandler.release();
814 m_page.websiteDataStore().deviceIdHashSaltStorage().deviceIdHashSaltForOrigin(requestOrigin, topOrigin, [this, weakThis = makeWeakPtr(*this), requestID, frameID, userMediaDocumentOrigin = WTFMove(userMediaDocumentOrigin), topLevelDocumentOrigin = WTFMove(topLevelDocumentOrigin), originHasPersistentAccess, completionHandler = WTFMove(completionHandler)](String&& deviceIDHashSalt) mutable {
815 auto callCompletionHandler = makeScopeExit([&completionHandler] {
816 completionHandler({ }, { });
817 });
818
819 if (!weakThis || !m_pendingDeviceRequests.remove(requestID))
820 return;
821
822 if (!m_page.hasRunningProcess())
823 return;
824
825 syncWithWebCorePrefs();
826
827 bool revealIdsAndLabels = originHasPersistentAccess || wasGrantedVideoOrAudioAccess(frameID, userMediaDocumentOrigin.get(), topLevelDocumentOrigin.get());
828
829 callCompletionHandler.release();
830 computeFilteredDeviceList(revealIdsAndLabels, [completionHandler = WTFMove(completionHandler), deviceIDHashSalt = WTFMove(deviceIDHashSalt)] (auto&& devices) mutable {
831 completionHandler(devices, deviceIDHashSalt);
832 });
833 });
834 };
835
836 getUserMediaPermissionInfo(frameID, WTFMove(userMediaDocumentOrigin), WTFMove(topLevelDocumentOrigin), WTFMove(callback));
837#else
838 UNUSED_PARAM(frameID);
839 UNUSED_PARAM(userMediaDocumentOrigin);
840 UNUSED_PARAM(topLevelDocumentOrigin);
841 completionHandler({ }, { });
842#endif
843}

869void UserMediaPermissionRequestManagerProxy::captureStateChanged(MediaProducer::MediaStateFlags oldState, MediaProducer::MediaStateFlags newState)
870{
871 if (!m_page.hasRunningProcess())
872 return;
873
874#if ENABLE(MEDIA_STREAM)
875 if (!m_hasPendingCapture)
876 UserMediaProcessManager::singleton().revokeSandboxExtensionsIfNeeded(page().process());
877
878 if (m_captureState == (newState & activeCaptureMask))
879 return;
880
881 ALWAYS_LOG(LOGIDENTIFIER, "state was: ", m_captureState.toRaw(), ", is now: ", (newState & activeCaptureMask).toRaw());
882 m_captureState = newState & activeCaptureMask;
883
884 Seconds interval;
885 if (m_captureState & activeCaptureMask)
886 interval = Seconds::fromHours(m_page.preferences().longRunningMediaCaptureStreamRepromptIntervalInHours());
887 else
888 interval = Seconds::fromMinutes(m_page.preferences().inactiveMediaCaptureSteamRepromptIntervalInMinutes());
889
890 if (interval == m_currentWatchdogInterval)
891 return;
892
893 ALWAYS_LOG(LOGIDENTIFIER, "watchdog set to ", interval.value());
894 m_currentWatchdogInterval = interval;
895 m_watchdogTimer.startOneShot(m_currentWatchdogInterval);
896#endif
