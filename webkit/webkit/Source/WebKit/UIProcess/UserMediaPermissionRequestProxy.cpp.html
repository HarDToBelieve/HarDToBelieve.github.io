48#if ENABLE(MEDIA_STREAM)
49static inline void setDeviceAsFirst(Vector<CaptureDevice>& devices, const String& deviceID)
50{
51 size_t index = devices.findMatching([&deviceID](const auto& device) {
52 return device.persistentId() == deviceID;
53 });
54 ASSERT(index != notFound);
55
56 if (index) {
57 auto device = devices[index];
58 ASSERT(device.enabled());
59
60 devices.remove(index);
61 devices.insert(0, WTFMove(device));
62 }
63}
64#endif

66void UserMediaPermissionRequestProxy::allow(const String& audioDeviceUID, const String& videoDeviceUID)
67{
68#if ENABLE(MEDIA_STREAM)
69 if (!audioDeviceUID.isEmpty())
70 setDeviceAsFirst(m_eligibleAudioDevices, audioDeviceUID);
71 if (!videoDeviceUID.isEmpty())
72 setDeviceAsFirst(m_eligibleVideoDevices, videoDeviceUID);
73#else
74 UNUSED_PARAM(audioDeviceUID);
75 UNUSED_PARAM(videoDeviceUID);
76#endif
77
78 allow();
79}
80
81void UserMediaPermissionRequestProxy::allow()
82{
83 ASSERT(m_manager);
84 if (!m_manager)
85 return;
86
87 m_manager->grantRequest(*this);
88 invalidate();
89}
90
91void UserMediaPermissionRequestProxy::deny(UserMediaAccessDenialReason reason)
92{
93 if (!m_manager)
94 return;
95
96 m_manager->denyRequest(*this, reason);
97 invalidate();
98}
99
100void UserMediaPermissionRequestProxy::invalidate()
101{
102 m_manager = nullptr;
103 if (m_decisionCompletionHandler)
104 m_decisionCompletionHandler(false);
105}

121String convertEnumerationToString(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason enumerationValue)
122{
123 static const NeverDestroyed<String> values[] = {
124 MAKE_STATIC_STRING_IMPL("NoConstraints"),
125 MAKE_STATIC_STRING_IMPL("UserMediaDisabled"),
126 MAKE_STATIC_STRING_IMPL("NoCaptureDevices"),
127 MAKE_STATIC_STRING_IMPL("InvalidConstraint"),
128 MAKE_STATIC_STRING_IMPL("HardwareError"),
129 MAKE_STATIC_STRING_IMPL("PermissionDenied"),
130 MAKE_STATIC_STRING_IMPL("OtherFailure"),
131 };
132 static_assert(static_cast<size_t>(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::NoConstraints) == 0, "UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::NoConstraints is not 0 as expected");
133 static_assert(static_cast<size_t>(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::UserMediaDisabled) == 1, "UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::UserMediaDisabled is not 1 as expected");
134 static_assert(static_cast<size_t>(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::NoCaptureDevices) == 2, "UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::NoCaptureDevices is not 2 as expected");
135 static_assert(static_cast<size_t>(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::InvalidConstraint) == 3, "UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::InvalidConstraint is not 3 as expected");
136 static_assert(static_cast<size_t>(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::HardwareError) == 4, "UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::HardwareError is not 4 as expected");
137 static_assert(static_cast<size_t>(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::PermissionDenied) == 5, "UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::PermissionDenied is not 5 as expected");
138 static_assert(static_cast<size_t>(UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::OtherFailure) == 6, "UserMediaPermissionRequestProxy::UserMediaAccessDenialReason::OtherFailure is not 6 as expected");
139 ASSERT(static_cast<size_t>(enumerationValue) < WTF_ARRAY_LENGTH(values));
140 return values[static_cast<size_t>(enumerationValue)];
141}

143void UserMediaPermissionRequestProxy::prompt()
144{
145#if ENABLE(MEDIA_STREAM) && PLATFORM(COCOA)
146 ASSERT(m_manager);
147 if (!m_manager) {
148 deny(UserMediaAccessDenialReason::PermissionDenied);
149 return;
150 }
151
152 if (requiresDisplayCapture()) {
153 // FIXME: Implement getDisplayMedia prompt, for now deny.
154 deny(UserMediaAccessDenialReason::PermissionDenied);
155 return;
156 }
157
158 MediaPermissionReason reason = MediaPermissionReason::Camera;
159 if (requiresAudioCapture())
160 reason = requiresVideoCapture() ? MediaPermissionReason::CameraAndMicrophone : MediaPermissionReason::Microphone;
161 alertForPermission(m_manager->page(), reason, topLevelDocumentSecurityOrigin().data(), [this, protectedThis = makeRef(*this)](bool granted) {
162 if (!granted)
163 deny(UserMediaAccessDenialReason::PermissionDenied);
164 else
165 allow();
166 });
167#else
168 deny();
169#endif
170}
