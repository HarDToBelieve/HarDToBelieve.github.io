152bool ViewGestureController::canSwipeInDirection(SwipeDirection direction) const
153{
154 if (!m_swipeGestureEnabled)
155 return false;
156
157#if ENABLE(FULLSCREEN_API)
158 if (m_webPageProxy.fullScreenManager() && m_webPageProxy.fullScreenManager()->isFullScreen())
159 return false;
160#endif
161
162 RefPtr<WebPageProxy> alternateBackForwardListSourcePage = m_alternateBackForwardListSourcePage.get();
163 auto& backForwardList = alternateBackForwardListSourcePage ? alternateBackForwardListSourcePage->backForwardList() : m_webPageProxy.backForwardList();
164 if (direction == SwipeDirection::Back)
165 return !!backForwardList.backItem();
166 return !!backForwardList.forwardItem();
167}

211void ViewGestureController::didReachNavigationTerminalState(API::Navigation* navigation)
212{
213 if (!m_pendingNavigation || navigation != m_pendingNavigation)
214 return;
215
216 if (m_snapshotRemovalTracker.isPaused() && m_snapshotRemovalTracker.hasRemovalCallback()) {
217 removeSwipeSnapshot();
218 return;
219 }
220
221 if (!m_snapshotRemovalTracker.eventOccurred(SnapshotRemovalTracker::MainFrameLoad))
222 return;
223
224 // Coming back from the back/forward cache will result in getting a load event, but no first visually non-empty layout.
225 // WebCore considers a loaded document enough to be considered visually non-empty, so that's good
226 // enough for us too.
227 m_snapshotRemovalTracker.cancelOutstandingEvent(SnapshotRemovalTracker::VisuallyNonEmptyLayout);
228
229 checkForActiveLoads();
230}
231
232void ViewGestureController::didSameDocumentNavigationForMainFrame(SameDocumentNavigationType type)
233{
234 didStartProvisionalOrSameDocumentLoadForMainFrame();
235
236 bool cancelledOutstandingEvent = false;
237
238 // Same-document navigations don't have a main frame load or first visually non-empty layout.
239 cancelledOutstandingEvent |= m_snapshotRemovalTracker.cancelOutstandingEvent(SnapshotRemovalTracker::MainFrameLoad);
240 cancelledOutstandingEvent |= m_snapshotRemovalTracker.cancelOutstandingEvent(SnapshotRemovalTracker::VisuallyNonEmptyLayout);
241
242 if (!cancelledOutstandingEvent)
243 return;
244
245 if (type != SameDocumentNavigationSessionStateReplace && type != SameDocumentNavigationSessionStatePop)
246 return;
247
248 checkForActiveLoads();
249}

251void ViewGestureController::checkForActiveLoads()
252{
253 if (m_webPageProxy.pageLoadState().isLoading()) {
254 if (!m_swipeActiveLoadMonitoringTimer.isActive())
255 m_swipeActiveLoadMonitoringTimer.startRepeating(swipeSnapshotRemovalActiveLoadMonitoringInterval);
256 return;
257 }
258
259 m_swipeActiveLoadMonitoringTimer.stop();
260 m_snapshotRemovalTracker.eventOccurred(SnapshotRemovalTracker::SubresourceLoads);
261}

268String ViewGestureController::SnapshotRemovalTracker::eventsDescription(Events event)
269{
270 StringBuilder description;
271
272 if (event & ViewGestureController::SnapshotRemovalTracker::VisuallyNonEmptyLayout)
273 description.append("VisuallyNonEmptyLayout ");
274
275 if (event & ViewGestureController::SnapshotRemovalTracker::RenderTreeSizeThreshold)
276 description.append("RenderTreeSizeThreshold ");
277
278 if (event & ViewGestureController::SnapshotRemovalTracker::RepaintAfterNavigation)
279 description.append("RepaintAfterNavigation ");
280
281 if (event & ViewGestureController::SnapshotRemovalTracker::MainFrameLoad)
282 description.append("MainFrameLoad ");
283
284 if (event & ViewGestureController::SnapshotRemovalTracker::SubresourceLoads)
285 description.append("SubresourceLoads ");
286
287 if (event & ViewGestureController::SnapshotRemovalTracker::ScrollPositionRestoration)
288 description.append("ScrollPositionRestoration ");
289
290 if (event & ViewGestureController::SnapshotRemovalTracker::SwipeAnimationEnd)
291 description.append("SwipeAnimationEnd ");
292
293 return description.toString();
294}

309void ViewGestureController::SnapshotRemovalTracker::start(Events desiredEvents, WTF::Function<void()>&& removalCallback)
310{
311 m_outstandingEvents = desiredEvents;
312 m_removalCallback = WTFMove(removalCallback);
313 m_startTime = MonotonicTime::now();
314
315 log("start");
316
317 startWatchdog(swipeSnapshotRemovalWatchdogDuration);
318
319 // Initially start out paused; we'll resume when the load is committed.
320 // This avoids processing callbacks from earlier loads.
321 pause();
322}

324void ViewGestureController::SnapshotRemovalTracker::reset()
325{
326 if (m_outstandingEvents)
327 log("reset; had outstanding events: " + eventsDescription(m_outstandingEvents));
328 m_outstandingEvents = 0;
329 m_watchdogTimer.stop();
330 m_removalCallback = nullptr;
331}

333bool ViewGestureController::SnapshotRemovalTracker:: stopWaitingForEvent(Events event, const String& logReason, ShouldIgnoreEventIfPaused shouldIgnoreEventIfPaused)
334{
335 ASSERT(hasOneBitSet(event));
336
337 if (!(m_outstandingEvents & event))
338 return false;
339
340 if (shouldIgnoreEventIfPaused == ShouldIgnoreEventIfPaused::Yes && isPaused()) {
341 log("is paused; ignoring event: " + eventsDescription(event));
342 return false;
343 }
344
345 log(logReason + eventsDescription(event));
346
347 m_outstandingEvents &= ~event;
348
349 fireRemovalCallbackIfPossible();
350 return true;
351}

355 return stopWaitingForEvent(event, "outstanding event occurred: ", shouldIgnoreEventIfPaused);

385 reset();
386 removalCallback();

414bool ViewGestureController::PendingSwipeTracker::scrollEventCanBecomeSwipe(PlatformScrollEvent event, ViewGestureController::SwipeDirection& potentialSwipeDirection)
415{
416 if (!scrollEventCanStartSwipe(event) || !scrollEventCanInfluenceSwipe(event))
417 return false;
418
419 FloatSize size = scrollEventGetScrollingDeltas(event);
420
421 if (deltaShouldCancelSwipe(size))
422 return false;
423
424 bool isPinnedToLeft = m_shouldIgnorePinnedState || m_webPageProxy.isPinnedToLeftSide();
425 bool isPinnedToRight = m_shouldIgnorePinnedState || m_webPageProxy.isPinnedToRightSide();
426
427 bool tryingToSwipeBack = size.width() > 0 && isPinnedToLeft;
428 bool tryingToSwipeForward = size.width() < 0 && isPinnedToRight;
429 if (m_webPageProxy.userInterfaceLayoutDirection() != WebCore::UserInterfaceLayoutDirection::LTR)
430 std::swap(tryingToSwipeBack, tryingToSwipeForward);
431
432 if (!tryingToSwipeBack && !tryingToSwipeForward)
433 return false;
434
435 potentialSwipeDirection = tryingToSwipeBack ? SwipeDirection::Back : SwipeDirection::Forward;
436 return m_viewGestureController.canSwipeInDirection(potentialSwipeDirection);
437}

439bool ViewGestureController::PendingSwipeTracker::handleEvent(PlatformScrollEvent event)
440{
441 LOG(ViewGestures, "PendingSwipeTracker::handleEvent - state %d", (int)m_state);
442
443 if (scrollEventCanEndSwipe(event)) {
444 reset("gesture ended");
445 return false;
446 }
447
448 if (m_state == State::None) {
449 LOG(ViewGestures, "PendingSwipeTracker::handleEvent - scroll can become swipe %d shouldIgnorePinnedState %d, page will handle scrolls %d", scrollEventCanBecomeSwipe(event, m_direction), m_shouldIgnorePinnedState, m_webPageProxy.willHandleHorizontalScrollEvents());
450
451 if (!scrollEventCanBecomeSwipe(event, m_direction))
452 return false;
453
454 if (!m_shouldIgnorePinnedState && m_webPageProxy.willHandleHorizontalScrollEvents()) {
455 m_state = State::WaitingForWebCore;
456 LOG(ViewGestures, "Swipe Start Hysteresis - waiting for WebCore to handle event");
457 }
458 }
459
460 if (m_state == State::WaitingForWebCore)
461 return false;
462
463 return tryToStartSwipe(event);
464}

466void ViewGestureController::PendingSwipeTracker::eventWasNotHandledByWebCore(PlatformScrollEvent event)
467{
468 LOG(ViewGestures, "Swipe Start Hysteresis - WebCore didn't handle event, state %d", (int)m_state);
469
470 if (m_state != State::WaitingForWebCore)
471 return;
472
473 m_state = State::None;
474 m_cumulativeDelta = FloatSize();
475 tryToStartSwipe(event);
476}
477
478bool ViewGestureController::PendingSwipeTracker::tryToStartSwipe(PlatformScrollEvent event)
479{
480 ASSERT(m_state != State::WaitingForWebCore);
481
482 if (m_state == State::None) {
483 SwipeDirection direction;
484 if (!scrollEventCanBecomeSwipe(event, direction))
485 return false;
486 }
487
488 if (!scrollEventCanInfluenceSwipe(event))
489 return false;
490
491 m_cumulativeDelta += scrollEventGetScrollingDeltas(event);
492 LOG(ViewGestures, "Swipe Start Hysteresis - consumed event, cumulative delta (%0.2f, %0.2f)", m_cumulativeDelta.width(), m_cumulativeDelta.height());
493
494 if (deltaShouldCancelSwipe(m_cumulativeDelta)) {
495 reset("cumulative delta became too vertical");
496 return false;
497 }
498
499 if (std::abs(m_cumulativeDelta.width()) >= minimumHorizontalSwipeDistance)
500 m_viewGestureController.startSwipeGesture(event, m_direction);
501 else
502 m_state = State::InsufficientMagnitude;
503
504 return true;
505}

507void ViewGestureController::PendingSwipeTracker::reset(const char* resetReasonForLogging)

509 if (m_state != State::None)
510 LOG(ViewGestures, "Swipe Start Hysteresis - reset; %s", resetReasonForLogging);
511
512 m_state = State::None;

516void ViewGestureController::startSwipeGesture(PlatformScrollEvent event, SwipeDirection direction)
517{
518 ASSERT(m_activeGestureType == ViewGestureType::None);
519
520 m_pendingSwipeTracker.reset("starting to track swipe");
521
522 m_webPageProxy.recordAutomaticNavigationSnapshot();
523
524 RefPtr<WebBackForwardListItem> targetItem = (direction == SwipeDirection::Back) ? m_webPageProxy.backForwardList().backItem() : m_webPageProxy.backForwardList().forwardItem();
525 if (!targetItem)
526 return;
527
528 trackSwipeGesture(event, direction, targetItem);
529}

538bool ViewGestureController::shouldUseSnapshotForSize(ViewSnapshot& snapshot, FloatSize swipeLayerSize, float topContentInset)
539{
540 float deviceScaleFactor = m_webPageProxy.deviceScaleFactor();
541 if (snapshot.deviceScaleFactor() != deviceScaleFactor)
542 return false;
543
544 FloatSize unobscuredSwipeLayerSizeInDeviceCoordinates = swipeLayerSize - FloatSize(0, topContentInset);
545 unobscuredSwipeLayerSizeInDeviceCoordinates.scale(deviceScaleFactor);
546 if (snapshot.size() != unobscuredSwipeLayerSizeInDeviceCoordinates)
547 return false;
548
549 return true;
550}

552void ViewGestureController::forceRepaintIfNeeded()
553{
554 if (m_activeGestureType != ViewGestureType::Swipe)
555 return;
556
557 if (m_hasOutstandingRepaintRequest)
558 return;
559
560 m_hasOutstandingRepaintRequest = true;
561
562 auto pageID = m_webPageProxy.identifier();
563 GestureID gestureID = m_currentGestureID;
564 m_webPageProxy.forceRepaint([pageID, gestureID] () {
565 if (auto gestureController = controllerForGesture(pageID, gestureID))
566 gestureController->removeSwipeSnapshot();
567 });
568}

570void ViewGestureController::willEndSwipeGesture(WebBackForwardListItem& targetItem, bool cancelled)
571{
572 m_webPageProxy.navigationGestureWillEnd(!cancelled, targetItem);
573
574 if (cancelled)
575 return;
576
577 uint64_t renderTreeSize = 0;
578 if (ViewSnapshot* snapshot = targetItem.snapshot())
579 renderTreeSize = snapshot->renderTreeSize();
580 auto renderTreeSizeThreshold = renderTreeSize * swipeSnapshotRemovalRenderTreeSizeTargetFraction;
581
582 m_didStartProvisionalLoad = false;
583 m_pendingNavigation = m_webPageProxy.goToBackForwardItem(targetItem);
584
585 auto* currentItem = m_webPageProxy.backForwardList().currentItem();
586 // The main frame will not be navigated so hide the snapshot right away.
587 if (currentItem && currentItem->itemIsClone(targetItem)) {
588 removeSwipeSnapshot();
589 return;
590 }
591
592 SnapshotRemovalTracker::Events desiredEvents = SnapshotRemovalTracker::VisuallyNonEmptyLayout
593 | SnapshotRemovalTracker::MainFrameLoad
594 | SnapshotRemovalTracker::SubresourceLoads
595 | SnapshotRemovalTracker::ScrollPositionRestoration
596 | SnapshotRemovalTracker::SwipeAnimationEnd;
597
598 if (renderTreeSizeThreshold) {
599 desiredEvents |= SnapshotRemovalTracker::RenderTreeSizeThreshold;
600 m_snapshotRemovalTracker.setRenderTreeSizeThreshold(renderTreeSizeThreshold);
601 }
602
603 m_snapshotRemovalTracker.start(desiredEvents, [this] { this->forceRepaintIfNeeded(); });
604
605 // FIXME: Like on iOS, we should ensure that even if one of the timeouts fires,
606 // we never show the old page content, instead showing the snapshot background color.
607
608 if (ViewSnapshot* snapshot = targetItem.snapshot())
609 m_backgroundColorForCurrentSnapshot = snapshot->backgroundColor();
610}
611
612void ViewGestureController::endSwipeGesture(WebBackForwardListItem* targetItem, bool cancelled)
613{
614 ASSERT(m_activeGestureType == ViewGestureType::Swipe);
615 ASSERT(targetItem);
616
617#if PLATFORM(MAC)
618 m_swipeCancellationTracker = nullptr;
619#endif
620
621 m_didCallEndSwipeGesture = true;
622
623 if (cancelled) {
624 removeSwipeSnapshot();
625 m_webPageProxy.navigationGestureDidEnd(false, *targetItem);
626 return;
627 }
628
629 m_webPageProxy.navigationGestureDidEnd(true, *targetItem);
630
631 m_snapshotRemovalTracker.eventOccurred(SnapshotRemovalTracker::SwipeAnimationEnd, SnapshotRemovalTracker::ShouldIgnoreEventIfPaused::No);
632
633 // removeSwipeSnapshot() was called between willEndSwipeGesture() and endSwipeGesture().
634 // We couldn't remove it then, because the animation was still running, but now we can!
635 if (m_removeSnapshotImmediatelyWhenGestureEnds) {
636 removeSwipeSnapshot();
637 return;
638 }
639}

641void ViewGestureController::requestRenderTreeSizeNotificationIfNeeded()
642{
643 if (!m_snapshotRemovalTracker.hasOutstandingEvent(SnapshotRemovalTracker::RenderTreeSizeThreshold))

646 auto threshold = m_snapshotRemovalTracker.renderTreeSizeThreshold();
647 auto* messageSender = m_webPageProxy.provisionalPageProxy() ? static_cast<IPC::MessageSender*>(m_webPageProxy.provisionalPageProxy()) : &m_webPageProxy;
648 messageSender->send(Messages::ViewGestureGeometryCollector::SetRenderTreeSizeNotificationThreshold(threshold));
649}

651FloatPoint ViewGestureController::scaledMagnificationOrigin(FloatPoint origin, double scale)
652{
653 FloatPoint scaledMagnificationOrigin(m_initialMagnificationOrigin);

693void ViewGestureController::endMagnificationGesture()
694{
695 if (m_activeGestureType != ViewGestureType::Magnification)
696 return;
697
698 double newMagnification = clampTo<double>(m_magnification, minMagnification, maxMagnification);
699
700 if (m_frameHandlesMagnificationGesture)
701 m_webPageProxy.scalePage(newMagnification, roundedIntPoint(m_magnificationOrigin));
702 else {
703 if (auto drawingArea = m_webPageProxy.drawingArea())
704 drawingArea->commitTransientZoom(newMagnification, scaledMagnificationOrigin(m_magnificationOrigin, newMagnification));
705 }
706
707#if PLATFORM(MAC)
708 m_webPageProxy.didEndMagnificationGesture();
709#endif
710
711 didEndGesture();
712 m_visibleContentRectIsValid = false;
713}

715double ViewGestureController::magnification() const
716{
717 if (m_activeGestureType == ViewGestureType::Magnification)
718 return m_magnification;
719
720 return m_webPageProxy.pageScaleFactor();
721}
