51class ViewSnapshot : public RefCounted<ViewSnapshot> {
52public:
53#if HAVE(IOSURFACE)
54 static Ref<ViewSnapshot> create(std::unique_ptr<WebCore::IOSurface>);
55#endif
56#if PLATFORM(GTK)
57#if USE(GTK4)
58 static Ref<ViewSnapshot> create(GRefPtr<GdkTexture>&&);
59#else
60 static Ref<ViewSnapshot> create(RefPtr<cairo_surface_t>&&);
61#endif
62#endif
63
64 ~ViewSnapshot();
65
66 void clearImage();
67 bool hasImage() const;
68
69#if HAVE(IOSURFACE)
70 id asLayerContents();
71 RetainPtr<CGImageRef> asImageForTesting();
72#endif
73
74 void setRenderTreeSize(uint64_t renderTreeSize) { m_renderTreeSize = renderTreeSize; }
75 uint64_t renderTreeSize() const { return m_renderTreeSize; }
76
77 void setBackgroundColor(const WebCore::Color& color) { m_backgroundColor = color; }
78 WebCore::Color backgroundColor() const { return m_backgroundColor; }
79
80 void setViewScrollPosition(WebCore::IntPoint scrollPosition) { m_viewScrollPosition = scrollPosition; }
81 WebCore::IntPoint viewScrollPosition() const { return m_viewScrollPosition; }
82
83 void setDeviceScaleFactor(float deviceScaleFactor) { m_deviceScaleFactor = deviceScaleFactor; }
84 float deviceScaleFactor() const { return m_deviceScaleFactor; }
85
86#if HAVE(IOSURFACE)
87 WebCore::IOSurface* surface() const { return m_surface.get(); }
88
89 size_t estimatedImageSizeInBytes() const { return m_surface ? m_surface->totalBytes() : 0; }
90 WebCore::IntSize size() const { return m_surface ? m_surface->size() : WebCore::IntSize(); }
91
92 void setSurface(std::unique_ptr<WebCore::IOSurface>);
93
94 WebCore::VolatilityState setVolatile(bool);
95#endif
96
97#if PLATFORM(GTK)
98#if USE(GTK4)
99 GdkTexture* texture() const { return m_texture.get(); }
100#else
101 cairo_surface_t* surface() const { return m_surface.get(); }
102#endif
103
104 size_t estimatedImageSizeInBytes() const;
105 WebCore::IntSize size() const;
106#endif
107
108private:
109#if HAVE(IOSURFACE)
110 explicit ViewSnapshot(std::unique_ptr<WebCore::IOSurface>);
111
112 std::unique_ptr<WebCore::IOSurface> m_surface;
113#endif
114
115#if PLATFORM(GTK)
116#if USE(GTK4)
117 explicit ViewSnapshot(GRefPtr<GdkTexture>&&);
118
119 GRefPtr<GdkTexture> m_texture;
120#else
121 explicit ViewSnapshot(RefPtr<cairo_surface_t>&&);
122
123 RefPtr<cairo_surface_t> m_surface;
124#endif
125#endif
126
127 uint64_t m_renderTreeSize;
128 float m_deviceScaleFactor;
129 WebCore::Color m_backgroundColor;
130 WebCore::IntPoint m_viewScrollPosition; // Scroll position at snapshot time. Integral to make comparison reliable.
131};
