39Ref<VisitedLinkStore> VisitedLinkStore::create()
40{
41 return adoptRef(*new VisitedLinkStore);
42}
43
44VisitedLinkStore::~VisitedLinkStore()
45{
46 RELEASE_ASSERT(m_processes.computesEmpty());
47}
48
49VisitedLinkStore::VisitedLinkStore()
50 : m_linkHashStore(*this)
51{
52}
53
54void VisitedLinkStore::addProcess(WebProcessProxy& process)
55{
56 ASSERT(!m_processes.contains(process));
57
58 if (!m_processes.add(process).isNewEntry)
59 return;
60
61 process.addMessageReceiver(Messages::VisitedLinkStore::messageReceiverName(), identifier(), *this);
62
63 if (m_linkHashStore.isEmpty())
64 return;
65
66 sendStoreHandleToProcess(process);
67}
68
69void VisitedLinkStore::removeProcess(WebProcessProxy& process)
70{
71 ASSERT(m_processes.contains(process));
72 if (!m_processes.remove(process))
73 return;
74
75 process.removeMessageReceiver(Messages::VisitedLinkStore::messageReceiverName(), identifier());
76}
77
78void VisitedLinkStore::addVisitedLinkHash(SharedStringHash linkHash)
79{
80 m_linkHashStore.scheduleAddition(linkHash);
81}
82
83bool VisitedLinkStore::containsVisitedLinkHash(WebCore::SharedStringHash linkHash)
84{
85 return m_linkHashStore.contains(linkHash);
86}
87
88void VisitedLinkStore::removeVisitedLinkHash(WebCore::SharedStringHash linkHash)
89{
90 m_linkHashStore.scheduleRemoval(linkHash);
91}
92
93void VisitedLinkStore::removeAll()
94{
95 m_linkHashStore.clear();
96
97 for (auto& process : m_processes) {
98 ASSERT(process.processPool().processes().contains(process));
99 process.send(Messages::VisitedLinkTableController::RemoveAllVisitedLinks(), identifier());
100 }
101}
102
103void VisitedLinkStore::addVisitedLinkHashFromPage(WebPageProxyIdentifier pageProxyID, SharedStringHash linkHash)
104{
105 if (auto* webPageProxy = WebProcessProxy::webPage(pageProxyID)) {
106 if (!webPageProxy->addsVisitedLinks())
107 return;
108 }
109
110 addVisitedLinkHash(linkHash);
111}
112
113void VisitedLinkStore::sendStoreHandleToProcess(WebProcessProxy& process)
114{
115 ASSERT(process.processPool().processes().contains(process));
116
117 SharedMemory::Handle handle;
118 if (!m_linkHashStore.createSharedMemoryHandle(handle))
119 return;
120
121 // FIXME: Get the actual size of data being sent from m_linkHashStore and send it in the SharedMemory::IPCHandle object.
122#if OS(DARWIN) || OS(WINDOWS)
123 uint64_t dataSize = handle.size();
124#else
125 uint64_t dataSize = 0;
126#endif
127 process.send(Messages::VisitedLinkTableController::SetVisitedLinkTable(SharedMemory::IPCHandle { WTFMove(handle), dataSize }), identifier());
128}
129
130void VisitedLinkStore::didInvalidateSharedMemory()
131{
132 for (auto& process : m_processes)
133 sendStoreHandleToProcess(process);
134}
135
136void VisitedLinkStore::didUpdateSharedStringHashes(const Vector<WebCore::SharedStringHash>& addedHashes, const Vector<WebCore::SharedStringHash>& removedHashes)
137{
138 ASSERT(!addedHashes.isEmpty() || !removedHashes.isEmpty());
139
140 for (auto& process : m_processes) {
141 ASSERT(process.processPool().processes().contains(process));
142
143 if (addedHashes.size() > 20 || !removedHashes.isEmpty())
144 process.send(Messages::VisitedLinkTableController::AllVisitedLinkStateChanged(), identifier());
145 else
146 process.send(Messages::VisitedLinkTableController::VisitedLinkStateChanged(addedHashes), identifier());
147 }
148}
