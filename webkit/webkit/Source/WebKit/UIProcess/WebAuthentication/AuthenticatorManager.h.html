48namespace WebKit {
49
50class AuthenticatorManager : public AuthenticatorTransportService::Observer, public Authenticator::Observer {
51 WTF_MAKE_FAST_ALLOCATED;
52 WTF_MAKE_NONCOPYABLE(AuthenticatorManager);
53public:
54 using Respond = Variant<Ref<WebCore::AuthenticatorResponse>, WebCore::ExceptionData>;
55 using Callback = CompletionHandler<void(Respond&&)>;
56 using TransportSet = HashSet<WebCore::AuthenticatorTransport, WTF::IntHash<WebCore::AuthenticatorTransport>, WTF::StrongEnumHashTraits<WebCore::AuthenticatorTransport>>;
57
58 using AuthenticatorTransportService::Observer::weakPtrFactory;
59 using WeakValueType = AuthenticatorTransportService::Observer::WeakValueType;
60
61 const static size_t maxTransportNumber;
62
63 AuthenticatorManager();
64 virtual ~AuthenticatorManager() = default;
65
66 void handleRequest(WebAuthenticationRequestData&&, Callback&&);
67 void cancelRequest(const WebCore::PageIdentifier&, const std::optional<WebCore::FrameIdentifier>&); // Called from WebPageProxy/WebProcessProxy.
68 void cancelRequest(const API::WebAuthenticationPanel&); // Called from panel clients.
69 void cancel(); // Called from the presenter.
70
71 virtual bool isMock() const { return false; }
72
73 void enableModernWebAuthentication();
74 void enableNativeSupport();
75
76protected:
77 RunLoop::Timer<AuthenticatorManager>& requestTimeOutTimer() { return m_requestTimeOutTimer; }
78 void clearStateAsync(); // To void cyclic dependence.
79 void clearState();
80 void invokePendingCompletionHandler(Respond&&);
81
82private:
83 enum class Mode {
84 Compatible,
85 Modern,
86 Native,
87 };
88
89 // AuthenticatorTransportService::Observer
90 void authenticatorAdded(Ref<Authenticator>&&) final;
91 void serviceStatusUpdated(WebAuthenticationStatus) final;
92
93 // Authenticator::Observer
94 void respondReceived(Respond&&) final;
95 void downgrade(Authenticator* id, Ref<Authenticator>&& downgradedAuthenticator) final;
96 void authenticatorStatusUpdated(WebAuthenticationStatus) final;
97 void requestPin(uint64_t retries, CompletionHandler<void(const WTF::String&)>&&) final;
98 void selectAssertionResponse(Vector<Ref<WebCore::AuthenticatorAssertionResponse>>&&, WebAuthenticationSource, CompletionHandler<void(WebCore::AuthenticatorAssertionResponse*)>&&) final;
99 void decidePolicyForLocalAuthenticator(CompletionHandler<void(LocalAuthenticatorPolicy)>&&) final;
100 void requestLAContextForUserVerification(CompletionHandler<void(LAContext *)>&&) final;
101 void cancelRequest() final;
102
103 // Overriden by MockAuthenticatorManager.
104 virtual UniqueRef<AuthenticatorTransportService> createService(WebCore::AuthenticatorTransport, AuthenticatorTransportService::Observer&) const;
105 // Overriden to return every exception for tests to confirm.
106 virtual void respondReceivedInternal(Respond&&) { }
107 virtual void filterTransports(TransportSet&) const;
108 virtual void runPresenterInternal(const TransportSet&);
109
110 void startDiscovery(const TransportSet&);
111 void initTimeOutTimer();
112 void timeOutTimerFired();
113 void runPanel();
114 void runPresenter();
115 void restartDiscovery();
116 TransportSet getTransports() const;
117 void dispatchPanelClientCall(Function<void(const API::WebAuthenticationPanel&)>&&) const;
118
119 // Request: We only allow one request per time. A new request will cancel any pending ones.
120 WebAuthenticationRequestData m_pendingRequestData;
121 Callback m_pendingCompletionHandler; // Should not be invoked directly, use invokePendingCompletionHandler.
122 RunLoop::Timer<AuthenticatorManager> m_requestTimeOutTimer;
123 std::unique_ptr<AuthenticatorPresenterCoordinator> m_presenter;
124
125 Vector<UniqueRef<AuthenticatorTransportService>> m_services;
126 HashSet<Ref<Authenticator>> m_authenticators;
127
128 Mode m_mode { Mode::Compatible };
129};

131} // namespace WebKit
