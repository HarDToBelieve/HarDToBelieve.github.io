41UniqueRef<AuthenticatorTransportService> AuthenticatorTransportService::create(WebCore::AuthenticatorTransport transport, Observer& observer)
42{
43 switch (transport) {
44 case WebCore::AuthenticatorTransport::Internal:
45 return makeUniqueRef<LocalService>(observer);
46 case WebCore::AuthenticatorTransport::Usb:
47 return makeUniqueRef<HidService>(observer);
48 case WebCore::AuthenticatorTransport::Nfc:
49 return makeUniqueRef<NfcService>(observer);
50 default:
51 ASSERT_NOT_REACHED();
52 return makeUniqueRef<LocalService>(observer);
53 }
54}
55
56UniqueRef<AuthenticatorTransportService> AuthenticatorTransportService::createMock(WebCore::AuthenticatorTransport transport, Observer& observer, const WebCore::MockWebAuthenticationConfiguration& configuration)
57{
58 switch (transport) {
59 case WebCore::AuthenticatorTransport::Internal:
60 return makeUniqueRef<MockLocalService>(observer, configuration);
61 case WebCore::AuthenticatorTransport::Usb:
62 return makeUniqueRef<MockHidService>(observer, configuration);
63 case WebCore::AuthenticatorTransport::Nfc:
64 return makeUniqueRef<MockNfcService>(observer, configuration);
65 default:
66 ASSERT_NOT_REACHED();
67 return makeUniqueRef<MockLocalService>(observer, configuration);
68 }
69}

71AuthenticatorTransportService::AuthenticatorTransportService(Observer& observer)

74}

76void AuthenticatorTransportService::startDiscovery()
77{
78 RunLoop::main().dispatch([weakThis = makeWeakPtr(*this)] {
79 if (!weakThis)
80 return;
81 weakThis->startDiscoveryInternal();
82 });
83}
84
85void AuthenticatorTransportService::restartDiscovery()
86{
87 RunLoop::main().dispatch([weakThis = makeWeakPtr(*this)] {
88 if (!weakThis)
89 return;
90 weakThis->restartDiscoveryInternal();
91 });
92}
