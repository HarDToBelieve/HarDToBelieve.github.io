26#pragma once
27
28#if ENABLE(WEB_AUTHN)
29
30#include "WebAuthenticationFlags.h"
31#include <WebCore/AuthenticatorTransport.h>
32#include <wtf/UniqueRef.h>
33#include <wtf/WeakPtr.h>
34
35namespace WebCore {
36struct MockWebAuthenticationConfiguration;
37}
38
39namespace WebKit {
40
41class Authenticator;
42
43class AuthenticatorTransportService : public CanMakeWeakPtr<AuthenticatorTransportService> {
44 WTF_MAKE_FAST_ALLOCATED;
45 WTF_MAKE_NONCOPYABLE(AuthenticatorTransportService);
46public:
47 class Observer : public CanMakeWeakPtr<Observer> {
48 public:
49 virtual ~Observer() = default;
50
51 virtual void authenticatorAdded(Ref<Authenticator>&&) = 0;
52 virtual void serviceStatusUpdated(WebAuthenticationStatus) = 0;
53 };
54
55 static UniqueRef<AuthenticatorTransportService> create(WebCore::AuthenticatorTransport, Observer&);
56 static UniqueRef<AuthenticatorTransportService> createMock(WebCore::AuthenticatorTransport, Observer&, const WebCore::MockWebAuthenticationConfiguration&);
57
58 virtual ~AuthenticatorTransportService() = default;
59
60 // These operations are guaranteed to execute asynchronously.
61 void startDiscovery();
62 void restartDiscovery();
63
64protected:
65 explicit AuthenticatorTransportService(Observer&);
66
67 Observer* observer() const { return m_observer.get(); }
68
69private:
70 virtual void startDiscoveryInternal() = 0;
71 // NFC service's polling is one shot. It halts after the first tags are detected.
72 // Therefore, a restart process is needed to resume polling after exceptions.
73 virtual void restartDiscoveryInternal() { };
74
75 WeakPtr<Observer> m_observer;
76};
77
78} // namespace WebKit
79
80#endif // ENABLE(WEB_AUTHN)
