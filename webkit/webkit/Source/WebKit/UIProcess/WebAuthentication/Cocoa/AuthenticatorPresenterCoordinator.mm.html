40AuthenticatorPresenterCoordinator::AuthenticatorPresenterCoordinator(const AuthenticatorManager& manager, const String& rpId, const TransportSet& transports, ClientDataType type, const String& username)
41 : m_manager(makeWeakPtr(manager))
42{
43#if HAVE(ASC_AUTH_UI)
44 m_context = adoptNS([allocASCAuthorizationPresentationContextInstance() initWithRequestContext:nullptr appIdentifier:nullptr]);
45 if ([getASCAuthorizationPresentationContextClass() instancesRespondToSelector:@selector(setServiceName:)])
46 [m_context setServiceName:rpId];
47
48 switch (type) {
49 case ClientDataType::Create: {
50 auto options = adoptNS([allocASCPublicKeyCredentialCreationOptionsInstance() init]);
51 [options setUserName:username];
52
53 if (transports.contains(AuthenticatorTransport::Internal))
54 [m_context addLoginChoice:adoptNS([allocASCPlatformPublicKeyCredentialLoginChoiceInstance() initRegistrationChoiceWithOptions:options.get()]).get()];
55 if (transports.contains(AuthenticatorTransport::Usb) || transports.contains(AuthenticatorTransport::Nfc))
56 [m_context addLoginChoice:adoptNS([allocASCSecurityKeyPublicKeyCredentialLoginChoiceInstance() initRegistrationChoiceWithOptions:options.get()]).get()];
57 break;
58 }
59 case ClientDataType::Get:
60 if ((transports.contains(AuthenticatorTransport::Usb) || transports.contains(AuthenticatorTransport::Nfc)) && !transports.contains(AuthenticatorTransport::Internal))
61 [m_context addLoginChoice:adoptNS([allocASCSecurityKeyPublicKeyCredentialLoginChoiceInstance() initAssertionPlaceholderChoice]).get()];
62 break;
63 default:
64 ASSERT_NOT_REACHED();
65 }
66
67 m_presenterDelegate = adoptNS([[WKASCAuthorizationPresenterDelegate alloc] initWithCoordinator:*this]);
68 m_presenter = adoptNS([allocASCAuthorizationPresenterInstance() init]);
69 [m_presenter setDelegate:m_presenterDelegate.get()];
70
71 auto completionHandler = makeBlockPtr([manager = m_manager] (id<ASCCredentialProtocol> credential, NSError *error) mutable {
72 if (credential || !error)
73 return;
74
75 LOG_ERROR("Couldn't complete the authenticator presentation context: %@", error);
76 // This block can be executed in another thread.
77 RunLoop::main().dispatch([manager] () mutable {
78 if (manager)
79 manager->cancel();
80 });
81 });
82
83 if (type == ClientDataType::Get && transports.contains(AuthenticatorTransport::Internal)) {
84 m_delayedPresentationNeedsSecurityKey = (transports.contains(AuthenticatorTransport::Usb) || transports.contains(AuthenticatorTransport::Nfc));
85 m_delayedPresentation = [completionHandler = WTFMove(completionHandler), this] {
86 [m_presenter presentAuthorizationWithContext:m_context.get() completionHandler:completionHandler.get()];
87 };
88 return;
89 }
90
91 [m_presenter presentAuthorizationWithContext:m_context.get() completionHandler:completionHandler.get()];
92#endif // HAVE(ASC_AUTH_UI)
93}

95AuthenticatorPresenterCoordinator::~AuthenticatorPresenterCoordinator()
96{
97 if (m_laContextHandler)
98 m_laContextHandler(nullptr);
99 if (m_responseHandler)
100 m_responseHandler(nullptr);
101 if (m_pinHandler)

103}

105void AuthenticatorPresenterCoordinator::updatePresenter(WebAuthenticationStatus status)
106{
107#if HAVE(ASC_AUTH_UI)
108 switch (status) {
109 case WebAuthenticationStatus::PinBlocked: {
110 auto error = adoptNS([[NSError alloc] initWithDomain:ASCAuthorizationErrorDomain code:ASCAuthorizationErrorAuthenticatorPermanentlyLocked userInfo:nil]);
111 m_credentialRequestHandler(nil, error.get());
112 break;
113 }
114 case WebAuthenticationStatus::PinAuthBlocked: {
115 auto error = adoptNS([[NSError alloc] initWithDomain:ASCAuthorizationErrorDomain code:ASCAuthorizationErrorAuthenticatorTemporarilyLocked userInfo:nil]);
116 m_credentialRequestHandler(nil, error.get());
117 break;
118 }
119 case WebAuthenticationStatus::PinInvalid: {
120 auto error = adoptNS([[NSError alloc] initWithDomain:ASCAuthorizationErrorDomain code:ASCAuthorizationErrorPINInvalid userInfo:nil]);
121 m_credentialRequestHandler(nil, error.get());
122 break;
123 }
124 case WebAuthenticationStatus::MultipleNFCTagsPresent: {
125 auto error = adoptNS([[NSError alloc] initWithDomain:ASCAuthorizationErrorDomain code:ASCAuthorizationErrorMultipleNFCTagsPresent userInfo:nil]);
126 [m_presenter updateInterfaceForUserVisibleError:error.get()];
127 break;
128 }
129 case WebAuthenticationStatus::LANoCredential: {
130 if (m_delayedPresentationNeedsSecurityKey) {
131 [m_context addLoginChoice:adoptNS([allocASCSecurityKeyPublicKeyCredentialLoginChoiceInstance() initAssertionPlaceholderChoice]).get()];
132 m_delayedPresentation();
133
134 break;
135 }
136
137 auto error = adoptNS([[NSError alloc] initWithDomain:ASCAuthorizationErrorDomain code:ASCAuthorizationErrorNoCredentialsFound userInfo:nil]);
138 [m_presenter presentError:error.get() forService:[m_context serviceName] completionHandler:makeBlockPtr([manager = m_manager] {
139 RunLoop::main().dispatch([manager] () mutable {
140 if (manager)
141 manager->cancel();
142 });
143 }).get()];
144 break;
145 }
146 case WebAuthenticationStatus::NoCredentialsFound: {
147 auto error = adoptNS([[NSError alloc] initWithDomain:ASCAuthorizationErrorDomain code:ASCAuthorizationErrorNoCredentialsFound userInfo:nil]);
148 [m_presenter updateInterfaceForUserVisibleError:error.get()];
149 break;
150 }
151 case WebAuthenticationStatus::LAError: {
152 auto error = adoptNS([[NSError alloc] initWithDomain:ASCAuthorizationErrorDomain code:ASCAuthorizationErrorLAError userInfo:nil]);
153 [m_presenter updateInterfaceForUserVisibleError:error.get()];
154 break;
155 }
156 case WebAuthenticationStatus::LAExcludeCredentialsMatched: {
157 auto error = adoptNS([[NSError alloc] initWithDomain:ASCAuthorizationErrorDomain code:ASCAuthorizationErrorLAExcludeCredentialsMatched userInfo:nil]);
158 [m_presenter updateInterfaceForUserVisibleError:error.get()];
159 break;
160 }
161 default:
162 ASSERT_NOT_REACHED();
163 break;
164 }
165#endif // HAVE(ASC_AUTH_UI)
166}

168void AuthenticatorPresenterCoordinator::requestPin(uint64_t, CompletionHandler<void(const String&)>&& completionHandler)
169{
170#if HAVE(ASC_AUTH_UI)
171 if (m_pinHandler)
172 m_pinHandler(String());
173 m_pinHandler = WTFMove(completionHandler);
174
175 if (m_presentedPIN)
176 return;
177 m_presentedPIN = true;
178 [m_presenter presentPINEntryInterface];
179#endif // HAVE(ASC_AUTH_UI)

182void AuthenticatorPresenterCoordinator::selectAssertionResponse(Vector<Ref<AuthenticatorAssertionResponse>>&& responses, WebAuthenticationSource source, CompletionHandler<void(AuthenticatorAssertionResponse*)>&& completionHandler)
183{
184#if HAVE(ASC_AUTH_UI)
185 if (m_responseHandler)
186 m_responseHandler(nullptr);
187 m_responseHandler = WTFMove(completionHandler);
188
189 if (source == WebAuthenticationSource::External) {
190 auto loginChoices = adoptNS([[NSMutableArray alloc] init]);
191
192 m_credentials.clear();
193 for (auto& response : responses) {
194 if (!response->name())
195 continue;
196
197 RetainPtr<NSData> userHandle;
198 if (response->userHandle())
199 userHandle = adoptNS([[NSData alloc] initWithBytes:response->userHandle()->data() length:response->userHandle()->byteLength()]);
200
201 auto loginChoice = adoptNS([allocASCSecurityKeyPublicKeyCredentialLoginChoiceInstance() initWithName:response->name() displayName:response->displayName() userHandle:userHandle.get()]);
202 [loginChoices addObject:loginChoice.get()];
203
204 m_credentials.add(response->name(), WTFMove(response));
205 }
206
207 [m_presenter updateInterfaceWithLoginChoices:loginChoices.get()];
208 return;
209 }
210
211 if (source == WebAuthenticationSource::Local) {
212 m_credentials.clear();
213 for (auto& response : responses) {
214 RetainPtr<NSData> userHandle;
215 if (response->userHandle())
216 userHandle = adoptNS([[NSData alloc] initWithBytes:response->userHandle()->data() length:response->userHandle()->byteLength()]);
217
218 auto loginChoice = adoptNS([allocASCPlatformPublicKeyCredentialLoginChoiceInstance() initWithName:response->name() displayName:response->displayName() userHandle:userHandle.get()]);
219 [m_context addLoginChoice:loginChoice.get()];
220
221 m_credentials.add(response->name(), WTFMove(response));
222 }
223
224 if (m_delayedPresentationNeedsSecurityKey)
225 [m_context addLoginChoice:adoptNS([allocASCSecurityKeyPublicKeyCredentialLoginChoiceInstance() initAssertionPlaceholderChoice]).get()];
226 m_delayedPresentation();
227 return;
228 }
229#endif // HAVE(ASC_AUTH_UI)
230}

232void AuthenticatorPresenterCoordinator::requestLAContextForUserVerification(CompletionHandler<void(LAContext *)>&& completionHandler)
233{
234 if (m_laContext) {
235 completionHandler(m_laContext.get());
236 return;

239 m_laContextHandler = WTFMove(completionHandler);

242void AuthenticatorPresenterCoordinator::dimissPresenter(WebAuthenticationResult result)
243{
244#if HAVE(ASC_AUTH_UI)
245 if (result == WebAuthenticationResult::Succeeded && m_credentialRequestHandler) {
246 // FIXME(219767): Replace the ASCAppleIDCredential with the upcoming WebAuthn credentials one.
247 // This is just a place holder to tell the UI that the ceremony succeeds.
248 m_credentialRequestHandler(adoptNS([WebKit::allocASCAppleIDCredentialInstance() initWithUser:@"" identityToken:adoptNS([[NSData alloc] init]).get() state:nil]).get(), nil);
249 return;
250 }
251
252 [m_presenter dismissWithError:nil];
253#endif // HAVE(ASC_AUTH_UI)
254}

256void AuthenticatorPresenterCoordinator::setLAContext(LAContext *context)
257{
258 if (m_laContextHandler) {
259 m_laContextHandler(context);
260 return;
261 }
262
263 m_laContext = context;
264}

266void AuthenticatorPresenterCoordinator::didSelectAssertionResponse(const String& credentialName, LAContext *context)
267{
268 auto response = m_credentials.take(credentialName);
269 if (!response)
270 return;
271
272 if (context)
273 response->setLAContext(context);
274
275 m_responseHandler(response.get());
276}
