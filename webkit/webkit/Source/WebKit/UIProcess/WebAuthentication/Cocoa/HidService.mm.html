51HidService::HidService(Observer& observer)
52 : FidoService(observer)
53{
54 m_manager = adoptCF(IOHIDManagerCreate(kCFAllocatorDefault, kIOHIDOptionsTypeNone));
55 NSDictionary *matchingDictionary = @{
56 @kIOHIDPrimaryUsagePageKey: @(kCtapHidUsagePage),
57 @kIOHIDPrimaryUsageKey: @(kCtapHidUsage),
58 };
59 IOHIDManagerSetDeviceMatching(m_manager.get(), (__bridge CFDictionaryRef)matchingDictionary);
60 IOHIDManagerRegisterDeviceMatchingCallback(m_manager.get(), deviceAddedCallback, this);
61 IOHIDManagerRegisterDeviceRemovalCallback(m_manager.get(), deviceRemovedCallback, this);
62}
63
64HidService::~HidService()
65{
66 IOHIDManagerUnscheduleFromRunLoop(m_manager.get(), CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
67 IOHIDManagerClose(m_manager.get(), kIOHIDOptionsTypeNone);
68}
69
70void HidService::startDiscoveryInternal()
71{
72 platformStartDiscovery();
73}
74
75void HidService::platformStartDiscovery()
76{
77 IOHIDManagerScheduleWithRunLoop(m_manager.get(), CFRunLoopGetCurrent(), kCFRunLoopDefaultMode);
78 IOHIDManagerOpen(m_manager.get(), kIOHIDOptionsTypeNone);
79}
80
81UniqueRef<HidConnection> HidService::createHidConnection(IOHIDDeviceRef device) const
82{
83 return makeUniqueRef<HidConnection>(device);
84}
85
86void HidService::deviceAdded(IOHIDDeviceRef device)
87{
88 getInfo(WTF::makeUnique<CtapHidDriver>(createHidConnection(device)));
89}
