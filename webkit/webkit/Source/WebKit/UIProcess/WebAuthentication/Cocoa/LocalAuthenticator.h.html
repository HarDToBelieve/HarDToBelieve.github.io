26#pragma once
27
28#if ENABLE(WEB_AUTHN)
29
30#include "Authenticator.h"
31#include "LocalConnection.h"
32#include <wtf/UniqueRef.h>
33
34OBJC_CLASS LAContext;
35
36namespace WebKit {
37
38class LocalAuthenticator final : public Authenticator {
39public:
40 // Here is the FSM.
41 // MakeCredential: Init => RequestReceived => PolicyDecided => UserVerified => (Attested) => End
42 // GetAssertion: Init => RequestReceived => ResponseSelected => UserVerified => End
43 enum class State {
44 Init,
45 RequestReceived,
46 UserVerified,
47 Attested,
48 ResponseSelected,
49 PolicyDecided,
50 };
51
52 static Ref<LocalAuthenticator> create(UniqueRef<LocalConnection>&& connection)
53 {
54 return adoptRef(*new LocalAuthenticator(WTFMove(connection)));
55 }
56
57 static void clearAllCredentials();
58
59private:
60 explicit LocalAuthenticator(UniqueRef<LocalConnection>&&);
61
62 void makeCredential() final;
63 void continueMakeCredentialAfterDecidePolicy(LocalAuthenticatorPolicy);
64 void continueMakeCredentialAfterReceivingLAContext(LAContext *);
65 void continueMakeCredentialAfterUserVerification(SecAccessControlRef, LocalConnection::UserVerification, LAContext *);
66 void continueMakeCredentialAfterAttested(Vector<uint8_t>&& credentialId, Vector<uint8_t>&& authData, NSArray *certificates, NSError *);
67
68 void getAssertion() final;
69 void continueGetAssertionAfterResponseSelected(Ref<WebCore::AuthenticatorAssertionResponse>&&);
70 void continueGetAssertionAfterUserVerification(Ref<WebCore::AuthenticatorAssertionResponse>&&, LocalConnection::UserVerification, LAContext *);
71
72 void receiveException(WebCore::ExceptionData&&, WebAuthenticationStatus = WebAuthenticationStatus::LAError) const;
73 void deleteDuplicateCredential() const;
74 bool validateUserVerification(LocalConnection::UserVerification) const;
75
76 State m_state { State::Init };
77 UniqueRef<LocalConnection> m_connection;
78 Vector<Ref<WebCore::AuthenticatorAssertionResponse>> m_existingCredentials;
79 RetainPtr<NSData> m_provisionalCredentialId;
80};
81
82} // namespace WebKit
83
84#endif // ENABLE(WEB_AUTHN)
