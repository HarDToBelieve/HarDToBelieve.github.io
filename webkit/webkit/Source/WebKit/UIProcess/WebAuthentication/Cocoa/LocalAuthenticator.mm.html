67// See https://www.w3.org/TR/webauthn/#flags.
68const uint8_t makeCredentialFlags = 0b01000101; // UP, UV and AT are set.
69const uint8_t otherMakeCredentialFlags = 0b01000001; // UP and AT are set.
70const uint8_t getAssertionFlags = 0b00000101; // UP and UV are set.
71const uint8_t otherGetAssertionFlags = 0b00000001; // UP is set.
72// Credential ID is currently SHA-1 of the corresponding public key.
73const uint16_t credentialIdLength = 20;
74const uint64_t counter = 0;
75const uint8_t aaguid[] = { 0xF2, 0x4A, 0x8E, 0x70, 0xD0, 0xD3, 0xF8, 0x2C, 0x29, 0x37, 0x32, 0x52, 0x3C, 0xC4, 0xDE, 0x5A }; // Randomly generated.

77static inline bool emptyTransportsOrContain(const Vector<AuthenticatorTransport>& transports, AuthenticatorTransport target)
78{
79 return transports.isEmpty() ? true : transports.contains(target);
80}

82// A Base64 encoded string of the Credential ID is used as the key of the hash set.
83static inline HashSet<String> produceHashSet(const Vector<PublicKeyCredentialDescriptor>& credentialDescriptors)
84{
85 HashSet<String> result;
86 for (auto& credentialDescriptor : credentialDescriptors) {
87 if (emptyTransportsOrContain(credentialDescriptor.transports, AuthenticatorTransport::Internal) && credentialDescriptor.type == PublicKeyCredentialType::PublicKey && credentialDescriptor.idVector.size() == credentialIdLength)
88 result.add(base64EncodeToString(credentialDescriptor.idVector.data(), credentialDescriptor.idVector.size()));
89 }
90 return result;
91}

95 static NeverDestroyed<Vector<uint8_t>> aaguidVector = { aaguid, aaguidLength };

120static std::optional<Vector<Ref<AuthenticatorAssertionResponse>>> getExistingCredentials(const String& rpId)
121{
122 // Search Keychain for existing credential matched the RP ID.
123 auto query = adoptNS([[NSMutableDictionary alloc] init]);
124 [query setDictionary:@{
125 (id)kSecClass: (id)kSecClassKey,
126 (id)kSecAttrKeyClass: (id)kSecAttrKeyClassPrivate,
127 (id)kSecAttrLabel: rpId,
128 (id)kSecReturnAttributes: @YES,
129 (id)kSecMatchLimit: (id)kSecMatchLimitAll,
130#if HAVE(DATA_PROTECTION_KEYCHAIN)
131 (id)kSecUseDataProtectionKeychain: @YES
132#else
133 (id)kSecAttrNoLegacy: @YES
134#endif
135 }];
136 updateQueryIfNecessary(query.get());
137
138 CFTypeRef attributesArrayRef = nullptr;
139 OSStatus status = SecItemCopyMatching((__bridge CFDictionaryRef)query.get(), &attributesArrayRef);
140 if (status && status != errSecItemNotFound)
141 return std::nullopt;
142 auto retainAttributesArray = adoptCF(attributesArrayRef);
143 NSArray *sortedAttributesArray = [(NSArray *)attributesArrayRef sortedArrayUsingComparator:^(NSDictionary *a, NSDictionary *b) {
144 return [b[(id)kSecAttrModificationDate] compare:a[(id)kSecAttrModificationDate]];
145 }];
146
147 Vector<Ref<AuthenticatorAssertionResponse>> result;
148 result.reserveInitialCapacity(sortedAttributesArray.count);
149 for (NSDictionary *attributes in sortedAttributesArray) {
150 auto decodedResponse = cbor::CBORReader::read(vectorFromNSData(attributes[(id)kSecAttrApplicationTag]));
151 if (!decodedResponse || !decodedResponse->isMap()) {
152 ASSERT_NOT_REACHED();
153 return std::nullopt;
154 }
155 auto& responseMap = decodedResponse->getMap();
156
157 auto it = responseMap.find(CBOR(fido::kEntityIdMapKey));
158 if (it == responseMap.end() || !it->second.isByteString()) {
159 ASSERT_NOT_REACHED();
160 return std::nullopt;
161 }
162 auto& userHandle = it->second.getByteString();
163
164 it = responseMap.find(CBOR(fido::kEntityNameMapKey));
165 if (it == responseMap.end() || !it->second.isString()) {
166 ASSERT_NOT_REACHED();
167 return std::nullopt;
168 }
169 auto& username = it->second.getString();
170
171 result.uncheckedAppend(AuthenticatorAssertionResponse::create(toArrayBuffer(attributes[(id)kSecAttrApplicationLabel]), toArrayBuffer(userHandle), String(username), (__bridge SecAccessControlRef)attributes[(id)kSecAttrAccessControl], AuthenticatorAttachment::Platform));
172 }
173 return result;
174}

203void LocalAuthenticator::makeCredential()
204{
205 using namespace LocalAuthenticatorInternal;
206 ASSERT(m_state == State::Init);
207 m_state = State::RequestReceived;
208 auto& creationOptions = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
209
210 // The following implements https://www.w3.org/TR/webauthn/#op-make-cred as of 5 December 2017.
211 // Skip Step 4-5 as requireResidentKey and requireUserVerification are enforced.
212 // Skip Step 9 as extensions are not supported yet.
213 // Step 8 is implicitly captured by all UnknownError exception receiveResponds.
214 // Skip Step 10 as counter is constantly 0.
215 // Step 2.
216 if (notFound == creationOptions.pubKeyCredParams.findMatching([] (auto& pubKeyCredParam) {
217 return pubKeyCredParam.type == PublicKeyCredentialType::PublicKey && pubKeyCredParam.alg == COSE::ES256;
218 })) {
219 receiveException({ NotSupportedError, "The platform attached authenticator doesn't support any provided PublicKeyCredentialParameters."_s });
220 return;
221 }
222
223 // Step 3.
224 auto existingCredentials = getExistingCredentials(creationOptions.rp.id);
225 if (!existingCredentials) {
226 receiveException({ UnknownError, makeString("Couldn't get existing credentials") });
227 return;
228 }
229 m_existingCredentials = WTFMove(*existingCredentials);
230
231 auto excludeCredentialIds = produceHashSet(creationOptions.excludeCredentials);
232 if (!excludeCredentialIds.isEmpty()) {
233 if (notFound != m_existingCredentials.findMatching([&excludeCredentialIds] (auto& credential) {
234 auto* rawId = credential->rawId();
235 ASSERT(rawId);
236 return excludeCredentialIds.contains(base64EncodeToString(rawId->data(), rawId->byteLength()));
237 })) {
238 receiveException({ NotAllowedError, "At least one credential matches an entry of the excludeCredentials list in the platform attached authenticator."_s }, WebAuthenticationStatus::LAExcludeCredentialsMatched);
239 return;
240 }
241 }
242
243 // Step 6.
244 // Get user consent.
245 if (webAuthenticationModernEnabled()) {
246 if (auto* observer = this->observer()) {
247 auto callback = [weakThis = makeWeakPtr(*this)] (LAContext *context) {
248 ASSERT(RunLoop::isMain());
249 if (!weakThis)
250 return;
251
252 weakThis->continueMakeCredentialAfterReceivingLAContext(context);
253 };
254 observer->requestLAContextForUserVerification(WTFMove(callback));
255 }
256
257 return;
258 }
259
260 if (auto* observer = this->observer()) {
261 auto callback = [weakThis = makeWeakPtr(*this)] (LocalAuthenticatorPolicy policy) {
262 ASSERT(RunLoop::isMain());
263 if (!weakThis)
264 return;
265
266 weakThis->continueMakeCredentialAfterDecidePolicy(policy);
267 };
268 observer->decidePolicyForLocalAuthenticator(WTFMove(callback));
269 }
270}

272void LocalAuthenticator::continueMakeCredentialAfterDecidePolicy(LocalAuthenticatorPolicy policy)
273{
274 ASSERT(m_state == State::RequestReceived);
275 m_state = State::PolicyDecided;
276
277 auto& creationOptions = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
278
279 if (policy == LocalAuthenticatorPolicy::Disallow) {
280 receiveRespond(ExceptionData { UnknownError, "Disallow local authenticator."_s });
281 return;
282 }
283
284 RetainPtr<SecAccessControlRef> accessControl;
285 {
286 CFErrorRef errorRef = nullptr;
287 accessControl = adoptCF(SecAccessControlCreateWithFlags(NULL, kSecAttrAccessibleWhenPasscodeSetThisDeviceOnly, kSecAccessControlPrivateKeyUsage | kSecAccessControlUserPresence, &errorRef));
288 auto retainError = adoptCF(errorRef);
289 if (errorRef) {
290 receiveException({ UnknownError, makeString("Couldn't create access control: ", String(((NSError*)errorRef).localizedDescription)) });
291 return;
292 }
293 }
294
295 SecAccessControlRef accessControlRef = accessControl.get();
296 auto callback = [accessControl = WTFMove(accessControl), weakThis = makeWeakPtr(*this)] (LocalConnection::UserVerification verification, LAContext *context) {
297 ASSERT(RunLoop::isMain());
298 if (!weakThis)
299 return;
300
301 weakThis->continueMakeCredentialAfterUserVerification(accessControl.get(), verification, context);
302 };
303 m_connection->verifyUser(creationOptions.rp.id, getClientDataType(requestData().options), accessControlRef, getUserVerificationRequirement(requestData().options), WTFMove(callback));
304}

333void LocalAuthenticator::continueMakeCredentialAfterUserVerification(SecAccessControlRef accessControlRef, LocalConnection::UserVerification verification, LAContext *context)
334{
335 using namespace LocalAuthenticatorInternal;
336
337 ASSERT(m_state == State::PolicyDecided);
338 m_state = State::UserVerified;
339 auto& creationOptions = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
340
341 if (!validateUserVerification(verification))
342 return;
343
344 // Here is the keychain schema.
345 // kSecAttrLabel: RP ID
346 // kSecAttrApplicationLabel: Credential ID (auto-gen by Keychain)
347 // kSecAttrApplicationTag: { "id": UserEntity.id, "name": UserEntity.name } (CBOR encoded)
348 // Noted, the vale of kSecAttrApplicationLabel is automatically generated by the Keychain, which is a SHA-1 hash of
349 // the public key.
350 const auto& secAttrLabel = creationOptions.rp.id;
351
352 cbor::CBORValue::MapValue userEntityMap;
353 userEntityMap[cbor::CBORValue(fido::kEntityIdMapKey)] = cbor::CBORValue(creationOptions.user.idVector);
354 userEntityMap[cbor::CBORValue(fido::kEntityNameMapKey)] = cbor::CBORValue(creationOptions.user.name);
355 auto userEntity = cbor::CBORWriter::write(cbor::CBORValue(WTFMove(userEntityMap)));
356 ASSERT(userEntity);
357 auto secAttrApplicationTag = toNSData(*userEntity);
358
359 // Step 7.
360 // The above-to-create private key will be inserted into keychain while using SEP.
361 auto privateKey = m_connection->createCredentialPrivateKey(context, accessControlRef, secAttrLabel, secAttrApplicationTag.get());
362 if (!privateKey) {
363 receiveException({ UnknownError, "Couldn't create private key."_s });
364 return;
365 }
366
367 RetainPtr<CFDataRef> publicKeyDataRef;
368 {
369 auto publicKey = adoptCF(SecKeyCopyPublicKey(privateKey.get()));
370 CFErrorRef errorRef = nullptr;
371 publicKeyDataRef = adoptCF(SecKeyCopyExternalRepresentation(publicKey.get(), &errorRef));
372 auto retainError = adoptCF(errorRef);
373 if (errorRef) {
374 receiveException({ UnknownError, makeString("Couldn't export the public key: ", String(((NSError*)errorRef).localizedDescription)) });
375 return;
376 }
377 ASSERT(((NSData *)publicKeyDataRef.get()).length == (1 + 2 * ES256FieldElementLength)); // 04 | X | Y
378 }
379 NSData *nsPublicKeyData = (NSData *)publicKeyDataRef.get();
380
381 // Query credentialId in the keychain could be racy as it is the only unique identifier
382 // of the key item. Instead we calculate that, and examine its equaity in DEBUG build.
383 Vector<uint8_t> credentialId;
384 {
385 auto digest = PAL::CryptoDigest::create(PAL::CryptoDigest::Algorithm::SHA_1);
386 digest->addBytes(nsPublicKeyData.bytes, nsPublicKeyData.length);
387 credentialId = digest->computeHash();
388 m_provisionalCredentialId = toNSData(credentialId);
389
390#ifndef NDEBUG
391 auto query = adoptNS([[NSMutableDictionary alloc] init]);
392 [query setDictionary:@{
393 (id)kSecClass: (id)kSecClassKey,
394 (id)kSecAttrKeyClass: (id)kSecAttrKeyClassPrivate,
395 (id)kSecAttrLabel: secAttrLabel,
396 (id)kSecAttrApplicationLabel: m_provisionalCredentialId.get(),
397#if HAVE(DATA_PROTECTION_KEYCHAIN)
398 (id)kSecUseDataProtectionKeychain: @YES
399#else
400 (id)kSecAttrNoLegacy: @YES
401#endif
402 }];
403 updateQueryIfNecessary(query.get());
404
405 OSStatus status = SecItemCopyMatching((__bridge CFDictionaryRef)query.get(), nullptr);
406 ASSERT(!status);
407#endif // NDEBUG
408 }
409
410 // Step 11. https://www.w3.org/TR/webauthn/#attested-credential-data
411 // credentialPublicKey
412 Vector<uint8_t> cosePublicKey;
413 {
414 // COSE Encoding
415 Vector<uint8_t> x(ES256FieldElementLength);
416 [nsPublicKeyData getBytes: x.data() range:NSMakeRange(1, ES256FieldElementLength)];
417 Vector<uint8_t> y(ES256FieldElementLength);
418 [nsPublicKeyData getBytes: y.data() range:NSMakeRange(1 + ES256FieldElementLength, ES256FieldElementLength)];
419 cosePublicKey = encodeES256PublicKeyAsCBOR(WTFMove(x), WTFMove(y));
420 }
421
422 auto flags = verification == LocalConnection::UserVerification::Presence ? otherMakeCredentialFlags : makeCredentialFlags;
423 // Step 12.
424 // Skip Apple Attestation for none attestation.
425 if (creationOptions.attestation == AttestationConveyancePreference::None) {
426 deleteDuplicateCredential();
427
428 auto authData = buildAuthData(creationOptions.rp.id, flags, counter, buildAttestedCredentialData(Vector<uint8_t>(aaguidLength, 0), credentialId, cosePublicKey));
429 auto attestationObject = buildAttestationObject(WTFMove(authData), "", { }, AttestationConveyancePreference::None);
430 receiveRespond(AuthenticatorAttestationResponse::create(credentialId, attestationObject, AuthenticatorAttachment::Platform));
431 return;
432 }
433
434 // Step 13. Apple Attestation
435 auto authData = buildAuthData(creationOptions.rp.id, flags, counter, buildAttestedCredentialData(aaguidVector(), credentialId, cosePublicKey));
436 auto nsAuthData = toNSData(authData);
437 auto callback = [credentialId = WTFMove(credentialId), authData = WTFMove(authData), weakThis = makeWeakPtr(*this)] (NSArray * _Nullable certificates, NSError * _Nullable error) mutable {
438 ASSERT(RunLoop::isMain());
439 if (!weakThis)
440 return;
441 weakThis->continueMakeCredentialAfterAttested(WTFMove(credentialId), WTFMove(authData), certificates, error);
442 };
443 m_connection->getAttestation(privateKey.get(), nsAuthData.get(), toNSData(requestData().hash).get(), WTFMove(callback));
444}

446void LocalAuthenticator::continueMakeCredentialAfterAttested(Vector<uint8_t>&& credentialId, Vector<uint8_t>&& authData, NSArray *certificates, NSError *error)
447{
448 using namespace LocalAuthenticatorInternal;
449
450 ASSERT(m_state == State::UserVerified);
451 m_state = State::Attested;
452 auto& creationOptions = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
453
454 if (error) {
455 receiveException({ UnknownError, makeString("Couldn't attest: ", String(error.localizedDescription)) });
456 return;
457 }
458 // Attestation Certificate and Attestation Issuing CA
459 ASSERT(certificates && ([certificates count] == 2));
460
461 // Step 13. Apple Attestation Cont'
462 // Assemble the attestation object:
463 // https://www.w3.org/TR/webauthn/#attestation-object
464 cbor::CBORValue::MapValue attestationStatementMap;
465 {
466 Vector<cbor::CBORValue> cborArray;
467 for (size_t i = 0; i < [certificates count]; i++)
468 cborArray.append(cbor::CBORValue(vectorFromNSData((NSData *)adoptCF(SecCertificateCopyData((__bridge SecCertificateRef)certificates[i])).get())));
469 attestationStatementMap[cbor::CBORValue("x5c")] = cbor::CBORValue(WTFMove(cborArray));
470 }
471 auto attestationObject = buildAttestationObject(WTFMove(authData), "apple", WTFMove(attestationStatementMap), creationOptions.attestation);
472
473 deleteDuplicateCredential();
474 receiveRespond(AuthenticatorAttestationResponse::create(credentialId, attestationObject, AuthenticatorAttachment::Platform));
475}

477void LocalAuthenticator::getAssertion()
478{
479 using namespace LocalAuthenticatorInternal;
480 ASSERT(m_state == State::Init);
481 m_state = State::RequestReceived;
482 auto& requestOptions = WTF::get<PublicKeyCredentialRequestOptions>(requestData().options);
483
484 // The following implements https://www.w3.org/TR/webauthn/#op-get-assertion as of 5 December 2017.
485 // Skip Step 2 as requireUserVerification is enforced.
486 // Skip Step 8 as extensions are not supported yet.
487 // Skip Step 9 as counter is constantly 0.
488 // Step 12 is implicitly captured by all UnknownError exception callbacks.
489 // Step 3-5. Unlike the spec, if an allow list is provided and there is no intersection between existing ones and the allow list, we always return NotAllowedError.
490 auto allowCredentialIds = produceHashSet(requestOptions.allowCredentials);
491 if (!requestOptions.allowCredentials.isEmpty() && allowCredentialIds.isEmpty()) {
492 receiveException({ NotAllowedError, "No matched credentials are found in the platform attached authenticator."_s }, WebAuthenticationStatus::LANoCredential);
493 return;
494 }
495
496 // Search Keychain for the RP ID.
497 auto existingCredentials = getExistingCredentials(requestOptions.rpId);
498 if (!existingCredentials) {
499 receiveException({ UnknownError, makeString("Couldn't get existing credentials") });
500 return;
501 }
502 m_existingCredentials = WTFMove(*existingCredentials);
503
504 Vector<Ref<WebCore::AuthenticatorAssertionResponse>> assertionResponses;
505 assertionResponses.reserveInitialCapacity(m_existingCredentials.size());
506 for (auto& credential : m_existingCredentials) {
507 if (allowCredentialIds.isEmpty()) {
508 assertionResponses.uncheckedAppend(credential.copyRef());
509 continue;
510 }
511
512 auto* rawId = credential->rawId();
513 if (allowCredentialIds.contains(base64EncodeToString(rawId->data(), rawId->byteLength())))
514 assertionResponses.uncheckedAppend(credential.copyRef());
515 }
516 if (assertionResponses.isEmpty()) {
517 receiveException({ NotAllowedError, "No matched credentials are found in the platform attached authenticator."_s }, WebAuthenticationStatus::LANoCredential);
518 return;
519 }
520
521 // Step 6-7. User consent is implicitly acquired by selecting responses.
522 m_connection->filterResponses(assertionResponses);
523
524 if (auto* observer = this->observer()) {
525 auto callback = [this, weakThis = makeWeakPtr(*this)] (AuthenticatorAssertionResponse* response) {
526 ASSERT(RunLoop::isMain());
527 if (!weakThis)
528 return;
529
530 auto result = m_existingCredentials.findMatching([expectedResponse = response] (auto& response) {
531 return response.ptr() == expectedResponse;
532 });
533 if (result == notFound)
534 return;
535 continueGetAssertionAfterResponseSelected(m_existingCredentials[result].copyRef());
536 };
537 observer->selectAssertionResponse(WTFMove(assertionResponses), WebAuthenticationSource::Local, WTFMove(callback));
538 }
539}

580void LocalAuthenticator::continueGetAssertionAfterUserVerification(Ref<WebCore::AuthenticatorAssertionResponse>&& response, LocalConnection::UserVerification verification, LAContext *context)
581{
582 using namespace LocalAuthenticatorInternal;
583 ASSERT(m_state == State::ResponseSelected);
584 m_state = State::UserVerified;
585
586 if (!validateUserVerification(verification))
587 return;
588
589 // Step 10.
590 auto requestOptions = WTF::get<PublicKeyCredentialRequestOptions>(requestData().options);
591 auto authData = buildAuthData(requestOptions.rpId, verification == LocalConnection::UserVerification::Presence ? otherGetAssertionFlags : getAssertionFlags, counter, { });
592
593 // Step 11.
594 RetainPtr<CFDataRef> signature;
595 auto nsCredentialId = toNSData(response->rawId());
596 {
597 NSMutableDictionary *queryDictionary = [@{
598 (id)kSecClass: (id)kSecClassKey,
599 (id)kSecAttrKeyClass: (id)kSecAttrKeyClassPrivate,
600 (id)kSecAttrApplicationLabel: nsCredentialId.get(),
601 (id)kSecReturnRef: @YES,
602#if HAVE(DATA_PROTECTION_KEYCHAIN)
603 (id)kSecUseDataProtectionKeychain: @YES
604#else
605 (id)kSecAttrNoLegacy: @YES
606#endif
607 } mutableCopy];
608
609 if (context)
610 queryDictionary[(id)kSecUseAuthenticationContext] = context;
611
612 auto query = adoptNS(queryDictionary);
613 updateQueryIfNecessary(query.get());
614
615 CFTypeRef privateKeyRef = nullptr;
616 OSStatus status = SecItemCopyMatching((__bridge CFDictionaryRef)query.get(), &privateKeyRef);
617 if (status) {
618 receiveException({ UnknownError, makeString("Couldn't get the private key reference: ", status) });
619 return;
620 }
621 auto privateKey = adoptCF(privateKeyRef);
622
623 NSMutableData *dataToSign = [NSMutableData dataWithBytes:authData.data() length:authData.size()];
624 [dataToSign appendBytes:requestData().hash.data() length:requestData().hash.size()];
625
626 CFErrorRef errorRef = nullptr;
627 // FIXME: Converting CFTypeRef to SecKeyRef is quite subtle here.
628 signature = adoptCF(SecKeyCreateSignature((__bridge SecKeyRef)((id)privateKeyRef), kSecKeyAlgorithmECDSASignatureMessageX962SHA256, (__bridge CFDataRef)dataToSign, &errorRef));
629 auto retainError = adoptCF(errorRef);
630 if (errorRef) {
631 receiveException({ UnknownError, makeString("Couldn't generate the signature: ", String(((NSError*)errorRef).localizedDescription)) });
632 return;
633 }
634 }
635
636 // Extra step: update the Keychain item with the same value to update its modification date such that LRU can be used
637 // for selectAssertionResponse
638 auto query = adoptNS([[NSMutableDictionary alloc] init]);
639 [query setDictionary:@{
640 (id)kSecClass: (id)kSecClassKey,
641 (id)kSecAttrKeyClass: (id)kSecAttrKeyClassPrivate,
642 (id)kSecAttrApplicationLabel: nsCredentialId.get(),
643#if HAVE(DATA_PROTECTION_KEYCHAIN)
644 (id)kSecUseDataProtectionKeychain: @YES
645#else
646 (id)kSecAttrNoLegacy: @YES
647#endif
648 }];
649 updateQueryIfNecessary(query.get());
650
651 NSDictionary *updateParams = @{
652 (id)kSecAttrLabel: requestOptions.rpId,
653 };
654 auto status = SecItemUpdate((__bridge CFDictionaryRef)query.get(), (__bridge CFDictionaryRef)updateParams);
655 if (status)
656 LOG_ERROR("Couldn't update the Keychain item: %d", status);
657
658 // Step 13.
659 response->setAuthenticatorData(WTFMove(authData));
660 response->setSignature(toArrayBuffer((NSData *)signature.get()));
661 receiveRespond(WTFMove(response));
662}

664void LocalAuthenticator::receiveException(ExceptionData&& exception, WebAuthenticationStatus status) const
665{
666 LOG_ERROR(exception.message.utf8().data());
667
668 // Roll back the just created credential.
669 if (m_provisionalCredentialId) {
670 auto query = adoptNS([[NSMutableDictionary alloc] init]);
671 [query setDictionary:@{
672 (id)kSecClass: (id)kSecClassKey,
673 (id)kSecAttrApplicationLabel: m_provisionalCredentialId.get(),
674#if HAVE(DATA_PROTECTION_KEYCHAIN)
675 (id)kSecUseDataProtectionKeychain: @YES
676#else
677 (id)kSecAttrNoLegacy: @YES
678#endif
679 }];
680 updateQueryIfNecessary(query.get());
681
682 OSStatus status = SecItemDelete((__bridge CFDictionaryRef)query.get());
683 if (status)
684 LOG_ERROR(makeString("Couldn't delete provisional credential while handling error: "_s, status).utf8().data());
685 }
686
687 if (auto* observer = this->observer())
688 observer->authenticatorStatusUpdated(status);
689
690 receiveRespond(WTFMove(exception));
691 return;
692}

694void LocalAuthenticator::deleteDuplicateCredential() const
695{
696 using namespace LocalAuthenticatorInternal;
697
698 auto& creationOptions = WTF::get<PublicKeyCredentialCreationOptions>(requestData().options);
699 m_existingCredentials.findMatching([creationOptions] (auto& credential) {
700 auto* userHandle = credential->userHandle();
701 ASSERT(userHandle);
702 if (userHandle->byteLength() != creationOptions.user.idVector.size())
703 return false;
704 if (memcmp(userHandle->data(), creationOptions.user.idVector.data(), userHandle->byteLength()))
705 return false;
706
707 auto query = adoptNS([[NSMutableDictionary alloc] init]);
708 [query setDictionary:@{
709 (id)kSecClass: (id)kSecClassKey,
710 (id)kSecAttrApplicationLabel: toNSData(credential->rawId()).get(),
711#if HAVE(DATA_PROTECTION_KEYCHAIN)
712 (id)kSecUseDataProtectionKeychain: @YES
713#else
714 (id)kSecAttrNoLegacy: @YES
715#endif
716 }];
717 updateQueryIfNecessary(query.get());
718
719 OSStatus status = SecItemDelete((__bridge CFDictionaryRef)query.get());
720 if (status && status != errSecItemNotFound)
721 LOG_ERROR(makeString("Couldn't delete older credential: "_s, status).utf8().data());
722 return true;
723 });
724}

726bool LocalAuthenticator::validateUserVerification(LocalConnection::UserVerification verification) const
727{
728 if (verification == LocalConnection::UserVerification::Cancel) {
729 if (auto* observer = this->observer())
730 observer->cancelRequest();
731 return false;
732 }
733
734 if (verification == LocalConnection::UserVerification::No) {
735 receiveException({ NotAllowedError, "Couldn't verify user."_s });
736 return false;
737 }
738
739 return true;
740}
