27
28#if ENABLE(WEB_AUTHN)
29
30#include <wtf/CompletionHandler.h>
31#include <wtf/FastMalloc.h>
32#include <wtf/Noncopyable.h>
33#include <wtf/Vector.h>
34#include <wtf/text/WTFString.h>
35
36OBJC_CLASS LAContext;
37
38namespace WebCore {
39class AuthenticatorAssertionResponse;
40enum class ClientDataType : bool;
41enum class UserVerificationRequirement;
42}
43
44namespace WebKit {
45
46// Local authenticators normally doesn't need to establish connections
47// between the platform and themselves as they are attached.
48// However, such abstraction is still provided to isolate operations
49// that are not allowed in auto test environment such that some mocking
50// mechnism can override them.
51class LocalConnection {
52 WTF_MAKE_FAST_ALLOCATED;
53 WTF_MAKE_NONCOPYABLE(LocalConnection);
54public:
55 enum class UserVerification : uint8_t {
56 No,
57 Yes,
58 Cancel,
59 Presence
60 };
61
62 using AttestationCallback = CompletionHandler<void(NSArray *, NSError *)>;
63 using UserVerificationCallback = CompletionHandler<void(UserVerification, LAContext *)>;
64
65 LocalConnection() = default;
66 virtual ~LocalConnection();
67
68 // Overrided by MockLocalConnection.
69 virtual void verifyUser(const String& rpId, WebCore::ClientDataType, SecAccessControlRef, WebCore::UserVerificationRequirement, UserVerificationCallback&&);
70 virtual void verifyUser(SecAccessControlRef, LAContext *, CompletionHandler<void(UserVerification)>&&);
71 virtual RetainPtr<SecKeyRef> createCredentialPrivateKey(LAContext *, SecAccessControlRef, const String& secAttrLabel, NSData *secAttrApplicationTag) const;
72 virtual void getAttestation(SecKeyRef, NSData *authData, NSData *hash, AttestationCallback&&) const;
73 virtual void filterResponses(Vector<Ref<WebCore::AuthenticatorAssertionResponse>>&) const { };
74
75private:
76 RetainPtr<LAContext> m_context;
77};
78
79} // namespace WebKit
80
81#endif // ENABLE(WEB_AUTHN)
