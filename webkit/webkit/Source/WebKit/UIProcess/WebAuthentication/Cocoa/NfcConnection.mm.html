41inline bool compareVersion(NSData *data, const uint8_t version[], size_t versionSize)
42{
43 if (!data)
44 return false;
45 if (data.length != versionSize)
46 return false;
47 return !memcmp(data.bytes, version, versionSize);
48}
49
50// Confirm the FIDO applet is avaliable.
51// https://fidoalliance.org/specs/fido-v2.0-ps-20190130/fido-client-to-authenticator-protocol-v2.0-ps-20190130.html#nfc-applet-selection
52static bool trySelectFidoApplet(NFReaderSession *session)
53{
54 auto *versionData = [session transceive:adoptNS([[NSData alloc] initWithBytes:kCtapNfcAppletSelectionCommand length:sizeof(kCtapNfcAppletSelectionCommand)]).get()];
55 if (compareVersion(versionData, kCtapNfcAppletSelectionU2f, sizeof(kCtapNfcAppletSelectionU2f))
56 || compareVersion(versionData, kCtapNfcAppletSelectionCtap, sizeof(kCtapNfcAppletSelectionCtap)))
57 return true;
58
59 // Some legacy U2F keys such as Google T1 Titan don't understand the FIDO applet command. Instead,
60 // they are configured to only have the FIDO applet. Therefore, when the above command fails, we
61 // use U2F_VERSION command to double check if the connected tag can actually speak U2F, indicating
62 // we are interacting with one of these legacy keys.
63 versionData = [session transceive:adoptNS([[NSData alloc] initWithBytes:kCtapNfcU2fVersionCommand length:sizeof(kCtapNfcU2fVersionCommand)]).get()];
64 if (compareVersion(versionData, kCtapNfcAppletSelectionU2f, sizeof(kCtapNfcAppletSelectionU2f)))
65 return true;
66
67 return false;
68}

72Ref<NfcConnection> NfcConnection::create(RetainPtr<NFReaderSession>&& session, NfcService& service)
73{
74 return adoptRef(*new NfcConnection(WTFMove(session), service));
75}

77NfcConnection::NfcConnection(RetainPtr<NFReaderSession>&& session, NfcService& service)
78 : m_session(WTFMove(session))
79 , m_delegate(adoptNS([[WKNFReaderSessionDelegate alloc] initWithConnection:*this]))
80 , m_service(makeWeakPtr(service))
81 , m_retryTimer(RunLoop::main(), this, &NfcConnection::startPolling)
82{
83 [m_session setDelegate:m_delegate.get()];
84 startPolling();
85}

87NfcConnection::~NfcConnection()
88{
89 stop();
90}

92Vector<uint8_t> NfcConnection::transact(Vector<uint8_t>&& data) const
93{
94 // The method will return an empty NSData if the tag is disconnected.
95 auto *responseData = [m_session transceive:adoptNS([[NSData alloc] initWithBytes:data.data() length:data.size()]).get()];
96 return vectorFromNSData(responseData);
97}

99void NfcConnection::stop() const
100{
101 [m_session disconnectTag];
102 [m_session stopPolling];
103 [m_session endSession];
104}

106void NfcConnection::didDetectTags(NSArray *tags)
107{
108 if (!m_service || !tags.count)
109 return;
110
111 // A physical NFC tag could have multiple interfaces.
112 // Therefore, we use tagID to detect if there are multiple physical tags.
113 NSData *tagID = ((NFTag *)tags[0]).tagID;
114 for (NFTag *tag : tags) {
115 if ([tagID isEqualToData:tag.tagID])
116 continue;
117 m_service->didDetectMultipleTags();
118 restartPolling();
119 return;
120 }
121
122 // FIXME(203234): Tell users to switch to a different tag if the tag is not of type NFTagTypeGeneric4A
123 // or can't speak U2F/FIDO2.
124 for (NFTag *tag : tags) {
125 if (tag.type != NFTagTypeGeneric4A || ![m_session connectTag:tag])
126 continue;
127
128 if (!trySelectFidoApplet(m_session.get())) {
129 [m_session disconnectTag];
130 continue;
131 }
132
133 m_service->didConnectTag();
134 return;
135 }
136 restartPolling();
137}

139// NearField polling is a one shot polling. It halts after tags are detected.
140// Therefore, a restart process is needed to resume polling after error.
141void NfcConnection::restartPolling()
142{
143 [m_session stopPolling];
144 m_retryTimer.startOneShot(1_s); // Magic number to give users enough time for reactions.
145}
146
147void NfcConnection::startPolling()
148{
149 NSError *error = nil;
150 [m_session startPollingWithError:&error];
151 if (error)
152 LOG_ERROR("Couldn't start NFC reader polling: %@", error);
153}
