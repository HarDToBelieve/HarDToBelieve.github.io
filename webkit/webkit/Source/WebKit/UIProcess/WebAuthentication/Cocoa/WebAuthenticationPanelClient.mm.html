44WebAuthenticationPanelClient::WebAuthenticationPanelClient(_WKWebAuthenticationPanel *panel, id <_WKWebAuthenticationPanelDelegate> delegate)
45 : m_panel(panel)
46 , m_delegate(delegate)
47{

51 m_delegateMethods.panelSelectAssertionResponseSourceCompletionHandler = [delegate respondsToSelector:@selector(panel:selectAssertionResponse:source:completionHandler:)];

54}

61static _WKWebAuthenticationPanelUpdate wkWebAuthenticationPanelUpdate(WebAuthenticationStatus status)
62{
63 if (status == WebAuthenticationStatus::MultipleNFCTagsPresent)
64 return _WKWebAuthenticationPanelUpdateMultipleNFCTagsPresent;
65 if (status == WebAuthenticationStatus::NoCredentialsFound)
66 return _WKWebAuthenticationPanelUpdateNoCredentialsFound;
67 if (status == WebAuthenticationStatus::PinBlocked)
68 return _WKWebAuthenticationPanelUpdatePINBlocked;
69 if (status == WebAuthenticationStatus::PinAuthBlocked)
70 return _WKWebAuthenticationPanelUpdatePINAuthBlocked;
71 if (status == WebAuthenticationStatus::PinInvalid)
72 return _WKWebAuthenticationPanelUpdatePINInvalid;
73 if (status == WebAuthenticationStatus::LAError)
74 return _WKWebAuthenticationPanelUpdateLAError;
75 if (status == WebAuthenticationStatus::LAExcludeCredentialsMatched)
76 return _WKWebAuthenticationPanelUpdateLAExcludeCredentialsMatched;
77 if (status == WebAuthenticationStatus::LANoCredential)
78 return _WKWebAuthenticationPanelUpdateLANoCredential;
79 ASSERT_NOT_REACHED();
80 return _WKWebAuthenticationPanelUpdateMultipleNFCTagsPresent;
81}

83void WebAuthenticationPanelClient::updatePanel(WebAuthenticationStatus status) const
84{
85 if (!m_delegateMethods.panelUpdateWebAuthenticationPanel)
86 return;
87
88 auto delegate = m_delegate.get();
89 if (!delegate)
90 return;
91
92 [delegate panel:m_panel updateWebAuthenticationPanel:wkWebAuthenticationPanelUpdate(status)];
93}

95static _WKWebAuthenticationResult wkWebAuthenticationResult(WebAuthenticationResult result)
96{
97 switch (result) {

99 return _WKWebAuthenticationResultSucceeded;

101 return _WKWebAuthenticationResultFailed;

103 ASSERT_NOT_REACHED();
104 return _WKWebAuthenticationResultFailed;

153void WebAuthenticationPanelClient::selectAssertionResponse(Vector<Ref<WebCore::AuthenticatorAssertionResponse>>&& responses, WebAuthenticationSource source, CompletionHandler<void(WebCore::AuthenticatorAssertionResponse*)>&& completionHandler) const
154{
155 ASSERT(!responses.isEmpty());
156
157 if (!m_delegateMethods.panelSelectAssertionResponseSourceCompletionHandler) {
158 completionHandler(nullptr);
159 return;
160 }
161
162 auto delegate = m_delegate.get();
163 if (!delegate) {
164 completionHandler(nullptr);
165 return;
166 }
167
168 Vector<RefPtr<API::Object>> apiResponses;
169 apiResponses.reserveInitialCapacity(responses.size());
170 for (auto& response : responses)
171 apiResponses.uncheckedAppend(API::WebAuthenticationAssertionResponse::create(response.copyRef()));
172
173 auto checker = CompletionHandlerCallChecker::create(delegate.get(), @selector(panel:selectAssertionResponse:source:completionHandler:));
174 [delegate panel:m_panel selectAssertionResponse:wrapper(API::Array::create(WTFMove(apiResponses))) source:wkWebAuthenticationSource(source) completionHandler:makeBlockPtr([completionHandler = WTFMove(completionHandler), checker = WTFMove(checker)](_WKWebAuthenticationAssertionResponse *response) mutable {
175 if (checker->completionHandlerHasBeenCalled())
176 return;
177 checker->didCallCompletionHandler();
178
179 if (!response) {
180 completionHandler(nullptr);
181 return;
182 }
183 completionHandler(static_cast<API::WebAuthenticationAssertionResponse&>([response _apiObject]).response());
184 }).get()];
185}

188{
