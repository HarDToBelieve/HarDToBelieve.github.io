48// https://fidoalliance.org/specs/fido-v2.0-ps-20170927/fido-client-to-authenticator-protocol-v2.0-ps-20170927.html#mandatory-commands
49const size_t CtapChannelIdSize = 4;

72auto MockHidConnection::sendSync(const Vector<uint8_t>& data) -> DataSent
73{
74 ASSERT(isInitialized());
75 if (m_configuration.hid->expectCancel) {
76 auto message = FidoHidMessage::createFromSerializedData(data);
77 ASSERT_UNUSED(message, message);
78 ASSERT(message->cmd() == FidoHidDeviceCommand::kCancel);
79 LOG_ERROR("Request cancelled.");
80 }
81 return DataSent::Yes;
82}
83
84void MockHidConnection::send(Vector<uint8_t>&& data, DataSentCallback&& callback)
85{
86 ASSERT(isInitialized());
87 auto task = makeBlockPtr([weakThis = makeWeakPtr(*this), data = WTFMove(data), callback = WTFMove(callback)]() mutable {
88 ASSERT(!RunLoop::isMain());
89 RunLoop::main().dispatch([weakThis, data = WTFMove(data), callback = WTFMove(callback)]() mutable {
90 if (!weakThis) {
91 callback(DataSent::No);
92 return;
93 }
94
95 weakThis->assembleRequest(WTFMove(data));
96
97 auto sent = DataSent::Yes;
98 if (weakThis->stagesMatch() && weakThis->m_configuration.hid->error == Mock::HidError::DataNotSent)
99 sent = DataSent::No;
100 callback(sent);
101 });
102 });
103 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), task.get());
104}

106void MockHidConnection::registerDataReceivedCallbackInternal()
107{
108 if (stagesMatch() && m_configuration.hid->error == Mock::HidError::EmptyReport) {
109 receiveReport({ });
110 shouldContinueFeedReports();
111 return;
112 }
113 if (!m_configuration.hid->fastDataArrival)
114 feedReports();
115}

117void MockHidConnection::assembleRequest(Vector<uint8_t>&& data)
118{
119 if (!m_requestMessage) {
120 m_requestMessage = FidoHidMessage::createFromSerializedData(data);
121 ASSERT(m_requestMessage);
122 } else {
123 auto status = m_requestMessage->addContinuationPacket(data);
124 ASSERT_UNUSED(status, status);
125 }
126
127 if (m_requestMessage->messageComplete())
128 parseRequest();
129}
130
131void MockHidConnection::parseRequest()
132{
133 using namespace MockHidConnectionInternal;
134
135 ASSERT(m_requestMessage);
136 // Set stages.
137 if (m_requestMessage->cmd() == FidoHidDeviceCommand::kInit) {
138 auto previousSubStage = m_subStage;
139 m_subStage = Mock::HidSubStage::Init;
140 if (previousSubStage == Mock::HidSubStage::Msg)
141 m_stage = Mock::HidStage::Request;
142 }
143 if (m_requestMessage->cmd() == FidoHidDeviceCommand::kCbor || m_requestMessage->cmd() == FidoHidDeviceCommand::kMsg)
144 m_subStage = Mock::HidSubStage::Msg;
145
146 if (m_stage == Mock::HidStage::Request && m_subStage == Mock::HidSubStage::Msg) {
147 // Make sure we issue different msg cmd for CTAP and U2F.
148 if (m_configuration.hid->canDowngrade && !m_configuration.hid->isU2f)
149 m_configuration.hid->isU2f = m_requestMessage->cmd() == FidoHidDeviceCommand::kMsg;
150 ASSERT(m_configuration.hid->isU2f ^ (m_requestMessage->cmd() != FidoHidDeviceCommand::kMsg));
151
152 // Set options.
153 if (m_requestMessage->cmd() == FidoHidDeviceCommand::kCbor) {
154 m_requireResidentKey = false;
155 m_requireUserVerification = false;
156
157 auto payload = m_requestMessage->getMessagePayload();
158 ASSERT(payload.size());
159 auto cmd = static_cast<CtapRequestCommand>(payload[0]);
160 payload.remove(0);
161 auto requestMap = CBORReader::read(payload);
162 ASSERT(requestMap || cmd == CtapRequestCommand::kAuthenticatorGetNextAssertion);
163
164 if (cmd == CtapRequestCommand::kAuthenticatorMakeCredential) {
165 auto it = requestMap->getMap().find(CBORValue(CtapMakeCredentialRequestOptionsKey)); // Find options.
166 if (it != requestMap->getMap().end()) {
167 auto& optionMap = it->second.getMap();
168
169 auto itr = optionMap.find(CBORValue(kResidentKeyMapKey));
170 if (itr != optionMap.end())
171 m_requireResidentKey = itr->second.getBool();
172
173 itr = optionMap.find(CBORValue(kUserVerificationMapKey));
174 if (itr != optionMap.end())
175 m_requireUserVerification = itr->second.getBool();
176 }
177 }
178
179 if (cmd == CtapRequestCommand::kAuthenticatorGetAssertion) {
180 auto it = requestMap->getMap().find(CBORValue(CtapGetAssertionRequestOptionsKey)); // Find options.
181 if (it != requestMap->getMap().end()) {
182 auto& optionMap = it->second.getMap();
183 auto itr = optionMap.find(CBORValue(kUserVerificationMapKey));
184 if (itr != optionMap.end())
185 m_requireUserVerification = itr->second.getBool();
186 }
187 }
188 }
189 }
190
191 // Store nonce.
192 if (m_subStage == Mock::HidSubStage::Init) {
193 m_nonce = m_requestMessage->getMessagePayload();
194 ASSERT(m_nonce.size() == kHidInitNonceLength);
195 }
196
197 m_currentChannel = m_requestMessage->channelId();
198 m_requestMessage = std::nullopt;
199 if (m_configuration.hid->fastDataArrival)
200 feedReports();
201}

203void MockHidConnection::feedReports()
204{
205 using namespace MockHidConnectionInternal;
206
207 if (m_subStage == Mock::HidSubStage::Init) {
208 Vector<uint8_t> payload;
209 payload.reserveInitialCapacity(kHidInitResponseSize);
210 payload.appendVector(m_nonce);
211 size_t writePosition = payload.size();
212 if (stagesMatch() && m_configuration.hid->error == Mock::HidError::WrongNonce)
213 payload[0]--;
214 payload.grow(kHidInitResponseSize);
215 cryptographicallyRandomValues(payload.data() + writePosition, CtapChannelIdSize);
216 auto channel = kHidBroadcastChannel;
217 if (stagesMatch() && m_configuration.hid->error == Mock::HidError::WrongChannelId)
218 channel--;
219 FidoHidInitPacket initPacket(channel, FidoHidDeviceCommand::kInit, WTFMove(payload), payload.size());
220 receiveReport(initPacket.getSerializedData());
221 shouldContinueFeedReports();
222 return;
223 }
224
225 std::optional<FidoHidMessage> message;
226 if (m_stage == Mock::HidStage::Info && m_subStage == Mock::HidSubStage::Msg) {
227 // FIXME(205839):
228 Vector<uint8_t> infoData;
229 if (m_configuration.hid->canDowngrade)
230 infoData = encodeAsCBOR(AuthenticatorGetInfoResponse({ ProtocolVersion::kCtap, ProtocolVersion::kU2f }, Vector<uint8_t>(aaguidLength, 0u)));
231 else if (m_configuration.hid->supportClientPin) {
232 AuthenticatorGetInfoResponse infoResponse({ ProtocolVersion::kCtap }, Vector<uint8_t>(aaguidLength, 0u));
233 infoResponse.setPinProtocols({ pin::kProtocolVersion });
234 AuthenticatorSupportedOptions options;
235 options.setClientPinAvailability(AuthenticatorSupportedOptions::ClientPinAvailability::kSupportedAndPinSet);
236 infoResponse.setOptions(WTFMove(options));
237 infoData = encodeAsCBOR(infoResponse);
238 } else
239 infoData = encodeAsCBOR(AuthenticatorGetInfoResponse({ ProtocolVersion::kCtap }, Vector<uint8_t>(aaguidLength, 0u)));
240 infoData.insert(0, static_cast<uint8_t>(CtapDeviceResponseCode::kSuccess)); // Prepend status code.
241 if (stagesMatch() && m_configuration.hid->error == Mock::HidError::WrongChannelId)
242 message = FidoHidMessage::create(m_currentChannel - 1, FidoHidDeviceCommand::kCbor, infoData);
243 else {
244 if (!m_configuration.hid->isU2f)
245 message = FidoHidMessage::create(m_currentChannel, FidoHidDeviceCommand::kCbor, infoData);
246 else
247 message = FidoHidMessage::create(m_currentChannel, FidoHidDeviceCommand::kError, { static_cast<uint8_t>(CtapDeviceResponseCode::kCtap1ErrInvalidCommand) });
248 }
249 }
250
251 if (m_stage == Mock::HidStage::Request && m_subStage == Mock::HidSubStage::Msg) {
252 if (m_configuration.hid->expectCancel)
253 return;
254 if (m_configuration.hid->keepAlive) {
255 m_configuration.hid->keepAlive = false;
256 FidoHidInitPacket initPacket(m_currentChannel, FidoHidDeviceCommand::kKeepAlive, { CtapKeepAliveStatusProcessing }, 1);
257 receiveReport(initPacket.getSerializedData());
258 continueFeedReports();
259 return;
260 }
261 if (stagesMatch() && m_configuration.hid->error == Mock::HidError::UnsupportedOptions && (m_requireResidentKey || m_requireUserVerification))
262 message = FidoHidMessage::create(m_currentChannel, FidoHidDeviceCommand::kCbor, { static_cast<uint8_t>(CtapDeviceResponseCode::kCtap2ErrUnsupportedOption) });
263 else {
264 ASSERT(!m_configuration.hid->payloadBase64.isEmpty());
265 auto payload = base64Decode(m_configuration.hid->payloadBase64[0]);
266 m_configuration.hid->payloadBase64.remove(0);
267 if (!m_configuration.hid->isU2f)
268 message = FidoHidMessage::create(m_currentChannel, FidoHidDeviceCommand::kCbor, WTFMove(*payload));
269 else
270 message = FidoHidMessage::create(m_currentChannel, FidoHidDeviceCommand::kMsg, WTFMove(*payload));
271 }
272 }
273
274 ASSERT(message);
275 bool isFirst = true;
276 while (message->numPackets()) {
277 auto report = message->popNextPacket();
278 if (!isFirst && stagesMatch() && m_configuration.hid->error == Mock::HidError::WrongChannelId)
279 report = FidoHidContinuationPacket(m_currentChannel - 1, 0, { }).getSerializedData();
280 // Packets are feed asynchronously to mimic actual data transmission.
281 RunLoop::main().dispatch([report = WTFMove(report), weakThis = makeWeakPtr(*this)]() mutable {
282 if (!weakThis)
283 return;
284 weakThis->receiveReport(WTFMove(report));
285 });
286 isFirst = false;
287 }
288}
289
290bool MockHidConnection::stagesMatch() const
291{
292 return m_configuration.hid->stage == m_stage && m_configuration.hid->subStage == m_subStage;
293}
294
295void MockHidConnection::shouldContinueFeedReports()
296{
297 if (!m_configuration.hid->continueAfterErrorData)
298 return;
299 m_configuration.hid->continueAfterErrorData = false;
300 m_configuration.hid->error = Mock::HidError::Success;
301 continueFeedReports();
302}
303
304void MockHidConnection::continueFeedReports()
305{
306 // Send actual response for the next run.
307 RunLoop::main().dispatch([weakThis = makeWeakPtr(*this)]() mutable {
308 if (!weakThis)
309 return;
310 weakThis->feedReports();
311 });
312}
