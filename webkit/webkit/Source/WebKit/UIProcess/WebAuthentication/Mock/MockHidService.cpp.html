34namespace WebKit {
35
36MockHidService::MockHidService(Observer& observer, const WebCore::MockWebAuthenticationConfiguration& configuration)
37 : HidService(observer)
38 , m_configuration(configuration)
39{
40}
41
42void MockHidService::platformStartDiscovery()
43{
44 if (!!m_configuration.hid) {
45 deviceAdded(nullptr);
46 return;
47 }
48 LOG_ERROR("No hid authenticators is available.");
49}
50
51UniqueRef<HidConnection> MockHidService::createHidConnection(IOHIDDeviceRef device) const
52{
53 return makeUniqueRef<MockHidConnection>(device, m_configuration);
54}
