43namespace WebKit {
44
45class CtapDriver;
46
47class CtapAuthenticator final : public FidoAuthenticator {
48public:
49 static Ref<CtapAuthenticator> create(std::unique_ptr<CtapDriver>&& driver, fido::AuthenticatorGetInfoResponse&& info)
50 {
51 return adoptRef(*new CtapAuthenticator(WTFMove(driver), WTFMove(info)));
52 }
53
54private:
55 explicit CtapAuthenticator(std::unique_ptr<CtapDriver>&&, fido::AuthenticatorGetInfoResponse&&);
56
57 void makeCredential() final;
58 void continueMakeCredentialAfterResponseReceived(Vector<uint8_t>&&);
59 void getAssertion() final;
60 void continueGetAssertionAfterResponseReceived(Vector<uint8_t>&&);
61 void continueGetNextAssertionAfterResponseReceived(Vector<uint8_t>&&);
62
63 void getRetries();
64 void continueGetKeyAgreementAfterGetRetries(Vector<uint8_t>&&);
65 void continueRequestPinAfterGetKeyAgreement(Vector<uint8_t>&&, uint64_t retries);
66 void continueGetPinTokenAfterRequestPin(const String& pin, const WebCore::CryptoKeyEC&);
67 void continueRequestAfterGetPinToken(Vector<uint8_t>&&, const fido::pin::TokenRequest&);
68 bool tryRestartPin(const fido::CtapDeviceResponseCode&);
69
70 bool tryDowngrade();
71 bool processGoogleLegacyAppIdSupportExtension();
72
73 fido::AuthenticatorGetInfoResponse m_info;
74 bool m_isDowngraded { false };
75 size_t m_remainingAssertionResponses { 0 };
76 Vector<Ref<WebCore::AuthenticatorAssertionResponse>> m_assertionResponses;
77 Vector<uint8_t> m_pinAuth;
78};
