41 : m_connection(WTFMove(connection))
42{
43 m_connection->initialize();
44}

46CtapHidDriver::Worker::~Worker()
47{
48 m_connection->terminate();
49}

51void CtapHidDriver::Worker::transact(fido::FidoHidMessage&& requestMessage, MessageCallback&& callback)
52{
53 ASSERT(m_state == State::Idle);
54 m_state = State::Write;
55 m_requestMessage = WTFMove(requestMessage);
56 m_responseMessage.reset();
57 m_callback = WTFMove(callback);
58
59 // HidConnection could hold data from other applications, and thereofore invalidate it before each transaction.
60 m_connection->invalidateCache();
61 m_connection->send(m_requestMessage->popNextPacket(), [weakThis = makeWeakPtr(*this)](HidConnection::DataSent sent) mutable {
62 ASSERT(RunLoop::isMain());
63 if (!weakThis)
64 return;
65 weakThis->write(sent);
66 });
67}
68
69void CtapHidDriver::Worker::write(HidConnection::DataSent sent)
70{
71 if (m_state != State::Write)
72 return;
73 if (sent != HidConnection::DataSent::Yes) {
74 m_responseMessage = std::nullopt;
75 returnMessage();
76 return;
77 }
78
79 if (!m_requestMessage->numPackets()) {
80 m_state = State::Read;
81 m_connection->registerDataReceivedCallback([weakThis = makeWeakPtr(*this)](Vector<uint8_t>&& data) mutable {
82 ASSERT(RunLoop::isMain());
83 if (!weakThis)
84 return;
85 weakThis->read(data);
86 });
87 return;
88 }
89
90 m_connection->send(m_requestMessage->popNextPacket(), [weakThis = makeWeakPtr(*this)](HidConnection::DataSent sent) mutable {
91 ASSERT(RunLoop::isMain());
92 if (!weakThis)
93 return;
94 weakThis->write(sent);
95 });
96}

98void CtapHidDriver::Worker::read(const Vector<uint8_t>& data)
99{
100 if (m_state != State::Read)
101 return;
102 if (!m_responseMessage) {
103 m_responseMessage = FidoHidMessage::createFromSerializedData(data);
104 // The first few reports could be for other applications, and therefore ignore those.
105 if (!m_responseMessage || m_responseMessage->channelId() != m_requestMessage->channelId()) {
106 LOG_ERROR("Couldn't parse a hid init packet: %s", m_responseMessage ? "wrong channel id." : "bad data.");
107 m_responseMessage.reset();
108 return;
109 }
110 } else {
111 if (!m_responseMessage->addContinuationPacket(data)) {
112 LOG_ERROR("Couldn't parse a hid continuation packet.");
113 m_responseMessage = std::nullopt;
114 returnMessage();
115 return;
116 }
117 }
118
119 if (m_responseMessage->messageComplete()) {
120 // A KeepAlive cmd could be sent between a request and a response to indicate that
121 // the authenticator is waiting for user consent. Keep listening for the response.
122 if (m_responseMessage->cmd() == FidoHidDeviceCommand::kKeepAlive) {
123 m_responseMessage.reset();
124 return;
125 }
126 returnMessage();
127 return;
128 }
129}

131void CtapHidDriver::Worker::returnMessage()
132{
133 // Reset state before calling the response callback to avoid being deleted.
134 auto callback = WTFMove(m_callback);
135 auto message = WTFMove(m_responseMessage);
136 reset();
137 callback(WTFMove(message));
138}

140void CtapHidDriver::Worker::reset()
141{
142 m_connection->unregisterDataReceivedCallback();
143 m_callback = nullptr;
144 m_responseMessage = std::nullopt;
145 m_requestMessage = std::nullopt;
146 m_state = State::Idle;
147}

149// This implements CTAPHID_CANCEL which violates the transaction semantics:
150// https://fidoalliance.org/specs/fido-v2.0-ps-20190130/fido-client-to-authenticator-protocol-v2.0-ps-20190130.html#usb-hid-cancel
151void CtapHidDriver::Worker::cancel(fido::FidoHidMessage&& requestMessage)
152{
153 reset();
154 m_connection->invalidateCache();
155 ASSERT(requestMessage.numPackets() == 1);
156 m_connection->sendSync(requestMessage.popNextPacket());
157}

165void CtapHidDriver::transact(Vector<uint8_t>&& data, ResponseCallback&& callback)
166{
167 ASSERT(m_state == State::Idle);
168 m_state = State::AllocateChannel;
169 m_channelId = kHidBroadcastChannel;
170 m_requestData = WTFMove(data);
171 m_responseCallback = WTFMove(callback);
172
173 // Allocate a channel.
174 // Use a pseudo random nonce instead of a cryptographically strong one as the nonce
175 // is mainly for identifications.
176 size_t steps = kHidInitNonceLength / sizeof(uint32_t);
177 ASSERT(!(kHidInitNonceLength % sizeof(uint32_t)) && steps >= 1);
178 for (size_t i = 0; i < steps; ++i) {
179 uint32_t weakRandom = weakRandomUint32();
180 memcpy(m_nonce.data() + i * sizeof(uint32_t), &weakRandom, sizeof(uint32_t));
181 }
182
183 auto initCommand = FidoHidMessage::create(m_channelId, FidoHidDeviceCommand::kInit, m_nonce);
184 ASSERT(initCommand);
185 m_worker->transact(WTFMove(*initCommand), [weakThis = makeWeakPtr(*this)](std::optional<FidoHidMessage>&& response) mutable {
186 ASSERT(RunLoop::isMain());
187 if (!weakThis)
188 return;
189 weakThis->continueAfterChannelAllocated(WTFMove(response));
190 });
191}

193void CtapHidDriver::continueAfterChannelAllocated(std::optional<FidoHidMessage>&& message)
194{
195 if (m_state != State::AllocateChannel)
196 return;
197 if (!message) {
198 returnResponse({ });
199 return;
200 }
201 ASSERT(message->channelId() == m_channelId);
202
203 auto payload = message->getMessagePayload();
204 ASSERT(payload.size() == kHidInitResponseSize);
205 // Restart the transaction in the next run loop when nonce mismatches.
206 if (memcmp(payload.data(), m_nonce.data(), m_nonce.size())) {
207 m_state = State::Idle;
208 RunLoop::main().dispatch([weakThis = makeWeakPtr(*this), data = WTFMove(m_requestData), callback = WTFMove(m_responseCallback)]() mutable {
209 if (!weakThis)
210 return;
211 weakThis->transact(WTFMove(data), WTFMove(callback));
212 });
213 return;
214 }
215
216 m_state = State::Ready;
217 auto index = kHidInitNonceLength;
218 m_channelId = static_cast<uint32_t>(payload[index++]) << 24;
219 m_channelId |= static_cast<uint32_t>(payload[index++]) << 16;
220 m_channelId |= static_cast<uint32_t>(payload[index++]) << 8;
221 m_channelId |= static_cast<uint32_t>(payload[index]);
222 // FIXME(191534): Check the rest of the payload.
223 auto cmd = FidoHidMessage::create(m_channelId, protocol() == ProtocolVersion::kCtap ? FidoHidDeviceCommand::kCbor : FidoHidDeviceCommand::kMsg, m_requestData);
224 ASSERT(cmd);
225 m_worker->transact(WTFMove(*cmd), [weakThis = makeWeakPtr(*this)](std::optional<FidoHidMessage>&& response) mutable {
226 ASSERT(RunLoop::isMain());
227 if (!weakThis)
228 return;
229 weakThis->continueAfterResponseReceived(WTFMove(response));
230 });
231}
232
233void CtapHidDriver::continueAfterResponseReceived(std::optional<fido::FidoHidMessage>&& message)
234{
235 if (m_state != State::Ready)
236 return;
237 ASSERT(!message || message->channelId() == m_channelId);
238 returnResponse(message ? message->getMessagePayload() : Vector<uint8_t>());
239}
240
241void CtapHidDriver::returnResponse(Vector<uint8_t>&& response)
242{
243 // Reset state before calling the response callback to avoid being deleted.
244 auto responseCallback = WTFMove(m_responseCallback);
245 reset();
246 responseCallback(WTFMove(response));
247}
248
249void CtapHidDriver::reset()
250{
251 m_responseCallback = nullptr;
252 m_channelId = fido::kHidBroadcastChannel;
253 m_state = State::Idle;
254}
255
256void CtapHidDriver::cancel()
257{
258 if (m_state == State::Idle || protocol() != ProtocolVersion::kCtap)
259 return;
260 // Cancel any outstanding requests.
261 if (m_state == State::Ready) {
262 auto cancelCommand = FidoHidMessage::create(m_channelId, FidoHidDeviceCommand::kCancel, { });
263 m_worker->cancel(WTFMove(*cancelCommand));
264 }
265 reset();
266}
