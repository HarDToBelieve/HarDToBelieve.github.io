39CtapNfcDriver::CtapNfcDriver(Ref<NfcConnection>&& connection)
40 : m_connection(WTFMove(connection))
41{
42}
43
44// FIXME(200934): Support NFCCTAP_GETRESPONSE
45void CtapNfcDriver::transact(Vector<uint8_t>&& data, ResponseCallback&& callback)
46{
47 // For CTAP2, commands follow:
48 // https://fidoalliance.org/specs/fido-v2.0-ps-20190130/fido-client-to-authenticator-protocol-v2.0-ps-20190130.html#nfc-command-framing
49 if (protocol() == ProtocolVersion::kCtap) {
50 ApduCommand command;
51 command.setCla(kCtapNfcApduCla);
52 command.setIns(kCtapNfcApduIns);
53 command.setData(WTFMove(data));
54 command.setResponseLength(ApduCommand::kApduMaxResponseLength);
55
56 auto apduResponse = ApduResponse::createFromMessage(m_connection->transact(command.getEncodedCommand()));
57 if (!apduResponse) {
58 respondAsync(WTFMove(callback), { });
59 return;
60 }
61 if (apduResponse->status() == ApduResponse::Status::SW_INS_NOT_SUPPORTED) {
62 // Return kCtap1ErrInvalidCommand instead of an empty response to signal FidoService to create a U2F authenticator
63 // for the getInfo stage.
64 respondAsync(WTFMove(callback), { static_cast<uint8_t>(CtapDeviceResponseCode::kCtap1ErrInvalidCommand) });
65 return;
66 }
67 if (apduResponse->status() != ApduResponse::Status::SW_NO_ERROR) {
68 respondAsync(WTFMove(callback), { });
69 return;
70 }
71
72 respondAsync(WTFMove(callback), WTFMove(apduResponse->data()));
73 return;
74 }
75
76 // For U2F, U2fAuthenticator would handle the APDU encoding.
77 // https://fidoalliance.org/specs/fido-u2f-v1.2-ps-20170411/fido-u2f-nfc-protocol-v1.2-ps-20170411.html#framing
78 callback(m_connection->transact(WTFMove(data)));
79}
80
81// Return the response async to match the HID behaviour, such that nfc could fit into the current infra.
82void CtapNfcDriver::respondAsync(ResponseCallback&& callback, Vector<uint8_t>&& response) const
83{
84 RunLoop::main().dispatch([callback = WTFMove(callback), response = WTFMove(response)] () mutable {
85 callback(WTFMove(response));
86 });
87}
