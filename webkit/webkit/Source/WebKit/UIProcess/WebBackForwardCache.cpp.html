41WebBackForwardCache::WebBackForwardCache(WebProcessPool& processPool)
42 : m_processPool(processPool)
43{
44}
45
46WebBackForwardCache::~WebBackForwardCache()
47{
48 clear();
49}
50
51inline void WebBackForwardCache::removeOldestEntry()
52{
53 ASSERT(!m_itemsWithCachedPage.isEmpty());
54 removeEntry(*m_itemsWithCachedPage.first());
55}
56
57void WebBackForwardCache::setCapacity(unsigned capacity)
58{
59 if (m_capacity == capacity)
60 return;
61
62 m_capacity = capacity;
63 while (size() > capacity)
64 removeOldestEntry();
65
66 m_processPool.sendToAllProcesses(Messages::WebProcess::SetBackForwardCacheCapacity(m_capacity));
67}

69void WebBackForwardCache::addEntry(WebBackForwardListItem& item, std::unique_ptr<WebBackForwardCacheEntry>&& backForwardCacheEntry)
70{
71 ASSERT(capacity());
72 ASSERT(backForwardCacheEntry);
73
74 if (item.backForwardCacheEntry()) {
75 ASSERT(m_itemsWithCachedPage.contains(&item));
76 m_itemsWithCachedPage.removeFirst(&item);
77 }
78
79 item.setBackForwardCacheEntry(WTFMove(backForwardCacheEntry));
80 m_itemsWithCachedPage.append(&item);
81
82 if (size() > capacity())
83 removeOldestEntry();
84 ASSERT(size() <= capacity());
85
86 RELEASE_LOG(BackForwardCache, "WebBackForwardCache::addEntry: item=%s, hasSuspendedPage=%d, size=%u/%u", item.itemID().string().utf8().data(), !!item.suspendedPage(), size(), capacity());
87}

89void WebBackForwardCache::addEntry(WebBackForwardListItem& item, std::unique_ptr<SuspendedPageProxy>&& suspendedPage)
90{
91 addEntry(item, makeUnique<WebBackForwardCacheEntry>(*this, item.itemID(), suspendedPage->process().coreProcessIdentifier(), WTFMove(suspendedPage)));
92}

94void WebBackForwardCache::addEntry(WebBackForwardListItem& item, WebCore::ProcessIdentifier processIdentifier)
95{
96 addEntry(item, makeUnique<WebBackForwardCacheEntry>(*this, item.itemID(), WTFMove(processIdentifier)));
97}

99void WebBackForwardCache::removeEntry(WebBackForwardListItem& item)
100{
101 ASSERT(m_itemsWithCachedPage.contains(&item));
102 m_itemsWithCachedPage.removeFirst(&item);
103 item.setBackForwardCacheEntry(nullptr);
104 RELEASE_LOG(BackForwardCache, "WebBackForwardCache::removeEntry: item=%s, size=%u/%u", item.itemID().string().utf8().data(), size(), capacity());
105}
106
107void WebBackForwardCache::removeEntry(SuspendedPageProxy& suspendedPage)
108{
109 removeEntriesMatching([&suspendedPage](auto& item) {
110 return item.suspendedPage() == &suspendedPage;
111 });
112}
113
114std::unique_ptr<SuspendedPageProxy> WebBackForwardCache::takeSuspendedPage(WebBackForwardListItem& item)
115{
116 RELEASE_LOG(BackForwardCache, "WebBackForwardCache::takeSuspendedPage: item=%s", item.itemID().string().utf8().data());
117
118 ASSERT(m_itemsWithCachedPage.contains(&item));
119 ASSERT(item.backForwardCacheEntry());
120 auto suspendedPage = item.backForwardCacheEntry()->takeSuspendedPage();
121 ASSERT(suspendedPage);
122 removeEntry(item);
123 return suspendedPage;
124}

126void WebBackForwardCache::removeEntriesForProcess(WebProcessProxy& process)
127{
128 removeEntriesMatching([processIdentifier = process.coreProcessIdentifier()](auto& entry) {
129 ASSERT(entry.backForwardCacheEntry());
130 return entry.backForwardCacheEntry()->processIdentifier() == processIdentifier;
131 });
132}

134void WebBackForwardCache::removeEntriesForSession(PAL::SessionID sessionID)
135{
136 removeEntriesMatching([sessionID](auto& item) {

141void WebBackForwardCache::removeEntriesForPage(WebPageProxy& page)

143 removeEntriesMatching([pageID = page.identifier()](auto& item) {
144 return item.pageID() == pageID;
145 });
146}

148void WebBackForwardCache::removeEntriesForPageAndProcess(WebPageProxy& page, WebProcessProxy& process)
149{
150 removeEntriesMatching([pageID = page.identifier(), processIdentifier = process.coreProcessIdentifier()](auto& item) {
151 ASSERT(item.backForwardCacheEntry());
152 return item.pageID() == pageID && item.backForwardCacheEntry()->processIdentifier() == processIdentifier;
153 });
154}
155
156void WebBackForwardCache::removeEntriesMatching(const Function<bool(WebBackForwardListItem&)>& matches)
157{
158 Vector<Ref<WebBackForwardListItem>> itemsWithEntriesToClear;
159 m_itemsWithCachedPage.removeAllMatching([&](auto* item) {
160 if (matches(*item)) {
161 itemsWithEntriesToClear.append(*item);
162 return true;
163 }
164 return false;
165 });
166
167 for (auto& item : itemsWithEntriesToClear)
168 item->setBackForwardCacheEntry(nullptr);
169}
170
171void WebBackForwardCache::clear()
172{
173 RELEASE_LOG(BackForwardCache, "WebBackForwardCache::clear");
174 auto itemsWithCachedPage = WTFMove(m_itemsWithCachedPage);
175 for (auto* item : itemsWithCachedPage)
176 item->setBackForwardCacheEntry(nullptr);
177}

179void WebBackForwardCache::pruneToSize(unsigned newSize)
180{
181 RELEASE_LOG(BackForwardCache, "WebBackForwardCache::pruneToSize(%u)", newSize);
182 while (size() > newSize)
183 removeOldestEntry();
184}
