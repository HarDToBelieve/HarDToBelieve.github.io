27
28#if ENABLE(CONTEXT_MENUS)
29
30#include "APIObject.h"
31#include <wtf/Forward.h>
32#include <wtf/WeakPtr.h>
33
34namespace WebKit {
35
36class WebContextMenuItem;
37
38class WebContextMenuListenerProxy : public API::ObjectImpl<API::Object::Type::ContextMenuListener> {
39public:
40 struct Client : CanMakeWeakPtr<Client> {
41 virtual ~Client() = default;
42 virtual void useContextMenuItems(Vector<Ref<WebContextMenuItem>>&&) = 0;
43 };
44
45 static Ref<WebContextMenuListenerProxy> create(Client& client)
46 {
47 return adoptRef(*new WebContextMenuListenerProxy(client));
48 }
49
50 virtual ~WebContextMenuListenerProxy();
51
52 void useContextMenuItems(Vector<Ref<WebContextMenuItem>>&&);
53
54private:
55 explicit WebContextMenuListenerProxy(Client&);
56
57 WeakPtr<Client> m_client;
58};
59
60} // namespace WebKit
61
62#endif // ENABLE(CONTEXT_MENUS)
