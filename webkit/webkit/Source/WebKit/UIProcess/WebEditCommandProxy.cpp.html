36namespace WebKit {
37using namespace WebCore;
38
39WebEditCommandProxy::WebEditCommandProxy(WebUndoStepID commandID, const String& label, WebPageProxy& page)
40 : m_commandID(commandID)
41 , m_label(label)
42 , m_page(makeWeakPtr(page))
43{
44 m_page->addEditCommand(*this);
45}
46
47WebEditCommandProxy::~WebEditCommandProxy()
48{
49 if (m_page)
50 m_page->removeEditCommand(*this);
51}
52
53void WebEditCommandProxy::unapply()
54{
55 if (!m_page || !m_page->hasRunningProcess())
56 return;
57
58 m_page->send(Messages::WebPage::UnapplyEditCommand(m_commandID), IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
59 m_page->registerEditCommand(*this, UndoOrRedo::Redo);
60}
61
62void WebEditCommandProxy::reapply()
63{
64 if (!m_page || !m_page->hasRunningProcess())
65 return;
66
67 m_page->send(Messages::WebPage::ReapplyEditCommand(m_commandID), IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
68 m_page->registerEditCommand(*this, UndoOrRedo::Undo);
69}
