26#ifndef WebPopupMenuProxy_h
27#define WebPopupMenuProxy_h
28
29#include <WebCore/WritingMode.h>
30#include <wtf/Forward.h>
31#include <wtf/RefCounted.h>
32
33namespace WebCore {
34 class IntRect;
35}
36
37namespace WebKit {
38
39struct PlatformPopupMenuData;
40struct WebPopupItem;
41class NativeWebMouseEvent;
42
43class WebPopupMenuProxy : public RefCounted<WebPopupMenuProxy> {
44public:
45 class Client {
46 protected:
47 virtual ~Client()
48 {
49 }
50
51 public:
52 virtual void valueChangedForPopupMenu(WebPopupMenuProxy*, int32_t newSelectedIndex) = 0;
53 virtual void setTextFromItemForPopupMenu(WebPopupMenuProxy*, int32_t index) = 0;
54 virtual NativeWebMouseEvent* currentlyProcessedMouseDownEvent() = 0;
55#if PLATFORM(GTK)
56 virtual void failedToShowPopupMenu() = 0;
57#endif
58 };
59
60 virtual ~WebPopupMenuProxy()
61 {
62 }
63
64 virtual void showPopupMenu(const WebCore::IntRect& rect, WebCore::TextDirection, double pageScaleFactor, const Vector<WebPopupItem>& items, const PlatformPopupMenuData&, int32_t selectedIndex) = 0;
65 virtual void hidePopupMenu() = 0;
66 virtual void cancelTracking() { }
67
68 void invalidate() { m_client = 0; }
69
70protected:
71 explicit WebPopupMenuProxy(Client& client)
72 : m_client(&client)
73 {
74 }
75
76 Client* m_client;
77};
78
79} // namespace WebKit
80
81#endif // WebPopupMenuProxy_h
