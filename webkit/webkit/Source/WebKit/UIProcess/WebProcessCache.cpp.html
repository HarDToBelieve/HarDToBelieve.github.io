57bool WebProcessCache::canCacheProcess(WebProcessProxy& process) const
58{
59 if (!capacity()) {
60 WEBPROCESSCACHE_RELEASE_LOG("canCacheProcess: Not caching process because the cache has no capacity", process.processIdentifier());
61 return false;
62 }
63
64 if (process.registrableDomain().isEmpty()) {
65 WEBPROCESSCACHE_RELEASE_LOG("canCacheProcess: Not caching process because it does not have an associated registrable domain", process.processIdentifier());
66 return false;
67 }
68
69 if (MemoryPressureHandler::singleton().isUnderMemoryPressure()) {
70 WEBPROCESSCACHE_RELEASE_LOG("canCacheProcess: Not caching process because we are under memory pressure", process.processIdentifier());
71 return false;
72 }
73
74 auto sessionID = process.websiteDataStore().sessionID();
75 if (sessionID.isEphemeral() && !process.processPool().hasPagesUsingWebsiteDataStore(process.websiteDataStore())) {
76 WEBPROCESSCACHE_RELEASE_LOG("canCacheProcess: Not caching process because this session has been destroyed", process.processIdentifier());
77 return false;
78 }
79
80 return true;
81}

83bool WebProcessCache::addProcessIfPossible(Ref<WebProcessProxy>&& process)
84{
85 ASSERT(!process->pageCount());
86 ASSERT(!process->provisionalPageCount());
87 ASSERT(!process->suspendedPageCount());
88 ASSERT(!process->isRunningServiceWorkers());
89
90 if (!canCacheProcess(process))
91 return false;
92
93 // CachedProcess can destroy the process pool (which owns the WebProcessCache), by making its reference weak in WebProcessProxy::setIsInProcessCache.
94 auto protectedProcessPool = makeRef(process->processPool());
95 uint64_t requestIdentifier = generateAddRequestIdentifier();
96 m_pendingAddRequests.add(requestIdentifier, makeUnique<CachedProcess>(process.copyRef()));
97
98 WEBPROCESSCACHE_RELEASE_LOG("addProcessIfPossible: Checking if process is responsive before caching it", process->processIdentifier());
99 process->isResponsive([this, processPool = WTFMove(protectedProcessPool), process, requestIdentifier](bool isResponsive) {
100 auto cachedProcess = m_pendingAddRequests.take(requestIdentifier);
101 if (!cachedProcess)
102 return;
103
104 if (!isResponsive) {
105 WEBPROCESSCACHE_RELEASE_LOG_ERROR("addProcessIfPossible(): Not caching process because it is not responsive", cachedProcess->process().processIdentifier());
106 return;
107 }
108 processPool->webProcessCache().addProcess(WTFMove(cachedProcess));
109 });
110 return true;
111}

113bool WebProcessCache::addProcess(std::unique_ptr<CachedProcess>&& cachedProcess)
114{
115 ASSERT(!cachedProcess->process().pageCount());
116 ASSERT(!cachedProcess->process().provisionalPageCount());
117 ASSERT(!cachedProcess->process().suspendedPageCount());
118 ASSERT(!cachedProcess->process().isRunningServiceWorkers());
119
120 if (!canCacheProcess(cachedProcess->process()))
121 return false;
122
123 auto registrableDomain = cachedProcess->process().registrableDomain();
124 RELEASE_ASSERT(!registrableDomain.isEmpty());
125
126 if (auto previousProcess = m_processesPerRegistrableDomain.take(registrableDomain))
127 WEBPROCESSCACHE_RELEASE_LOG("addProcess: Evicting process from WebProcess cache because a new process was added for the same domain", previousProcess->process().processIdentifier());
128
129 while (m_processesPerRegistrableDomain.size() >= capacity()) {
130 auto it = m_processesPerRegistrableDomain.random();
131 WEBPROCESSCACHE_RELEASE_LOG("addProcess: Evicting process from WebProcess cache because capacity was reached", it->value->process().processIdentifier());
132 m_processesPerRegistrableDomain.remove(it);
133 }
134
135 WEBPROCESSCACHE_RELEASE_LOG("addProcess: Added process to WebProcess cache (size=%u, capacity=%u)", cachedProcess->process().processIdentifier(), size() + 1, capacity());
136 m_processesPerRegistrableDomain.add(registrableDomain, WTFMove(cachedProcess));
137
138 return true;
139}

141RefPtr<WebProcessProxy> WebProcessCache::takeProcess(const WebCore::RegistrableDomain& registrableDomain, WebsiteDataStore& dataStore)
142{
143 auto it = m_processesPerRegistrableDomain.find(registrableDomain);
144 if (it == m_processesPerRegistrableDomain.end())
145 return nullptr;
146
147 if (&it->value->process().websiteDataStore() != &dataStore)
148 return nullptr;
149
150 auto process = it->value->takeProcess();
151 m_processesPerRegistrableDomain.remove(it);
152 WEBPROCESSCACHE_RELEASE_LOG("takeProcess: Taking process from WebProcess cache (size=%u, capacity=%u)", process->processIdentifier(), size(), capacity());
153
154 ASSERT(!process->pageCount());
155 ASSERT(!process->provisionalPageCount());
156 ASSERT(!process->suspendedPageCount());
157
158 return process;
159}

161void WebProcessCache::updateCapacity(WebProcessPool& processPool)
162{
163 if (!processPool.configuration().processSwapsOnNavigation() || !processPool.configuration().usesWebProcessCache() || LegacyGlobalSettings::singleton().cacheModel() != CacheModel::PrimaryWebBrowser || processPool.configuration().usesSingleWebProcess()) {
164 if (!processPool.configuration().processSwapsOnNavigation())
165 WEBPROCESSCACHE_RELEASE_LOG("updateCapacity: Cache is disabled because process swap on navigation is disabled", 0);
166 else if (!processPool.configuration().usesWebProcessCache())
167 WEBPROCESSCACHE_RELEASE_LOG("updateCapacity: Cache is disabled by client", 0);
168 else if (processPool.configuration().usesSingleWebProcess())
169 WEBPROCESSCACHE_RELEASE_LOG("updateCapacity: Cache is disabled because process-per-tab was disabled", 0);
170 else
171 WEBPROCESSCACHE_RELEASE_LOG("updateCapacity: Cache is disabled because cache model is not PrimaryWebBrowser", 0);
172 m_capacity = 0;
173 } else {
174 size_t memorySize = ramSize() / GB;
175 if (memorySize < 3) {
176 m_capacity = 0;
177 WEBPROCESSCACHE_RELEASE_LOG("updateCapacity: Cache is disabled because device does not have enough RAM", 0);
178 } else {
179 // Allow 4 processes in the cache per GB of RAM, up to 30 processes.
180 m_capacity = std::min<unsigned>(memorySize * 4, 30);
181 WEBPROCESSCACHE_RELEASE_LOG("updateCapacity: Cache has a capacity of %u processes", 0, capacity());
182 }
183 }
184
185 if (!m_capacity)
186 clear();
187}

189void WebProcessCache::clear()
190{
191 if (m_pendingAddRequests.isEmpty() && m_processesPerRegistrableDomain.isEmpty())
192 return;
193
194 WEBPROCESSCACHE_RELEASE_LOG("clear: Evicting %u processes", 0, m_pendingAddRequests.size() + m_processesPerRegistrableDomain.size());
195 m_pendingAddRequests.clear();
196 m_processesPerRegistrableDomain.clear();
197}

199void WebProcessCache::clearAllProcessesForSession(PAL::SessionID sessionID)
200{
201 Vector<WebCore::RegistrableDomain> keysToRemove;
202 for (auto& pair : m_processesPerRegistrableDomain) {
203 if (pair.value->process().websiteDataStore().sessionID() == sessionID) {
204 WEBPROCESSCACHE_RELEASE_LOG("clearAllProcessesForSession: Evicting process because its session was destroyed", pair.value->process().processIdentifier());
205 keysToRemove.append(pair.key);
206 }
207 }
208 for (auto& key : keysToRemove)
209 m_processesPerRegistrableDomain.remove(key);
210
211 Vector<uint64_t> pendingRequestsToRemove;
212 for (auto& pair : m_pendingAddRequests) {
213 if (pair.value->process().websiteDataStore().sessionID() == sessionID) {
214 WEBPROCESSCACHE_RELEASE_LOG("clearAllProcessesForSession: Evicting process because its session was destroyed", pair.value->process().processIdentifier());
215 pendingRequestsToRemove.append(pair.key);
216 }
217 }
218 for (auto& key : pendingRequestsToRemove)
219 m_pendingAddRequests.remove(key);
220}

222void WebProcessCache::setApplicationIsActive(bool isActive)
223{
224 WEBPROCESSCACHE_RELEASE_LOG("setApplicationIsActive: (isActive=%d)", 0, isActive);
225 if (isActive)
226 m_evictionTimer.stop();
227 else if (!m_processesPerRegistrableDomain.isEmpty())
228 m_evictionTimer.startOneShot(clearingDelayAfterApplicationResignsActive);
229}

231void WebProcessCache::removeProcess(WebProcessProxy& process, ShouldShutDownProcess shouldShutDownProcess)
232{
233 RELEASE_ASSERT(!process.registrableDomain().isEmpty());
234 WEBPROCESSCACHE_RELEASE_LOG("removeProcess: Evicting process from WebProcess cache because it expired", process.processIdentifier());
235
236 std::unique_ptr<CachedProcess> cachedProcess;
237 auto it = m_processesPerRegistrableDomain.find(process.registrableDomain());
238 if (it != m_processesPerRegistrableDomain.end() && &it->value->process() == &process) {
239 cachedProcess = WTFMove(it->value);
240 m_processesPerRegistrableDomain.remove(it);
241 } else {
242 for (auto& pair : m_pendingAddRequests) {
243 if (&pair.value->process() == &process) {
244 cachedProcess = WTFMove(pair.value);
245 m_pendingAddRequests.remove(pair.key);
246 break;
247 }
248 }
249 }
250 ASSERT(cachedProcess);
251 if (!cachedProcess)
252 return;
253
254 ASSERT(&cachedProcess->process() == &process);
255 if (shouldShutDownProcess == ShouldShutDownProcess::No)
256 cachedProcess->takeProcess();
257}

259WebProcessCache::CachedProcess::CachedProcess(Ref<WebProcessProxy>&& process)
260 : m_process(WTFMove(process))
261 , m_evictionTimer(RunLoop::main(), this, &CachedProcess::evictionTimerFired)
262{
263 RELEASE_ASSERT(!m_process->pageCount());
264 RELEASE_ASSERT_WITH_MESSAGE(!m_process->websiteDataStore().processes().contains(*m_process), "Only processes with pages should be registered with the data store");
265 m_process->setIsInProcessCache(true);
266 m_evictionTimer.startOneShot(cachedProcessLifetime);
267}
268
269WebProcessCache::CachedProcess::~CachedProcess()
270{
271 if (!m_process)
272 return;
273
274 ASSERT(!m_process->pageCount());
275 ASSERT(!m_process->provisionalPageCount());
276 ASSERT(!m_process->suspendedPageCount());
277
278 m_process->setIsInProcessCache(false);
279 m_process->shutDown();
280}
281
282Ref<WebProcessProxy> WebProcessCache::CachedProcess::takeProcess()
283{
284 ASSERT(m_process);
285 m_evictionTimer.stop();
286 m_process->setIsInProcessCache(false);
287 return m_process.releaseNonNull();
288}
289
290void WebProcessCache::CachedProcess::evictionTimerFired()
291{
292 ASSERT(m_process);
293 m_process->processPool().webProcessCache().removeProcess(*m_process, ShouldShutDownProcess::Yes);
294}
