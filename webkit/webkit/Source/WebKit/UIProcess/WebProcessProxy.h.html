526 enum class IsWeak { No, Yes };
527 template<typename T> class WeakOrStrongPtr {
528 public:
529 WeakOrStrongPtr(T& object, IsWeak isWeak)
530 : m_isWeak(isWeak)
531 , m_weakObject(makeWeakPtr(object))
532 {
533 updateStrongReference();
534 }
535
536 void setIsWeak(IsWeak isWeak)
537 {
538 m_isWeak = isWeak;
539 updateStrongReference();
540 }
541
542 T* get() const { return m_weakObject.get(); }
543 T* operator->() const { return m_weakObject.get(); }
544 T& operator*() const { return *m_weakObject; }
545 explicit operator bool() const { return !!m_weakObject; }
546
547 private:
548 void updateStrongReference()
549 {
550 m_strongObject = m_isWeak == IsWeak::Yes ? nullptr : m_weakObject.get();
551 }
552
553 IsWeak m_isWeak;
554 WeakPtr<T> m_weakObject;
555 RefPtr<T> m_strongObject;
556 };
