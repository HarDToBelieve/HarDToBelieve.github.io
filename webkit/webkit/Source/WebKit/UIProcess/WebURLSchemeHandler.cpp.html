36static uint64_t generateWebURLSchemeHandlerIdentifier()
37{
38 static uint64_t nextIdentifier = 1;
39 return nextIdentifier++;
40}

42WebURLSchemeHandler::WebURLSchemeHandler()
43 : m_identifier(generateWebURLSchemeHandlerIdentifier())
44{
45}

47WebURLSchemeHandler::~WebURLSchemeHandler()
48{
49 ASSERT(m_tasks.isEmpty());
50}

52void WebURLSchemeHandler::startTask(WebPageProxy& page, WebProcessProxy& process, PageIdentifier webPageID, URLSchemeTaskParameters&& parameters, SyncLoadCompletionHandler&& completionHandler)
53{
54 auto taskIdentifier = parameters.taskIdentifier;
55 auto result = m_tasks.add(taskIdentifier, WebURLSchemeTask::create(*this, page, process, webPageID, WTFMove(parameters), WTFMove(completionHandler)));
56 ASSERT(result.isNewEntry);
57
58 auto pageEntry = m_tasksByPageIdentifier.add(page.identifier(), HashSet<uint64_t>());
59 ASSERT(!pageEntry.iterator->value.contains(taskIdentifier));
60 pageEntry.iterator->value.add(taskIdentifier);
61
62 platformStartTask(page, result.iterator->value);
63}

65WebProcessProxy* WebURLSchemeHandler::processForTaskIdentifier(uint64_t taskIdentifier) const
66{
67 if (!decltype(m_tasks)::isValidKey(taskIdentifier))
68 return nullptr;
69 auto iterator = m_tasks.find(taskIdentifier);
70 if (iterator == m_tasks.end())
71 return nullptr;
72 return iterator->value->process();
73}
74
75void WebURLSchemeHandler::stopAllTasksForPage(WebPageProxy& page, WebProcessProxy* process)
76{
77 auto iterator = m_tasksByPageIdentifier.find(page.identifier());
78 if (iterator == m_tasksByPageIdentifier.end())
79 return;
80
81 auto& tasksByPage = iterator->value;
82 Vector<uint64_t> taskIdentifiersToStop;
83 taskIdentifiersToStop.reserveInitialCapacity(tasksByPage.size());
84 for (auto taskIdentifier : tasksByPage) {
85 if (!process || processForTaskIdentifier(taskIdentifier) == process)
86 taskIdentifiersToStop.uncheckedAppend(taskIdentifier);
87 }
88
89 for (auto& taskIdentifier : taskIdentifiersToStop)
90 stopTask(page, taskIdentifier);
91
92}
93
94void WebURLSchemeHandler::stopTask(WebPageProxy& page, uint64_t taskIdentifier)
95{
96 if (!decltype(m_tasks)::isValidKey(taskIdentifier))
97 return;
98 auto iterator = m_tasks.find(taskIdentifier);
99 if (iterator == m_tasks.end())
100 return;
101
102 iterator->value->stop();
103 platformStopTask(page, iterator->value);
104
105 removeTaskFromPageMap(page.identifier(), taskIdentifier);
106 m_tasks.remove(iterator);
107}
108
109void WebURLSchemeHandler::taskCompleted(WebURLSchemeTask& task)
110{
111 auto takenTask = m_tasks.take(task.identifier());
112 ASSERT_UNUSED(takenTask, takenTask == &task);
113 removeTaskFromPageMap(task.pageProxyID(), task.identifier());
114
115 platformTaskCompleted(task);
116}
117
118void WebURLSchemeHandler::removeTaskFromPageMap(WebPageProxyIdentifier pageID, uint64_t taskID)
119{
120 auto iterator = m_tasksByPageIdentifier.find(pageID);
121 ASSERT(iterator != m_tasksByPageIdentifier.end());
122 ASSERT(iterator->value.contains(taskID));
123 if (!decltype(iterator->value)::isValidValue(taskID))
124 return;
125 iterator->value.remove(taskID);
126 if (iterator->value.isEmpty())
127 m_tasksByPageIdentifier.remove(iterator);
128}
