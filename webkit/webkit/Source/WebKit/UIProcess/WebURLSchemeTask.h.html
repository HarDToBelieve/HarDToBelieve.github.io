57class WebURLSchemeTask : public API::ObjectImpl<API::Object::Type::URLSchemeTask>, public InstanceCounted<WebURLSchemeTask> {
58 WTF_MAKE_NONCOPYABLE(WebURLSchemeTask);

61
62 ~WebURLSchemeTask();
63
64 uint64_t identifier() const { ASSERT(RunLoop::isMain()); return m_identifier; }
65 WebPageProxyIdentifier pageProxyID() const { ASSERT(RunLoop::isMain()); return m_pageProxyID; }
66 WebCore::PageIdentifier webPageID() const { ASSERT(RunLoop::isMain()); return m_webPageID; }

76 DataAlreadySent,

78 RedirectAfterResponse,
79 TaskAlreadyStopped,

81 WaitingForRedirectCompletionHandler,
82 None,

84 ExceptionType willPerformRedirection(WebCore::ResourceResponse&&, WebCore::ResourceRequest&&, Function<void(WebCore::ResourceRequest&&)>&&);
85 ExceptionType didPerformRedirection(WebCore::ResourceResponse&&, WebCore::ResourceRequest&&);
86 ExceptionType didReceiveResponse(const WebCore::ResourceResponse&);
87 ExceptionType didReceiveData(Ref<WebCore::SharedBuffer>&&);
88 ExceptionType didComplete(const WebCore::ResourceError&);
89
90 void stop();
91 void pageDestroyed();
92
93 void suppressTaskStoppedExceptions() { m_shouldSuppressTaskStoppedExceptions = true; }
94
95 bool waitingForRedirectCompletionHandlerCallback() const { return m_waitingForRedirectCompletionHandlerCallback; }
96
97private:
98 WebURLSchemeTask(WebURLSchemeHandler&, WebPageProxy&, WebProcessProxy&, WebCore::PageIdentifier, URLSchemeTaskParameters&&, SyncLoadCompletionHandler&&);
99
100 bool isSync() const { return !!m_syncCompletionHandler; }
101
102 Ref<WebURLSchemeHandler> m_urlSchemeHandler;
103 RefPtr<WebProcessProxy> m_process;
104 uint64_t m_identifier;
105 WebPageProxyIdentifier m_pageProxyID;
106 WebCore::PageIdentifier m_webPageID;
107 WebCore::ResourceRequest m_request WTF_GUARDED_BY_LOCK(m_requestLock);

110 bool m_stopped { false };

120 bool m_waitingForRedirectCompletionHandlerCallback { false };
