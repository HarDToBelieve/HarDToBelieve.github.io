33WebsiteDataStoreConfiguration::WebsiteDataStoreConfiguration(IsPersistent isPersistent, WillCopyPathsFromExistingConfiguration willCopyPaths)
34 : m_isPersistent(isPersistent)
35{
36 if (isPersistent == IsPersistent::Yes && willCopyPaths == WillCopyPathsFromExistingConfiguration::No) {
37 setApplicationCacheDirectory(WebsiteDataStore::defaultApplicationCacheDirectory());
38 setCacheStorageDirectory(WebsiteDataStore::defaultCacheStorageDirectory());
39 setNetworkCacheDirectory(WebsiteDataStore::defaultNetworkCacheDirectory());
40 setAlternativeServicesDirectory(WebsiteDataStore::defaultAlternativeServicesDirectory());
41 setMediaCacheDirectory(WebsiteDataStore::defaultMediaCacheDirectory());
42 setIndexedDBDatabaseDirectory(WebsiteDataStore::defaultIndexedDBDatabaseDirectory());
43 setServiceWorkerRegistrationDirectory(WebsiteDataStore::defaultServiceWorkerRegistrationDirectory());
44 setWebSQLDatabaseDirectory(WebsiteDataStore::defaultWebSQLDatabaseDirectory());
45 setLocalStorageDirectory(WebsiteDataStore::defaultLocalStorageDirectory());
46 setMediaKeysStorageDirectory(WebsiteDataStore::defaultMediaKeysStorageDirectory());
47 setResourceLoadStatisticsDirectory(WebsiteDataStore::defaultResourceLoadStatisticsDirectory());
48 setDeviceIdHashSaltsStorageDirectory(WebsiteDataStore::defaultDeviceIdHashSaltsStorageDirectory());
49 setJavaScriptConfigurationDirectory(WebsiteDataStore::defaultJavaScriptConfigurationDirectory());
50#if HAVE(ARKIT_INLINE_PREVIEW)
51 setModelElementCacheDirectory(WebsiteDataStore::defaultModelElementCacheDirectory());
52#endif
53 }
54}
