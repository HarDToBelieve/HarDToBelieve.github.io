150bool PageClientImpl::canTakeForegroundAssertions()
151{
152 if (EndowmentStateTracker::singleton().isVisible()) {
153 // If the application is visible according to the UIKit visibility endowment then we can take
154 // foreground assertions. Note that for view services, the visibility endownment from the host
155 // application gets propagated to the view service.
156 return true;
157 }
158
159 // If there is no run time limitation, then it means that the process is allowed to run for an extended
160 // period of time in the background (e.g. a daemon) and we let such processes take foreground assertions.
161 return [RBSProcessHandle currentProcess].activeLimitations.runTime == RBSProcessTimeLimitationNone;
162}

323void PageClientImpl::registerEditCommand(Ref<WebEditCommandProxy>&& command, UndoOrRedo undoOrRedo)
324{
325 auto actionName = command->label();
326 auto commandObjC = adoptNS([[WKEditCommand alloc] initWithWebEditCommandProxy:WTFMove(command)]);
327
328 NSUndoManager *undoManager = [m_contentView undoManager];
329 [undoManager registerUndoWithTarget:m_undoTarget.get() selector:((undoOrRedo == UndoOrRedo::Undo) ? @selector(undoEditing:) : @selector(redoEditing:)) object:commandObjC.get()];
330 if (!actionName.isEmpty())
331 [undoManager setActionName:(NSString *)actionName];
332}

335{

618void PageClientImpl::elementDidFocus(const FocusedElementInformation& nodeInformation, bool userIsInteracting, bool blurPreviousNode, OptionSet<WebCore::ActivityState::Flag> activityStateChanges, API::Object* userData)
619{
620 MESSAGE_CHECK(!userData || userData->type() == API::Object::Type::Data);
621
622 NSObject <NSSecureCoding> *userObject = nil;
623 if (API::Data* data = static_cast<API::Data*>(userData)) {
624 auto nsData = adoptNS([[NSData alloc] initWithBytesNoCopy:const_cast<unsigned char*>(data->bytes()) length:data->size() freeWhenDone:NO]);
625 auto unarchiver = adoptNS([[NSKeyedUnarchiver alloc] initForReadingFromData:nsData.get() error:nullptr]);
626 unarchiver.get().decodingFailurePolicy = NSDecodingFailurePolicyRaiseException;
627 @try {
628 auto* allowedClasses = m_webView.get()->_page->process().processPool().allowedClassesForParameterCoding();
629 userObject = [unarchiver decodeObjectOfClasses:allowedClasses forKey:@"userObject"];
630 } @catch (NSException *exception) {
631 LOG_ERROR("Failed to decode user data: %@", exception);
632 }
633 }
634
635 [m_contentView _elementDidFocus:nodeInformation userIsInteracting:userIsInteracting blurPreviousNode:blurPreviousNode activityStateChanges:activityStateChanges userObject:userObject];
636}
