70- (instancetype)initWithViewGestureController:(WebKit::ViewGestureController*)gestureController gestureRecognizerView:(UIView *)gestureRecognizerView
71{
72 self = [super init];
73 if (self) {
74 _gestureController = gestureController;
75 _gestureRecognizerView = gestureRecognizerView;
76
77 _backTransitionController = adoptNS([_UINavigationInteractiveTransitionBase alloc]);
78 _backTransitionController = [_backTransitionController initWithGestureRecognizerView:gestureRecognizerView animator:nil delegate:self];
79
80 _forwardTransitionController = adoptNS([_UINavigationInteractiveTransitionBase alloc]);
81 _forwardTransitionController = [_forwardTransitionController initWithGestureRecognizerView:gestureRecognizerView animator:nil delegate:self];
82 [_forwardTransitionController setShouldReverseTranslation:YES];
83 }
84 return self;
85}
86
87- (void)invalidate
88{
89 _gestureController = nullptr;
90}

92- (WebKit::ViewGestureController::SwipeDirection)directionForTransition:(_UINavigationInteractiveTransitionBase *)transition
93{
94 return transition == _backTransitionController ? WebKit::ViewGestureController::SwipeDirection::Back : WebKit::ViewGestureController::SwipeDirection::Forward;
95}
96
97- (_UINavigationInteractiveTransitionBase *)transitionForDirection:(WebKit::ViewGestureController::SwipeDirection)direction
98{
99 return direction == WebKit::ViewGestureController::SwipeDirection::Back ? _backTransitionController.get() : _forwardTransitionController.get();
100}
101
102- (void)startInteractiveTransition:(_UINavigationInteractiveTransitionBase *)transition
103{
104 _gestureController->beginSwipeGesture(transition, [self directionForTransition:transition]);
105}
106
107- (BOOL)shouldBeginInteractiveTransition:(_UINavigationInteractiveTransitionBase *)transition
108{
109 return _gestureController->canSwipeInDirection([self directionForTransition:transition]);
110}

112- (BOOL)interactiveTransition:(_UINavigationInteractiveTransitionBase *)transition gestureRecognizer:(UIGestureRecognizer *)gestureRecognizer shouldRecognizeSimultaneouslyWithGestureRecognizer:(UIGestureRecognizer *)otherGestureRecognizer
113{
114 return [otherGestureRecognizer isKindOfClass:[UITapGestureRecognizer class]];
115}
116
117- (BOOL)interactiveTransition:(_UINavigationInteractiveTransitionBase *)transition gestureRecognizer:(UIGestureRecognizer *)gestureRecognizer shouldReceiveTouch:(UITouch *)touch
118{
119 return YES;
120}

122- (UIPanGestureRecognizer *)gestureRecognizerForInteractiveTransition:(_UINavigationInteractiveTransitionBase *)transition WithTarget:(id)target action:(SEL)action
123{
124#if HAVE(UI_PARALLAX_TRANSITION_GESTURE_RECOGNIZER)
125ALLOW_NEW_API_WITHOUT_GUARDS_BEGIN
126 auto recognizer = adoptNS([[_UIParallaxTransitionPanGestureRecognizer alloc] initWithTarget:target action:action]);
127ALLOW_NEW_API_WITHOUT_GUARDS_END
128#else
129 auto recognizer = adoptNS([[UIScreenEdgePanGestureRecognizer alloc] initWithTarget:target action:action]);
130#endif
131 bool isLTR = [UIView userInterfaceLayoutDirectionForSemanticContentAttribute:[_gestureRecognizerView.get() semanticContentAttribute]] == UIUserInterfaceLayoutDirectionLeftToRight;
132
133 switch ([self directionForTransition:transition]) {
134 case WebKit::ViewGestureController::SwipeDirection::Back:
135 [recognizer setEdges:isLTR ? UIRectEdgeLeft : UIRectEdgeRight];
136 break;
137 case WebKit::ViewGestureController::SwipeDirection::Forward:
138 [recognizer setEdges:isLTR ? UIRectEdgeRight : UIRectEdgeLeft];
139 break;
140 }
141 return recognizer.autorelease();
142}
143
144- (BOOL)isNavigationSwipeGestureRecognizer:(UIGestureRecognizer *)recognizer
145{
146 return recognizer == [_backTransitionController gestureRecognizer] || recognizer == [_forwardTransitionController gestureRecognizer];
147}

166void ViewGestureController::installSwipeHandler(UIView *gestureRecognizerView, UIView *swipingView)
167{
168 ASSERT(!m_swipeInteractiveTransitionDelegate);
169 m_swipeInteractiveTransitionDelegate = adoptNS([[WKSwipeTransitionController alloc] initWithViewGestureController:this gestureRecognizerView:gestureRecognizerView]);
170 m_liveSwipeView = swipingView;
171}

173void ViewGestureController::beginSwipeGesture(_UINavigationInteractiveTransitionBase *transition, SwipeDirection direction)
174{
175 if (m_activeGestureType != ViewGestureType::None)
176 return;
177
178 willBeginGesture(ViewGestureType::Swipe);
179
180 m_webPageProxy.recordAutomaticNavigationSnapshot();
181
182 RefPtr<WebPageProxy> alternateBackForwardListSourcePage = m_alternateBackForwardListSourcePage.get();
183 m_webPageProxyForBackForwardListForCurrentSwipe = alternateBackForwardListSourcePage ? alternateBackForwardListSourcePage.get() : &m_webPageProxy;
184
185 auto& backForwardList = m_webPageProxyForBackForwardListForCurrentSwipe->backForwardList();
186 auto targetItem = makeRefPtr(direction == SwipeDirection::Back ? backForwardList.backItem() : backForwardList.forwardItem());
187 if (!targetItem) {
188 RELEASE_LOG_ERROR(ViewGestures, "Failed to find %s item when beginning swipe.", direction == SwipeDirection::Back ? "back" : "forward");
189 didEndGesture();
190 return;
191 }
192
193 m_webPageProxyForBackForwardListForCurrentSwipe->navigationGestureDidBegin();
194 if (&m_webPageProxy != m_webPageProxyForBackForwardListForCurrentSwipe)
195 m_webPageProxy.navigationGestureDidBegin();
196
197 // Copy the snapshot from this view to the one that owns the back forward list, so that
198 // swiping forward will have the correct snapshot.
199 if (m_webPageProxyForBackForwardListForCurrentSwipe != &m_webPageProxy) {
200 if (auto* currentViewHistoryItem = m_webPageProxy.backForwardList().currentItem())
201 backForwardList.currentItem()->setSnapshot(currentViewHistoryItem->snapshot());
202 }
203
204 CGRect liveSwipeViewFrame = [m_liveSwipeView frame];
205
206 RetainPtr<UIViewController> snapshotViewController = adoptNS([[UIViewController alloc] init]);
207 m_snapshotView = adoptNS([[UIView alloc] initWithFrame:liveSwipeViewFrame]);
208
209 // Disabling user interaction on the snapshot view lets the gestures go through the snapshot view, to the
210 // actual underlying view.
211 [m_snapshotView setUserInteractionEnabled:NO];
212
213 [m_snapshotView layer].name = @"SwipeSnapshot";
214
215 RetainPtr<UIColor> backgroundColor = [UIColor whiteColor];
216 if (ViewSnapshot* snapshot = targetItem->snapshot()) {
217 float deviceScaleFactor = m_webPageProxy.deviceScaleFactor();
218 WebCore::FloatSize swipeLayerSizeInDeviceCoordinates(liveSwipeViewFrame.size);
219 swipeLayerSizeInDeviceCoordinates.scale(deviceScaleFactor);
220
221 BOOL shouldRestoreScrollPosition = targetItem->pageState().mainFrameState.shouldRestoreScrollPosition;
222 WebCore::IntPoint currentScrollPosition = WebCore::roundedIntPoint(m_webPageProxy.viewScrollPosition());
223
224 if (snapshot->hasImage() && snapshot->size() == swipeLayerSizeInDeviceCoordinates && deviceScaleFactor == snapshot->deviceScaleFactor() && (shouldRestoreScrollPosition || (currentScrollPosition == snapshot->viewScrollPosition())))
225 [m_snapshotView layer].contents = snapshot->asLayerContents();
226 WebCore::Color coreColor = snapshot->backgroundColor();
227 if (coreColor.isValid())
228 backgroundColor = adoptNS([[UIColor alloc] initWithCGColor:WebCore::cachedCGColor(coreColor)]);
229 }
230
231 [m_snapshotView setBackgroundColor:backgroundColor.get()];
232 [m_snapshotView layer].contentsGravity = kCAGravityTopLeft;
233 [m_snapshotView layer].contentsScale = m_liveSwipeView.window.screen.scale;
234 [snapshotViewController setView:m_snapshotView.get()];
235
236 m_transitionContainerView = adoptNS([[UIView alloc] initWithFrame:liveSwipeViewFrame]);
237 [m_transitionContainerView layer].name = @"SwipeTransitionContainer";
238 m_liveSwipeViewClippingView = adoptNS([[UIView alloc] initWithFrame:liveSwipeViewFrame]);
239 [m_liveSwipeViewClippingView layer].name = @"LiveSwipeViewClipping";
240
241 [m_liveSwipeViewClippingView setClipsToBounds:YES];
242
243 [m_liveSwipeView.superview insertSubview:m_transitionContainerView.get() belowSubview:m_liveSwipeView];
244 [m_transitionContainerView addSubview:m_liveSwipeViewClippingView.get()];
245 [m_liveSwipeViewClippingView addSubview:m_liveSwipeView];
246
247 RetainPtr<UIViewController> targettedViewController = adoptNS([[UIViewController alloc] init]);
248 [targettedViewController setView:m_liveSwipeViewClippingView.get()];
249
250 UINavigationControllerOperation transitionOperation = direction == SwipeDirection::Back ? UINavigationControllerOperationPop : UINavigationControllerOperationPush;
251 RetainPtr<_UINavigationParallaxTransition> animationController = adoptNS([[_UINavigationParallaxTransition alloc] initWithCurrentOperation:transitionOperation]);
252
253 m_swipeTransitionContext = adoptNS([[_UIViewControllerOneToOneTransitionContext alloc] init]);
254 [m_swipeTransitionContext _setFromViewController:targettedViewController.get()];
255 [m_swipeTransitionContext _setToViewController:snapshotViewController.get()];
256 [m_swipeTransitionContext _setContainerView:m_transitionContainerView.get()];
257 [m_swipeTransitionContext _setFromStartFrame:liveSwipeViewFrame];
258 [m_swipeTransitionContext _setToEndFrame:liveSwipeViewFrame];
259 [m_swipeTransitionContext _setToStartFrame:CGRectZero];
260 [m_swipeTransitionContext _setFromEndFrame:CGRectZero];
261 [m_swipeTransitionContext _setAnimator:animationController.get()];
262 [m_swipeTransitionContext _setInteractor:transition];
263 [m_swipeTransitionContext _setTransitionIsInFlight:YES];
264 m_didCallWillEndSwipeGesture = false;
265 m_didCallEndSwipeGesture = false;
266 m_removeSnapshotImmediatelyWhenGestureEnds = false;
267 [m_swipeTransitionContext _setInteractiveUpdateHandler:^(BOOL finish, CGFloat percent, BOOL transitionCompleted, _UIViewControllerTransitionContext *) {
268 if (finish)
269 willEndSwipeGesture(*targetItem, !transitionCompleted);
270 }];
271 auto pageID = m_webPageProxy.identifier();
272 GestureID gestureID = m_currentGestureID;
273 [m_swipeTransitionContext _setCompletionHandler:[pageID, gestureID, targetItem] (_UIViewControllerTransitionContext *context, BOOL didComplete) {
274 if (auto gestureController = controllerForGesture(pageID, gestureID))
275 gestureController->endSwipeGesture(targetItem.get(), context, !didComplete);
276 }];
277
278 [transition setAnimationController:animationController.get()];
279 [transition startInteractiveTransition:m_swipeTransitionContext.get()];
280}

282void ViewGestureController::willEndSwipeGesture(WebBackForwardListItem& targetItem, bool cancelled)
283{
284 m_didCallWillEndSwipeGesture = true;
285 m_webPageProxyForBackForwardListForCurrentSwipe->navigationGestureWillEnd(!cancelled, targetItem);
286
287 if (cancelled)
288 return;
289
290 m_snapshotRemovalTargetRenderTreeSize = 0;
291 if (ViewSnapshot* snapshot = targetItem.snapshot())
292 m_snapshotRemovalTargetRenderTreeSize = snapshot->renderTreeSize() * swipeSnapshotRemovalRenderTreeSizeTargetFraction;
293
294 m_didStartProvisionalLoad = false;
295 m_pendingNavigation = m_webPageProxyForBackForwardListForCurrentSwipe->goToBackForwardItem(targetItem);
296
297 auto* currentItem = m_webPageProxyForBackForwardListForCurrentSwipe->backForwardList().currentItem();
298 // The main frame will not be navigated so hide the snapshot right away.
299 if (currentItem && currentItem->itemIsClone(targetItem)) {
300 removeSwipeSnapshot();
301 return;
302 }
303
304 // FIXME: Should we wait for VisuallyNonEmptyLayout like we do on Mac?
305 m_snapshotRemovalTracker.start(SnapshotRemovalTracker::RenderTreeSizeThreshold
306 | SnapshotRemovalTracker::RepaintAfterNavigation
307 | SnapshotRemovalTracker::MainFrameLoad
308 | SnapshotRemovalTracker::SubresourceLoads
309 | SnapshotRemovalTracker::ScrollPositionRestoration, [this] {
310 this->removeSwipeSnapshot();
311 });
312
313 if (ViewSnapshot* snapshot = targetItem.snapshot()) {
314 m_backgroundColorForCurrentSnapshot = snapshot->backgroundColor();
315 m_webPageProxy.didChangeBackgroundColor();
316 }
317}

319void ViewGestureController::endSwipeGesture(WebBackForwardListItem* targetItem, _UIViewControllerTransitionContext *context, bool cancelled)
320{
321 // At least in the context of a simulated swipe, it is possible that endSwipeGesture() gets called but not willEndSwipeGesture().
322 if (!m_didCallWillEndSwipeGesture)
323 willEndSwipeGesture(*targetItem, cancelled);
324
325 m_didCallEndSwipeGesture = true;
326
327 [context _setTransitionIsInFlight:NO];
328 [context _setInteractor:nil];
329 [context _setAnimator:nil];
330
331 [[m_transitionContainerView superview] insertSubview:m_snapshotView.get() aboveSubview:m_transitionContainerView.get()];
332 [[m_transitionContainerView superview] insertSubview:m_liveSwipeView aboveSubview:m_transitionContainerView.get()];
333 [m_liveSwipeViewClippingView removeFromSuperview];
334 m_liveSwipeViewClippingView = nullptr;
335 [m_transitionContainerView removeFromSuperview];
336 m_transitionContainerView = nullptr;
337
338 if (cancelled) {
339 // removeSwipeSnapshot will clear m_webPageProxyForBackForwardListForCurrentSwipe, so hold on to it here.
340 RefPtr<WebPageProxy> webPageProxyForBackForwardListForCurrentSwipe = m_webPageProxyForBackForwardListForCurrentSwipe;
341 removeSwipeSnapshot();
342 webPageProxyForBackForwardListForCurrentSwipe->navigationGestureDidEnd(false, *targetItem);
343 if (&m_webPageProxy != webPageProxyForBackForwardListForCurrentSwipe)
344 m_webPageProxy.navigationGestureDidEnd();
345 return;
346 }
347
348 m_webPageProxyForBackForwardListForCurrentSwipe->navigationGestureDidEnd(true, *targetItem);
349 if (&m_webPageProxy != m_webPageProxyForBackForwardListForCurrentSwipe)
350 m_webPageProxy.navigationGestureDidEnd();
351
352 if (!m_webPageProxy.provisionalDrawingArea()) {
353 removeSwipeSnapshot();
354 return;
355 }
356
357 // removeSwipeSnapshot() was called between willEndSwipeGesture() and endSwipeGesture().
358 // We couldn't remove it then, because the animation was still running, but now we can!
359 if (m_removeSnapshotImmediatelyWhenGestureEnds) {
360 removeSwipeSnapshot();
361 return;
362 }
363
364 auto pageID = m_webPageProxy.identifier();
365 GestureID gestureID = m_currentGestureID;
366
367 auto doAfterLoadStart = [this, pageID, gestureID] {
368 auto* drawingArea = m_webPageProxy.provisionalDrawingArea();
369 if (!drawingArea) {
370 removeSwipeSnapshot();
371 return;
372 }
373
374 drawingArea->dispatchAfterEnsuringDrawing([pageID, gestureID] (CallbackBase::Error error) {
375 if (auto gestureController = controllerForGesture(pageID, gestureID))
376 gestureController->willCommitPostSwipeTransitionLayerTree(error == CallbackBase::Error::None);
377 });
378 drawingArea->hideContentUntilPendingUpdate();
379 };
380
381 if (m_didStartProvisionalLoad)
382 doAfterLoadStart();
383 else
384 m_loadCallback = WTFMove(doAfterLoadStart);
385}

387void ViewGestureController::setRenderTreeSize(uint64_t renderTreeSize)
388{
389 if (m_activeGestureType != ViewGestureType::Swipe)
390 return;
391
392 if (!m_snapshotRemovalTargetRenderTreeSize || renderTreeSize > m_snapshotRemovalTargetRenderTreeSize)
393 didHitRenderTreeSizeThreshold();
394}
395
396void ViewGestureController::willCommitPostSwipeTransitionLayerTree(bool successful)
397{
398 if (m_activeGestureType != ViewGestureType::Swipe)
399 return;
400
401 if (!successful) {
402 removeSwipeSnapshot();
403 return;
404 }
405
406 didRepaintAfterNavigation();
407}

409void ViewGestureController::removeSwipeSnapshot()
410{
411 m_snapshotRemovalTracker.reset();
412
413 if (m_activeGestureType != ViewGestureType::Swipe)
414 return;
415
416 if (!m_didCallEndSwipeGesture) {
417 m_removeSnapshotImmediatelyWhenGestureEnds = true;
418 return;
419 }
420
421 resetState();
422}
423
424void ViewGestureController::resetState()
425{
426 [m_snapshotView removeFromSuperview];
427 m_snapshotView = nullptr;
428
429 m_snapshotRemovalTargetRenderTreeSize = 0;
430
431 if (m_webPageProxyForBackForwardListForCurrentSwipe) {
432 m_webPageProxyForBackForwardListForCurrentSwipe->navigationGestureSnapshotWasRemoved();
433 m_webPageProxyForBackForwardListForCurrentSwipe = nullptr;
434 }
435
436 m_swipeTransitionContext = nullptr;
437
438 m_backgroundColorForCurrentSnapshot = WebCore::Color();
439
440 m_pendingNavigation = nullptr;
441
442 didEndGesture();
443}
444
445void ViewGestureController::reset()
446{
447 removeSwipeSnapshot();
448 resetState();
449}

451bool ViewGestureController::beginSimulatedSwipeInDirectionForTesting(SwipeDirection direction)
452{
453 if (!canSwipeInDirection(direction))
454 return false;
455
456 _UINavigationInteractiveTransitionBase *transition = [m_swipeInteractiveTransitionDelegate transitionForDirection:direction];
457 beginSwipeGesture(transition, direction);
458 [transition _stopInteractiveTransition];
459
460 return true;
461}
462
463bool ViewGestureController::completeSimulatedSwipeInDirectionForTesting(SwipeDirection direction)
464{
465 _UINavigationInteractiveTransitionBase *transition = [m_swipeInteractiveTransitionDelegate transitionForDirection:direction];
466 [transition _completeStoppedInteractiveTransition];
467
468 return true;
469}
