167- (instancetype)_commonInitializationWithProcessPool:(WebKit::WebProcessPool&)processPool configuration:(Ref<API::PageConfiguration>&&)configuration
168{
169 ASSERT(_pageClient);
170
171 _page = processPool.createWebPage(*_pageClient, WTFMove(configuration));
172 _page->initializeWebPage();
173 _page->setIntrinsicDeviceScaleFactor(WebCore::screenScaleFactor([UIScreen mainScreen]));
174 _page->setUseFixedLayout(true);
175 _page->setDelegatesScrolling(true);
176 _page->setScreenIsBeingCaptured([[[self window] screen] isCaptured]);
177
178 // In order to ensure that we get a unique DisplayRefreshMonitor per-DrawingArea (necessary because DisplayRefreshMonitor
179 // is driven by this class), give each page a unique DisplayID derived from WebPage's unique ID.
180 // FIXME: While using the high end of the range of DisplayIDs makes a collision with real, non-RemoteLayerTreeDrawingArea
181 // DisplayIDs less likely, it is not entirely safe to have a RemoteLayerTreeDrawingArea and TiledCoreAnimationDrawingArea
182 // coeexist in the same process.
183 _page->windowScreenDidChange(std::numeric_limits<uint32_t>::max() - _page->webPageID().toUInt64(), std::nullopt);
184
185#if ENABLE(FULLSCREEN_API)
186 _page->setFullscreenClient(makeUnique<WebKit::FullscreenClient>(self.webView));
187#endif
188
189 WebKit::WebProcessPool::statistics().wkViewCount++;
190
191 _rootContentView = adoptNS([[UIView alloc] init]);
192 [_rootContentView layer].name = @"RootContent";
193 [_rootContentView layer].masksToBounds = NO;
194 [_rootContentView setAutoresizingMask:UIViewAutoresizingFlexibleWidth | UIViewAutoresizingFlexibleHeight];
195
196 _fixedClippingView = adoptNS([[UIView alloc] init]);
197 [_fixedClippingView layer].name = @"FixedClipping";
198 [_fixedClippingView layer].masksToBounds = YES;
199 [_fixedClippingView layer].anchorPoint = CGPointZero;
200
201 [self addSubview:_fixedClippingView.get()];
202 [_fixedClippingView addSubview:_rootContentView.get()];
203
204 if (!linkedOnOrAfter(WebCore::SDKVersion::FirstWithLazyGestureRecognizerInstallation))
205 [self setUpInteraction];
206 [self setUserInteractionEnabled:YES];
207
208 self.layer.hitTestsAsOpaque = YES;
209
210#if HAVE(UI_FOCUS_EFFECT)
211 if ([self respondsToSelector:@selector(setFocusEffect:)])
212 self.focusEffect = nil;
213#endif
214
215#if HAVE(VISIBILITY_PROPAGATION_VIEW)
216 [self _setupVisibilityPropagationViewForWebProcess];
217#if ENABLE(GPU_PROCESS)
218 [self _setupVisibilityPropagationViewForGPUProcess];
219#endif
220#endif
221
222 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(_applicationWillResignActive:) name:UIApplicationWillResignActiveNotification object:[UIApplication sharedApplication]];
223 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(_applicationDidBecomeActive:) name:UIApplicationDidBecomeActiveNotification object:[UIApplication sharedApplication]];
224 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(_applicationDidEnterBackground:) name:UIApplicationDidEnterBackgroundNotification object:[UIApplication sharedApplication]];
225 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(_applicationWillEnterForeground:) name:UIApplicationWillEnterForegroundNotification object:[UIApplication sharedApplication]];
226 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(_screenCapturedDidChange:) name:UIScreenCapturedDidChangeNotification object:[UIScreen mainScreen]];
227
228#if USE(UIKIT_KEYBOARD_ADDITIONS)
229 if (WebCore::IOSApplication::isEvernote() && !linkedOnOrAfter(WebCore::SDKVersion::FirstWhereWKContentViewDoesNotOverrideKeyCommands))
230 class_addMethod(self.class, @selector(keyCommands), reinterpret_cast<IMP>(&keyCommandsPlaceholderHackForEvernote), method_getTypeEncoding(class_getInstanceMethod(self.class, @selector(keyCommands))));
231#endif
232
233 return self;
234}

237- (void)_setupVisibilityPropagationViewForWebProcess
238{
239 auto processIdentifier = _page->process().processIdentifier();
240 auto contextID = _page->contextIDForVisibilityPropagationInWebProcess();
241 if (!processIdentifier || !contextID)
242 return;
243
244 ASSERT(!_visibilityPropagationViewForWebProcess);
245 // Propagate the view's visibility state to the WebContent process so that it is marked as "Foreground Running" when necessary.
246 _visibilityPropagationViewForWebProcess = adoptNS([[_UILayerHostView alloc] initWithFrame:CGRectZero pid:processIdentifier contextID:contextID]);
247 RELEASE_LOG(Process, "Created visibility propagation view %p (contextID=%u) for WebContent process with PID=%d", _visibilityPropagationViewForWebProcess.get(), contextID, processIdentifier);
248 [self addSubview:_visibilityPropagationViewForWebProcess.get()];
249}
250
251#if ENABLE(GPU_PROCESS)
252- (void)_setupVisibilityPropagationViewForGPUProcess
253{
254 auto* gpuProcess = _page->process().processPool().gpuProcess();
255 if (!gpuProcess)
256 return;
257 auto processIdentifier = gpuProcess->processIdentifier();
258 auto contextID = _page->contextIDForVisibilityPropagationInGPUProcess();
259 if (!processIdentifier || !contextID)
260 return;
261
262 if (_visibilityPropagationViewForGPUProcess)
263 return;
264
265 // Propagate the view's visibility state to the GPU process so that it is marked as "Foreground Running" when necessary.
266 _visibilityPropagationViewForGPUProcess = adoptNS([[_UILayerHostView alloc] initWithFrame:CGRectZero pid:processIdentifier contextID:contextID]);
267 RELEASE_LOG(Process, "Created visibility propagation view %p (contextID=%u) for GPU process with PID=%d", _visibilityPropagationViewForGPUProcess.get(), contextID, processIdentifier);
268 [self addSubview:_visibilityPropagationViewForGPUProcess.get()];
269}
270#endif // ENABLE(GPU_PROCESS)

438- (CGRect)_computeUnobscuredContentRectRespectingInputViewBounds:(CGRect)unobscuredContentRect inputViewBounds:(CGRect)inputViewBounds
439{
440 // The input view bounds are in window coordinates, but the unobscured rect is in content coordinates. Account for this by converting input view bounds to content coordinates.
441 CGRect inputViewBoundsInContentCoordinates = [self.window convertRect:inputViewBounds toView:self];
442 if (CGRectGetHeight(inputViewBoundsInContentCoordinates))
443 unobscuredContentRect.size.height = std::min<float>(CGRectGetHeight(unobscuredContentRect), CGRectGetMinY(inputViewBoundsInContentCoordinates) - CGRectGetMinY(unobscuredContentRect));
444 return unobscuredContentRect;
445}

447- (void)didUpdateVisibleRect:(CGRect)visibleContentRect
448 unobscuredRect:(CGRect)unobscuredContentRect
449 contentInsets:(UIEdgeInsets)contentInsets
450 unobscuredRectInScrollViewCoordinates:(CGRect)unobscuredRectInScrollViewCoordinates
451 obscuredInsets:(UIEdgeInsets)obscuredInsets
452 unobscuredSafeAreaInsets:(UIEdgeInsets)unobscuredSafeAreaInsets
453 inputViewBounds:(CGRect)inputViewBounds
454 scale:(CGFloat)zoomScale minimumScale:(CGFloat)minimumScale
455 viewStability:(OptionSet<WebKit::ViewStabilityFlag>)viewStability
456 enclosedInScrollableAncestorView:(BOOL)enclosedInScrollableAncestorView
457 sendEvenIfUnchanged:(BOOL)sendEvenIfUnchanged
458{
459 auto drawingArea = _page->drawingArea();
460 if (!drawingArea)
461 return;
462
463 MonotonicTime timestamp = MonotonicTime::now();
464 WebCore::VelocityData velocityData;
465 bool inStableState = viewStability.isEmpty();
466 if (!inStableState)
467 velocityData = _historicalKinematicData.velocityForNewData(visibleContentRect.origin, zoomScale, timestamp);
468 else {
469 _historicalKinematicData.clear();
470 velocityData = { 0, 0, 0, timestamp };
471 }
472
473 CGRect unobscuredContentRectRespectingInputViewBounds = [self _computeUnobscuredContentRectRespectingInputViewBounds:unobscuredContentRect inputViewBounds:inputViewBounds];
474 WebCore::FloatRect fixedPositionRectForLayout = _page->computeLayoutViewportRect(unobscuredContentRect, unobscuredContentRectRespectingInputViewBounds, _page->layoutViewportRect(), zoomScale, WebCore::FrameView::LayoutViewportConstraint::ConstrainedToDocumentRect);
475
476 WebKit::VisibleContentRectUpdateInfo visibleContentRectUpdateInfo(
477 visibleContentRect,
478 unobscuredContentRect,
479 floatBoxExtent(contentInsets),
480 unobscuredRectInScrollViewCoordinates,
481 unobscuredContentRectRespectingInputViewBounds,
482 fixedPositionRectForLayout,
483 floatBoxExtent(obscuredInsets),
484 floatBoxExtent(unobscuredSafeAreaInsets),
485 zoomScale,
486 viewStability,
487 _sizeChangedSinceLastVisibleContentRectUpdate,
488 self.webView._allowsViewportShrinkToFit,
489 enclosedInScrollableAncestorView,
490 velocityData,
491 downcast<WebKit::RemoteLayerTreeDrawingAreaProxy>(*drawingArea).lastCommittedLayerTreeTransactionID());
492
493 LOG_WITH_STREAM(VisibleRects, stream << "-[WKContentView didUpdateVisibleRect]" << visibleContentRectUpdateInfo.dump());
494
495 bool wasStableState = _page->inStableState();
496
497 _page->updateVisibleContentRects(visibleContentRectUpdateInfo, sendEvenIfUnchanged);
498
499 auto layoutViewport = _page->unconstrainedLayoutViewportRect();
500 _page->adjustLayersForLayoutViewport(layoutViewport);
501
502 _sizeChangedSinceLastVisibleContentRectUpdate = NO;
503
504 drawingArea->updateDebugIndicator();
505
506 [self updateFixedClippingView:layoutViewport];
507
508 if (wasStableState && !inStableState)
509 [self _didExitStableState];
510}

618- (void)_accessibilityRegisterUIProcessTokens
619{
620 auto uuid = [NSUUID UUID];
621 NSData *remoteElementToken = WebKit::newAccessibilityRemoteToken(uuid);
622
623 // Store information about the WebProcess that can later be retrieved by the iOS Accessibility runtime.
624 if (_page->process().state() == WebKit::WebProcessProxy::State::Running) {
625 [self _updateRemoteAccessibilityRegistration:YES];
626 IPC::Connection* connection = _page->process().connection();
627 storeAccessibilityRemoteConnectionInformation(self, _page->process().processIdentifier(), connection->identifier().port, uuid);
628
629 IPC::DataReference elementToken = IPC::DataReference(reinterpret_cast<const uint8_t*>([remoteElementToken bytes]), [remoteElementToken length]);
630 _page->registerUIProcessAccessibilityTokens(elementToken, elementToken);
631 }

700- (void)_didCommitLayerTree:(const WebKit::RemoteLayerTreeTransaction&)layerTreeTransaction
701{
702 CGSize contentsSize = layerTreeTransaction.contentsSize();
703 CGPoint scrollOrigin = -layerTreeTransaction.scrollOrigin();
704 CGRect contentBounds = { scrollOrigin, contentsSize };
705
706 LOG_WITH_STREAM(VisibleRects, stream << "-[WKContentView _didCommitLayerTree:] transactionID " << layerTreeTransaction.transactionID() << " contentBounds " << WebCore::FloatRect(contentBounds));
707
708 BOOL boundsChanged = !CGRectEqualToRect([self bounds], contentBounds);
709 if (boundsChanged)
710 [self setBounds:contentBounds];
711
712 [_webView _didCommitLayerTree:layerTreeTransaction];
713
714 if (_interactionViewsContainerView) {
715 WebCore::FloatPoint scaledOrigin = layerTreeTransaction.scrollOrigin();
716 float scale = self.webView.scrollView.zoomScale;
717 scaledOrigin.scale(scale);
718 [_interactionViewsContainerView setFrame:CGRectMake(scaledOrigin.x(), scaledOrigin.y(), 0, 0)];
719 }
720
721 if (boundsChanged) {
722 // FIXME: factor computeLayoutViewportRect() into something that gives us this rect.
723 WebCore::FloatRect fixedPositionRect = _page->computeLayoutViewportRect(_page->unobscuredContentRect(), _page->unobscuredContentRectRespectingInputViewBounds(), _page->layoutViewportRect(), self.webView.scrollView.zoomScale);
724 [self updateFixedClippingView:fixedPositionRect];
725
726 // We need to push the new content bounds to the webview to update fixed position rects.
727 [_webView _scheduleVisibleContentRectUpdate];
728 }
729
730 // Updating the selection requires a full editor state. If the editor state is missing post layout
731 // data then it means there is a layout pending and we're going to be called again after the layout
732 // so we delay the selection update.
733 if (!_page->editorState().isMissingPostLayoutData)
734 [self _updateChangedSelection];
735}

832- (NSUInteger)_wk_pageCountForPrintFormatter:(_WKWebViewPrintFormatter *)printFormatter
833{
834 if (_pdfPrintCallbackID)
835 [self _waitForDrawToPDFCallback];
836
837 WebCore::FrameIdentifier frameID;
838 if (_WKFrameHandle *handle = printFormatter.frameToPrint)
839 frameID = handle->_frameHandle->frameID();
840 else if (auto mainFrame = _page->mainFrame())
841 frameID = mainFrame->frameID();
842 else
843 return 0;
844
845 // The first page can have a smaller content rect than subsequent pages if a top content inset
846 // is specified. Since WebKit requires a uniform content rect for each page during layout, use
847 // the intersection of the first and non-first page rects.
848 // FIXME: Teach WebCore::PrintContext to accept an initial content offset when paginating.
849 CGRect printingRect = CGRectIntersection([printFormatter _pageContentRect:YES], [printFormatter _pageContentRect:NO]);
850 if (CGRectIsEmpty(printingRect))
851 return 0;
852
853 WebKit::PrintInfo printInfo;
854 printInfo.pageSetupScaleFactor = 1;
855 printInfo.snapshotFirstPage = printFormatter.snapshotFirstPage;
856
857 // FIXME: Paginate when exporting PDFs taller than 200"
858 if (printInfo.snapshotFirstPage) {
859 static const CGFloat maximumPDFHeight = 200 * 72; // maximum PDF height for a single page is 200 inches
860 CGSize contentSize = self.bounds.size;
861 printingRect = (CGRect) { CGPointZero, { contentSize.width, std::min(contentSize.height, maximumPDFHeight) } };
862 [printFormatter _setSnapshotPaperRect:printingRect];
863 }
864 printInfo.availablePaperWidth = CGRectGetWidth(printingRect);
865 printInfo.availablePaperHeight = CGRectGetHeight(printingRect);
866
867 auto retainedSelf = retainPtr(self);
868 auto pair = _page->computePagesForPrintingAndDrawToPDF(frameID, printInfo, [retainedSelf](const IPC::DataReference& pdfData) {
869 retainedSelf->_pdfPrintCallbackID = 0;
870 if (pdfData.isEmpty())
871 return;
872
873 auto data = adoptCF(CFDataCreate(kCFAllocatorDefault, pdfData.data(), pdfData.size()));
874 auto dataProvider = adoptCF(CGDataProviderCreateWithCFData(data.get()));
875 retainedSelf->_printedDocument = adoptCF(CGPDFDocumentCreateWithProvider(dataProvider.get()));
876 });
877 _pdfPrintCallbackID = pair.second;
878 return pair.first;
879}
880
881- (BOOL)_waitForDrawToPDFCallback
882{
883 ASSERT(_pdfPrintCallbackID);
884 if (!_page->process().connection()->waitForAsyncCallbackAndDispatchImmediately<Messages::WebPage::DrawToPDFiOS>(std::exchange(_pdfPrintCallbackID, 0), Seconds::infinity()))
885 return false;
886 ASSERT(!_pdfPrintCallbackID);
887 return true;
888}
889
890- (CGPDFDocumentRef)_wk_printedDocument
891{
892 if (_pdfPrintCallbackID) {
893 if (![self _waitForDrawToPDFCallback])
894 return nullptr;
895 }
896
897 return _printedDocument.get();
898}
