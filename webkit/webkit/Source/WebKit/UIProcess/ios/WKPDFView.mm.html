133- (void)web_setContentProviderData:(NSData *)data suggestedFilename:(NSString *)filename
134{
135 _data = adoptNS([data copy]);
136 _suggestedFilename = adoptNS([filename copy]);
137
138 [PDFHostViewController createHostView:[self, weakSelf = WeakObjCPtr<WKPDFView>(self)](PDFHostViewController *hostViewController) {
139 ASSERT(isMainRunLoop());
140
141 WKPDFView *autoreleasedSelf = weakSelf.getAutoreleased();
142 if (!autoreleasedSelf)
143 return;
144
145 WKWebView *webView = _webView.getAutoreleased();
146 if (!webView)
147 return;
148
149 if (!hostViewController)
150 return;
151 _hostViewController = hostViewController;
152
153 UIView *hostView = hostViewController.view;
154 hostView.frame = webView.bounds;
155
156 UIScrollView *scrollView = webView.scrollView;
157 [self removeFromSuperview];
158 [scrollView addSubview:hostView];
159
160 _actionSheetAssistant = adoptNS([[WKActionSheetAssistant alloc] initWithView:hostView]);
161 [_actionSheetAssistant setDelegate:self];
162
163 _pageNumberIndicator = hostViewController.pageNumberIndicator;
164 [_fixedOverlayView addSubview:_pageNumberIndicator.get()];
165
166 hostViewController.delegate = self;
167 [hostViewController setDocumentData:_data.get() withScrollView:scrollView];
168 } forExtensionIdentifier:nil];
169}

242static NSStringCompareOptions stringCompareOptions(_WKFindOptions findOptions)
243{
244 NSStringCompareOptions compareOptions = 0;
245 if (findOptions & _WKFindOptionsBackwards)
246 compareOptions |= NSBackwardsSearch;
247 if (findOptions & _WKFindOptionsCaseInsensitive)
248 compareOptions |= NSCaseInsensitiveSearch;
249 return compareOptions;
250}

284- (BOOL)_computeFocusedSearchResultIndexWithOptions:(_WKFindOptions)options didWrapAround:(BOOL *)didWrapAround
285{
286 BOOL isBackwards = options & _WKFindOptionsBackwards;
287 NSInteger singleOffset = isBackwards ? -1 : 1;
288
289 if (_findCompletion) {
290 ASSERT(!_focusedSearchResultIndex);
291 _focusedSearchResultPendingOffset += singleOffset;
292 return NO;
293 }
294
295 if (!_findStringCount)
296 return NO;
297
298 NSInteger newIndex;
299 if (_focusedSearchResultIndex) {
300 ASSERT(!_focusedSearchResultPendingOffset);
301 newIndex = *_focusedSearchResultIndex + singleOffset;
302 } else {
303 newIndex = isBackwards ? _findStringCount - 1 : 0;
304 newIndex += std::exchange(_focusedSearchResultPendingOffset, 0);
305 }
306
307 if (newIndex < 0 || static_cast<NSUInteger>(newIndex) >= _findStringCount) {
308 if (!(options & _WKFindOptionsWrapAround))
309 return NO;
310
311 NSUInteger wrappedIndex = std::abs(newIndex) % _findStringCount;
312 if (newIndex < 0)
313 wrappedIndex = _findStringCount - wrappedIndex;
314 newIndex = wrappedIndex;
315 *didWrapAround = YES;
316 }
317
318 _focusedSearchResultIndex = newIndex;
319 ASSERT(*_focusedSearchResultIndex < _findStringCount);
320 return YES;
321}

323- (void)_focusOnSearchResultWithOptions:(_WKFindOptions)options
324{
325 auto page = [_webView _page];
326 if (!page)
327 return;
328
329 BOOL didWrapAround = NO;
330 if (![self _computeFocusedSearchResultIndexWithOptions:options didWrapAround:&didWrapAround]) {
331 if (!_findCompletion)
332 page->findClient().didFailToFindString(page, _findString.get());
333 return;
334 }
335
336 auto focusedIndex = *_focusedSearchResultIndex;
337 [_hostViewController focusOnSearchResultAtIndex:focusedIndex];
338 page->findClient().didFindString(page, _findString.get(), { }, _findStringCount, focusedIndex, didWrapAround);
339}

341- (void)web_findString:(NSString *)string options:(_WKFindOptions)options maxCount:(NSUInteger)maxCount
342{
343 if ([_findString isEqualToString:string]) {
344 [self _focusOnSearchResultWithOptions:options];
345 return;
346 }
347
348 [self _findString:string withOptions:options maxCount:maxCount completion:^{
349 ASSERT([_findString isEqualToString:string]);
350 [self _focusOnSearchResultWithOptions:options];
351 }];
352}

433- (void)pdfHostViewController:(PDFHostViewController *)controller findStringUpdate:(NSUInteger)numFound done:(BOOL)done
434{
435 if (numFound > _findStringMaxCount && !done) {
436 [controller cancelFindStringWithHighlightsCleared:NO];
437 done = YES;
438 }
439
440 if (!done)
441 return;
442
443 if (auto findCompletion = std::exchange(_findCompletion, nil)) {
444 _findStringCount = numFound;
445 findCompletion();
446 }

492 return;
493 }
494#endif
495
496 [_actionSheetAssistant showLinkSheet];
497}

613- (CGPDFDocumentRef)_ensureDocumentForPrinting
614{
615 if (_documentForPrinting)
616 return _documentForPrinting.get();
617
618 auto dataProvider = adoptCF(CGDataProviderCreateWithCFData((CFDataRef)_data.get()));
619 auto pdfDocument = adoptCF(CGPDFDocumentCreateWithProvider(dataProvider.get()));
620 if (!CGPDFDocumentIsUnlocked(pdfDocument.get())) {
621 if (!CGPDFDocumentUnlockWithPassword(pdfDocument.get(), _passwordForPrinting.data()))
622 return nullptr;
623 }
624
625 if (!CGPDFDocumentAllowsPrinting(pdfDocument.get()))
626 return nullptr;
627
628 _documentForPrinting = WTFMove(pdfDocument);
629 return _documentForPrinting.get();
630}

632- (NSUInteger)_wk_pageCountForPrintFormatter:(_WKWebViewPrintFormatter *)printFormatter
633{
634 CGPDFDocumentRef documentForPrinting = [self _ensureDocumentForPrinting];
635 if (!documentForPrinting)
636 return 0;
637
638 size_t pageCount = CGPDFDocumentGetNumberOfPages(documentForPrinting);
639 if (printFormatter.snapshotFirstPage)
640 return std::min<NSUInteger>(pageCount, 1);
641 return pageCount;
642}
