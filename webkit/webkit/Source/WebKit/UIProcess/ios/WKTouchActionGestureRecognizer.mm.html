34@implementation WKTouchActionGestureRecognizer {
35 HashMap<unsigned, OptionSet<WebCore::TouchAction>> _touchActionsByTouchIdentifier;
36 id <WKTouchActionGestureRecognizerDelegate> _touchActionDelegate;
37}
38
39- (id)initWithTouchActionDelegate:(id <WKTouchActionGestureRecognizerDelegate>)touchActionDelegate
40{
41 if (self = [super init])
42 _touchActionDelegate = touchActionDelegate;
43 return self;
44}
45
46- (void)setTouchActions:(OptionSet<WebCore::TouchAction>)touchActions forTouchIdentifier:(unsigned)touchIdentifier
47{
48 ASSERT(!touchActions.contains(WebCore::TouchAction::Auto));
49 _touchActionsByTouchIdentifier.set(touchIdentifier, touchActions);
50}
51
52- (void)clearTouchActionsForTouchIdentifier:(unsigned)touchIdentifier
53{
54 _touchActionsByTouchIdentifier.remove(touchIdentifier);
55}

77- (void)_updateState
78{
79 // We always want to be in a recognized state so that we may always prevent another gesture recognizer.
80 [self setState:UIGestureRecognizerStateRecognized];
81}
82
83- (BOOL)canBePreventedByGestureRecognizer:(UIGestureRecognizer *)preventingGestureRecognizer
84{
85 // This allows this gesture recognizer to persist, even if other gesture recognizers are recognized.
86 return NO;
87}
88
89- (BOOL)canPreventGestureRecognizer:(UIGestureRecognizer *)preventedGestureRecognizer
90{
91 if (_touchActionsByTouchIdentifier.isEmpty())
92 return NO;
93
94 auto mayPan = [_touchActionDelegate gestureRecognizerMayPanWebView:preventedGestureRecognizer];
95 auto mayPinchToZoom = [_touchActionDelegate gestureRecognizerMayPinchToZoomWebView:preventedGestureRecognizer];
96 auto mayDoubleTapToZoom = [_touchActionDelegate gestureRecognizerMayDoubleTapToZoomWebView:preventedGestureRecognizer];
97
98 if (!mayPan && !mayPinchToZoom && !mayDoubleTapToZoom)
99 return NO;
100
101 // Now that we've established that this gesture recognizer may yield an interaction that is preventable by the "touch-action"
102 // CSS property we iterate over all active touches, check whether that touch matches the gesture recognizer, see if we have
103 // any touch-action specified for it, and then check for each type of interaction whether the touch-action property has a
104 // value that should prevent the interaction.
105 auto* activeTouches = [_touchActionDelegate touchActionActiveTouches];
106 for (NSNumber *touchIdentifier in activeTouches) {
107 auto iterator = _touchActionsByTouchIdentifier.find([touchIdentifier unsignedIntegerValue]);
108 if (iterator != _touchActionsByTouchIdentifier.end() && [[activeTouches objectForKey:touchIdentifier].gestureRecognizers containsObject:preventedGestureRecognizer]) {
109 // Panning is only allowed if "pan-x", "pan-y" or "manipulation" is specified. Additional work is needed to respect individual values, but this takes
110 // care of the case where no panning is allowed.
111 if (mayPan && !iterator->value.containsAny({ WebCore::TouchAction::PanX, WebCore::TouchAction::PanY, WebCore::TouchAction::Manipulation }))
112 return YES;
113 // Pinch-to-zoom is only allowed if "pinch-zoom" or "manipulation" is specified.
114 if (mayPinchToZoom && !iterator->value.containsAny({ WebCore::TouchAction::PinchZoom, WebCore::TouchAction::Manipulation }))
115 return YES;
116 // Double-tap-to-zoom is only disallowed if "none" is specified.
117 if (mayDoubleTapToZoom && iterator->value.contains(WebCore::TouchAction::None))
118 return YES;
119 }
120 }
121
122 return NO;
123}
