102void WebDataListSuggestionsDropdownIOS::show(WebCore::DataListSuggestionInformation&& information)
103{
104 if (m_suggestionsControl) {
105 [m_suggestionsControl updateWithInformation:WTFMove(information)];
106 return;
107 }
108
109 WebCore::DataListSuggestionActivationType type = information.activationType;
110
111#if ENABLE(IOS_FORM_CONTROL_REFRESH)
112 if (m_contentView._shouldUseContextMenusForFormControls) {
113 m_suggestionsControl = adoptNS([[WKDataListSuggestionsDropdown alloc] initWithInformation:WTFMove(information) inView:m_contentView]);
114 [m_suggestionsControl showSuggestionsDropdown:*this activationType:type];
115 return;
116 }
117#endif
118
119 if (currentUserInterfaceIdiomIsPadOrMac())
120 m_suggestionsControl = adoptNS([[WKDataListSuggestionsPopover alloc] initWithInformation:WTFMove(information) inView:m_contentView]);
121 else
122 m_suggestionsControl = adoptNS([[WKDataListSuggestionsPicker alloc] initWithInformation:WTFMove(information) inView:m_contentView]);
123
124 [m_suggestionsControl showSuggestionsDropdown:*this activationType:type];
125}

188- (NSArray<WKDataListTextSuggestion *> *)textSuggestions
189{
190 NSMutableArray *suggestions = [NSMutableArray array];
191
192 for (const auto& suggestion : _suggestions) {
193 [suggestions addObject:[WKDataListTextSuggestion textSuggestionWithInputText:suggestion.value]];
194 if (suggestions.count == 3)
195 break;
196 }
197
198 return suggestions;
199}

224- (instancetype)initWithInformation:(WebCore::DataListSuggestionInformation&&)information inView:(WKContentView *)view
225{
226 if (!(self = [super initWithInformation:WTFMove(information) inView:view]))
227 return nil;
228
229 _pickerView = adoptNS([[WKDataListSuggestionsPickerView alloc] initWithFrame:CGRectZero]);
230 [_pickerView setAutoresizingMask:UIViewAutoresizingFlexibleWidth | UIViewAutoresizingFlexibleHeight];
231 [_pickerView setDataSource:self];
232 [_pickerView setDelegate:self];
233 [_pickerView setControl:self];
234 [_pickerView setSize:[UIKeyboard defaultSizeForInterfaceOrientation:view.interfaceOrientation]];
235
236 return self;
237}

239- (void)updateWithInformation:(WebCore::DataListSuggestionInformation&&)information
240{
241 [super updateWithInformation:WTFMove(information)];
242 if (information.activationType != WebCore::DataListSuggestionActivationType::IndicatorClicked) {
243 self.view.dataListTextSuggestionsInputView = nil;
244 self.view.dataListTextSuggestions = self.textSuggestions;
245 return;
246 }
247
248 self.view.dataListTextSuggestionsInputView = _pickerView.get();
249
250 [_pickerView reloadAllComponents];
251 [_pickerView selectRow:0 inComponent:0 animated:NO];
252}
253
254- (void)showSuggestionsDropdown:(WebKit::WebDataListSuggestionsDropdownIOS&)dropdown activationType:(WebCore::DataListSuggestionActivationType)activationType
255{
256 [super showSuggestionsDropdown:dropdown activationType:activationType];
257 if (activationType == WebCore::DataListSuggestionActivationType::IndicatorClicked) {
258 self.view.dataListTextSuggestionsInputView = _pickerView.get();
259 [_pickerView selectRow:0 inComponent:0 animated:NO];
260 } else
261 self.view.dataListTextSuggestions = self.textSuggestions;
262}

365- (void)reloadData
366{
367 [self.tableView reloadData];
368
369 NSInteger suggestionsCount = [self.control suggestionsCount];
370 if (suggestionsCount > maxVisibleSuggestions)
371 [self setPreferredContentSize:CGSizeMake(suggestionsPopoverWidth, maxVisibleSuggestions * suggestionsPopoverCellHeight + suggestionsPopoverCellHeight / 2)];
372 else
373 [self setPreferredContentSize:CGSizeMake(suggestionsPopoverWidth, suggestionsCount * suggestionsPopoverCellHeight)];
374}
375
376- (NSInteger)tableView:(UITableView *)tableView numberOfRowsInSection:(NSInteger)section
377{
378 return [self.control suggestionsCount];
379}
380
381- (UITableViewCell *)tableView:(UITableView *)tableView cellForRowAtIndexPath:(NSIndexPath *)indexPath
382{
383 auto cell = retainPtr([tableView dequeueReusableCellWithIdentifier:suggestionCellReuseIdentifier]);
384 if (!cell)
385 cell = adoptNS([[UITableViewCell alloc] initWithStyle:UITableViewCellStyleDefault reuseIdentifier:suggestionCellReuseIdentifier]);
386
387 [cell textLabel].text = [self.control suggestionAtIndex:indexPath.row];
388 [cell textLabel].lineBreakMode = NSLineBreakByTruncatingTail;
389 [cell textLabel].textAlignment = [self.control textAlignment];
390
391 return cell.autorelease();
392}

447- (void)_displayWithActivationType:(WebCore::DataListSuggestionActivationType)activationType
448{
449 if (activationType == WebCore::DataListSuggestionActivationType::IndicatorClicked)
450 [self.view updateFocusedElementFocusedWithDataListDropdown:YES];
451 else if (activationType == WebCore::DataListSuggestionActivationType::ControlClicked)
452 [self.view updateFocusedElementFocusedWithDataListDropdown:NO];
453
454 [self _updateTextSuggestions];
455
456 if (![UIKeyboard isInHardwareKeyboardMode] && activationType != WebCore::DataListSuggestionActivationType::IndicatorClicked)
457 return;
458
459 [self _showSuggestions];
460}

462- (void)_showSuggestions
463{
464#if USE(UICONTEXTMENU)
465 [self _updateSuggestionsMenuElements];
466
467 if (!_suggestionsContextMenuInteraction) {
468 _suggestionsContextMenuInteraction = adoptNS([[UIContextMenuInteraction alloc] initWithDelegate:self]);
469 [self.view addInteraction:_suggestionsContextMenuInteraction.get()];
470
471 [self.view doAfterEditorStateUpdateAfterFocusingElement:[weakSelf = WeakObjCPtr<WKDataListSuggestionsDropdown>(self)] {
472 auto strongSelf = weakSelf.get();
473 if (!strongSelf)
474 return;
475
476 [strongSelf->_suggestionsContextMenuInteraction _presentMenuAtLocation:[[strongSelf view] lastInteractionLocation]];
477 }];
478 } else {
479 [_suggestionsContextMenuInteraction updateVisibleMenuWithBlock:[&](UIMenu *visibleMenu) -> UIMenu * {
480 return [visibleMenu menuByReplacingChildren:_suggestionsMenuElements.get()];
481 }];
482 }
483#endif
484}

491#if USE(UICONTEXTMENU)

493- (void)_updateSuggestionsMenuElements
494{
495 NSMutableArray *suggestions = [NSMutableArray arrayWithCapacity:self.suggestionsCount];
496
497 for (NSInteger index = 0; index < self.suggestionsCount; index++) {
498 UIAction *suggestionAction = [UIAction actionWithTitle:[self suggestionAtIndex:index] image:nil identifier:nil handler:[weakSelf = WeakObjCPtr<WKDataListSuggestionsDropdown>(self), index] (UIAction *) {
499 auto strongSelf = weakSelf.get();
500 if (!strongSelf)
501 return;
502
503 [strongSelf didSelectOptionAtIndex:index];
504 }];
505
506 [suggestions addObject:suggestionAction];
507 }
508
509 _suggestionsMenuElements = adoptNS([suggestions copy]);
510}

521}

540- (UIEdgeInsets)_preferredEdgeInsetsForSuggestionsMenu
541{
542 CGRect windowBounds = self.view.textEffectsWindow.bounds;
543 CGRect elementFrameInWindowCoordinates = [self.view convertRect:self.view.focusedElementInformation.interactionRect toView:nil];
544
545 if (CGRectGetMidY(elementFrameInWindowCoordinates) > CGRectGetMidY(windowBounds))
546 return UIEdgeInsetsMake(0, 0, CGRectGetMaxY(windowBounds) - CGRectGetMinY(elementFrameInWindowCoordinates), 0);
547
548 // Use MinY rather than MaxY to account for the hint preview.
549 return UIEdgeInsetsMake(CGRectGetMinY(elementFrameInWindowCoordinates), 0, 0, 0);
550}
