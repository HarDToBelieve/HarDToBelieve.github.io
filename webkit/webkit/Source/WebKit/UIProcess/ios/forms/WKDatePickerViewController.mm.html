373- (NSDate *)defaultMinimumDate
374{
375 auto components = adoptNS([[NSDateComponents alloc] init]);
376 [components setDay:1];
377 [components setMonth:1];
378 [components setYear:1];
379 NSDate *minDateInGregorianCalendar = [[NSCalendar calendarWithIdentifier:NSCalendarIdentifierGregorian] dateFromComponents:components.get()];
380 NSDate *minDateInCurrentCalendar = nil;
381 if (![_calendar rangeOfUnit:NSCalendarUnitYear startDate:&minDateInCurrentCalendar interval:nil forDate:minDateInGregorianCalendar]) {
382 // Fall back to the Gregorian calendar date if the year containing the min date couldn't be computed using the current calendar.
383 ASSERT_NOT_REACHED();
384 return minDateInGregorianCalendar;
385 }
386
387 if ([_calendar compareDate:minDateInCurrentCalendar toDate:minDateInGregorianCalendar toUnitGranularity:NSCalendarUnitDay] == NSOrderedAscending)
388 minDateInCurrentCalendar = [_calendar dateByAddingUnit:NSCalendarUnitYear value:1 toDate:minDateInCurrentCalendar options:0];
389
390 return minDateInCurrentCalendar;
391}
392
393- (NSDate *)defaultMaximumDate
394{
395 auto components = adoptNS([[NSDateComponents alloc] init]);
396 [components setDay:1];
397 [components setMonth:1];
398 [components setYear:10000];
399 NSDate *dayAfterMaxDateInGregorianCalendar = [[NSCalendar calendarWithIdentifier:NSCalendarIdentifierGregorian] dateFromComponents:components.get()];
400 NSDate *dayAfterMaxDateInCurrentCalendar = nil;
401 NSDate *dayAfterMaxDate = nil;
402 if ([_calendar rangeOfUnit:NSCalendarUnitYear startDate:&dayAfterMaxDateInCurrentCalendar interval:nil forDate:dayAfterMaxDateInGregorianCalendar])
403 dayAfterMaxDate = dayAfterMaxDateInCurrentCalendar;
404 else {
405 // Fall back to the Gregorian calendar date if the year containing the max date couldn't be computed using the current calendar.
406 ASSERT_NOT_REACHED();
407 dayAfterMaxDate = dayAfterMaxDateInGregorianCalendar;
408 }
409 return [_calendar dateByAddingUnit:NSCalendarUnitDay value:-1 toDate:dayAfterMaxDate options:0];
410}

476- (void)setDate:(NSDate *)date
477{
478 if ([self.date isEqualToDate:date])
479 return;
480
481 if ([date compare:_minimumDate.get()] == NSOrderedAscending)
482 date = _minimumDate.get();
483
484 if ([date compare:_maximumDate.get()] == NSOrderedDescending)
485 date = _maximumDate.get();
486
487 [self setDateFromComponents:[_calendar components:NSCalendarUnitDay | NSCalendarUnitMonth | NSCalendarUnitYear | NSCalendarUnitEra fromDate:date]];
488}

522- (void)_adjustDateToValidDateIfNecessary
523{
524 NSDateComponents *adjustedComponents = [self _dateComponentForDay:_day month:_month year:_year era:_era];
525 BOOL didAdjust = NO;
526 while (![adjustedComponents isValidDateInCalendar:_calendar.get()]) {
527 if (adjustedComponents.day <= 1) {
528 ASSERT_NOT_REACHED();
529 break;
530 }
531 --adjustedComponents.day;
532 didAdjust = YES;
533 }
534
535 [self setDateFromComponents:adjustedComponents];
536
537 if (didAdjust)
538 [self _updateSelectedPickerViewIndices];
539}

565- (NSInteger)numberOfItemsInPickerView:(WKDatePickerWheel *)pickerView
566{
567 if (pickerView == _monthPicker)
568 return [_calendar rangeOfUnit:NSCalendarUnitMonth inUnit:NSCalendarUnitYear forDate:self.date].length;
569
570 if (pickerView == _dayPicker)
571 return [_calendar rangeOfUnit:NSCalendarUnitDay inUnit:NSCalendarUnitMonth forDate:self.date].length;
572
573 if (pickerView == _yearAndEraPicker) {
574 NSDateComponents *yearAndEraComponents = [_calendar components:NSCalendarUnitYear | NSCalendarUnitEra fromDate:self.maximumDate];
575 return 1 + [self _indexFromYear:yearAndEraComponents.year era:yearAndEraComponents.era];
576 }
577
578 return 0;
579}

581- (UIView *)pickerView:(WKDatePickerWheel *)pickerView viewForItemAtIndex:(NSInteger)index
582{
583 auto label = retainPtr((WKDatePickerWheelLabel *)[pickerView dequeueReusableItemView]) ?: adoptNS([[WKDatePickerWheelLabel alloc] init]);
584 if (![label needsUpdateForIndex:index selectedDate:self.date])
585 return label.autorelease();
586
587 if (pickerView == _monthPicker) {
588 NSDateComponents *dateComponentsAtIndex = [self _dateComponentForDay:NSNotFound month:[self _monthFromIndex:index] year:_year era:_era];
589 [label setText:[_standaloneMonthFormatter stringFromDate:dateComponentsAtIndex.date].localizedUppercaseString];
590 }
591
592 if (pickerView == _dayPicker) {
593 NSDateComponents *dateComponentsAtIndex = [self _dateComponentForDay:[self _dayFromIndex:index] month:_month year:_year era:_era];
594 [label setText:[_standaloneDayFormatter stringFromDate:dateComponentsAtIndex.date]];
595 }
596
597 if (pickerView == _yearAndEraPicker) {
598 auto eraAndYear = [self _eraAndYearFromIndex:index];
599 NSDate *dateForEraAndYear = [self _dateComponentForDay:NSNotFound month:NSNotFound year:eraAndYear.year era:eraAndYear.era].date;
600 NSString *eraPrefix = _displayCombinedEraAndYear ? [[[_calendar eraSymbols] objectAtIndex:eraAndYear.era] substringToIndex:1] : @"";
601 [label setText:[eraPrefix stringByAppendingString:[_standaloneYearFormatter stringFromDate:dateForEraAndYear]]];
602 }
603
604 [label setIndex:index];
605 [label setLastSelectedDate:self.date];
606 return label.autorelease();
607}

609- (void)didBeginInteractingWithDatePicker:(WKDatePickerWheel *)datePicker
610{
611 if (datePicker == _focusedPicker)
612 return;
613
614 [_monthPicker setDrawsFocusOutline:datePicker == _monthPicker];
