191- (CGSize)preferredDatePickerSize
192{
193 CGSize fittingSize = UILayoutFittingCompressedSize;
194 UILayoutPriority horizontalPriority = UILayoutPriorityFittingSizeLevel;
195 UILayoutPriority verticalPriority = UILayoutPriorityFittingSizeLevel;
196 if ([_datePicker datePickerMode] != UIDatePickerModeTime)
197 fittingSize.width = kDateTimePickerDefaultWidth;
198 else {
199 fittingSize.width = kDateTimePickerTimeControlWidth;
200 fittingSize.height = kDateTimePickerTimeControlHeight;
201 horizontalPriority = UILayoutPriorityRequired;
202 verticalPriority = UILayoutPriorityRequired;
203 }
204
205 CGSize layoutSize = [_datePicker systemLayoutSizeFittingSize:fittingSize withHorizontalFittingPriority:horizontalPriority verticalFittingPriority:verticalPriority];
206 return layoutSize;
207}

209- (CGSize)preferredContentSize
210{
211 // Cache the content size to workaround rdar://74749942.
212 if (CGSizeEqualToSize(_contentSize, CGSizeZero)) {
213 CGSize datePickerSize = [self preferredDatePickerSize];
214 UIEdgeInsets datePickerInsets = [self datePickerInsets];
215
216 _contentSize = CGSizeMake(datePickerSize.width + datePickerInsets.left + datePickerInsets.right, datePickerSize.height + kDateTimePickerToolbarHeight + datePickerInsets.top + datePickerInsets.bottom);
217 }
218
219 return _contentSize;
220}

266- (id)initWithView:(WKContentView *)view datePickerMode:(UIDatePickerMode)mode
267{
268 if (!(self = [super init]))
269 return nil;
270
271 _view = view;
272 _interactionPoint = [_view lastInteractionLocation];
273 _shouldRemoveTimeZoneInformation = NO;
274
275 switch (view.focusedElementInformation.elementType) {
276 case WebKit::InputType::Date:
277 _formatString = kDateFormatString;
278 break;
279 case WebKit::InputType::Month:
280 _formatString = kMonthFormatString;
281 break;
282 case WebKit::InputType::Time:
283 _formatString = kTimeFormatString;
284 break;
285 case WebKit::InputType::DateTimeLocal:
286 _shouldRemoveTimeZoneInformation = YES;
287 break;
288 default:
289 break;
290 }
291
292 _datePicker = adoptNS([[UIDatePicker alloc] init]);
293 [_datePicker addTarget:self action:@selector(datePickerChanged:) forControlEvents:UIControlEventValueChanged];
294
295 if ([self shouldForceGregorianCalendar])
296 [_datePicker setCalendar:[NSCalendar calendarWithIdentifier:NSCalendarIdentifierGregorian]];
297
298 [_datePicker setDatePickerMode:mode];
299
300#if HAVE(UIDATEPICKER_STYLE)
301 if (mode == UIDatePickerModeTime || mode == (UIDatePickerMode)UIDatePickerModeYearAndMonth)
302 [_datePicker setPreferredDatePickerStyle:UIDatePickerStyleWheels];
303 else
304 [_datePicker setPreferredDatePickerStyle:UIDatePickerStyleInline];
305#endif
306
307#if HAVE(UIDATEPICKER_OVERLAY_PRESENTATION)
308 _isDismissingDatePicker = NO;
309
310 _accessoryView = adoptNS([[UIToolbar alloc] init]);
311 [[_accessoryView heightAnchor] constraintEqualToConstant:kDateTimePickerToolbarHeight].active = YES;
312
313#if HAVE(UITOOLBAR_STANDARD_APPEARANCE)
314 auto toolbarAppearance = adoptNS([[UIToolbarAppearance alloc] init]);
315 [toolbarAppearance setBackgroundEffect:nil];
316 [_accessoryView setStandardAppearance:toolbarAppearance.get()];
317#endif
318
319 auto resetButton = adoptNS([[UIBarButtonItem alloc] initWithTitle:WEB_UI_STRING_KEY("Reset", "Reset Button Date/Time Context Menu", "Reset button in date input context menu") style:UIBarButtonItemStylePlain target:self action:@selector(reset:)]);
320 auto doneButton = adoptNS([[UIBarButtonItem alloc] initWithBarButtonSystemItem:UIBarButtonSystemItemDone target:self action:@selector(done:)]);
321
322 [_accessoryView setItems:@[ resetButton.get(), UIBarButtonItem.flexibleSpaceItem, doneButton.get() ]];
323#elif USE(UICONTEXTMENU)
324 _dateTimePickerViewController = adoptNS([[WKDateTimePickerViewController alloc] initWithDatePicker:_datePicker.get()]);
325 [_dateTimePickerViewController setDelegate:self];
326#endif
327
328 return self;
329}

357- (UIEdgeInsets)_preferredEdgeInsetsForDateTimePicker
358{
359 CGSize pickerSize = [_dateTimePickerViewController preferredContentSize];
360 CGRect windowBounds = _view.textEffectsWindow.bounds;
361 CGRect elementFrameInWindowCoordinates = [_view convertRect:_view.focusedElementInformation.interactionRect toView:nil];
362
363 // Attempt to present the date picker in a way that does not obscure the element.
364
365 CGFloat topInsetForBottomAlignment = CGRectGetMaxY(elementFrameInWindowCoordinates) + kDateTimePickerControlMargin;
366 CGFloat rightInsetForRightAlignment = CGRectGetWidth(windowBounds) - CGRectGetMaxX(elementFrameInWindowCoordinates);
367
368 BOOL canPresentBelowElement = (topInsetForBottomAlignment + pickerSize.height) < CGRectGetHeight(windowBounds);
369 BOOL canAlignToElementRight = (rightInsetForRightAlignment + pickerSize.width) < CGRectGetWidth(windowBounds);
370
371 // Try to present the picker from the bottom right of the element.
372 if (canPresentBelowElement && canAlignToElementRight)
373 return UIEdgeInsetsMake(topInsetForBottomAlignment, 0, 0, rightInsetForRightAlignment);
374
375 CGFloat leftInsetForLeftAlignment = CGRectGetMinX(elementFrameInWindowCoordinates);
376
377 BOOL canAlignToElementLeft = (leftInsetForLeftAlignment + pickerSize.width) <= CGRectGetWidth(windowBounds);
378
379 // Try to present the picker from the bottom left of the element.
380 if (canPresentBelowElement && canAlignToElementLeft)
381 return UIEdgeInsetsMake(topInsetForBottomAlignment, leftInsetForLeftAlignment, 0, 0);
382
383 // Try to present the picker underneath the element.
384 if (canPresentBelowElement)
385 return UIEdgeInsetsMake(topInsetForBottomAlignment, 0, 0, 0);
386
387 CGFloat bottomInsetForTopAlignment = CGRectGetHeight(windowBounds) - CGRectGetMinY(elementFrameInWindowCoordinates) + kDateTimePickerControlMargin;
388
389 BOOL canPresentAboveElement = (bottomInsetForTopAlignment + pickerSize.height) < CGRectGetHeight(windowBounds);
390
391 // Try to present the picker from the top right of the element.
392 if (canPresentAboveElement && canAlignToElementRight)
393 return UIEdgeInsetsMake(0, 0, bottomInsetForTopAlignment, rightInsetForRightAlignment);
394
395 // Try to present the picker from the top left of the element.
396 if (canPresentAboveElement && canAlignToElementLeft)
397 return UIEdgeInsetsMake(0, leftInsetForLeftAlignment, bottomInsetForTopAlignment, 0);
398
399 // Try to present the picker above the element.
400 if (canPresentAboveElement)
401 return UIEdgeInsetsMake(0, 0, bottomInsetForTopAlignment, 0);
402
403 CGFloat rightInsetForPresentingBesideElementLeft = CGRectGetWidth(windowBounds) - CGRectGetMinX(elementFrameInWindowCoordinates) + kDateTimePickerControlMargin;
404 BOOL canPresentBesideElementLeft = (rightInsetForPresentingBesideElementLeft + pickerSize.width) < CGRectGetWidth(windowBounds);
405
406 // Try to present the picker to the left of the element.
407 if (canPresentBesideElementLeft)
408 return UIEdgeInsetsMake(0, 0, 0, rightInsetForPresentingBesideElementLeft);
409
410 CGFloat leftInsetForPresentingBesideElementRight = CGRectGetMaxX(elementFrameInWindowCoordinates) + kDateTimePickerControlMargin;
411 BOOL canPresentBesideElementRight = (leftInsetForPresentingBesideElementRight + pickerSize.width) < CGRectGetWidth(windowBounds);
412
413 // Try to present the picker to the right of the element.
414 if (canPresentBesideElementRight)
415 return UIEdgeInsetsMake(0, leftInsetForPresentingBesideElementRight, 0, 0);
416
417 // Present the picker from the center of the element.
418 return UIEdgeInsetsZero;
419}

558- (NSInteger)_timeZoneOffsetFromGMT:(NSDate *)date
559{
560 if (!_shouldRemoveTimeZoneInformation)
561 return 0;
562
563 return [[_datePicker timeZone] secondsFromGMTForDate:date];

566- (NSString *)_sanitizeInputValueForFormatter:(NSString *)value
567{
568 // The "time" input type may have seconds and milliseconds information which we
569 // just ignore. For example: "01:56:20.391" is shortened to just "01:56".
570 if (_view.focusedElementInformation.elementType == WebKit::InputType::Time)
571 return [value substringToIndex:[kTimeFormatString length]];
572
573 return value;
574}

599- (void)_dateChanged
600{
601 // Internally, DOMHTMLInputElement setValueAs* each take different values for
602 // different date types. It is sometimes easier to set the date in different ways:
603 // - use setValueAsString for "date", "month", and "time".
604 // - use setValueAsNumber for "datetime-local".
605 if (_formatString)
606 [self _dateChangedSetAsString];
607 else
608 [self _dateChangedSetAsNumber];
609}
610
611- (void)setDateTimePickerToInitialValue
612{
613 if ([_initialValue isEqual: @""]) {
614 [_datePicker setDate:[NSDate date]];
615 [self _dateChanged];
616 } else if (_formatString) {
617 // Convert the string value to a date object for the fields where we have a format string.
618 RetainPtr<NSDateFormatter> dateFormatter = [self dateFormatterForPicker];
619 NSDate *parsedDate = [dateFormatter dateFromString:[self _sanitizeInputValueForFormatter:_initialValue.get()]];
620 [_datePicker setDate:parsedDate ? parsedDate : [NSDate date]];
621 } else {
622 // Convert the number value to a date object for the fields affected by timezones.
623 NSTimeInterval secondsSince1970 = _initialValueAsNumber / kMillisecondsPerSecond;
624 NSInteger timeZoneOffset = [self _timeZoneOffsetFromGMT:[NSDate dateWithTimeIntervalSince1970:secondsSince1970]];
625 NSTimeInterval adjustedSecondsSince1970 = secondsSince1970 - timeZoneOffset;
626 [_datePicker setDate:[NSDate dateWithTimeIntervalSince1970:adjustedSecondsSince1970]];
627 }
