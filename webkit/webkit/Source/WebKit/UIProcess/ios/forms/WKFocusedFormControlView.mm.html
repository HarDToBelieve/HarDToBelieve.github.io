54static UIBezierPath *pathWithRoundedRectInFrame(CGRect rect, CGFloat borderRadius, CGRect frame)
55{
56 UIBezierPath *path = [UIBezierPath bezierPathWithRoundedRect:rect cornerRadius:borderRadius];
57 [path appendPath:[UIBezierPath bezierPathWithRect:frame]];
58 return path;
59}

85- (instancetype)initWithFrame:(CGRect)frame delegate:(id <WKFocusedFormControlViewDelegate>)delegate
86{
87 if (!(self = [super initWithFrame:frame]))
88 return nil;
89
90 _delegate = delegate;
91 _highlightedFrame = CGRectZero;
92
93 _dimmingView = adoptNS([[UIView alloc] init]);
94 [_dimmingView setOpaque:NO];
95 [_dimmingView setBackgroundColor:[UIColor colorWithWhite:0 alpha:focusedRectDimmingViewAlpha]];
96 auto maskLayer = adoptNS([[CAShapeLayer alloc] init]);
97 [maskLayer setBackgroundColor:[UIColor blackColor].CGColor];
98 [maskLayer setFillRule:kCAFillRuleEvenOdd];
99 [_dimmingView layer].mask = maskLayer.get();
100 [maskLayer web_disableAllActions];
101
102 // FIXME: If we decide to turn on the focused form control overlay by default, we will need to move the submit and dismiss
103 // buttons to the status bar. See <rdar://problem/39264218>.
104 _submitButton = [UIButton buttonWithType:UIButtonTypeSystem];
105 [_submitButton addTarget:self action:@selector(didSubmit) forControlEvents:UIControlEventTouchUpInside];
106
107 _dismissButton = [UIButton buttonWithType:UIButtonTypeSystem];
108 [_dismissButton addTarget:self action:@selector(didDismiss) forControlEvents:UIControlEventTouchUpInside];
109 [_dismissButton setTitle:WebCore::formControlHideButtonTitle() forState:UIControlStateNormal];
110 [_dismissButton setTitleColor:[UIColor whiteColor] forState:UIControlStateNormal];
111
112 _tapGestureRecognizer = adoptNS([[UITapGestureRecognizer alloc] initWithTarget:self action:@selector(handleTap)]);
113 [_tapGestureRecognizer setNumberOfTapsRequired:1];
114
115 self.opaque = NO;
116 self.layer.masksToBounds = YES;
117 [self addGestureRecognizer:_tapGestureRecognizer.get()];
118
119 _submitButtonBackgroundView = adoptNS([[UIView alloc] init]);
120 [_submitButtonBackgroundView setOpaque:YES];
121 [_submitButtonBackgroundView setBackgroundColor:[UIColor blackColor]];
122
123 [_submitButtonBackgroundView addSubview:_dismissButton.get()];
124 [_submitButtonBackgroundView addSubview:_submitButton.get()];
125 [self addSubview:_dimmingView.get()];
126 [self addSubview:_submitButtonBackgroundView.get()];
127
128 _hasPendingFocusRequest = NO;
129 _initialScrollViewContentOffset = std::nullopt;
130
131 return self;
132}

140- (void)show:(BOOL)animated
141{
142 if (!self.hidden)
143 return;
144
145 self.hidden = NO;
146
147 auto fadeInViewBlock = [view = retainPtr(self)] {
148 [view setAlpha:1];
149 };
150
151 if (animated)
152 [UIView animateWithDuration:formControlViewAnimationDuration delay:0 options:UIViewAnimationOptionBeginFromCurrentState animations:fadeInViewBlock completion:nil];
153 else
154 fadeInViewBlock();
155}
156
157- (void)hide:(BOOL)animated
158{
159 if (self.hidden)
160 return;
161
162 if (!animated) {
163 self.alpha = 0;
164 self.hidden = YES;
165 return;
166 }
167
168 [UIView animateWithDuration:formControlViewAnimationDuration delay:0 options:UIViewAnimationOptionBeginFromCurrentState animations:[view = retainPtr(self)] {
169 [view setAlpha:0];
170 } completion:[view = retainPtr(self)] (BOOL finished)
171 {
172 if (finished)
173 [view setHidden:YES];
174 }];
175}

238- (UIBezierPath *)computeDimmingViewCutoutPath
239{
240 if (CGRectIsEmpty(_highlightedFrame))
241 return nil;
242
243 CGRect inflatedHighlightFrame = UIRectInsetEdges(_highlightedFrame, UIRectEdgeAll, -focusedRectDimmingViewCornerRadius);
244 return pathWithRoundedRectInFrame([self convertRect:inflatedHighlightFrame toView:_dimmingView.get()], focusedRectDimmingViewCornerRadius, [_dimmingView bounds]);
245}

255- (void)engageFocusedFormControlNavigation
256{
257 if (!_crownInputSequencer) {
258 _crownInputSequencer = adoptNS([[PUICCrownInputSequencer alloc] init]);
259 [_crownInputSequencer setRubberBandingEnabled:YES];
260 [_crownInputSequencer setUseWideIdleCheck:YES];
261 [_crownInputSequencer setStart:-1];
262 [_crownInputSequencer setEnd:1];
263 [_crownInputSequencer setScreenSpaceMultiplier:digitalCrownPointsPerUnit];
264 }
265
266 _initialScrollViewContentOffset = self.scrollViewForCrownInputSequencer.contentOffset;
267 [_crownInputSequencer stopVelocityTrackingAndDecelerationImmediately];
268 [_crownInputSequencer setOffset:0];
269 [_crownInputSequencer setDelegate:self];
270}

285- (void)setMaskLayerPosition:(CGPoint)position animated:(BOOL)animated
286{
287 if (animated) {
288 CABasicAnimation *maskAnimation = [CABasicAnimation animationWithKeyPath:@"position"];
289 maskAnimation.fromValue = [NSValue valueWithCGPoint:self.dimmingMaskLayer.position];
290 maskAnimation.toValue = [NSValue valueWithCGPoint:position];
291 maskAnimation.duration = dimmingViewAnimationDuration;
292 maskAnimation.timingFunction = [CAMediaTimingFunction functionWithName:kCAMediaTimingFunctionEaseInEaseOut];
293 [self.dimmingMaskLayer addAnimation:maskAnimation forKey:nil];
294 }
295 self.dimmingMaskLayer.position = position;
296}

298- (void)setHighlightedFrame:(CGRect)highlightedFrame animated:(BOOL)animated
299{
300 if (CGRectEqualToRect(_highlightedFrame, highlightedFrame))
301 return;
302
303 _highlightedFrame = highlightedFrame;
304
305 UIBezierPath *updatedCutoutPath = [self computeDimmingViewCutoutPath];
306 if (!animated || !self.dimmingMaskLayer.path || !updatedCutoutPath) {
307 self.dimmingMaskLayer.path = updatedCutoutPath.CGPath;
308 return;
309 }
310
311 CABasicAnimation *maskAnimation = [CABasicAnimation animationWithKeyPath:@"path"];
312 maskAnimation.fromValue = (__bridge UIBezierPath *)self.dimmingMaskLayer.path;
313 maskAnimation.toValue = updatedCutoutPath;
314 maskAnimation.duration = dimmingViewAnimationDuration;
315 maskAnimation.timingFunction = [CAMediaTimingFunction functionWithName:kCAMediaTimingFunctionEaseInEaseOut];
316 [self.dimmingMaskLayer addAnimation:maskAnimation forKey:nil];
317 self.dimmingMaskLayer.path = updatedCutoutPath.CGPath;
318}

351- (void)updateViewForCurrentCrownInputSequencerState
352{
353 if (!_initialScrollViewContentOffset)
354 return;
355
356 CGFloat crownOffset = [_crownInputSequencer offset];
357 UIScrollView *scrollView = self.scrollViewForCrownInputSequencer;
358 CGPoint targetContentOffset = [self scrollOffsetForCrownInputOffset:crownOffset];
359 scrollView.contentOffset = targetContentOffset;
360 self.dimmingMaskLayer.position = CGPointMake(_initialScrollViewContentOffset->x - targetContentOffset.x, _initialScrollViewContentOffset->y - targetContentOffset.y);
361
362 if (_hasPendingFocusRequest)
363 return;
364
365 if (crownOffset > 1 && self.hasNextNode) {
366 [_delegate focusedFormControlViewDidRequestNextNode:self];
367 _hasPendingFocusRequest = YES;
368 }
369
370 if (crownOffset < -1 && self.hasPreviousNode) {
371 [_delegate focusedFormControlViewDidRequestPreviousNode:self];
372 _hasPendingFocusRequest = YES;
373 }
374}

376- (CGPoint)scrollOffsetForCrownInputOffset:(CGFloat)crownOffset
377{
378 ASSERT(_initialScrollViewContentOffset);
379 if (!crownOffset)
380 return *_initialScrollViewContentOffset;
381
382 CGPoint highlightedFrameMidpoint { CGRectGetMidX(_highlightedFrame), CGRectGetMidY(_highlightedFrame) };
383 CGVector unitVectorToTarget = CGVectorMake(0, crownOffset < 0 ? -1 : 1);
384 CGFloat targetHighlightedFrameDeltaX = 0;
385 CGFloat targetHighlightedFrameDeltaY = 0;
386 BOOL isMovingTowardsNextOrPreviousNode = NO;
387 if (crownOffset < 0 && self.hasPreviousNode) {
388 targetHighlightedFrameDeltaX = CGRectGetMidX(_previousHighlightedFrame) - highlightedFrameMidpoint.x;
389 targetHighlightedFrameDeltaY = CGRectGetMidY(_previousHighlightedFrame) - highlightedFrameMidpoint.y;
390 isMovingTowardsNextOrPreviousNode = YES;
391 } else if (crownOffset > 0 && self.hasNextNode) {
392 targetHighlightedFrameDeltaX = CGRectGetMidX(_nextHighlightedFrame) - highlightedFrameMidpoint.x;
393 targetHighlightedFrameDeltaY = CGRectGetMidY(_nextHighlightedFrame) - highlightedFrameMidpoint.y;
394 isMovingTowardsNextOrPreviousNode = YES;
395 }
396
397 if (targetHighlightedFrameDeltaX || targetHighlightedFrameDeltaY) {
398 CGFloat distanceToTarget = std::hypot(targetHighlightedFrameDeltaX, targetHighlightedFrameDeltaY);
399 unitVectorToTarget = CGVector { targetHighlightedFrameDeltaX / distanceToTarget, targetHighlightedFrameDeltaY / distanceToTarget };
400 }
401
402 CGFloat distancePerUnit = isMovingTowardsNextOrPreviousNode ? digitalCrownActiveScrollDistancePerUnit : digitalCrownInactiveScrollDistancePerUnit;
403 CGFloat absoluteClampedCrownOffset = std::min<CGFloat>(digitalCrownMaximumScrollDistance / distancePerUnit, ABS(crownOffset));
404 return {
405 _initialScrollViewContentOffset->x + distancePerUnit * unitVectorToTarget.dx * absoluteClampedCrownOffset,
406 _initialScrollViewContentOffset->y + distancePerUnit * unitVectorToTarget.dy * absoluteClampedCrownOffset
407 };
408}

410- (void)_crownInputSequencerTimerFired
411{
412 CGFloat currentCrownInputSequencerOffset = [_crownInputSequencer offset];
413 if (ABS(currentCrownInputSequencerOffset) < 0.01) {
414 [_crownInputSequencer setOffset:0];
415 [self updateViewForCurrentCrownInputSequencerState];
416 return;
417 }
418
419 [_crownInputSequencer setOffset:0.75 * currentCrownInputSequencerOffset];
420 [self updateViewForCurrentCrownInputSequencerState];
421 [self scheduleCrownInputSequencerUpdate];
422}

424- (void)cancelPendingCrownInputSequencerUpdate
425{
426 [NSObject cancelPreviousPerformRequestsWithTarget:self selector:@selector(_crownInputSequencerTimerFired) object:nil];
427}

429- (void)scheduleCrownInputSequencerUpdate

431 [self performSelector:@selector(_crownInputSequencerTimerFired) withObject:nil afterDelay:1 / 60.];
432}

462- (void)setSuggestions:(NSArray<UITextSuggestion *> *)suggestions
463{
464 RetainPtr<NSMutableArray> displayableTextSuggestions;
465 if (suggestions) {
466 displayableTextSuggestions = adoptNS([[NSMutableArray alloc] initWithCapacity:suggestions.count]);
467 for (UITextSuggestion *suggestion in suggestions) {
468 if (!suggestion.displayText.length)
469 continue;
470
471 [displayableTextSuggestions addObject:suggestion];
472 }
473 }
474
475 if (_textSuggestions == displayableTextSuggestions.get() || [_textSuggestions isEqualToArray:displayableTextSuggestions.get()])
476 return;
477
478 _textSuggestions = WTFMove(displayableTextSuggestions);
479 [_delegate focusedFormControllerDidUpdateSuggestions:self];
480}
