146- (instancetype)initWithView:(WKContentView *)view
147{
148 if (!(self = [super initWithFrame:CGRectZero]))
149 return nil;
150
151 _view = view;
152 _allowsMultipleSelection = _view.focusedElementInformation.isMultiSelect;
153 _singleSelectionIndex = NSNotFound;
154 [self setAutoresizingMask:UIViewAutoresizingFlexibleWidth | UIViewAutoresizingFlexibleHeight];
155 [self setDataSource:self];
156 [self setDelegate:self];
157 [self _setUsesCheckedSelection:YES];
158
159 [self _setMagnifierEnabled:NO];
160 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
161 UITextWritingDirection writingDirection = UITextWritingDirectionLeftToRight;
162 // FIXME: retrieve from WebProcess writing direction.
163 _textAlignment = (writingDirection == UITextWritingDirectionLeftToRight) ? NSTextAlignmentLeft : NSTextAlignmentRight;
164 ALLOW_DEPRECATED_DECLARATIONS_END

166 [self setAllowsMultipleSelection:_allowsMultipleSelection];
167 [self setSize:[UIKeyboard defaultSizeForInterfaceOrientation:view.interfaceOrientation]];
168 [self reloadAllComponents];
169
170 if (!_allowsMultipleSelection) {
171 const Vector<OptionItem>& selectOptions = [_view focusedSelectElementOptions];
172 for (size_t i = 0; i < selectOptions.size(); ++i) {
173 const OptionItem& item = selectOptions[i];
174 if (item.isGroup)
175 continue;
176
177 if (item.isSelected) {
178 _singleSelectionIndex = i;
179 [self selectRow:_singleSelectionIndex inComponent:0 animated:NO];
180 break;
181 }
182 }
183 }

223- (UIView *)pickerView:(UIPickerView *)pickerView viewForRow:(NSInteger)rowIndex forComponent:(NSInteger)columnIndex reusingView:(UIView *)view
224{
225 auto& item = [_view focusedSelectElementOptions][rowIndex];
226 RetainPtr<WKOptionPickerCell> pickerItem;
227 if (item.isGroup)
228 pickerItem = adoptNS([[WKOptionGroupPickerCell alloc] initWithOptionItem:item]);
229 else
230 pickerItem = adoptNS([[WKOptionPickerCell alloc] initWithOptionItem:item]);
231
232 // The cell starts out with a null frame. We need to set its frame now so we can find the right font size.
233 UITableView *table = [pickerView tableViewForColumn:0];
234 CGRect frame = [table rectForRowAtIndexPath:[NSIndexPath indexPathForRow:rowIndex inSection:0]];
235 [pickerItem setFrame:frame];
236
237 UILabel *titleTextLabel = [pickerItem titleLabel];
238 float width = [pickerItem labelWidthForBounds:CGRectMake(0, 0, CGRectGetWidth(frame), CGRectGetHeight(frame))];
239 ASSERT(width > 0);
240
241 // Assume all cells have the same available text width.
242 UIFont *font = titleTextLabel.font;
243 if (width != _maximumTextWidth || _fontSize == 0) {
244 _maximumTextWidth = width;
245 _fontSize = adjustedFontSize(_maximumTextWidth, font, titleTextLabel.font.pointSize, [_view focusedSelectElementOptions]);
246 }
247
248 [titleTextLabel setFont:[font fontWithSize:_fontSize]];
249 [titleTextLabel setLineBreakMode:NSLineBreakByWordWrapping];
250 [titleTextLabel setNumberOfLines:2];
251 [titleTextLabel setTextAlignment:_textAlignment];
252
253 return pickerItem.autorelease();
254}

266- (NSInteger)findItemIndexAt:(int)rowIndex
267{
268 ASSERT(rowIndex >= 0 && (size_t)rowIndex < [_view focusedSelectElementOptions].size());
269 NSInteger itemIndex = 0;
270 for (int i = 0; i < rowIndex; ++i) {
271 if ([_view focusedSelectElementOptions][i].isGroup)
272 continue;
273 itemIndex++;
274 }
275
276 ASSERT(itemIndex >= 0);
277 return itemIndex;

280- (void)pickerView:(UIPickerView *)pickerView row:(int)rowIndex column:(int)columnIndex checked:(BOOL)isChecked
281{
282 auto numberOfOptions = static_cast<NSUInteger>([_view focusedSelectElementOptions].size());
283 if (numberOfOptions <= static_cast<NSUInteger>(rowIndex))
284 return;
285
286 auto& item = [_view focusedSelectElementOptions][rowIndex];
287
288 // FIXME: Remove this workaround once <rdar://problem/18745253> is fixed.
289 // Group rows and disabled rows should not be checkable, but we are getting
290 // this delegate for those rows. As a workaround, if we get this delegate
291 // for a group or disabled row, reset the styles for the content view so it
292 // still appears unselected.
293 if (item.isGroup || item.disabled) {
294 UIPickerContentView *view = (UIPickerContentView *)[self viewForRow:rowIndex forComponent:columnIndex];
295 [view setChecked:NO];
296 [[view titleLabel] setTextColor:[UIColor colorWithWhite:0.0 alpha:item.isGroup ? GroupOptionTextColorAlpha : DisabledOptionAlpha]];
297 return;
298 }
299
300 if ([self allowsMultipleSelection]) {
301 [_view page]->setFocusedElementSelectedIndex([self findItemIndexAt:rowIndex], true);
302 item.isSelected = isChecked;
303 } else if (isChecked) {
304 // Single selection.
305 if (_singleSelectionIndex < numberOfOptions)
306 [_view focusedSelectElementOptions][_singleSelectionIndex].isSelected = false;
307
308 _singleSelectionIndex = rowIndex;
309
310 // This private delegate often gets called for multiple rows in the picker,
311 // so we only activate and set as selected the checked item in single selection.
312 [_view page]->setFocusedElementSelectedIndex([self findItemIndexAt:rowIndex]);
313 item.isSelected = true;
314 } else
315 item.isSelected = false;
316}

408- (NSAttributedString *)pickerView:(UIPickerView *)pickerView attributedTitleForRow:(NSInteger)row forComponent:(NSInteger)component
409{
410 if (row < 0 || row >= (NSInteger)[_view focusedSelectElementOptions].size())
411 return nil;
412
413 const OptionItem& option = [_view focusedSelectElementOptions][row];
414 auto trimmedText = adoptNS([option.text mutableCopy]);
415 CFStringTrimWhitespace((CFMutableStringRef)trimmedText.get());
416
417 auto attributedString = adoptNS([[NSMutableAttributedString alloc] initWithString:trimmedText.get()]);
418 if (option.disabled)
419 [attributedString addAttribute:NSForegroundColorAttributeName value:[UIColor colorWithWhite:0.0 alpha:DisabledOptionAlpha] range:NSMakeRange(0, [trimmedText length])];
420
421 return attributedString.autorelease();
422}

424- (void)pickerView:(UIPickerView *)pickerView didSelectRow:(NSInteger)row inComponent:(NSInteger)component
425{
426 if (row < 0 || row >= (NSInteger)[_view focusedSelectElementOptions].size())
427 return;
428
429 const OptionItem& newSelectedOption = [_view focusedSelectElementOptions][row];
430 if (newSelectedOption.disabled) {
431 NSInteger rowToSelect = NSNotFound;
432
433 // Search backwards for the previous enabled option.
434 for (NSInteger i = row - 1; i >= 0; --i) {
435 const OptionItem& earlierOption = [_view focusedSelectElementOptions][i];
436 if (!earlierOption.disabled) {
437 rowToSelect = i;
438 break;
439 }
440 }
441
442 // If nothing previous, search forwards for the next enabled option.
443 if (rowToSelect == NSNotFound) {
444 for (size_t i = row + 1; i < [_view focusedSelectElementOptions].size(); ++i) {
445 const OptionItem& laterOption = [_view focusedSelectElementOptions][i];
446 if (!laterOption.disabled) {
447 rowToSelect = i;
448 break;
449 }
450 }
451 }
452
453 if (rowToSelect == NSNotFound)
454 return;
455
456 [self selectRow:rowToSelect inComponent:0 animated:YES];
457 row = rowToSelect;
458 }
459
460 _selectedIndex = row;
461}

536- (void)didSelectOptionIndex:(NSInteger)index
537{
538 NSInteger optionIndex = 0;
539 for (auto& option : _view.focusedSelectElementOptions) {
540 if (option.isGroup)
541 continue;
542
543 option.isSelected = optionIndex == index;
544 optionIndex++;
545 }

552- (UIMenu *)createMenu
553{
554 if (!_view.focusedSelectElementOptions.size()) {
555 UIAction *emptyAction = [UIAction actionWithTitle:WEB_UI_STRING_KEY("No Options", "No Options Select Popover", "Empty select list") image:nil identifier:nil handler:^(__kindof UIAction *action) { }];
556 emptyAction.attributes = UIMenuElementAttributesDisabled;
557 return [UIMenu menuWithTitle:@"" children:@[emptyAction]];
558 }
559
560 NSMutableArray *items = [NSMutableArray array];
561 NSInteger optionIndex = 0;
562
563 size_t currentIndex = 0;
564 while (currentIndex < _view.focusedSelectElementOptions.size()) {
565 auto& optionItem = _view.focusedSelectElementOptions[currentIndex];
566 if (optionItem.isGroup) {
567 NSString *groupText = optionItem.text;
568 NSMutableArray *groupedItems = [NSMutableArray array];
569
570 if (groupText.length) {
571 UIAction *action = [UIAction actionWithTitle:groupText image:nil identifier:nil handler:^(UIAction *action) { }];
572 action.attributes = UIMenuElementAttributesDisabled;
573 [groupedItems addObject:action];
574 }
575
576 currentIndex++;
577 while (currentIndex < _view.focusedSelectElementOptions.size()) {
578 auto& childOptionItem = _view.focusedSelectElementOptions[currentIndex];
579 if (childOptionItem.isGroup)
580 break;
581
582 UIAction *action = [self actionForOptionItem:childOptionItem withIndex:optionIndex];
583 [groupedItems addObject:action];
584 optionIndex++;
585 currentIndex++;
586 }
587
588 UIMenu *groupMenu = [UIMenu menuWithTitle:groupText image:nil identifier:nil options:(UIMenuOptionsDisplayInline | UIMenuOptionsPrivateRemoveLineLimitForChildren) children:groupedItems];
589 [items addObject:groupMenu];
590 continue;
591 }
592
593 UIAction *action = [self actionForOptionItem:optionItem withIndex:optionIndex];
594 [items addObject:action];
595 optionIndex++;
596 currentIndex++;
597 }
598
599 UIMenuOptions options = UIMenuOptionsPrivateRemoveLineLimitForChildren;
600#if HAVE(UIMENUOPTIONS_SINGLE_SELECTION)
601 options |= UIMenuOptionsSingleSelection;
602#endif
603
604 return [UIMenu menuWithTitle:@"" image:nil identifier:nil options:options children:items];
605}

607- (UIAction *)actionForOptionItem:(const OptionItem&)option withIndex:(NSInteger)optionIndex
608{
609 UIAction *optionAction = [UIAction actionWithTitle:option.text image:nil identifier:nil handler:^(__kindof UIAction *action) {
610 [self didSelectOptionIndex:optionIndex];
611 }];
612
613 if (option.disabled)
614 optionAction.attributes = UIMenuElementAttributesDisabled;
615
616 if (option.isSelected)
617 optionAction.state = UIMenuElementStateOn;
618
619 return optionAction;
620}
621
622- (UIAction *)actionForOptionIndex:(NSInteger)optionIndex
623{
624 NSInteger currentIndex = 0;
625
626 NSArray<UIMenuElement *> *menuElements = [_selectMenu children];
627 for (UIMenuElement *menuElement in menuElements) {
628 if ([menuElement isKindOfClass:UIAction.class]) {
629 if (currentIndex == optionIndex)
630 return (UIAction *)menuElement;
631
632 currentIndex++;
633 continue;
634 }
635
636 UIMenu *groupedMenu = (UIMenu *)menuElement;
637 NSUInteger numGroupedOptions = groupedMenu.children.count;
638
639 // The first child of a grouped menu with a title represents the title, and is not a selectable option.
640 if (groupedMenu.title.length)
641 numGroupedOptions--;
642
643 if (currentIndex + numGroupedOptions <= (NSUInteger)optionIndex)
644 currentIndex += numGroupedOptions;
645 else
646 return (UIAction *)[groupedMenu.children objectAtIndex:(groupedMenu.children.count - numGroupedOptions) + (optionIndex - currentIndex)];
647 }
648
649 return nil;
650}
