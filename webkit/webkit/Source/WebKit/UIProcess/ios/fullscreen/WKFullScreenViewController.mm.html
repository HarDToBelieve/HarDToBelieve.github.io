50class WKFullScreenViewControllerPlaybackSessionModelClient : WebCore::PlaybackSessionModelClient {
51public:
52 void setParent(WKFullScreenViewController *parent) { m_parent = parent; }
53
54 void rateChanged(OptionSet<WebCore::PlaybackSessionModel::PlaybackState> playbackState, double /* playbackRate */, double /* defaultPlaybackRate */) override
55 {
56 if (auto *controller = m_parent.getAutoreleased())
57 controller.playing = playbackState.contains(WebCore::PlaybackSessionModel::PlaybackState::Playing);
58 }
59
60 void isPictureInPictureSupportedChanged(bool) override
61 {
62 }
63
64 void pictureInPictureActiveChanged(bool active) override
65 {
66 if (auto *controller = m_parent.getAutoreleased())
67 controller.pictureInPictureActive = active;
68 }
69
70 void setInterface(WebCore::PlaybackSessionInterfaceAVKit* interface)
71 {
72 if (m_interface == interface)
73 return;
74
75 if (m_interface && m_interface->playbackSessionModel())
76 m_interface->playbackSessionModel()->removeClient(*this);
77 m_interface = interface;
78 if (m_interface && m_interface->playbackSessionModel())
79 m_interface->playbackSessionModel()->addClient(*this);
80 }
81
82private:
83 WeakObjCPtr<WKFullScreenViewController> m_parent;
84 RefPtr<WebCore::PlaybackSessionInterfaceAVKit> m_interface;
85};

177- (void)showUI
178{
179 ASSERT(_valid);
180 [NSObject cancelPreviousPerformRequestsWithTarget:self selector:@selector(hideUI) object:nil];
181
182 if (_playing) {
183 NSTimeInterval hideDelay = autoHideDelay;
184 [self performSelector:@selector(hideUI) withObject:nil afterDelay:hideDelay];
185 }
186 [UIView animateWithDuration:showHideAnimationDuration animations:^{
187 [_stackView setHidden:NO];
188 [_stackView setAlpha:1];
189 self.prefersStatusBarHidden = NO;
190 self.prefersHomeIndicatorAutoHidden = NO;
191 if (_topConstraint)
192 [NSLayoutConstraint deactivateConstraints:@[_topConstraint.get()]];
193 _topConstraint = [[_topGuide topAnchor] constraintEqualToAnchor:self.view.safeAreaLayoutGuide.topAnchor];
194 [_topConstraint setActive:YES];
195 if (auto* manager = self._manager)
196 manager->setFullscreenControlsHidden(false);
197 }];
198}
199
200- (void)hideUI
201{
202 ASSERT(_valid);
203 [NSObject cancelPreviousPerformRequestsWithTarget:self selector:@selector(hideUI) object:nil];
204 [UIView animateWithDuration:showHideAnimationDuration animations:^{
205
206 if (_topConstraint)
207 [NSLayoutConstraint deactivateConstraints:@[_topConstraint.get()]];
208 _topConstraint = [[_topGuide topAnchor] constraintEqualToAnchor:self.view.topAnchor constant:self.view.safeAreaInsets.top];
209 [_topConstraint setActive:YES];
210 [_stackView setAlpha:0];
211 self.prefersStatusBarHidden = YES;
212 self.prefersHomeIndicatorAutoHidden = YES;
213 if (auto* manager = self._manager)
214 manager->setFullscreenControlsHidden(true);
215 } completion:^(BOOL finished) {
216 if (!finished)
217 return;
218
219 [_stackView setHidden:YES];
220 }];
221}

223- (void)videoControlsManagerDidChange
224{
225 ASSERT(_valid);
226 auto page = [self._webView _page];
227 auto* videoFullscreenManager = page ? page->videoFullscreenManager() : nullptr;
228 auto* videoFullscreenInterface = videoFullscreenManager ? videoFullscreenManager->controlsManagerInterface() : nullptr;
229 auto* playbackSessionInterface = videoFullscreenInterface ? &videoFullscreenInterface->playbackSessionInterface() : nullptr;
230
231 _playbackClient.setInterface(playbackSessionInterface);
232
233 WebCore::PlaybackSessionModel* playbackSessionModel = playbackSessionInterface ? playbackSessionInterface->playbackSessionModel() : nullptr;
234 self.playing = playbackSessionModel ? playbackSessionModel->isPlaying() : NO;
235 bool isPiPEnabled = false;
236 if (auto page = [self._webView _page])
237 isPiPEnabled = page->preferences().pictureInPictureAPIEnabled() && page->preferences().allowsPictureInPictureMediaPlayback();
238 bool isPiPSupported = playbackSessionModel && playbackSessionModel->isPictureInPictureSupported();
239 [_pipButton setHidden:!isPiPEnabled || !isPiPSupported];
240}

265- (void)setPlaying:(BOOL)isPlaying
266{
267 ASSERT(_valid);
268 if (_playing == isPlaying)
269 return;
270
271 _playing = isPlaying;
272
273 if (![self viewIfLoaded])
274 return;
275
276 if (!_playing)
277 [self showUI];
278 else {
279 [NSObject cancelPreviousPerformRequestsWithTarget:self selector:@selector(hideUI) object:nil];
280 NSTimeInterval hideDelay = autoHideDelay;
281 [self performSelector:@selector(hideUI) withObject:nil afterDelay:hideDelay];
282 }
283}

450@dynamic _effectiveFullscreenInsets;
451- (WebCore::FloatBoxExtent)_effectiveFullscreenInsets
452{
453 ASSERT(_valid);
454 auto safeAreaInsets = self.view.safeAreaInsets;
455 WebCore::FloatBoxExtent insets { safeAreaInsets.top, safeAreaInsets.right, safeAreaInsets.bottom, safeAreaInsets.left };
456
457 CGRect cancelFrame = _cancelButton.get().frame;
458 CGPoint maxXY = CGPointMake(CGRectGetMaxX(cancelFrame), CGRectGetMaxY(cancelFrame));
459 insets.setTop([_cancelButton convertPoint:maxXY toView:self.view].y);
460 return insets;
461}

469- (void)_togglePiPAction:(id)sender
470{
471 ASSERT(_valid);
472 auto page = [self._webView _page];
473 if (!page)
474 return;
475
476 WebKit::PlaybackSessionManagerProxy* playbackSessionManager = page->playbackSessionManager();
477 if (!playbackSessionManager)
478 return;
479
480 PlatformPlaybackSessionInterface* playbackSessionInterface = playbackSessionManager->controlsManagerInterface();
481 if (!playbackSessionInterface)
482 return;
483
484 WebCore::PlaybackSessionModel* playbackSessionModel = playbackSessionInterface->playbackSessionModel();
485 if (!playbackSessionModel)
486 return;
487
488 playbackSessionModel->togglePictureInPicture();
489}

491- (void)_touchDetected:(id)sender
492{
493 ASSERT(_valid);
494 if ([_touchGestureRecognizer state] == UIGestureRecognizerStateEnded) {
495 double score = _secheuristic.scoreOfNextTouch([_touchGestureRecognizer locationInView:self.view]);
496 if (score > _secheuristic.requiredScore())
497 [self _showPhishingAlert];
498 }
499 if (!self.animating)
500 [self showUI];
501}

523- (void)_showPhishingAlert
524{
525 ASSERT(_valid);
526 NSString *alertTitle = WEB_UI_STRING("It looks like you are typing while in full screen", "Full Screen Deceptive Website Warning Sheet Title");
527 NSString *alertMessage = [NSString stringWithFormat:WEB_UI_STRING("Typing is not allowed in full screen websites. “%@” may be showing a fake keyboard to trick you into disclosing personal or financial information.", "Full Screen Deceptive Website Warning Sheet Content Text"), (NSString *)self.location];
528 UIAlertController* alert = [UIAlertController alertControllerWithTitle:alertTitle message:alertMessage preferredStyle:UIAlertControllerStyleAlert];
529
530 if (auto page = [self._webView _page]) {
531 page->suspendAllMediaPlayback([] { });
532 page->suspendActiveDOMObjectsAndAnimations();
533 }
534
535 UIAlertAction* exitAction = [UIAlertAction actionWithTitle:WEB_UI_STRING_KEY("Exit Full Screen", "Exit Full Screen (Element Full Screen)", "Full Screen Deceptive Website Exit Action") style:UIAlertActionStyleCancel handler:^(UIAlertAction * action) {
536 [self _cancelAction:action];
537 if (auto page = [self._webView _page]) {
538 page->resumeActiveDOMObjectsAndAnimations();
539 page->resumeAllMediaPlayback([] { });
540 }
541 }];
542
543 UIAlertAction* stayAction = [UIAlertAction actionWithTitle:WEB_UI_STRING_KEY("Stay in Full Screen", "Stay in Full Screen (Element Full Screen)", "Full Screen Deceptive Website Stay Action") style:UIAlertActionStyleDefault handler:^(UIAlertAction * action) {
544 if (auto page = [self._webView _page]) {
545 page->resumeActiveDOMObjectsAndAnimations();
546 page->resumeAllMediaPlayback([] { });
547 }
548 _secheuristic.reset();
549 }];
550
551 [alert addAction:exitAction];
552 [alert addAction:stayAction];
553 [self presentViewController:alert animated:YES completion:nil];
554}
