43DisplayLink::DisplayLink(WebCore::PlatformDisplayID displayID)
44 : m_displayID(displayID)
45{
46 // FIXME: We can get here with displayID == 0 (webkit.org/b/212120), in which case CVDisplayLinkCreateWithCGDisplay()
47 // probably defaults to the main screen.
48 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
49 CVReturn error = CVDisplayLinkCreateWithCGDisplay(displayID, &m_displayLink);
50 if (error) {
51 WTFLogAlways("Could not create a display link for display %u: error %d", displayID, error);
52 return;
53 }
54
55 error = CVDisplayLinkSetOutputCallback(m_displayLink, displayLinkCallback, this);
56 if (error) {
57 WTFLogAlways("Could not set the display link output callback for display %u: error %d", displayID, error);
58 return;
59 }
60
61 m_displayNominalFramesPerSecond = nominalFramesPerSecondFromDisplayLink(m_displayLink);
62
63 LOG_WITH_STREAM(DisplayLink, stream << "[UI ] Created DisplayLink " << this << " for display " << displayID << " with nominal fps " << m_displayNominalFramesPerSecond);
64}
65
66DisplayLink::~DisplayLink()
67{
68 LOG_WITH_STREAM(DisplayLink, stream << "[UI ] Destroying DisplayLink " << this << " for display " << m_displayID);
69
70 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
71 ASSERT(m_displayLink);
72 if (!m_displayLink)
73 return;
74
75 CVDisplayLinkStop(m_displayLink);
76 CVDisplayLinkRelease(m_displayLink);
77}

79WebCore::FramesPerSecond DisplayLink::nominalFramesPerSecondFromDisplayLink(CVDisplayLinkRef displayLink)
80{
81 CVTime refreshPeriod = CVDisplayLinkGetNominalOutputVideoRefreshPeriod(displayLink);
82 return round((double)refreshPeriod.timeScale / (double)refreshPeriod.timeValue);
83}

85void DisplayLink::addObserver(IPC::Connection& connection, DisplayLinkObserverID observerID, WebCore::FramesPerSecond preferredFramesPerSecond)
86{
87 ASSERT(RunLoop::isMain());
88
89 LOG_WITH_STREAM(DisplayLink, stream << "[UI ] DisplayLink " << this << " for display display " << m_displayID << " add observer " << observerID << " fps " << preferredFramesPerSecond);
90
91 {
92 Locker locker { m_observersLock };
93 m_observers.ensure(connection.uniqueID(), [] {
94 return ConnectionClientInfo { };
95 }).iterator->value.observers.append({ observerID, preferredFramesPerSecond });
96 }
97
98 if (!CVDisplayLinkIsRunning(m_displayLink)) {
99 LOG_WITH_STREAM(DisplayLink, stream << "[UI ] DisplayLink for display " << m_displayID << " starting CVDisplayLink with fps " << m_displayNominalFramesPerSecond);
100 CVReturn error = CVDisplayLinkStart(m_displayLink);
101 if (error)
102 WTFLogAlways("Could not start the display link: %d", error);
103
104 m_currentUpdate = { 0, m_displayNominalFramesPerSecond };
105 }
106}

108void DisplayLink::removeObserver(IPC::Connection& connection, DisplayLinkObserverID observerID)
109{
110 ASSERT(RunLoop::isMain());
111
112 Locker locker { m_observersLock };
113
114 auto it = m_observers.find(connection.uniqueID());
115 if (it == m_observers.end())
116 return;
117
118 auto& connectionInfo = it->value;
119
120 bool removed = connectionInfo.observers.removeFirstMatching([observerID](const auto& value) {
121 return value.observerID == observerID;
122 });
123 ASSERT_UNUSED(removed, removed);
124
125 LOG_WITH_STREAM(DisplayLink, stream << "[UI ] DisplayLink " << this << " for display " << m_displayID << " remove observer " << observerID);
126
127 removeInfoForConnectionIfPossible(connection);
128
129 // We do not stop the display link right away when |m_observers| becomes empty. Instead, we
130 // let the display link fire up to |maxFireCountWithoutObservers| times without observers to avoid
131 // killing & restarting too many threads when observers gets removed & added in quick succession.
132}
133
134void DisplayLink::removeObservers(IPC::Connection& connection)
135{
136 ASSERT(RunLoop::isMain());
137
138 Locker locker { m_observersLock };
139 m_observers.remove(connection.uniqueID());
140
141 // We do not stop the display link right away when |m_observers| becomes empty. Instead, we
142 // let the display link fire up to |maxFireCountWithoutObservers| times without observers to avoid
143 // killing & restarting too many threads when observers gets removed & added in quick succession.
144}

146void DisplayLink::removeInfoForConnectionIfPossible(IPC::Connection& connection)
147{
148 auto it = m_observers.find(connection.uniqueID());
149 if (it == m_observers.end())
150 return;
151
152 auto& connectionInfo = it->value;
153 if (connectionInfo.observers.isEmpty() && !connectionInfo.fullSpeedUpdatesClientCount)
154 m_observers.remove(it);
155}

157void DisplayLink::incrementFullSpeedRequestClientCount(IPC::Connection& connection)
158{
159 Locker locker { m_observersLock };
160
161 auto& connectionInfo = m_observers.ensure(connection.uniqueID(), [] {
162 return ConnectionClientInfo { };
163 }).iterator->value;
164
165 ++connectionInfo.fullSpeedUpdatesClientCount;
166}
167
168void DisplayLink::decrementFullSpeedRequestClientCount(IPC::Connection& connection)
169{
170 Locker locker { m_observersLock };
171
172 auto it = m_observers.find(connection.uniqueID());
173 if (it == m_observers.end())
174 return;
175
176 auto& connectionInfo = it->value;
177 ASSERT(connectionInfo.fullSpeedUpdatesClientCount);
178 --connectionInfo.fullSpeedUpdatesClientCount;
179 removeInfoForConnectionIfPossible(connection);
180}

182void DisplayLink::setPreferredFramesPerSecond(IPC::Connection& connection, DisplayLinkObserverID observerID, WebCore::FramesPerSecond preferredFramesPerSecond)
183{
184 LOG_WITH_STREAM(DisplayLink, stream << "[UI ] DisplayLink " << this << " setPreferredFramesPerSecond - display " << m_displayID << " observer " << observerID << " fps " << preferredFramesPerSecond);
185
186 Locker locker { m_observersLock };
187
188 auto it = m_observers.find(connection.uniqueID());
189 if (it == m_observers.end())
190 return;
191
192 auto& connectionInfo = it->value;
193 auto index = connectionInfo.observers.findMatching([observerID](const auto& observer) {
194 return observer.observerID == observerID;
195 });
196
197 if (index != notFound)
198 connectionInfo.observers[index].preferredFramesPerSecond = preferredFramesPerSecond;
199}

201CVReturn DisplayLink::displayLinkCallback(CVDisplayLinkRef displayLinkRef, const CVTimeStamp*, const CVTimeStamp*, CVOptionFlags, CVOptionFlags*, void* data)
202{
203 static_cast<DisplayLink*>(data)->notifyObserversDisplayWasRefreshed();
204 return kCVReturnSuccess;
205}

207void DisplayLink::notifyObserversDisplayWasRefreshed()
208{
209 ASSERT(!RunLoop::isMain());
210
211 Locker locker { m_observersLock };
212
213 auto maxFramesPerSecond = [](const Vector<ObserverInfo>& observers) {
214 std::optional<WebCore::FramesPerSecond> observersMaxFramesPerSecond;
215 for (const auto& observer : observers)
216 observersMaxFramesPerSecond = std::max(observersMaxFramesPerSecond.value_or(0), observer.preferredFramesPerSecond);
217 return observersMaxFramesPerSecond;
218 };
219
220 bool anyConnectionHadObservers = false;
221 for (auto& [connectionID, connectionInfo] : m_observers) {
222 if (connectionInfo.observers.isEmpty())
223 continue;
224
225 anyConnectionHadObservers = true;
226
227 auto observersMaxFramesPerSecond = maxFramesPerSecond(connectionInfo.observers);
228 auto mainThreadWantsUpdate = m_currentUpdate.relevantForUpdateFrequency(observersMaxFramesPerSecond.value_or(WebCore::FullSpeedFramesPerSecond));
229
230 LOG_WITH_STREAM(DisplayLink, stream << "[UI ] DisplayLink " << this << " for display " << m_displayID << " (display fps " << m_displayNominalFramesPerSecond << ") update " << m_currentUpdate << " " << connectionInfo.observers.size()
231 << " observers, on background queue " << shouldSendIPCOnBackgroundQueue << " maxFramesPerSecond " << observersMaxFramesPerSecond << " full speed clients " << connectionInfo.fullSpeedUpdatesClientCount << " relevant " << mainThreadWantsUpdate);
232
233 if (connectionInfo.fullSpeedUpdatesClientCount) {
234 IPC::Connection::send(connectionID, Messages::EventDispatcher::DisplayWasRefreshed(m_displayID, m_currentUpdate, mainThreadWantsUpdate), 0);
235 } else if (mainThreadWantsUpdate)
236 IPC::Connection::send(connectionID, Messages::WebProcess::DisplayWasRefreshed(m_displayID, m_currentUpdate), 0);
237 }
238
239 m_currentUpdate = m_currentUpdate.nextUpdate();
240
241 if (!anyConnectionHadObservers) {
242 if (++m_fireCountWithoutObservers >= maxFireCountWithoutObservers) {
243 LOG_WITH_STREAM(DisplayLink, stream << "[UI ] DisplayLink for display " << m_displayID << " fired " << m_fireCountWithoutObservers << " times with no observers; stopping CVDisplayLink");
244 CVDisplayLinkStop(m_displayLink);
245 }
246 return;
247 }
248 m_fireCountWithoutObservers = 0;
249}
