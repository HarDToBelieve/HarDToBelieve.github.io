38
39namespace IPC {
40class Connection;
41}
42
43namespace WebKit {
44
45class DisplayLink {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 explicit DisplayLink(WebCore::PlatformDisplayID);
49 ~DisplayLink();
50
51 void addObserver(IPC::Connection&, DisplayLinkObserverID, WebCore::FramesPerSecond);
52 void removeObserver(IPC::Connection&, DisplayLinkObserverID);
53 void removeObservers(IPC::Connection&);
54
55 void incrementFullSpeedRequestClientCount(IPC::Connection&);
56 void decrementFullSpeedRequestClientCount(IPC::Connection&);
57
58 void setPreferredFramesPerSecond(IPC::Connection&, DisplayLinkObserverID, WebCore::FramesPerSecond);
59
60 WebCore::PlatformDisplayID displayID() const { return m_displayID; }
61
62 WebCore::FramesPerSecond nominalFramesPerSecond() const { return m_displayNominalFramesPerSecond; }
63
64 // When responsiveness is critical, we send the IPC to a background queue. Otherwise, we send it to the
65 // main thread to avoid unnecessary thread hopping and save power.
66 static void setShouldSendIPCOnBackgroundQueue(bool value) { shouldSendIPCOnBackgroundQueue = value; }
67
68private:
69 static CVReturn displayLinkCallback(CVDisplayLinkRef, const CVTimeStamp*, const CVTimeStamp*, CVOptionFlags, CVOptionFlags*, void* data);
70 void notifyObserversDisplayWasRefreshed();
71
72 void removeInfoForConnectionIfPossible(IPC::Connection&) WTF_REQUIRES_LOCK(m_observersLock);
73
74 static WebCore::FramesPerSecond nominalFramesPerSecondFromDisplayLink(CVDisplayLinkRef);
75
76 struct ObserverInfo {
77 DisplayLinkObserverID observerID;
78 WebCore::FramesPerSecond preferredFramesPerSecond;
79 };
80
81 struct ConnectionClientInfo {
82 unsigned fullSpeedUpdatesClientCount { 0 };
83 Vector<ObserverInfo> observers;
84 };
85
86 CVDisplayLinkRef m_displayLink { nullptr };
87 Lock m_observersLock;
88 HashMap<IPC::Connection::UniqueID, ConnectionClientInfo> m_observers WTF_GUARDED_BY_LOCK(m_observersLock);
89 WebCore::PlatformDisplayID m_displayID;
90 WebCore::FramesPerSecond m_displayNominalFramesPerSecond { 0 };
91 WebCore::DisplayUpdate m_currentUpdate;
92 unsigned m_fireCountWithoutObservers { 0 };
93 static bool shouldSendIPCOnBackgroundQueue;
94};
95
96} // namespace WebKit
97
98#endif // HAVE(CVDISPLAYLINK)
