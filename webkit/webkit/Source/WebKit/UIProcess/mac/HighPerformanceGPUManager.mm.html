42HighPerformanceGPUManager& HighPerformanceGPUManager::singleton()

45 return sharedManager;
46}

48HighPerformanceGPUManager::HighPerformanceGPUManager()
49 : m_updateStateTimer(*this, &HighPerformanceGPUManager::updateState)
50{
51}
52
53HighPerformanceGPUManager::~HighPerformanceGPUManager() = default;
54
55void HighPerformanceGPUManager::addProcessRequiringHighPerformance(WebProcessProxy& process)
56{
57 if (!WebCore::hasLowAndHighPowerGPUs())
58 return;
59
60 if (m_processesRequiringHighPerformance.add(process)) {
61 LOG(WebGL, "HighPerformanceGPUManager::addProcessRequiringHighPerformance() - adding process %p", &process);
62 updateState();
63 return;
64 }
65
66 LOG(WebGL, "HighPerformanceGPUManager::addProcessRequiringHighPerformance() - process %p was already requesting high performance", &process);
67}
68
69void HighPerformanceGPUManager::removeProcessRequiringHighPerformance(WebProcessProxy& process)
70{
71 if (!WebCore::hasLowAndHighPowerGPUs())
72 return;
73
74 if (m_processesRequiringHighPerformance.remove(process)) {
75 LOG(WebGL, "HighPerformanceGPUManager::removeProcessRequiringHighPerformance() - removing process %p", &process);
76 static const Seconds timeToKeepHighPerformanceGPUAlive { 10_s };
77 m_updateStateTimer.startOneShot(timeToKeepHighPerformanceGPUAlive);
78 return;
79 }
80
81 LOG(WebGL, "HighPerformanceGPUManager::removeProcessRequiringHighPerformance() - process %p was not requesting high performance", &process);
82}
83
84void HighPerformanceGPUManager::updateState()
85{
86 if (m_updateStateTimer.isActive())
87 m_updateStateTimer.stop();
88 if (m_processesRequiringHighPerformance.computeSize()) {
89 if (!m_pixelFormatObj) {
90 LOG(WebGL, "HighPerformanceGPUManager - turning on high-performance GPU.");
91
92 CGLPixelFormatAttribute attributes[] = { kCGLPFAAccelerated, kCGLPFAColorSize, static_cast<CGLPixelFormatAttribute>(32), static_cast<CGLPixelFormatAttribute>(0) };
93 GLint numPixelFormats = 0;
94 CGLChoosePixelFormat(attributes, &m_pixelFormatObj, &numPixelFormats);
95
96 LOG(WebGL, "HighPerformanceGPUManager - CGLPixelFormatObj is %p", m_pixelFormatObj);
97 }
98 } else if (m_pixelFormatObj) {
99 LOG(WebGL, "HighPerformanceGPUManager - turning off high-performance GPU.");
100 CGLReleasePixelFormat(m_pixelFormatObj);
101 m_pixelFormatObj = nullptr;
102 }
103}
