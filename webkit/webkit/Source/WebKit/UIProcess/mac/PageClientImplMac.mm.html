191bool PageClientImpl::isViewVisible()
192{
193 NSView *activeView = this->activeView();
194 NSWindow *activeViewWindow = activeWindow();
195
196 auto windowIsOccluded = [&]()->bool {
197 return m_impl && m_impl->windowOcclusionDetectionEnabled() && (activeViewWindow.occlusionState & NSWindowOcclusionStateVisible) != NSWindowOcclusionStateVisible;
198 };
199
200 LOG_WITH_STREAM(ActivityState, stream << "PageClientImpl " << this << " isViewVisible(): activeViewWindow " << activeViewWindow
201 << " (window visible " << activeViewWindow.isVisible << ", view hidden " << activeView.isHiddenOrHasHiddenAncestor << ", window occluded " << windowIsOccluded() << ")");
202
203 if (!activeViewWindow)
204 return false;
205
206 if (!activeViewWindow.isVisible)
207 return false;
208
209 if (activeView.isHiddenOrHasHiddenAncestor)
210 return false;
211
212 if (windowIsOccluded())
213 return false;
214
215 return true;
216}

305void PageClientImpl::setCursor(const WebCore::Cursor& cursor)
306{
307 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
308 // FIXME: Would be nice to share this code with WebKit1's WebChromeClient.
309
310 // The Web process may have asked to change the cursor when the view was in an active window, but
311 // if it is no longer in a window or the window is not active, then the cursor should not change.
312 if (!isViewWindowActive())
313 return;
314
315 if ([NSApp _cursorRectCursor])
316 return;
317
318 if (!m_view)
319 return;
320
321 NSWindow *window = [m_view window];
322 if (!window)
323 return;
324
325 if ([window windowNumber] != [NSWindow windowNumberAtPoint:[NSEvent mouseLocation] belowWindowWithWindowNumber:0])
326 return;
327
328 NSCursor *platformCursor = cursor.platformCursor();
329 if ([NSCursor currentCursor] == platformCursor)
330 return;
331
332 [platformCursor set];
333
334 if (cursor.type() == WebCore::Cursor::Type::None) {
335 if ([NSCursor respondsToSelector:@selector(hideUntilChanged)])
336 [NSCursor hideUntilChanged];
337 }
338}

703static inline NSCorrectionResponse toCorrectionResponse(AutocorrectionResponse response)
704{
705 switch (response) {
706 case WebCore::AutocorrectionResponse::Reverted:

708 case WebCore::AutocorrectionResponse::Edited:

710 case WebCore::AutocorrectionResponse::Accepted:
711 return NSCorrectionResponseAccepted;
712 }
713
714 ASSERT_NOT_REACHED();
715 return NSCorrectionResponseAccepted;
716}

723void PageClientImpl::recommendedScrollbarStyleDidChange(ScrollbarStyle newStyle)
724{
725 // Now re-create a tracking area with the appropriate options given the new scrollbar style
726 NSTrackingAreaOptions options = NSTrackingMouseMoved | NSTrackingMouseEnteredAndExited | NSTrackingInVisibleRect | NSTrackingCursorUpdate;
727 if (newStyle == ScrollbarStyle::AlwaysVisible)
728 options |= NSTrackingActiveAlways;
729 else
