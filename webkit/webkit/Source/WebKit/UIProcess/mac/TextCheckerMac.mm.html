90static TextCheckerState& mutableState()
91{
92 static NeverDestroyed<TextCheckerState> state = makeNeverDestroyed([] {
93 TextCheckerState initialState;
94 initialState.isContinuousSpellCheckingEnabled = [[NSUserDefaults standardUserDefaults] boolForKey:WebContinuousSpellCheckingEnabled] && TextChecker::isContinuousSpellCheckingAllowed();
95 initialState.isGrammarCheckingEnabled = [[NSUserDefaults standardUserDefaults] boolForKey:WebGrammarCheckingEnabled];
96 initialState.isAutomaticTextReplacementEnabled = shouldAutomaticTextReplacementBeEnabled();
97 initialState.isAutomaticSpellingCorrectionEnabled = shouldAutomaticSpellingCorrectionBeEnabled();
98 initialState.isAutomaticQuoteSubstitutionEnabled = shouldAutomaticQuoteSubstitutionBeEnabled();
99 initialState.isAutomaticDashSubstitutionEnabled = shouldAutomaticDashSubstitutionBeEnabled();
100 initialState.isAutomaticLinkDetectionEnabled = [[NSUserDefaults standardUserDefaults] boolForKey:WebAutomaticLinkDetectionEnabled];
101 return initialState;
102 }());
103 return state;
104}

133bool TextChecker::isContinuousSpellCheckingAllowed()
134{
135 static bool allowContinuousSpellChecking = true;
136 static bool readAllowContinuousSpellCheckingDefault = false;
137
138 if (!readAllowContinuousSpellCheckingDefault) {
139 if ([[NSUserDefaults standardUserDefaults] objectForKey:@"NSAllowContinuousSpellChecking"])
140 allowContinuousSpellChecking = [[NSUserDefaults standardUserDefaults] boolForKey:@"NSAllowContinuousSpellChecking"];
141
142 readAllowContinuousSpellCheckingDefault = true;
143 }
144
145 return allowContinuousSpellChecking;
146}

148bool TextChecker::setContinuousSpellCheckingEnabled(bool isContinuousSpellCheckingEnabled)
149{
150 if (state().isContinuousSpellCheckingEnabled == isContinuousSpellCheckingEnabled)
151 return false;
152
153 mutableState().isContinuousSpellCheckingEnabled = isContinuousSpellCheckingEnabled;
154 if (!testingModeEnabled)
155 [[NSUserDefaults standardUserDefaults] setBool:isContinuousSpellCheckingEnabled forKey:WebContinuousSpellCheckingEnabled];
156
157 // FIXME: preflight the spell checker.
158 return true;
159}

322Vector<TextCheckingResult> TextChecker::checkTextOfParagraph(SpellDocumentTag spellDocumentTag, StringView text, int32_t insertionPoint, OptionSet<TextCheckingType> checkingTypes, bool initialCapitalizationEnabled)
323{
324 Vector<TextCheckingResult> results;
325
326 RetainPtr<NSString> textString = text.createNSStringWithoutCopying();
327 NSDictionary *options = @{
328 NSTextCheckingInsertionPointKey : @(insertionPoint),
329 NSTextCheckingSuppressInitialCapitalizationKey : @(!initialCapitalizationEnabled)
330 };
331 NSArray *incomingResults = [[NSSpellChecker sharedSpellChecker] checkString:textString.get()
332 range:NSMakeRange(0, text.length())
333 types:nsTextCheckingTypes(checkingTypes) | NSTextCheckingTypeOrthography
334 options:options
335 inSpellDocumentWithTag:spellDocumentTag
336 orthography:NULL
337 wordCount:NULL];
338 for (NSTextCheckingResult *incomingResult in incomingResults) {
339 NSTextCheckingType resultType = [incomingResult resultType];
340 ASSERT(incomingResult.range.location != NSNotFound);
341 ASSERT(incomingResult.range.length > 0);
342 auto resultRange = incomingResult.range;
343 if (resultType == NSTextCheckingTypeSpelling && checkingTypes.contains(TextCheckingType::Spelling)) {
344 TextCheckingResult result;
345 result.type = TextCheckingType::Spelling;
346 result.range = resultRange;
347 results.append(result);
348 } else if (resultType == NSTextCheckingTypeGrammar && checkingTypes.contains(TextCheckingType::Grammar)) {
349 TextCheckingResult result;
350 NSArray *details = [incomingResult grammarDetails];
351 result.type = TextCheckingType::Grammar;
352 result.range = resultRange;
353 for (NSDictionary *incomingDetail in details) {
354 ASSERT(incomingDetail);
355 GrammarDetail detail;
356 NSValue *detailRangeAsNSValue = [incomingDetail objectForKey:NSGrammarRange];
357 ASSERT(detailRangeAsNSValue);
358 NSRange detailNSRange = [detailRangeAsNSValue rangeValue];
359 ASSERT(detailNSRange.location != NSNotFound);
360 ASSERT(detailNSRange.length > 0);
361 detail.range = detailNSRange;
362 detail.userDescription = [incomingDetail objectForKey:NSGrammarUserDescription];
363 NSArray *guesses = [incomingDetail objectForKey:NSGrammarCorrections];
364 for (NSString *guess in guesses)
365 detail.guesses.append(String(guess));
366 result.details.append(detail);
367 }
368 results.append(result);
369 } else if (resultType == NSTextCheckingTypeLink && checkingTypes.contains(TextCheckingType::Link)) {
370 TextCheckingResult result;
371 result.type = TextCheckingType::Link;
372 result.range = resultRange;
373 result.replacement = [[incomingResult URL] absoluteString];
374 results.append(result);
375 } else if (resultType == NSTextCheckingTypeQuote && checkingTypes.contains(TextCheckingType::Quote)) {
376 TextCheckingResult result;
377 result.type = TextCheckingType::Quote;
378 result.range = resultRange;
379 result.replacement = [incomingResult replacementString];
380 results.append(result);
381 } else if (resultType == NSTextCheckingTypeDash && checkingTypes.contains(TextCheckingType::Dash)) {
382 TextCheckingResult result;
383 result.type = TextCheckingType::Dash;
384 result.range = resultRange;
385 result.replacement = [incomingResult replacementString];
386 results.append(result);
387 } else if (resultType == NSTextCheckingTypeReplacement && checkingTypes.contains(TextCheckingType::Replacement)) {
388 TextCheckingResult result;
389 result.type = TextCheckingType::Replacement;
390 result.range = resultRange;
391 result.replacement = [incomingResult replacementString];
392 results.append(result);
393 } else if (resultType == NSTextCheckingTypeCorrection && checkingTypes.contains(TextCheckingType::Correction)) {
394 TextCheckingResult result;
395 result.type = TextCheckingType::Correction;
396 result.range = resultRange;
397 result.replacement = [incomingResult replacementString];
398 results.append(result);
399 }
400 }
401
402 return results;
403}

438void TextChecker::updateSpellingUIWithGrammarString(SpellDocumentTag, const String& badGrammarPhrase, const GrammarDetail& grammarDetail)
439{
440 CheckedUint64 endOfRangeChecked = grammarDetail.range.location;
441 endOfRangeChecked += grammarDetail.range.length;
442
443 if (endOfRangeChecked.hasOverflowed() || endOfRangeChecked >= badGrammarPhrase.length())
444 return;
445
446 NSDictionary *detail = @{
447 NSGrammarRange : [NSValue valueWithRange:grammarDetail.range],
448 NSGrammarUserDescription : grammarDetail.userDescription,
449 NSGrammarCorrections : createNSArray(grammarDetail.guesses).get(),
450 };
451 [[NSSpellChecker sharedSpellChecker] updateSpellingPanelWithGrammarString:badGrammarPhrase detail:detail];
452}

454void TextChecker::getGuessesForWord(SpellDocumentTag spellDocumentTag, const String& word, const String& context, int32_t insertionPoint, Vector<String>& guesses, bool initialCapitalizationEnabled)
455{
456 NSString* language = nil;
457 NSOrthography* orthography = nil;
458 NSSpellChecker *checker = [NSSpellChecker sharedSpellChecker];
459 NSDictionary *options = @{
460 NSTextCheckingInsertionPointKey : @(insertionPoint),
461 NSTextCheckingSuppressInitialCapitalizationKey : @(!initialCapitalizationEnabled)
462 };
463 if (context.length()) {
464 [checker checkString:context range:NSMakeRange(0, context.length()) types:NSTextCheckingTypeOrthography options:options inSpellDocumentWithTag:spellDocumentTag orthography:&orthography wordCount:0];
465 language = [checker languageForWordRange:NSMakeRange(0, context.length()) inString:context orthography:orthography];
466 }
467 guesses = makeVector<String>([checker guessesForWordRange:NSMakeRange(0, word.length()) inString:word language:language inSpellDocumentWithTag:spellDocumentTag]);
468}
