61void TiledCoreAnimationDrawingAreaProxy::sizeDidChange()
62{
63 if (!m_webPageProxy.hasRunningProcess())
64 return;
65
66 // We only want one UpdateGeometry message in flight at once, so if we've already sent one but
67 // haven't yet received the reply we'll just return early here.
68 if (m_isWaitingForDidUpdateGeometry)
69 return;
70
71 sendUpdateGeometry();
72}

120void TiledCoreAnimationDrawingAreaProxy::didUpdateGeometry()
121{
122 ASSERT(m_isWaitingForDidUpdateGeometry);
123
124 m_isWaitingForDidUpdateGeometry = false;
125
126 IntSize minimumSizeForAutoLayout = m_webPageProxy.minimumSizeForAutoLayout();
127 IntSize sizeToContentAutoSizeMaximumSize = m_webPageProxy.sizeToContentAutoSizeMaximumSize();
128
129 // If the WKView was resized while we were waiting for a DidUpdateGeometry reply from the web process,
130 // we need to resend the new size here.
131 if (m_lastSentSize != m_size || m_lastSentMinimumSizeForAutoLayout != minimumSizeForAutoLayout || m_lastSentSizeToContentAutoSizeMaximumSize != sizeToContentAutoSizeMaximumSize)
132 sendUpdateGeometry();
133}

141void TiledCoreAnimationDrawingAreaProxy::willSendUpdateGeometry()
142{
143 m_lastSentMinimumSizeForAutoLayout = m_webPageProxy.minimumSizeForAutoLayout();
144 m_lastSentSizeToContentAutoSizeMaximumSize = m_webPageProxy.sizeToContentAutoSizeMaximumSize();
145 m_lastSentSize = m_size;
146 m_isWaitingForDidUpdateGeometry = true;
147}

149MachSendRight TiledCoreAnimationDrawingAreaProxy::createFence()
150{
151 if (!m_webPageProxy.hasRunningProcess())
152 return MachSendRight();
153
154 RetainPtr<CAContext> rootLayerContext = [m_webPageProxy.acceleratedCompositingRootLayer() context];
155 if (!rootLayerContext)
156 return MachSendRight();
157
158 // Don't fence if we don't have a connection, because the message
159 // will likely get dropped on the floor (if the Web process is terminated)
160 // or queued up until process launch completes, and there's nothing useful
161 // to synchronize in these cases.
162 if (!process().connection())
163 return MachSendRight();
164
165 // Don't fence if we have incoming synchronous messages, because we may not
166 // be able to reply to the message until the fence times out.
167 if (process().connection()->hasIncomingSyncMessage())
168 return MachSendRight();
169
170 MachSendRight fencePort = MachSendRight::adopt([rootLayerContext createFencePort]);
171
172 // Invalidate the fence if a synchronous message arrives while it's installed,
173 // because we won't be able to reply during the fence-wait.
174 uint64_t callbackID = process().connection()->installIncomingSyncMessageCallback([rootLayerContext] {
175 [rootLayerContext invalidateFences];
176 });
177 RefPtr<WebPageProxy> retainedPage = &m_webPageProxy;
178 [CATransaction addCommitHandler:[callbackID, retainedPage] {
179 if (!retainedPage->hasRunningProcess())
180 return;
181 if (Connection* connection = retainedPage->process().connection())
182 connection->uninstallIncomingSyncMessageCallback(callbackID);
183 } forPhase:kCATransactionPhasePostCommit];
184
185 return fencePort;
186}

188void TiledCoreAnimationDrawingAreaProxy::sendUpdateGeometry()
189{
190 ASSERT(!m_isWaitingForDidUpdateGeometry);
191
192 willSendUpdateGeometry();
193 send(Messages::DrawingArea::UpdateGeometry(m_size, true /* flushSynchronously */, createFence()));
194}

206void TiledCoreAnimationDrawingAreaProxy::dispatchAfterEnsuringDrawing(WTF::Function<void (CallbackBase::Error)>&& callback)
207{
208 if (!m_webPageProxy.hasRunningProcess()) {
209 callback(CallbackBase::Error::OwnerWasInvalidated);
210 return;
211 }
212
213 send(Messages::DrawingArea::AddTransactionCallbackID(m_callbacks.put(WTFMove(callback), nullptr)));
214}
