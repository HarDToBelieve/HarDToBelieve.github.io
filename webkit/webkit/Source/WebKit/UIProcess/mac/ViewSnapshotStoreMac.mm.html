37namespace WebKit {
38
39Ref<ViewSnapshot> ViewSnapshot::create(std::unique_ptr<WebCore::IOSurface> surface)
40{
41 return adoptRef(*new ViewSnapshot(WTFMove(surface)));
42}
43
44ViewSnapshot::ViewSnapshot(std::unique_ptr<WebCore::IOSurface> surface)
45 : m_surface(WTFMove(surface))
46{
47 if (hasImage())
48 ViewSnapshotStore::singleton().didAddImageToSnapshot(*this);
49}
50
51void ViewSnapshot::setSurface(std::unique_ptr<WebCore::IOSurface> surface)
52{
53 ASSERT(!m_surface);
54 if (!surface) {
55 clearImage();
56 return;
57 }
58
59 m_surface = WTFMove(surface);
60 ViewSnapshotStore::singleton().didAddImageToSnapshot(*this);
61}
62
63bool ViewSnapshot::hasImage() const
64{
65 return !!m_surface;
66}
67
68void ViewSnapshot::clearImage()
69{
70 if (!hasImage())
71 return;
72
73 ViewSnapshotStore::singleton().willRemoveImageFromSnapshot(*this);
74
75 m_surface = nullptr;
76}
77
78WebCore::VolatilityState ViewSnapshot::setVolatile(bool becomeVolatile)
79{
80 if (ViewSnapshotStore::singleton().disableSnapshotVolatilityForTesting())
81 return WebCore::VolatilityState::Valid;
82
83 if (!m_surface)
84 return WebCore::VolatilityState::Empty;
85
86 return m_surface->setVolatile(becomeVolatile);
87}
88
89id ViewSnapshot::asLayerContents()
90{
91 if (!m_surface)
92 return nullptr;
93
94 if (setVolatile(false) != WebCore::VolatilityState::Valid) {
95 clearImage();
96 return nullptr;
97 }
98
99 return m_surface->asLayerContents();
100}
101
102RetainPtr<CGImageRef> ViewSnapshot::asImageForTesting()
103{
104 if (!m_surface)
105 return nullptr;
106
107 ASSERT(ViewSnapshotStore::singleton().disableSnapshotVolatilityForTesting());
108 return m_surface->createImage();
109}
