59- (instancetype)initWithPage:(NakedRef<WebKit::WebPageProxy>)page view:(NSView *)view viewImpl:(NakedRef<WebKit::WebViewImpl>)viewImpl recognizer:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
60{
61 self = [super init];
62
63 if (!self)
64 return nil;
65
66 _page = page.ptr();
67 _view = view;
68 _viewImpl = viewImpl.ptr();
69 _type = kWKImmediateActionNone;
70 _immediateActionRecognizer = immediateActionRecognizer;
71 _hasActiveImmediateAction = NO;
72
73 return self;
74}

76- (void)willDestroyView:(NSView *)view
77{
78 _page = nullptr;
79 _view = nil;
80 _viewImpl = nullptr;
81 _hitTestResultData = WebKit::WebHitTestResultData();
82 _contentPreventsDefault = NO;
83
84 id animationController = [_immediateActionRecognizer animationController];
85 if (PAL::isQuickLookUIFrameworkAvailable() && [animationController isKindOfClass:PAL::getQLPreviewMenuItemClass()]) {
86 QLPreviewMenuItem *menuItem = (QLPreviewMenuItem *)animationController;
87 menuItem.delegate = nil;
88 }
89
90 _immediateActionRecognizer = nil;
91 _currentActionContext = nil;
92 _hasActiveImmediateAction = NO;
93}

95- (void)_cancelImmediateAction
96{
97 // Reset the recognizer by turning it off and on again.
98 [_immediateActionRecognizer setEnabled:NO];
99 [_immediateActionRecognizer setEnabled:YES];
100
101 [self _clearImmediateActionState];
102}

104- (void)_cancelImmediateActionIfNeeded
105{
106 if (![_immediateActionRecognizer animationController])
107 [self _cancelImmediateAction];
108}

110- (void)_clearImmediateActionState
111{
112 if (_page)
113 _page->clearTextIndicator();
114
115 if (_currentActionContext && _hasActivatedActionContext) {
116 _hasActivatedActionContext = NO;
117 if (PAL::isDataDetectorsFrameworkAvailable())
118 [PAL::getDDActionsManagerClass() didUseActions];
119 }
120
121 _state = WebKit::ImmediateActionState::None;
122 _hitTestResultData = WebKit::WebHitTestResultData();
123 _contentPreventsDefault = NO;
124 _type = kWKImmediateActionNone;
125 _currentActionContext = nil;
126 _userData = nil;
127 _currentQLPreviewMenuItem = nil;
128 _hasActiveImmediateAction = NO;
129}

131- (void)didPerformImmediateActionHitTest:(const WebKit::WebHitTestResultData&)hitTestResult contentPreventsDefault:(BOOL)contentPreventsDefault userData:(API::Object*)userData
132{
133 // If we've already given up on this gesture (either because it was canceled or the
134 // willBeginAnimation timeout expired), we shouldn't build a new animationController for it.
135 if (_state != WebKit::ImmediateActionState::Pending)
136 return;
137
138 // FIXME: This needs to use the WebKit2 callback mechanism to avoid out-of-order replies.
139 _state = WebKit::ImmediateActionState::Ready;
140 _hitTestResultData = hitTestResult;
141 _contentPreventsDefault = contentPreventsDefault;
142 _userData = userData;
143
144 [self _updateImmediateActionItem];
145 [self _cancelImmediateActionIfNeeded];
146}

159#pragma mark NSImmediateActionGestureRecognizerDelegate

161- (void)immediateActionRecognizerWillPrepare:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
162{
163 if (immediateActionRecognizer != _immediateActionRecognizer)
164 return;
165
166 _viewImpl->prepareForImmediateActionAnimation();
167
168 _viewImpl->dismissContentRelativeChildWindowsWithAnimation(true);
169
170 _page->setMaintainsInactiveSelection(true);
171
172 _state = WebKit::ImmediateActionState::Pending;
173 immediateActionRecognizer.animationController = nil;
174
175 _page->performImmediateActionHitTestAtLocation([immediateActionRecognizer locationInView:immediateActionRecognizer.view]);
176}
177
178- (void)immediateActionRecognizerWillBeginAnimation:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
179{
180 if (immediateActionRecognizer != _immediateActionRecognizer)
181 return;
182
183 if (_state == WebKit::ImmediateActionState::None)
184 return;
185
186 _hasActiveImmediateAction = YES;
187
188 // FIXME: We need to be able to cancel this if the gesture recognizer is cancelled.
189 // FIXME: Connection can be null if the process is closed; we should clean up better in that case.
190 if (_state == WebKit::ImmediateActionState::Pending) {
191 if (auto* connection = _page->process().connection()) {
192 bool receivedReply = connection->waitForAndDispatchImmediately<Messages::WebPageProxy::DidPerformImmediateActionHitTest>(_page->webPageID(), Seconds::fromMilliseconds(500));
193 if (!receivedReply)
194 _state = WebKit::ImmediateActionState::TimedOut;
195 }
196 }
197
198 if (_state != WebKit::ImmediateActionState::Ready) {
199 [self _updateImmediateActionItem];
200 [self _cancelImmediateActionIfNeeded];
201 }
202
203 if (_currentActionContext) {
204 _hasActivatedActionContext = YES;
205 if (PAL::isDataDetectorsFrameworkAvailable()) {
206 if (![PAL::getDDActionsManagerClass() shouldUseActionsWithContext:_currentActionContext.get()])
207 [self _cancelImmediateAction];
208 }
209 }
210}

212- (void)immediateActionRecognizerDidUpdateAnimation:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
213{
214 if (immediateActionRecognizer != _immediateActionRecognizer)
215 return;
216
217 _page->immediateActionDidUpdate();
218 if (_contentPreventsDefault)
219 return;
220
221 _page->setTextIndicatorAnimationProgress([immediateActionRecognizer animationProgress]);
222}
223
224- (void)immediateActionRecognizerDidCancelAnimation:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
225{
226 if (immediateActionRecognizer != _immediateActionRecognizer)
227 return;
228
229 _page->immediateActionDidCancel();
230
231 _viewImpl->cancelImmediateActionAnimation();
232
233 _page->setTextIndicatorAnimationProgress(0);
234 [self _clearImmediateActionState];
235 _page->setMaintainsInactiveSelection(false);
236}
237
238- (void)immediateActionRecognizerDidCompleteAnimation:(NSImmediateActionGestureRecognizer *)immediateActionRecognizer
239{
240 if (immediateActionRecognizer != _immediateActionRecognizer)
241 return;
242
243 _page->immediateActionDidComplete();
244
245 _viewImpl->completeImmediateActionAnimation();
246
247 _page->setTextIndicatorAnimationProgress(1);
248}

250- (RefPtr<API::HitTestResult>)_webHitTestResult
251{
252 RefPtr<API::HitTestResult> hitTestResult;
253 if (_state == WebKit::ImmediateActionState::Ready)
254 hitTestResult = API::HitTestResult::create(_hitTestResultData);
255 else
256 hitTestResult = _page->lastMouseMoveHitTestResult();
257
258 return hitTestResult;
259}

263- (id <NSImmediateActionAnimationController>)_defaultAnimationController
264{
265 if (_contentPreventsDefault)
266 return adoptNS([[WKAnimationController alloc] init]).autorelease();
267
268 RefPtr<API::HitTestResult> hitTestResult = [self _webHitTestResult];
269
270 if (!hitTestResult)
271 return nil;
272
273 String absoluteLinkURL = hitTestResult->absoluteLinkURL();
274 if (!absoluteLinkURL.isEmpty()) {
275 if (WTF::protocolIs(absoluteLinkURL, "file")) {
276 _type = kWKImmediateActionNone;
277 return nil;
278 }
279
280 if (WTF::protocolIs(absoluteLinkURL, "mailto")) {
281 _type = kWKImmediateActionMailtoLink;
282 return [self _animationControllerForDataDetectedLink];
283 }
284
285 if (WTF::protocolIs(absoluteLinkURL, "tel")) {
286 _type = kWKImmediateActionTelLink;
287 return [self _animationControllerForDataDetectedLink];
288 }
289
290 if (WTF::protocolIsInHTTPFamily(absoluteLinkURL)) {
291 _type = kWKImmediateActionLinkPreview;
292
293 QLPreviewMenuItem *item = [NSMenuItem standardQuickLookMenuItem];
294 item.previewStyle = QLPreviewStylePopover;
295 item.delegate = self;
296 _currentQLPreviewMenuItem = item;
297
298 if (auto textIndicator = _hitTestResultData.linkTextIndicator.get())
299 _page->setTextIndicator(textIndicator->data());
300
301 return (id<NSImmediateActionAnimationController>)item;
302 }
303 }
304
305 if (hitTestResult->isTextNode() || hitTestResult->isOverTextInsideFormControlElement()) {
306 if (auto animationController = [self _animationControllerForDataDetectedText]) {
307 _type = kWKImmediateActionDataDetectedItem;
308 return animationController;
309 }
310
311 if (auto animationController = [self _animationControllerForText]) {
312 _type = kWKImmediateActionLookupText;
313 return animationController;
314 }
315 }
316
317 return nil;
318}

320- (void)_updateImmediateActionItem
321{
322 _type = kWKImmediateActionNone;
323
324 id <NSImmediateActionAnimationController> defaultAnimationController = [self _defaultAnimationController];
325
326 if (_contentPreventsDefault) {
327 [_immediateActionRecognizer.get() setAnimationController:defaultAnimationController];
328 return;
329 }
330
331 RefPtr<API::HitTestResult> hitTestResult = [self _webHitTestResult];
332 if (!hitTestResult) {
333 [self _cancelImmediateAction];
334 return;
335 }
336
337 id customClientAnimationController = _page->immediateActionAnimationControllerForHitTestResult(hitTestResult, _type, _userData);
338 if (customClientAnimationController == [NSNull null]) {
339 [self _cancelImmediateAction];
340 return;
341 }
342
343 if (customClientAnimationController && [customClientAnimationController conformsToProtocol:@protocol(NSImmediateActionAnimationController)])
344 [_immediateActionRecognizer setAnimationController:(id <NSImmediateActionAnimationController>)customClientAnimationController];
345 else
346 [_immediateActionRecognizer setAnimationController:defaultAnimationController];
347}

400#pragma mark Data Detectors actions

402- (id<NSImmediateActionAnimationController>)_animationControllerForDataDetectedText
403{
404 if (!PAL::isDataDetectorsFrameworkAvailable())
405 return nil;
406
407 DDActionContext *actionContext = _hitTestResultData.detectedDataActionContext.get();
408 if (!actionContext)
409 return nil;
410
411 actionContext.altMode = YES;
412 actionContext.immediate = YES;
413 if (![[PAL::getDDActionsManagerClass() sharedManager] hasActionsForResult:actionContext.mainResult actionContext:actionContext])
414 return nil;
415
416 RefPtr<WebKit::WebPageProxy> page = _page.get();
417 WebCore::PageOverlay::PageOverlayID overlayID = _hitTestResultData.detectedDataOriginatingPageOverlay;
418 _currentActionContext = [actionContext contextForView:_view altMode:YES interactionStartedHandler:^() {
419 page->send(Messages::WebPage::DataDetectorsDidPresentUI(overlayID));
420 } interactionChangedHandler:^() {
421 if (_hitTestResultData.detectedDataTextIndicator)
422 page->setTextIndicator(_hitTestResultData.detectedDataTextIndicator->data());
423 page->send(Messages::WebPage::DataDetectorsDidChangeUI(overlayID));
424 } interactionStoppedHandler:^() {
425 page->send(Messages::WebPage::DataDetectorsDidHideUI(overlayID));
426 [self _clearImmediateActionState];
427 }];
428
429 [_currentActionContext setHighlightFrame:[_view.window convertRectToScreen:[_view convertRect:_hitTestResultData.detectedDataBoundingBox toView:nil]]];
430
431 NSArray *menuItems = [[PAL::getDDActionsManagerClass() sharedManager] menuItemsForResult:[_currentActionContext mainResult] actionContext:_currentActionContext.get()];
432
433 if (menuItems.count != 1)
434 return nil;
435
436 return (id<NSImmediateActionAnimationController>)menuItems.lastObject;
437}

439- (id<NSImmediateActionAnimationController>)_animationControllerForDataDetectedLink
440{
441 if (!PAL::isDataDetectorsFrameworkAvailable())
442 return nil;
443
444 auto actionContext = adoptNS([PAL::allocDDActionContextInstance() init]);
445
446 if (!actionContext)
447 return nil;
448
449 [actionContext setAltMode:YES];
450 [actionContext setImmediate:YES];
451
452 RefPtr<WebKit::WebPageProxy> page = _page.get();
453 _currentActionContext = [actionContext contextForView:_view altMode:YES interactionStartedHandler:^() {
454 } interactionChangedHandler:^() {
455 if (_hitTestResultData.linkTextIndicator)
456 page->setTextIndicator(_hitTestResultData.linkTextIndicator->data());
457 } interactionStoppedHandler:^() {
458 [self _clearImmediateActionState];
459 }];
460
461 [_currentActionContext setHighlightFrame:[_view.window convertRectToScreen:[_view convertRect:_hitTestResultData.elementBoundingBox toView:nil]]];
462
463 RefPtr<API::HitTestResult> hitTestResult = [self _webHitTestResult];
464 if (!hitTestResult)
465 return nil;
466
467 NSArray *menuItems = [[PAL::getDDActionsManagerClass() sharedManager] menuItemsForTargetURL:hitTestResult->absoluteLinkURL() actionContext:_currentActionContext.get()];
468
469 if (menuItems.count != 1)
470 return nil;
471
472 return (id<NSImmediateActionAnimationController>)menuItems.lastObject;
473}

475#pragma mark Text action
476
477- (id<NSImmediateActionAnimationController>)_animationControllerForText
478{
479 if (_state != WebKit::ImmediateActionState::Ready)
480 return nil;
481
482 WebCore::DictionaryPopupInfo dictionaryPopupInfo = _hitTestResultData.dictionaryPopupInfo;
483 if (!dictionaryPopupInfo.attributedString)
484 return nil;
485
486 _viewImpl->prepareForDictionaryLookup();
487
488 return WebCore::DictionaryLookup::animationControllerForPopup(dictionaryPopupInfo, _view, [self](WebCore::TextIndicator& textIndicator) {
489 _viewImpl->setTextIndicator(textIndicator, WebCore::TextIndicatorLifetime::Permanent);
490 }, nullptr, [self]() {
491 _viewImpl->clearTextIndicatorWithAnimation(WebCore::TextIndicatorDismissalAnimation::None);
492 });
493}
