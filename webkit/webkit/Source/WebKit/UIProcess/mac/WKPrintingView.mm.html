96- (void)_suspendAutodisplay
97{
98 // A drawRect: call on WKView causes a switch to screen mode, which is slow due to relayout, and we want to avoid that.
99 // Disabling autodisplay will prevent random updates from causing this, but resizing the window will still work.
100 if (_autodisplayResumeTimer) {
101 [_autodisplayResumeTimer invalidate];
102 _autodisplayResumeTimer = nil;
103 } else
104 [self _setAutodisplay:NO];
105}

107- (void)_delayedResumeAutodisplayTimerFired
108{
109 ASSERT(RunLoop::isMain());
110
111 _autodisplayResumeTimer = nil;
112 [self _setAutodisplay:YES];
113
114 // Enabling autodisplay normally implicitly calls endPrinting() via -[WKView drawRect:], but not when content is in accelerated compositing mode.
115 if (_webFrame->page())
116 _webFrame->page()->endPrinting();
117}

119- (void)_delayedResumeAutodisplay
120{
121 // AppKit calls endDocument/beginDocument when print option change. We don't want to switch between print and screen mode just for that,
122 // and enabling autodisplay may result in switching into screen mode. So, autodisplay is only resumed on next run loop iteration.
123 if (!_autodisplayResumeTimer) {
124 _autodisplayResumeTimer = [NSTimer timerWithTimeInterval:0 target:self selector:@selector(_delayedResumeAutodisplayTimerFired) userInfo:nil repeats:NO];
125 // The timer must be scheduled on main thread, because printing thread may finish before it fires.
126 [[NSRunLoop mainRunLoop] addTimer:_autodisplayResumeTimer forMode:NSDefaultRunLoopMode];
127 }

130- (void)_adjustPrintingMarginsForHeaderAndFooter
131{
132 ASSERT(RunLoop::isMain()); // This function calls the client, which should only be done on main thread.
133
134 if (!_printOperation)
135 return;
136
137 NSPrintInfo *info = [_printOperation.get() printInfo];
138 NSMutableDictionary *infoDictionary = [info dictionary];
139
140 // We need to modify the top and bottom margins in the NSPrintInfo to account for the space needed by the
141 // header and footer. Because this method can be called more than once on the same NSPrintInfo (see 5038087),
142 // we stash away the unmodified top and bottom margins the first time this method is called, and we read from
143 // those stashed-away values on subsequent calls.
144 double originalTopMargin;
145 double originalBottomMargin;
146 NSNumber *originalTopMarginNumber = [infoDictionary objectForKey:WebKitOriginalTopPrintingMarginKey];
147 if (!originalTopMarginNumber) {
148 ASSERT(![infoDictionary objectForKey:WebKitOriginalBottomPrintingMarginKey]);
149 originalTopMargin = [info topMargin];
150 originalBottomMargin = [info bottomMargin];
151 [infoDictionary setObject:@(originalTopMargin) forKey:WebKitOriginalTopPrintingMarginKey];
152 [infoDictionary setObject:@(originalBottomMargin) forKey:WebKitOriginalBottomPrintingMarginKey];
153 } else {
154 ASSERT([originalTopMarginNumber isKindOfClass:[NSNumber class]]);
155 ASSERT([[infoDictionary objectForKey:WebKitOriginalBottomPrintingMarginKey] isKindOfClass:[NSNumber class]]);
156 originalTopMargin = [originalTopMarginNumber doubleValue];
157 originalBottomMargin = [[infoDictionary objectForKey:WebKitOriginalBottomPrintingMarginKey] doubleValue];
158 }
159
160 CGFloat scale = [info scalingFactor];
161 [info setTopMargin:originalTopMargin + _webFrame->page()->headerHeight(*_webFrame) * scale];
162 [info setBottomMargin:originalBottomMargin + _webFrame->page()->footerHeight(*_webFrame) * scale];
163}

196- (NSUInteger)_lastPrintedPageNumber
197{
198 ASSERT([self _hasPageRects]);
199
200 if (!_printOperation)
201 return 0;
202
203 // Need to directly access the dictionary because -[NSPrintOperation pageRange] verifies pagination, potentially causing recursion.
204 NSUInteger firstPage = [[[[_printOperation.get() printInfo] dictionary] objectForKey:NSPrintFirstPage] unsignedIntegerValue];
205 NSUInteger lastPage = [[[[_printOperation.get() printInfo] dictionary] objectForKey:NSPrintLastPage] unsignedIntegerValue];
206 if (lastPage - firstPage >= _printingPageRects.size())
207 return _printingPageRects.size();

225static void pageDidDrawToImage(const WebKit::ShareableBitmap::Handle& imageHandle, IPCCallbackContext* context)
226{
227 ASSERT(RunLoop::isMain());
228
229 WKPrintingView *view = context->view.get();
230
231 // If the user has already changed print setup, then this response is obsolete. And if this callback is not in response to the latest request,
232 // then the user has already moved to another page - we'll cache the response, but won't draw it.
233 HashMap<uint64_t, WebCore::IntRect>::iterator iter = view->_expectedPreviewCallbacks.find(context->callbackID);
234 if (iter != view->_expectedPreviewCallbacks.end()) {
235 ASSERT([view _isPrintingPreview]);
236
237 if (!imageHandle.isNull()) {
238 auto image = WebKit::ShareableBitmap::create(imageHandle, WebKit::SharedMemory::Protection::ReadOnly);
239
240 if (image)
241 view->_pagePreviews.add(iter->value, image);
242 }
243
244 view->_expectedPreviewCallbacks.remove(context->callbackID);
245 bool receivedResponseToLatestRequest = view->_latestExpectedPreviewCallback == context->callbackID;
246 if (receivedResponseToLatestRequest) {
247 view->_latestExpectedPreviewCallback = 0;
248 [view _updatePreview];
249 }
250 }
251}

253- (void)_preparePDFDataForPrintingOnSecondaryThread
254{
255 ASSERT(RunLoop::isMain());
256
257 if (!_webFrame->page() || !_printOperation) {
258 _printingCallbackCondition.notifyOne();
259 return;
260 }
261
262 Locker locker { _printingCallbackMutex };
263
264 ASSERT([self _hasPageRects]);
265 ASSERT(_printedPagesData.isEmpty());
266 ASSERT(!_printedPagesPDFDocument);
267 ASSERT(!_expectedPrintCallback);
268
269 NSUInteger firstPage = [self _firstPrintedPageNumber];
270 NSUInteger lastPage = [self _lastPrintedPageNumber];
271
272 ASSERT(firstPage > 0);
273 ASSERT(firstPage <= lastPage);
274 LOG(Printing, "WKPrintingView requesting PDF data for pages %u...%u", firstPage, lastPage);
275
276 WebKit::PrintInfo printInfo([_printOperation.get() printInfo]);
277 // Return to printing mode if we're already back to screen (e.g. due to window resizing).
278 _webFrame->page()->beginPrinting(_webFrame.get(), printInfo);
279
280 IPCCallbackContext* context = new IPCCallbackContext;
281 auto callback = [context](API::Data* data) {
282 ASSERT(RunLoop::isMain());
283
284 std::unique_ptr<IPCCallbackContext> contextDeleter(context);
285 WKPrintingView *view = context->view.get();
286
287 if (context->callbackID == view->_expectedPrintCallback) {
288 ASSERT(![view _isPrintingPreview]);
289 ASSERT(view->_printedPagesData.isEmpty());
290 ASSERT(!view->_printedPagesPDFDocument);
291 if (data)
292 view->_printedPagesData.append(data->bytes(), data->size());
293 view->_expectedPrintCallback = 0;
294 view->_printingCallbackCondition.notifyOne();
295 }
296 };
297 _expectedPrintCallback = _webFrame->page()->drawPagesToPDF(_webFrame.get(), printInfo, firstPage - 1, lastPage - firstPage + 1, WTFMove(callback));
298 context->view = self;
299 context->callbackID = _expectedPrintCallback;
300}

302static void pageDidComputePageRects(const Vector<WebCore::IntRect>& pageRects, double totalScaleFactorForPrinting, const WebCore::FloatBoxExtent& computedPageMargin, IPCCallbackContext* context)
303{
304 ASSERT(RunLoop::isMain());
305
306 WKPrintingView *view = context->view.get();
307 if (!view->_printOperation)
308 return;
309
310 // If the user has already changed print setup, then this response is obsolete.
311 if (context->callbackID == view->_expectedComputedPagesCallback) {
312 ASSERT(RunLoop::isMain());
313 ASSERT(view->_expectedPreviewCallbacks.isEmpty());
314 ASSERT(!view->_latestExpectedPreviewCallback);
315 ASSERT(!view->_expectedPrintCallback);
316 ASSERT(view->_pagePreviews.isEmpty());
317 view->_expectedComputedPagesCallback = 0;
318
319 view->_printingPageRects = pageRects;
320 view->_totalScaleFactorForPrinting = totalScaleFactorForPrinting;
321
322 // Sanitize a response coming from the Web process.
323 if (view->_printingPageRects.isEmpty())
324 view->_printingPageRects.append(WebCore::IntRect(0, 0, 1, 1));
325 if (view->_totalScaleFactorForPrinting <= 0)
326 view->_totalScaleFactorForPrinting = 1;
327
328 const WebCore::IntRect& lastPrintingPageRect = view->_printingPageRects[view->_printingPageRects.size() - 1];
329 NSRect newFrameSize = NSMakeRect(0, 0,
330 ceil(lastPrintingPageRect.maxX() * view->_totalScaleFactorForPrinting),
331 ceil(lastPrintingPageRect.maxY() * view->_totalScaleFactorForPrinting));
332 LOG(Printing, "WKPrintingView setting frame size to x:%g y:%g width:%g height:%g", newFrameSize.origin.x, newFrameSize.origin.y, newFrameSize.size.width, newFrameSize.size.height);
333 [view setFrame:newFrameSize];
334 // Set @page margin.
335 auto *printInfo = [view->_printOperation.get() printInfo];
336 [printInfo setTopMargin:computedPageMargin.top()];
337 [printInfo setBottomMargin:computedPageMargin.bottom()];
338 [printInfo setLeftMargin:computedPageMargin.left()];
339 [printInfo setRightMargin:computedPageMargin.right()];
340
341 if ([view _isPrintingPreview]) {
342 // Show page count, and ask for an actual image to replace placeholder.
343 [view _updatePreview];
344 } else {
345 // When printing, request everything we'll need beforehand.
346 [view _preparePDFDataForPrintingOnSecondaryThread];
347 }
348 }
349}

351- (BOOL)_askPageToComputePageRects
352{
353 ASSERT(RunLoop::isMain());
354
355 if (!_webFrame->page() || !_printOperation)
356 return NO;
357
358 ASSERT(!_expectedComputedPagesCallback);
359
360 IPCCallbackContext* context = new IPCCallbackContext;
361 auto callback = [context](const Vector<WebCore::IntRect>& pageRects, double totalScaleFactorForPrinting, const WebCore::FloatBoxExtent& computedPageMargin) {
362 std::unique_ptr<IPCCallbackContext> contextDeleter(context);
363 pageDidComputePageRects(pageRects, totalScaleFactorForPrinting, computedPageMargin, context);
364 };
365 _expectedComputedPagesCallback = _webFrame->page()->computePagesForPrinting(_webFrame.get(), WebKit::PrintInfo([_printOperation.get() printInfo]), WTFMove(callback));
366 context->view = self;
367 context->callbackID = _expectedComputedPagesCallback;
368
369 return YES;
370}

372static void prepareDataForPrintingOnSecondaryThread(WKPrintingView *view)
373{
374 ASSERT(RunLoop::isMain());
375
376 Locker locker { view->_printingCallbackMutex };
377
378 // We may have received page rects while a message to call this function traveled from secondary thread to main one.
379 if ([view _hasPageRects]) {
380 [view _preparePDFDataForPrintingOnSecondaryThread];
381 return;
382 }
383
384 // A request for pages has already been made, just wait for it to finish.
385 if (view->_expectedComputedPagesCallback)
386 return;
387
388 [view _askPageToComputePageRects];
389}

391- (BOOL)knowsPageRange:(NSRangePointer)range
392{
393 LOG(Printing, "-[WKPrintingView %p knowsPageRange:], %s, %s", self, [self _hasPageRects] ? "print data is available" : "print data is not available yet", RunLoop::isMain() ? "on main thread" : "on secondary thread");
394
395 if (!_printOperation)
396 return NO;
397
398 ASSERT(_printOperation.get() == [NSPrintOperation currentOperation]);
399
400 // Assuming that once we switch to printing from a secondary thread, we don't go back.
401 ASSERT(!_isPrintingFromSecondaryThread || !RunLoop::isMain());
402 if (!RunLoop::isMain())
403 _isPrintingFromSecondaryThread = YES;
404
405 if (_webFrame->pageIsClosed()) {
406 *range = NSMakeRange(1, NSIntegerMax);
407 return YES;
408 }
409
410 [self _suspendAutodisplay];
411
412 [self performSelectorOnMainThread:@selector(_adjustPrintingMarginsForHeaderAndFooter) withObject:nil waitUntilDone:YES];
413
414 if ([self _hasPageRects])
415 *range = NSMakeRange(1, _printingPageRects.size());
416 else if (!RunLoop::isMain()) {
417 ASSERT(![self _isPrintingPreview]);
418 Locker lock { _printingCallbackMutex };
419
420 RunLoop::main().dispatch([self] {
421 prepareDataForPrintingOnSecondaryThread(self);
422 });
423
424 _printingCallbackCondition.wait(_printingCallbackMutex);
425 *range = NSMakeRange(1, _printingPageRects.size());
426 } else {
427 ASSERT([self _isPrintingPreview]);
428
429 // If a request for pages hasn't already been made, make it now.
430 if (!_expectedComputedPagesCallback)
431 [self _askPageToComputePageRects];
432
433 *range = NSMakeRange(1, NSIntegerMax);
434 }
435 return YES;
436}

438- (unsigned)_pageForRect:(NSRect)rect
439{
440 // Assuming that rect exactly matches one of the pages.
441 for (size_t i = 0; i < _printingPageRects.size(); ++i) {
442 WebCore::IntRect currentRect(_printingPageRects[i]);
443 currentRect.scale(_totalScaleFactorForPrinting);
444 if (rect.origin.y == currentRect.y() && rect.origin.x == currentRect.x())
445 return i + 1;
446 }
447 ASSERT_NOT_REACHED();
448 return 0; // Invalid page number.
449}

456- (void)_drawPDFDocument:(PDFDocument *)pdfDocument page:(unsigned)page atPoint:(NSPoint)point
457{
458 if (!pdfDocument) {
459 LOG_ERROR("Couldn't create a PDF document with data passed for preview");
460 return;
461 }
462
463 PDFPage *pdfPage;
464 @try {
465 pdfPage = [pdfDocument pageAtIndex:page];
466 } @catch (id exception) {
467 LOG_ERROR("Preview data doesn't have page %d: %@", page, exception);
468 return;
469 }
470
471 CGContextRef context = [[NSGraphicsContext currentContext] CGContext];
472
473 CGContextSaveGState(context);
474 CGContextTranslateCTM(context, point.x, point.y);
475 CGContextScaleCTM(context, _totalScaleFactorForPrinting, -_totalScaleFactorForPrinting);
476 CGContextTranslateCTM(context, 0, -[pdfPage boundsForBox:kPDFDisplayBoxMediaBox].size.height);
477 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
478 [pdfPage drawWithBox:kPDFDisplayBoxMediaBox];
479 ALLOW_DEPRECATED_DECLARATIONS_END
480
481 CGAffineTransform transform = CGContextGetCTM(context);
482
483 for (const auto& destination : _linkDestinationsPerPage[page]) {
484 CGPoint destinationPoint = CGPointApplyAffineTransform(NSPointToCGPoint([destination point]), transform);
485 CGPDFContextAddDestinationAtPoint(context, (__bridge CFStringRef)linkDestinationName(pdfDocument, destination.get()), destinationPoint);
486 }
487
488 for (PDFAnnotation *annotation in [pdfPage annotations]) {
489 if (![annotation isKindOfClass:WebKit::getPDFAnnotationLinkClass()])
490 continue;
491
492 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
493 PDFAnnotationLink *linkAnnotation = (PDFAnnotationLink *)annotation;
494 ALLOW_DEPRECATED_DECLARATIONS_END
495 NSURL *url = [linkAnnotation URL];
496 CGRect transformedRect = CGRectApplyAffineTransform(NSRectToCGRect([linkAnnotation bounds]), transform);
497
498 if (!url) {
499 PDFDestination *destination = [linkAnnotation destination];
500 if (!destination)
501 continue;
502 CGPDFContextSetDestinationForRect(context, (__bridge CFStringRef)linkDestinationName(pdfDocument, destination), transformedRect);
503 continue;
504 }
505
506 CGPDFContextSetURLForRect(context, (CFURLRef)url, transformedRect);
507 }
508
509 CGContextRestoreGState(context);
510}

512- (void)_drawPreview:(NSRect)nsRect
513{
514 ASSERT(RunLoop::isMain());
515
516 if (!_webFrame || !_webFrame->page())
517 return;
518
519 WebCore::IntRect scaledPrintingRect(nsRect);
520 scaledPrintingRect.scale(1 / _totalScaleFactorForPrinting);
521 WebCore::IntSize imageSize(nsRect.size);
522 imageSize.scale(_webFrame->page()->deviceScaleFactor());
523 HashMap<WebCore::IntRect, RefPtr<WebKit::ShareableBitmap>>::iterator pagePreviewIterator = _pagePreviews.find(scaledPrintingRect);
524 if (pagePreviewIterator == _pagePreviews.end()) {
525 // It's too early to ask for page preview if we don't even know page size and scale.
526 if ([self _hasPageRects]) {
527 if (uint64_t existingCallback = [self _expectedPreviewCallbackForRect:scaledPrintingRect]) {
528 // We've already asked for a preview of this page, and are waiting for response.
529 // There is no need to ask again.
530 _latestExpectedPreviewCallback = existingCallback;
531 } else {
532 if (!_printOperation)
533 return;
534
535 // Preview isn't available yet, request it asynchronously.
536 // Return to printing mode if we're already back to screen (e.g. due to window resizing).
537 _webFrame->page()->beginPrinting(_webFrame.get(), WebKit::PrintInfo([_printOperation.get() printInfo]));
538
539 IPCCallbackContext* context = new IPCCallbackContext;
540 auto callback = [context](const WebKit::ShareableBitmap::Handle& imageHandle) {
541 std::unique_ptr<IPCCallbackContext> contextDeleter(context);
542 pageDidDrawToImage(imageHandle, context);
543 };
544 _latestExpectedPreviewCallback = _webFrame->page()->drawRectToImage(_webFrame.get(), WebKit::PrintInfo([_printOperation.get() printInfo]), scaledPrintingRect, imageSize, WTFMove(callback));
545 _expectedPreviewCallbacks.add(_latestExpectedPreviewCallback, scaledPrintingRect);
546
547 context->view = self;
548 context->callbackID = _latestExpectedPreviewCallback;
549 return;
550 }
551 }
552
553 // FIXME: Draw a placeholder
554 return;
555 }
556
557 RefPtr<WebKit::ShareableBitmap> bitmap = pagePreviewIterator->value;
558
559 WebCore::GraphicsContextCG context([[NSGraphicsContext currentContext] CGContext]);
560 WebCore::GraphicsContextStateSaver stateSaver(context);
561
562 bitmap->paint(context, _webFrame->page()->deviceScaleFactor(), WebCore::IntPoint(nsRect.origin), bitmap->bounds());
563}

565- (void)drawRect:(NSRect)nsRect
566{
567 LOG(Printing, "WKPrintingView %p printing rect x:%g, y:%g, width:%g, height:%g%s", self, nsRect.origin.x, nsRect.origin.y, nsRect.size.width, nsRect.size.height, [self _isPrintingPreview] ? " for preview" : "");
568
569 if (!_printOperation)
570 return;
571
572 ASSERT(_printOperation.get() == [NSPrintOperation currentOperation]);
573
574 // Always use the light appearance when printing.
575 WebCore::LocalDefaultSystemAppearance localAppearance(false);
576
577 if ([self _isPrintingPreview]) {
578 [self _drawPreview:nsRect];
579 return;
580 }
581
582 ASSERT(!RunLoop::isMain());
583 ASSERT(!_printedPagesData.isEmpty()); // Prepared by knowsPageRange:
584
585 if (!_printedPagesPDFDocument) {
586 RetainPtr<NSData> pdfData = adoptNS([[NSData alloc] initWithBytes:_printedPagesData.data() length:_printedPagesData.size()]);
587 _printedPagesPDFDocument = adoptNS([WebKit::allocPDFDocumentInstance() initWithData:pdfData.get()]);
588
589 unsigned pageCount = [_printedPagesPDFDocument pageCount];
590 _linkDestinationsPerPage.clear();
591 _linkDestinationsPerPage.resize(pageCount);
592 for (unsigned i = 0; i < pageCount; i++) {
593 PDFPage *page = [_printedPagesPDFDocument pageAtIndex:i];
594 for (PDFAnnotation *annotation in page.annotations) {
595 if (![annotation isKindOfClass:WebKit::getPDFAnnotationLinkClass()])
596 continue;
597
598 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
599 PDFAnnotationLink *linkAnnotation = (PDFAnnotationLink *)annotation;
600 ALLOW_DEPRECATED_DECLARATIONS_END
601 if (linkAnnotation.URL)
602 continue;
603
604 PDFDestination *destination = linkAnnotation.destination;
605 if (!destination)
606 continue;
607
608 unsigned destinationPageIndex = [_printedPagesPDFDocument indexForPage:destination.page];
609 _linkDestinationsPerPage[destinationPageIndex].append(destination);
610 }
611 }
612 }
613
614 unsigned printedPageNumber = [self _pageForRect:nsRect] - [self _firstPrintedPageNumber];
615 [self _drawPDFDocument:_printedPagesPDFDocument.get() page:printedPageNumber atPoint:NSMakePoint(nsRect.origin.x, nsRect.origin.y)];
616}

638- (void)drawPageBorderWithSize:(NSSize)borderSize
639{
640 if (!_printOperation)
641 return;
642
643 ASSERT(NSEqualSizes(borderSize, [[_printOperation.get() printInfo] paperSize]));
644 ASSERT(_printOperation.get() == [NSPrintOperation currentOperation]);
645
646 if (!RunLoop::isMain()) {
647 // Don't call the client from a secondary thread.
648 NSInvocation *invocation = [NSInvocation invocationWithMethodSignature:[WKPrintingView instanceMethodSignatureForSelector:@selector(_drawPageBorderWithSizeOnMainThread:)]];
649 [invocation setSelector:@selector(_drawPageBorderWithSizeOnMainThread:)];
650 [invocation setArgument:&borderSize atIndex:2];
651 [invocation performSelectorOnMainThread:@selector(invokeWithTarget:) withObject:self waitUntilDone:YES];
652 return;
653 }
654
655 if (!_webFrame->page())
656 return;
657
658 // The header and footer rect height scales with the page, but the width is always
659 // all the way across the printed page (inset by printing margins).
660 NSPrintInfo *printInfo = [_printOperation.get() printInfo];
661 CGFloat scale = [printInfo scalingFactor];
662 NSSize paperSize = [printInfo paperSize];
663 CGFloat headerFooterLeft = [printInfo leftMargin] / scale;
664 CGFloat headerFooterWidth = (paperSize.width - ([printInfo leftMargin] + [printInfo rightMargin])) / scale;
665 NSRect footerRect = NSMakeRect(headerFooterLeft, [printInfo bottomMargin] / scale - _webFrame->page()->footerHeight(*_webFrame), headerFooterWidth, _webFrame->page()->footerHeight(*_webFrame));
666 NSRect headerRect = NSMakeRect(headerFooterLeft, (paperSize.height - [printInfo topMargin]) / scale, headerFooterWidth, _webFrame->page()->headerHeight(*_webFrame));
667
668 NSGraphicsContext *currentContext = [NSGraphicsContext currentContext];
669 [currentContext saveGraphicsState];
670 NSRectClip(headerRect);
671 _webFrame->page()->drawHeader(*_webFrame, headerRect);
672 [currentContext restoreGraphicsState];
673
674 [currentContext saveGraphicsState];
675 NSRectClip(footerRect);
676 _webFrame->page()->drawFooter(*_webFrame, footerRect);
677 [currentContext restoreGraphicsState];
678}

680- (NSRect)rectForPage:(NSInteger)page
681{
682 if (!_printOperation)
683 return NSZeroRect;
684
685 ASSERT(_printOperation.get() == [NSPrintOperation currentOperation]);
686 if (![self _hasPageRects]) {
687 LOG(Printing, "-[WKPrintingView %p rectForPage:%d] - data is not yet available", self, (int)page);
688 if (!_webFrame->page()) {
689 // We may have not told AppKit how many pages there are, so it will try to print until a null rect is returned.
690 return NSZeroRect;
691 }
692 // We must be still calculating the page range.
693 ASSERT(_expectedComputedPagesCallback);
694 return NSMakeRect(0, 0, 1, 1);
695 }
696
697 // If Web process crashes while computing page rects, we never tell AppKit how many pages there are.
698 // Returning a null rect prevents selecting non-existent pages in preview dialog.
699 if (static_cast<unsigned>(page) > _printingPageRects.size()) {
700 ASSERT(!_webFrame->page());
701 return NSZeroRect;
702 }
703
704 WebCore::IntRect rect = _printingPageRects[page - 1];
705 rect.scale(_totalScaleFactorForPrinting);
706 LOG(Printing, "-[WKPrintingView %p rectForPage:%d] -> x %d, y %d, width %d, height %d", self, (int)page, rect.x(), rect.y(), rect.width(), rect.height());
707 return rect;
708}
