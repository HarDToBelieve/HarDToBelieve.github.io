31#if PLATFORM(MAC) && ENABLE(REVEAL)
32
33#import "WebViewImpl.h"
34#import <wtf/RetainPtr.h>
35#import <wtf/WeakPtr.h>
36#import <pal/cocoa/RevealSoftLink.h>
37
38@interface WKRevealItemPresenter () <RVPresenterHighlightDelegate>
39@end
40
41@implementation WKRevealItemPresenter {
42 WeakPtr<WebKit::WebViewImpl> _impl;
43 RetainPtr<RVPresenter> _presenter;
44 RetainPtr<RVPresentingContext> _presentingContext;
45 RetainPtr<RVItem> _item;
46 CGRect _frameInView;
47 CGPoint _menuLocationInView;
48 BOOL _isHighlightingItem;
49}
50
51- (instancetype)initWithWebViewImpl:(const WebKit::WebViewImpl&)impl item:(RVItem *)item frame:(CGRect)frameInView menuLocation:(CGPoint)menuLocationInView
52{
53 if (!(self = [super init]))
54 return nil;
55
56 _impl = makeWeakPtr(impl);
57 _presenter = adoptNS([PAL::allocRVPresenterInstance() init]);
58 _presentingContext = adoptNS([PAL::allocRVPresentingContextInstance() initWithPointerLocationInView:menuLocationInView inView:impl.view() highlightDelegate:self]);
59 _item = item;
60 _frameInView = frameInView;
61 _menuLocationInView = menuLocationInView;
62 return self;
63}
64
65- (void)showContextMenu
66{
67 if (!_impl)
68 return;
69
70 auto view = _impl->view();
71 if (!view)
72 return;
73
74 auto menuItems = retainPtr([_presenter menuItemsForItem:_item.get() documentContext:nil presentingContext:_presentingContext.get() options:nil]);
75 if (![menuItems count])
76 return;
77
78 auto menu = adoptNS([[NSMenu alloc] initWithTitle:emptyString()]);
79 [menu setAutoenablesItems:NO];
80 [menu setItemArray:menuItems.get()];
81
82 auto clickLocationInWindow = [view convertPoint:_menuLocationInView toView:nil];
83 NSEvent *event = [NSEvent mouseEventWithType:NSEventTypeLeftMouseDown location:clickLocationInWindow modifierFlags:0 timestamp:0 windowNumber:view.window.windowNumber context:0 eventNumber:0 clickCount:1 pressure:1];
84 [NSMenu popUpContextMenu:menu.get() withEvent:event forView:view];
85
86 [self _callDidFinishPresentationIfNeeded];
87}
88
89- (void)_callDidFinishPresentationIfNeeded
90{
91 if (!_impl || _isHighlightingItem)
92 return;
93
94 _impl->didFinishPresentation(self);
95}
96
97#pragma mark - RVPresenterHighlightDelegate
98
99- (NSArray<NSValue *> *)revealContext:(RVPresentingContext *)context rectsForItem:(RVItem *)item
100{
101 return @[ [NSValue valueWithRect:_frameInView] ];
102}
103
104- (BOOL)revealContext:(RVPresentingContext *)context shouldUseDefaultHighlightForItem:(RVItem *)item
105{
106 return self.shouldUseDefaultHighlight;
107}
108
109- (void)revealContext:(RVPresentingContext *)context startHighlightingItem:(RVItem *)item
110{
111 _isHighlightingItem = YES;
112}
113
114- (void)revealContext:(RVPresentingContext *)context stopHighlightingItem:(RVItem *)item
115{
116 _isHighlightingItem = NO;
117
118 [self _callDidFinishPresentationIfNeeded];
119}
120
121@end
