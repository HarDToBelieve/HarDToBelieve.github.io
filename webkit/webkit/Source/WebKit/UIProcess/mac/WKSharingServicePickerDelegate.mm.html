73- (NSArray *)sharingServicePicker:(NSSharingServicePicker *)sharingServicePicker sharingServicesForItems:(NSArray *)items mask:(NSSharingServiceMask)mask proposedSharingServices:(NSArray *)proposedServices
74{
75 if (!_filterEditingServices)
76 return proposedServices;
77
78 NSMutableArray *services = [NSMutableArray arrayWithCapacity:proposedServices.count];
79
80 for (NSSharingService *service in proposedServices) {
81 if (service.type != NSSharingServiceTypeEditor)
82 [services addObject:service];
83 }
84
85 return services;
86}

98- (void)sharingService:(NSSharingService *)sharingService didShareItems:(NSArray *)items
99{
100 // We only care about what item was shared if we were interested in editor services
101 // (i.e., if we plan on replacing the selection with the returned item)
102 if (!_handleEditingReplacement)
103 return;
104
105 if (!items.count)
106 return;
107
108 Vector<String> types;
109 IPC::DataReference dataReference;
110
111 id item = [items objectAtIndex:0];
112
113 if ([item isKindOfClass:[NSAttributedString class]]) {
114 NSData *data = [item RTFDFromRange:NSMakeRange(0, [item length]) documentAttributes:@{ }];
115 dataReference = IPC::DataReference(static_cast<const uint8_t*>([data bytes]), [data length]);
116
117 types.append(NSPasteboardTypeRTFD);
118 types.append(WebCore::legacyRTFDPasteboardType());
119 } else if ([item isKindOfClass:[NSData class]]) {
120 NSData *data = (NSData *)item;
121 RetainPtr<CGImageSourceRef> source = adoptCF(CGImageSourceCreateWithData((CFDataRef)data, NULL));
122 RetainPtr<CGImageRef> image = adoptCF(CGImageSourceCreateImageAtIndex(source.get(), 0, NULL));
123
124 if (!image)
125 return;
126
127 dataReference = IPC::DataReference(static_cast<const uint8_t*>([data bytes]), [data length]);
128 types.append(NSPasteboardTypeTIFF);
129 } else {
130 LOG_ERROR("sharingService:didShareItems: - Unknown item type returned\n");
131 return;
132 }
133
134 // FIXME: We should adopt replaceSelectionWithAttributedString instead of bouncing through the (fake) pasteboard.
135 _menuProxy->page()->replaceSelectionWithPasteboardData(types, dataReference);
136}
