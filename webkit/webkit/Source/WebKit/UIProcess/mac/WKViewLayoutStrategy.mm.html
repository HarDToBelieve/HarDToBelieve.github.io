50+ (instancetype)layoutStrategyWithPage:(NakedRef<WebKit::WebPageProxy>)page view:(NSView *)view viewImpl:(NakedRef<WebKit::WebViewImpl>)webViewImpl mode:(WKLayoutMode)mode
51{
52 RetainPtr<WKViewLayoutStrategy> strategy;
53
54 switch (mode) {
55 case kWKLayoutModeFixedSize:
56 strategy = adoptNS([[WKViewFixedSizeLayoutStrategy alloc] initWithPage:page.copyRef() view:view viewImpl:webViewImpl.copyRef() mode:mode]);
57 break;
58 case kWKLayoutModeDynamicSizeComputedFromViewScale:
59 strategy = adoptNS([[WKViewDynamicSizeComputedFromViewScaleLayoutStrategy alloc] initWithPage:page.copyRef() view:view viewImpl:webViewImpl.copyRef() mode:mode]);
60 break;
61 case kWKLayoutModeDynamicSizeComputedFromMinimumDocumentSize:
62 strategy = adoptNS([[WKViewDynamicSizeComputedFromMinimumDocumentSizeLayoutStrategy alloc] initWithPage:page.copyRef() view:view viewImpl:webViewImpl.copyRef() mode:mode]);
63 break;
64 case kWKLayoutModeViewSize:
65 default:
66 strategy = adoptNS([[WKViewViewSizeLayoutStrategy alloc] initWithPage:page.copyRef() view:view viewImpl:webViewImpl.copyRef() mode:mode]);
67 break;
68 }
69
70 [strategy updateLayout];
71
72 return strategy.autorelease();
73}

75- (instancetype)initWithPage:(NakedRef<WebKit::WebPageProxy>)page view:(NSView *)view viewImpl:(NakedRef<WebKit::WebViewImpl>)webViewImpl mode:(WKLayoutMode)mode
76{
77 self = [super init];
78
79 if (!self)
80 return nil;

82 _page = page.ptr();
83 _webViewImpl = webViewImpl.ptr();
84 _view = view;
85 _layoutMode = mode;
86
87 return self;
88}

137- (void)didChangeFrameSize
138{
139 if ([self frameSizeUpdatesDisabled])

207- (void)updateLayout
208{
209 CGFloat inverseScale = 1 / _page->viewScaleFactor();
210 _webViewImpl->setFixedLayoutSize(CGSizeMake(_view.frame.size.width * inverseScale, _view.frame.size.height * inverseScale));
211}
