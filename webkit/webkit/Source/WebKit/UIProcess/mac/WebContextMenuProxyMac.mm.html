129- (void)forwardContextMenuAction:(id)sender
130{
131 id representedObject = [sender representedObject];
132
133 // NSMenuItems with a represented selection handler belong solely to the UI process
134 // and don't need any further processing after the selection handler is called.
135 if ([representedObject isKindOfClass:[WKSelectionHandlerWrapper class]]) {
136 [representedObject executeSelectionHandler];
137 return;
138 }
139
140 ASSERT(!sender || [sender isKindOfClass:NSMenuItem.class]);
141 WebKit::WebContextMenuItemData item(WebCore::ActionType, static_cast<WebCore::ContextMenuAction>([sender tag]), [sender title], [sender isEnabled], [(NSMenuItem *)sender state] == NSControlStateValueOn);
142 if (representedObject) {
143 ASSERT([representedObject isKindOfClass:[WKUserDataWrapper class]]);
144 item.setUserData([static_cast<WKUserDataWrapper *>(representedObject) userData]);
145 }
146
147 _menuProxy->contextMenuItemSelected(item);
148}

208void WebContextMenuProxyMac::setupServicesMenu()
209{
210 bool includeEditorServices = m_context.controlledDataIsEditable();
211 bool hasControlledImage = m_context.controlledImage();
212 NSArray *items = nil;
213 if (hasControlledImage) {
214 RefPtr<ShareableBitmap> image = m_context.controlledImage();
215 if (!image)
216 return;
217
218 auto cgImage = image->makeCGImage();
219 auto nsImage = adoptNS([[NSImage alloc] initWithCGImage:cgImage.get() size:image->size()]);
220
221ALLOW_DEPRECATED_DECLARATIONS_BEGIN
222 auto itemProvider = adoptNS([[NSItemProvider alloc] initWithItem:[nsImage TIFFRepresentation] typeIdentifier:(__bridge NSString *)kUTTypeTIFF]);
223ALLOW_DEPRECATED_DECLARATIONS_END
224 items = @[ itemProvider.get() ];
225 } else if (!m_context.controlledSelectionData().isEmpty()) {
226 auto selectionData = adoptNS([[NSData alloc] initWithBytes:static_cast<const void*>(m_context.controlledSelectionData().data()) length:m_context.controlledSelectionData().size()]);
227 auto selection = adoptNS([[NSAttributedString alloc] initWithRTFD:selectionData.get() documentAttributes:nil]);
228
229 items = @[ selection.get() ];
230 } else {
231 LOG_ERROR("No service controlled item represented in the context");
232 return;
233 }
234
235 RetainPtr<NSSharingServicePicker> picker = adoptNS([[NSSharingServicePicker alloc] initWithItems:items]);
236 [picker setStyle:hasControlledImage ? NSSharingServicePickerStyleRollover : NSSharingServicePickerStyleTextSelection];
237 [picker setDelegate:[WKSharingServicePickerDelegate sharedSharingServicePickerDelegate]];
238 [[WKSharingServicePickerDelegate sharedSharingServicePickerDelegate] setPicker:picker.get()];
239 [[WKSharingServicePickerDelegate sharedSharingServicePickerDelegate] setFiltersEditingServices:!includeEditorServices];
240 [[WKSharingServicePickerDelegate sharedSharingServicePickerDelegate] setHandlesEditingReplacement:includeEditorServices];
241
242 m_menu = adoptNS([[picker menu] copy]);
243
244 if (!hasControlledImage)
245 [m_menu setShowsStateColumn:YES];
246
247 // Explicitly add a menu item for each telephone number that is in the selection.
248 Vector<RetainPtr<NSMenuItem>> telephoneNumberMenuItems;
249
250 for (auto& telephoneNumber : m_context.selectedTelephoneNumbers()) {
251 if (NSMenuItem *item = menuItemForTelephoneNumber(telephoneNumber)) {
252 [item setIndentationLevel:1];
253 telephoneNumberMenuItems.append(item);
254 }
255 }
256
257 if (!telephoneNumberMenuItems.isEmpty()) {
258 if (m_menu)
259 [m_menu insertItem:[NSMenuItem separatorItem] atIndex:0];
260 else
261 m_menu = adoptNS([[NSMenu alloc] init]);
262 int itemPosition = 0;
263 auto groupEntry = adoptNS([[NSMenuItem alloc] initWithTitle:menuItemTitleForTelephoneNumberGroup() action:nil keyEquivalent:@""]);
264 [groupEntry setEnabled:NO];
265 [m_menu insertItem:groupEntry.get() atIndex:itemPosition++];
266 for (auto& menuItem : telephoneNumberMenuItems)
267 [m_menu insertItem:menuItem.get() atIndex:itemPosition++];
268 }
269
270 // If there is no services menu, then the existing services on the system have changed, so refresh that list of services.
271 // If <rdar://problem/17954709> is resolved then we can more accurately keep the list up to date without this call.
272 if (!m_menu)
273 ServicesController::singleton().refreshExistingServices();
274}

292{
293#if HAVE(NSSHARINGSERVICEPICKER_ASYNC_MENUS)
294 // FIXME (<rdar://problem/54551500>): Replace this with the async variant of +[NSMenuItem standardShareMenuItemForItems:] when it's available.
295 auto sharingServicePicker = adoptNS([[NSSharingServicePicker alloc] initWithItems:items]);
296 [sharingServicePicker setStyle:NSSharingServicePickerStyleMenu];
297 [sharingServicePicker getMenuWithCompletion:^(NSMenu *shareMenu) {
298 ASSERT(isMainThread());
299 shareMenu.delegate = (id <NSMenuDelegate>)sharingServicePicker.get();
300 auto shareMenuItem = adoptNS([[NSMenuItem alloc] initWithTitle:WEB_UI_STRING("Share", "Title for Share context menu item.") action:nil keyEquivalent:@""]);
301 [shareMenuItem setRepresentedObject:sharingServicePicker.get()];
302 [shareMenuItem setSubmenu:shareMenu];
303 completionHandler(shareMenuItem.get());
304 }];
305#else
306 completionHandler([NSMenuItem standardShareMenuItemForItems:items]);
307#endif
308}

310void WebContextMenuProxyMac::getShareMenuItem(CompletionHandler<void(NSMenuItem *)>&& completionHandler)
311{
312 const WebHitTestResultData& hitTestData = m_context.webHitTestResultData();
313
314 auto items = adoptNS([[NSMutableArray alloc] init]);
315
316 if (!hitTestData.absoluteLinkURL.isEmpty()) {
317 auto absoluteLinkURL = URL({ }, hitTestData.absoluteLinkURL);
318 if (!absoluteLinkURL.isEmpty())
319 [items addObject:(NSURL *)absoluteLinkURL];
320 }
321
322 if (hitTestData.isDownloadableMedia && !hitTestData.absoluteMediaURL.isEmpty()) {
323 auto downloadableMediaURL = URL({ }, hitTestData.absoluteMediaURL);
324 if (!downloadableMediaURL.isEmpty())
325 [items addObject:(NSURL *)downloadableMediaURL];
326 }
327
328 if (hitTestData.imageSharedMemory && hitTestData.imageSize) {
329 if (auto image = adoptNS([[NSImage alloc] initWithData:[NSData dataWithBytes:(unsigned char*)hitTestData.imageSharedMemory->data() length:hitTestData.imageSize]]))
330 [items addObject:image.get()];
331 }
332
333 if (!m_context.selectedText().isEmpty())
334 [items addObject:(NSString *)m_context.selectedText()];
335
336 if (![items count]) {
337 completionHandler(nil);
338 return;
339 }
340
341 getStandardShareMenuItem(items.get(), makeBlockPtr([completionHandler = WTFMove(completionHandler), protectedThis = makeRef(*this), this](NSMenuItem *item) mutable {
342 if (!item) {
343 completionHandler(nil);
344 return;
345 }
346
347 NSSharingServicePicker *sharingServicePicker = item.representedObject;
348 WKSharingServicePickerDelegate *sharingServicePickerDelegate = WKSharingServicePickerDelegate.sharedSharingServicePickerDelegate;
349 sharingServicePicker.delegate = sharingServicePickerDelegate;
350
351 sharingServicePickerDelegate.filtersEditingServices = NO;
352 sharingServicePickerDelegate.handlesEditingReplacement = NO;
353 sharingServicePickerDelegate.menuProxy = this;
354
355 // Setting the picker lets the delegate retain it to keep it alive, but this picker is kept alive by the menu item.
356 sharingServicePickerDelegate.picker = nil;
357
358 item.identifier = _WKMenuItemIdentifierShareMenu;
359
360 completionHandler(item);
361 }).get());
362}

469static RetainPtr<NSMenuItem> createMenuActionItem(const WebContextMenuItemData& item)
470{
471 auto type = item.type();
472 ASSERT_UNUSED(type, type == WebCore::ActionType || type == WebCore::CheckableActionType);
473
474 auto menuItem = adoptNS([[NSMenuItem alloc] initWithTitle:item.title() action:@selector(forwardContextMenuAction:) keyEquivalent:@""]);
475
476 [menuItem setTag:item.action()];
477 [menuItem setEnabled:item.enabled()];
478 [menuItem setState:item.checked() ? NSControlStateValueOn : NSControlStateValueOff];
479 [menuItem setIndentationLevel:item.indentationLevel()];
480 [menuItem setTarget:[WKMenuTarget sharedMenuTarget]];
481 [menuItem setIdentifier:menuItemIdentifier(item.action())];
482
483 if (item.userData())
484 [menuItem setRepresentedObject:adoptNS([[WKUserDataWrapper alloc] initWithUserData:item.userData()]).get()];
485
486 return menuItem;
487}

489void WebContextMenuProxyMac::getContextMenuFromItems(const Vector<WebContextMenuItemData>& items, CompletionHandler<void(NSMenu *)>&& completionHandler)
490{
491 auto menu = adoptNS([[NSMenu alloc] initWithTitle:@""]);
492 [menu setAutoenablesItems:NO];
493
494 if (items.isEmpty()) {
495 completionHandler(menu.get());
496 return;
497 }
498
499 auto filteredItems = items;
500 auto webView = m_webView.get();
501
502 bool isPopover = webView.get().window._childWindowOrderingPriority == NSWindowChildOrderingPriorityPopover;
503 bool isLookupDisabled = [NSUserDefaults.standardUserDefaults boolForKey:@"LULookupDisabled"];
504
505 if (isLookupDisabled || isPopover) {
506 filteredItems.removeAllMatching([] (auto& item) {
507 return item.action() == WebCore::ContextMenuItemTagLookUpInDictionary;
508 });
509 }
510
511 bool shouldUpdateQuickLookItemTitle = false;
512 std::optional<WebContextMenuItemData> quickLookItemToInsertIfNeeded;
513
514#if ENABLE(IMAGE_ANALYSIS)
515 auto indexOfQuickLookItem = filteredItems.findMatching([&] (auto& item) {
516 return item.action() == WebCore::ContextMenuItemTagQuickLookImage;
517 });
518
519 if (indexOfQuickLookItem != notFound) {
520 if (auto page = this->page(); page && page->preferences().preferInlineTextSelectionInImages()) {
521 quickLookItemToInsertIfNeeded = filteredItems[indexOfQuickLookItem];
522 filteredItems.remove(indexOfQuickLookItem);
523 } else
524 shouldUpdateQuickLookItemTitle = true;
525 }
526#endif // ENABLE(IMAGE_ANALYSIS)
527
528#if HAVE(TRANSLATION_UI_SERVICES)
529 if (!page()->canHandleContextMenuTranslation() || isPopover) {
530 filteredItems.removeAllMatching([] (auto& item) {
531 return item.action() == ContextMenuItemTagTranslate;
532 });
533 }
534#endif
535
536 auto imageURL = URL { URL { }, m_context.webHitTestResultData().absoluteImageURL };
537 auto imageBitmap = m_context.webHitTestResultData().imageBitmap;
538
539 auto sparseMenuItems = retainPtr([NSPointerArray strongObjectsPointerArray]);
540 auto insertMenuItem = makeBlockPtr([protectedThis = makeRef(*this), weakPage = makeWeakPtr(page()), imageURL = WTFMove(imageURL), imageBitmap = WTFMove(imageBitmap), shouldUpdateQuickLookItemTitle, quickLookItemToInsertIfNeeded = WTFMove(quickLookItemToInsertIfNeeded), completionHandler = WTFMove(completionHandler), itemsRemaining = filteredItems.size(), menu = WTFMove(menu), sparseMenuItems](NSMenuItem *item, NSUInteger index) mutable {
541 ASSERT(index < [sparseMenuItems count]);
542 ASSERT(![sparseMenuItems pointerAtIndex:index]);
543 [sparseMenuItems replacePointerAtIndex:index withPointer:item];
544 if (--itemsRemaining)
545 return;
546
547 [menu setItemArray:[sparseMenuItems allObjects]];
548
549 auto page = makeRefPtr(weakPage.get());
550 if (page && imageBitmap) {
551#if ENABLE(IMAGE_ANALYSIS)
552 protectedThis->insertOrUpdateQuickLookImageItem(imageURL, imageBitmap.releaseNonNull(), WTFMove(quickLookItemToInsertIfNeeded), shouldUpdateQuickLookItemTitle);
553#else
554 UNUSED_PARAM(quickLookItemToInsertIfNeeded);
555 UNUSED_PARAM(shouldUpdateQuickLookItemTitle);
556 UNUSED_PARAM(imageURL);
557#endif
558 }
559
560 completionHandler(menu.get());
561 });
562
563 for (size_t i = 0; i < filteredItems.size(); ++i) {
564 [sparseMenuItems addPointer:nullptr];
565 getContextMenuItem(filteredItems[i], [insertMenuItem, i](NSMenuItem *menuItem) {
566 insertMenuItem(menuItem, i);
567 });
568 }
569}

573void WebContextMenuProxyMac::insertOrUpdateQuickLookImageItem(const URL& imageURL, Ref<ShareableBitmap>&& imageBitmap, std::optional<WebContextMenuItemData>&& quickLookItemToInsertIfNeeded, bool shouldUpdateQuickLookItemTitle)
574{
575 auto page = makeRef(*this->page());
576 if (quickLookItemToInsertIfNeeded) {
577 page->computeHasImageAnalysisResults(imageURL, imageBitmap.get(), ImageAnalysisType::VisualSearch, [weakThis = makeWeakPtr(*this), quickLookItemToInsertIfNeeded = WTFMove(*quickLookItemToInsertIfNeeded)] (bool hasVisualSearchResults) mutable {
578 if (auto protectedThis = makeRefPtr(weakThis.get()); protectedThis && hasVisualSearchResults) {
579 protectedThis->m_quickLookPreviewActivity = QuickLookPreviewActivity::VisualSearch;
580 [protectedThis->m_menu addItem:NSMenuItem.separatorItem];
581 [protectedThis->m_menu addItem:createMenuActionItem(quickLookItemToInsertIfNeeded).get()];
582 }
583 });
584 return;
585 }
586
587 if (shouldUpdateQuickLookItemTitle) {
588 page->computeHasImageAnalysisResults(imageURL, imageBitmap.get(), ImageAnalysisType::VisualSearch, [weakThis = makeWeakPtr(*this), weakPage = makeWeakPtr(page.get()), imageURL, imageBitmap = WTFMove(imageBitmap)] (bool hasVisualSearchResults) mutable {
589 auto protectedThis = makeRefPtr(weakThis.get());
590 if (!protectedThis)
591 return;
592
593 auto page = makeRefPtr(weakPage.get());
594 if (!page)
595 return;
596
597 if (hasVisualSearchResults) {
598 protectedThis->m_quickLookPreviewActivity = QuickLookPreviewActivity::VisualSearch;
599 protectedThis->updateQuickLookContextMenuItemTitle(contextMenuItemTagQuickLookImageForVisualSearch());
600 return;
601 }
602
603 page->computeHasImageAnalysisResults(imageURL, imageBitmap.get(), ImageAnalysisType::Text, [weakThis = WTFMove(weakThis), weakPage] (bool hasText) mutable {
604 auto protectedThis = makeRefPtr(weakThis.get());
605 if (!protectedThis)
606 return;
607
608 if (auto page = makeRefPtr(weakPage.get()); page && hasText)
609 protectedThis->updateQuickLookContextMenuItemTitle(contextMenuItemTagQuickLookImageForTextSelection());
610 });
611 });
612 }
613}

628void WebContextMenuProxyMac::getContextMenuItem(const WebContextMenuItemData& item, CompletionHandler<void(NSMenuItem *)>&& completionHandler)
629{
630#if ENABLE(SERVICE_CONTROLS)
631 if (item.action() == ContextMenuItemTagShareMenu) {
632 getShareMenuItem(WTFMove(completionHandler));
633 return;
634 }
635#endif
636
637 switch (item.type()) {
638 case WebCore::ActionType:
639 case WebCore::CheckableActionType:
640 completionHandler(createMenuActionItem(item).get());
641 return;
642
643 case WebCore::SeparatorType:
644 completionHandler(NSMenuItem.separatorItem);
645 return;
646
647 case WebCore::SubmenuType: {
648 getContextMenuFromItems(item.submenu(), [action = item.action(), completionHandler = WTFMove(completionHandler), enabled = item.enabled(), title = item.title(), indentationLevel = item.indentationLevel()](NSMenu *menu) mutable {
649 auto menuItem = adoptNS([[NSMenuItem alloc] initWithTitle:title action:nullptr keyEquivalent:@""]);
650 [menuItem setEnabled:enabled];
651 [menuItem setIndentationLevel:indentationLevel];
652 [menuItem setSubmenu:menu];
653 [menuItem setIdentifier:menuItemIdentifier(action)];
654 completionHandler(menuItem.get());
655 });
656 return;
657 }
658 }

661void WebContextMenuProxyMac::showContextMenuWithItems(Vector<Ref<WebContextMenuItem>>&& items)
662{
663#if ENABLE(SERVICE_CONTROLS)
664 if (m_context.isServicesMenu()) {
665 ASSERT(items.isEmpty());
666 showServicesMenu();
667 return;
668 }
669#endif
670
671 if (page()->contextMenuClient().canShowContextMenu()) {
672 page()->contextMenuClient().showContextMenu(*page(), m_context.menuLocation(), items);
673 return;
674 }
675
676 ASSERT(items.isEmpty());
677 if (!m_menu)
678 return;
679
680 auto webView = m_webView.get();
681 NSPoint menuLocation = [webView convertPoint:m_context.menuLocation() toView:nil];
682 NSEvent *event = [NSEvent mouseEventWithType:NSEventTypeRightMouseUp location:menuLocation modifierFlags:0 timestamp:0 windowNumber:[webView window].windowNumber context:nil eventNumber:0 clickCount:0 pressure:0];
683 [NSMenu popUpContextMenu:m_menu.get() withEvent:event forView:webView.get()];
684}

686void WebContextMenuProxyMac::useContextMenuItems(Vector<Ref<WebContextMenuItem>>&& items)
687{
688 if (items.isEmpty() || !page() || page()->contextMenuClient().canShowContextMenu()) {
689 WebContextMenuProxy::useContextMenuItems(WTFMove(items));
690 return;
691 }
692
693 auto data = WTF::map(items, [](auto& item) {
694 return item->data();
695 });
696
697 getContextMenuFromItems(data, [this, protectedThis = makeRef(*this)](NSMenu *menu) mutable {
698 if (!page()) {

733static NSDictionary *contentsOfContextMenuItem(NSMenuItem *item)
734{
735 NSMutableDictionary* result = NSMutableDictionary.dictionary;
736
737 if (item.title.length)
738 result[@"title"] = item.title;
739
740 if (item.isSeparatorItem)
741 result[@"separator"] = @YES;
742 else if (!item.enabled)
743 result[@"enabled"] = @NO;
744
745 if (NSInteger indentationLevel = item.indentationLevel)
746 result[@"indentationLevel"] = [NSNumber numberWithInteger:indentationLevel];
747
748 if (item.state == NSControlStateValueOn)
749 result[@"checked"] = @YES;
750
751 if (NSArray<NSMenuItem *> *submenuItems = item.submenu.itemArray) {
752 NSMutableArray *children = [NSMutableArray arrayWithCapacity:submenuItems.count];
753 for (NSMenuItem *submenuItem : submenuItems)
754 [children addObject:contentsOfContextMenuItem(submenuItem)];
755 result[@"children"] = children;
756 }
757
758 return result;
759}
