84void WebDataListSuggestionsDropdownMac::show(WebCore::DataListSuggestionInformation&& information)
85{
86 if (m_dropdownUI) {
87 [m_dropdownUI updateWithInformation:WTFMove(information)];
88 return;
89 }
90
91 m_dropdownUI = adoptNS([[WKDataListSuggestionsController alloc] initWithInformation:WTFMove(information) inView:m_view]);
92 [m_dropdownUI showSuggestionsDropdown:*this];
93}

95void WebDataListSuggestionsDropdownMac::didSelectOption(const String& selectedOption)
96{
97 if (!m_page)
98 return;
99
100 m_page->didSelectOption(selectedOption);
101 close();
102}
103
104void WebDataListSuggestionsDropdownMac::selectOption()
105{
106 if (!m_page)
107 return;
108
109 String selectedOption = [m_dropdownUI currentSelectedString];
110 if (!selectedOption.isNull())
111 m_page->didSelectOption(selectedOption);
112
113 close();
114}
115
116void WebDataListSuggestionsDropdownMac::handleKeydownWithIdentifier(const String& key)
117{
118 if (key == "Enter")
119 selectOption();
120 else if (key == "Up" || key == "Down")
121 [m_dropdownUI moveSelectionByDirection:key];
122}
123
124void WebDataListSuggestionsDropdownMac::close()
125{
126 [m_dropdownUI invalidate];
127 m_dropdownUI = nil;
128 WebDataListSuggestionsDropdown::close();
129}

179- (id)initWithFrame:(NSRect)frameRect
180{
181 if (!(self = [super initWithFrame:frameRect]))
182 return self;
183
184 _valueField = adoptNS([[NSTextField alloc] init]);
185 _labelField = adoptNS([[NSTextField alloc] init]);
186 _bottomDivider = adoptNS([[NSView alloc] init]);
187 [_bottomDivider setWantsLayer:YES];
188 [_bottomDivider setHidden:YES];
189 [_bottomDivider layer].backgroundColor = NSColor.separatorColor.CGColor;
190 [self addSubview:_bottomDivider.get()];
191
192 auto setUpTextField = [&](NSTextField *textField) {
193 textField.editable = NO;
194 textField.bezeled = NO;
195 textField.font = [NSFont menuFontOfSize:0];
196 textField.drawsBackground = NO;
197 [self addSubview:textField];
198 };
199
200 setUpTextField(_valueField.get());
201 setUpTextField(_labelField.get());
202 self.identifier = suggestionCellReuseIdentifier;
203
204 return self;
205}
206
207- (void)layout
208{
209 [super layout];
210
211 auto bounds = self.bounds;
212 auto width = bounds.size.width;
213 if (![_labelField isHidden]) {
214 auto halfOfHeight = bounds.size.height / 2;
215 [_valueField setFrame:NSMakeRect(0, halfOfHeight, width, halfOfHeight)];
216 [_labelField setFrame:NSMakeRect(0, 0, width, halfOfHeight)];
217 } else
218 [_valueField setFrame:bounds];
219
220 if (![_bottomDivider isHidden])
221 [_bottomDivider setFrame:NSMakeRect(0, 0, width, 0.5)];
222}
223
224- (void)setValue:(NSString *)value label:(NSString *)label
225{
226 if ([[_valueField stringValue] isEqualToString:value] && [[_labelField stringValue] isEqualToString:label])
227 return;
228
229 [_valueField setStringValue:value];
230 [_labelField setStringValue:label];
231 [_labelField setHidden:!label.length];
232 self.needsLayout = YES;
233}
234
235- (void)setShouldShowBottomDivider:(BOOL)showBottomDivider
236{
237 if ([_bottomDivider isHidden] == !showBottomDivider)
238 return;
239
240 [_bottomDivider setHidden:!showBottomDivider];
241 self.needsLayout = YES;
242}

244- (BOOL)shouldShowBottomDivider

308static BOOL shouldShowDividersBetweenCells(const Vector<WebCore::DataListSuggestion>& suggestions)
309{
310 return notFound != suggestions.findMatching([](auto& suggestion) {
311 return !suggestion.label.isEmpty();
312 });
313}

315@implementation WKDataListSuggestionsController {
316 WeakPtr<WebKit::WebDataListSuggestionsDropdownMac> _dropdown;
317 Vector<WebCore::DataListSuggestion> _suggestions;
318 NSView *_presentingView;
319
320 RetainPtr<NSScrollView> _scrollView;
321 RetainPtr<WKDataListSuggestionWindow> _enclosingWindow;
322 RetainPtr<WKDataListSuggestionTableView> _table;
323 BOOL _showDividersBetweenCells;
324}

326- (id)initWithInformation:(WebCore::DataListSuggestionInformation&&)information inView:(NSView *)presentingView
327{
328 if (!(self = [super init]))
329 return self;
330
331 _presentingView = presentingView;
332 _suggestions = WTFMove(information.suggestions);
333 _showDividersBetweenCells = shouldShowDividersBetweenCells(_suggestions);
334 _table = adoptNS([[WKDataListSuggestionTableView alloc] initWithElementRect:information.elementRect]);
335
336 _enclosingWindow = adoptNS([[WKDataListSuggestionWindow alloc] initWithContentRect:NSZeroRect styleMask:(NSWindowStyleMaskTitled | NSWindowStyleMaskFullSizeContentView) backing:NSBackingStoreBuffered defer:NO]);
337 [_enclosingWindow setReleasedWhenClosed:NO];
338 [_enclosingWindow setFrame:[self dropdownRectForElementRect:information.elementRect] display:YES];
339 [_enclosingWindow setTitleVisibility:NSWindowTitleHidden];
340 [_enclosingWindow setTitlebarAppearsTransparent:YES];
341 [_enclosingWindow setMovable:NO];
342 [_enclosingWindow setBackgroundColor:[NSColor clearColor]];
343 [_enclosingWindow setOpaque:NO];
344
345 _scrollView = adoptNS([[NSScrollView alloc] initWithFrame:[_enclosingWindow contentView].bounds]);
346 [_scrollView setHasVerticalScroller:YES];
347 [_scrollView setVerticalScrollElasticity:NSScrollElasticityAllowed];
348 [_scrollView setHorizontalScrollElasticity:NSScrollElasticityNone];
349 [_scrollView setDocumentView:_table.get()];
350 [_scrollView setDrawsBackground:NO];
351
352 auto insetView =
353#if HAVE(NSTABLEVIEWSTYLE)
354 _scrollView;
355#else
356 [_scrollView contentView];
357#endif
358 [insetView setAutomaticallyAdjustsContentInsets:NO];
359 [insetView setContentInsets:NSEdgeInsetsMake(dropdownVerticalPadding, 0, dropdownVerticalPadding, 0)];
360
361 [_table setDelegate:self];
362 [_table setDataSource:self];
363 [_table setAction:@selector(selectedRow:)];
364 [_table setTarget:self];
365
366 return self;
367}

369- (String)currentSelectedString
370{
371 NSInteger selectedRow = [_table selectedRow];
372
373 if (selectedRow >= 0 && static_cast<size_t>(selectedRow) < _suggestions.size())
374 return _suggestions.at(selectedRow).value;
375
376 return String();
377}

379- (void)updateWithInformation:(WebCore::DataListSuggestionInformation&&)information
380{
381 _suggestions = WTFMove(information.suggestions);
382 _showDividersBetweenCells = shouldShowDividersBetweenCells(_suggestions);
383 [_table reload];
384
385 [_enclosingWindow setFrame:[self dropdownRectForElementRect:information.elementRect] display:YES];
386 [_scrollView setFrame:[_enclosingWindow contentView].bounds];
387}

397- (void)moveSelectionByDirection:(const String&)direction
398{
399 size_t size = _suggestions.size();
400 NSInteger oldSelection = [_table selectedRow];
401
402 size_t newSelection;
403 if (oldSelection != -1) {
404 if (direction == "Up")
405 newSelection = oldSelection ? (oldSelection - 1) : (size - 1);
406 else
407 newSelection = (oldSelection + 1) % size;
408 } else
409 newSelection = (direction == "Up") ? (size - 1) : 0;
410
411 [_table selectRowIndexes:[NSIndexSet indexSetWithIndex:newSelection] byExtendingSelection:NO];
412 [_table scrollRowToVisible:newSelection];
413
414 // Notify accessibility clients of new selection.
415 NSString *currentSelectedString = [self currentSelectedString];
416 [self notifyAccessibilityClients:currentSelectedString];
417}
418
419- (void)invalidate
420{
421 [_table removeFromSuperviewWithoutNeedingDisplay];
422 [_scrollView removeFromSuperviewWithoutNeedingDisplay];
423
424 [_table setDelegate:nil];
425 [_table setDataSource:nil];
426 [_table setTarget:nil];
427
428 _table = nil;
429 _scrollView = nil;
430
431 [[_presentingView window] removeChildWindow:_enclosingWindow.get()];
432 [_enclosingWindow close];
433 _enclosingWindow = nil;
434
435 // Notify accessibility clients that datalist went away.
436 NSString *info = WEB_UI_STRING("Suggestions list hidden.", "Accessibility announcement for the data list suggestions dropdown going away.");
437 [self notifyAccessibilityClients:info];
438}
439
440- (NSRect)dropdownRectForElementRect:(const WebCore::IntRect&)rect
441{
442 NSRect windowRect = [[_presentingView window] convertRectToScreen:[_presentingView convertRect:rect toView:nil]];
443
444 CGFloat totalCellHeight = 0;
445 for (auto& suggestion : _suggestions)
446 totalCellHeight += suggestion.label.isEmpty() ? dropdownRowHeightWithoutLabel : dropdownRowHeightWithLabel;
447
448 CGFloat totalIntercellSpacingAndPadding = dropdownVerticalPadding * 2;
449 if (_suggestions.size() > 1)
450 totalIntercellSpacingAndPadding += (_suggestions.size() - 1) * [_table intercellSpacing].height;
451
452 CGFloat height = totalIntercellSpacingAndPadding + std::min(totalCellHeight, maximumTotalHeightForDropdownCells);
453 return NSMakeRect(NSMinX(windowRect), NSMinY(windowRect) - height - dropdownTopMargin, rect.width(), height);
454}
455
456- (void)showSuggestionsDropdown:(WebKit::WebDataListSuggestionsDropdownMac&)dropdown
457{
458 _dropdown = makeWeakPtr(dropdown);
459 [[_enclosingWindow contentView] addSubview:_scrollView.get()];
460 [_table reload];
461 [[_presentingView window] addChildWindow:_enclosingWindow.get() ordered:NSWindowAbove];
462 [_scrollView flashScrollers];
463
464 // Notify accessibility clients of datalist becoming visible.
465 NSString *currentSelectedString = [self currentSelectedString];
466 NSString *info = [NSString stringWithFormat:WEB_UI_STRING("Suggestions list visible, %@", "Accessibility announcement that the suggestions list became visible. The format argument is for the first option in the list."), currentSelectedString];
467 [self notifyAccessibilityClients:info];
468}

470- (void)selectedRow:(NSTableView *)sender
471{
472 auto selectedString = self.currentSelectedString;
473 if (!selectedString)
474 return;
475
476 _dropdown->didSelectOption(selectedString);
477}

479- (NSInteger)numberOfRowsInTableView:(NSTableView *)tableView

484- (CGFloat)tableView:(NSTableView *)tableView heightOfRow:(NSInteger)row
485{
486 auto suggestionIndex = static_cast<size_t>(row);
487 if (suggestionIndex >= _suggestions.size()) {
488 ASSERT_NOT_REACHED();
489 return 0;
490 }
491
492 return _suggestions.at(suggestionIndex).label.isEmpty() ? dropdownRowHeightWithoutLabel : dropdownRowHeightWithLabel;
493}
494
495- (NSTableRowView *)tableView:(NSTableView *)tableView rowViewForRow:(NSInteger)row
496{
497 return adoptNS([[WKDataListSuggestionTableRowView alloc] init]).autorelease();
498}
499
500- (NSView *)tableView:(NSTableView *)tableView viewForTableColumn:(NSTableColumn *)tableColumn row:(NSInteger)row
501{
502 auto result = retainPtr([tableView makeViewWithIdentifier:suggestionCellReuseIdentifier owner:self]);
503
504 if (!result) {
505 result = adoptNS([[WKDataListSuggestionView alloc] init]);
506 [result setIdentifier:suggestionCellReuseIdentifier];
507 }
508
509 auto& suggestion = _suggestions.at(row);
510 [result setShouldShowBottomDivider:_showDividersBetweenCells && row < static_cast<NSInteger>(_suggestions.size() - 1)];
511 [result setValue:suggestion.value label:suggestion.label];
512
513 return result.autorelease();
514}
