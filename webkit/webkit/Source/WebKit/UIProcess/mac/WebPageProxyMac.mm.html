118static inline bool expectsLegacyImplicitRubberBandControl()
119{
120 if (MacApplication::isSafari()) {
121 const int32_t firstVersionOfSafariNotExpectingImplicitRubberBandControl = 0x021A0F00; // 538.15.0
122 bool linkedAgainstSafariExpectingImplicitRubberBandControl = NSVersionOfLinkTimeLibrary("Safari") < firstVersionOfSafariNotExpectingImplicitRubberBandControl;
123 return linkedAgainstSafariExpectingImplicitRubberBandControl;
124 }
125
126 const int32_t firstVersionOfWebKit2WithNoImplicitRubberBandControl = 0x021A0200; // 538.2.0
127 int32_t linkedWebKit2Version = NSVersionOfLinkTimeLibrary("WebKit2");
128 return linkedWebKit2Version != -1 && linkedWebKit2Version < firstVersionOfWebKit2WithNoImplicitRubberBandControl;
129}

230RefPtr<WebCore::SharedBuffer> WebPageProxy::dataSelectionForPasteboard(const String& pasteboardType)
231{
232 if (!hasRunningProcess())
233 return nullptr;
234
235 SharedMemory::IPCHandle ipcHandle;
236 const Seconds messageTimeout(20);
237 sendSync(Messages::WebPage::GetDataSelectionForPasteboard(pasteboardType), Messages::WebPage::GetDataSelectionForPasteboard::Reply(ipcHandle), messageTimeout);
238 MESSAGE_CHECK_WITH_RETURN_VALUE(!ipcHandle.handle.isNull(), nullptr);
239
240 auto sharedMemoryBuffer = SharedMemory::map(ipcHandle.handle, SharedMemory::Protection::ReadOnly);
241 if (!sharedMemoryBuffer)
242 return nullptr;
243 return SharedBuffer::create(static_cast<unsigned char *>(sharedMemoryBuffer->data()), ipcHandle.dataSize);
244}

268void WebPageProxy::setPromisedDataForImage(const String& pasteboardName, const SharedMemory::IPCHandle& imageHandle, const String& filename, const String& extension,
269 const String& title, const String& url, const String& visibleURL, const SharedMemory::IPCHandle& archiveHandle, const String& originIdentifier)
270{
271 MESSAGE_CHECK_URL(url);
272 MESSAGE_CHECK_URL(visibleURL);
273 MESSAGE_CHECK(!imageHandle.handle.isNull());
274
275 auto sharedMemoryImage = SharedMemory::map(imageHandle.handle, SharedMemory::Protection::ReadOnly);
276 if (!sharedMemoryImage)
277 return;
278
279 auto imageBuffer = SharedBuffer::create(static_cast<unsigned char*>(sharedMemoryImage->data()), static_cast<size_t>(imageHandle.dataSize));
280 RefPtr<SharedBuffer> archiveBuffer;
281
282 if (!archiveHandle.handle.isNull()) {
283 auto sharedMemoryArchive = SharedMemory::map(archiveHandle.handle, SharedMemory::Protection::ReadOnly);
284 if (!sharedMemoryArchive)
285 return;
286 archiveBuffer = SharedBuffer::create(static_cast<unsigned char*>(sharedMemoryArchive->data()), static_cast<size_t>(archiveHandle.dataSize));
287 }
288 pageClient().setPromisedDataForImage(pasteboardName, WTFMove(imageBuffer), ResourceResponseBase::sanitizeSuggestedFilename(filename), extension, title, url, visibleURL, WTFMove(archiveBuffer), originIdentifier);
289}

415static NSString *temporaryPDFDirectoryPath()
416{
417 static auto temporaryPDFDirectoryPath = makeNeverDestroyed([] {
418 NSString *temporaryDirectoryTemplate = [NSTemporaryDirectory() stringByAppendingPathComponent:@"WebKitPDFs-XXXXXX"];
419 CString templateRepresentation = [temporaryDirectoryTemplate fileSystemRepresentation];
420
421 if (mkdtemp(templateRepresentation.mutableData()))
422 return adoptNS([[[NSFileManager defaultManager] stringWithFileSystemRepresentation:templateRepresentation.data() length:templateRepresentation.length()] copy]);
423 return RetainPtr<id> { };
424 }());
425
426 return temporaryPDFDirectoryPath.get().get();
427}

429static NSString *pathToPDFOnDisk(const String& suggestedFilename)
430{
431 NSString *pdfDirectoryPath = temporaryPDFDirectoryPath();
432 if (!pdfDirectoryPath) {
433 WTFLogAlways("Cannot create temporary PDF download directory.");
434 return nil;
435 }
436
437 NSString *path = [pdfDirectoryPath stringByAppendingPathComponent:suggestedFilename];
438
439 NSFileManager *fileManager = [NSFileManager defaultManager];
440 if ([fileManager fileExistsAtPath:path]) {
441 NSString *pathTemplatePrefix = [pdfDirectoryPath stringByAppendingPathComponent:@"XXXXXX-"];
442 NSString *pathTemplate = [pathTemplatePrefix stringByAppendingString:suggestedFilename];
443 CString pathTemplateRepresentation = [pathTemplate fileSystemRepresentation];
444
445 int fd = mkstemps(pathTemplateRepresentation.mutableData(), pathTemplateRepresentation.length() - strlen([pathTemplatePrefix fileSystemRepresentation]) + 1);
446 if (fd < 0) {
447 WTFLogAlways("Cannot create PDF file in the temporary directory (%s).", suggestedFilename.utf8().data());
448 return nil;
449 }
450
451 close(fd);
452 path = [fileManager stringWithFileSystemRepresentation:pathTemplateRepresentation.data() length:pathTemplateRepresentation.length()];
453 }
454
455 return path;
456}

458void WebPageProxy::savePDFToTemporaryFolderAndOpenWithNativeApplication(const String& suggestedFilename, FrameInfoData&& frameInfo, const IPC::DataReference& data, const String& pdfUUID)
459{
460 if (data.isEmpty()) {
461 WTFLogAlways("Cannot save empty PDF file to the temporary directory.");
462 return;
463 }
464
465 auto sanitizedFilename = ResourceResponseBase::sanitizeSuggestedFilename(suggestedFilename);
466 if (!sanitizedFilename.endsWithIgnoringASCIICase(".pdf")) {
467 WTFLogAlways("Cannot save file without .pdf extension to the temporary directory.");
468 return;
469 }
470
471 auto nsPath = retainPtr(pathToPDFOnDisk(sanitizedFilename));
472
473 if (!nsPath)
474 return;
475
476 auto permissions = adoptNS([[NSNumber alloc] initWithInt:S_IRUSR]);
477 auto fileAttributes = adoptNS([[NSDictionary alloc] initWithObjectsAndKeys:permissions.get(), NSFilePosixPermissions, nil]);
478 auto nsData = adoptNS([[NSData alloc] initWithBytesNoCopy:(void*)data.data() length:data.size() freeWhenDone:NO]);
479
480 if (![[NSFileManager defaultManager] createFileAtPath:nsPath.get() contents:nsData.get() attributes:fileAttributes.get()]) {
481 WTFLogAlways("Cannot create PDF file in the temporary directory (%s).", sanitizedFilename.utf8().data());
482 return;
483 }
484 auto originatingURLString = frameInfo.request.url().string();
485 FileSystem::setMetadataURL(nsPath.get(), originatingURLString);
486
487 if (TemporaryPDFFileMap::isValidKey(pdfUUID))
488 m_temporaryPDFFiles.add(pdfUUID, nsPath.get());
489
490 auto pdfFileURL = URL::fileURLWithFileSystemPath(String(nsPath.get()));
491 m_uiClient->confirmPDFOpening(*this, pdfFileURL, WTFMove(frameInfo), [pdfFileURL] (bool allowed) {
492 if (!allowed)
493 return;
494 [[NSWorkspace sharedWorkspace] openURL:pdfFileURL];
495 });
496}

517#if ENABLE(PDFKIT_PLUGIN)
518void WebPageProxy::showPDFContextMenu(const WebKit::PDFContextMenu& contextMenu, PDFPluginIdentifier identifier, CompletionHandler<void(std::optional<int32_t>&&)>&& completionHandler)
519{
520 if (!contextMenu.items.size())
521 return completionHandler(std::nullopt);
522
523 RetainPtr<WKPDFMenuTarget> menuTarget = adoptNS([[WKPDFMenuTarget alloc] init]);
524 RetainPtr<NSMenu> nsMenu = adoptNS([[NSMenu alloc] init]);
525 [nsMenu setAllowsContextMenuPlugIns:false];
526 for (unsigned i = 0; i < contextMenu.items.size(); i++) {
527 auto& item = contextMenu.items[i];
528
529 if (item.separator) {
530 [nsMenu insertItem:[NSMenuItem separatorItem] atIndex:i];
531 continue;
532 }
533
534 RetainPtr<NSMenuItem> nsItem = adoptNS([[NSMenuItem alloc] init]);
535 [nsItem setTitle:item.title];
536 [nsItem setEnabled:item.enabled];
537 [nsItem setState:item.state];
538 if (item.hasAction) {
539 [nsItem setTarget:menuTarget.get()];
540 [nsItem setAction:@selector(contextMenuAction:)];
541 }
542 [nsItem setTag:item.tag];
543 [nsMenu insertItem:nsItem.get() atIndex:i];
544 }
545 NSWindow *window = pageClient().platformWindow();
546 auto windowNumber = [window windowNumber];
547 auto location = [window convertRectFromScreen: { contextMenu.point, NSZeroSize }].origin;
548 NSEvent* event = [NSEvent mouseEventWithType:NSEventTypeRightMouseDown location:location modifierFlags:0 timestamp:0 windowNumber:windowNumber context:0 eventNumber:0 clickCount:1 pressure:1];
549
550 auto view = [pageClient().platformWindow() contentView];
551 [NSMenu popUpContextMenu:nsMenu.get() withEvent:event forView:view];
552
553 if (auto selectedMenuItem = [menuTarget selectedMenuItem]) {
554 NSInteger tag = selectedMenuItem.tag;
555#if ENABLE(UI_PROCESS_PDF_HUD)
556 if (contextMenu.openInPreviewIndex && *contextMenu.openInPreviewIndex == tag)
557 pdfOpenWithPreview(identifier);
558#else
559 UNUSED_PARAM(identifier);
560#endif
561 return completionHandler(tag);
562 }
563 completionHandler(std::nullopt);
564}
565#endif

580void WebPageProxy::didUpdateEditorState(const EditorState& oldEditorState, const EditorState& newEditorState)
581{
582 bool couldChangeSecureInputState = newEditorState.isInPasswordField != oldEditorState.isInPasswordField || oldEditorState.selectionIsNone;
583 // Selection being none is a temporary state when editing. Flipping secure input state too quickly was causing trouble (not fully understood).
584 if (couldChangeSecureInputState && !newEditorState.selectionIsNone)
585 pageClient().updateSecureInputState();
586
587 if (newEditorState.shouldIgnoreSelectionChanges)
588 return;
589
590 pageClient().selectionDidChange();
591 updateFontAttributesAfterEditorStateChange();
592}

758void WebPageProxy::showImageInQuickLookPreviewPanel(ShareableBitmap& imageBitmap, const String& tooltip, const URL& imageURL, QuickLookPreviewActivity activity)
759{
760 if (!PAL::isQuickLookUIFrameworkAvailable() || !PAL::getQLPreviewPanelClass() || ![PAL::getQLItemClass() instancesRespondToSelector:@selector(initWithDataProvider:contentType:previewTitle:)])
761 return;
762
763 auto image = imageBitmap.makeCGImage();
764 if (!image)
765 return;
766
767 auto imageData = adoptCF(CFDataCreateMutable(kCFAllocatorDefault, 0));
768 auto destination = adoptCF(CGImageDestinationCreateWithData(imageData.get(), (__bridge CFStringRef)UTTypePNG.identifier, 1, nullptr));
769 if (!destination)
770 return;
771
772 CGImageDestinationAddImage(destination.get(), image.get(), nil);
773 if (!CGImageDestinationFinalize(destination.get()))
774 return;
775
776 m_quickLookPreviewController = adoptNS([[WKQuickLookPreviewController alloc] initWithPage:*this imageData:(__bridge NSData *)imageData.get() title:tooltip imageURL:imageURL activity:activity]);
777
778 // When presenting the shared QLPreviewPanel, QuickLook will search the responder chain for a suitable panel controller.
779 // Make sure that we (by default) start the search at the web view, which knows how to vend the Visual Search preview
780 // controller as a delegate and data source for the preview panel.
781 pageClient().makeFirstResponder();
782
783 auto previewPanel = [PAL::getQLPreviewPanelClass() sharedPreviewPanel];
784 [previewPanel makeKeyAndOrderFront:nil];
785
786 if (![m_quickLookPreviewController isControlling:previewPanel]) {
787 // The WebKit client may have overridden QLPreviewPanelController methods on the view without calling into the superclass.
788 // In this case, hand over control to the client and clear out our state eagerly, since we don't expect any further delegate
789 // calls once the preview panel is dismissed.
790 m_quickLookPreviewController.clear();
791 }
792}
