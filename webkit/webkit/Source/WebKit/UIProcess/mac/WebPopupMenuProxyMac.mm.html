58void WebPopupMenuProxyMac::populate(const Vector<WebPopupItem>& items, NSFont *font, TextDirection menuTextDirection)
59{
60 if (m_popup)
61 [m_popup removeAllItems];
62 else {
63 m_popup = adoptNS([[NSPopUpButtonCell alloc] initTextCell:@"" pullsDown:NO]);
64 [m_popup setUsesItemFromMenu:NO];
65 [m_popup setAutoenablesItems:NO];
66 }
67
68 int size = items.size();
69
70 for (int i = 0; i < size; i++) {
71 if (items[i].m_type == WebPopupItem::Type::Separator)
72 [[m_popup menu] addItem:[NSMenuItem separatorItem]];
73 else {
74 [m_popup addItemWithTitle:@""];
75 NSMenuItem *menuItem = [m_popup lastItem];
76
77 RetainPtr<NSMutableParagraphStyle> paragraphStyle = adoptNS([[NSParagraphStyle defaultParagraphStyle] mutableCopy]);
78 NSWritingDirection writingDirection = items[i].m_textDirection == TextDirection::LTR ? NSWritingDirectionLeftToRight : NSWritingDirectionRightToLeft;
79 [paragraphStyle setBaseWritingDirection:writingDirection];
80 [paragraphStyle setAlignment:menuTextDirection == TextDirection::LTR ? NSTextAlignmentLeft : NSTextAlignmentRight];
81 RetainPtr<NSMutableDictionary> attributes = adoptNS([[NSMutableDictionary alloc] initWithObjectsAndKeys:
82 paragraphStyle.get(), NSParagraphStyleAttributeName,
83 font, NSFontAttributeName,
84 nil]);
85 if (items[i].m_hasTextDirectionOverride) {
86 RetainPtr<NSNumber> writingDirectionValue = adoptNS([[NSNumber alloc] initWithInteger:writingDirection + NSWritingDirectionOverride]);
87 RetainPtr<NSArray> writingDirectionArray = adoptNS([[NSArray alloc] initWithObjects:writingDirectionValue.get(), nil]);
88 [attributes setObject:writingDirectionArray.get() forKey:NSWritingDirectionAttributeName];
89 }
90 RetainPtr<NSAttributedString> string = adoptNS([[NSAttributedString alloc] initWithString:nsStringFromWebCoreString(items[i].m_text) attributes:attributes.get()]);
91
92 [menuItem setAttributedTitle:string.get()];
93 // We set the title as well as the attributed title here. The attributed title will be displayed in the menu,
94 // but typeahead will use the non-attributed string that doesn't contain any leading or trailing whitespace.
95 [menuItem setTitle:[[string string] stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]]];
96 [menuItem setEnabled:items[i].m_isEnabled];
97 [menuItem setToolTip:nsStringFromWebCoreString(items[i].m_toolTip)];
98 }
99 }
100}

102void WebPopupMenuProxyMac::showPopupMenu(const IntRect& rect, TextDirection textDirection, double pageScaleFactor, const Vector<WebPopupItem>& items, const PlatformPopupMenuData& data, int32_t selectedIndex)
103{
104 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanCommunicateWithWindowServer));
105 NSFont *font;
106
107 BEGIN_BLOCK_OBJC_EXCEPTIONS
108
109 if (NSDictionary *fontAttributes = static_cast<NSDictionary *>(data.fontInfo.fontAttributeDictionary.get())) {
110 auto scaledFontSize = [fontAttributes[NSFontSizeAttribute] floatValue] * pageScaleFactor;
111
112 font = fontWithAttributes(fontAttributes, ((pageScaleFactor != 1) ? scaledFontSize : 0));
113 // font will be nil when using a custom font. However, we should still
114 // honor the font size, matching other browsers.
115 if (!font)
116 font = [NSFont menuFontOfSize:scaledFontSize];
117 } else
118 font = [NSFont menuFontOfSize:0];
119
120 END_BLOCK_OBJC_EXCEPTIONS
121
122 populate(items, font, textDirection);
123
124 [m_popup attachPopUpWithFrame:rect inView:m_webView];
125 [m_popup selectItemAtIndex:selectedIndex];
126 [m_popup setUserInterfaceLayoutDirection:textDirection == TextDirection::LTR ? NSUserInterfaceLayoutDirectionLeftToRight : NSUserInterfaceLayoutDirectionRightToLeft];
127
128 NSMenu *menu = [m_popup menu];
129 [menu setUserInterfaceLayoutDirection:textDirection == TextDirection::LTR ? NSUserInterfaceLayoutDirectionLeftToRight : NSUserInterfaceLayoutDirectionRightToLeft];
130
131 // These values were borrowed from AppKit to match their placement of the menu.
132 const int popOverHorizontalAdjust = -13;
133 const int popUnderHorizontalAdjust = 6;
134 const int popUnderVerticalAdjust = 6;
135
136 // Menus that pop-over directly obscure the node that generated the popup menu.
137 // Menus that pop-under are offset underneath it.
138 NSPoint location;
139 if (data.shouldPopOver) {
140 NSRect titleFrame = [m_popup titleRectForBounds:rect];
141 if (titleFrame.size.width <= 0 || titleFrame.size.height <= 0)
142 titleFrame = rect;
143 float verticalOffset = roundf((NSMaxY(rect) - NSMaxY(titleFrame)) + NSHeight(titleFrame)) + 1;
144 if (textDirection == TextDirection::LTR)
145 location = NSMakePoint(NSMinX(rect) + popOverHorizontalAdjust, NSMaxY(rect) - verticalOffset);
146 else
147 location = NSMakePoint(NSMaxX(rect) - popOverHorizontalAdjust, NSMaxY(rect) - verticalOffset);
148 } else {
149 if (textDirection == TextDirection::LTR)
150 location = NSMakePoint(NSMinX(rect) + popUnderHorizontalAdjust, NSMaxY(rect) + popUnderVerticalAdjust);
151 else
152 location = NSMakePoint(NSMaxX(rect) - popUnderHorizontalAdjust, NSMaxY(rect) + popUnderVerticalAdjust);
153 }
154 RetainPtr<NSView> dummyView = adoptNS([[NSView alloc] initWithFrame:rect]);
155 [dummyView.get() setUserInterfaceLayoutDirection:textDirection == TextDirection::LTR ? NSUserInterfaceLayoutDirectionLeftToRight : NSUserInterfaceLayoutDirectionRightToLeft];
156 [m_webView addSubview:dummyView.get()];
157 location = [dummyView convertPoint:location fromView:m_webView];
158
159 NSControlSize controlSize;
160 switch (data.menuSize) {
161 case WebCore::PopupMenuStyle::PopupMenuSizeNormal:
162 controlSize = NSControlSizeRegular;
163 break;
164 case WebCore::PopupMenuStyle::PopupMenuSizeSmall:
165 controlSize = NSControlSizeSmall;
166 break;
167 case WebCore::PopupMenuStyle::PopupMenuSizeMini:
168 controlSize = NSControlSizeMini;
169 break;
170#if HAVE(LARGE_CONTROL_SIZE)
171 case PopupMenuStyle::PopupMenuSizeLarge:
172 controlSize = NSControlSizeLarge;
173 break;
174#endif
175 }
176
177 Ref<WebPopupMenuProxyMac> protect(*this);
178 PAL::popUpMenu(menu, location, roundf(NSWidth(rect)), dummyView.get(), selectedIndex, font, controlSize, data.hideArrows);
179
180 [m_popup dismissPopUp];
181 [dummyView removeFromSuperview];
182
183 if (!m_client || m_wasCanceled)
184 return;
185
186 m_client->valueChangedForPopupMenu(this, [m_popup indexOfSelectedItem]);
187
188 // <https://bugs.webkit.org/show_bug.cgi?id=57904> This code is adopted from EventHandler::sendFakeEventsAfterWidgetTracking().
189 if (!m_client->currentlyProcessedMouseDownEvent())
190 return;
191
192 NSEvent* initiatingNSEvent = m_client->currentlyProcessedMouseDownEvent()->nativeEvent();
193 if ([initiatingNSEvent type] != NSEventTypeLeftMouseDown)
194 return;
195
196 NSEvent *fakeEvent = [NSEvent mouseEventWithType:NSEventTypeLeftMouseUp
197 location:[initiatingNSEvent locationInWindow]
198 modifierFlags:[initiatingNSEvent modifierFlags]
199 timestamp:[initiatingNSEvent timestamp]
200 windowNumber:[initiatingNSEvent windowNumber]
201 context:nullptr
202 eventNumber:[initiatingNSEvent eventNumber]
203 clickCount:[initiatingNSEvent clickCount]
204 pressure:[initiatingNSEvent pressure]];
205
206 [NSApp postEvent:fakeEvent atStart:YES];
207 fakeEvent = [NSEvent mouseEventWithType:NSEventTypeMouseMoved
208 location:[[m_webView window]
209ALLOW_DEPRECATED_DECLARATIONS_BEGIN
210 convertScreenToBase:[NSEvent mouseLocation]]
211ALLOW_DEPRECATED_DECLARATIONS_END
212 modifierFlags:[initiatingNSEvent modifierFlags]
213 timestamp:[initiatingNSEvent timestamp]
214 windowNumber:[initiatingNSEvent windowNumber]
215 context:nullptr
216 eventNumber:0
217 clickCount:0
218 pressure:0];
219 [NSApp postEvent:fakeEvent atStart:YES];
220}

222void WebPopupMenuProxyMac::hidePopupMenu()
223{
224 [[m_popup menu] cancelTracking];
225}

227void WebPopupMenuProxyMac::cancelTracking()
228{
229 [[m_popup menu] cancelTracking];
230 m_wasCanceled = true;
