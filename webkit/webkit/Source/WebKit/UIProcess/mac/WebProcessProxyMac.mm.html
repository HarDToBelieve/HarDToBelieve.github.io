45bool WebProcessProxy::shouldAllowNonValidInjectedCode() const
46{
47 static bool isSystemWebKit = [] {
48 NSBundle *webkit2Bundle = [NSBundle bundleForClass:NSClassFromString(@"WKWebView")];
49 return [webkit2Bundle.bundlePath hasPrefix:@"/System/"];
50 }();
51
52 if (!isSystemWebKit)
53 return false;
54
55 static bool isPlatformBinary = SecTaskGetCodeSignStatus(adoptCF(SecTaskCreateFromSelf(kCFAllocatorDefault)).get()) & CS_PLATFORM_BINARY;
56 if (isPlatformBinary)
57 return false;
58
59 const String& path = m_processPool->configuration().injectedBundlePath();
60 return !path.isEmpty() && !path.startsWith("/System/");
61}

82void WebProcessProxy::platformSuspendProcess()
83{
84 RELEASE_LOG(Process, "%p - [PID=%i] WebProcessProxy::platformSuspendProcess", this, processIdentifier());
85 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
86 if (auto* connection = this->connection())
87 xpc_connection_kill(connection->xpcConnection(), SIGSTOP);
88 ALLOW_DEPRECATED_DECLARATIONS_END
89}
90
91void WebProcessProxy::platformResumeProcess()
92{
93 RELEASE_LOG(Process, "%p - [PID=%i] WebProcessProxy::platformResumeProcess", this, processIdentifier());
94 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
95 if (auto* connection = this->connection())
96 xpc_connection_kill(connection->xpcConnection(), SIGCONT);
97 ALLOW_DEPRECATED_DECLARATIONS_END
98}
