36enum class AuthenticatorAttachment;

38struct ExceptionData;
39struct MockWebAuthenticationConfiguration;

44namespace WebKit {
45
46class WebAuthnProcess;
47struct WebAuthenticationRequestData;
48
49class WebAuthnConnectionToWebProcess
50 : public RefCounted<WebAuthnConnectionToWebProcess>
51 , IPC::Connection::Client {
52public:
53 static Ref<WebAuthnConnectionToWebProcess> create(WebAuthnProcess&, WebCore::ProcessIdentifier, IPC::Connection::Identifier);
54 virtual ~WebAuthnConnectionToWebProcess();
55
56 IPC::Connection& connection() { return m_connection.get(); }
57 WebAuthnProcess& WebAuthnProcessProcess() { return m_WebAuthnProcess.get(); }
58 WebCore::ProcessIdentifier webProcessIdentifier() const { return m_webProcessIdentifier; }
59
60private:
61 using RequestCompletionHandler = CompletionHandler<void(const WebCore::AuthenticatorResponseData&, WebCore::AuthenticatorAttachment, const WebCore::ExceptionData&)>;
62 using QueryCompletionHandler = CompletionHandler<void(bool)>;
63
64 WebAuthnConnectionToWebProcess(WebAuthnProcess&, WebCore::ProcessIdentifier, IPC::Connection::Identifier);
65
66 // IPC::Connection::Client
67 void didClose(IPC::Connection&) final;
68 void didReceiveInvalidMessage(IPC::Connection&, IPC::MessageName) final;
69 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) final;
70
71 // Receivers.
72 void makeCredential(Vector<uint8_t>&& hash, WebCore::PublicKeyCredentialCreationOptions&&, bool processingUserGesture, RequestCompletionHandler&&);
73 void getAssertion(Vector<uint8_t>&& hash, WebCore::PublicKeyCredentialRequestOptions&&, bool processingUserGesture, RequestCompletionHandler&&);
74 void isUserVerifyingPlatformAuthenticatorAvailable(QueryCompletionHandler&&);
75 void setMockWebAuthenticationConfiguration(WebCore::MockWebAuthenticationConfiguration&&);
76
77 void handleRequest(WebAuthenticationRequestData&&, RequestCompletionHandler&&);
78
79 Ref<IPC::Connection> m_connection;
80 Ref<WebAuthnProcess> m_WebAuthnProcess;
81 const WebCore::ProcessIdentifier m_webProcessIdentifier;
82};
