52void WebAuthnProcess::createWebAuthnConnectionToWebProcess(ProcessIdentifier identifier, CompletionHandler<void(std::optional<IPC::Attachment>&&)>&& completionHandler)
53{
54 auto ipcConnection = createIPCConnectionPair();
55 if (!ipcConnection) {
56 completionHandler({ });
57 return;
58 }
59
60 auto newConnection = WebAuthnConnectionToWebProcess::create(*this, identifier, ipcConnection->first);
61
62 ASSERT(!m_webProcessConnections.contains(identifier));
63 m_webProcessConnections.add(identifier, WTFMove(newConnection));
64
65 completionHandler(WTFMove(ipcConnection->second));
66}
67
68void WebAuthnProcess::removeWebAuthnConnectionToWebProcess(WebAuthnConnectionToWebProcess& connection)
69{
70 ASSERT(m_webProcessConnections.contains(connection.webProcessIdentifier()));
71 m_webProcessConnections.remove(connection.webProcessIdentifier());
72}

129void WebAuthnProcess::setMockWebAuthenticationConfiguration(WebCore::MockWebAuthenticationConfiguration&& configuration)
130{
131 if (!m_authenticatorManager->isMock()) {
132 m_authenticatorManager = makeUniqueRef<MockAuthenticatorManager>(WTFMove(configuration));
133 m_authenticatorManager->enableModernWebAuthentication();
134 return;
135 }
136 static_cast<MockAuthenticatorManager*>(&m_authenticatorManager)->setTestConfiguration(WTFMove(configuration));
137}
