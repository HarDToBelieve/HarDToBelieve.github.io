68std::optional<String> WebPaymentCoordinator::validatedPaymentNetwork(const String& paymentNetwork)
69{
70 if (!m_availablePaymentNetworks)
71 m_availablePaymentNetworks = platformAvailablePaymentNetworks();
72
73 auto result = m_availablePaymentNetworks->find(paymentNetwork);
74 if (result == m_availablePaymentNetworks->end())
75 return std::nullopt;
76 return *result;
77}

79bool WebPaymentCoordinator::canMakePayments()
80{
81 bool canMakePayments;
82 if (!sendSync(Messages::WebPaymentCoordinatorProxy::CanMakePayments(), Messages::WebPaymentCoordinatorProxy::CanMakePayments::Reply(canMakePayments)))

85 return canMakePayments;

98bool WebPaymentCoordinator::showPaymentUI(const URL& originatingURL, const Vector<URL>& linkIconURLs, const WebCore::ApplePaySessionPaymentRequest& paymentRequest)
99{
100 Vector<String> linkIconURLStrings;
101 for (const auto& linkIconURL : linkIconURLs)
102 linkIconURLStrings.append(linkIconURL.string());
103
104 bool result;
105 if (!sendSync(Messages::WebPaymentCoordinatorProxy::ShowPaymentUI(m_webPage.identifier(), originatingURL.string(), linkIconURLStrings, paymentRequest), Messages::WebPaymentCoordinatorProxy::ShowPaymentUI::Reply(result)))
106 return false;
107
108 return result;
109}
