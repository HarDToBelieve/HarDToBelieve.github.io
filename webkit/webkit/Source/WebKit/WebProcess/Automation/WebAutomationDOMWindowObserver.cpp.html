32namespace WebKit {
33
34WebAutomationDOMWindowObserver::WebAutomationDOMWindowObserver(WebCore::DOMWindow& window, WTF::Function<void(WebAutomationDOMWindowObserver&)>&& callback)
35 : m_window(makeWeakPtr(window))
36 , m_callback(WTFMove(callback))
37{
38 ASSERT(m_window->frame());
39 m_window->registerObserver(*this);
40}
41
42WebAutomationDOMWindowObserver::~WebAutomationDOMWindowObserver()
43{
44 if (m_window)
45 m_window->unregisterObserver(*this);
46}
47
48void WebAutomationDOMWindowObserver::willDestroyGlobalObjectInCachedFrame()
49{
50 Ref<WebAutomationDOMWindowObserver> protectedThis(*this);
51
52 if (!m_wasDetached) {
53 ASSERT(m_window && m_window->frame());
54 m_callback(*this);
55 }
56
57 ASSERT(m_window);
58 if (m_window)
59 m_window->unregisterObserver(*this);
60 m_window = nullptr;
61}
62
63void WebAutomationDOMWindowObserver::willDestroyGlobalObjectInFrame()
64{
65 Ref<WebAutomationDOMWindowObserver> protectedThis(*this);
66
67 if (!m_wasDetached) {
68 ASSERT(m_window && m_window->frame());
69 m_callback(*this);
70 }
71
72 ASSERT(m_window);
73 if (m_window)
74 m_window->unregisterObserver(*this);
75 m_window = nullptr;
76}
77
78void WebAutomationDOMWindowObserver::willDetachGlobalObjectFromFrame()
79{
80 ASSERT(!m_wasDetached);
81
82 Ref<WebAutomationDOMWindowObserver> protectedThis(*this);
83
84 m_wasDetached = true;
85
86 m_callback(*this);
87
88 ASSERT(m_window);
89 if (m_window)
90 m_window->unregisterObserver(*this);
91 m_window = nullptr;
92}
