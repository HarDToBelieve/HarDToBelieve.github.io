38static HashMap<uint64_t, WebDatabaseProvider*>& databaseProviders()
39{
40 static NeverDestroyed<HashMap<uint64_t, WebDatabaseProvider*>> databaseProviders;
41
42 return databaseProviders;
43}
44
45Ref<WebDatabaseProvider> WebDatabaseProvider::getOrCreate(uint64_t identifier)
46{
47 auto& slot = databaseProviders().add(identifier, nullptr).iterator->value;
48 if (slot)
49 return *slot;
50
51 Ref<WebDatabaseProvider> databaseProvider = adoptRef(*new WebDatabaseProvider(identifier));
52 slot = databaseProvider.ptr();
53
54 return databaseProvider;
55}
56
57WebDatabaseProvider::WebDatabaseProvider(uint64_t identifier)
58 : m_identifier(identifier)
59{
60}
61
62WebDatabaseProvider::~WebDatabaseProvider()
63{
64 ASSERT(databaseProviders().contains(m_identifier));
65
66 databaseProviders().remove(m_identifier);
67}

69WebCore::IDBClient::IDBConnectionToServer& WebDatabaseProvider::idbConnectionToServerForSession(PAL::SessionID)
70{
71 return WebProcess::singleton().ensureNetworkProcessConnection().idbConnectionToServer().coreConnectionToServer();
72}
