53static WebCore::IntRect screenRectOfContents(WebCore::Element* element)
54{
55 ASSERT(element);
56 if (!element)
57 return { };
58
59 if (element->renderer() && element->renderer()->hasLayer() && element->renderer()->enclosingLayer()->isComposited()) {
60 WebCore::FloatQuad contentsBox = static_cast<WebCore::FloatRect>(element->renderer()->enclosingLayer()->backing()->contentsBox());
61 contentsBox = element->renderer()->localToAbsoluteQuad(contentsBox);
62 return element->renderer()->view().frameView().contentsToScreen(contentsBox.enclosingBoundingBox());
63 }
64
65 return element->screenRect();
66}

87void WebFullScreenManager::videoControlsManagerDidChange()
88{
89#if PLATFORM(IOS_FAMILY) || (PLATFORM(MAC) && ENABLE(VIDEO_PRESENTATION_MODE))
90 LOG(Fullscreen, "WebFullScreenManager %p videoControlsManagerDidChange()", this);
91
92 auto* currentPlaybackControlsElement = m_page->playbackSessionManager().currentPlaybackControlsElement();
93 if (!m_element || !is<WebCore::HTMLVideoElement>(currentPlaybackControlsElement)) {
94 setPIPStandbyElement(nullptr);
95 return;
96 }
97
98 setPIPStandbyElement(downcast<WebCore::HTMLVideoElement>(currentPlaybackControlsElement));
99#endif
100}

102void WebFullScreenManager::setPIPStandbyElement(WebCore::HTMLVideoElement* pipStandbyElement)
103{
104#if ENABLE(VIDEO)
105 if (pipStandbyElement == m_pipStandbyElement)
106 return;
107
108 LOG(Fullscreen, "WebFullScreenManager %p setPIPStandbyElement() - old element %p, new element %p", this, m_pipStandbyElement.get(), pipStandbyElement);
109
110 if (m_pipStandbyElement)
111 m_pipStandbyElement->setVideoFullscreenStandby(false);
112
113 m_pipStandbyElement = pipStandbyElement;
114
115 if (m_pipStandbyElement)
116 m_pipStandbyElement->setVideoFullscreenStandby(true);
117#endif
118}

133void WebFullScreenManager::enterFullScreenForElement(WebCore::Element* element)
134{
135 LOG(Fullscreen, "WebFullScreenManager %p enterFullScreenForElement(%p)", this, element);
136
137 ASSERT(element);
138 if (!element)
139 return;
140 m_element = element;
141
142#if PLATFORM(IOS_FAMILY) || (PLATFORM(MAC) && ENABLE(VIDEO_PRESENTATION_MODE))
143 if (auto* currentPlaybackControlsElement = m_page->playbackSessionManager().currentPlaybackControlsElement())
144 currentPlaybackControlsElement->prepareForVideoFullscreenStandby();
145#endif
146
147 m_initialFrame = screenRectOfContents(m_element.get());
148 m_page->injectedBundleFullScreenClient().enterFullScreenForElement(m_page.get(), element, m_element->document().quirks().blocksReturnToFullscreenFromPictureInPictureQuirk());
149}

154 m_page->injectedBundleFullScreenClient().exitFullScreenForElement(m_page.get(), element);

157void WebFullScreenManager::willEnterFullScreen()
158{
159 LOG(Fullscreen, "WebFullScreenManager %p willEnterFullScreen() - element %p", this, m_element.get());
160 ASSERT(m_element);
161 if (!m_element)
162 return;
163
164 if (!m_element->document().fullscreenManager().willEnterFullscreen(*m_element)) {
165 close();
166 return;
167 }
168
169#if !PLATFORM(IOS_FAMILY)
170 m_page->hidePageBanners();
171#endif
172 m_element->document().updateLayout();
173 m_page->forceRepaintWithoutCallback();
174 m_finalFrame = screenRectOfContents(m_element.get());
175 m_page->injectedBundleFullScreenClient().beganEnterFullScreen(m_page.get(), m_initialFrame, m_finalFrame);
176}

178void WebFullScreenManager::didEnterFullScreen()
179{
180 LOG(Fullscreen, "WebFullScreenManager %p didEnterFullScreen() - element %p", this, m_element.get());
181 ASSERT(m_element);
182 if (!m_element)
183 return;
184
185 if (!m_element->document().fullscreenManager().didEnterFullscreen()) {
186 close();
187 return;
188 }
189
190#if PLATFORM(IOS_FAMILY) || (PLATFORM(MAC) && ENABLE(VIDEO_PRESENTATION_MODE))
191 auto* currentPlaybackControlsElement = m_page->playbackSessionManager().currentPlaybackControlsElement();
192 setPIPStandbyElement(is<HTMLVideoElement>(currentPlaybackControlsElement) ? downcast<HTMLVideoElement>(currentPlaybackControlsElement) : nullptr);
193#endif
194}

196void WebFullScreenManager::willExitFullScreen()
197{
198 LOG(Fullscreen, "WebFullScreenManager %p willExitFullScreen() - element %p", this, m_element.get());
199 ASSERT(m_element);
200 if (!m_element)
201 return;
202
203#if ENABLE(VIDEO)
204 setPIPStandbyElement(nullptr);
205#endif
206
207 m_finalFrame = screenRectOfContents(m_element.get());
208 if (!m_element->document().fullscreenManager().willExitFullscreen()) {
209 close();
210 return;
211 }
212#if !PLATFORM(IOS_FAMILY)
213 m_page->showPageBanners();
214#endif
215 m_page->injectedBundleFullScreenClient().beganExitFullScreen(m_page.get(), m_finalFrame, m_initialFrame);
216}

218void WebFullScreenManager::didExitFullScreen()
219{
220 LOG(Fullscreen, "WebFullScreenManager %p didExitFullScreen() - element %p", this, m_element.get());
221 ASSERT(m_element);
222 if (!m_element)
223 return;
224
225 setFullscreenInsets(WebCore::FloatBoxExtent());
226 setFullscreenAutoHideDuration(0_s);
227 m_element->document().fullscreenManager().didExitFullscreen();
228}

238void WebFullScreenManager::requestEnterFullScreen()
239{
240 ASSERT(m_element);
241 if (!m_element)
242 return;
243
244 WebCore::UserGestureIndicator gestureIndicator(WebCore::ProcessingUserGesture);
245 m_element->document().fullscreenManager().requestFullscreenForElement(m_element.get(), WebCore::FullscreenManager::ExemptIFrameAllowFullscreenRequirement);
246}
247
248void WebFullScreenManager::requestExitFullScreen()
249{
250 ASSERT(m_element);
251 if (!m_element) {
252 close();
253 return;
254 }
255
256 auto& topDocument = m_element->document().topDocument();
257 if (!topDocument.fullscreenManager().fullscreenElement()) {
258 close();
259 return;
260 }
261 m_element->document().fullscreenManager().cancelFullscreen();
262}
