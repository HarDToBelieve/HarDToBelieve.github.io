95static void languagesChanged(void* context)
96{

98}

100GPUProcessConnection::GPUProcessConnection(IPC::Connection::Identifier connectionIdentifier)
101 : m_connection(IPC::Connection::createClientConnection(connectionIdentifier, *this))
102{
103 m_connection->open();
104
105 addLanguageChangeObserver(this, languagesChanged);
106
107 if (WebProcess::singleton().shouldUseRemoteRenderingFor(RenderingPurpose::MediaPainting)) {
108#if ENABLE(VP9)
109 enableVP9Decoders(PlatformMediaSessionManager::shouldEnableVP8Decoder(), PlatformMediaSessionManager::shouldEnableVP9Decoder(), PlatformMediaSessionManager::shouldEnableVP9SWDecoder());
110#endif
111 }
112}
113
114GPUProcessConnection::~GPUProcessConnection()
115{
116 m_connection->invalidate();
117#if PLATFORM(COCOA) && ENABLE(WEB_AUDIO)
118 if (m_audioSourceProviderManager)
119 m_audioSourceProviderManager->stopListeningForIPC();
120#endif
121 removeLanguageChangeObserver(this);
122}

124void GPUProcessConnection::didClose(IPC::Connection&)
125{
126 auto protector = makeRef(*this);
127 WebProcess::singleton().gpuProcessConnectionClosed(*this);
128
129 auto clients = m_clients;
130 for (auto& client : clients)
131 client.gpuProcessConnectionDidClose(*this);
132}

161bool GPUProcessConnection::dispatchMessage(IPC::Connection& connection, IPC::Decoder& decoder)
162{
163 if (decoder.messageReceiverName() == Messages::MediaPlayerPrivateRemote::messageReceiverName()) {
164 WebProcess::singleton().supplement<RemoteMediaPlayerManager>()->didReceivePlayerMessage(connection, decoder);
165 return true;
166 }
167
168#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
169 if (decoder.messageReceiverName() == Messages::UserMediaCaptureManager::messageReceiverName()) {
170 if (auto* captureManager = WebProcess::singleton().supplement<UserMediaCaptureManager>())
171 captureManager->didReceiveMessageFromGPUProcess(connection, decoder);
172 return true;
173 }
174
175 if (decoder.messageReceiverName() == Messages::SampleBufferDisplayLayer::messageReceiverName()) {
176 sampleBufferDisplayLayerManager().didReceiveLayerMessage(connection, decoder);
177 return true;
178 }
179#endif // PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
180
181#if ENABLE(ENCRYPTED_MEDIA)
182 if (decoder.messageReceiverName() == Messages::RemoteCDMInstanceSession::messageReceiverName()) {
183 WebProcess::singleton().cdmFactory().didReceiveSessionMessage(connection, decoder);
184 return true;
185 }
186#endif
187 if (messageReceiverMap().dispatchMessage(connection, decoder))
188 return true;
189
190 // Skip messages intended for already removed messageReceiverMap() destinations.
191#if ENABLE(WEBGL)
192 if (decoder.messageReceiverName() == Messages::RemoteGraphicsContextGLProxy::messageReceiverName()) {
193 RELEASE_LOG_ERROR(WebGL, "The RemoteGraphicsContextGLProxy object has beed destroyed");
194 return true;
195 }
196#endif
197
198#if USE(AUDIO_SESSION)
199 if (decoder.messageReceiverName() == Messages::RemoteAudioSession::messageReceiverName()) {
200 RELEASE_LOG_ERROR(Media, "The RemoteAudioSession object has beed destroyed");
201 return true;
202 }
203#endif
204
205#if ENABLE(MEDIA_SOURCE)
206 if (decoder.messageReceiverName() == Messages::MediaSourcePrivateRemote::messageReceiverName()) {
207 RELEASE_LOG_ERROR(Media, "The MediaSourcePrivateRemote object has beed destroyed");
208 return true;
209 }
210
211 if (decoder.messageReceiverName() == Messages::SourceBufferPrivateRemote::messageReceiverName()) {
212 RELEASE_LOG_ERROR(Media, "The SourceBufferPrivateRemote object has beed destroyed");
213 return true;
214 }
215#endif
216
217 if (decoder.messageReceiverName() == Messages::RemoteAudioHardwareListener::messageReceiverName()) {
218 RELEASE_LOG_ERROR(Media, "The RemoteAudioHardwareListener object has beed destroyed");
219 return true;
220 }
221
222 if (decoder.messageReceiverName() == Messages::RemoteRemoteCommandListener::messageReceiverName()) {
223 RELEASE_LOG_ERROR(Media, "The RemoteRemoteCommandListener object has beed destroyed");
224 return true;
225 }
226
227 return false;
228}

293void GPUProcessConnection::updateMediaConfiguration()
294{
295#if PLATFORM(COCOA)
296 bool settingsChanged = false;
297
298 if (m_mediaOverridesForTesting.systemHasAC != SystemBatteryStatusTestingOverrides::singleton().hasAC() || m_mediaOverridesForTesting.systemHasBattery != SystemBatteryStatusTestingOverrides::singleton().hasBattery())
299 settingsChanged = true;
300
301#if ENABLE(VP9)
302 if (m_mediaOverridesForTesting.vp9HardwareDecoderDisabled != VP9TestingOverrides::singleton().hardwareDecoderDisabled() || m_mediaOverridesForTesting.vp9ScreenSizeAndScale != VP9TestingOverrides::singleton().vp9ScreenSizeAndScale())
303 settingsChanged = true;
304#endif
305
306 if (!settingsChanged)
307 return;
308
309 m_mediaOverridesForTesting = {
310 .systemHasAC = SystemBatteryStatusTestingOverrides::singleton().hasAC(),
311 .systemHasBattery = SystemBatteryStatusTestingOverrides::singleton().hasBattery(),
312
313#if ENABLE(VP9)
314 .vp9HardwareDecoderDisabled = VP9TestingOverrides::singleton().hardwareDecoderDisabled(),
315 .vp9ScreenSizeAndScale = VP9TestingOverrides::singleton().vp9ScreenSizeAndScale(),
316#endif
317 };
318
319 connection().send(Messages::GPUConnectionToWebProcess::SetMediaOverridesForTesting(m_mediaOverridesForTesting), { });
320#endif
321}
