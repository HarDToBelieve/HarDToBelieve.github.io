59class GPUProcessConnection : public RefCounted<GPUProcessConnection>, public IPC::Connection::Client {
60public:
61 static Ref<GPUProcessConnection> create(IPC::Connection::Identifier connectionIdentifier)
62 {
63 return adoptRef(*new GPUProcessConnection(connectionIdentifier));
64 }
65 ~GPUProcessConnection();
66
67 IPC::Connection& connection() { return m_connection.get(); }
68 IPC::MessageReceiverMap& messageReceiverMap() { return m_messageReceiverMap; }
69
70#if HAVE(AUDIT_TOKEN)
71 void setAuditToken(std::optional<audit_token_t> auditToken) { m_auditToken = auditToken; }
72 std::optional<audit_token_t> auditToken() const { return m_auditToken; }
73#endif
74#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
75 SampleBufferDisplayLayerManager& sampleBufferDisplayLayerManager();
76#endif
77
78 RemoteMediaPlayerManager& mediaPlayerManager();
79
80#if PLATFORM(COCOA) && ENABLE(WEB_AUDIO)
81 RemoteAudioSourceProviderManager& audioSourceProviderManager();
82#endif
83
84 void updateMediaConfiguration();
85
86#if ENABLE(VP9)
87 void enableVP9Decoders(bool enableVP8Decoder, bool enableVP9Decoder, bool enableVP9SWDecoder);
88
89 bool isVP8DecoderEnabled() const { return m_enableVP8Decoder; }
90 bool isVP9DecoderEnabled() const { return m_enableVP9Decoder; }
91 bool isVPSWDecoderEnabled() const { return m_enableVP9SWDecoder; }
92#endif
93
94#if HAVE(VISIBILITY_PROPAGATION_VIEW)
95 void createVisibilityPropagationContextForPage(WebPage&);
96 void destroyVisibilityPropagationContextForPage(WebPage&);
97#endif
98
99 void configureLoggingChannel(const String&, WTFLogChannelState, WTFLogLevel);
100
101 class Client : public CanMakeWeakPtr<Client> {
102 public:
103 virtual ~Client() = default;
104
105 virtual void gpuProcessConnectionDidClose(GPUProcessConnection&) { }
106 };
107 void addClient(const Client& client) { m_clients.add(client); }
108 void removeClient(const Client& client) { m_clients.remove(client); }
109
110private:
111 GPUProcessConnection(IPC::Connection::Identifier);
112
113 // IPC::Connection::Client
114 void didClose(IPC::Connection&) override;
115 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) final;
116 bool didReceiveSyncMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&) final;
117 void didReceiveInvalidMessage(IPC::Connection&, IPC::MessageName) override;
118
119 bool dispatchMessage(IPC::Connection&, IPC::Decoder&);
120 bool dispatchSyncMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&);
121
122 void didReceiveRemoteCommand(WebCore::PlatformMediaSession::RemoteControlCommandType, const WebCore::PlatformMediaSession::RemoteCommandArgument&);
123
124#if ENABLE(ROUTING_ARBITRATION)
125 void beginRoutingArbitrationWithCategory(WebCore::AudioSession::CategoryType, WebCore::AudioSessionRoutingArbitrationClient::ArbitrationCallback&&);
126 void endRoutingArbitration();
127#endif
128
129 // The connection from the web process to the GPU process.
130 Ref<IPC::Connection> m_connection;
131 IPC::MessageReceiverMap m_messageReceiverMap;
132
133#if HAVE(AUDIT_TOKEN)
134 std::optional<audit_token_t> m_auditToken;
135#endif
136#if PLATFORM(COCOA) && ENABLE(MEDIA_STREAM)
137 std::unique_ptr<SampleBufferDisplayLayerManager> m_sampleBufferDisplayLayerManager;
138#endif
139#if PLATFORM(COCOA) && ENABLE(WEB_AUDIO)
140 RefPtr<RemoteAudioSourceProviderManager> m_audioSourceProviderManager;
141#endif
142#if ENABLE(VP9)
143 bool m_enableVP8Decoder { false };
144 bool m_enableVP9Decoder { false };
145 bool m_enableVP9SWDecoder { false };
146#endif
147
148#if PLATFORM(COCOA)
149 MediaOverridesForTesting m_mediaOverridesForTesting;
150#endif
151
152 WeakHashSet<Client> m_clients;
153};
