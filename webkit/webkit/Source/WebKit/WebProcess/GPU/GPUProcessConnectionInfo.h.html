26#pragma once
27
28#if ENABLE(GPU_PROCESS)
29
30namespace WebKit {
31
32struct GPUProcessConnectionInfo {
33 IPC::Attachment connection;
34#if HAVE(AUDIT_TOKEN)
35 std::optional<audit_token_t> auditToken;
36#endif
37
38 IPC::Connection::Identifier identifier() const
39 {
40#if USE(UNIX_DOMAIN_SOCKETS)
41 return IPC::Connection::Identifier(connection.fileDescriptor());
42#elif OS(DARWIN)
43 return IPC::Connection::Identifier(connection.port());
44#elif OS(WINDOWS)
45 return IPC::Connection::Identifier(connection.handle());
46#else
47 ASSERT_NOT_REACHED();
48 return IPC::Connection::Identifier();
49#endif
50 }
51
52 IPC::Connection::Identifier releaseIdentifier()
53 {
54#if USE(UNIX_DOMAIN_SOCKETS)
55 auto returnValue = IPC::Connection::Identifier(connection.releaseFileDescriptor());
56#else
57 auto returnValue = identifier();
58#endif
59 connection = { };
60 return returnValue;
61 }
62
63 void encode(IPC::Encoder& encoder) const
64 {
65 encoder << connection;
66#if HAVE(AUDIT_TOKEN)
67 encoder << auditToken;
68#endif
69 }
70
71 static WARN_UNUSED_RETURN bool decode(IPC::Decoder& decoder, GPUProcessConnectionInfo& info)
72 {
73 if (!decoder.decode(info.connection))
74 return false;
75#if HAVE(AUDIT_TOKEN)
76 if (!decoder.decode(info.auditToken))
77 return false;
78#endif
79 return true;
80 }
81};
82
83} // namespace WebKit
84
85#endif // ENABLE(GPU_PROCESS)
