31namespace WebKit {
32using namespace WebCore;
33
34size_t DisplayListWriterHandle::advance(size_t amount)
35{
36 m_writableOffset = CheckedSize { m_writableOffset } + amount;
37 return CheckedSize { header().unreadBytes.exchangeAdd(amount) } + amount;
38}
39
40size_t DisplayListWriterHandle::availableCapacity() const
41{
42 if (UNLIKELY(sharedMemory().size() < writableOffset()))
43 RELEASE_ASSERT_NOT_REACHED();
44
45 return sharedMemory().size() - writableOffset();
46}
47
48DisplayList::ItemBufferHandle DisplayListWriterHandle::createHandle() const
49{
50 return {
51 identifier(),
52 data() + writableOffset(),
53 availableCapacity()
54 };
55}
56
57bool DisplayListWriterHandle::moveWritableOffsetToStartIfPossible()
58{
59 if (m_writableOffset <= SharedDisplayListHandle::headerSize()) {
60 RELEASE_ASSERT(m_writableOffset == SharedDisplayListHandle::headerSize());
61 return true;
62 }
63
64 if (unreadBytes())
65 return false;
66
67 m_writableOffset = SharedDisplayListHandle::headerSize();
68 return true;
69}
