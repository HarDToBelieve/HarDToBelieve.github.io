51IntSize ImageBufferShareableBitmapBackend::calculateSafeBackendSize(const Parameters& parameters)
52{
53 IntSize backendSize = calculateBackendSize(parameters);
54 if (backendSize.isEmpty())
55 return { };
56
57 CheckedUint32 bytesPerRow = ShareableBitmap::calculateBytesPerRow(backendSize, configuration(parameters));
58 if (bytesPerRow.hasOverflowed())
59 return { };
60
61 CheckedSize numBytes = CheckedUint32(backendSize.height()) * bytesPerRow;
62 if (numBytes.hasOverflowed())
63 return { };
64
65 return backendSize;
66}
67
68unsigned ImageBufferShareableBitmapBackend::calculateBytesPerRow(const Parameters& parameters, const IntSize& backendSize)
69{
70 ASSERT(!backendSize.isEmpty());
71 return ShareableBitmap::calculateBytesPerRow(backendSize, configuration(parameters));
72}

74size_t ImageBufferShareableBitmapBackend::calculateMemoryCost(const Parameters& parameters)
75{
76 IntSize backendSize = calculateBackendSize(parameters);
77 return ImageBufferBackend::calculateMemoryCost(backendSize, calculateBytesPerRow(parameters, backendSize));
78}

80std::unique_ptr<ImageBufferShareableBitmapBackend> ImageBufferShareableBitmapBackend::create(const Parameters& parameters, const HostWindow*)
81{
82 ASSERT(parameters.pixelFormat == PixelFormat::BGRA8);
83
84 IntSize backendSize = calculateSafeBackendSize(parameters);
85 if (backendSize.isEmpty())
86 return nullptr;
87
88 auto bitmap = ShareableBitmap::createShareable(backendSize, configuration(parameters));
89 if (!bitmap)
90 return nullptr;
91
92 auto context = bitmap->createGraphicsContext();
93 if (!context)
94 return nullptr;
95
96 return makeUnique<ImageBufferShareableBitmapBackend>(parameters, WTFMove(bitmap), WTFMove(context));
97}
98
99std::unique_ptr<ImageBufferShareableBitmapBackend> ImageBufferShareableBitmapBackend::create(const Parameters& parameters, ImageBufferBackendHandle handle)
100{
101 if (!WTF::holds_alternative<ShareableBitmap::Handle>(handle)) {
102 ASSERT_NOT_REACHED();
103 return nullptr;
104 }
105
106 auto bitmap = ShareableBitmap::create(WTFMove(WTF::get<ShareableBitmap::Handle>(handle)));
107 if (!bitmap)
108 return nullptr;
109
110 auto context = bitmap->createGraphicsContext();
111 if (!context)
112 return nullptr;
113
114 return makeUnique<ImageBufferShareableBitmapBackend>(parameters, WTFMove(bitmap), WTFMove(context));
115}

117ImageBufferShareableBitmapBackend::ImageBufferShareableBitmapBackend(const Parameters& parameters, RefPtr<ShareableBitmap>&& bitmap, std::unique_ptr<GraphicsContext>&& context)
118 : PlatformImageBufferBackend(parameters)
119 , m_bitmap(WTFMove(bitmap))
120 , m_context(WTFMove(context))
121{
122 // ShareableBitmap ensures that the coordinate space in the context that we're adopting
123 // has a top-left origin, so we don't ever need to flip here, so we don't call setupContext().
124 // However, ShareableBitmap does not have a notion of scale, so we must apply the device
125 // scale factor to the context ourselves.
126 m_context->applyDeviceScaleFactor(resolutionScale());
127}

134}
