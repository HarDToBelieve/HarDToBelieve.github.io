49template<typename BackendType>
50class RemoteImageBufferProxy : public WebCore::DisplayList::ImageBuffer<BackendType>, public WebCore::DisplayList::Recorder::Delegate, public WebCore::DisplayList::ItemBufferWritingClient {
51 using BaseDisplayListImageBuffer = WebCore::DisplayList::ImageBuffer<BackendType>;
52 using BaseDisplayListImageBuffer::m_backend;
53 using BaseDisplayListImageBuffer::m_drawingContext;
54 using BaseDisplayListImageBuffer::m_renderingResourceIdentifier;
55 using BaseDisplayListImageBuffer::resolutionScale;
56
57public:
58 static RefPtr<RemoteImageBufferProxy> create(const WebCore::FloatSize& size, float resolutionScale, const WebCore::DestinationColorSpace& colorSpace, WebCore::PixelFormat pixelFormat, RemoteRenderingBackendProxy& remoteRenderingBackendProxy)
59 {
60 auto parameters = WebCore::ImageBufferBackend::Parameters { size, resolutionScale, colorSpace, pixelFormat };
61 if (BackendType::calculateSafeBackendSize(parameters).isEmpty())
62 return nullptr;
63 return adoptRef(new RemoteImageBufferProxy(parameters, remoteRenderingBackendProxy));
64 }
65
66 ~RemoteImageBufferProxy()
67 {
68 if (!m_remoteRenderingBackendProxy) {
69 clearDisplayList();
70 return;
71 }
72
73 flushDrawingContext();
74 m_remoteRenderingBackendProxy->remoteResourceCacheProxy().releaseImageBuffer(m_renderingResourceIdentifier);
75 m_remoteRenderingBackendProxy->releaseRemoteResource(m_renderingResourceIdentifier);
76 }
77
78 ImageBufferBackendHandle createImageBufferBackendHandle()
79 {
80 ensureBackendCreated();
81 return m_backend->createImageBufferBackendHandle();
82 }
83
84 WebCore::DisplayList::FlushIdentifier lastSentFlushIdentifier() const { return m_sentFlushIdentifier; }
85
86 void waitForDidFlushOnSecondaryThread(WebCore::DisplayList::FlushIdentifier targetFlushIdentifier)
87 {
88 ASSERT(!isMainRunLoop());
89 Locker locker { m_receivedFlushIdentifierLock };
90 m_receivedFlushIdentifierChangedCondition.wait(m_receivedFlushIdentifierLock, [&] {
91 assertIsHeld(m_receivedFlushIdentifierLock);
92 return m_receivedFlushIdentifier == targetFlushIdentifier;
93 });
94
95 // Nothing should have sent more drawing commands to the GPU process
96 // while waiting for this ImageBuffer to be flushed.
97 ASSERT(m_sentFlushIdentifier == targetFlushIdentifier);
98 }
99
100protected:
101 RemoteImageBufferProxy(const WebCore::ImageBufferBackend::Parameters& parameters, RemoteRenderingBackendProxy& remoteRenderingBackendProxy)
102 : BaseDisplayListImageBuffer(parameters, this)
103 , m_remoteRenderingBackendProxy(makeWeakPtr(remoteRenderingBackendProxy))
104 {
105 ASSERT(m_remoteRenderingBackendProxy);
106 m_remoteRenderingBackendProxy->remoteResourceCacheProxy().cacheImageBuffer(*this);
107
108 m_drawingContext.displayList().setItemBufferWritingClient(this);
109 m_drawingContext.displayList().setItemBufferReadingClient(nullptr);
110 m_drawingContext.displayList().setTracksDrawingItemExtents(false);
111 }
112
113 WebCore::RenderingMode renderingMode() const override { return BaseDisplayListImageBuffer::renderingMode(); }
114
115 // It is safe to access m_receivedFlushIdentifier from the main thread without locking since it
116 // only gets modified on the main thread.
117 bool hasPendingFlush() const WTF_IGNORES_THREAD_SAFETY_ANALYSIS
118 {
119 ASSERT(isMainRunLoop());
120 return m_sentFlushIdentifier != m_receivedFlushIdentifier;
121 }
122
123 void didFlush(WebCore::DisplayList::FlushIdentifier flushIdentifier) override
124 {
125 ASSERT(isMainRunLoop());
126 Locker locker { m_receivedFlushIdentifierLock };
127 m_receivedFlushIdentifier = flushIdentifier;
128 m_receivedFlushIdentifierChangedCondition.notifyAll();
129 }
130
131 void waitForDidFlushWithTimeout()
132 {
133 if (!m_remoteRenderingBackendProxy)
134 return;
135
136 // Wait for our DisplayList to be flushed but do not hang.
137 static constexpr unsigned maximumNumberOfTimeouts = 3;
138 unsigned numberOfTimeouts = 0;
139#if !LOG_DISABLED
140 auto startTime = MonotonicTime::now();
141#endif
142 LOG_WITH_STREAM(SharedDisplayLists, stream << "Waiting for Flush{" << m_sentFlushIdentifier << "} in Image(" << m_renderingResourceIdentifier << ")");
143 while (numberOfTimeouts < maximumNumberOfTimeouts && hasPendingFlush()) {
144 if (!m_remoteRenderingBackendProxy->waitForDidFlush())
145 ++numberOfTimeouts;
146 }
147 LOG_WITH_STREAM(SharedDisplayLists, stream << "Done waiting: " << MonotonicTime::now() - startTime << "; " << numberOfTimeouts << " timeout(s)");
148 }
149
150 WebCore::ImageBufferBackend* ensureBackendCreated() const override
151 {
152 if (!m_remoteRenderingBackendProxy)
153 return m_backend.get();
154
155 static constexpr unsigned maximumTimeoutOrFailureCount = 3;
156 unsigned numberOfTimeoutsOrFailures = 0;
157 while (!m_backend && numberOfTimeoutsOrFailures < maximumTimeoutOrFailureCount) {
158 if (m_remoteRenderingBackendProxy->waitForDidCreateImageBufferBackend() == RemoteRenderingBackendProxy::DidReceiveBackendCreationResult::TimeoutOrIPCFailure)
159 ++numberOfTimeoutsOrFailures;
160 }
161 return m_backend.get();
162 }

182 RefPtr<WebCore::NativeImage> copyNativeImage(WebCore::BackingStoreCopy = WebCore::BackingStoreCopy::CopyBackingStore) const override
183 {
184 if (UNLIKELY(!m_remoteRenderingBackendProxy))
185 return { };
186 const_cast<RemoteImageBufferProxy*>(this)->flushDrawingContext();
187 auto bitmap = m_remoteRenderingBackendProxy->getShareableBitmap(m_renderingResourceIdentifier, WebCore::PreserveResolution::Yes);
188 if (!bitmap)
189 return { };
190 return WebCore::NativeImage::create(bitmap->createPlatformImage());
191 }

193 RefPtr<WebCore::Image> copyImage(WebCore::BackingStoreCopy = WebCore::BackingStoreCopy::CopyBackingStore, WebCore::PreserveResolution preserveResolution = WebCore::PreserveResolution::No) const override
194 {
195 if (UNLIKELY(!m_remoteRenderingBackendProxy))
196 return { };
197 const_cast<RemoteImageBufferProxy*>(this)->flushDrawingContext();
198 auto bitmap = m_remoteRenderingBackendProxy->getShareableBitmap(m_renderingResourceIdentifier, preserveResolution);
199 if (!bitmap)
200 return { };
201 return bitmap->createImage();
202 }

204 std::optional<WebCore::PixelBuffer> getPixelBuffer(const WebCore::PixelBufferFormat& destinationFormat, const WebCore::IntRect& srcRect) const override
205 {
206 if (UNLIKELY(!m_remoteRenderingBackendProxy))
207 return std::nullopt;
208
209 auto pixelBuffer = WebCore::PixelBuffer::tryCreate(destinationFormat, srcRect.size());
210 if (!pixelBuffer)
211 return std::nullopt;
212 size_t dataSize = pixelBuffer->data().byteLength();
213
214 IPC::Timeout timeout = 5_s;
215 SharedMemory* sharedMemory = m_remoteRenderingBackendProxy->sharedMemoryForGetPixelBuffer(dataSize, timeout);
216 if (!sharedMemory)
217 return std::nullopt;
218
219 auto& mutableThis = const_cast<RemoteImageBufferProxy&>(*this);
220 mutableThis.m_drawingContext.recorder().getPixelBuffer(destinationFormat, srcRect);
221 mutableThis.flushDrawingContextAsync();
222
223 if (m_remoteRenderingBackendProxy->waitForGetPixelBufferToComplete(timeout))
224 memcpy(pixelBuffer->data().data(), sharedMemory->data(), dataSize);
225 else
226 memset(pixelBuffer->data().data(), 0, dataSize);
227 return pixelBuffer;
228 }
229
230 void putPixelBuffer(const WebCore::PixelBuffer& pixelBuffer, const WebCore::IntRect& srcRect, const WebCore::IntPoint& destPoint = { }, WebCore::AlphaPremultiplication destFormat = WebCore::AlphaPremultiplication::Premultiplied) override
231 {
232 // The math inside PixelBuffer::create() doesn't agree with the math inside ImageBufferBackend::putPixelBuffer() about how m_resolutionScale interacts with the data in the ImageBuffer.
233 // This means that putPixelBuffer() is only called when resolutionScale() == 1.
234 ASSERT(resolutionScale() == 1);
235 m_drawingContext.recorder().putPixelBuffer(pixelBuffer, srcRect, destPoint, destFormat);
236 }

238 bool prefersPreparationForDisplay() override { return true; }
239
240 void flushContext() override
241 {
242 flushDrawingContext();
243 m_backend->flushContext();
244 }
245
246 void flushDrawingContext() override
247 {
248 if (UNLIKELY(!m_remoteRenderingBackendProxy))
249 return;
250
251 TraceScope tracingScope(FlushRemoteImageBufferStart, FlushRemoteImageBufferEnd);
252 flushDrawingContextAsync();
253 waitForDidFlushWithTimeout();
254 }
255
256 void flushDrawingContextAsync() override
257 {
258 if (UNLIKELY(!m_remoteRenderingBackendProxy))
259 return;
260
261 if (!m_drawingContext.displayList().isEmpty()) {
262 m_sentFlushIdentifier = WebCore::DisplayList::FlushIdentifier::generate();
263 m_drawingContext.recorder().flushContext(m_sentFlushIdentifier);
264 }
265
266 m_remoteRenderingBackendProxy->sendDeferredWakeupMessageIfNeeded();
267 clearDisplayList();
268 }

276 bool isCachedImageBuffer(const WebCore::ImageBuffer& imageBuffer) const override
277 {
278 if (!m_remoteRenderingBackendProxy)
279 return false;
280 auto cachedImageBuffer = m_remoteRenderingBackendProxy->remoteResourceCacheProxy().cachedImageBuffer(imageBuffer.renderingResourceIdentifier());
281 ASSERT(!cachedImageBuffer || cachedImageBuffer == &imageBuffer);
282 return cachedImageBuffer;
283 }

332 RefPtr<WebCore::SharedBuffer> encodeItemOutOfLine(const WebCore::DisplayList::DisplayListItem& item) const final
333 {
334 return WTF::visit([](const auto& displayListItem) -> RefPtr<WebCore::SharedBuffer> {
335 using DisplayListItemType = typename WTF::RemoveCVAndReference<decltype(displayListItem)>::type;
336 if constexpr (!DisplayListItemType::isInlineItem)
337 return IPC::Encoder::encodeSingleObject<DisplayListItemType>(displayListItem);
338 RELEASE_ASSERT_NOT_REACHED();
339 return nullptr;
340 }, item);
341 }
342
343 std::unique_ptr<WebCore::ThreadSafeImageBufferFlusher> createFlusher() override
344 {
345 return WTF::makeUnique<ThreadSafeRemoteImageBufferFlusher<BackendType>>(*this);
346 }
347
348 WebCore::DisplayList::FlushIdentifier m_sentFlushIdentifier;
349 Lock m_receivedFlushIdentifierLock;
350 Condition m_receivedFlushIdentifierChangedCondition;
351 WebCore::DisplayList::FlushIdentifier m_receivedFlushIdentifier WTF_GUARDED_BY_LOCK(m_receivedFlushIdentifierLock); // Only modified on the main thread but may get queried on a secondary thread.
352 WeakPtr<RemoteRenderingBackendProxy> m_remoteRenderingBackendProxy;
353};
354
355template<typename BackendType>
356class ThreadSafeRemoteImageBufferFlusher final : public WebCore::ThreadSafeImageBufferFlusher {
357 WTF_MAKE_FAST_ALLOCATED;
358public:
359 ThreadSafeRemoteImageBufferFlusher(RemoteImageBufferProxy<BackendType>& imageBuffer)
360 : m_imageBuffer(imageBuffer)
361 , m_targetFlushIdentifier(imageBuffer.lastSentFlushIdentifier())
362 {
363 }
364
365 void flush() final
366 {
367 m_imageBuffer->waitForDidFlushOnSecondaryThread(m_targetFlushIdentifier);
368 }
369
370private:
371 Ref<RemoteImageBufferProxy<BackendType>> m_imageBuffer;
372 WebCore::DisplayList::FlushIdentifier m_targetFlushIdentifier;
373};
