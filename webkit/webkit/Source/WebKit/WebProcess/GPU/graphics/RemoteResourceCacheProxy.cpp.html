37RemoteResourceCacheProxy::RemoteResourceCacheProxy(RemoteRenderingBackendProxy& remoteRenderingBackendProxy)
38 : m_remoteRenderingBackendProxy(remoteRenderingBackendProxy)

40}

42RemoteResourceCacheProxy::~RemoteResourceCacheProxy()
43{
44 for (auto& image : m_nativeImages.values())
45 image->removeObserver(*this);

48void RemoteResourceCacheProxy::cacheImageBuffer(WebCore::ImageBuffer& imageBuffer)
49{
50 auto addResult = m_imageBuffers.add(imageBuffer.renderingResourceIdentifier(), makeWeakPtr(imageBuffer));
51 ASSERT_UNUSED(addResult, addResult.isNewEntry);
52}
53
54ImageBuffer* RemoteResourceCacheProxy::cachedImageBuffer(RenderingResourceIdentifier renderingResourceIdentifier)
55{
56 return m_imageBuffers.get(renderingResourceIdentifier).get();
57}
58
59void RemoteResourceCacheProxy::releaseImageBuffer(RenderingResourceIdentifier renderingResourceIdentifier)
60{
61 bool found = m_imageBuffers.remove(renderingResourceIdentifier);
62 ASSERT_UNUSED(found, found);
63}

65inline static RefPtr<ShareableBitmap> createShareableBitmapFromNativeImage(NativeImage& image)
66{
67 auto imageSize = image.size();
68
69 auto bitmap = ShareableBitmap::createShareable(image.size(), { });
70 if (!bitmap)
71 return nullptr;
72
73 auto context = bitmap->createGraphicsContext();
74 if (!context)
75 return nullptr;
76
77 context->drawNativeImage(image, imageSize, FloatRect({ }, imageSize), FloatRect({ }, imageSize), { WebCore::CompositeOperator::Copy });
78 return bitmap;
79}
80
81void RemoteResourceCacheProxy::cacheNativeImage(NativeImage& image)
82{
83 if (m_nativeImages.contains(image.renderingResourceIdentifier()))
84 return;
85
86 auto bitmap = createShareableBitmapFromNativeImage(image);
87 if (!bitmap)
88 return;
89
90 ShareableBitmap::Handle handle;
91 bitmap->createHandle(handle);
92 if (handle.isNull())
93 return;
94
95 m_nativeImages.add(image.renderingResourceIdentifier(), makeWeakPtr(image));
96
97 // Set itself as an observer to NativeImage, so releaseNativeImage()
98 // gets called when NativeImage is being deleleted.
99 image.addObserver(*this);
100
101 // Tell the GPU process to cache this resource.
102 m_remoteRenderingBackendProxy.cacheNativeImage(handle, image.renderingResourceIdentifier());
103}

105void RemoteResourceCacheProxy::cacheFont(Font& font)
106{
107 auto result = m_fontIdentifierToLastRenderingUpdateVersionMap.ensure(font.renderingResourceIdentifier(), [&] {
108 return 0;
109 });
110 auto& lastVersion = result.iterator->value;
111 if (lastVersion != m_currentRenderingUpdateVersion) {
112 lastVersion = m_currentRenderingUpdateVersion;
113 ++m_numberOfFontsUsedInCurrentRenderingUpdate;
114 }
115 if (result.isNewEntry)
116 m_remoteRenderingBackendProxy.cacheFont(makeRef(font));
117}
118
119void RemoteResourceCacheProxy::releaseNativeImage(RenderingResourceIdentifier renderingResourceIdentifier)
120{
121 if (!m_nativeImages.remove(renderingResourceIdentifier))
122 return;
123
124 // Tell the GPU process to remove this resource.
125 m_remoteRenderingBackendProxy.releaseRemoteResource(renderingResourceIdentifier);
126}

128void RemoteResourceCacheProxy::didFinalizeRenderingUpdate()
129{
130 static constexpr unsigned minimumRenderingUpdateCountToKeepFontAlive = 4;
131 static constexpr double minimumFractionOfUnusedFontCountToTriggerRemoval = 0.25;
132 static constexpr unsigned maximumUnusedFontCountToSkipRemoval = 0;
133
134 unsigned totalFontCount = m_fontIdentifierToLastRenderingUpdateVersionMap.size();
135 RELEASE_ASSERT(m_numberOfFontsUsedInCurrentRenderingUpdate <= totalFontCount);
136 unsigned unusedFontCount = totalFontCount - m_numberOfFontsUsedInCurrentRenderingUpdate;
137 if (unusedFontCount < minimumFractionOfUnusedFontCountToTriggerRemoval * totalFontCount && unusedFontCount <= maximumUnusedFontCountToSkipRemoval)
138 return;
139
140 for (auto& item : m_fontIdentifierToLastRenderingUpdateVersionMap) {
141 if (m_currentRenderingUpdateVersion - item.value >= minimumRenderingUpdateCountToKeepFontAlive)
142 m_remoteRenderingBackendProxy.releaseRemoteResource(item.key);
143 }
144
145 ++m_currentRenderingUpdateVersion;
146 m_numberOfFontsUsedInCurrentRenderingUpdate = 0;
147}
148
149void RemoteResourceCacheProxy::remoteResourceCacheWasDestroyed()
150{
151 for (auto& imageBuffer : m_imageBuffers.values()) {
152 if (!imageBuffer)
153 continue;
154 m_remoteRenderingBackendProxy.createRemoteImageBuffer(*imageBuffer);
155 imageBuffer->clearBackend();
156 }
157 m_nativeImages.clear();
158 m_fontIdentifierToLastRenderingUpdateVersionMap.clear();
159 m_numberOfFontsUsedInCurrentRenderingUpdate = 0;
160}
161
162void RemoteResourceCacheProxy::releaseMemory()
163{
164 m_fontIdentifierToLastRenderingUpdateVersionMap.clear();
165 m_numberOfFontsUsedInCurrentRenderingUpdate = 0;
166 m_remoteRenderingBackendProxy.deleteAllFonts();
167}
