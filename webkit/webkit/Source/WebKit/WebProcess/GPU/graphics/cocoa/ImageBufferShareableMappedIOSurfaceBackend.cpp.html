42std::unique_ptr<ImageBufferShareableMappedIOSurfaceBackend> ImageBufferShareableMappedIOSurfaceBackend::create(const Parameters& parameters, const HostWindow*)
43{
44 IntSize backendSize = calculateSafeBackendSize(parameters);
45 if (backendSize.isEmpty())
46 return nullptr;
47
48 auto surface = IOSurface::create(backendSize, backendSize, parameters.colorSpace, IOSurface::formatForPixelFormat(parameters.pixelFormat));
49 if (!surface)
50 return nullptr;
51
52 RetainPtr<CGContextRef> cgContext = surface->ensurePlatformContext();
53 if (!cgContext)
54 return nullptr;
55
56 CGContextClearRect(cgContext.get(), FloatRect(FloatPoint::zero(), backendSize));
57
58 return makeUnique<ImageBufferShareableMappedIOSurfaceBackend>(parameters, WTFMove(surface));
59}
60
61std::unique_ptr<ImageBufferShareableMappedIOSurfaceBackend> ImageBufferShareableMappedIOSurfaceBackend::create(const Parameters& parameters, ImageBufferBackendHandle handle)
62{
63 if (!WTF::holds_alternative<MachSendRight>(handle)) {
64 ASSERT_NOT_REACHED();
65 return nullptr;
66 }
67
68 auto surface = IOSurface::createFromSendRight(WTFMove(WTF::get<MachSendRight>(handle)), parameters.colorSpace);
69 if (!surface)
70 return nullptr;
71
72 return makeUnique<ImageBufferShareableMappedIOSurfaceBackend>(parameters, WTFMove(surface));
73}
74
75ImageBufferBackendHandle ImageBufferShareableMappedIOSurfaceBackend::createImageBufferBackendHandle() const
76{
77 return ImageBufferBackendHandle(m_surface->createSendRight());
78}
