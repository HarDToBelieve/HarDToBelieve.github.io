39AudioTrackPrivateRemote::AudioTrackPrivateRemote(GPUProcessConnection& gpuProcessConnection, WebCore::MediaPlayerIdentifier playerIdentifier, TrackPrivateRemoteIdentifier idendifier, TrackPrivateRemoteConfiguration&& configuration)
40 : m_gpuProcessConnection(makeWeakPtr(gpuProcessConnection))
41 , m_playerIdentifier(playerIdentifier)
42 , m_idendifier(idendifier)
43{
44 updateConfiguration(WTFMove(configuration));
45}
46
47void AudioTrackPrivateRemote::setEnabled(bool enabled)
48{
49 if (!m_gpuProcessConnection)
50 return;
51
52 if (enabled != this->enabled())
53 m_gpuProcessConnection->connection().send(Messages::RemoteMediaPlayerProxy::AudioTrackSetEnabled(m_idendifier, enabled), m_playerIdentifier);
54
55 AudioTrackPrivate::setEnabled(enabled);
56}
57
58void AudioTrackPrivateRemote::updateConfiguration(TrackPrivateRemoteConfiguration&& configuration)
59{
60 if (configuration.trackId != m_id) {
61 auto changed = !m_id.isEmpty();
62 m_id = configuration.trackId;
63 if (changed && client())
64 client()->idChanged(m_id);
65 }
66
67 if (configuration.label != m_label) {
68 auto changed = !m_label.isEmpty();
69 m_label = configuration.label;
70 if (changed && client())
71 client()->labelChanged(m_label);
72 }
73
74 if (configuration.language != m_language) {
75 auto changed = !m_language.isEmpty();
76 m_language = configuration.language;
77 if (changed && client())
78 client()->languageChanged(m_language);
79 }
80
81 m_trackIndex = configuration.trackIndex;
82 m_startTimeVariance = configuration.startTimeVariance;
83 m_kind = configuration.audioKind;
84 AudioTrackPrivate::setEnabled(configuration.enabled);
85}
