134void MediaPlayerPrivateRemote::prepareForPlayback(bool privateMode, MediaPlayer::Preload preload, bool preservesPitch, bool prepare)
135{
136 auto player = makeRefPtr(m_player.get());
137 if (!player)
138 return;
139
140 auto scale = player->playerContentsScale();
141 auto preferredDynamicRangeMode = m_player->preferredDynamicRangeMode();
142
143 connection().sendWithAsyncReply(Messages::RemoteMediaPlayerProxy::PrepareForPlayback(privateMode, preload, preservesPitch, prepare, scale, preferredDynamicRangeMode), [weakThis = makeWeakPtr(*this), this](auto inlineLayerHostingContextId) mutable {
144 if (!weakThis)
145 return;
146
147 if (!inlineLayerHostingContextId)
148 return;
149
150 auto player = makeRefPtr(m_player.get());
151 if (!player)
152 return;
153
154 m_videoLayer = createVideoLayerRemote(this, inlineLayerHostingContextId.value(), m_videoFullscreenGravity);
155#if PLATFORM(COCOA)
156 m_videoLayerManager->setVideoLayer(m_videoLayer.get(), snappedIntRect(player->playerContentBoxRect()).size());
157#endif
158 }, m_id);
159}

161void MediaPlayerPrivateRemote::load(const URL& url, const ContentType& contentType, const String& keySystem)
162{
163 std::optional<SandboxExtension::Handle> sandboxExtensionHandle;
164 if (url.isLocalFile()) {
165 SandboxExtension::Handle handle;
166 auto fileSystemPath = url.fileSystemPath();
167
168 auto createExtension = [&] {
169#if HAVE(AUDIT_TOKEN)
170 if (auto auditToken = m_manager.gpuProcessConnection().auditToken())
171 return SandboxExtension::createHandleForReadByAuditToken(fileSystemPath, auditToken.value(), handle);
172#endif
173
174 return SandboxExtension::createHandle(fileSystemPath, SandboxExtension::Type::ReadOnly, handle);
175 };
176
177 if (!createExtension()) {
178 WTFLogAlways("Unable to create sandbox extension handle for GPUProcess url.\n");
179 m_cachedState.networkState = MediaPlayer::NetworkState::FormatError;
180 if (auto player = makeRefPtr(m_player.get()))
181 player->networkStateChanged();
182 return;
183 }
184
185 sandboxExtensionHandle = WTFMove(handle);
186 }
187
188 connection().sendWithAsyncReply(Messages::RemoteMediaPlayerProxy::Load(url, sandboxExtensionHandle, contentType, keySystem), [weakThis = makeWeakPtr(*this), this](auto&& configuration) {
189 if (!weakThis)
190 return;
191
192 auto player = makeRefPtr(m_player.get());
193 if (!player)
194 return;
195
196 m_configuration = WTFMove(configuration);
197 player->mediaEngineUpdated();
198 }, m_id);
199}

217void MediaPlayerPrivateRemote::pause()
218{
219 m_cachedState.paused = true;
220 auto now = MonotonicTime::now();
221 m_cachedMediaTime += MediaTime::createWithDouble(m_rate * (now - m_cachedMediaTimeQueryTime).value());
222 m_cachedMediaTimeQueryTime = now;
223 connection().send(Messages::RemoteMediaPlayerProxy::Pause(), m_id);
224}

468void MediaPlayerPrivateRemote::updateCachedState(RemoteMediaPlayerState&& state)
469{
470 const Seconds playbackQualityMetricsTimeout = 30_s;
471
472 m_cachedState.duration = state.duration;
473 m_cachedState.minTimeSeekable = state.minTimeSeekable;
474 m_cachedState.maxTimeSeekable = state.maxTimeSeekable;
475 m_cachedState.networkState = state.networkState;
476 m_cachedState.readyState = state.readyState;
477 m_cachedState.paused = state.paused;
478 m_cachedState.naturalSize = state.naturalSize;
479 m_cachedState.movieLoadType = state.movieLoadType;
480 m_cachedState.wirelessPlaybackTargetType = state.wirelessPlaybackTargetType;
481 m_cachedState.wirelessPlaybackTargetName = state.wirelessPlaybackTargetName;
482
483 m_cachedState.startDate = state.startDate;
484 m_cachedState.startTime = state.startTime;
485 m_cachedState.languageOfPrimaryAudioTrack = state.languageOfPrimaryAudioTrack;
486 m_cachedState.maxFastForwardRate = state.maxFastForwardRate;
487 m_cachedState.minFastReverseRate = state.minFastReverseRate;
488 m_cachedState.seekableTimeRangesLastModifiedTime = state.seekableTimeRangesLastModifiedTime;
489 m_cachedState.liveUpdateInterval = state.liveUpdateInterval;
490 m_cachedState.canSaveMediaData = state.canSaveMediaData;
491 m_cachedState.hasAudio = state.hasAudio;
492 m_cachedState.hasVideo = state.hasVideo;
493
494 if (state.videoMetrics)
495 m_cachedState.videoMetrics = state.videoMetrics;
496 if (m_videoPlaybackMetricsUpdateInterval && (MonotonicTime::now() - m_lastPlaybackQualityMetricsQueryTime) > playbackQualityMetricsTimeout)
497 updateVideoPlaybackMetricsUpdateInterval(0_s);
498
499 m_cachedState.hasClosedCaptions = state.hasClosedCaptions;
500 m_cachedState.hasAvailableVideoFrame = state.hasAvailableVideoFrame;
501 m_cachedState.wirelessVideoPlaybackDisabled = state.wirelessVideoPlaybackDisabled;
502 m_cachedState.hasSingleSecurityOrigin = state.hasSingleSecurityOrigin;
503 m_cachedState.didPassCORSAccessCheck = state.didPassCORSAccessCheck;
504 m_cachedState.wouldTaintDocumentSecurityOrigin = state.wouldTaintDocumentSecurityOrigin;
505
506 if (state.bufferedRanges.length())
507 m_cachedBufferedTimeRanges = makeUnique<PlatformTimeRanges>(state.bufferedRanges);
508}
