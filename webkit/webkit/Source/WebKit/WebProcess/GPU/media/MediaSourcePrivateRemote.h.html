52class MediaSourcePrivateRemote final
53 : public WebCore::MediaSourcePrivate
54 , public IPC::MessageReceiver
55#if !RELEASE_LOG_DISABLED
56 , private LoggerHelper
57#endif
58{
59public:
60 static Ref<MediaSourcePrivateRemote> create(GPUProcessConnection&, RemoteMediaSourceIdentifier, RemoteMediaPlayerMIMETypeCache&, const MediaPlayerPrivateRemote&, WebCore::MediaSourcePrivateClient*);
61 virtual ~MediaSourcePrivateRemote();
62
63 // MediaSourcePrivate overrides
64 AddStatus addSourceBuffer(const WebCore::ContentType&, bool webMParserEnabled, RefPtr<WebCore::SourceBufferPrivate>&) final;
65 void durationChanged(const MediaTime&) final;
66 void bufferedChanged(const WebCore::PlatformTimeRanges&) final;
67 void markEndOfStream(EndOfStreamStatus) final;
68 void unmarkEndOfStream() final;
69 bool isEnded() const final;
70 WebCore::MediaPlayer::ReadyState readyState() const final;
71 void setReadyState(WebCore::MediaPlayer::ReadyState) final;
72 void setIsSeeking(bool) final;
73 void waitForSeekCompleted() final;
74 void seekCompleted() final;
75 void setTimeFudgeFactor(const MediaTime&) final;
76
77 MediaTime duration() const { return m_client->duration(); }
78
79#if !RELEASE_LOG_DISABLED
80 const Logger& logger() const final { return m_logger.get(); }
81 const void* nextSourceBufferLogIdentifier() { return childLogIdentifier(m_logIdentifier, ++m_nextSourceBufferID); }
82#endif
83
84private:
85 MediaSourcePrivateRemote(GPUProcessConnection&, RemoteMediaSourceIdentifier, RemoteMediaPlayerMIMETypeCache&, const MediaPlayerPrivateRemote&, WebCore::MediaSourcePrivateClient*);
86
87 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) final;
88 void seekToTime(const MediaTime&);
89
90 WeakPtr<GPUProcessConnection> m_gpuProcessConnection;
91 RemoteMediaSourceIdentifier m_identifier;
92 RemoteMediaPlayerMIMETypeCache& m_mimeTypeCache;
93 WeakPtr<MediaPlayerPrivateRemote> m_mediaPlayerPrivate;
94 RefPtr<WebCore::MediaSourcePrivateClient> m_client;
95 Vector<RefPtr<SourceBufferPrivateRemote>> m_sourceBuffers;
96
97#if !RELEASE_LOG_DISABLED
98 const char* logClassName() const override { return "MediaSourcePrivateRemote"; }
99 const void* logIdentifier() const final { return m_logIdentifier; }
100 WTFLogChannel& logChannel() const final;
101
102 Ref<const Logger> m_logger;
103 const void* m_logIdentifier;
104 uint64_t m_nextSourceBufferID { 0 };
105#endif
106};
