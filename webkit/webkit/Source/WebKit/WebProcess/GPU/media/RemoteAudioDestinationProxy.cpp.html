47// Allocate a ring buffer large enough to contain 2 seconds of audio.
48constexpr size_t ringBufferSizeInSecond = 2;
49
50using AudioDestination = WebCore::AudioDestination;
51using AudioIOCallback = WebCore::AudioIOCallback;

53Ref<AudioDestination> RemoteAudioDestinationProxy::create(AudioIOCallback& callback,
54 const String& inputDeviceId, unsigned numberOfInputChannels, unsigned numberOfOutputChannels, float sampleRate)
55{
56 return adoptRef(*new RemoteAudioDestinationProxy(callback, inputDeviceId, numberOfInputChannels, numberOfOutputChannels, sampleRate));
57}

59RemoteAudioDestinationProxy::RemoteAudioDestinationProxy(AudioIOCallback& callback, const String& inputDeviceId, unsigned numberOfInputChannels, unsigned numberOfOutputChannels, float sampleRate)
60#if PLATFORM(COCOA)
61 : WebCore::AudioDestinationCocoa(callback, numberOfOutputChannels, sampleRate, false)
62 , m_numberOfFrames(hardwareSampleRate() * ringBufferSizeInSecond)
63 , m_ringBuffer(makeUnique<WebCore::CARingBuffer>(makeUniqueRef<SharedRingBufferStorage>(std::bind(&RemoteAudioDestinationProxy::storageChanged, this, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3))))
64 , m_sampleRate(hardwareSampleRate())
65#else
66 : WebCore::AudioDestinationGStreamer(callback, numberOfOutputChannels, sampleRate)
67 , m_numberOfOutputChannels(numberOfOutputChannels)
68#endif
69 , m_inputDeviceId(inputDeviceId)
70 , m_numberOfInputChannels(numberOfInputChannels)
71{
72}

74void RemoteAudioDestinationProxy::startRenderingThread()
75{
76 ASSERT(!m_renderThread);
77
78 auto offThreadRendering = [this]() mutable {
79 do {
80 m_renderSemaphore.wait();
81 if (m_shouldStopThread)
82 break;
83
84 renderQuantum();
85 } while (!m_shouldStopThread);
86 };
87 m_renderThread = Thread::create("RemoteAudioDestinationProxy render thread", WTFMove(offThreadRendering), ThreadType::Audio, Thread::QOS::UserInteractive);
88}
89
90void RemoteAudioDestinationProxy::stopRenderingThread()
91{
92 if (!m_renderThread)
93 return;
94
95 m_shouldStopThread = true;
96 m_renderSemaphore.signal();
97 m_renderThread->waitForCompletion();
98 m_renderThread = nullptr;
99}

101IPC::Connection* RemoteAudioDestinationProxy::connection()
102{
103 if (!m_gpuProcessConnection) {
104 m_gpuProcessConnection = makeWeakPtr(WebProcess::singleton().ensureGPUProcessConnection());
105 m_gpuProcessConnection->addClient(*this);
106
107 if (!m_gpuProcessConnection->connection().sendSync(Messages::RemoteAudioDestinationManager::CreateAudioDestination(m_inputDeviceId, m_numberOfInputChannels, numberOfOutputChannels(), sampleRate(), hardwareSampleRate(), m_renderSemaphore), Messages::RemoteAudioDestinationManager::CreateAudioDestination::Reply(m_destinationID), 0)) {
108 // The GPUProcess likely crashed during this synchronous IPC. gpuProcessConnectionDidClose() will get called to reconnect to the GPUProcess.
109 RELEASE_LOG_ERROR(Media, "RemoteAudioDestinationProxy::destinationID: IPC to create the audio destination failed, the GPUProcess likely crashed.");
110 return nullptr;
111 }
112
113#if PLATFORM(COCOA)
114 m_currentFrame = 0;
115 AudioStreamBasicDescription streamFormat;
116 getAudioStreamBasicDescription(streamFormat);
117 m_ringBuffer->allocate(streamFormat, m_numberOfFrames);
118 m_audioBufferList = makeUnique<WebCore::WebAudioBufferList>(streamFormat);
119 m_audioBufferList->setSampleCount(WebCore::AudioUtilities::renderQuantumSize);
120#endif
121
122 startRenderingThread();
123 }
124 return m_destinationID ? &m_gpuProcessConnection->connection() : nullptr;
125}
126
127IPC::Connection* RemoteAudioDestinationProxy::existingConnection()
128{
129 return m_gpuProcessConnection && m_destinationID ? &m_gpuProcessConnection->connection() : nullptr;
130}

132RemoteAudioDestinationProxy::~RemoteAudioDestinationProxy()
133{
134 if (m_gpuProcessConnection && m_destinationID) {
135 m_gpuProcessConnection->connection().sendWithAsyncReply(
136 Messages::RemoteAudioDestinationManager::DeleteAudioDestination(m_destinationID), [] {
137 // Can't remove this from proxyMap() here because the object would have been already deleted.
138 });
139 }
140
141 stopRenderingThread();
142}

144void RemoteAudioDestinationProxy::startRendering(CompletionHandler<void(bool)>&& completionHandler)
145{
146 auto* connection = this->connection();
147 if (!connection)
148 return completionHandler(false);
149
150 connection->sendWithAsyncReply(Messages::RemoteAudioDestinationManager::StartAudioDestination(m_destinationID), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)](bool isPlaying) mutable {
151 setIsPlaying(isPlaying);
152 completionHandler(isPlaying);
153 });
154}
155
156void RemoteAudioDestinationProxy::stopRendering(CompletionHandler<void(bool)>&& completionHandler)
157{
158 auto* connection = this->connection();
159 if (!connection)
160 return completionHandler(false);
161
162 connection->sendWithAsyncReply(Messages::RemoteAudioDestinationManager::StopAudioDestination(m_destinationID), [this, protectedThis = makeRef(*this), completionHandler = WTFMove(completionHandler)](bool isPlaying) mutable {
163 setIsPlaying(isPlaying);
164 completionHandler(!isPlaying);
165 });
166}
167
168void RemoteAudioDestinationProxy::renderQuantum()
169{
170 ASSERT(!isMainRunLoop());
171
172#if PLATFORM(COCOA)
173 AudioDestinationCocoa::render(m_currentFrame / static_cast<double>(m_sampleRate), mach_absolute_time(), WebCore::AudioUtilities::renderQuantumSize, m_audioBufferList->list());
174 m_ringBuffer->store(m_audioBufferList->list(), WebCore::AudioUtilities::renderQuantumSize, m_currentFrame);
175 m_currentFrame += WebCore::AudioUtilities::renderQuantumSize;
176#endif
177}

179#if PLATFORM(COCOA)
180void RemoteAudioDestinationProxy::storageChanged(SharedMemory* storage, const WebCore::CAAudioStreamDescription& format, size_t frameCount)
181{
182 auto* connection = existingConnection();
183 if (!connection)
184 return;
185
186 SharedMemory::Handle handle;
187 if (storage)
188 storage->createHandle(handle, SharedMemory::Protection::ReadOnly);
189
190 // FIXME: Send the actual data size with IPCHandle.
191#if OS(DARWIN) || OS(WINDOWS)
192 uint64_t dataSize = handle.size();
193#else
194 uint64_t dataSize = 0;
195#endif
196
197 m_gpuProcessConnection->connection().send(Messages::RemoteAudioDestinationManager::AudioSamplesStorageChanged { m_destinationID, SharedMemory::IPCHandle { WTFMove(handle), dataSize }, format, frameCount }, 0);
198}
199#endif
200
201void RemoteAudioDestinationProxy::gpuProcessConnectionDidClose(GPUProcessConnection& oldConnection)
202{
203 oldConnection.removeClient(*this);
204
205 stopRenderingThread();
206 m_gpuProcessConnection = nullptr;
207 m_destinationID = { };
208
209 if (isPlaying())
210 startRendering([](bool) { });
211}
