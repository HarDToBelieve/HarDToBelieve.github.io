40Ref<RemoteAudioHardwareListener> RemoteAudioHardwareListener::create(AudioHardwareListener::Client& client, WebProcess& webProcess)
41{
42 return adoptRef(*new RemoteAudioHardwareListener(client, webProcess));
43}
44
45RemoteAudioHardwareListener::RemoteAudioHardwareListener(AudioHardwareListener::Client& client, WebProcess& webProcess)
46 : AudioHardwareListener(client)
47 , m_process(webProcess)
48 , m_identifier(RemoteAudioHardwareListenerIdentifier::generate())
49{
50 auto& connection = WebProcess::singleton().ensureGPUProcessConnection();
51 connection.addClient(*this);
52 connection.messageReceiverMap().addMessageReceiver(Messages::RemoteAudioHardwareListener::messageReceiverName(), m_identifier.toUInt64(), *this);
53 connection.connection().send(Messages::GPUConnectionToWebProcess::CreateAudioHardwareListener(m_identifier), { });
54}
55
56RemoteAudioHardwareListener::~RemoteAudioHardwareListener()
57{
58 if (auto* connection = WebProcess::singleton().existingGPUProcessConnection()) {
59 connection->messageReceiverMap().removeMessageReceiver(*this);
60 connection->connection().send(Messages::GPUConnectionToWebProcess::ReleaseAudioHardwareListener(m_identifier), 0);
61 }
62}
63
64void RemoteAudioHardwareListener::gpuProcessConnectionDidClose(GPUProcessConnection&)
65{
66 audioHardwareDidBecomeInactive();
67}
68
69void RemoteAudioHardwareListener::audioHardwareDidBecomeActive()
70{
71 setHardwareActivity(AudioHardwareActivityType::IsActive);
72 m_client.audioHardwareDidBecomeActive();
73}
74
75void RemoteAudioHardwareListener::audioHardwareDidBecomeInactive()
76{
77 setHardwareActivity(AudioHardwareActivityType::IsInactive);
78 m_client.audioHardwareDidBecomeInactive();
79}

81void RemoteAudioHardwareListener::audioOutputDeviceChanged(size_t bufferSizeMinimum, size_t bufferSizeMaximum)

83 setSupportedBufferSizes({ bufferSizeMinimum, bufferSizeMaximum });
84 m_client.audioOutputDeviceChanged();
