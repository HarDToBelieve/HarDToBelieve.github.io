27
28#if ENABLE(GPU_PROCESS) && USE(AUDIO_SESSION)
29
30#include <WebCore/AudioSession.h>
31
32namespace WebKit {
33
34struct RemoteAudioSessionConfiguration {
35 String routingContextUID;
36 float sampleRate { 0 };
37 size_t bufferSize { 0 };
38 size_t numberOfOutputChannels { 0 };
39 size_t maximumNumberOfOutputChannels { 0 };
40 size_t preferredBufferSize { 0 };
41 bool isMuted { false };
42 bool isActive { false };
43
44 template<class Encoder>
45 void encode(Encoder& encoder) const
46 {
47 encoder << routingContextUID;
48 encoder << sampleRate;
49 encoder << bufferSize;
50 encoder << numberOfOutputChannels;
51 encoder << maximumNumberOfOutputChannels;
52 encoder << preferredBufferSize;
53 encoder << isMuted;
54 encoder << isActive;
55 }
56
57 template <class Decoder>
58 static std::optional<RemoteAudioSessionConfiguration> decode(Decoder& decoder)
59 {
60 std::optional<String> routingContextUID;
61 decoder >> routingContextUID;
62 if (!routingContextUID)
63 return std::nullopt;
64
65 std::optional<float> sampleRate;
66 decoder >> sampleRate;
67 if (!sampleRate)
68 return std::nullopt;
69
70 std::optional<size_t> bufferSize;
71 decoder >> bufferSize;
72 if (!bufferSize)
73 return std::nullopt;
74
75 std::optional<size_t> numberOfOutputChannels;
76 decoder >> numberOfOutputChannels;
77 if (!numberOfOutputChannels)
78 return std::nullopt;
79
80 std::optional<size_t> maximumNumberOfOutputChannels;
81 decoder >> maximumNumberOfOutputChannels;
82 if (!maximumNumberOfOutputChannels)
83 return std::nullopt;
84
85 std::optional<size_t> preferredBufferSize;
86 decoder >> preferredBufferSize;
87 if (!preferredBufferSize)
88 return std::nullopt;
89
90 std::optional<bool> isMuted;
91 decoder >> isMuted;
92 if (!isMuted)
93 return std::nullopt;
94
95 std::optional<bool> isActive;
96 decoder >> isActive;
97 if (!isActive)
98 return std::nullopt;
99
100 return {{
101 WTFMove(*routingContextUID),
102 *sampleRate,
103 *bufferSize,
104 *numberOfOutputChannels,
105 *maximumNumberOfOutputChannels,
106 *preferredBufferSize,
107 *isMuted,
108 *isActive,
109 }};
110 }
111};
112
113} // namespace WebKit

115#endif
