44Ref<RemoteAudioSourceProvider> RemoteAudioSourceProvider::create(WebCore::MediaPlayerIdentifier identifier, WTF::LoggerHelper& helper)
45{
46 auto provider = adoptRef(*new RemoteAudioSourceProvider(identifier, helper));
47 provider->m_gpuProcessConnection->audioSourceProviderManager().addProvider(provider.copyRef());
48 return provider;
49}
50
51RemoteAudioSourceProvider::RemoteAudioSourceProvider(MediaPlayerIdentifier identifier, WTF::LoggerHelper& helper)
52 : m_identifier(identifier)
53 , m_gpuProcessConnection(makeWeakPtr(WebProcess::singleton().ensureGPUProcessConnection()))
54#if !RELEASE_LOG_DISABLED
55 , m_logger(helper.logger())
56 , m_logIdentifier(helper.logIdentifier())
57#endif
58{
59 ASSERT(isMainRunLoop());
60 UNUSED_PARAM(helper);
61
62#if ENABLE(WEB_AUDIO)
63 m_gpuProcessConnection->connection().send(Messages::RemoteMediaPlayerProxy::CreateAudioSourceProvider { }, identifier);
64#endif
65}
66
67RemoteAudioSourceProvider::~RemoteAudioSourceProvider()
68{
69}
70
71void RemoteAudioSourceProvider::close()
72{
73 ASSERT(isMainRunLoop());
74 if (m_gpuProcessConnection)
75 m_gpuProcessConnection->audioSourceProviderManager().removeProvider(m_identifier);
76}
77
78void RemoteAudioSourceProvider::hasNewClient(AudioSourceProviderClient* client)
79{
80 if (m_gpuProcessConnection)
81 m_gpuProcessConnection->connection().send(Messages::RemoteMediaPlayerProxy::SetShouldEnableAudioSourceProvider { !!client }, m_identifier);
82}
83
84void RemoteAudioSourceProvider::audioSamplesAvailable(const PlatformAudioData& data, const AudioStreamDescription& description, size_t size)
85{
86 receivedNewAudioSamples(data, description, size);
87}
