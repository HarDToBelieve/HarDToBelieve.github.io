38namespace WebKit {

40class RemoteAudioSourceProvider;
41
42class RemoteAudioSourceProviderManager : public IPC::Connection::WorkQueueMessageReceiver {
43public:
44 static Ref<RemoteAudioSourceProviderManager> create() { return adoptRef(*new RemoteAudioSourceProviderManager()); }
45 ~RemoteAudioSourceProviderManager();
46 void stopListeningForIPC();
47
48 void addProvider(Ref<RemoteAudioSourceProvider>&&);
49 void removeProvider(WebCore::MediaPlayerIdentifier);
50
51 void didReceiveMessage(IPC::Connection&, IPC::Decoder&);
52
53private:
54 RemoteAudioSourceProviderManager();
55
56 // Messages
57 void audioStorageChanged(WebCore::MediaPlayerIdentifier, const SharedMemory::IPCHandle&, const WebCore::CAAudioStreamDescription&, uint64_t numberOfFrames);
58 void audioSamplesAvailable(WebCore::MediaPlayerIdentifier, uint64_t startFrame, uint64_t numberOfFrames);
59
60 void setConnection(IPC::Connection*);
61
62 class RemoteAudio {
63 WTF_MAKE_FAST_ALLOCATED;
64 public:
65 explicit RemoteAudio(Ref<RemoteAudioSourceProvider>&&);
66
67 void setStorage(const SharedMemory::Handle&, const WebCore::CAAudioStreamDescription&, uint64_t numberOfFrames);
68 void audioSamplesAvailable(uint64_t startFrame, uint64_t numberOfFrames);
69
70 private:
71 Ref<RemoteAudioSourceProvider> m_provider;
72 WebCore::CAAudioStreamDescription m_description;
73 std::unique_ptr<WebCore::CARingBuffer> m_ringBuffer;
74 std::unique_ptr<WebCore::WebAudioBufferList> m_buffer;
75 };
76
77 Ref<WorkQueue> m_queue;
78 RefPtr<IPC::Connection> m_connection;
79
80 // background thread member
81 HashMap<WebCore::MediaPlayerIdentifier, std::unique_ptr<RemoteAudio>> m_providers;
82};
