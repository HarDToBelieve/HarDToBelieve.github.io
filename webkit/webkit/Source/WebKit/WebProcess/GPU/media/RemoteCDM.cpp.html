58void RemoteCDM::getSupportedConfiguration(CDMKeySystemConfiguration&& configuration, LocalStorageAccess access, SupportedConfigurationCallback&& callback)
59{
60 if (!m_factory) {
61 callback(std::nullopt);
62 return;
63 }
64
65 m_factory->gpuProcessConnection().connection().sendWithAsyncReply(Messages::RemoteCDMProxy::GetSupportedConfiguration(WTFMove(configuration), access), WTFMove(callback), m_identifier);
66}

111RefPtr<CDMInstance> RemoteCDM::createInstance()
112{
113 if (!m_factory)
114 return nullptr;
115
116 RemoteCDMInstanceIdentifier identifier;
117 RemoteCDMInstanceConfiguration configuration;
118 m_factory->gpuProcessConnection().connection().sendSync(Messages::RemoteCDMProxy::CreateInstance(), Messages::RemoteCDMProxy::CreateInstance::Reply(identifier, configuration), m_identifier);
119 if (!identifier)
120 return nullptr;
121 return RemoteCDMInstance::create(makeWeakPtr(m_factory.get()), WTFMove(identifier), WTFMove(configuration));
122}
123
124void RemoteCDM::loadAndInitialize()
125{
126 if (!m_factory)
127 return;
128
129 m_factory->gpuProcessConnection().connection().send(Messages::RemoteCDMProxy::LoadAndInitialize(), m_identifier);
130}
