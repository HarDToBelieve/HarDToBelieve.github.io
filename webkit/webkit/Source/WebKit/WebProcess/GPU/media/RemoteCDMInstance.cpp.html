45Ref<RemoteCDMInstance> RemoteCDMInstance::create(WeakPtr<RemoteCDMFactory>&& factory, RemoteCDMInstanceIdentifier&& identifier, RemoteCDMInstanceConfiguration&& configuration)
46{
47 return adoptRef(*new RemoteCDMInstance(WTFMove(factory), WTFMove(identifier), WTFMove(configuration)));
48}
49
50RemoteCDMInstance::RemoteCDMInstance(WeakPtr<RemoteCDMFactory>&& factory, RemoteCDMInstanceIdentifier&& identifier, RemoteCDMInstanceConfiguration&& configuration)
51 : m_factory(WTFMove(factory))
52 , m_identifier(WTFMove(identifier))
53 , m_configuration(WTFMove(configuration))
54{
55 if (m_factory)
56 m_factory->gpuProcessConnection().messageReceiverMap().addMessageReceiver(Messages::RemoteCDMInstance::messageReceiverName(), m_identifier.toUInt64(), *this);
57}
58
59RemoteCDMInstance::~RemoteCDMInstance()
60{
61 if (m_factory)
62 m_factory->gpuProcessConnection().messageReceiverMap().removeMessageReceiver(Messages::RemoteCDMInstance::messageReceiverName(), m_identifier.toUInt64());
63}

65void RemoteCDMInstance::unrequestedInitializationDataReceived(const String& type, IPC::SharedBufferCopy&& initData)
66{
67 if (m_client && initData.buffer())
68 m_client->unrequestedInitializationDataReceived(type, initData.buffer().releaseNonNull());
69}

71void RemoteCDMInstance::initializeWithConfiguration(const WebCore::CDMKeySystemConfiguration& configuration, AllowDistinctiveIdentifiers distinctiveIdentifiers, AllowPersistentState persistentState, SuccessCallback&& callback)
72{
73 if (!m_factory) {
74 callback(Failed);
75 return;
76 }
77
78 m_factory->gpuProcessConnection().connection().sendWithAsyncReply(Messages::RemoteCDMInstanceProxy::InitializeWithConfiguration(configuration, distinctiveIdentifiers, persistentState), WTFMove(callback), m_identifier);
79}
80
81void RemoteCDMInstance::setServerCertificate(Ref<WebCore::SharedBuffer>&& certificate, SuccessCallback&& callback)
82{
83 if (!m_factory) {
84 callback(Failed);
85 return;
86 }
87
88 m_factory->gpuProcessConnection().connection().sendWithAsyncReply(Messages::RemoteCDMInstanceProxy::SetServerCertificate(certificate.get()), WTFMove(callback), m_identifier);
89}
90
91void RemoteCDMInstance::setStorageDirectory(const String& directory)
92{
93 if (!m_factory)
94 return;
95
96 m_factory->gpuProcessConnection().connection().send(Messages::RemoteCDMInstanceProxy::SetStorageDirectory(directory), m_identifier);
97}
98
99RefPtr<WebCore::CDMInstanceSession> RemoteCDMInstance::createSession()
100{
101 if (!m_factory)
102 return nullptr;
103
104 RemoteCDMInstanceSessionIdentifier identifier;
105 m_factory->gpuProcessConnection().connection().sendSync(Messages::RemoteCDMInstanceProxy::CreateSession(), Messages::RemoteCDMInstanceProxy::CreateSession::Reply(identifier), m_identifier);
106 if (!identifier)
107 return nullptr;
108 auto session = RemoteCDMInstanceSession::create(makeWeakPtr(m_factory.get()), WTFMove(identifier));
109 m_factory->addSession(session.copyRef());
110 return session;
111}
