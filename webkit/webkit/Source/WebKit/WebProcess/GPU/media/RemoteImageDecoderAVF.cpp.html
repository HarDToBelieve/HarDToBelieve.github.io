75EncodedDataStatus RemoteImageDecoderAVF::encodedDataStatus() const
76{
77 if (m_frameCount)
78 return EncodedDataStatus::Complete;
79 if (m_size)
80 return EncodedDataStatus::SizeAvailable;
81 if (m_hasTrack)
82 return EncodedDataStatus::TypeAvailable;
83
84 return EncodedDataStatus::Unknown;
85}

160PlatformImagePtr RemoteImageDecoderAVF::createFrameImageAtIndex(size_t index, SubsamplingLevel, const DecodingOptions&)
161{
162 if (m_frameImages.contains(index))
163 return m_frameImages.get(index);
164
165 auto createFrameImage = [&] {
166 if (!m_gpuProcessConnection)
167 return;
168
169 std::optional<MachSendRight> sendRight;
170 std::optional<DestinationColorSpace> colorSpace;
171 if (!m_gpuProcessConnection->connection().sendSync(Messages::RemoteImageDecoderAVFProxy::CreateFrameImageAtIndex(m_identifier, index), Messages::RemoteImageDecoderAVFProxy::CreateFrameImageAtIndex::Reply(sendRight, colorSpace), 0))
172 return;
173
174 if (!sendRight || !colorSpace)
175 return;
176
177 auto surface = WebCore::IOSurface::createFromSendRight(WTFMove(*sendRight), WTFMove(*colorSpace));
178 if (!surface)
179 return;
180
181 if (auto image = IOSurface::sinkIntoImage(WTFMove(surface)))
182 m_frameImages.add(index, image);
183 };
184
185 callOnMainRunLoopAndWait(WTFMove(createFrameImage));
186
187 if (m_frameImages.contains(index))
188 return m_frameImages.get(index);
189
190 return nullptr;
191}

201// If allDataReceived is true, the caller expects encodedDataStatus() to be >= EncodedDataStatus::SizeAvailable
202// after this function returns (in the same run loop).
203void RemoteImageDecoderAVF::setData(SharedBuffer& data, bool allDataReceived)
204{
205 if (!m_gpuProcessConnection)
206 return;
207
208 IPC::SharedBufferDataReference dataReference { data };
209
210 size_t frameCount;
211 IntSize size;
212 bool hasTrack;
213 std::optional<Vector<ImageDecoder::FrameInfo>> frameInfos;
214 if (!m_gpuProcessConnection->connection().sendSync(Messages::RemoteImageDecoderAVFProxy::SetData(m_identifier, dataReference, allDataReceived), Messages::RemoteImageDecoderAVFProxy::SetData::Reply(frameCount, size, hasTrack, frameInfos), 0))
215 return;
216
217 m_isAllDataReceived = allDataReceived;
218 m_frameCount = frameCount;
219 if (!size.isEmpty())
220 m_size = size;
221 m_hasTrack = hasTrack;
222
223 if (frameInfos)
224 m_frameInfos.swap(*frameInfos);
225}

227void RemoteImageDecoderAVF::clearFrameBufferCache(size_t index)
228{
229 for (size_t i = 0; i < std::min(index + 1, m_frameCount); ++i)
230 m_frameImages.remove(i);
231
232 m_gpuProcessConnection->connection().send(Messages::RemoteImageDecoderAVFProxy::ClearFrameBufferCache(m_identifier, index), 0);
233}

235void RemoteImageDecoderAVF::encodedDataStatusChanged(size_t frameCount, const WebCore::IntSize& size, bool hasTrack)
236{
237 m_frameCount = frameCount;
238 if (!size.isEmpty())
239 m_size = size;
240 m_hasTrack = hasTrack;
241
242 if (m_encodedDataStatusChangedCallback)
243 m_encodedDataStatusChangedCallback(encodedDataStatus());
244}
