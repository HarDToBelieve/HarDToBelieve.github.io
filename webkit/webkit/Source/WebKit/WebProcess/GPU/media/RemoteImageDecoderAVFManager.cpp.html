42RefPtr<RemoteImageDecoderAVF> RemoteImageDecoderAVFManager::createImageDecoder(SharedBuffer& data, const String& mimeType, AlphaOption alphaOption, GammaAndColorProfileOption gammaAndColorProfileOption)
43{
44 std::optional<ImageDecoderIdentifier> imageDecoderIdentifier;
45 IPC::SharedBufferDataReference dataReference { data };
46 if (!ensureGPUProcessConnection().connection().sendSync(Messages::RemoteImageDecoderAVFProxy::CreateDecoder(dataReference, mimeType), Messages::RemoteImageDecoderAVFProxy::CreateDecoder::Reply(imageDecoderIdentifier), 0))
47 return nullptr;
48
49 if (!imageDecoderIdentifier)
50 return nullptr;
51
52 auto remoteImageDecoder = RemoteImageDecoderAVF::create(*this, *imageDecoderIdentifier, data, mimeType);
53 m_remoteImageDecoders.add(*imageDecoderIdentifier, makeWeakPtr(remoteImageDecoder.get()));
54
55 return remoteImageDecoder;
56}
57
58void RemoteImageDecoderAVFManager::deleteRemoteImageDecoder(const ImageDecoderIdentifier& identifier)
59{
60 m_remoteImageDecoders.take(identifier);
61 if (m_gpuProcessConnection)
62 m_gpuProcessConnection->connection().send(Messages::RemoteImageDecoderAVFProxy::DeleteDecoder(identifier), 0);
63}

65RemoteImageDecoderAVFManager::RemoteImageDecoderAVFManager(WebProcess& process)
66 : m_process(process)
67{
68}

70RemoteImageDecoderAVFManager::~RemoteImageDecoderAVFManager()
71{
72 if (m_gpuProcessConnection)
73 m_gpuProcessConnection->messageReceiverMap().removeMessageReceiver(Messages::RemoteImageDecoderAVFManager::messageReceiverName());
74}
75
76void RemoteImageDecoderAVFManager::gpuProcessConnectionDidClose(GPUProcessConnection& connection)
77{
78 ASSERT(m_gpuProcessConnection == &connection);
79 connection.removeClient(*this);
80 m_gpuProcessConnection->messageReceiverMap().removeMessageReceiver(Messages::RemoteImageDecoderAVFManager::messageReceiverName());
81 m_gpuProcessConnection = nullptr;
82 // FIXME: Do we need to do more when m_remoteImageDecoders is not empty to re-create them?
83}
84
85const char* RemoteImageDecoderAVFManager::supplementName()
86{
87 return "RemoteImageDecoderAVFManager";
88}
89
90GPUProcessConnection& RemoteImageDecoderAVFManager::ensureGPUProcessConnection()
91{
92 if (!m_gpuProcessConnection) {
93 m_gpuProcessConnection = makeWeakPtr(m_process.ensureGPUProcessConnection());
94 m_gpuProcessConnection->addClient(*this);
95 m_gpuProcessConnection->messageReceiverMap().addMessageReceiver(Messages::RemoteImageDecoderAVFManager::messageReceiverName(), *this);
96 }
97 return *m_gpuProcessConnection;
98}
99
100void RemoteImageDecoderAVFManager::setUseGPUProcess(bool useGPUProcess)
101{
102 if (!useGPUProcess) {
103 ImageDecoder::resetFactories();
104 return;
105 }
106
107 ImageDecoder::clearFactories();
108 ImageDecoder::installFactory({
109 RemoteImageDecoderAVF::supportsMediaType,
110 RemoteImageDecoderAVF::canDecodeType,
111 [this](SharedBuffer& data, const String& mimeType, AlphaOption alphaOption, GammaAndColorProfileOption gammaAndColorProfileOption) {
112 return createImageDecoder(data, mimeType, alphaOption, gammaAndColorProfileOption);
113 }
114 });
115}
116
117void RemoteImageDecoderAVFManager::encodedDataStatusChanged(const ImageDecoderIdentifier& identifier, size_t frameCount, const WebCore::IntSize& size, bool hasTrack)
118{
119 if (!m_remoteImageDecoders.contains(identifier))
120 return;
121
122 auto remoteImageDecoder = m_remoteImageDecoders.get(identifier);
123 if (!remoteImageDecoder)
124 return;
125
126 remoteImageDecoder->encodedDataStatusChanged(frameCount, size, hasTrack);
127}
