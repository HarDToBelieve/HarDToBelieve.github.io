40std::unique_ptr<RemoteLegacyCDM> RemoteLegacyCDM::create(WeakPtr<RemoteLegacyCDMFactory>&& factory, RemoteLegacyCDMIdentifier identifier)
41{
42 return std::unique_ptr<RemoteLegacyCDM>(new RemoteLegacyCDM(WTFMove(factory), WTFMove(identifier)));

45RemoteLegacyCDM::RemoteLegacyCDM(WeakPtr<RemoteLegacyCDMFactory>&& factory, RemoteLegacyCDMIdentifier&& identifier)
46 : m_factory(WTFMove(factory))
47 , m_identifier(WTFMove(identifier))
48{
49}

51RemoteLegacyCDM::~RemoteLegacyCDM() = default;
52
53bool RemoteLegacyCDM::supportsMIMEType(const String& mimeType)
54{
55 if (!m_factory)
56 return false;
57
58 bool supported = false;
59 m_factory->gpuProcessConnection().connection().sendSync(Messages::RemoteLegacyCDMProxy::SupportsMIMEType(mimeType), Messages::RemoteLegacyCDMProxy::SupportsMIMEType::Reply(supported), m_identifier);
60 return supported;
61}
62
63std::unique_ptr<WebCore::LegacyCDMSession> RemoteLegacyCDM::createSession(WebCore::LegacyCDMSessionClient* client)
64{
65 if (!m_factory)
66 return nullptr;
67
68 String storageDirectory = client ? client->mediaKeysStorageDirectory() : emptyString();
69
70 RemoteLegacyCDMSessionIdentifier identifier;
71 m_factory->gpuProcessConnection().connection().sendSync(Messages::RemoteLegacyCDMProxy::CreateSession(storageDirectory), Messages::RemoteLegacyCDMProxy::CreateSession::Reply(identifier), m_identifier);
72 if (!identifier)
73 return nullptr;
74 return RemoteLegacyCDMSession::create(m_factory, WTFMove(identifier));
75}
76
77void RemoteLegacyCDM::setPlayerId(MediaPlayerIdentifier identifier)
78{
79 if (!m_factory)
80 return;
81
82 std::optional<MediaPlayerIdentifier> optionalId;
83 if (identifier)
84 optionalId = identifier;
85 m_factory->gpuProcessConnection().connection().send(Messages::RemoteLegacyCDMProxy::SetPlayerId(optionalId), m_identifier);
86}
