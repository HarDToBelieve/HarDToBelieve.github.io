43static RefPtr<ArrayBuffer> convertToArrayBuffer(IPC::SharedBufferCopy&& buffer)
44{
45 if (buffer.buffer())
46 return ArrayBuffer::create(buffer.data(), buffer.size());
47 return nullptr;
48}
49
50static RefPtr<Uint8Array> convertToUint8Array(IPC::SharedBufferCopy&& buffer)
51{
52 auto arrayBuffer = convertToArrayBuffer(WTFMove(buffer));
53 if (!arrayBuffer)
54 return nullptr;
55
56 size_t sizeInBytes = arrayBuffer->byteLength();
57 return Uint8Array::create(WTFMove(arrayBuffer), 0, sizeInBytes);
58}
59
60template <typename T>
61static IPC::SharedBufferCopy convertToSharedBufferCopy(T array)
62{
63 if (!array)
64 return { };
65 return { SharedBuffer::create(array->data(), array->byteLength()) };
66}

79RefPtr<Uint8Array> RemoteLegacyCDMSession::generateKeyRequest(const String& mimeType, Uint8Array* initData, String& destinationURL, unsigned short& errorCode, uint32_t& systemCode)
80{
81 if (!m_factory || !initData)
82 return nullptr;
83
84 auto ipcInitData = convertToSharedBufferCopy(initData);
85 std::optional<IPC::SharedBufferCopy> ipcNextMessage;
86 m_factory->gpuProcessConnection().connection().sendSync(Messages::RemoteLegacyCDMSessionProxy::GenerateKeyRequest(mimeType, ipcInitData), Messages::RemoteLegacyCDMSessionProxy::GenerateKeyRequest::Reply(ipcNextMessage, destinationURL, errorCode, systemCode), m_identifier);
87
88 if (!ipcNextMessage)
89 return nullptr;
90
91 return convertToUint8Array(WTFMove(*ipcNextMessage));
92}

94void RemoteLegacyCDMSession::releaseKeys()
95{
96 if (!m_factory)
97 return;
98
99 m_factory->gpuProcessConnection().connection().send(Messages::RemoteLegacyCDMSessionProxy::ReleaseKeys(), m_identifier);
100 m_cachedKeyCache.clear();
101}

103bool RemoteLegacyCDMSession::update(Uint8Array* keyData, RefPtr<Uint8Array>& nextMessage, unsigned short& errorCode, uint32_t& systemCode)
104{
105 if (!m_factory || !keyData)
106 return false;
107
108 auto ipcKeyData = convertToSharedBufferCopy(WTFMove(keyData));
109 bool succeeded { false };
110 std::optional<IPC::SharedBufferCopy> ipcNextMessage;
111 m_factory->gpuProcessConnection().connection().sendSync(Messages::RemoteLegacyCDMSessionProxy::Update(ipcKeyData), Messages::RemoteLegacyCDMSessionProxy::Update::Reply(succeeded, ipcNextMessage, errorCode, systemCode), m_identifier);
112
113 if (ipcNextMessage)
114 nextMessage = convertToUint8Array(WTFMove(*ipcNextMessage));
115
116 return succeeded;
117}

119RefPtr<ArrayBuffer> RemoteLegacyCDMSession::cachedKeyForKeyID(const String& keyId) const
120{
121 if (!m_factory)
122 return nullptr;
123
124 auto foundInCache = m_cachedKeyCache.find(keyId);
125 if (foundInCache != m_cachedKeyCache.end())
126 return foundInCache->value;
127
128 std::optional<IPC::SharedBufferCopy> ipcKey;
129 m_factory->gpuProcessConnection().connection().sendSync(Messages::RemoteLegacyCDMSessionProxy::CachedKeyForKeyID(keyId), Messages::RemoteLegacyCDMSessionProxy::CachedKeyForKeyID::Reply(ipcKey), m_identifier);
130
131 if (!ipcKey)
132 return nullptr;
133
134 auto ipcKeyBuffer = convertToArrayBuffer(WTFMove(*ipcKey));
135 m_cachedKeyCache.set(keyId, ipcKeyBuffer);
136 return ipcKeyBuffer;
137}

139void RemoteLegacyCDMSession::sendMessage(std::optional<IPC::SharedBufferCopy>&& message, const String& destinationURL)
140{
141 if (!m_client)
142 return;
143
144 if (!message) {
145 m_client->sendMessage(nullptr, destinationURL);
146 return;
147 }
148
149 m_client->sendMessage(convertToUint8Array(WTFMove(*message)).get(), destinationURL);
150}

155 m_client->sendError(errorCode, systemCode);
