38RemoteMediaPlayerMIMETypeCache::RemoteMediaPlayerMIMETypeCache(RemoteMediaPlayerManager& manager, MediaPlayerEnums::MediaEngineIdentifier engineIdentifier)
39 : m_manager(manager)
40 , m_engineIdentifier(engineIdentifier)
41{
42}
43
44void RemoteMediaPlayerMIMETypeCache::addSupportedTypes(const Vector<String>& newTypes)
45{
46 if (!m_supportedTypesCache)
47 m_supportedTypesCache = HashSet<String, ASCIICaseInsensitiveHash> { };
48
49 for (auto& type : newTypes)
50 m_supportedTypesCache->add(type);
51}
52
53bool RemoteMediaPlayerMIMETypeCache::isEmpty() const
54{
55 return m_supportedTypesCache && m_supportedTypesCache->isEmpty();
56}
57
58HashSet<String, ASCIICaseInsensitiveHash>& RemoteMediaPlayerMIMETypeCache::supportedTypes()
59{
60 if (m_supportedTypesCache)
61 return *m_supportedTypesCache;
62
63 Vector<String> types;
64 if (m_manager.gpuProcessConnection().connection().sendSync(Messages::RemoteMediaPlayerManagerProxy::GetSupportedTypes(m_engineIdentifier), Messages::RemoteMediaPlayerManagerProxy::GetSupportedTypes::Reply(types), 0))
65 addSupportedTypes(types);
66
67 return *m_supportedTypesCache;
68}
69
70MediaPlayerEnums::SupportsType RemoteMediaPlayerMIMETypeCache::supportsTypeAndCodecs(const MediaEngineSupportParameters& parameters)
71{
72 if (parameters.type.raw().isEmpty())
73 return MediaPlayerEnums::SupportsType::MayBeSupported;
74
75 SupportedTypesAndCodecsKey searchKey { parameters.type.raw(), parameters.isMediaSource, parameters.isMediaStream };
76
77 if (m_supportsTypeAndCodecsCache) {
78 auto it = m_supportsTypeAndCodecsCache->find(searchKey);
79 if (it != m_supportsTypeAndCodecsCache->end())
80 return it->value;
81 }
82
83 if (!m_supportsTypeAndCodecsCache)
84 m_supportsTypeAndCodecsCache = HashMap<SupportedTypesAndCodecsKey, MediaPlayerEnums::SupportsType> { };
85
86 MediaPlayerEnums::SupportsType result = MediaPlayerEnums::SupportsType::IsNotSupported;
87 if (m_manager.gpuProcessConnection().connection().sendSync(Messages::RemoteMediaPlayerManagerProxy::SupportsTypeAndCodecs(m_engineIdentifier, parameters), Messages::RemoteMediaPlayerManagerProxy::SupportsTypeAndCodecs::Reply(result), 0))
88 m_supportsTypeAndCodecsCache->add(searchKey, result);
89
90 return result;
91}
