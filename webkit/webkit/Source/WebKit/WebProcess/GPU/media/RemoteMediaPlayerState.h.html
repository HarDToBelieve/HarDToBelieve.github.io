39struct RemoteMediaPlayerState {
40 MediaTime duration;
41 MediaTime minTimeSeekable;
42 MediaTime maxTimeSeekable;
43 MediaTime startDate;
44 MediaTime startTime;
45 String languageOfPrimaryAudioTrack;
46 String wirelessPlaybackTargetName;
47 WebCore::PlatformTimeRanges bufferedRanges;
48 WebCore::MediaPlayerEnums::NetworkState networkState { WebCore::MediaPlayerEnums::NetworkState::Empty };
49 WebCore::MediaPlayerEnums::ReadyState readyState { WebCore::MediaPlayerEnums::ReadyState::HaveNothing };
50 WebCore::MediaPlayerEnums::MovieLoadType movieLoadType { WebCore::MediaPlayerEnums::MovieLoadType::Unknown };
51 WebCore::MediaPlayerEnums::WirelessPlaybackTargetType wirelessPlaybackTargetType { WebCore::MediaPlayerEnums::WirelessPlaybackTargetType::TargetTypeNone };
52 WebCore::FloatSize naturalSize;
53 double maxFastForwardRate { 0 };
54 double minFastReverseRate { 0 };
55 double seekableTimeRangesLastModifiedTime { 0 };
56 double liveUpdateInterval { 0 };
57 uint64_t totalBytes { 0 };
58 std::optional<WebCore::VideoPlaybackQualityMetrics> videoMetrics;
59 std::optional<bool> wouldTaintDocumentSecurityOrigin { true };
60 bool paused { true };
61 bool canSaveMediaData { false };
62 bool hasAudio { false };
63 bool hasVideo { false };
64 bool hasClosedCaptions { false };
65 bool hasAvailableVideoFrame { false };
66 bool wirelessVideoPlaybackDisabled { false };
67 bool hasSingleSecurityOrigin { false };
68 bool didPassCORSAccessCheck { false };

73 encoder << duration;
74 encoder << minTimeSeekable;
75 encoder << maxTimeSeekable;

77 encoder << startTime;
78 encoder << languageOfPrimaryAudioTrack;
79 encoder << wirelessPlaybackTargetName;
80 encoder << bufferedRanges;
81 encoder << networkState;
82 encoder << readyState;
83 encoder << movieLoadType;
84 encoder << wirelessPlaybackTargetType;
85 encoder << naturalSize;
86 encoder << maxFastForwardRate;
87 encoder << minFastReverseRate;
88 encoder << seekableTimeRangesLastModifiedTime;
89 encoder << liveUpdateInterval;
90 encoder << totalBytes;
91 encoder << videoMetrics;
92 encoder << wouldTaintDocumentSecurityOrigin;
93 encoder << paused;
94 encoder << canSaveMediaData;
95 encoder << hasAudio;
96 encoder << hasVideo;
97 encoder << hasClosedCaptions;
98 encoder << hasAvailableVideoFrame;
99 encoder << wirelessVideoPlaybackDisabled;
100 encoder << hasSingleSecurityOrigin;
101 encoder << didPassCORSAccessCheck;
102 }
103
104 template <class Decoder>
105 static std::optional<RemoteMediaPlayerState> decode(Decoder& decoder)
106 {
107 std::optional<MediaTime> duration;
108 decoder >> duration;
109 if (!duration)
110 return std::nullopt;
111
112 std::optional<MediaTime> minTimeSeekable;
113 decoder >> minTimeSeekable;
114 if (!minTimeSeekable)
115 return std::nullopt;
116
117 std::optional<MediaTime> maxTimeSeekable;
118 decoder >> maxTimeSeekable;
119 if (!maxTimeSeekable)
120 return std::nullopt;
121
122 std::optional<MediaTime> startDate;
123 decoder >> startDate;
124 if (!startDate)
125 return std::nullopt;
126
127 std::optional<MediaTime> startTime;
128 decoder >> startTime;
129 if (!startTime)
130 return std::nullopt;
131
132 std::optional<String> languageOfPrimaryAudioTrack;
133 decoder >> languageOfPrimaryAudioTrack;
134 if (!languageOfPrimaryAudioTrack)
135 return std::nullopt;
136
137 std::optional<String> wirelessPlaybackTargetName;
138 decoder >> wirelessPlaybackTargetName;
139 if (!wirelessPlaybackTargetName)
140 return std::nullopt;
141
142 std::optional<WebCore::PlatformTimeRanges> bufferedRanges;
143 decoder >> bufferedRanges;
144 if (!bufferedRanges)
145 return std::nullopt;
146
147 WebCore::MediaPlayerEnums::NetworkState networkState;
148 if (!decoder.decode(networkState))
149 return std::nullopt;
150
151 WebCore::MediaPlayerEnums::ReadyState readyState;
152 if (!decoder.decode(readyState))
153 return std::nullopt;
154
155 WebCore::MediaPlayerEnums::MovieLoadType movieLoadType;
156 if (!decoder.decode(movieLoadType))
157 return std::nullopt;
158
159 WebCore::MediaPlayerEnums::WirelessPlaybackTargetType wirelessPlaybackTargetType;
160 if (!decoder.decode(wirelessPlaybackTargetType))
161 return std::nullopt;
162
163 std::optional<WebCore::FloatSize> naturalSize;
164 decoder >> naturalSize;
165 if (!naturalSize)
166 return std::nullopt;
167
168 std::optional<double> maxFastForwardRate;
169 decoder >> maxFastForwardRate;
170 if (!maxFastForwardRate)
171 return std::nullopt;
172
173 std::optional<double> minFastReverseRate;
174 decoder >> minFastReverseRate;
175 if (!minFastReverseRate)
176 return std::nullopt;
177
178 std::optional<double> seekableTimeRangesLastModifiedTime;
179 decoder >> seekableTimeRangesLastModifiedTime;
180 if (!seekableTimeRangesLastModifiedTime)
181 return std::nullopt;
182
183 std::optional<double> liveUpdateInterval;
184 decoder >> liveUpdateInterval;
185 if (!liveUpdateInterval)
186 return std::nullopt;
187
188 std::optional<uint64_t> totalBytes;
189 decoder >> totalBytes;
190 if (!totalBytes)
191 return std::nullopt;
192
193 std::optional<std::optional<WebCore::VideoPlaybackQualityMetrics>> videoMetrics;
194 decoder >> videoMetrics;
195 if (!videoMetrics)
196 return std::nullopt;
197
198 std::optional<std::optional<bool>> wouldTaintDocumentSecurityOrigin;
199 decoder >> wouldTaintDocumentSecurityOrigin;
200 if (!wouldTaintDocumentSecurityOrigin)
201 return std::nullopt;
202
203 std::optional<bool> paused;
204 decoder >> paused;
205 if (!paused)
206 return std::nullopt;
207
208 std::optional<bool> canSaveMediaData;
209 decoder >> canSaveMediaData;
210 if (!canSaveMediaData)
211 return std::nullopt;
212
213 std::optional<bool> hasAudio;
214 decoder >> hasAudio;
215 if (!hasAudio)
216 return std::nullopt;
217
218 std::optional<bool> hasVideo;
219 decoder >> hasVideo;
220 if (!hasVideo)
221 return std::nullopt;
222
223 std::optional<bool> hasClosedCaptions;
224 decoder >> hasClosedCaptions;
225 if (!hasClosedCaptions)
226 return std::nullopt;
227
228 std::optional<bool> hasAvailableVideoFrame;
229 decoder >> hasAvailableVideoFrame;
230 if (!hasAvailableVideoFrame)
231 return std::nullopt;
232
233 std::optional<bool> wirelessVideoPlaybackDisabled;
234 decoder >> wirelessVideoPlaybackDisabled;
235 if (!wirelessVideoPlaybackDisabled)
236 return std::nullopt;
237
238 std::optional<bool> hasSingleSecurityOrigin;
239 decoder >> hasSingleSecurityOrigin;
240 if (!hasSingleSecurityOrigin)
241 return std::nullopt;
242
243 std::optional<bool> didPassCORSAccessCheck;
244 decoder >> didPassCORSAccessCheck;
245 if (!didPassCORSAccessCheck)
246 return std::nullopt;
247
248 return {{
249 WTFMove(*duration),
250 WTFMove(*minTimeSeekable),
251 WTFMove(*maxTimeSeekable),
252 WTFMove(*startDate),
253 WTFMove(*startTime),
254 WTFMove(*languageOfPrimaryAudioTrack),
255 WTFMove(*wirelessPlaybackTargetName),
256 WTFMove(*bufferedRanges),
257 networkState,
258 readyState,
259 movieLoadType,
260 wirelessPlaybackTargetType,
261 WTFMove(*naturalSize),
262 *maxFastForwardRate,
263 *minFastReverseRate,
264 *seekableTimeRangesLastModifiedTime,
265 *liveUpdateInterval,
266 *totalBytes,
267 WTFMove(*videoMetrics),
268 WTFMove(*wouldTaintDocumentSecurityOrigin),
269 *paused,
270 *canSaveMediaData,
271 *hasAudio,
272 *hasVideo,
273 *hasClosedCaptions,
274 *hasAvailableVideoFrame,
275 *wirelessVideoPlaybackDisabled,
276 *hasSingleSecurityOrigin,
277 *didPassCORSAccessCheck
278 }};
279 }
