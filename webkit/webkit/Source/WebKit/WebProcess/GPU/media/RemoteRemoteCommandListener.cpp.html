41std::unique_ptr<RemoteRemoteCommandListener> RemoteRemoteCommandListener::create(RemoteCommandListenerClient& client, WebProcess& webProcess)
42{
43 return makeUnique<RemoteRemoteCommandListener>(client, webProcess);
44}
45
46RemoteRemoteCommandListener::RemoteRemoteCommandListener(RemoteCommandListenerClient& client, WebProcess& webProcess)
47 : RemoteCommandListener(client)
48 , m_process(webProcess)
49 , m_identifier(RemoteRemoteCommandListenerIdentifier::generate())
50{
51 auto& connection = m_process.ensureGPUProcessConnection();
52 connection.addClient(*this);
53 connection.messageReceiverMap().addMessageReceiver(Messages::RemoteRemoteCommandListener::messageReceiverName(), m_identifier.toUInt64(), *this);
54 connection.connection().send(Messages::GPUConnectionToWebProcess::CreateRemoteCommandListener(m_identifier), { });
55}
56
57RemoteRemoteCommandListener::~RemoteRemoteCommandListener()
58{
59 if (auto* gpuProcessConnection = m_process.existingGPUProcessConnection()) {
60 gpuProcessConnection->messageReceiverMap().removeMessageReceiver(*this);
61 gpuProcessConnection->connection().send(Messages::GPUConnectionToWebProcess::ReleaseRemoteCommandListener(m_identifier), 0);
62 }
63}

75void RemoteRemoteCommandListener::updateSupportedCommands()
76{
77 auto& supportedCommands = this->supportedCommands();
78 if (m_currentCommands == supportedCommands && m_currentSupportSeeking == supportsSeeking())
79 return;
80
81 m_currentCommands = supportedCommands;
82 m_currentSupportSeeking = supportsSeeking();
83
84 Vector<PlatformMediaSession::RemoteControlCommandType> commands;
85 commands.reserveInitialCapacity(supportedCommands.size());
86 for (auto command : supportedCommands)
87 commands.uncheckedAppend(command);
88
89 m_process.ensureGPUProcessConnection().connection().send(Messages::RemoteRemoteCommandListenerProxy::UpdateSupportedCommands { WTFMove(commands), m_currentSupportSeeking }, m_identifier);
90}
