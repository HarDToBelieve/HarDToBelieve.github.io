40TextTrackPrivateRemote::TextTrackPrivateRemote(GPUProcessConnection& gpuProcessConnection, MediaPlayerIdentifier playerIdentifier, TrackPrivateRemoteIdentifier idendifier, TextTrackPrivateRemoteConfiguration&& configuration)
41 : WebCore::InbandTextTrackPrivate(configuration.cueFormat)
42 , m_gpuProcessConnection(makeWeakPtr(gpuProcessConnection))
43 , m_playerIdentifier(playerIdentifier)
44 , m_identifier(idendifier)
45{
46 updateConfiguration(WTFMove(configuration));
47}

49void TextTrackPrivateRemote::setMode(TextTrackMode mode)
50{
51 if (!m_gpuProcessConnection)
52 return;
53
54 if (mode == InbandTextTrackPrivate::mode())
55 return;
56
57 m_gpuProcessConnection->connection().send(Messages::RemoteMediaPlayerProxy::TextTrackSetMode(m_identifier, mode), m_playerIdentifier);
58 InbandTextTrackPrivate::setMode(mode);
59}

61void TextTrackPrivateRemote::updateConfiguration(TextTrackPrivateRemoteConfiguration&& configuration)
62{
63 if (configuration.trackId != m_id) {
64 auto changed = !m_id.isEmpty();
65 m_id = configuration.trackId;
66 if (changed && client())
67 client()->idChanged(m_id);
68 }
69
70 if (configuration.label != m_label) {
71 auto changed = !m_label.isEmpty();
72 m_label = configuration.label;
73 if (changed && client())
74 client()->labelChanged(m_label);
75 }
76
77 if (configuration.language != m_language) {
78 auto changed = !m_language.isEmpty();
79 m_language = configuration.language;
80 if (changed && client())
81 client()->languageChanged(m_language);
82 }
83
84 m_trackIndex = configuration.trackIndex;
85 m_inBandMetadataTrackDispatchType = configuration.inBandMetadataTrackDispatchType;
86 m_startTimeVariance = configuration.startTimeVariance;
87
88 m_kind = configuration.kind;
89 m_isClosedCaptions = configuration.isClosedCaptions;
90 m_isSDH = configuration.isSDH;
91 m_containsOnlyForcedSubtitles = configuration.containsOnlyForcedSubtitles;
92 m_isMainProgramContent = configuration.isMainProgramContent;
93 m_isEasyToRead = configuration.isEasyToRead;
94 m_isDefault = configuration.isDefault;
95}
