42class VideoTrackPrivateRemote
43 : public WebCore::VideoTrackPrivate {
44 WTF_MAKE_NONCOPYABLE(VideoTrackPrivateRemote)
45public:
46 static Ref<VideoTrackPrivateRemote> create(GPUProcessConnection& gpuProcessConnection, WebCore::MediaPlayerIdentifier playerIdentifier, TrackPrivateRemoteIdentifier idendifier, TrackPrivateRemoteConfiguration&& configuration)

48 return adoptRef(*new VideoTrackPrivateRemote(gpuProcessConnection, playerIdentifier, idendifier, WTFMove(configuration)));
49 }
50
51 void updateConfiguration(TrackPrivateRemoteConfiguration&&);
52
53 using VideoTrackKind = WebCore::VideoTrackPrivate::Kind;
54 VideoTrackKind kind() const final { return m_kind; }
55 AtomString id() const final { return m_id; }
56 AtomString label() const final { return m_label; }
57 AtomString language() const final { return m_language; }
58 int trackIndex() const final { return m_trackIndex; }
59 MediaTime startTimeVariance() const final { return m_startTimeVariance; }
60
61private:
62 VideoTrackPrivateRemote(GPUProcessConnection&, WebCore::MediaPlayerIdentifier, TrackPrivateRemoteIdentifier, TrackPrivateRemoteConfiguration&&);
63
64 void setSelected(bool) final;
65
66 WeakPtr<GPUProcessConnection> m_gpuProcessConnection;
67 WebCore::MediaPlayerIdentifier m_playerIdentifier;
68 VideoTrackKind m_kind { None };
69 AtomString m_id;
70 AtomString m_label;
71 AtomString m_language;
72 int m_trackIndex { -1 };
73 MediaTime m_startTimeVariance { MediaTime::zeroTime() };
74 TrackPrivateRemoteIdentifier m_idendifier;
75};
