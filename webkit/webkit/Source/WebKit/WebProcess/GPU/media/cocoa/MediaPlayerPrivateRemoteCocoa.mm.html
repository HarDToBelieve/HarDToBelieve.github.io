65RefPtr<NativeImage> MediaPlayerPrivateRemote::nativeImageForCurrentTime()
66{
67 std::optional<MachSendRight> sendRight;
68 if (!connection().sendSync(Messages::RemoteMediaPlayerProxy::NativeImageForCurrentTime(), Messages::RemoteMediaPlayerProxy::NativeImageForCurrentTime::Reply(sendRight), m_id))
69 return nullptr;
70
71 if (!sendRight)
72 return nullptr;
73
74 auto surface = WebCore::IOSurface::createFromSendRight(WTFMove(*sendRight), WebCore::DestinationColorSpace::SRGB());
75 if (!surface)
76 return nullptr;
77
78 auto platformImage = WebCore::IOSurface::sinkIntoImage(WTFMove(surface));
79 if (!platformImage)
80 return nullptr;
81
82 return NativeImage::create(WTFMove(platformImage));
83}

85RetainPtr<CVPixelBufferRef> MediaPlayerPrivateRemote::pixelBufferForCurrentTime()
86{
87
88 RetainPtr<CVPixelBufferRef> result;
89 if (!connection().sendSync(Messages::RemoteMediaPlayerProxy::PixelBufferForCurrentTime(), Messages::RemoteMediaPlayerProxy::PixelBufferForCurrentTime::Reply(result), m_id))
90 return nullptr;
91 return result;
92}
