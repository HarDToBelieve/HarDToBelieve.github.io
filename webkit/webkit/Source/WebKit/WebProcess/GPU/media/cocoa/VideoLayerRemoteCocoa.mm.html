41// We want to wait for a short time after the completion of the animation (we choose 100 ms here) to fire the timer
42// to avoid excessive XPC messages from the Web process to the GPU process.
43static const Seconds PostAnimationDelay { 100_ms };
44
45@implementation WKVideoLayerRemote {
46 WeakPtr<WebKit::MediaPlayerPrivateRemote> _mediaPlayerPrivateRemote;
47 RetainPtr<CAContext> _context;
48 WebCore::MediaPlayerEnums::VideoGravity _videoGravity;
49
50 std::unique_ptr<WebCore::Timer> _resolveBoundsTimer;
51 bool _shouldRestartWhenTimerFires;
52 Seconds _delay;
53}
54
55- (instancetype)init
56{
57 self = [super init];
58 if (!self)
59 return nil;
60
61 self.masksToBounds = YES;
62 _resolveBoundsTimer = makeUnique<WebCore::Timer>([weakSelf = WeakObjCPtr<WKVideoLayerRemote>(self)] {
63 auto localSelf = weakSelf.get();
64 if (!localSelf)
65 return;
66
67 [localSelf resolveBounds];
68 });
69 _shouldRestartWhenTimerFires = false;
70
71 return self;
72}
73
74- (WebKit::MediaPlayerPrivateRemote*)mediaPlayerPrivateRemote
75{
76 return _mediaPlayerPrivateRemote.get();
77}

91 _videoGravity = videoGravity;

94- (bool)resizePreservingGravity
95{
96 auto* player = self.mediaPlayerPrivateRemote;
97 if (player && player->inVideoFullscreenOrPictureInPicture())
98 return true;
99
100 return _videoGravity != WebCore::MediaPlayer::VideoGravity::Resize;
101}
102
103- (void)layoutSublayers
104{
105 auto* sublayers = [self sublayers];
106
107 if ([sublayers count] != 1) {
108 ASSERT_NOT_REACHED();
109 return;
110 }
111
112 WebCore::FloatRect sourceVideoFrame = self.videoLayerFrame;
113 WebCore::FloatRect targetVideoFrame = self.bounds;
114 CGAffineTransform transform = CGAffineTransformIdentity;
115 if (!sourceVideoFrame.isEmpty()) {
116 if ([self resizePreservingGravity]) {
117 auto scale = std::fmax(targetVideoFrame.width() / sourceVideoFrame.width(), targetVideoFrame.height() / sourceVideoFrame.height());
118 transform = CGAffineTransformMakeScale(scale, scale);
119 } else
120 transform = CGAffineTransformMakeScale(targetVideoFrame.width() / sourceVideoFrame.width(), targetVideoFrame.height() / sourceVideoFrame.height());
121 }
122
123 auto* videoSublayer = [sublayers objectAtIndex:0];
124 [CATransaction begin];
125 [CATransaction setDisableActions:YES];
126 [videoSublayer setPosition:CGPointMake(CGRectGetMidX(self.bounds), CGRectGetMidY(self.bounds))];
127 [videoSublayer setAffineTransform:transform];
128 [CATransaction commit];
129
130 _context = [CAContext currentContext];
131 NSTimeInterval animationDuration = [CATransaction animationDuration];
132
133 _delay = Seconds(animationDuration) + PostAnimationDelay;
134 if (_resolveBoundsTimer->isActive()) {
135 _shouldRestartWhenTimerFires = true;
136 _delay -= _resolveBoundsTimer->nextFireInterval();
137 return;
138 }
139 _resolveBoundsTimer->startOneShot(_delay);
140}
141
142- (void)resolveBounds
143{
144 if (_shouldRestartWhenTimerFires) {
145 _shouldRestartWhenTimerFires = false;
146 _resolveBoundsTimer->startOneShot(_delay);
147 return;
148 }
149
150 auto* sublayers = [self sublayers];
151 if ([sublayers count] != 1) {
152 ASSERT_NOT_REACHED();
153 return;
154 }
155
156 if (CGRectEqualToRect(self.videoLayerFrame, self.bounds) && CGAffineTransformIsIdentity(self.affineTransform))
157 return;
158
159 [CATransaction begin];
160 [CATransaction setDisableActions:YES];
161
162 if (!CGRectEqualToRect(self.videoLayerFrame, self.bounds)) {
163 self.videoLayerFrame = self.bounds;
164 if (auto* mediaPlayerPrivateRemote = self.mediaPlayerPrivateRemote) {
165 MachSendRight fenceSendRight = MachSendRight::adopt([_context createFencePort]);
166 mediaPlayerPrivateRemote->setVideoInlineSizeFenced(WebCore::FloatSize(self.videoLayerFrame.size), fenceSendRight);
167 }
168 }
169
170 auto* videoSublayer = [sublayers objectAtIndex:0];
171 [videoSublayer setAffineTransform:CGAffineTransformIdentity];
172 [videoSublayer setFrame:self.bounds];
173
174 [CATransaction commit];
175}
176
177@end

181PlatformLayerContainer createVideoLayerRemote(MediaPlayerPrivateRemote* mediaPlayerPrivateRemote, LayerHostingContextID contextId, WebCore::MediaPlayerEnums::VideoGravity videoGravity)
182{
183 // Initially, all the layers will be empty (both width and height are 0) and invisible.
184 // The renderer will change the sizes of WKVideoLayerRemote to trigger layout of sublayers and make them visible.
185 auto videoLayerRemote = adoptNS([[WKVideoLayerRemote alloc] init]);
186 [videoLayerRemote setName:@"WKVideoLayerRemote"];
187 [videoLayerRemote setVideoGravity:videoGravity];
188 [videoLayerRemote setMediaPlayerPrivateRemote:mediaPlayerPrivateRemote];
189 [videoLayerRemote addSublayer:LayerHostingContext::createPlatformLayerForHostingContext(contextId).get()];
190
191 return videoLayerRemote;
192}
