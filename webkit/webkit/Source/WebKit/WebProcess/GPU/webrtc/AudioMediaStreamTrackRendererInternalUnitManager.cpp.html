119AudioMediaStreamTrackRendererInternalUnitManager::Proxy::~Proxy()
120{
121 WebProcess::singleton().audioMediaStreamTrackRendererInternalUnitManager().remove(*this);
122 WebProcess::singleton().ensureGPUProcessConnection().connection().send(Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::DeleteUnit { m_identifier }, 0);
123
124 while (!m_descriptionCallbacks.isEmpty())
125 m_descriptionCallbacks.takeFirst()(nullptr);
126}

136void AudioMediaStreamTrackRendererInternalUnitManager::Proxy::initialize(const WebCore::CAAudioStreamDescription& description, size_t frameChunkSize)
137{
138 if (m_semaphore)
139 stopThread();
140
141 m_semaphore = makeUnique<IPC::Semaphore>();
142 m_description = description;
143 m_frameChunkSize = frameChunkSize;
144
145 while (!m_descriptionCallbacks.isEmpty())
146 m_descriptionCallbacks.takeFirst()(&description);
147
148 if (m_isPlaying)
149 start();
150}
151
152void AudioMediaStreamTrackRendererInternalUnitManager::Proxy::start()
153{
154 if (!m_description) {
155 m_isPlaying = true;
156 return;
157 }
158
159 if (m_didClose) {
160 m_didClose = false;
161 createRemoteUnit();
162 if (!m_deviceId.isEmpty())
163 setAudioOutputDevice(m_deviceId);
164 m_isPlaying = true;
165 return;
166 }
167
168 stopThread();
169
170 m_isPlaying = true;
171
172 m_numberOfFrames = m_description->sampleRate() * 2;
173 m_ringBuffer.reset();
174 m_ringBuffer = makeUnique<CARingBuffer>(makeUniqueRef<SharedRingBufferStorage>(std::bind(&AudioMediaStreamTrackRendererInternalUnitManager::Proxy::storageChanged, this, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3)));
175 m_ringBuffer->allocate(m_description->streamDescription(), m_numberOfFrames);
176
177 m_buffer = makeUnique<WebAudioBufferList>(*m_description, m_numberOfFrames);
178 m_buffer->setSampleCount(m_frameChunkSize);
179
180 startThread();
181}

183void AudioMediaStreamTrackRendererInternalUnitManager::Proxy::storageChanged(SharedMemory* memory, const CAAudioStreamDescription& format, size_t frameCount)
184{
185 if (!frameCount)
186 return;
187
188 SharedMemory::Handle handle;
189 if (memory)
190 memory->createHandle(handle, SharedMemory::Protection::ReadOnly);
191
192 // FIXME: Send the actual data size with IPCHandle.
193#if OS(DARWIN) || OS(WINDOWS)
194 uint64_t dataSize = handle.size();
195#else
196 uint64_t dataSize = 0;
197#endif
198 WebProcess::singleton().ensureGPUProcessConnection().connection().send(Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::StartUnit { m_identifier, SharedMemory::IPCHandle { WTFMove(handle), dataSize }, format, frameCount, *m_semaphore }, 0);
199}

213void AudioMediaStreamTrackRendererInternalUnitManager::Proxy::retrieveFormatDescription(CompletionHandler<void(const WebCore::CAAudioStreamDescription*)>&& callback)
214{
215 if (!m_description || !m_descriptionCallbacks.isEmpty()) {
216 m_descriptionCallbacks.append(WTFMove(callback));
217 return;

219 callback(&m_description.value());
220}

222void AudioMediaStreamTrackRendererInternalUnitManager::Proxy::stopThread()
223{
224 if (!m_thread)
225 return;
226
227 m_shouldStopThread = true;
228 m_semaphore->signal();
229 m_thread->waitForCompletion();
230 m_thread = nullptr;
231}
232
233void AudioMediaStreamTrackRendererInternalUnitManager::Proxy::startThread()
234{
235 ASSERT(!m_thread);
236 m_shouldStopThread = false;
237 auto threadLoop = [this]() mutable {
238 m_writeOffset = 0;
239 do {
240 // If wait fails, the semaphore on the other side was probably destroyed and we should just exit here and wait to launch a new thread.
241 if (!m_semaphore->wait())
242 break;
243 if (m_shouldStopThread)
244 break;
245
246 auto& bufferList = *m_buffer->list();
247
248 AudioUnitRenderActionFlags flags = 0;
249 m_renderCallback(m_frameChunkSize, bufferList, m_writeOffset, mach_absolute_time(), flags);
250
251 if (flags == kAudioUnitRenderAction_OutputIsSilence)
252 AudioSampleBufferList::zeroABL(bufferList, static_cast<size_t>(m_frameChunkSize * m_description->bytesPerFrame()));
253
254 m_ringBuffer->store(&bufferList, m_frameChunkSize, m_writeOffset);
255 m_writeOffset += m_frameChunkSize;
256 } while (!m_shouldStopThread);
257 };
258 m_thread = Thread::create("AudioMediaStreamTrackRendererInternalUnit thread", WTFMove(threadLoop), ThreadType::Audio, Thread::QOS::UserInteractive);
259}
260
261void AudioMediaStreamTrackRendererInternalUnitManager::Proxy::restartIfNeeded()
262{
263 stopThread();
264 m_didClose = true;
265 if (m_isPlaying)
266 start();
267}
