51static webrtc::WebKitVideoDecoder createVideoDecoder(const webrtc::SdpVideoFormat& format)
52{
53 auto& codecs = WebProcess::singleton().libWebRTCCodecs();
54 if (format.name == "H264")
55 return codecs.createDecoder(LibWebRTCCodecs::Type::H264);
56
57 if (format.name == "H265")
58 return codecs.createDecoder(LibWebRTCCodecs::Type::H265);
59
60 if (format.name == "VP9" && codecs.supportVP9VTB())
61 return codecs.createDecoder(LibWebRTCCodecs::Type::VP9);
62
63 return nullptr;
64}

103static inline MediaSample::VideoRotation toMediaSampleVideoRotation(webrtc::VideoRotation rotation)
104{
105 switch (rotation) {
106 case webrtc::kVideoRotation_0:
107 return MediaSample::VideoRotation::None;
108 case webrtc::kVideoRotation_180:
109 return MediaSample::VideoRotation::UpsideDown;
110 case webrtc::kVideoRotation_90:
111 return MediaSample::VideoRotation::Right;
112 case webrtc::kVideoRotation_270:
113 return MediaSample::VideoRotation::Left;
114 }
115 ASSERT_NOT_REACHED();
116 return MediaSample::VideoRotation::None;

177void LibWebRTCCodecs::ensureGPUProcessConnectionOnMainThreadWithLock()
178{
179 ASSERT(isMainRunLoop());
180 if (m_connection)
181 return;
182
183 auto& gpuConnection = WebProcess::singleton().ensureGPUProcessConnection();
184 gpuConnection.addClient(*this);
185 m_connection = makeRef(gpuConnection.connection());
186 m_connection->addThreadMessageReceiver(Messages::LibWebRTCCodecs::messageReceiverName(), this);
187
188 if (m_loggingLevel)
189 m_connection->send(Messages::LibWebRTCCodecsProxy::SetRTCLoggingLevel(*m_loggingLevel), 0);
190}

192// May be called on any thread.
193void LibWebRTCCodecs::ensureGPUProcessConnectionAndDispatchToThread(Function<void()>&& task)
194{
195 m_needsGPUProcessConnection = true;
196
197 Locker locker { m_connectionLock };
198
199 // Fast path when we already have a connection.
200 if (m_connection) {
201 dispatchToThread(WTFMove(task));
202 return;
203 }
204
205 // We don't have a connection to the GPUProcess yet, we need to hop to the main thread to initiate it.
206 m_tasksToDispatchAfterEstablishingConnection.append(WTFMove(task));
207 if (m_tasksToDispatchAfterEstablishingConnection.size() != 1)
208 return;
209
210 callOnMainRunLoop([this] {
211 Locker locker { m_connectionLock };
212 ensureGPUProcessConnectionOnMainThreadWithLock();
213 for (auto& task : std::exchange(m_tasksToDispatchAfterEstablishingConnection, { }))
214 dispatchToThread(WTFMove(task));
215 });
216}

230void LibWebRTCCodecs::setCallbacks(bool useGPUProcess)
231{
232 ASSERT(isMainRunLoop());
233
234 if (!useGPUProcess) {
235 webrtc::setVideoDecoderCallbacks(nullptr, nullptr, nullptr, nullptr);
236 webrtc::setVideoEncoderCallbacks(nullptr, nullptr, nullptr, nullptr, nullptr, nullptr);
237 return;
238 }
239
240 // Let's create WebProcess libWebRTCCodecs since it may be called from various threads.
241 WebProcess::singleton().libWebRTCCodecs();
242
243#if ENABLE(VP9)
244 // FIMXE: We should disable VP9VTB if VP9 hardware decoding is enabled but there is no support for it.
245 WebProcess::singleton().libWebRTCCodecs().setVP9VTBSupport(PlatformMediaSessionManager::shouldEnableVP9Decoder() || PlatformMediaSessionManager::shouldEnableVP9SWDecoder());
246#endif
247
248 webrtc::setVideoDecoderCallbacks(createVideoDecoder, releaseVideoDecoder, decodeVideoFrame, registerDecodeCompleteCallback);
249 webrtc::setVideoEncoderCallbacks(createVideoEncoder, releaseVideoEncoder, initializeVideoEncoder, encodeVideoFrame, registerEncodeCompleteCallback, setEncodeRatesCallback);
250}

252LibWebRTCCodecs::Decoder* LibWebRTCCodecs::createDecoder(Type type)
253{
254 auto decoder = makeUnique<Decoder>();
255 auto* result = decoder.get();
256 decoder->identifier = RTCDecoderIdentifier::generateThreadSafe();
257 decoder->type = type;
258
259 ensureGPUProcessConnectionAndDispatchToThread([this, decoder = WTFMove(decoder)]() mutable {
260 Locker locker { m_connectionLock };
261 decoder->connection = m_connection;
262 createRemoteDecoder(*decoder, *m_connection);
263
264 auto decoderIdentifier = decoder->identifier;
265 ASSERT(!m_decoders.contains(decoderIdentifier));
266 m_decoders.add(decoderIdentifier, WTFMove(decoder));
267 });
268 return result;
269}

271int32_t LibWebRTCCodecs::releaseDecoder(Decoder& decoder)
272{
273#if ASSERT_ENABLED
274 {
275 Locker locker { decoder.decodedImageCallbackLock };
276 ASSERT(!decoder.decodedImageCallback);
277 }
278#endif
279 ensureGPUProcessConnectionAndDispatchToThread([this, decoderIdentifier = decoder.identifier] {
280 ASSERT(m_decoders.contains(decoderIdentifier));
281 if (auto decoder = m_decoders.take(decoderIdentifier)) {
282 decoder->connection->send(Messages::LibWebRTCCodecsProxy::ReleaseDecoder { decoderIdentifier }, 0);
283 gpuProcessConnectionMayNoLongerBeNeeded();
284 }
285 });
286 return 0;

289int32_t LibWebRTCCodecs::decodeFrame(Decoder& decoder, uint32_t timeStamp, const uint8_t* data, size_t size, uint16_t width, uint16_t height)
290{
291 Locker locker { m_connectionLock };
292 if (!decoder.connection || decoder.hasError) {
293 decoder.hasError = false;
294 return WEBRTC_VIDEO_CODEC_ERROR;
295 }
296
297 if (decoder.type == Type::VP9 && (width || height))
298 decoder.connection->send(Messages::LibWebRTCCodecsProxy::SetFrameSize { decoder.identifier, width, height }, 0);
299
300 decoder.connection->send(Messages::LibWebRTCCodecsProxy::DecodeFrame { decoder.identifier, timeStamp, IPC::DataReference { data, size } }, 0);
301 return WEBRTC_VIDEO_CODEC_OK;
302}

318void LibWebRTCCodecs::completedDecoding(RTCDecoderIdentifier decoderIdentifier, uint32_t timeStamp, WebCore::RemoteVideoSample&& remoteSample)
319{
320 ASSERT(!isMainRunLoop());
321
322 // FIXME: Do error logging.
323 auto* decoder = m_decoders.get(decoderIdentifier);
324 if (!decoder)
325 return;
326
327 if (!decoder->decodedImageCallbackLock.tryLock())
328 return;
329
330 Locker locker { AdoptLock, decoder->decodedImageCallbackLock };
331
332 if (!decoder->decodedImageCallback)
333 return;
334
335 if (!m_imageTransferSession || m_imageTransferSession->pixelFormat() != remoteSample.videoFormat())
336 m_imageTransferSession = WebCore::ImageTransferSessionVT::create(remoteSample.videoFormat());
337
338 if (!m_imageTransferSession) {
339 ASSERT_NOT_REACHED();
340 return;
341 }
342
343 auto pixelBuffer = m_imageTransferSession->createPixelBuffer(remoteSample.surface(), remoteSample.size());
344 if (!pixelBuffer) {
345 ASSERT_NOT_REACHED();
346 return;
347 }
348
349 webrtc::videoDecoderTaskComplete(decoder->decodedImageCallback, timeStamp, pixelBuffer.get(), remoteSample.time().toDouble());
350}

376LibWebRTCCodecs::Encoder* LibWebRTCCodecs::createEncoder(Type type, const std::map<std::string, std::string>& formatParameters)
377{
378 auto encoder = makeUnique<Encoder>();
379 auto* result = encoder.get();
380 encoder->identifier = RTCEncoderIdentifier::generateThreadSafe();
381 encoder->codecType = toWebRTCCodecType(type);
382
383 Vector<std::pair<String, String>> parameters;
384 for (auto& keyValue : formatParameters)
385 parameters.append(std::make_pair(String::fromUTF8(keyValue.first.data(), keyValue.first.length()), String::fromUTF8(keyValue.second.data(), keyValue.second.length())));
386
387 ensureGPUProcessConnectionAndDispatchToThread([this, encoder = WTFMove(encoder), type, parameters = WTFMove(parameters)]() mutable {
388 Locker locker { m_connectionLock };
389 encoder->connection = m_connection;
390 encoder->connection->send(Messages::LibWebRTCCodecsProxy::CreateEncoder { encoder->identifier, formatNameFromCodecType(type), parameters, RuntimeEnabledFeatures::sharedFeatures().webRTCH264LowLatencyEncoderEnabled() }, 0);
391 encoder->parameters = WTFMove(parameters);
392
393 auto encoderIdentifier = encoder->identifier;
394 ASSERT(!m_encoders.contains(encoderIdentifier));
395 m_encoders.add(encoderIdentifier, WTFMove(encoder));
396 });
397 return result;
398}

427int32_t LibWebRTCCodecs::encodeFrame(Encoder& encoder, const webrtc::VideoFrame& frame, bool shouldEncodeAsKeyFrame)
428{
429 Locker locker { m_connectionLock };
430
431 if (!encoder.connection)
432 return WEBRTC_VIDEO_CODEC_ERROR;
433
434 RetainPtr<CVPixelBufferRef> newPixelBuffer;
435 auto pixelBuffer = webrtc::pixelBufferFromFrame(frame, [&newPixelBuffer](size_t width, size_t height, webrtc::BufferType bufferType) -> CVPixelBufferRef {
436 OSType poolBufferType;
437 switch (bufferType) {
438 case webrtc::BufferType::I420:
439 poolBufferType = kCVPixelFormatType_420YpCbCr8BiPlanarFullRange;
440 break;
441 case webrtc::BufferType::I010:
442 poolBufferType = kCVPixelFormatType_420YpCbCr10BiPlanarFullRange;
443 }
444 auto pixelBufferPool = WebProcess::singleton().libWebRTCCodecs().pixelBufferPool(width, height, poolBufferType);
445 if (!pixelBufferPool)
446 return nullptr;
447
448 newPixelBuffer = WebCore::createPixelBufferFromPool(pixelBufferPool);
449 return newPixelBuffer.get();
450 });
451
452 if (!pixelBuffer)
453 return WEBRTC_VIDEO_CODEC_ERROR;
454
455 auto sample = RemoteVideoSample::create(pixelBuffer, MediaTime(frame.timestamp_us() * 1000, 1000000), toMediaSampleVideoRotation(frame.rotation()));
456 if (!sample) {
457 // FIXME: Optimize this code path, currently we have non BGRA for muted frames at least.
458 newPixelBuffer = convertToBGRA(pixelBuffer);
459 sample = RemoteVideoSample::create(newPixelBuffer.get(), MediaTime(frame.timestamp_us() * 1000, 1000000), toMediaSampleVideoRotation(frame.rotation()));
460 }
461
462 encoder.connection->send(Messages::LibWebRTCCodecsProxy::EncodeFrame { encoder.identifier, *sample, frame.timestamp(), shouldEncodeAsKeyFrame }, 0);
463 return WEBRTC_VIDEO_CODEC_OK;
464}

521void LibWebRTCCodecs::gpuProcessConnectionDidClose(GPUProcessConnection&)
522{
523 ASSERT(isMainRunLoop());
524 Locker locker { m_connectionLock };
525 std::exchange(m_connection, nullptr)->removeThreadMessageReceiver(Messages::LibWebRTCCodecs::messageReceiverName());
526 if (!m_needsGPUProcessConnection)
527 return;
528
529 ensureGPUProcessConnectionOnMainThreadWithLock();
530 dispatchToThread([this]() {
531 // Lock everything so that we can update encoder/decoder connection.
532 Locker locker { m_connectionLock };
533 // Recreate encoders and initialize them, recreate decoders.
534 for (auto& encoder : m_encoders.values()) {
535 m_connection->send(Messages::LibWebRTCCodecsProxy::CreateEncoder { encoder->identifier, formatNameFromWebRTCCodecType(encoder->codecType), encoder->parameters, RuntimeEnabledFeatures::sharedFeatures().webRTCH264LowLatencyEncoderEnabled() }, 0);
536 if (encoder->initializationData)
537 m_connection->send(Messages::LibWebRTCCodecsProxy::InitializeEncoder { encoder->identifier, encoder->initializationData->width, encoder->initializationData->height, encoder->initializationData->startBitRate, encoder->initializationData->maxBitRate, encoder->initializationData->minBitRate, encoder->initializationData->maxFrameRate }, 0);
538 encoder->connection = m_connection.get();
539 }
540 for (auto& decoder : m_decoders.values()) {
541 createRemoteDecoder(*decoder, *m_connection);
542 decoder->connection = m_connection.get();
543 }
544 });
545}
