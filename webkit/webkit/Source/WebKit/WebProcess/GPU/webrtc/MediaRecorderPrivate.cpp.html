59void MediaRecorderPrivate::startRecording(StartRecordingCallback&& callback)
60{
61 // FIXME: we will need to implement support for multiple audio/video tracks
62 // Currently we only choose the first track as the recorded track.
63
64 auto selectedTracks = MediaRecorderPrivate::selectTracks(m_stream);
65 if (selectedTracks.audioTrack)
66 m_ringBuffer = makeUnique<CARingBuffer>(makeUniqueRef<SharedRingBufferStorage>(std::bind(&MediaRecorderPrivate::storageChanged, this, std::placeholders::_1, std::placeholders::_2, std::placeholders::_3)));
67
68 m_connection->sendWithAsyncReply(Messages::RemoteMediaRecorderManager::CreateRecorder { m_identifier, !!selectedTracks.audioTrack, !!selectedTracks.videoTrack, m_options }, [this, weakThis = makeWeakPtr(this), audioTrack = makeRefPtr(selectedTracks.audioTrack), videoTrack = makeRefPtr(selectedTracks.videoTrack), callback = WTFMove(callback)](auto&& exception, String&& mimeType, unsigned audioBitRate, unsigned videoBitRate) mutable {
69 if (!weakThis) {
70 callback(Exception { InvalidStateError }, 0, 0);
71 return;
72 }
73 if (exception) {
74 callback(Exception { exception->code, WTFMove(exception->message) }, 0, 0);
75 return;
76 }
77 if (!m_isStopped) {
78 if (audioTrack)
79 setAudioSource(&audioTrack->source());
80 if (videoTrack)
81 setVideoSource(&videoTrack->source());
82 }
83 callback(WTFMove(mimeType), audioBitRate, videoBitRate);
84 }, 0);
85}

92void MediaRecorderPrivate::videoSampleAvailable(MediaSample& sample)
93{
94 std::unique_ptr<RemoteVideoSample> remoteSample;
95 if (shouldMuteVideo()) {
96 if (!m_blackFrame) {
97 auto blackFrameDescription = CMSampleBufferGetFormatDescription(sample.platformSample().sample.cmSampleBuffer);
98 auto dimensions = CMVideoFormatDescriptionGetDimensions(blackFrameDescription);
99 auto blackFrame = createBlackPixelBuffer(dimensions.width, dimensions.height);
100 // FIXME: We convert to get an IOSurface. We could optimize this.
101 m_blackFrame = convertToBGRA(blackFrame.get());
102 }
103 remoteSample = RemoteVideoSample::create(m_blackFrame.get(), sample.presentationTime(), sample.videoRotation());
104 } else {
105 m_blackFrame = nullptr;
106 remoteSample = RemoteVideoSample::create(sample);
107 if (!remoteSample) {
108 // FIXME: Optimize this code path.
109 auto pixelBuffer = static_cast<CVPixelBufferRef>(CMSampleBufferGetImageBuffer(sample.platformSample().sample.cmSampleBuffer));
110 auto newPixelBuffer = convertToBGRA(pixelBuffer);
111 remoteSample = RemoteVideoSample::create(newPixelBuffer.get(), sample.presentationTime(), sample.videoRotation());
112 }
113 }
114
115 if (remoteSample)
116 m_connection->send(Messages::RemoteMediaRecorder::VideoSampleAvailable { WTFMove(*remoteSample) }, m_identifier);
117}

119void MediaRecorderPrivate::audioSamplesAvailable(const MediaTime& time, const PlatformAudioData& audioData, const AudioStreamDescription& description, size_t numberOfFrames)
120{
121 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
122 // explicitly allow the following allocation(s).
123 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
124
125 if (m_description != description) {
126 ASSERT(description.platformDescription().type == PlatformDescription::CAAudioStreamBasicType);
127 m_description = *WTF::get<const AudioStreamBasicDescription*>(description.platformDescription().description);
128
129 // Allocate a ring buffer large enough to contain 2 seconds of audio.
130 m_numberOfFrames = m_description.sampleRate() * 2;
131 m_ringBuffer->allocate(m_description.streamDescription(), m_numberOfFrames);
132 m_silenceAudioBuffer = nullptr;
133 }
134
135 ASSERT(is<WebAudioBufferList>(audioData));
136
137 if (shouldMuteAudio()) {
138 if (!m_silenceAudioBuffer)
139 m_silenceAudioBuffer = makeUnique<WebAudioBufferList>(m_description, numberOfFrames);
140 else
141 m_silenceAudioBuffer->setSampleCount(numberOfFrames);
142 m_silenceAudioBuffer->zeroFlatBuffer();
143 m_ringBuffer->store(m_silenceAudioBuffer->list(), numberOfFrames, time.timeValue());
144 } else
145 m_ringBuffer->store(downcast<WebAudioBufferList>(audioData).list(), numberOfFrames, time.timeValue());
146 m_connection->send(Messages::RemoteMediaRecorder::AudioSamplesAvailable { time, numberOfFrames }, m_identifier);
147}

149void MediaRecorderPrivate::storageChanged(SharedMemory* storage, const WebCore::CAAudioStreamDescription& format, size_t frameCount)
150{
151 // Heap allocations are forbidden on the audio thread for performance reasons so we need to
152 // explicitly allow the following allocation(s).
153 DisableMallocRestrictionsForCurrentThreadScope disableMallocRestrictions;
154
155 SharedMemory::Handle handle;
156 if (storage)
157 storage->createHandle(handle, SharedMemory::Protection::ReadOnly);
158
159 // FIXME: Send the actual data size with IPCHandle.
160#if OS(DARWIN) || OS(WINDOWS)
161 uint64_t dataSize = handle.size();
162#else
163 uint64_t dataSize = 0;
164#endif
165 m_connection->send(Messages::RemoteMediaRecorder::AudioSamplesStorageChanged { SharedMemory::IPCHandle { WTFMove(handle), dataSize }, format, frameCount }, m_identifier);
166}

168void MediaRecorderPrivate::fetchData(CompletionHandler<void(RefPtr<WebCore::SharedBuffer>&&, const String& mimeType, double)>&& completionHandler)
169{
170 m_connection->sendWithAsyncReply(Messages::RemoteMediaRecorder::FetchData { }, [completionHandler = WTFMove(completionHandler), mimeType = mimeType()](auto&& data, double timeCode) mutable {
171 RefPtr<SharedBuffer> buffer;
172 if (data.size())
173 buffer = SharedBuffer::create(data.data(), data.size());
174 completionHandler(WTFMove(buffer), mimeType, timeCode);
175 }, m_identifier);
176}

194const String& MediaRecorderPrivate::mimeType() const
195{
196 static NeverDestroyed<const String> audioMP4(MAKE_STATIC_STRING_IMPL("audio/mp4"));
197 static NeverDestroyed<const String> videoMP4(MAKE_STATIC_STRING_IMPL("video/mp4"));
198 return m_hasVideo ? videoMP4 : audioMP4;
199}
