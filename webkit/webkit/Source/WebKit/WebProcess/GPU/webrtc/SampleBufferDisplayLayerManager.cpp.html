37void SampleBufferDisplayLayerManager::didReceiveLayerMessage(IPC::Connection& connection, IPC::Decoder& decoder)
38{
39 if (auto* layer = m_layers.get(makeObjectIdentifier<SampleBufferDisplayLayerIdentifierType>(decoder.destinationID())).get())
40 layer->didReceiveMessage(connection, decoder);
41}
42
43std::unique_ptr<WebCore::SampleBufferDisplayLayer> SampleBufferDisplayLayerManager::createLayer(SampleBufferDisplayLayer::Client& client)
44{
45 auto layer = SampleBufferDisplayLayer::create(*this, client);
46 if (!layer)
47 return { };
48
49 m_layers.add(layer->identifier(), makeWeakPtr(*layer));
50 return layer;
51}
52
53void SampleBufferDisplayLayerManager::addLayer(SampleBufferDisplayLayer& layer)
54{
55 ASSERT(!m_layers.contains(layer.identifier()));
56 m_layers.add(layer.identifier(), makeWeakPtr(layer));
57}
58
59void SampleBufferDisplayLayerManager::removeLayer(SampleBufferDisplayLayer& layer)
60{
61 ASSERT(m_layers.contains(layer.identifier()));
62 m_layers.remove(layer.identifier());
63}
