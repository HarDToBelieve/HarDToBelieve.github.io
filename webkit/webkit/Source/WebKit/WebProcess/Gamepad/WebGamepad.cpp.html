34using WebCore::SharedGamepadValue;

36namespace WebKit {
37
38WebGamepad::WebGamepad(const GamepadData& gamepadData)
39 : PlatformGamepad(gamepadData.index())
40{
41 LOG(Gamepad, "Connecting WebGamepad %u", gamepadData.index());
42
43 m_id = gamepadData.id();
44 m_mapping = gamepadData.mapping();
45 m_axisValues.resize(gamepadData.axisValues().size());
46 m_buttonValues.resize(gamepadData.buttonValues().size());
47
48 updateValues(gamepadData);
49}
50
51const Vector<SharedGamepadValue>& WebGamepad::axisValues() const
52{
53 return m_axisValues;
54}
55
56const Vector<SharedGamepadValue>& WebGamepad::buttonValues() const
57{
58 return m_buttonValues;
59}
60
61void WebGamepad::updateValues(const GamepadData& gamepadData)
62{
63 ASSERT(!gamepadData.isNull());
64 ASSERT(gamepadData.index() == index());
65 ASSERT(m_axisValues.size() == gamepadData.axisValues().size());
66 ASSERT(m_buttonValues.size() == gamepadData.buttonValues().size());
67
68
69 m_axisValues = WTF::map(gamepadData.axisValues(), [](auto value) { return SharedGamepadValue(value); });
70 m_buttonValues = WTF::map(gamepadData.buttonValues(), [](auto value) { return SharedGamepadValue(value); });
71 m_lastUpdateTime = gamepadData.lastUpdateTime();
72}
73
74}
