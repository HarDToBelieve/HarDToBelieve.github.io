47GeolocationPermissionRequestManager::GeolocationPermissionRequestManager(WebPage& page)
48 : m_page(page)
49{
50}
51
52void GeolocationPermissionRequestManager::startRequestForGeolocation(Geolocation& geolocation)
53{
54 Frame* frame = geolocation.frame();
55
56 ASSERT_WITH_MESSAGE(frame, "It is not well understood in which cases the Geolocation is alive after its frame goes away. If you hit this assertion, please add a test covering this case.");
57 if (!frame) {
58 geolocation.setIsAllowed(false, { });
59 return;
60 }
61
62 GeolocationIdentifier geolocationID = GeolocationIdentifier::generate();
63
64 m_geolocationToIDMap.set(&geolocation, geolocationID);
65 m_idToGeolocationMap.set(geolocationID, &geolocation);
66
67 WebFrame* webFrame = WebFrame::fromCoreFrame(*frame);
68 ASSERT(webFrame);
69
70 m_page.send(Messages::WebPageProxy::RequestGeolocationPermissionForFrame(geolocationID, webFrame->info()));
71}
72
73void GeolocationPermissionRequestManager::revokeAuthorizationToken(const String& authorizationToken)
74{
75 m_page.send(Messages::WebPageProxy::RevokeGeolocationAuthorizationToken(authorizationToken));
76}
77
78void GeolocationPermissionRequestManager::cancelRequestForGeolocation(Geolocation& geolocation)
79{
80 GeolocationIdentifier geolocationID = m_geolocationToIDMap.take(&geolocation);
81 if (!geolocationID)
82 return;
83 m_idToGeolocationMap.remove(geolocationID);
84}
85
86void GeolocationPermissionRequestManager::didReceiveGeolocationPermissionDecision(GeolocationIdentifier geolocationID, const String& authorizationToken)
87{
88 Geolocation* geolocation = m_idToGeolocationMap.take(geolocationID);
89 if (!geolocation)
90 return;
91 m_geolocationToIDMap.remove(geolocation);
92
93 geolocation->setIsAllowed(!authorizationToken.isNull(), authorizationToken);
94}
