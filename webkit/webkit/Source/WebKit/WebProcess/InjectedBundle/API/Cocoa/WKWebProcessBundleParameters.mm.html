26#import "config.h"
27#import "WKWebProcessBundleParameters.h"
28
29#import <wtf/RetainPtr.h>
30
31@implementation WKWebProcessBundleParameters {
32 RetainPtr<NSMutableDictionary> _parameters;
33}
34
35- (instancetype)initWithDictionary:(NSDictionary *)dictionary
36{
37 if (!(self = [super init]))
38 return nil;
39
40 _parameters = adoptNS([[NSMutableDictionary alloc] initWithDictionary:dictionary]);
41
42 return self;
43}
44
45- (NSString *)description
46{
47 return [NSString stringWithFormat:@"<%@: %p; parameters = %@>", NSStringFromClass(self.class), self, _parameters.get()];
48}
49
50- (NSString *)valueForKey:(NSString *)key
51{
52 return [_parameters valueForKey:key];
53}
54
55- (void)setParameter:(id)parameter forKey:(NSString *)key
56{
57 [self willChangeValueForKey:key];
58 [_parameters setValue:parameter forKey:key];
59 [self didChangeValueForKey:key];
60}
61
62- (void)setParametersForKeyWithDictionary:(NSDictionary *)dictionary
63{
64 [dictionary enumerateKeysAndObjectsUsingBlock:^(NSString *key, id parameter, BOOL*) {
65 [self setParameter:parameter forKey:key];
66 }];
67}
68
69@end
