61- (void)dealloc
62{
63 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKWebProcessPlugInFrame.class, self))
64 return;
65 _frame->~WebFrame();
66 [super dealloc];

69- (JSContext *)jsContextForWorld:(WKWebProcessPlugInScriptWorld *)world

74- (WKWebProcessPlugInHitTestResult *)hitTest:(CGPoint)point
75{

79- (WKWebProcessPlugInHitTestResult *)hitTest:(CGPoint)point options:(WKHitTestOptions)options
80{
81 auto types = WebKit::WebFrame::defaultHitTestRequestTypes();
82 if (options & WKHitTestOptionAllowUserAgentShadowRootContent)
83 types.remove(WebCore::HitTestRequest::Type::DisallowUserAgentShadowContent);
84 return wrapper(_frame->hitTest(WebCore::IntPoint(point), types));
85}

88{

137static RetainPtr<NSArray> collectIcons(WebCore::Frame* frame, OptionSet<WebCore::LinkIconType> iconTypes)
138{
139 if (!frame)
140 return @[];
141 auto document = frame->document();
142 if (!document)
143 return @[];
144 return createNSArray(WebCore::LinkIconCollector(*document).iconsOfTypes(iconTypes), [] (auto& icon) -> NSURL * {
145 return icon.url;
146 });
147}

152}
