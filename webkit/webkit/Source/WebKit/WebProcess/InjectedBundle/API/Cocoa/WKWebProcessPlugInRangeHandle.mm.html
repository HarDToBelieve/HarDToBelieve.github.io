42
43- (void)dealloc
44{
45 if (WebCoreObjCScheduleDeallocateOnMainRunLoop(WKWebProcessPlugInRangeHandle.class, self))
46 return;
47 _rangeHandle->~InjectedBundleRangeHandle();
48 [super dealloc];
49}
50
51+ (WKWebProcessPlugInRangeHandle *)rangeHandleWithJSValue:(JSValue *)value inContext:(JSContext *)context
52{
53 JSContextRef contextRef = [context JSGlobalContextRef];
54 JSObjectRef objectRef = JSValueToObject(contextRef, [value JSValueRef], nullptr);
55 return wrapper(WebKit::InjectedBundleRangeHandle::getOrCreate(contextRef, objectRef));
56}
57
58- (WKWebProcessPlugInFrame *)frame
59{
60 return wrapper(_rangeHandle->document()->documentFrame());
61}
62
63- (NSString *)text
64{
65 return _rangeHandle->text();
66}
