70inline WKInsertActionType toAPI(WebCore::EditorInsertAction action)
71{
72 switch (action) {
73 case WebCore::EditorInsertAction::Typed:
74 return kWKInsertActionTyped;
75 case WebCore::EditorInsertAction::Pasted:
76 return kWKInsertActionPasted;
77 case WebCore::EditorInsertAction::Dropped:
78 return kWKInsertActionDropped;
79 }
80 ASSERT_NOT_REACHED();
81 return kWKInsertActionTyped;
82}
83
84inline WKAffinityType toAPI(WebCore::Affinity affinity)
85{
86 switch (affinity) {
87 case WebCore::Affinity::Upstream:
88 return kWKAffinityUpstream;
89 case WebCore::Affinity::Downstream:
90 return kWKAffinityDownstream;
91 }
92 ASSERT_NOT_REACHED();
93 return kWKAffinityUpstream;
94}
