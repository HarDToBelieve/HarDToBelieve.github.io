34WKTypeID WKBundleDOMWindowExtensionGetTypeID()
35{
36 return WebKit::toAPI(WebKit::InjectedBundleDOMWindowExtension::APIType);
37}
38
39WKBundleDOMWindowExtensionRef WKBundleDOMWindowExtensionCreate(WKBundleFrameRef frame, WKBundleScriptWorldRef world)
40{
41 RefPtr<WebKit::InjectedBundleDOMWindowExtension> extension = WebKit::InjectedBundleDOMWindowExtension::create(WebKit::toImpl(frame), WebKit::toImpl(world));
42 return toAPI(extension.leakRef());
43}
44
45WKBundleFrameRef WKBundleDOMWindowExtensionGetFrame(WKBundleDOMWindowExtensionRef extension)
46{
47 return toAPI(WebKit::toImpl(extension)->frame());
48}
49
50WKBundleScriptWorldRef WKBundleDOMWindowExtensionGetScriptWorld(WKBundleDOMWindowExtensionRef extension)
51{
52 return toAPI(WebKit::toImpl(extension)->world());
53}
