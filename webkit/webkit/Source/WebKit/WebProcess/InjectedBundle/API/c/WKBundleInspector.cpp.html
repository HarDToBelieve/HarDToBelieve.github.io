33WKTypeID WKBundleInspectorGetTypeID()
34{
35 return WebKit::toAPI(WebKit::WebInspector::APIType);
36}
37
38void WKBundleInspectorShow(WKBundleInspectorRef inspectorRef)
39{
40 WebKit::toImpl(inspectorRef)->show();
41}
42
43void WKBundleInspectorClose(WKBundleInspectorRef inspectorRef)
44{
45 WebKit::toImpl(inspectorRef)->close();
46}
47
48void WKBundleInspectorEvaluateScriptForTest(WKBundleInspectorRef inspectorRef, WKStringRef script)
49{
50 WebKit::toImpl(inspectorRef)->evaluateScriptForTest(WebKit::toWTFString(script));
51}
52
53void WKBundleInspectorSetPageProfilingEnabled(WKBundleInspectorRef inspectorRef, bool enabled)
54{
55 if (enabled)
56 WebKit::toImpl(inspectorRef)->startPageProfiling();
57 else
58 WebKit::toImpl(inspectorRef)->stopPageProfiling();
59}
