33WKTypeID WKBundleScriptWorldGetTypeID()
34{
35 return WebKit::toAPI(WebKit::InjectedBundleScriptWorld::APIType);
36}
37
38WKBundleScriptWorldRef WKBundleScriptWorldCreateWorld()
39{
40 RefPtr<WebKit::InjectedBundleScriptWorld> world = WebKit::InjectedBundleScriptWorld::create();
41 return toAPI(world.leakRef());
42}
43
44WKBundleScriptWorldRef WKBundleScriptWorldNormalWorld()
45{
46 return toAPI(&WebKit::InjectedBundleScriptWorld::normalWorld());
47}

49void WKBundleScriptWorldClearWrappers(WKBundleScriptWorldRef scriptWorldRef)

52}

54void WKBundleScriptWorldMakeAllShadowRootsOpen(WKBundleScriptWorldRef scriptWorldRef)

59void WKBundleScriptWorldDisableOverrideBuiltinsBehavior(WKBundleScriptWorldRef scriptWorldRef)

61 WebKit::toImpl(scriptWorldRef)->disableOverrideBuiltinsBehavior();
62}

64WKStringRef WKBundleScriptWorldCopyName(WKBundleScriptWorldRef scriptWorldRef)
