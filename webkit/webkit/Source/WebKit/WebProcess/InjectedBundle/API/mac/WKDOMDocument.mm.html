38@interface WKDOMDocumentParserYieldToken : NSObject
39
40@end
41
42@implementation WKDOMDocumentParserYieldToken {
43 std::unique_ptr<WebCore::DocumentParserYieldToken> _token;
44}
45
46- (instancetype)initWithDocument:(NakedRef<WebCore::Document>)document
47{
48 if (self = [super init])
49 _token = document->createParserYieldToken();
50 return self;
51}
52
53@end
54
55@implementation WKDOMDocument
56
57- (WKDOMElement *)createElement:(NSString *)tagName
58{
59 // FIXME: Do something about the exception.
60 auto result = downcast<WebCore::Document>(*_impl).createElementForBindings(tagName);
61 if (result.hasException())
62 return nil;
63 return WebKit::toWKDOMElement(result.releaseReturnValue().ptr());
64}
65
66- (WKDOMText *)createTextNode:(NSString *)data
67{
68 return WebKit::toWKDOMText(downcast<WebCore::Document>(*_impl).createTextNode(data).ptr());
69}
70
71- (WKDOMElement *)body
72{
73 return WebKit::toWKDOMElement(downcast<WebCore::Document>(*_impl).bodyOrFrameset());
74}
75
76- (WKDOMNode *)createDocumentFragmentWithMarkupString:(NSString *)markupString baseURL:(NSURL *)baseURL
77{
78 return WebKit::toWKDOMNode(createFragmentFromMarkup(downcast<WebCore::Document>(*_impl), markupString, baseURL.absoluteString).ptr());
79}
80
81- (WKDOMNode *)createDocumentFragmentWithText:(NSString *)text
82{
83 return WebKit::toWKDOMNode(createFragmentFromText(makeRangeSelectingNodeContents(downcast<WebCore::Document>(*_impl)), text).ptr());
84}
85
86- (id)parserYieldToken
87{
88 return adoptNS([[WKDOMDocumentParserYieldToken alloc] initWithDocument:downcast<WebCore::Document>(*_impl)]).autorelease();
89}
90
91@end
