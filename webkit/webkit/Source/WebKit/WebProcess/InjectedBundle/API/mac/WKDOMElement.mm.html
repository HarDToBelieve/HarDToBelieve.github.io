35{
36 return downcast<WebCore::Element>(*_impl).hasAttribute(attribute);
37}
38
39- (NSString *)getAttribute:(NSString *)attribute
40{
41 return downcast<WebCore::Element>(*_impl).getAttribute(attribute);
42}
43
44- (void)setAttribute:(NSString *)name value:(NSString *)value
45{
46 downcast<WebCore::Element>(*_impl).setAttribute(name, value);
47}
48
49- (NSString *)tagName
50{
51 return downcast<WebCore::Element>(*_impl).tagName();
52}

54@end
