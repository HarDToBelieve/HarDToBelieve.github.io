57static void didCreatePage(WKBundleRef bundle, WKBundlePageRef page, const void* clientInfo)
58{
59 auto plugInController = (__bridge WKWebProcessPlugInController *)clientInfo;
60 id <WKWebProcessPlugIn> principalClassInstance = plugInController->_principalClassInstance.get();
61
62 if ([principalClassInstance respondsToSelector:@selector(webProcessPlugIn:didCreateBrowserContextController:)])
63 [principalClassInstance webProcessPlugIn:plugInController didCreateBrowserContextController:wrapper(*WebKit::toImpl(page))];
64}
65
66static void willDestroyPage(WKBundleRef bundle, WKBundlePageRef page, const void* clientInfo)
67{
68 auto plugInController = (__bridge WKWebProcessPlugInController *)clientInfo;
69 id <WKWebProcessPlugIn> principalClassInstance = plugInController->_principalClassInstance.get();
70
71 if ([principalClassInstance respondsToSelector:@selector(webProcessPlugIn:willDestroyBrowserContextController:)])
72 [principalClassInstance webProcessPlugIn:plugInController willDestroyBrowserContextController:wrapper(*WebKit::toImpl(page))];
73}

75static void setUpBundleClient(WKWebProcessPlugInController *plugInController, WebKit::InjectedBundle& bundle)
76{
77 WKBundleClientV1 bundleClient;
78 memset(&bundleClient, 0, sizeof(bundleClient));
79
80 bundleClient.base.version = 1;
81 bundleClient.base.clientInfo = (__bridge void*)plugInController;
82 bundleClient.didCreatePage = didCreatePage;
83 bundleClient.willDestroyPage = willDestroyPage;
84
85 WKBundleSetClient(toAPI(&bundle), &bundleClient.base);
86}
87
88- (void)_setPrincipalClassInstance:(id <WKWebProcessPlugIn>)principalClassInstance
89{
90 ASSERT(!_principalClassInstance);
91 _principalClassInstance = principalClassInstance;
92
93 setUpBundleClient(self, *_bundle);
94}

108static Ref<API::Array> createWKArray(NSArray *array)
109{
110 NSUInteger count = [array count];
111 Vector<RefPtr<API::Object>> strings;
112 strings.reserveInitialCapacity(count);
113
114 for (id entry in array) {
115 if ([entry isKindOfClass:[NSString class]])
116 strings.uncheckedAppend(adoptRef(WebKit::toImpl(WKStringCreateWithCFString((__bridge CFStringRef)entry))));
117 }
118
119 return API::Array::create(WTFMove(strings));
120}
121
122- (void)extendClassesForParameterCoder:(NSArray *)classes
123{
124 auto classList = createWKArray(classes);
125 _bundle->extendClassesForParameterCoder(classList.get());
126}
