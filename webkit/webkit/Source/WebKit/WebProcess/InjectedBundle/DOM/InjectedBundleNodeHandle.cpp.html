87Ref<InjectedBundleNodeHandle> InjectedBundleNodeHandle::getOrCreate(Node& node)
88{
89 if (auto* existingHandle = domNodeHandleCache().get(&node))
90 return Ref<InjectedBundleNodeHandle>(*existingHandle);
91
92 auto nodeHandle = InjectedBundleNodeHandle::create(node);
93 if (nodeHandle->coreNode())
94 domNodeHandleCache().add(nodeHandle->coreNode(), nodeHandle.ptr());
95 return nodeHandle;
96}

98Ref<InjectedBundleNodeHandle> InjectedBundleNodeHandle::create(Node& node)
99{
100 auto handle = adoptRef(*new InjectedBundleNodeHandle(node));
101 handle->suspendIfNeeded();
102 return handle;
103}

149static RefPtr<WebImage> imageForRect(FrameView* frameView, const IntRect& paintingRect, const std::optional<float>& bitmapWidth, SnapshotOptions options)
150{
151 if (paintingRect.isEmpty())
152 return nullptr;
153
154 float bitmapScaleFactor;
155 IntSize bitmapSize;
156 if (bitmapWidth) {
157 bitmapScaleFactor = bitmapWidth.value() / paintingRect.width();
158 bitmapSize = roundedIntSize(FloatSize(bitmapWidth.value(), paintingRect.height() * bitmapScaleFactor));
159 } else {
160 bitmapScaleFactor = 1;
161 bitmapSize = paintingRect.size();
162 }
163
164 float deviceScaleFactor = frameView->frame().page()->deviceScaleFactor();
165 bitmapSize.scale(deviceScaleFactor);
166
167 if (bitmapSize.isEmpty())
168 return nullptr;
169
170 auto snapshot = WebImage::create(bitmapSize, snapshotOptionsToImageOptions(options));
171 if (!snapshot)
172 return nullptr;
173
174 auto graphicsContext = snapshot->bitmap().createGraphicsContext();
175 if (!graphicsContext)
176 return nullptr;
177
178 graphicsContext->clearRect(IntRect(IntPoint(), bitmapSize));
179 graphicsContext->applyDeviceScaleFactor(deviceScaleFactor);
180 graphicsContext->scale(bitmapScaleFactor);
181 graphicsContext->translate(-paintingRect.location());
182
183 FrameView::SelectionInSnapshot shouldPaintSelection = FrameView::IncludeSelection;
184 if (options & SnapshotOptionsExcludeSelectionHighlighting)
185 shouldPaintSelection = FrameView::ExcludeSelection;
186
187 auto paintBehavior = frameView->paintBehavior() | PaintBehavior::FlattenCompositingLayers | PaintBehavior::Snapshotting;
188 if (options & SnapshotOptionsForceBlackText)
189 paintBehavior.add(PaintBehavior::ForceBlackText);
190 if (options & SnapshotOptionsForceWhiteText)
191 paintBehavior.add(PaintBehavior::ForceWhiteText);
192
193 auto oldPaintBehavior = frameView->paintBehavior();
194 frameView->setPaintBehavior(paintBehavior);
195 frameView->paintContentsForSnapshot(*graphicsContext.get(), paintingRect, shouldPaintSelection, FrameView::DocumentCoordinates);
196 frameView->setPaintBehavior(oldPaintBehavior);
197
198 return snapshot;
199}

201RefPtr<WebImage> InjectedBundleNodeHandle::renderedImage(SnapshotOptions options, bool shouldExcludeOverflow, const std::optional<float>& bitmapWidth)
202{
203 if (!m_node)
204 return nullptr;
205
206 Frame* frame = m_node->document().frame();
207 if (!frame)
208 return nullptr;
209
210 FrameView* frameView = frame->view();
211 if (!frameView)
212 return nullptr;
213
214 m_node->document().updateLayout();
215
216 RenderObject* renderer = m_node->renderer();
217 if (!renderer)
218 return nullptr;
219
220 IntRect paintingRect;
221 if (shouldExcludeOverflow)
222 paintingRect = renderer->absoluteBoundingBoxRectIgnoringTransforms();
223 else {
224 LayoutRect topLevelRect;
225 paintingRect = snappedIntRect(renderer->paintingRootRect(topLevelRect));
226 }
227
228 frameView->setNodeToDraw(m_node.get());
229 auto image = imageForRect(frameView, paintingRect, bitmapWidth, options);
230 frameView->setNodeToDraw(0);
231
232 return image;
233}

432void InjectedBundleNodeHandle::stop()
433{
434 // Invalidate handles to nodes inside documents that are about to be destroyed in order to prevent leaks.
435 if (m_node) {
436 domNodeHandleCache().remove(m_node.get());
437 m_node = nullptr;
438 }
