138void InjectedBundlePageEditorClient::getPasteboardDataForRange(WebPage& page, const std::optional<SimpleRange>& range, Vector<String>& pasteboardTypes, Vector<RefPtr<SharedBuffer>>& pasteboardData)
139{
140 if (m_client.getPasteboardDataForRange) {
141 WKArrayRef types = nullptr;
142 WKArrayRef data = nullptr;
143 m_client.getPasteboardDataForRange(toAPI(&page), toAPI(createHandle(range).get()), &types, &data, m_client.base.clientInfo);
144 auto typesArray = adoptRef(toImpl(types));
145 auto dataArray = adoptRef(toImpl(data));
146
147 pasteboardTypes.clear();
148 pasteboardData.clear();
149
150 if (!typesArray || !dataArray)
151 return;
152
153 ASSERT(typesArray->size() == dataArray->size());
154
155 for (auto type : typesArray->elementsOfType<API::String>())
156 pasteboardTypes.append(type->string());
157
158 for (auto item : dataArray->elementsOfType<API::Data>()) {
159 auto buffer = SharedBuffer::create(item->bytes(), item->size());
160 pasteboardData.append(WTFMove(buffer));
161 }
162 }
163}

165bool InjectedBundlePageEditorClient::performTwoStepDrop(WebPage& page, DocumentFragment& fragment, const SimpleRange& destination, bool isMove)
166{
167 if (!m_client.performTwoStepDrop)
168 return false;
169
170 auto nodeHandle = InjectedBundleNodeHandle::getOrCreate(&fragment);
171 return m_client.performTwoStepDrop(toAPI(&page), toAPI(nodeHandle.get()), toAPI(createHandle(destination).get()), isMove, m_client.base.clientInfo);
172}
