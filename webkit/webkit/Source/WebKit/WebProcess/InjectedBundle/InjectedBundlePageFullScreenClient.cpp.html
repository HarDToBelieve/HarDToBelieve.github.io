41namespace WebKit {

44bool InjectedBundlePageFullScreenClient::supportsFullScreen(WebPage *page, bool withKeyboard)
45{
46 if (m_client.supportsFullScreen)
47 return m_client.supportsFullScreen(toAPI(page), withKeyboard);
48
49 bool supports = true;
50 page->sendSync(Messages::WebFullScreenManagerProxy::SupportsFullScreen(withKeyboard), Messages::WebFullScreenManagerProxy::SupportsFullScreen::Reply(supports));
51 return supports;
52}
53
54void InjectedBundlePageFullScreenClient::enterFullScreenForElement(WebPage *page, WebCore::Element *element, bool blocksReturnToFullscreenFromPictureInPicture)
55{
56 if (m_client.enterFullScreenForElement) {
57 RefPtr<InjectedBundleNodeHandle> nodeHandle = InjectedBundleNodeHandle::getOrCreate(element);
58 m_client.enterFullScreenForElement(toAPI(page), toAPI(nodeHandle.get()));
59 } else
60 page->send(Messages::WebFullScreenManagerProxy::EnterFullScreen(blocksReturnToFullscreenFromPictureInPicture));
61}
62
63void InjectedBundlePageFullScreenClient::exitFullScreenForElement(WebPage *page, WebCore::Element *element)
64{
65 if (m_client.exitFullScreenForElement) {
66 RefPtr<InjectedBundleNodeHandle> nodeHandle = InjectedBundleNodeHandle::getOrCreate(element);
67 m_client.exitFullScreenForElement(toAPI(page), toAPI(nodeHandle.get()));
68 } else
69 page->send(Messages::WebFullScreenManagerProxy::ExitFullScreen());
70}
71
72void InjectedBundlePageFullScreenClient::beganEnterFullScreen(WebPage *page, IntRect& initialFrame, IntRect& finalFrame)
73{
74 if (m_client.beganEnterFullScreen)
75 m_client.beganEnterFullScreen(toAPI(page), toAPI(initialFrame), toAPI(finalFrame));
76 else
77 page->send(Messages::WebFullScreenManagerProxy::BeganEnterFullScreen(initialFrame, finalFrame));
78}
79
80void InjectedBundlePageFullScreenClient::beganExitFullScreen(WebPage *page, IntRect& initialFrame, IntRect& finalFrame)
81{
82 if (m_client.beganExitFullScreen)
83 m_client.beganExitFullScreen(toAPI(page), toAPI(initialFrame), toAPI(finalFrame));
84 else
85 page->send(Messages::WebFullScreenManagerProxy::BeganExitFullScreen(initialFrame, finalFrame));
86}
87
88void InjectedBundlePageFullScreenClient::closeFullScreen(WebPage *page)
89{
90 if (m_client.closeFullScreen)
91 m_client.closeFullScreen(toAPI(page));
92 else
93 page->send(Messages::WebFullScreenManagerProxy::Close());
94}
