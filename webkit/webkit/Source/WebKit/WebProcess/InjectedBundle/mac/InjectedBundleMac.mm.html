56static NSEventModifierFlags currentModifierFlags(id self, SEL _cmd)
57{
58 auto currentModifiers = PlatformKeyboardEvent::currentStateOfModifierKeys();
59 NSEventModifierFlags modifiers = 0;
60
61 if (currentModifiers.contains(PlatformEvent::Modifier::ShiftKey))
62 modifiers |= NSEventModifierFlagShift;
63 if (currentModifiers.contains(PlatformEvent::Modifier::ControlKey))
64 modifiers |= NSEventModifierFlagControl;
65 if (currentModifiers.contains(PlatformEvent::Modifier::AltKey))
66 modifiers |= NSEventModifierFlagOption;
67 if (currentModifiers.contains(PlatformEvent::Modifier::MetaKey))
68 modifiers |= NSEventModifierFlagCommand;
69 if (currentModifiers.contains(PlatformEvent::Modifier::CapsLockKey))
70 modifiers |= NSEventModifierFlagCapsLock;
71
72 return modifiers;
73}
74#endif

76static RetainPtr<NSKeyedUnarchiver> createUnarchiver(const unsigned char* bytes, NSUInteger length)
77{
78 auto data = adoptNS([[NSData alloc] initWithBytesNoCopy:const_cast<unsigned char*>(bytes) length:length freeWhenDone:NO]);
79 auto unarchiver = adoptNS([[NSKeyedUnarchiver alloc] initForReadingFromData:data.get() error:nullptr]);
80 unarchiver.get().decodingFailurePolicy = NSDecodingFailurePolicyRaiseException;
81 return unarchiver;
82}
83
84static RetainPtr<NSKeyedUnarchiver> createUnarchiver(const API::Data& data)
85{
86 return createUnarchiver(data.bytes(), data.size());
87}
88
89static RetainPtr<NSKeyedUnarchiver> createUnarchiver(const IPC::DataReference& data)
90{
91 return createUnarchiver(data.data(), data.size());
92}

94bool InjectedBundle::decodeBundleParameters(API::Data* bundleParameterDataPtr)
95{
96 if (!bundleParameterDataPtr)
97 return true;
98
99 auto unarchiver = createUnarchiver(*bundleParameterDataPtr);
100
101 NSDictionary *dictionary = nil;
102 @try {
103 dictionary = [unarchiver.get() decodeObjectOfClasses:classesForCoder() forKey:@"parameters"];
104 if (![dictionary isKindOfClass:[NSDictionary class]]) {
105 WTFLogAlways("InjectedBundle::decodeBundleParameters failed - Resulting object was not an NSDictionary.\n");
106 return false;
107 }
108 } @catch (NSException *exception) {
109 LOG_ERROR("InjectedBundle::decodeBundleParameters failed to decode bundle parameters: %@." , exception);
110 return false;
111 }
112
113 ASSERT(!m_bundleParameters || m_bundleParameters.get());
114 m_bundleParameters = adoptNS([[WKWebProcessBundleParameters alloc] initWithDictionary:dictionary]);
115 return true;
116}

118bool InjectedBundle::initialize(const WebProcessCreationParameters& parameters, API::Object* initializationUserData)
119{
120 if (auto sandboxExtension = std::exchange(m_sandboxExtension, nullptr)) {
121 if (!sandboxExtension->consumePermanently()) {
122 WTFLogAlways("InjectedBundle::load failed - Could not consume bundle sandbox extension for [%s].\n", m_path.utf8().data());
123 return false;
124 }
125 }
126
127 m_platformBundle = [[NSBundle alloc] initWithPath:m_path];
128 if (!m_platformBundle) {
129 WTFLogAlways("InjectedBundle::load failed - Could not create the bundle.\n");
130 return false;
131 }
132
133 WKBundleAdditionalClassesForParameterCoderFunctionPtr additionalClassesForParameterCoderFunction = nullptr;
134 WKBundleInitializeFunctionPtr initializeFunction = nullptr;
135 if (NSString *executablePath = m_platformBundle.executablePath) {
136 if (dlopen_preflight(executablePath.fileSystemRepresentation)) {
137 // We don't hold onto this handle anywhere more permanent since we never dlclose.
138 if (void* handle = dlopen(executablePath.fileSystemRepresentation, RTLD_LAZY | RTLD_GLOBAL | RTLD_FIRST)) {
139 additionalClassesForParameterCoderFunction = bitwise_cast<WKBundleAdditionalClassesForParameterCoderFunctionPtr>(dlsym(handle, "WKBundleAdditionalClassesForParameterCoder"));
140 initializeFunction = bitwise_cast<WKBundleInitializeFunctionPtr>(dlsym(handle, "WKBundleInitialize"));
141 }
142 }
143 }
144
145 if (!initializeFunction) {
146 NSError *error;
147 if (![m_platformBundle preflightAndReturnError:&error]) {
148 NSLog(@"InjectedBundle::load failed - preflightAndReturnError failed, error: %@", error);
149 return nil;
150 }
151 if (![m_platformBundle loadAndReturnError:&error]) {
152 NSLog(@"InjectedBundle::load failed - loadAndReturnError failed, error: %@", error);
153 return false;
154 }
155 initializeFunction = bitwise_cast<WKBundleInitializeFunctionPtr>(CFBundleGetFunctionPointerForName([m_platformBundle _cfBundle], CFSTR("WKBundleInitialize")));
156 }
157
158 if (!additionalClassesForParameterCoderFunction)
159 additionalClassesForParameterCoderFunction = bitwise_cast<WKBundleAdditionalClassesForParameterCoderFunctionPtr>(CFBundleGetFunctionPointerForName([m_platformBundle _cfBundle], CFSTR("WKBundleAdditionalClassesForParameterCoder")));
160
161 // Update list of valid classes for the parameter coder
162 if (additionalClassesForParameterCoderFunction)
163 additionalClassesForParameterCoderFunction(toAPI(this), toAPI(initializationUserData));
164
165#if PLATFORM(MAC)
166 // Swizzle [NSEvent modiferFlags], since it always returns 0 when the WindowServer is blocked.
167 Method method = class_getClassMethod([NSEvent class], @selector(modifierFlags));
168 method_setImplementation(method, reinterpret_cast<IMP>(currentModifierFlags));
169#endif
170
171 // First check to see if the bundle has a WKBundleInitialize function.
172 if (initializeFunction) {
173 if (!decodeBundleParameters(parameters.bundleParameterData.get()))
174 return false;
175 initializeFunction(toAPI(this), toAPI(initializationUserData));
176 return true;
177 }
178
179 // Otherwise, look to see if the bundle has a principal class
180 Class principalClass = [m_platformBundle principalClass];
181 if (!principalClass) {
182 WTFLogAlways("InjectedBundle::load failed - No initialize function or principal class found in the bundle executable.\n");
183 return false;
184 }
185
186 if (![principalClass conformsToProtocol:@protocol(WKWebProcessPlugIn)]) {
187 WTFLogAlways("InjectedBundle::load failed - Principal class does not conform to the WKWebProcessPlugIn protocol.\n");
188 return false;
189 }
190
191 auto instance = adoptNS((id <WKWebProcessPlugIn>)[(NSObject *)[principalClass alloc] init]);
192 if (!instance) {
193 WTFLogAlways("InjectedBundle::load failed - Could not initialize an instance of the principal class.\n");
194 return false;
195 }
196
197 WKWebProcessPlugInController* plugInController = WebKit::wrapper(*this);
198 [plugInController _setPrincipalClassInstance:instance.get()];
199
200 if ([instance respondsToSelector:@selector(additionalClassesForParameterCoder)])
201 [plugInController extendClassesForParameterCoder:[instance additionalClassesForParameterCoder]];
202
203 if (!decodeBundleParameters(parameters.bundleParameterData.get()))
204 return false;
205
206 if ([instance respondsToSelector:@selector(webProcessPlugIn:initializeWithObject:)]) {
207 RetainPtr<id> objCInitializationUserData;
208 if (initializationUserData && initializationUserData->type() == API::Object::Type::ObjCObjectGraph)
209 objCInitializationUserData = static_cast<ObjCObjectGraph*>(initializationUserData)->rootObject();
210 [instance webProcessPlugIn:plugInController initializeWithObject:objCInitializationUserData.get()];
211 }
212
213 return true;
214}

216WKWebProcessBundleParameters *InjectedBundle::bundleParameters()
217{
218 // We must not return nil even if no parameters are currently set, in order to allow the client
219 // to use KVO.
220 if (!m_bundleParameters)
221 m_bundleParameters = adoptNS([[WKWebProcessBundleParameters alloc] initWithDictionary:@{ }]);
222
223 return m_bundleParameters.get();
224}

226void InjectedBundle::extendClassesForParameterCoder(API::Array& classes)
227{
228 size_t size = classes.size();
229
230 auto mutableSet = adoptNS([classesForCoder() mutableCopy]);
231
232 for (size_t i = 0; i < size; ++i) {
233 API::String* classNameString = classes.at<API::String>(i);
234 if (!classNameString) {
235 WTFLogAlways("InjectedBundle::extendClassesForParameterCoder - No class provided as argument %d.\n", i);
236 break;
237 }
238
239 CString className = classNameString->string().utf8();
240 Class objectClass = objc_lookUpClass(className.data());
241 if (!objectClass) {
242 WTFLogAlways("InjectedBundle::extendClassesForParameterCoder - Class %s is not a valid Objective C class.\n", className.data());
243 break;
244 }
245
246 [mutableSet.get() addObject:objectClass];
247 }
248
249 m_classesForCoder = mutableSet;
250}

252NSSet* InjectedBundle::classesForCoder()
253{
254 if (!m_classesForCoder)
255 m_classesForCoder = [NSSet setWithObjects:[NSArray class], [NSData class], [NSDate class], [NSDictionary class], [NSNull class], [NSNumber class], [NSSet class], [NSString class], [NSTimeZone class], [NSURL class], [NSUUID class], [WKBrowsingContextHandle class], nil];
256
257 return m_classesForCoder.get();
258}

260void InjectedBundle::setBundleParameter(const String& key, const IPC::DataReference& value)
261{
262 id parameter = nil;
263 auto unarchiver = createUnarchiver(value);
264 @try {
265 parameter = [unarchiver decodeObjectOfClasses:classesForCoder() forKey:@"parameter"];
266 } @catch (NSException *exception) {
267 LOG_ERROR("Failed to decode bundle parameter: %@", exception);
268 return;
269 }
270
271 if (!m_bundleParameters && parameter)
272 m_bundleParameters = adoptNS([[WKWebProcessBundleParameters alloc] initWithDictionary:@{ }]);
273
274 [m_bundleParameters setParameter:parameter forKey:key];
275}
276
277void InjectedBundle::setBundleParameters(const IPC::DataReference& value)
278{
279 NSDictionary *parameters = nil;
280 auto unarchiver = createUnarchiver(value);
281 @try {
282 parameters = [unarchiver decodeObjectOfClasses:classesForCoder() forKey:@"parameters"];
283 } @catch (NSException *exception) {
284 LOG_ERROR("Failed to decode bundle parameter: %@", exception);
285 }
286
287 if (!parameters)
288 return;
289
290 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION([parameters isKindOfClass:[NSDictionary class]]);
291
292 if (!m_bundleParameters) {
293 m_bundleParameters = adoptNS([[WKWebProcessBundleParameters alloc] initWithDictionary:parameters]);
294 return;
295 }
296
297 [m_bundleParameters setParametersForKeyWithDictionary:parameters];
298}
