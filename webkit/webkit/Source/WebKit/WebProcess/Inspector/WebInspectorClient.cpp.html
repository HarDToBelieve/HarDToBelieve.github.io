46class RepaintIndicatorLayerClient final : public GraphicsLayerClient {
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 RepaintIndicatorLayerClient(WebInspectorClient& inspectorClient)
50 : m_inspectorClient(inspectorClient)
51 {
52 }
53 virtual ~RepaintIndicatorLayerClient() { }
54private:
55 void notifyAnimationEnded(const GraphicsLayer* layer, const String&) override
56 {
57 m_inspectorClient.animationEndedForLayer(layer);
58 }
59
60 WebInspectorClient& m_inspectorClient;
61};

69WebInspectorClient::~WebInspectorClient()
70{
71 for (auto& layer : m_paintRectLayers)
72 layer->removeFromParent();
73
74 m_paintRectLayers.clear();
75
76 if (m_paintRectOverlay && m_page->corePage())
77 m_page->corePage()->pageOverlayController().uninstallPageOverlay(*m_paintRectOverlay, PageOverlay::FadeMode::Fade);
78}

112void WebInspectorClient::highlight()
113{
114 if (!m_page->corePage()->settings().acceleratedCompositingEnabled()) {
115#if PLATFORM(GTK) || PLATFORM(WIN) || PLATFORM(PLAYSTATION)
116 // FIXME: It can be optimized by marking only highlighted rect dirty.
117 // setNeedsDisplay always makes whole rect dirty, and could lead to poor performance.
118 // https://bugs.webkit.org/show_bug.cgi?id=195933
119 m_page->drawingArea()->setNeedsDisplay();
120#endif
121 return;
122 }
123
124#if !PLATFORM(IOS_FAMILY)
125 if (!m_highlightOverlay) {
126 auto highlightOverlay = PageOverlay::create(*this);
127 m_highlightOverlay = highlightOverlay.ptr();
128 m_page->corePage()->pageOverlayController().installPageOverlay(WTFMove(highlightOverlay), PageOverlay::FadeMode::Fade);
129 m_highlightOverlay->setNeedsDisplay();
130 } else {
131 m_highlightOverlay->stopFadeOutAnimation();
132 m_highlightOverlay->setNeedsDisplay();
133 }
134#else
135 InspectorOverlay::Highlight highlight;
136 m_page->corePage()->inspectorController().getHighlight(highlight, InspectorOverlay::CoordinateSystem::Document);
137 m_page->showInspectorHighlight(highlight);
138#endif
139}

141void WebInspectorClient::hideHighlight()
142{
143#if PLATFORM(GTK) || PLATFORM(WIN) || PLATFORM(PLAYSTATION)
144 if (!m_page->corePage()->settings().acceleratedCompositingEnabled()) {
145 // FIXME: It can be optimized by marking only highlighted rect dirty.
146 // setNeedsDisplay always makes whole rect dirty, and could lead to poor performance.
147 // https://bugs.webkit.org/show_bug.cgi?id=195933
148 m_page->drawingArea()->setNeedsDisplay();
149 return;
150 }
151#endif
152
153#if !PLATFORM(IOS_FAMILY)
154 if (m_highlightOverlay)
155 m_page->corePage()->pageOverlayController().uninstallPageOverlay(*m_highlightOverlay, PageOverlay::FadeMode::Fade);
156#else
157 m_page->hideInspectorHighlight();
158#endif
159}

161void WebInspectorClient::showPaintRect(const FloatRect& rect)
162{
163 if (!m_page->corePage()->settings().acceleratedCompositingEnabled())
164 return;
165
166 if (!m_paintRectOverlay) {
167 m_paintRectOverlay = PageOverlay::create(*this, PageOverlay::OverlayType::Document);
168 m_page->corePage()->pageOverlayController().installPageOverlay(*m_paintRectOverlay, PageOverlay::FadeMode::DoNotFade);
169 }
170
171 if (!m_paintIndicatorLayerClient)
172 m_paintIndicatorLayerClient = makeUnique<RepaintIndicatorLayerClient>(*this);
173
174 auto paintLayer = GraphicsLayer::create(m_page->drawingArea()->graphicsLayerFactory(), *m_paintIndicatorLayerClient);
175
176 paintLayer->setName(MAKE_STATIC_STRING_IMPL("paint rect"));
177 paintLayer->setAnchorPoint(FloatPoint3D());
178 paintLayer->setPosition(rect.location());
179 paintLayer->setSize(rect.size());
180 paintLayer->setBackgroundColor(Color::red.colorWithAlphaByte(51));
181
182 KeyframeValueList fadeKeyframes(AnimatedPropertyOpacity);
183 fadeKeyframes.insert(makeUnique<FloatAnimationValue>(0, 1));
184
185 fadeKeyframes.insert(makeUnique<FloatAnimationValue>(0.25, 0));
186
187 auto opacityAnimation = Animation::create();
188 opacityAnimation->setDuration(0.25);
189
190 paintLayer->addAnimation(fadeKeyframes, FloatSize(), opacityAnimation.ptr(), "opacity"_s, 0);
191
192 GraphicsLayer& rawLayer = paintLayer.get();
193 m_paintRectLayers.add(WTFMove(paintLayer));
194
195 GraphicsLayer& overlayRootLayer = m_paintRectOverlay->layer();
196 overlayRootLayer.addChild(rawLayer);
197}
198
199void WebInspectorClient::animationEndedForLayer(const GraphicsLayer* layer)
200{
201 GraphicsLayer* nonConstLayer = const_cast<GraphicsLayer*>(layer);
202 nonConstLayer->removeFromParent();
203 m_paintRectLayers.remove(*nonConstLayer);
204}

244void WebInspectorClient::willMoveToPage(PageOverlay&, Page* page)
245{
246 if (page)
247 return;
248
249 // The page overlay is moving away from the web page, reset it.
250 ASSERT(m_highlightOverlay);
251 m_highlightOverlay = nullptr;
252}
