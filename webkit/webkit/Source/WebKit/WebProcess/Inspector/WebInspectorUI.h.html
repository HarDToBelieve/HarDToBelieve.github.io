77 void showMainResourceForFrame(const String& frameIdentifier);

111 void startWindowDrag() override;
112 void moveWindowBy(float x, float y) override;
113
114 bool isRemote() const final { return false; }
115 String localizedStringsURL() const override;
116 String backendCommandsURL() const final { return String(); }

118 String targetPlatformName() const override;
119 String targetBuildVersion() const override;

124 void bringToFront() override;
125 void closeWindow() override;
126 void reopen() override;
127 void resetState() override;

129 void setForcedAppearance(WebCore::InspectorFrontendClient::Appearance) override;

133 bool supportsDockSide(DockSide) override;

187 WebPageProxyIdentifier m_inspectedPageIdentifier;

195
196 DockSide m_dockSide { DockSide::Undocked };
197 unsigned m_inspectionLevel { 1 };
