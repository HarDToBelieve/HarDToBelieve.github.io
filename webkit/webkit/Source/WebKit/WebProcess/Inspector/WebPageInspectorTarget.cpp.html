36using namespace Inspector;
37
38WebPageInspectorTarget::WebPageInspectorTarget(WebPage& page)
39 : m_page(page)
40{
41}
42
43String WebPageInspectorTarget::identifier() const
44{
45 return toTargetID(m_page.identifier());
46}
47
48void WebPageInspectorTarget::connect(Inspector::FrontendChannel::ConnectionType connectionType)
49{
50 if (m_channel)
51 return;
52 m_channel = makeUnique<WebPageInspectorTargetFrontendChannel>(m_page, identifier(), connectionType);
53 if (m_page.corePage())
54 m_page.corePage()->inspectorController().connectFrontend(*m_channel);
55}
56
57void WebPageInspectorTarget::disconnect()
58{
59 if (!m_channel)
60 return;
61 if (m_page.corePage())
62 m_page.corePage()->inspectorController().disconnectFrontend(*m_channel);
63 m_channel.reset();
64}
65
66void WebPageInspectorTarget::sendMessageToTargetBackend(const String& message)
67{
68 if (m_page.corePage())
69 m_page.corePage()->inspectorController().dispatchMessageFromFrontend(message);
70}
71
72String WebPageInspectorTarget::toTargetID(WebCore::PageIdentifier pageID)
73{
74 return makeString("page-", pageID.toUInt64());
75}
