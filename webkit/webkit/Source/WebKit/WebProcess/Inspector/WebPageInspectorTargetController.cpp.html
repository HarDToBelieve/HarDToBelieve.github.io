33namespace WebKit {
34
35WebPageInspectorTargetController::WebPageInspectorTargetController(WebPage& page)
36 : m_page(page)
37 , m_pageTarget(page)
38{
39 // Do not send the page target to the UIProcess, the WebPageProxy will manager this for us.
40 m_targets.set(m_pageTarget.identifier(), &m_pageTarget);
41}
42
43WebPageInspectorTargetController::~WebPageInspectorTargetController()
44{
45}
46
47void WebPageInspectorTargetController::addTarget(Inspector::InspectorTarget& target)
48{
49 auto addResult = m_targets.set(target.identifier(), &target);
50 ASSERT_UNUSED(addResult, addResult.isNewEntry);
51
52 m_page.send(Messages::WebPageProxy::CreateInspectorTarget(target.identifier(), target.type()));
53}
54
55void WebPageInspectorTargetController::removeTarget(Inspector::InspectorTarget& target)
56{
57 ASSERT_WITH_MESSAGE(target.identifier() != m_pageTarget.identifier(), "Should never remove the main target.");
58
59 m_page.send(Messages::WebPageProxy::DestroyInspectorTarget(target.identifier()));
60
61 m_targets.remove(target.identifier());
62}
63
64void WebPageInspectorTargetController::connectInspector(const String& targetId, Inspector::FrontendChannel::ConnectionType connectionType)
65{
66 auto target = m_targets.get(targetId);
67 if (!target)
68 return;
69
70 target->connect(connectionType);
71}
72
73void WebPageInspectorTargetController::disconnectInspector(const String& targetId)
74{
75 auto target = m_targets.get(targetId);
76 if (!target)
77 return;
78
79 target->disconnect();
80}
81
82void WebPageInspectorTargetController::sendMessageToTargetBackend(const String& targetId, const String& message)
83{
84 auto target = m_targets.get(targetId);
85 if (!target)
86 return;
87
88 target->sendMessageToTargetBackend(message);
89}
