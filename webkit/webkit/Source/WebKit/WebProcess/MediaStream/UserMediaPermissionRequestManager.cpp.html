48void UserMediaPermissionRequestManager::startUserMediaRequest(UserMediaRequest& request)
49{
50 Document* document = request.document();
51 Frame* frame = document ? document->frame() : nullptr;
52
53 if (!frame || !document->page()) {
54 request.deny(UserMediaRequest::OtherFailure, emptyString());
55 return;
56 }
57
58 if (document->page()->canStartMedia()) {
59 sendUserMediaRequest(request);
60 return;
61 }
62
63 auto& pendingRequests = m_pendingUserMediaRequests.add(document, Vector<Ref<UserMediaRequest>>()).iterator->value;
64 if (pendingRequests.isEmpty())
65 document->addMediaCanStartListener(*this);
66 pendingRequests.append(request);
67}
68
69void UserMediaPermissionRequestManager::sendUserMediaRequest(UserMediaRequest& userRequest)
70{
71 auto* frame = userRequest.document() ? userRequest.document()->frame() : nullptr;
72 if (!frame) {
73 userRequest.deny(UserMediaRequest::OtherFailure, emptyString());
74 return;
75 }
76
77 m_ongoingUserMediaRequests.add(userRequest.identifier(), makeRef(userRequest));
78
79 WebFrame* webFrame = WebFrame::fromCoreFrame(*frame);
80 ASSERT(webFrame);
81
82 auto* topLevelDocumentOrigin = userRequest.topLevelDocumentOrigin();
83 m_page.send(Messages::WebPageProxy::RequestUserMediaPermissionForFrame(userRequest.identifier().toUInt64(), webFrame->frameID(), userRequest.userMediaDocumentOrigin()->data(), topLevelDocumentOrigin->data(), userRequest.request()));
84}
85
86void UserMediaPermissionRequestManager::cancelUserMediaRequest(UserMediaRequest& request)
87{
88 if (auto removedRequest = m_ongoingUserMediaRequests.take(request.identifier()))
89 return;
90
91 auto* document = request.document();
92 if (!document)
93 return;
94
95 auto iterator = m_pendingUserMediaRequests.find(document);
96 if (iterator == m_pendingUserMediaRequests.end())
97 return;
98
99 auto& pendingRequests = iterator->value;
100 pendingRequests.removeFirstMatching([&request](auto& item) {
101 return &request == item.ptr();
102 });
103
104 if (!pendingRequests.isEmpty())
105 return;
106
107 document->removeMediaCanStartListener(*this);
108 m_pendingUserMediaRequests.remove(iterator);
109}
110
111void UserMediaPermissionRequestManager::mediaCanStart(Document& document)
112{
113 ASSERT(document.page()->canStartMedia());
114
115 auto pendingRequests = m_pendingUserMediaRequests.take(&document);
116 for (auto& pendingRequest : pendingRequests)
117 sendUserMediaRequest(pendingRequest);
118}

120void UserMediaPermissionRequestManager::userMediaAccessWasGranted(uint64_t requestID, CaptureDevice&& audioDevice, CaptureDevice&& videoDevice, String&& deviceIdentifierHashSalt, CompletionHandler<void()>&& completionHandler)
121{
122 auto request = m_ongoingUserMediaRequests.take(makeObjectIdentifier<UserMediaRequestIdentifierType>(requestID));
123 if (!request) {
124 completionHandler();
125 return;
126 }
127
128 request->allow(WTFMove(audioDevice), WTFMove(videoDevice), WTFMove(deviceIdentifierHashSalt), WTFMove(completionHandler));
129}
130
131void UserMediaPermissionRequestManager::userMediaAccessWasDenied(uint64_t requestID, UserMediaRequest::MediaAccessDenialReason reason, String&& invalidConstraint)
132{
133 auto request = m_ongoingUserMediaRequests.take(makeObjectIdentifier<UserMediaRequestIdentifierType>(requestID));
134 if (!request)
135 return;
136
137 request->deny(reason, WTFMove(invalidConstraint));
138}

140void UserMediaPermissionRequestManager::enumerateMediaDevices(Document& document, CompletionHandler<void(const Vector<CaptureDevice>&, const String&)>&& completionHandler)
141{
142 auto* frame = document.frame();
143 if (!frame) {
144 completionHandler({ }, emptyString());
145 return;
146 }
147
148 m_page.sendWithAsyncReply(Messages::WebPageProxy::EnumerateMediaDevicesForFrame { WebFrame::fromCoreFrame(*frame)->frameID(), document.securityOrigin().data(), document.topOrigin().data() }, WTFMove(completionHandler));
149}

151#if USE(GSTREAMER)
152void UserMediaPermissionRequestManager::updateCaptureDevices(ShouldNotify shouldNotify)
153{
154 WebCore::RealtimeMediaSourceCenter::singleton().getMediaStreamDevices([weakThis = makeWeakPtr(*this), this, shouldNotify](auto&& newDevices) mutable {
155 if (!weakThis)
156 return;
157
158 if (!haveDevicesChanged(m_captureDevices, newDevices))
159 return;
160
161 m_captureDevices = WTFMove(newDevices);
162 if (shouldNotify == ShouldNotify::Yes)
163 captureDevicesChanged();
164 });
165}

167void UserMediaPermissionRequestManager::devicesChanged()
168{
169 updateCaptureDevices(ShouldNotify::Yes);

173UserMediaClient::DeviceChangeObserverToken UserMediaPermissionRequestManager::addDeviceChangeObserver(Function<void()>&& observer)
174{
175 auto identifier = UserMediaClient::DeviceChangeObserverToken::generate();
176 m_deviceChangeObserverMap.add(identifier, WTFMove(observer));
177
178 if (!m_monitoringDeviceChange) {
179 m_monitoringDeviceChange = true;
180#if USE(GSTREAMER)
181 updateCaptureDevices(ShouldNotify::No);
182 WebCore::RealtimeMediaSourceCenter::singleton().addDevicesChangedObserver(*this);
183#else
184 m_page.send(Messages::WebPageProxy::BeginMonitoringCaptureDevices());
185#endif
186 }
187 return identifier;
188}

190void UserMediaPermissionRequestManager::removeDeviceChangeObserver(UserMediaClient::DeviceChangeObserverToken token)
191{
192 bool wasRemoved = m_deviceChangeObserverMap.remove(token);
193 ASSERT_UNUSED(wasRemoved, wasRemoved);
194}

196void UserMediaPermissionRequestManager::captureDevicesChanged()
197{
198 // When new media input and/or output devices are made available, or any available input and/or
199 // output device becomes unavailable, the User Agent MUST run the following steps in browsing
200 // contexts where at least one of the following criteria are met, but in no other contexts:
201
202 // * The permission state of the "device-info" permission is "granted",
203 // * any of the input devices are attached to an active MediaStream in the browsing context, or
204 // * the active document is fully active and has focus.
205
206 auto identifiers = m_deviceChangeObserverMap.keys();
207 for (auto& identifier : identifiers) {
208 auto iterator = m_deviceChangeObserverMap.find(identifier);
209 if (iterator != m_deviceChangeObserverMap.end())
210 (iterator->value)();
211 }
212}
