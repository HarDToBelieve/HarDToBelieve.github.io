75NetworkProcessConnection::NetworkProcessConnection(IPC::Connection::Identifier connectionIdentifier, HTTPCookieAcceptPolicy cookieAcceptPolicy)
76 : m_connection(IPC::Connection::createClientConnection(connectionIdentifier, *this))
77 , m_cookieAcceptPolicy(cookieAcceptPolicy)
78{
79 m_connection->open();
80
81 if (LibWebRTCProvider::webRTCAvailable())
82 WebProcess::singleton().libWebRTCNetwork().setConnection(m_connection.copyRef());
83}

90void NetworkProcessConnection::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
91{
92 if (decoder.messageReceiverName() == Messages::WebResourceLoader::messageReceiverName()) {
93 if (auto* webResourceLoader = WebProcess::singleton().webLoaderStrategy().webResourceLoaderForIdentifier(decoder.destinationID()))
94 webResourceLoader->didReceiveWebResourceLoaderMessage(connection, decoder);
95 return;
96 }
97 if (decoder.messageReceiverName() == Messages::WebSocketStream::messageReceiverName()) {
98 if (auto* stream = WebSocketStream::streamWithIdentifier(makeObjectIdentifier<WebSocketIdentifierType>(decoder.destinationID())))
99 stream->didReceiveMessage(connection, decoder);
100 return;
101 }
102 if (decoder.messageReceiverName() == Messages::WebSocketChannel::messageReceiverName()) {
103 WebProcess::singleton().webSocketChannelManager().didReceiveMessage(connection, decoder);
104 return;
105 }
106 if (decoder.messageReceiverName() == Messages::WebPage::messageReceiverName()) {
107 if (auto* webPage = WebProcess::singleton().webPage(makeObjectIdentifier<PageIdentifierType>(decoder.destinationID())))
108 webPage->didReceiveWebPageMessage(connection, decoder);
109 return;
110 }
111 if (decoder.messageReceiverName() == Messages::StorageAreaMap::messageReceiverName()) {
112 if (auto* storageAreaMap = WebProcess::singleton().storageAreaMap(makeObjectIdentifier<StorageAreaIdentifierType>(decoder.destinationID())))
113 storageAreaMap->didReceiveMessage(connection, decoder);
114 return;
115 }
116
117#if USE(LIBWEBRTC)
118 if (decoder.messageReceiverName() == Messages::WebRTCMonitor::messageReceiverName()) {
119 auto& network = WebProcess::singleton().libWebRTCNetwork();
120 if (network.isActive())
121 network.monitor().didReceiveMessage(connection, decoder);
122 else
123 RELEASE_LOG_ERROR(WebRTC, "Received WebRTCMonitor message while libWebRTCNetwork is not active");
124 return;
125 }
126 if (decoder.messageReceiverName() == Messages::WebRTCResolver::messageReceiverName()) {
127 auto& network = WebProcess::singleton().libWebRTCNetwork();
128 if (network.isActive())
129 network.resolver(makeObjectIdentifier<LibWebRTCResolverIdentifierType>(decoder.destinationID())).didReceiveMessage(connection, decoder);
130 else
131 RELEASE_LOG_ERROR(WebRTC, "Received WebRTCResolver message while libWebRTCNetwork is not active");
132 return;
133 }
134#endif
135#if ENABLE(WEB_RTC)
136 if (decoder.messageReceiverName() == Messages::WebMDNSRegister::messageReceiverName()) {
137 auto& network = WebProcess::singleton().libWebRTCNetwork();
138 if (network.isActive())
139 network.mdnsRegister().didReceiveMessage(connection, decoder);
140 else
141 RELEASE_LOG_ERROR(WebRTC, "Received WebMDNSRegister message while libWebRTCNetwork is not active");
142 return;
143 }
144#endif
145
146 if (decoder.messageReceiverName() == Messages::WebIDBConnectionToServer::messageReceiverName()) {
147 if (m_webIDBConnection)
148 m_webIDBConnection->didReceiveMessage(connection, decoder);
149 return;
150 }
151
152#if ENABLE(SERVICE_WORKER)
153 if (decoder.messageReceiverName() == Messages::WebSWClientConnection::messageReceiverName()) {
154 serviceWorkerConnection().didReceiveMessage(connection, decoder);
155 return;
156 }
157 if (decoder.messageReceiverName() == Messages::WebSWContextManagerConnection::messageReceiverName()) {
158 ASSERT(SWContextManager::singleton().connection());
159 if (auto* contextManagerConnection = SWContextManager::singleton().connection())
160 static_cast<WebSWContextManagerConnection&>(*contextManagerConnection).didReceiveMessage(connection, decoder);
161 return;
162 }
163#endif
164
165#if ENABLE(APPLE_PAY_REMOTE_UI)
166 if (decoder.messageReceiverName() == Messages::WebPaymentCoordinator::messageReceiverName()) {
167 if (auto webPage = WebProcess::singleton().webPage(makeObjectIdentifier<PageIdentifierType>(decoder.destinationID())))
168 webPage->paymentCoordinator()->didReceiveMessage(connection, decoder);
169 return;
170 }
171#endif
172
173 didReceiveNetworkProcessConnectionMessage(connection, decoder);
174}

176bool NetworkProcessConnection::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
177{
178#if ENABLE(SERVICE_WORKER)
179 if (decoder.messageReceiverName() == Messages::WebSWContextManagerConnection::messageReceiverName()) {
180 ASSERT(SWContextManager::singleton().connection());
181 if (auto* contextManagerConnection = SWContextManager::singleton().connection())
182 return static_cast<WebSWContextManagerConnection&>(*contextManagerConnection).didReceiveSyncMessage(connection, decoder, replyEncoder);
183 return false;
184 }
185#endif
186
187#if ENABLE(APPLE_PAY_REMOTE_UI)
188 if (decoder.messageReceiverName() == Messages::WebPaymentCoordinator::messageReceiverName()) {
189 if (auto webPage = WebProcess::singleton().webPage(makeObjectIdentifier<PageIdentifierType>(decoder.destinationID())))
190 return webPage->paymentCoordinator()->didReceiveSyncMessage(connection, decoder, replyEncoder);
191 return false;
192 }
193#endif
194
195 ASSERT_NOT_REACHED();
196 return false;
197}

199void NetworkProcessConnection::didClose(IPC::Connection&)
200{
201 // The NetworkProcess probably crashed.
202 Ref<NetworkProcessConnection> protector(*this);
203 WebProcess::singleton().networkProcessConnectionClosed(this);
204
205 if (auto idbConnection = std::exchange(m_webIDBConnection, nullptr))
206 idbConnection->connectionToServerLost();
207
208#if ENABLE(SERVICE_WORKER)
209 if (auto swConnection = std::exchange(m_swConnection, nullptr))
210 swConnection->connectionToServerLost();
211#endif
212}

265#if ENABLE(SHAREABLE_RESOURCE)
266void NetworkProcessConnection::didCacheResource(const ResourceRequest& request, const ShareableResource::Handle& handle)
267{
268 auto* resource = MemoryCache::singleton().resourceForRequest(request, WebProcess::singleton().sessionID());
269 if (!resource)
270 return;
271
272 RefPtr<SharedBuffer> buffer = handle.tryWrapInSharedBuffer();
273 if (!buffer) {
274 LOG_ERROR("Unable to create SharedBuffer from ShareableResource handle for resource url %s", request.url().string().utf8().data());
275 return;
276 }
277
278 resource->tryReplaceEncodedData(*buffer);
279}
280#endif

308void NetworkProcessConnection::broadcastConsoleMessage(MessageSource source, MessageLevel level, const String& message)
309{
310 FAST_RETURN_IF_NO_FRONTENDS(void());
311
312 for (auto* frame : WebProcess::singleton().webFrames()) {
313 if (frame->isMainFrame())
314 frame->addConsoleMessage(source, level, message);
315 }
316}
