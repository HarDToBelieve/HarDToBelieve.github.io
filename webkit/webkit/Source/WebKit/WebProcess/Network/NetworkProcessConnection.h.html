48namespace WebKit {
49
50class WebIDBConnectionToServer;
51class WebSWClientConnection;
52
53typedef uint64_t ResourceLoadIdentifier;
54
55class NetworkProcessConnection : public RefCounted<NetworkProcessConnection>, IPC::Connection::Client {
56public:
57 static Ref<NetworkProcessConnection> create(IPC::Connection::Identifier connectionIdentifier, WebCore::HTTPCookieAcceptPolicy httpCookieAcceptPolicy)
58 {
59 return adoptRef(*new NetworkProcessConnection(connectionIdentifier, httpCookieAcceptPolicy));
60 }
61 ~NetworkProcessConnection();
62
63 IPC::Connection& connection() { return m_connection.get(); }
64
65 void didReceiveNetworkProcessConnectionMessage(IPC::Connection&, IPC::Decoder&);
66
67 void writeBlobsToTemporaryFiles(const Vector<String>& blobURLs, CompletionHandler<void(Vector<String>&& filePaths)>&&);
68
69 WebIDBConnectionToServer* existingIDBConnectionToServer() const { return m_webIDBConnection.get(); };
70 WebIDBConnectionToServer& idbConnectionToServer();
71
72#if ENABLE(SERVICE_WORKER)
73 WebSWClientConnection& serviceWorkerConnection();
74#endif
75
76#if HAVE(AUDIT_TOKEN)
77 void setNetworkProcessAuditToken(std::optional<audit_token_t> auditToken) { m_networkProcessAuditToken = auditToken; }
78 std::optional<audit_token_t> networkProcessAuditToken() const { return m_networkProcessAuditToken; }
79#endif
80
81 WebCore::HTTPCookieAcceptPolicy cookieAcceptPolicy() const { return m_cookieAcceptPolicy; }
82 bool cookiesEnabled() const;
83
84#if HAVE(COOKIE_CHANGE_LISTENER_API)
85 void cookiesAdded(const String& host, const Vector<WebCore::Cookie>&);
86 void cookiesDeleted(const String& host, const Vector<WebCore::Cookie>&);
87 void allCookiesDeleted();
88#endif
89
90private:
91 NetworkProcessConnection(IPC::Connection::Identifier, WebCore::HTTPCookieAcceptPolicy);
92
93 // IPC::Connection::Client
94 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
95 bool didReceiveSyncMessage(IPC::Connection&, IPC::Decoder&, UniqueRef<IPC::Encoder>&) override;
96 void didClose(IPC::Connection&) override;
97 void didReceiveInvalidMessage(IPC::Connection&, IPC::MessageName) override;
98
99 void didFinishPingLoad(uint64_t pingLoadIdentifier, WebCore::ResourceError&&, WebCore::ResourceResponse&&);
100 void didFinishPreconnection(uint64_t preconnectionIdentifier, WebCore::ResourceError&&);
101 void setOnLineState(bool isOnLine);
102 void cookieAcceptPolicyChanged(WebCore::HTTPCookieAcceptPolicy);
103
104 void checkProcessLocalPortForActivity(const WebCore::MessagePortIdentifier&, CompletionHandler<void(WebCore::MessagePortChannelProvider::HasActivity)>&&);
105 void messagesAvailableForPort(const WebCore::MessagePortIdentifier&);
106
107#if ENABLE(SHAREABLE_RESOURCE)
108 // Message handlers.
109 void didCacheResource(const WebCore::ResourceRequest&, const ShareableResource::Handle&);
110#endif
111#if ENABLE(WEB_RTC)
112 void connectToRTCDataChannelRemoteSource(WebCore::RTCDataChannelIdentifier source, WebCore::RTCDataChannelIdentifier handler, CompletionHandler<void(std::optional<bool>)>&&);
113#endif
114
115 void broadcastConsoleMessage(MessageSource, MessageLevel, const String& message);
116
117 // The connection from the web process to the network process.
118 Ref<IPC::Connection> m_connection;
119#if HAVE(AUDIT_TOKEN)
120 std::optional<audit_token_t> m_networkProcessAuditToken;
121#endif
122
123 RefPtr<WebIDBConnectionToServer> m_webIDBConnection;
124
125#if ENABLE(SERVICE_WORKER)
126 RefPtr<WebSWClientConnection> m_swConnection;
127#endif
128 WebCore::HTTPCookieAcceptPolicy m_cookieAcceptPolicy;
129};
