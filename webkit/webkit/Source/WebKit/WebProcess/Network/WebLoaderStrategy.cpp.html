105void WebLoaderStrategy::loadResource(Frame& frame, CachedResource& resource, ResourceRequest&& request, const ResourceLoaderOptions& options, CompletionHandler<void(RefPtr<SubresourceLoader>&&)>&& completionHandler)
106{
107 if (resource.type() != CachedResource::Type::MainResource || !frame.isMainFrame()) {
108 if (auto* document = frame.mainFrame().document()) {
109 if (document && document->loader())
110 request.setIsAppBound(document->loader()->lastNavigationWasAppBound());
111 }
112 }
113
114 SubresourceLoader::create(frame, resource, WTFMove(request), options, [this, referrerPolicy = options.referrerPolicy, completionHandler = WTFMove(completionHandler), resource = CachedResourceHandle<CachedResource>(&resource), frame = makeRef(frame)] (RefPtr<SubresourceLoader>&& loader) mutable {
115 if (loader)
116 scheduleLoad(*loader, resource.get(), referrerPolicy == ReferrerPolicy::NoReferrerWhenDowngrade);
117 else
118 RELEASE_LOG(Network, "%p - [webPageID=%" PRIu64 ", frameID=%" PRIu64 "] WebLoaderStrategy::loadResource: Unable to create SubresourceLoader", this, frame->pageID().value_or(PageIdentifier()).toUInt64(), frame->frameID().value_or(FrameIdentifier()).toUInt64());
119 completionHandler(WTFMove(loader));
120 });
121}

132static Seconds maximumBufferingTime(CachedResource* resource)
133{
134 if (!resource)
135 return 0_s;
136
137 switch (resource->type()) {
138 case CachedResource::Type::Beacon:
139 case CachedResource::Type::Ping:
140 case CachedResource::Type::CSSStyleSheet:
141 case CachedResource::Type::Script:
142 case CachedResource::Type::SVGFontResource:
143 case CachedResource::Type::FontResource:
144#if ENABLE(APPLICATION_MANIFEST)
145 case CachedResource::Type::ApplicationManifest:
146#endif
147#if ENABLE(MODEL_ELEMENT)
148 case CachedResource::Type::ModelResource:
149#endif
150 return Seconds::infinity();
151 case CachedResource::Type::ImageResource:
152 return 500_ms;
153 case CachedResource::Type::MediaResource:
154 return 50_ms;
155 case CachedResource::Type::MainResource:
156 case CachedResource::Type::Icon:
157 case CachedResource::Type::RawResource:
158 case CachedResource::Type::SVGDocumentResource:
159 case CachedResource::Type::LinkPrefetch:
160 case CachedResource::Type::TextTrackResource:
161#if ENABLE(XSLT)
162 case CachedResource::Type::XSLStyleSheet:
163#endif
164 return 0_s;
165 }
166
167 ASSERT_NOT_REACHED();
168 return 0_s;
169}

171void WebLoaderStrategy::scheduleLoad(ResourceLoader& resourceLoader, CachedResource* resource, bool shouldClearReferrerOnHTTPSToHTTPRedirect)
172{
173 ResourceLoadIdentifier identifier = resourceLoader.identifier();
174 ASSERT(identifier);
175
176 auto& frameLoaderClient = resourceLoader.frameLoader()->client();
177
178 WebResourceLoader::TrackingParameters trackingParameters;
179 if (auto* webFrameLoaderClient = toWebFrameLoaderClient(frameLoaderClient))
180 trackingParameters.webPageProxyID = webFrameLoaderClient->webPageProxyID().value_or(WebPageProxyIdentifier { });
181#if ENABLE(SERVICE_WORKER)
182 else if (is<ServiceWorkerFrameLoaderClient>(frameLoaderClient))
183 trackingParameters.webPageProxyID = downcast<ServiceWorkerFrameLoaderClient>(frameLoaderClient).webPageProxyID();
184#endif
185 trackingParameters.pageID = frameLoaderClient.pageID().value_or(PageIdentifier { });
186 trackingParameters.frameID = frameLoaderClient.frameID().value_or(FrameIdentifier { });
187 trackingParameters.resourceID = identifier;
188
189#if ENABLE(WEB_ARCHIVE) || ENABLE(MHTML)
190 // If the DocumentLoader schedules this as an archive resource load,
191 // then we should remember the ResourceLoader in our records but not schedule it in the NetworkProcess.
192 if (resourceLoader.documentLoader()->scheduleArchiveLoad(resourceLoader, resourceLoader.request())) {
193 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::scheduleLoad, url '%s' will be handled as an archive resource.", resourceLoader.url().string().utf8().data());
194 WEBLOADERSTRATEGY_RELEASE_LOG("scheduleLoad: URL will be handled as an archive resource");
195 m_webResourceLoaders.set(identifier, WebResourceLoader::create(resourceLoader, trackingParameters));
196 return;
197 }
198#endif
199
200 if (resourceLoader.documentLoader()->applicationCacheHost().maybeLoadResource(resourceLoader, resourceLoader.request(), resourceLoader.request().url())) {
201 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::scheduleLoad, url '%s' will be loaded from application cache.", resourceLoader.url().string().utf8().data());
202 WEBLOADERSTRATEGY_RELEASE_LOG("scheduleLoad: URL will be loaded from application cache");
203 m_webResourceLoaders.set(identifier, WebResourceLoader::create(resourceLoader, trackingParameters));
204 return;
205 }
206
207 if (resourceLoader.request().url().protocolIsData()) {
208 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::scheduleLoad, url '%s' will be loaded as data.", resourceLoader.url().string().utf8().data());
209 WEBLOADERSTRATEGY_RELEASE_LOG("scheduleLoad: URL will be loaded as data");
210 startLocalLoad(resourceLoader);
211 return;
212 }
213
214#if USE(QUICK_LOOK)
215 if (isQuickLookPreviewURL(resourceLoader.request().url())) {
216 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::scheduleLoad, url '%s' will be handled as a QuickLook resource.", resourceLoader.url().string().utf8().data());
217 WEBLOADERSTRATEGY_RELEASE_LOG("scheduleLoad: URL will be handled as a QuickLook resource");
218 startLocalLoad(resourceLoader);
219 return;
220 }
221#endif
222
223#if USE(SOUP)
224 // For apps that call g_resource_load in a web extension.
225 // https://blogs.gnome.org/alexl/2012/01/26/resources-in-glib/
226 if (resourceLoader.request().url().protocolIs("resource")) {
227 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::scheduleLoad, url '%s' will be handled as a GResource.", resourceLoader.url().string().utf8().data());
228 WEBLOADERSTRATEGY_RELEASE_LOG("scheduleLoad: URL will be handled as a GResource");
229 startLocalLoad(resourceLoader);
230 return;
231 }
232#endif
233
234 if (!tryLoadingUsingURLSchemeHandler(resourceLoader, trackingParameters)) {
235 WEBLOADERSTRATEGY_RELEASE_LOG("scheduleLoad: URL will be scheduled with the NetworkProcess");
236
237#if ENABLE(SERVICE_WORKER)
238 if (!resourceLoader.options().serviceWorkerRegistrationIdentifier && InspectorInstrumentationWebKit::shouldInterceptRequest(resourceLoader.frame(), resourceLoader.request())) {
239 InspectorInstrumentationWebKit::interceptRequest(resourceLoader, [this, protectedResourceLoader = makeRefPtr(&resourceLoader), trackingParameters, shouldClearReferrerOnHTTPSToHTTPRedirect, resource](const ResourceRequest& request) {
240 scheduleLoadFromNetworkProcess(*protectedResourceLoader, request, trackingParameters, shouldClearReferrerOnHTTPSToHTTPRedirect, maximumBufferingTime(resource));
241 });
242 return;
243 }
244#endif
245 scheduleLoadFromNetworkProcess(resourceLoader, resourceLoader.request(), trackingParameters, shouldClearReferrerOnHTTPSToHTTPRedirect, maximumBufferingTime(resource));
246 return;
247 }
248}

250bool WebLoaderStrategy::tryLoadingUsingURLSchemeHandler(ResourceLoader& resourceLoader, const WebResourceLoader::TrackingParameters& trackingParameters)
251{
252 auto* webFrameLoaderClient = toWebFrameLoaderClient(resourceLoader.frameLoader()->client());
253 if (!webFrameLoaderClient)
254 return false;
255
256 auto& webFrame = webFrameLoaderClient->webFrame();
257 auto* webPage = webFrame.page();
258 if (!webPage)
259 return false;
260
261 auto* handler = webPage->urlSchemeHandlerForScheme(resourceLoader.request().url().protocol().toStringWithoutCopying());
262 if (!handler)
263 return false;
264
265 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::scheduleLoad, URL '%s' will be handled by a UIProcess URL scheme handler.", resourceLoader.url().string().utf8().data());
266 WEBLOADERSTRATEGY_RELEASE_LOG("tryLoadingUsingURLSchemeHandler: URL will be handled by a UIProcess URL scheme handler");
267
268 handler->startNewTask(resourceLoader, webFrame);
269 return true;
270}

272static void addParametersShared(const Frame* frame, NetworkResourceLoadParameters& parameters)
273{
274 parameters.crossOriginAccessControlCheckEnabled = CrossOriginAccessControlCheckDisabler::singleton().crossOriginAccessControlCheckEnabled();
275
276 if (!frame)
277 return;
278
279 if (auto* page = frame->page()) {
280 parameters.pageHasResourceLoadClient = page->hasResourceLoadClient();
281 parameters.shouldRelaxThirdPartyCookieBlocking = page->shouldRelaxThirdPartyCookieBlocking();
282 page->logMediaDiagnosticMessage(parameters.request.httpBody());
283 }
284
285 if (auto* ownerElement = frame->ownerElement()) {
286 if (auto* parentFrame = ownerElement->document().frame())
287 parameters.parentFrameID = parentFrame->loader().frameID();
288 }
289}

291void WebLoaderStrategy::scheduleLoadFromNetworkProcess(ResourceLoader& resourceLoader, const ResourceRequest& request, const WebResourceLoader::TrackingParameters& trackingParameters, bool shouldClearReferrerOnHTTPSToHTTPRedirect, Seconds maximumBufferingTime)
292{
293 ResourceLoadIdentifier identifier = resourceLoader.identifier();
294 ASSERT(identifier);
295
296 ContentSniffingPolicy contentSniffingPolicy = resourceLoader.shouldSniffContent() ? ContentSniffingPolicy::SniffContent : ContentSniffingPolicy::DoNotSniffContent;
297 ContentEncodingSniffingPolicy contentEncodingSniffingPolicy = resourceLoader.shouldSniffContentEncoding() ? ContentEncodingSniffingPolicy::Sniff : ContentEncodingSniffingPolicy::DoNotSniff;
298 StoredCredentialsPolicy storedCredentialsPolicy = resourceLoader.shouldUseCredentialStorage() ? StoredCredentialsPolicy::Use : StoredCredentialsPolicy::DoNotUse;
299
300 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::scheduleLoad, url '%s' will be scheduled with the NetworkProcess with priority %d, storedCredentialsPolicy %i", resourceLoader.url().string().latin1().data(), static_cast<int>(resourceLoader.request().priority()), (int)storedCredentialsPolicy);
301
302 auto* frame = resourceLoader.frame();
303
304 NetworkResourceLoadParameters loadParameters;
305 loadParameters.identifier = identifier;
306 loadParameters.webPageProxyID = trackingParameters.webPageProxyID;
307 loadParameters.webPageID = trackingParameters.pageID;
308 loadParameters.webFrameID = trackingParameters.frameID;
309 loadParameters.parentPID = presentingApplicationPID();
310#if HAVE(AUDIT_TOKEN)
311 loadParameters.networkProcessAuditToken = WebProcess::singleton().ensureNetworkProcessConnection().networkProcessAuditToken();
312#endif
313 loadParameters.request = request;
314 loadParameters.contentSniffingPolicy = contentSniffingPolicy;
315 loadParameters.contentEncodingSniffingPolicy = contentEncodingSniffingPolicy;
316 loadParameters.storedCredentialsPolicy = storedCredentialsPolicy;
317 // If there is no WebFrame then this resource cannot be authenticated with the client.
318 loadParameters.clientCredentialPolicy = (loadParameters.webFrameID && loadParameters.webPageID && resourceLoader.isAllowedToAskUserForCredentials()) ? ClientCredentialPolicy::MayAskClientForCredentials : ClientCredentialPolicy::CannotAskClientForCredentials;
319 loadParameters.shouldClearReferrerOnHTTPSToHTTPRedirect = shouldClearReferrerOnHTTPSToHTTPRedirect;
320 loadParameters.needsCertificateInfo = resourceLoader.shouldIncludeCertificateInfo();
321 loadParameters.maximumBufferingTime = maximumBufferingTime;
322 loadParameters.options = resourceLoader.options();
323 loadParameters.preflightPolicy = resourceLoader.options().preflightPolicy;
324 addParametersShared(frame, loadParameters);
325
326#if ENABLE(SERVICE_WORKER)
327 loadParameters.serviceWorkersMode = resourceLoader.options().loadedFromOpaqueSource == LoadedFromOpaqueSource::No ? resourceLoader.options().serviceWorkersMode : ServiceWorkersMode::None;
328 loadParameters.serviceWorkerRegistrationIdentifier = resourceLoader.options().serviceWorkerRegistrationIdentifier;
329 loadParameters.httpHeadersToKeep = resourceLoader.options().httpHeadersToKeep;
330#endif
331
332 auto* document = frame ? frame->document() : nullptr;
333 if (resourceLoader.options().cspResponseHeaders)
334 loadParameters.cspResponseHeaders = resourceLoader.options().cspResponseHeaders;
335 else if (document && !document->shouldBypassMainWorldContentSecurityPolicy() && resourceLoader.options().contentSecurityPolicyImposition == ContentSecurityPolicyImposition::DoPolicyCheck) {
336 if (auto* contentSecurityPolicy = document->contentSecurityPolicy())
337 loadParameters.cspResponseHeaders = contentSecurityPolicy->responseHeaders();
338 }
339
340#if ENABLE(APP_BOUND_DOMAINS) || ENABLE(CONTENT_EXTENSIONS)
341 auto* webFrameLoaderClient = frame ? toWebFrameLoaderClient(frame->loader().client()) : nullptr;
342 auto* webFrame = webFrameLoaderClient ? &webFrameLoaderClient->webFrame() : nullptr;
343#endif
344
345#if ENABLE(APP_BOUND_DOMAINS)
346 if (webFrame)
347 loadParameters.isNavigatingToAppBoundDomain = webFrame->isTopFrameNavigatingToAppBoundDomain();
348#endif
349
350#if ENABLE(CONTENT_EXTENSIONS)
351 if (document) {
352 loadParameters.mainDocumentURL = document->topDocument().url();
353 // FIXME: Instead of passing userContentControllerIdentifier, the NetworkProcess should be able to get it using webPageId.
354 if (auto* webPage = webFrame ? webFrame->page() : nullptr)
355 loadParameters.userContentControllerIdentifier = webPage->userContentControllerIdentifier();
356 }
357#endif
358
359 // FIXME: All loaders should provide their origin if navigation mode is cors/no-cors/same-origin.
360 // As a temporary approach, we use the document origin if available or the HTTP Origin header otherwise.
361 if (is<SubresourceLoader>(resourceLoader)) {
362 auto& loader = downcast<SubresourceLoader>(resourceLoader);
363 loadParameters.sourceOrigin = loader.origin();
364
365 if (auto* headers = loader.originalHeaders())
366 loadParameters.originalRequestHeaders = *headers;
367 }
368
369 if (!loadParameters.sourceOrigin && document)
370 loadParameters.sourceOrigin = &document->securityOrigin();
371 if (!loadParameters.sourceOrigin) {
372 auto origin = request.httpOrigin();
373 if (!origin.isNull())
374 loadParameters.sourceOrigin = SecurityOrigin::createFromString(origin);
375 }
376 if (document) {
377 loadParameters.topOrigin = &document->topOrigin();
378 loadParameters.documentURL = document->url();
379 }
380
381 if (loadParameters.options.mode != FetchOptions::Mode::Navigate) {
382 ASSERT(loadParameters.sourceOrigin);
383 if (!loadParameters.sourceOrigin) {
384 WEBLOADERSTRATEGY_RELEASE_LOG_ERROR("scheduleLoad: no sourceOrigin (priority=%d)", static_cast<int>(resourceLoader.request().priority()));
385 scheduleInternallyFailedLoad(resourceLoader);
386 return;
387 }
388 }
389
390 loadParameters.shouldRestrictHTTPResponseAccess = shouldPerformSecurityChecks();
391
392 loadParameters.isMainFrameNavigation = resourceLoader.frame() && resourceLoader.frame()->isMainFrame() && resourceLoader.options().mode == FetchOptions::Mode::Navigate;
393
394 loadParameters.isMainResourceNavigationForAnyFrame = resourceLoader.frame() && resourceLoader.options().mode == FetchOptions::Mode::Navigate;
395
396 loadParameters.shouldEnableCrossOriginResourcePolicy = !loadParameters.isMainFrameNavigation;
397
398 if (resourceLoader.options().mode == FetchOptions::Mode::Navigate) {
399 Vector<RefPtr<SecurityOrigin>> frameAncestorOrigins;
400 for (auto* frame = resourceLoader.frame()->tree().parent(); frame; frame = frame->tree().parent())
401 frameAncestorOrigins.append(makeRefPtr(frame->document()->securityOrigin()));
402 loadParameters.frameAncestorOrigins = WTFMove(frameAncestorOrigins);
403 }
404
405 ASSERT((loadParameters.webPageID && loadParameters.webFrameID) || loadParameters.clientCredentialPolicy == ClientCredentialPolicy::CannotAskClientForCredentials);
406
407 WEBLOADERSTRATEGY_RELEASE_LOG("scheduleLoad: Resource is being scheduled with the NetworkProcess (priority=%d)", static_cast<int>(resourceLoader.request().priority()));
408 if (!WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::ScheduleResourceLoad(loadParameters), 0)) {
409 WEBLOADERSTRATEGY_RELEASE_LOG_ERROR("scheduleLoad: Unable to schedule resource with the NetworkProcess (priority=%d)", static_cast<int>(resourceLoader.request().priority()));
410 // We probably failed to schedule this load with the NetworkProcess because it had crashed.
411 // This load will never succeed so we will schedule it to fail asynchronously.
412 scheduleInternallyFailedLoad(resourceLoader);
413 return;
414 }
415
416 auto loader = WebResourceLoader::create(resourceLoader, trackingParameters);
417 m_webResourceLoaders.set(identifier, WTFMove(loader));
418}

449void WebLoaderStrategy::remove(ResourceLoader* resourceLoader)
450{
451 ASSERT(resourceLoader);
452 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::remove, url '%s'", resourceLoader->url().string().utf8().data());
453
454 if (auto task = m_urlSchemeTasks.take(resourceLoader->identifier())) {
455 ASSERT(!m_internallyFailedResourceLoaders.contains(resourceLoader));
456 task->stopLoading();
457 return;
458 }
459
460 if (m_internallyFailedResourceLoaders.contains(resourceLoader)) {
461 m_internallyFailedResourceLoaders.remove(resourceLoader);
462 return;
463 }
464
465 ResourceLoadIdentifier identifier = resourceLoader->identifier();
466 if (!identifier) {
467 LOG_ERROR("WebLoaderStrategy removing a ResourceLoader that has no identifier.");
468 return;
469 }
470
471 RefPtr<WebResourceLoader> loader = m_webResourceLoaders.take(identifier);
472 // Loader may not be registered if we created it, but haven't scheduled yet (a bundle client can decide to cancel such request via willSendRequest).
473 if (!loader)
474 return;
475
476 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::RemoveLoadIdentifier(identifier), 0);
477
478 // It's possible that this WebResourceLoader might be just about to message back to the NetworkProcess (e.g. ContinueWillSendRequest)
479 // but there's no point in doing so anymore.
480 loader->detachFromCoreLoader();
481}

509void WebLoaderStrategy::networkProcessCrashed()
510{
511 WEBLOADERSTRATEGY_RELEASE_LOG_ERROR_BASIC("networkProcessCrashed: failing all pending resource loaders");
512
513 for (auto& loader : m_webResourceLoaders.values()) {
514 scheduleInternallyFailedLoad(*loader->resourceLoader());
515 loader->detachFromCoreLoader();
516 }
517
518 m_webResourceLoaders.clear();
519
520 auto pingLoadCompletionHandlers = WTFMove(m_pingLoadCompletionHandlers);
521 for (auto& pingLoadCompletionHandler : pingLoadCompletionHandlers.values())
522 pingLoadCompletionHandler(internalError(URL()), { });
523
524 auto preconnectCompletionHandlers = WTFMove(m_preconnectCompletionHandlers);
525 for (auto& preconnectCompletionHandler : preconnectCompletionHandlers.values())
526 preconnectCompletionHandler(internalError(URL()));
527}

538WebLoaderStrategy::SyncLoadResult WebLoaderStrategy::loadDataURLSynchronously(const ResourceRequest& request)
539{
540 auto mode = DataURLDecoder::Mode::Legacy;
541 if (request.requester() == ResourceRequest::Requester::Fetch)
542 mode = DataURLDecoder::Mode::ForgivingBase64;
543
544 SyncLoadResult result;
545 auto decodeResult = DataURLDecoder::decode(request.url(), mode);
546 if (!decodeResult) {
547 WEBLOADERSTRATEGY_RELEASE_LOG_BASIC("loadDataURLSynchronously: decoding of data failed");
548 result.error = internalError(request.url());
549 return result;
550 }
551
552 result.response = ResourceResponse::dataURLResponse(request.url(), decodeResult.value());
553 result.data = WTFMove(decodeResult->data);
554
555 return result;
556}

558std::optional<WebLoaderStrategy::SyncLoadResult> WebLoaderStrategy::tryLoadingSynchronouslyUsingURLSchemeHandler(FrameLoader& frameLoader, ResourceLoadIdentifier identifier, const ResourceRequest& request)
559{
560 auto* webFrameLoaderClient = toWebFrameLoaderClient(frameLoader.client());
561 auto* webFrame = webFrameLoaderClient ? &webFrameLoaderClient->webFrame() : nullptr;
562 auto* webPage = webFrame ? webFrame->page() : nullptr;
563 if (!webPage)
564 return std::nullopt;
565
566 auto* handler = webPage->urlSchemeHandlerForScheme(request.url().protocol().toStringWithoutCopying());
567 if (!handler)
568 return std::nullopt;
569
570 LOG(NetworkScheduling, "(WebProcess) WebLoaderStrategy::scheduleLoad, sync load to URL '%s' will be handled by a UIProcess URL scheme handler.", request.url().string().utf8().data());
571
572 SyncLoadResult result;
573 handler->loadSynchronously(identifier, *webFrame, request, result.response, result.error, result.data);
574
575 return result;
576}

578void WebLoaderStrategy::loadResourceSynchronously(FrameLoader& frameLoader, unsigned long resourceLoadIdentifier, const ResourceRequest& request, ClientCredentialPolicy clientCredentialPolicy, const FetchOptions& options, const HTTPHeaderMap& originalRequestHeaders, ResourceError& error, ResourceResponse& response, Vector<uint8_t>& data)
579{
580 auto* webFrameLoaderClient = toWebFrameLoaderClient(frameLoader.client());
581 auto* webFrame = webFrameLoaderClient ? &webFrameLoaderClient->webFrame() : nullptr;
582 auto* webPage = webFrame ? webFrame->page() : nullptr;
583 auto* page = webPage ? webPage->corePage() : nullptr;
584
585 auto webPageProxyID = webPage ? webPage->webPageProxyIdentifier() : WebPageProxyIdentifier { };
586 auto pageID = webPage ? webPage->identifier() : PageIdentifier { };
587 auto frameID = webFrame ? webFrame->frameID() : FrameIdentifier { };
588
589 WebResourceLoader::TrackingParameters trackingParameters;
590 trackingParameters.pageID = pageID;
591 trackingParameters.frameID = frameID;
592 trackingParameters.resourceID = resourceLoadIdentifier;
593
594 auto* document = frameLoader.frame().document();
595 if (!document) {
596 WEBLOADERSTRATEGY_WITH_FRAMELOADER_RELEASE_LOG_ERROR("loadResourceSynchronously: no document");
597 error = internalError(request.url());
598 return;
599 }
600
601 if (request.url().protocolIsData()) {
602 WEBLOADERSTRATEGY_WITH_FRAMELOADER_RELEASE_LOG_ERROR("loadResourceSynchronously: URL will be loaded as data");
603 auto syncLoadResult = loadDataURLSynchronously(request);
604 error = WTFMove(syncLoadResult.error);
605 response = WTFMove(syncLoadResult.response);
606 data = WTFMove(syncLoadResult.data);
607 return;
608 }
609
610 if (auto syncLoadResult = tryLoadingSynchronouslyUsingURLSchemeHandler(frameLoader, resourceLoadIdentifier, request)) {
611 WEBLOADERSTRATEGY_WITH_FRAMELOADER_RELEASE_LOG_ERROR("loadResourceSynchronously: failed calling tryLoadingSynchronouslyUsingURLSchemeHandler (error=%d)", syncLoadResult->error.errorCode());
612 error = WTFMove(syncLoadResult->error);
613 response = WTFMove(syncLoadResult->response);
614 data = WTFMove(syncLoadResult->data);
615 return;
616 }
617
618 NetworkResourceLoadParameters loadParameters;
619 loadParameters.identifier = resourceLoadIdentifier;
620 loadParameters.webPageProxyID = webPageProxyID;
621 loadParameters.webPageID = pageID;
622 loadParameters.webFrameID = frameID;
623 loadParameters.parentPID = presentingApplicationPID();
624#if HAVE(AUDIT_TOKEN)
625 loadParameters.networkProcessAuditToken = WebProcess::singleton().ensureNetworkProcessConnection().networkProcessAuditToken();
626#endif
627 loadParameters.request = request;
628 loadParameters.contentSniffingPolicy = ContentSniffingPolicy::SniffContent;
629 loadParameters.contentEncodingSniffingPolicy = ContentEncodingSniffingPolicy::Sniff;
630 loadParameters.storedCredentialsPolicy = options.credentials == FetchOptions::Credentials::Omit ? StoredCredentialsPolicy::DoNotUse : StoredCredentialsPolicy::Use;
631 loadParameters.clientCredentialPolicy = clientCredentialPolicy;
632 loadParameters.shouldClearReferrerOnHTTPSToHTTPRedirect = shouldClearReferrerOnHTTPSToHTTPRedirect(webFrame ? webFrame->coreFrame() : nullptr);
633 loadParameters.shouldRestrictHTTPResponseAccess = shouldPerformSecurityChecks();
634
635 loadParameters.options = options;
636 loadParameters.sourceOrigin = &document->securityOrigin();
637 loadParameters.topOrigin = &document->topOrigin();
638 if (!document->shouldBypassMainWorldContentSecurityPolicy()) {
639 if (auto* contentSecurityPolicy = document->contentSecurityPolicy())
640 loadParameters.cspResponseHeaders = contentSecurityPolicy->responseHeaders();
641 }
642 loadParameters.originalRequestHeaders = originalRequestHeaders;
643#if ENABLE(APP_BOUND_DOMAINS)
644 if (webFrame)
645 loadParameters.isNavigatingToAppBoundDomain = webFrame->isTopFrameNavigatingToAppBoundDomain();
646#endif
647 addParametersShared(webFrame->coreFrame(), loadParameters);
648
649 data.shrink(0);
650
651 HangDetectionDisabler hangDetectionDisabler;
652 IPC::UnboundedSynchronousIPCScope unboundedSynchronousIPCScope;
653
654 if (!WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::NetworkConnectionToWebProcess::PerformSynchronousLoad(loadParameters), Messages::NetworkConnectionToWebProcess::PerformSynchronousLoad::Reply(error, response, data), 0)) {
655 WEBLOADERSTRATEGY_WITH_FRAMELOADER_RELEASE_LOG_ERROR("loadResourceSynchronously: failed sending synchronous network process message");
656 if (page)
657 page->diagnosticLoggingClient().logDiagnosticMessage(WebCore::DiagnosticLoggingKeys::internalErrorKey(), WebCore::DiagnosticLoggingKeys::synchronousMessageFailedKey(), WebCore::ShouldSample::No);
658 response = ResourceResponse();
659 error = internalError(request.url());
660 }
661}

686void WebLoaderStrategy::startPingLoad(Frame& frame, ResourceRequest& request, const HTTPHeaderMap& originalRequestHeaders, const FetchOptions& options, ContentSecurityPolicyImposition policyCheck, PingLoadCompletionHandler&& completionHandler)
687{
688 auto* webFrame = WebFrame::fromCoreFrame(frame);
689 auto* document = frame.document();
690 if (!document || !webFrame) {
691 if (completionHandler)
692 completionHandler(internalError(request.url()), { });
693 return;
694 }
695
696 auto* webPage = webFrame->page();
697 if (!webPage) {
698 if (completionHandler)
699 completionHandler(internalError(request.url()), { });
700 return;
701 }
702
703 NetworkResourceLoadParameters loadParameters;
704 loadParameters.identifier = WebLoaderStrategy::generateLoadIdentifier();
705 loadParameters.webPageProxyID = webPage->webPageProxyIdentifier();
706 loadParameters.webPageID = webPage->identifier();
707 loadParameters.webFrameID = webFrame->frameID();
708 loadParameters.request = request;
709 loadParameters.sourceOrigin = &document->securityOrigin();
710 loadParameters.topOrigin = &document->topOrigin();
711 loadParameters.parentPID = presentingApplicationPID();
712#if HAVE(AUDIT_TOKEN)
713 loadParameters.networkProcessAuditToken = WebProcess::singleton().ensureNetworkProcessConnection().networkProcessAuditToken();
714#endif
715 loadParameters.storedCredentialsPolicy = options.credentials == FetchOptions::Credentials::Omit ? StoredCredentialsPolicy::DoNotUse : StoredCredentialsPolicy::Use;
716 loadParameters.options = options;
717 loadParameters.originalRequestHeaders = originalRequestHeaders;
718 loadParameters.shouldClearReferrerOnHTTPSToHTTPRedirect = shouldClearReferrerOnHTTPSToHTTPRedirect(&frame);
719 loadParameters.shouldRestrictHTTPResponseAccess = shouldPerformSecurityChecks();
720 if (policyCheck == ContentSecurityPolicyImposition::DoPolicyCheck && !document->shouldBypassMainWorldContentSecurityPolicy()) {
721 if (auto* contentSecurityPolicy = document->contentSecurityPolicy())
722 loadParameters.cspResponseHeaders = contentSecurityPolicy->responseHeaders();
723 }
724 addParametersShared(&frame, loadParameters);
725#if ENABLE(APP_BOUND_DOMAINS)
726 loadParameters.isNavigatingToAppBoundDomain = webFrame->isTopFrameNavigatingToAppBoundDomain();
727#endif
728
729#if ENABLE(CONTENT_EXTENSIONS)
730 loadParameters.mainDocumentURL = document->topDocument().url();
731 // FIXME: Instead of passing userContentControllerIdentifier, we should just pass webPageId to NetworkProcess.
732 loadParameters.userContentControllerIdentifier = webPage->userContentControllerIdentifier();
733#endif
734
735 if (completionHandler)
736 m_pingLoadCompletionHandlers.add(loadParameters.identifier, WTFMove(completionHandler));
737
738 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::LoadPing { loadParameters }, 0);
739}

748{

765void WebLoaderStrategy::preconnectTo(WebCore::ResourceRequest&& request, WebPage& webPage, WebFrame& webFrame, WebCore::StoredCredentialsPolicy storedCredentialsPolicy, PreconnectCompletionHandler&& completionHandler)
766{
767 if (auto* document = webPage.mainFrame()->document()) {
768 request.setFirstPartyForCookies(document->firstPartyForCookies());
769 if (auto* loader = document->loader())
770 request.setIsAppBound(loader->lastNavigationWasAppBound());
771 }
772
773 std::optional<uint64_t> preconnectionIdentifier;
774 if (completionHandler) {
775 preconnectionIdentifier = WebLoaderStrategy::generateLoadIdentifier();
776 auto addResult = m_preconnectCompletionHandlers.add(*preconnectionIdentifier, WTFMove(completionHandler));
777 ASSERT_UNUSED(addResult, addResult.isNewEntry);
778 }
779
780 NetworkResourceLoadParameters parameters;
781 parameters.request = WTFMove(request);
782 if (parameters.request.httpUserAgent().isEmpty()) {
783 // FIXME: we add user-agent to the preconnect request because otherwise the preconnect
784 // gets thrown away by CFNetwork when using an HTTPS proxy (<rdar://problem/59434166>).
785 String webPageUserAgent = webPage.userAgent(parameters.request.url());
786 if (!webPageUserAgent.isEmpty())
787 parameters.request.setHTTPUserAgent(webPageUserAgent);
788 }
789 parameters.webPageProxyID = webPage.webPageProxyIdentifier();
790 parameters.webPageID = webPage.identifier();
791 parameters.webFrameID = webFrame.frameID();
792 parameters.parentPID = presentingApplicationPID();
793#if HAVE(AUDIT_TOKEN)
794 parameters.networkProcessAuditToken = WebProcess::singleton().ensureNetworkProcessConnection().networkProcessAuditToken();
795#endif
796 parameters.storedCredentialsPolicy = storedCredentialsPolicy;
797 parameters.shouldPreconnectOnly = PreconnectOnly::Yes;
798 parameters.shouldRestrictHTTPResponseAccess = shouldPerformSecurityChecks();
799 // FIXME: Use the proper destination once all fetch options are passed.
800 parameters.options.destination = FetchOptions::Destination::EmptyString;
801#if ENABLE(APP_BOUND_DOMAINS)
802 parameters.isNavigatingToAppBoundDomain = webFrame.isTopFrameNavigatingToAppBoundDomain();
803#endif
804
805 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::PreconnectTo(preconnectionIdentifier, WTFMove(parameters)), 0);
806}

875bool WebLoaderStrategy::havePerformedSecurityChecks(const ResourceResponse& response) const
876{

883 case ResourceResponse::Source::MemoryCacheAfterValidation:
