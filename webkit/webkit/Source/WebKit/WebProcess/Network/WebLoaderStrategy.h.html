63 void servePendingRequests(WebCore::ResourceLoadPriority minimumPriority) final;

65 void suspendPendingRequests() final;

68 bool usePingLoad() const final;

71
72 void preconnectTo(WebCore::ResourceRequest&&, WebPage&, WebFrame&, WebCore::StoredCredentialsPolicy, PreconnectCompletionHandler&& = nullptr);

80
81 void networkProcessCrashed();
82
83 void addURLSchemeTaskProxy(WebURLSchemeTaskProxy&);
84 void removeURLSchemeTaskProxy(WebURLSchemeTaskProxy&);

136 bool m_isOnLine { true };
