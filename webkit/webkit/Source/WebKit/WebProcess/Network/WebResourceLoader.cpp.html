95void WebResourceLoader::willSendRequest(ResourceRequest&& proposedRequest, IPC::FormDataReference&& proposedRequestBody, ResourceResponse&& redirectResponse)
96{
97 Ref<WebResourceLoader> protectedThis(*this);
98
99 // Make the request whole again as we do not normally encode the request's body when sending it over IPC, for performance reasons.
100 proposedRequest.setHTTPBody(proposedRequestBody.takeData());
101
102 LOG(Network, "(WebProcess) WebResourceLoader::willSendRequest to '%s'", proposedRequest.url().string().latin1().data());
103 WEBRESOURCELOADER_RELEASE_LOG("willSendRequest:");
104
105 if (m_coreLoader->documentLoader()->applicationCacheHost().maybeLoadFallbackForRedirect(m_coreLoader.get(), proposedRequest, redirectResponse)) {
106 WEBRESOURCELOADER_RELEASE_LOG("willSendRequest: exiting early because maybeLoadFallbackForRedirect returned false");
107 return;
108 }
109
110 m_coreLoader->willSendRequest(WTFMove(proposedRequest), redirectResponse, [this, protectedThis = WTFMove(protectedThis)](ResourceRequest&& request) {
111 if (!m_coreLoader || !m_coreLoader->identifier()) {
112 WEBRESOURCELOADER_RELEASE_LOG("willSendRequest: exiting early because no coreloader or identifier");
113 return;
114 }
115
116 WEBRESOURCELOADER_RELEASE_LOG("willSendRequest: returning ContinueWillSendRequest");
117 send(Messages::NetworkResourceLoader::ContinueWillSendRequest(request, m_coreLoader->isAllowedToAskUserForCredentials()));
118 });
119}

126void WebResourceLoader::didReceiveResponse(const ResourceResponse& response, bool needsContinueDidReceiveResponseMessage)
127{
128 LOG(Network, "(WebProcess) WebResourceLoader::didReceiveResponse for '%s'. Status %d.", m_coreLoader->url().string().latin1().data(), response.httpStatusCode());
129 WEBRESOURCELOADER_RELEASE_LOG("didReceiveResponse: (httpStatusCode=%d)", response.httpStatusCode());
130
131 Ref<WebResourceLoader> protectedThis(*this);
132
133 if (m_coreLoader->documentLoader()->applicationCacheHost().maybeLoadFallbackForResponse(m_coreLoader.get(), response)) {
134 WEBRESOURCELOADER_RELEASE_LOG("didReceiveResponse: not continuing load because the content is already cached");
135 return;
136 }
137
138 CompletionHandler<void()> policyDecisionCompletionHandler;
139 if (needsContinueDidReceiveResponseMessage) {
140#if ASSERT_ENABLED
141 m_isProcessingNetworkResponse = true;
142#endif
143 policyDecisionCompletionHandler = [this, protectedThis = WTFMove(protectedThis)] {
144#if ASSERT_ENABLED
145 m_isProcessingNetworkResponse = false;
146#endif
147 // If m_coreLoader becomes null as a result of the didReceiveResponse callback, we can't use the send function().
148 if (m_coreLoader && m_coreLoader->identifier())
149 send(Messages::NetworkResourceLoader::ContinueDidReceiveResponse());
150 else
151 WEBRESOURCELOADER_RELEASE_LOG("didReceiveResponse: not continuing load because no coreLoader or no ID");
152 };
153 }
154
155 if (InspectorInstrumentationWebKit::shouldInterceptResponse(m_coreLoader->frame(), response)) {
156 unsigned long interceptedRequestIdentifier = m_coreLoader->identifier();
157 m_interceptController.beginInterceptingResponse(interceptedRequestIdentifier);
158 InspectorInstrumentationWebKit::interceptResponse(m_coreLoader->frame(), response, interceptedRequestIdentifier, [this, protectedThis = makeRef(*this), interceptedRequestIdentifier, policyDecisionCompletionHandler = WTFMove(policyDecisionCompletionHandler)](const ResourceResponse& inspectorResponse, RefPtr<SharedBuffer> overrideData) mutable {
159 if (!m_coreLoader || !m_coreLoader->identifier()) {
160 WEBRESOURCELOADER_RELEASE_LOG("didReceiveResponse: not continuing intercept load because no coreLoader or no ID");
161 m_interceptController.continueResponse(interceptedRequestIdentifier);
162 return;
163 }
164
165 m_coreLoader->didReceiveResponse(inspectorResponse, [this, protectedThis = WTFMove(protectedThis), interceptedRequestIdentifier, policyDecisionCompletionHandler = WTFMove(policyDecisionCompletionHandler), overrideData = WTFMove(overrideData)]() mutable {
166 if (policyDecisionCompletionHandler)
167 policyDecisionCompletionHandler();
168
169 if (!m_coreLoader || !m_coreLoader->identifier()) {
170 m_interceptController.continueResponse(interceptedRequestIdentifier);
171 return;
172 }
173
174 RefPtr<WebCore::ResourceLoader> protectedCoreLoader = m_coreLoader;
175 if (!overrideData)
176 m_interceptController.continueResponse(interceptedRequestIdentifier);
177 else {
178 m_interceptController.interceptedResponse(interceptedRequestIdentifier);
179 if (unsigned bufferSize = overrideData->size())
180 protectedCoreLoader->didReceiveBuffer(overrideData.releaseNonNull(), bufferSize, DataPayloadWholeResource);
181 WebCore::NetworkLoadMetrics emptyMetrics;
182 protectedCoreLoader->didFinishLoading(emptyMetrics);
183 }
184 });
185 });
186 return;
187 }
188
189 m_coreLoader->didReceiveResponse(response, WTFMove(policyDecisionCompletionHandler));
190}

192void WebResourceLoader::didReceiveData(const IPC::DataReference& data, int64_t encodedDataLength)
193{
194 LOG(Network, "(WebProcess) WebResourceLoader::didReceiveData of size %lu for '%s'", data.size(), m_coreLoader->url().string().latin1().data());
195 ASSERT_WITH_MESSAGE(!m_isProcessingNetworkResponse, "Network process should not send data until we've validated the response");
196
197 if (UNLIKELY(m_interceptController.isIntercepting(m_coreLoader->identifier()))) {
198 auto buffer = SharedBuffer::create(data.data(), data.size());
199 m_interceptController.defer(m_coreLoader->identifier(), [this, protectedThis = makeRef(*this), buffer = WTFMove(buffer), encodedDataLength]() mutable {
200 if (m_coreLoader)
201 didReceiveData({ buffer->data(), buffer->size() }, encodedDataLength);
202 });
203 return;
204 }
205
206 if (!m_numBytesReceived)
207 WEBRESOURCELOADER_RELEASE_LOG("didReceiveData: Started receiving data");
208 m_numBytesReceived += data.size();
209
210 m_coreLoader->didReceiveData(data.data(), data.size(), encodedDataLength, DataPayloadBytes);
211}

213void WebResourceLoader::didFinishResourceLoad(const NetworkLoadMetrics& networkLoadMetrics)
214{
215 LOG(Network, "(WebProcess) WebResourceLoader::didFinishResourceLoad for '%s'", m_coreLoader->url().string().latin1().data());
216 WEBRESOURCELOADER_RELEASE_LOG("didFinishResourceLoad: (length=%zd)", m_numBytesReceived);
217
218 if (UNLIKELY(m_interceptController.isIntercepting(m_coreLoader->identifier()))) {
219 m_interceptController.defer(m_coreLoader->identifier(), [this, protectedThis = makeRef(*this), networkLoadMetrics]() mutable {
220 if (m_coreLoader)
221 didFinishResourceLoad(networkLoadMetrics);
222 });
223 return;
224 }
225
226 ASSERT_WITH_MESSAGE(!m_isProcessingNetworkResponse, "Load should not be able to finish before we've validated the response");
227 m_coreLoader->didFinishLoading(networkLoadMetrics);
228}

230void WebResourceLoader::didFailServiceWorkerLoad(const ResourceError& error)
231{
232 if (auto* document = m_coreLoader->frame() ? m_coreLoader->frame()->document() : nullptr) {
233 if (m_coreLoader->options().destination != FetchOptions::Destination::EmptyString || error.isGeneral())
234 document->addConsoleMessage(MessageSource::JS, MessageLevel::Error, error.localizedDescription());
235 if (m_coreLoader->options().destination != FetchOptions::Destination::EmptyString)
236 document->addConsoleMessage(MessageSource::JS, MessageLevel::Error, makeString("Cannot load ", error.failingURL().string(), "."));
237 }
238
239 didFailResourceLoad(error);
240}

256void WebResourceLoader::didFailResourceLoad(const ResourceError& error)
257{
258 LOG(Network, "(WebProcess) WebResourceLoader::didFailResourceLoad for '%s'", m_coreLoader->url().string().latin1().data());
259 WEBRESOURCELOADER_RELEASE_LOG("didFailResourceLoad:");
260
261 if (UNLIKELY(m_interceptController.isIntercepting(m_coreLoader->identifier()))) {
262 m_interceptController.defer(m_coreLoader->identifier(), [this, protectedThis = makeRef(*this), error]() mutable {
263 if (m_coreLoader)
264 didFailResourceLoad(error);
265 });
266 return;
267 }
268
269 ASSERT_WITH_MESSAGE(!m_isProcessingNetworkResponse, "Load should not be able to finish before we've validated the response");
270
271 if (m_coreLoader->documentLoader()->applicationCacheHost().maybeLoadFallbackForError(m_coreLoader.get(), error))
272 return;
273 m_coreLoader->didFail(error);
274}

292#if ENABLE(SHAREABLE_RESOURCE)
293void WebResourceLoader::didReceiveResource(const ShareableResource::Handle& handle)
294{
295 LOG(Network, "(WebProcess) WebResourceLoader::didReceiveResource for '%s'", m_coreLoader->url().string().latin1().data());
296 WEBRESOURCELOADER_RELEASE_LOG("didReceiveResource:");
297
298 RefPtr<SharedBuffer> buffer = handle.tryWrapInSharedBuffer();
299
300 if (!buffer) {
301 LOG_ERROR("Unable to create buffer from ShareableResource sent from the network process.");
302 WEBRESOURCELOADER_RELEASE_LOG("didReceiveResource: Unable to create SharedBuffer");
303 if (auto* frame = m_coreLoader->frame()) {
304 if (auto* page = frame->page())
305 page->diagnosticLoggingClient().logDiagnosticMessage(WebCore::DiagnosticLoggingKeys::internalErrorKey(), WebCore::DiagnosticLoggingKeys::createSharedBufferFailedKey(), WebCore::ShouldSample::No);
306 }
307 m_coreLoader->didFail(internalError(m_coreLoader->request().url()));
308 return;
309 }
310
311 Ref<WebResourceLoader> protect(*this);
312
313 // Only send data to the didReceiveData callback if it exists.
314 if (unsigned bufferSize = buffer->size())
315 m_coreLoader->didReceiveBuffer(buffer.releaseNonNull(), bufferSize, DataPayloadWholeResource);
316
317 if (!m_coreLoader)
318 return;
319
320 NetworkLoadMetrics emptyMetrics;
321 m_coreLoader->didFinishLoading(emptyMetrics);
322}
323#endif
