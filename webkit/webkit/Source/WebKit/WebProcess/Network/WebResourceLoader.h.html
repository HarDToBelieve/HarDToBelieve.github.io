53typedef uint64_t ResourceLoadIdentifier;
54
55class WebResourceLoader : public RefCounted<WebResourceLoader>, public IPC::MessageSender {
56public:
57 struct TrackingParameters {
58 WebPageProxyIdentifier webPageProxyID;
59 WebCore::PageIdentifier pageID;
60 WebCore::FrameIdentifier frameID;
61 ResourceLoadIdentifier resourceID { 0 };
62 };
63
64 static Ref<WebResourceLoader> create(Ref<WebCore::ResourceLoader>&&, const TrackingParameters&);
65
66 ~WebResourceLoader();
67
68 void didReceiveWebResourceLoaderMessage(IPC::Connection&, IPC::Decoder&);
69
70 WebCore::ResourceLoader* resourceLoader() const { return m_coreLoader.get(); }
71
72 void detachFromCoreLoader();
73
74private:
75 WebResourceLoader(Ref<WebCore::ResourceLoader>&&, const TrackingParameters&);
76
77 // IPC::MessageSender
78 IPC::Connection* messageSenderConnection() const override;
79 uint64_t messageSenderDestinationID() const override;
80
81 void willSendRequest(WebCore::ResourceRequest&&, IPC::FormDataReference&& requestBody, WebCore::ResourceResponse&&);
82 void didSendData(uint64_t bytesSent, uint64_t totalBytesToBeSent);
83 void didReceiveResponse(const WebCore::ResourceResponse&, bool needsContinueDidReceiveResponseMessage);
84 void didReceiveData(const IPC::DataReference&, int64_t encodedDataLength);
85 void didFinishResourceLoad(const WebCore::NetworkLoadMetrics&);
86 void didFailResourceLoad(const WebCore::ResourceError&);
87 void didFailServiceWorkerLoad(const WebCore::ResourceError&);
88 void serviceWorkerDidNotHandle();
89 void didBlockAuthenticationChallenge();
90
91 void stopLoadingAfterXFrameOptionsOrContentSecurityPolicyDenied(const WebCore::ResourceResponse&);
92
93#if ENABLE(SHAREABLE_RESOURCE)
94 void didReceiveResource(const ShareableResource::Handle&);
95#endif
96
97 RefPtr<WebCore::ResourceLoader> m_coreLoader;
98 TrackingParameters m_trackingParameters;
99 WebResourceInterceptController m_interceptController;
100 size_t m_numBytesReceived { 0 };
101
102#if ASSERT_ENABLED
103 bool m_isProcessingNetworkResponse { false };
104#endif
105};
