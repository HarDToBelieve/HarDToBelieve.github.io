56NetworkSendQueue WebSocketChannel::createMessageQueue(Document& document, WebSocketChannel& channel)
57{
58 return { document, [&channel](auto& utf8String) {
59 channel.notifySendFrame(WebSocketFrame::OpCode::OpCodeText, utf8String.dataAsUInt8Ptr(), utf8String.length());
60 channel.sendMessage(Messages::NetworkSocketChannel::SendString { IPC::DataReference { utf8String.dataAsUInt8Ptr(), utf8String.length() } }, utf8String.length());
61 }, [&channel](auto& span) {
62 channel.notifySendFrame(WebSocketFrame::OpCode::OpCodeBinary, span.data(), span.size());
63 channel.sendMessage(Messages::NetworkSocketChannel::SendData { IPC::DataReference { span.data(), span.size() } }, span.size());
64 }, [&channel](ExceptionCode exceptionCode) {
65 auto code = static_cast<int>(exceptionCode);
66 channel.fail(makeString("Failed to load Blob: exception code = ", code));
67 return NetworkSendQueue::Continue::No;
68 } };
69}

106WebSocketChannel::ConnectStatus WebSocketChannel::connect(const URL& url, const String& protocol)
107{
108 if (!m_document)
109 return ConnectStatus::KO;
110
111 auto request = webSocketConnectRequest(*m_document, url);
112 if (!request)
113 return ConnectStatus::KO;
114
115 if (request->url() != url && m_client)
116 m_client->didUpgradeURL();
117
118 m_inspector.didCreateWebSocket(m_document.get(), url);
119 m_url = request->url();
120 MessageSender::send(Messages::NetworkConnectionToWebProcess::CreateSocketChannel { *request, protocol, m_identifier, m_webPageProxyID });
121 return ConnectStatus::OK;
122}

124bool WebSocketChannel::increaseBufferedAmount(size_t byteLength)
125{
126 if (!byteLength)
127 return true;
128
129 CheckedSize checkedNewBufferedAmount = m_bufferedAmount;
130 checkedNewBufferedAmount += byteLength;
131 if (UNLIKELY(checkedNewBufferedAmount.hasOverflowed())) {
132 fail("Failed to send WebSocket frame: buffer has no more space");
133 return false;
134 }
135
136 m_bufferedAmount = checkedNewBufferedAmount;
137 if (m_client)
138 m_client->didUpdateBufferedAmount(m_bufferedAmount);
139 return true;
140}
141
142void WebSocketChannel::decreaseBufferedAmount(size_t byteLength)
143{
144 if (!byteLength)
145 return;
146
147 ASSERT(m_bufferedAmount >= byteLength);
148 m_bufferedAmount -= byteLength;
149 if (m_client)
150 m_client->didUpdateBufferedAmount(m_bufferedAmount);
151}

153template<typename T> void WebSocketChannel::sendMessage(T&& message, size_t byteLength)
154{
155 CompletionHandler<void()> completionHandler = [this, protectedThis = makeRef(*this), byteLength] {
156 decreaseBufferedAmount(byteLength);
157 };
158 sendWithAsyncReply(WTFMove(message), WTFMove(completionHandler));
159}
160
161WebSocketChannel::SendResult WebSocketChannel::send(const String& message)
162{
163 auto utf8 = message.utf8(StrictConversionReplacingUnpairedSurrogatesWithFFFD);
164 if (!increaseBufferedAmount(utf8.length()))
165 return SendFail;
166
167 m_messageQueue.enqueue(WTFMove(utf8));
168 return SendSuccess;
169}
170
171WebSocketChannel::SendResult WebSocketChannel::send(const JSC::ArrayBuffer& binaryData, unsigned byteOffset, unsigned byteLength)
172{
173 if (!increaseBufferedAmount(byteLength))
174 return SendFail;
175
176 m_messageQueue.enqueue(binaryData, byteOffset, byteLength);
177 return SendSuccess;
178}
179
180WebSocketChannel::SendResult WebSocketChannel::send(Blob& blob)
181{
182 auto byteLength = blob.size();
183 if (!blob.size())
184 return send(JSC::ArrayBuffer::create(byteLength, 1), 0, 0);
185
186 if (!increaseBufferedAmount(byteLength))
187 return SendFail;
188
189 m_messageQueue.enqueue(blob);
190 return SendSuccess;
191}

198void WebSocketChannel::close(int code, const String& reason)
199{
200 // An attempt to send closing handshake may fail, which will get the channel closed and dereferenced.
201 auto protectedThis = makeRef(*this);
202
203 m_isClosing = true;
204 if (m_client)
205 m_client->didStartClosingHandshake();
206
207 ASSERT(code >= 0 || code == WebCore::WebSocketChannel::CloseEventCodeNotSpecified);
208
209 WebSocketFrame closingFrame(WebSocketFrame::OpCodeClose, true, false, true);
210 m_inspector.didSendWebSocketFrame(m_document.get(), closingFrame);
211
212 MessageSender::send(Messages::NetworkSocketChannel::Close { code, reason });
213}
214
215void WebSocketChannel::fail(const String& reason)
216{
217 // The client can close the channel, potentially removing the last reference.
218 auto protectedThis = makeRef(*this);
219
220 logErrorMessage(reason);
221 if (m_client)
222 m_client->didReceiveMessageError();
223
224 if (m_isClosing)
225 return;
226
227 MessageSender::send(Messages::NetworkSocketChannel::Close { 0, reason });
228 didClose(WebCore::WebSocketChannel::CloseEventCodeAbnormalClosure, { });
229}

231void WebSocketChannel::disconnect()
232{
233 m_client = nullptr;
234 m_document = nullptr;
235 m_pendingTasks.clear();
236 m_messageQueue.clear();
237
238
239 m_inspector.didCloseWebSocket(m_document.get());
240
241 MessageSender::send(Messages::NetworkSocketChannel::Close { 0, { } });
242}

244void WebSocketChannel::didConnect(String&& subprotocol, String&& extensions)
245{
246 if (m_isClosing)
247 return;
248
249 if (!m_client)
250 return;
251
252 if (m_isSuspended) {
253 enqueueTask([this, subprotocol = WTFMove(subprotocol), extensions = WTFMove(extensions)] () mutable {
254 didConnect(WTFMove(subprotocol), WTFMove(extensions));
255 });
256 return;
257 }
258
259 m_subprotocol = WTFMove(subprotocol);
260 m_extensions = WTFMove(extensions);
261 m_client->didConnect();
262}

264static inline WebSocketFrame createWebSocketFrameForWebInspector(const uint8_t* data, size_t length, WebSocketFrame::OpCode opCode)
265{
266 // This is an approximation since frames can be merged on a single message.
267 WebSocketFrame frame;
268 frame.opCode = opCode;
269 frame.masked = false;
270 frame.payload = data;
271 frame.payloadLength = length;
272
273 // WebInspector does not use them.
274 frame.final = false;
275 frame.compress = false;
276 frame.reserved2 = false;
277 frame.reserved3 = false;
278
279 return frame;
280}

282void WebSocketChannel::didReceiveText(String&& message)
283{
284 if (m_isClosing)
285 return;
286
287 if (!m_client)
288 return;
289
290 if (m_isSuspended) {
291 enqueueTask([this, message = WTFMove(message)] () mutable {
292 didReceiveText(WTFMove(message));
293 });
294 return;
295 }
296
297 auto utf8Message = message.utf8();
298 m_inspector.didReceiveWebSocketFrame(m_document.get(), createWebSocketFrameForWebInspector(utf8Message.dataAsUInt8Ptr(), utf8Message.length(), WebSocketFrame::OpCode::OpCodeText));
299
300 m_client->didReceiveMessage(message);
301}

303void WebSocketChannel::didReceiveBinaryData(IPC::DataReference&& data)
304{
305 if (m_isClosing)
306 return;
307
308 if (!m_client)
309 return;
310
311 if (m_isSuspended) {
312 enqueueTask([this, data = data.vector()] () mutable {
313 if (!m_isClosing && m_client)
314 m_client->didReceiveBinaryData(WTFMove(data));
315 });
316 return;
317 }
318
319 m_inspector.didReceiveWebSocketFrame(m_document.get(), createWebSocketFrameForWebInspector(data.data(), data.size(), WebSocketFrame::OpCode::OpCodeBinary));
320
321 m_client->didReceiveBinaryData(data.vector());
322}

324void WebSocketChannel::didClose(unsigned short code, String&& reason)
325{
326 if (!m_client)
327 return;
328
329 if (m_isSuspended) {
330 enqueueTask([this, code, reason = WTFMove(reason)] () mutable {
331 didClose(code, WTFMove(reason));
332 });
333 return;
334 }
335
336 WebSocketFrame closingFrame(WebSocketFrame::OpCodeClose, true, false, false);
337 m_inspector.didReceiveWebSocketFrame(m_document.get(), closingFrame);
338 m_inspector.didCloseWebSocket(m_document.get());
339
340 // An attempt to send closing handshake may fail, which will get the channel closed and dereferenced.
341 auto protectedThis = makeRef(*this);
342
343 bool receivedClosingHandshake = code != WebCore::WebSocketChannel::CloseEventCodeAbnormalClosure;
344 if (receivedClosingHandshake)
345 m_client->didStartClosingHandshake();
346
347 m_client->didClose(m_bufferedAmount, (m_isClosing || receivedClosingHandshake) ? WebCore::WebSocketChannelClient::ClosingHandshakeComplete : WebCore::WebSocketChannelClient::ClosingHandshakeIncomplete, code, reason);
348}

350void WebSocketChannel::logErrorMessage(const String& errorMessage)
351{
352 if (!m_document)
353 return;
354
355 String consoleMessage;
356 if (!m_url.isNull())
357 consoleMessage = makeString("WebSocket connection to '", m_url.string(), "' failed: ", errorMessage);
358 else
359 consoleMessage = makeString("WebSocket connection failed: ", errorMessage);
360 m_document->addConsoleMessage(MessageSource::Network, MessageLevel::Error, consoleMessage);
361}

363void WebSocketChannel::didReceiveMessageError(String&& errorMessage)
364{
365 if (!m_client)
366 return;
367
368 if (m_isSuspended) {
369 enqueueTask([this, errorMessage = WTFMove(errorMessage)] () mutable {
370 didReceiveMessageError(WTFMove(errorMessage));
371 });
372 return;
373 }
374
375 logErrorMessage(errorMessage);
376 m_client->didReceiveMessageError();
377}

389void WebSocketChannel::resume()
390{
391 auto protectedThis = makeRef(*this);
392 m_isSuspended = false;
393 while (!m_isSuspended && !m_pendingTasks.isEmpty())
394 m_pendingTasks.takeFirst()();
395}

402void WebSocketChannel::didSendHandshakeRequest(ResourceRequest&& request)
403{
404 if (m_isSuspended) {
405 enqueueTask([this, request = WTFMove(request)]() mutable {
406 didSendHandshakeRequest(WTFMove(request));
407 });
408 return;
409 }
410
411 m_inspector.willSendWebSocketHandshakeRequest(m_document.get(), request);
412 m_handshakeRequest = WTFMove(request);
413}
