29#include <WebCore/WebSocketIdentifier.h>

31namespace WebKit {
32
33void WebSocketChannelManager::addChannel(WebSocketChannel& channel)
34{
35 ASSERT(!m_channels.contains(channel.identifier()));
36 m_channels.add(channel.identifier(), makeWeakPtr(channel));
37}
38
39void WebSocketChannelManager::networkProcessCrashed()
40{
41 auto channels = WTFMove(m_channels);
42 for (auto& channel : channels.values())
43 channel->networkProcessCrashed();
44}
45
46void WebSocketChannelManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
47{
48 auto iterator = m_channels.find(makeObjectIdentifier<WebCore::WebSocketIdentifierType>(decoder.destinationID()));
49 if (iterator != m_channels.end())
50 iterator->value->didReceiveMessage(connection, decoder);
51}
