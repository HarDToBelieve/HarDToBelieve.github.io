41Ref<SocketStreamHandle> WebSocketProvider::createSocketStreamHandle(const URL& url, SocketStreamHandleClient& client, WebSocketIdentifier identifier, PAL::SessionID sessionID, const String& credentialPartition, const StorageSessionProvider*)
42{
43 ASSERT_UNUSED(sessionID, sessionID == WebProcess::singleton().sessionID());
44 return WebSocketStream::create(url, client, identifier, credentialPartition);
45}
46
47RefPtr<ThreadableWebSocketChannel> WebSocketProvider::createWebSocketChannel(Document& document, WebSocketChannelClient& client)
48{
49 return WebSocketChannel::create(m_webPageProxyID, document, client);
50}
