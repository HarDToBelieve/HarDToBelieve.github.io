27
28#if USE(LIBWEBRTC) && PLATFORM(COCOA)
29#include <WebCore/LibWebRTCProviderCocoa.h>
30#elif USE(LIBWEBRTC) && USE(GSTREAMER)
31#include <WebCore/LibWebRTCProviderGStreamer.h>
32#else
33#include <WebCore/LibWebRTCProvider.h>
34#endif
35
36namespace WebKit {
37
38class WebPage;
39
40#if USE(LIBWEBRTC)
41
42#if PLATFORM(COCOA)
43using LibWebRTCProviderBase = WebCore::LibWebRTCProviderCocoa;
44#elif USE(GSTREAMER)
45using LibWebRTCProviderBase = WebCore::LibWebRTCProviderGStreamer;
46#else
47using LibWebRTCProviderBase = WebCore::LibWebRTCProvider;
48#endif
49
50class LibWebRTCProvider final : public LibWebRTCProviderBase {
51public:
52 explicit LibWebRTCProvider(WebPage&);
53
54private:
55 std::unique_ptr<SuspendableSocketFactory> createSocketFactory(String&& /* userAgent */, bool /* isFirstParty */, WebCore::RegistrableDomain&&) final;
56
57 rtc::scoped_refptr<webrtc::PeerConnectionInterface> createPeerConnection(webrtc::PeerConnectionObserver&, rtc::PacketSocketFactory*, webrtc::PeerConnectionInterface::RTCConfiguration&&) final;
58
59 void unregisterMDNSNames(WebCore::DocumentIdentifier) final;
60 void registerMDNSName(WebCore::DocumentIdentifier, const String& ipAddress, CompletionHandler<void(MDNSNameOrError&&)>&&) final;
61 void disableNonLocalhostConnections() final;
62 void startedNetworkThread() final;
63 RefPtr<WebCore::RTCDataChannelRemoteHandlerConnection> createRTCDataChannelRemoteHandlerConnection() final;
64 void setLoggingLevel(WTFLogLevel) final;
65
66#if ENABLE(GPU_PROCESS) && PLATFORM(COCOA) && !PLATFORM(MACCATALYST)
67 WebPage& m_webPage;
68 bool m_didInitializeCallback { false };
69#endif
70};
71
72inline LibWebRTCProvider::LibWebRTCProvider(WebPage& webPage)
73#if ENABLE(GPU_PROCESS) && PLATFORM(COCOA) && !PLATFORM(MACCATALYST)
74 : m_webPage(webPage)
75#endif
76{
77 m_useNetworkThreadWithSocketServer = false;
78}
79
80inline UniqueRef<LibWebRTCProvider> createLibWebRTCProvider(WebPage& page)
81{
82 return makeUniqueRef<LibWebRTCProvider>(page);
83}
84#else
85using LibWebRTCProvider = WebCore::LibWebRTCProvider;
86
87inline UniqueRef<LibWebRTCProvider> createLibWebRTCProvider(WebPage&)
88{
89 return makeUniqueRef<LibWebRTCProvider>();
90}
91#endif // USE(LIBWEBRTC)
92
93} // namespace WebKit
