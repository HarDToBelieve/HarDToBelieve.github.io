41namespace WebKit {
42
43class LibWebRTCSocketFactory;
44
45class LibWebRTCSocket final : public rtc::AsyncPacketSocket {
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 enum class Type { UDP, ServerTCP, ClientTCP, ServerConnectionTCP };
49
50 LibWebRTCSocket(LibWebRTCSocketFactory&, const void* socketGroup, Type, const rtc::SocketAddress& localAddress, const rtc::SocketAddress& remoteAddress);
51 ~LibWebRTCSocket();
52
53 const void* socketGroup() const { return m_socketGroup; }
54 WebCore::LibWebRTCSocketIdentifier identifier() const { return m_identifier; }
55 const rtc::SocketAddress& localAddress() const { return m_localAddress; }
56 const rtc::SocketAddress& remoteAddress() const { return m_remoteAddress; }
57
58 void setError(int error) { m_error = error; }
59 void setState(State state) { m_state = state; }
60
61 void suspend();
62 void resume();
63
64private:
65 bool willSend(size_t);
66
67 friend class LibWebRTCNetwork;
68 void signalReadPacket(const uint8_t*, size_t, rtc::SocketAddress&&, int64_t);
69 void signalSentPacket(int, int64_t);
70 void signalAddressReady(const rtc::SocketAddress&);
71 void signalConnect();
72 void signalClose(int);
73 void signalNewConnection(rtc::AsyncPacketSocket*);
74
75 // AsyncPacketSocket API
76 int GetError() const final { return m_error; }
77 void SetError(int error) final { setError(error); }
78 rtc::SocketAddress GetLocalAddress() const final;
79 rtc::SocketAddress GetRemoteAddress() const final;
80 int Send(const void *pv, size_t cb, const rtc::PacketOptions& options) final { return SendTo(pv, cb, m_remoteAddress, options); }
81 int SendTo(const void *, size_t, const rtc::SocketAddress&, const rtc::PacketOptions&) final;
82 int Close() final;
83 State GetState() const final { return m_state; }
84 int GetOption(rtc::Socket::Option, int*) final;
85 int SetOption(rtc::Socket::Option, int) final;
86
87 LibWebRTCSocketFactory& m_factory;
88 WebCore::LibWebRTCSocketIdentifier m_identifier;
89 Type m_type;
90 rtc::SocketAddress m_localAddress;
91 rtc::SocketAddress m_remoteAddress;
92
93 int m_error { 0 };
94 State m_state { STATE_BINDING };
95
96 static const unsigned MAX_SOCKET_OPTION { rtc::Socket::OPT_RTP_SENDTIME_EXTN_ID + 1 };
97 std::optional<int> m_options[MAX_SOCKET_OPTION];
98
99 Deque<size_t> m_beingSentPacketSizes;
100 size_t m_availableSendingBytes { 65536 };
101 bool m_shouldSignalReadyToSend { false };
102 bool m_isSuspended { false };
103 const void* m_socketGroup { nullptr };
104};

106} // namespace WebKit
