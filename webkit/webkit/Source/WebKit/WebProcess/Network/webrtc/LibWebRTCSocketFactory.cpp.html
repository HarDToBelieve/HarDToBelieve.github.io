42static inline rtc::SocketAddress prepareSocketAddress(const rtc::SocketAddress& address, bool disableNonLocalhostConnections)
43{
44 auto result = RTCNetwork::isolatedCopy(address);
45 if (disableNonLocalhostConnections)
46 result.SetIP("127.0.0.1");
47 return result;
48}

50void LibWebRTCSocketFactory::setConnection(RefPtr<IPC::Connection>&& connection)
51{
52 ASSERT(!WTF::isMainRunLoop());
53 m_connection = WTFMove(connection);
54 if (!m_connection)
55 return;
56
57 m_connection->send(Messages::NetworkRTCProvider::SetPlatformTCPSocketsEnabled(RuntimeEnabledFeatures::sharedFeatures().webRTCPlatformTCPSocketsEnabled()), 0);
58 m_connection->send(Messages::NetworkRTCProvider::SetPlatformUDPSocketsEnabled(RuntimeEnabledFeatures::sharedFeatures().webRTCPlatformUDPSocketsEnabled()), 0);
59
60 while (!m_pendingMessageTasks.isEmpty())
61 m_pendingMessageTasks.takeFirst()(*m_connection);
62}

64IPC::Connection* LibWebRTCSocketFactory::connection()
65{
66 ASSERT(!WTF::isMainRunLoop());
67 return m_connection.get();
68}

70rtc::AsyncPacketSocket* LibWebRTCSocketFactory::createServerTcpSocket(const void* socketGroup, const rtc::SocketAddress& address, uint16_t minPort, uint16_t maxPort, int options)
71{
72 ASSERT(!WTF::isMainRunLoop());
73 auto socket = makeUnique<LibWebRTCSocket>(*this, socketGroup, LibWebRTCSocket::Type::ServerTCP, address, rtc::SocketAddress());
74
75 if (m_connection)
76 m_connection->send(Messages::NetworkRTCProvider::CreateServerTCPSocket(socket->identifier(), RTCNetwork::SocketAddress(address), minPort, maxPort, options), 0);
77 else {
78 callOnMainRunLoop([] {
79 WebProcess::singleton().ensureNetworkProcessConnection();
80 });
81 m_pendingMessageTasks.append([identifier = socket->identifier(), address = RTCNetwork::SocketAddress(address), minPort, maxPort, options](auto& connection) {
82 connection.send(Messages::NetworkRTCProvider::CreateServerTCPSocket(identifier, address, minPort, maxPort, options), 0);
83 });
84 }
85 return socket.release();
86}

107rtc::AsyncPacketSocket* LibWebRTCSocketFactory::createClientTcpSocket(const void* socketGroup, const rtc::SocketAddress& localAddress, const rtc::SocketAddress& remoteAddress, String&& userAgent, const rtc::PacketSocketTcpOptions& options)
108{
109 ASSERT(!WTF::isMainRunLoop());
110
111 auto socket = makeUnique<LibWebRTCSocket>(*this, socketGroup, LibWebRTCSocket::Type::ClientTCP, localAddress, remoteAddress);
112 socket->setState(LibWebRTCSocket::STATE_CONNECTING);
113
114 // FIXME: We only transfer options.opts but should also handle other members.
115 if (m_connection)
116 m_connection->send(Messages::NetworkRTCProvider::CreateClientTCPSocket(socket->identifier(), RTCNetwork::SocketAddress(prepareSocketAddress(localAddress, m_disableNonLocalhostConnections)), RTCNetwork::SocketAddress(prepareSocketAddress(remoteAddress, m_disableNonLocalhostConnections)), userAgent, options.opts), 0);
117 else {
118 callOnMainRunLoop([] {
119 WebProcess::singleton().ensureNetworkProcessConnection();
120 });
121 m_pendingMessageTasks.append([identifier = socket->identifier(), localAddress = RTCNetwork::SocketAddress(prepareSocketAddress(localAddress, m_disableNonLocalhostConnections)), remoteAddress = RTCNetwork::SocketAddress(prepareSocketAddress(remoteAddress, m_disableNonLocalhostConnections)), userAgent, opts = options.opts](auto& connection) {
122 connection.send(Messages::NetworkRTCProvider::CreateClientTCPSocket(identifier, localAddress, remoteAddress, userAgent, opts), 0);
123 });
124 }
125
126 return socket.release();
127}

129rtc::AsyncPacketSocket* LibWebRTCSocketFactory::createNewConnectionSocket(LibWebRTCSocket& serverSocket, LibWebRTCSocketIdentifier newConnectionSocketIdentifier, const rtc::SocketAddress& remoteAddress)
130{
131 ASSERT(!WTF::isMainRunLoop());
132 if (!m_connection) {
133 // No need to enqueue a message in this case since it means the network process handling the incoming socket is gone.
134 RELEASE_LOG(WebRTC, "No connection to create incoming TCP socket");
135 return nullptr;
136 }
137
138 auto socket = makeUnique<LibWebRTCSocket>(*this, serverSocket.socketGroup(), LibWebRTCSocket::Type::ServerConnectionTCP, serverSocket.localAddress(), remoteAddress);
139 socket->setState(LibWebRTCSocket::STATE_CONNECTED);
140
141 m_connection->send(Messages::NetworkRTCProvider::WrapNewTCPConnection(socket->identifier(), newConnectionSocketIdentifier), 0);
142
143 return socket.release();
144}

146void LibWebRTCSocketFactory::addSocket(LibWebRTCSocket& socket)
147{
148 ASSERT(!WTF::isMainRunLoop());
149 ASSERT(!m_sockets.contains(socket.identifier()));
150 m_sockets.add(socket.identifier(), &socket);
151}

153void LibWebRTCSocketFactory::removeSocket(LibWebRTCSocket& socket)
154{
155 ASSERT(!WTF::isMainRunLoop());
156 ASSERT(m_sockets.contains(socket.identifier()));
157 m_sockets.remove(socket.identifier());
158}

160void LibWebRTCSocketFactory::forSocketInGroup(const void* socketGroup, const Function<void(LibWebRTCSocket&)>& callback)
161{
162 ASSERT(!WTF::isMainRunLoop());
163 for (auto* socket : m_sockets.values()) {
164 if (socket->socketGroup() == socketGroup)
165 callback(*socket);
166 }
167}

169rtc::AsyncResolverInterface* LibWebRTCSocketFactory::createAsyncResolver()
170{
171 auto resolver = makeUnique<LibWebRTCResolver>();
172 auto* resolverPointer = resolver.get();
173 m_resolvers.set(resolverPointer->identifier(), WTFMove(resolver));
174 return resolverPointer;
175}
