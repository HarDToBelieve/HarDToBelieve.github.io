26#pragma once
27
28#if USE(LIBWEBRTC)
29
30#include "LibWebRTCResolver.h"
31#include "LibWebRTCSocket.h"
32#include <WebCore/LibWebRTCMacros.h>
33#include <WebCore/LibWebRTCSocketIdentifier.h>
34#include <webrtc/rtc_base/net_helpers.h>
35#include <webrtc/api/packet_socket_factory.h>
36#include <wtf/Deque.h>
37#include <wtf/Function.h>
38#include <wtf/HashMap.h>
39
40namespace WebKit {
41
42class LibWebRTCNetwork;
43class LibWebRTCSocket;
44
45class LibWebRTCSocketFactory {
46public:
47 LibWebRTCSocketFactory() = default;
48
49 void addSocket(LibWebRTCSocket&);
50 void removeSocket(LibWebRTCSocket&);
51 LibWebRTCSocket* socket(WebCore::LibWebRTCSocketIdentifier identifier) { return m_sockets.get(identifier); }
52
53 void forSocketInGroup(const void* socketGroup, const Function<void(LibWebRTCSocket&)>&);
54 rtc::AsyncPacketSocket* createUdpSocket(const void* socketGroup, const rtc::SocketAddress&, uint16_t minPort, uint16_t maxPort, bool isFirstParty, bool isRelayDisabled, const WebCore::RegistrableDomain&);
55 rtc::AsyncPacketSocket* createServerTcpSocket(const void* socketGroup, const rtc::SocketAddress&, uint16_t minPort, uint16_t maxPort, int options);
56 rtc::AsyncPacketSocket* createClientTcpSocket(const void* socketGroup, const rtc::SocketAddress& localAddress, const rtc::SocketAddress& remoteAddress, String&& userAgent, const rtc::PacketSocketTcpOptions&);
57 rtc::AsyncPacketSocket* createNewConnectionSocket(LibWebRTCSocket&, WebCore::LibWebRTCSocketIdentifier newConnectionSocketIdentifier, const rtc::SocketAddress&);
58
59 LibWebRTCResolver* resolver(LibWebRTCResolverIdentifier identifier) { return m_resolvers.get(identifier); }
60 std::unique_ptr<LibWebRTCResolver> takeResolver(LibWebRTCResolverIdentifier identifier) { return m_resolvers.take(identifier); }
61 rtc::AsyncResolverInterface* createAsyncResolver();
62
63 void disableNonLocalhostConnections() { m_disableNonLocalhostConnections = true; }
64
65 void setConnection(RefPtr<IPC::Connection>&&);
66 IPC::Connection* connection();
67
68private:
69 // We cannot own sockets, clients of the factory are responsible to free them.
70 HashMap<WebCore::LibWebRTCSocketIdentifier, LibWebRTCSocket*> m_sockets;
71
72 // We can own resolvers as we control their Destroy method.
73 HashMap<LibWebRTCResolverIdentifier, std::unique_ptr<LibWebRTCResolver>> m_resolvers;
74 bool m_disableNonLocalhostConnections { false };
75
76 RefPtr<IPC::Connection> m_connection;
77 Deque<Function<void(IPC::Connection&)>> m_pendingMessageTasks;
78};
79
80} // namespace WebKit
81
82#endif // USE(LIBWEBRTC)
