39void WebMDNSRegister::finishedRegisteringMDNSName(MDNSRegisterIdentifier identifier, LibWebRTCProvider::MDNSNameOrError&& result)
40{
41 auto pendingRegistration = m_pendingRegistrations.take(identifier);
42 if (!pendingRegistration.callback)
43 return;
44
45 if (result.has_value()) {
46 auto iterator = m_registeringDocuments.find(pendingRegistration.documentIdentifier);
47 if (iterator == m_registeringDocuments.end()) {
48 pendingRegistration.callback(makeUnexpected(WebCore::MDNSRegisterError::DNSSD));
49 return;
50 }
51 iterator->value.add(pendingRegistration.ipAddress, result.value());
52 }
53
54 pendingRegistration.callback(WTFMove(result));
55}
56
57void WebMDNSRegister::unregisterMDNSNames(DocumentIdentifier identifier)
58{
59 if (m_registeringDocuments.take(identifier).isEmpty())
60 return;
61
62 auto& connection = WebProcess::singleton().ensureNetworkProcessConnection().connection();
63 connection.send(Messages::NetworkMDNSRegister::UnregisterMDNSNames { identifier }, 0);
64}
65
66void WebMDNSRegister::registerMDNSName(DocumentIdentifier identifier, const String& ipAddress, CompletionHandler<void(LibWebRTCProvider::MDNSNameOrError&&)>&& callback)
67{
68 auto& map = m_registeringDocuments.ensure(identifier, [] {
69 return HashMap<String, String> { };
70 }).iterator->value;
71
72 auto iterator = map.find(ipAddress);
73 if (iterator != map.end()) {
74 callback(iterator->value);
75 return;
76 }
77
78 auto requestIdentifier = MDNSRegisterIdentifier::generate();
79 m_pendingRegistrations.add(requestIdentifier, PendingRegistration { WTFMove(callback), identifier, ipAddress });
80
81 auto& connection = WebProcess::singleton().ensureNetworkProcessConnection().connection();
82 if (!connection.send(Messages::NetworkMDNSRegister::RegisterMDNSName { requestIdentifier, identifier, ipAddress }, 0))
83 finishedRegisteringMDNSName(requestIdentifier, makeUnexpected(MDNSRegisterError::Internal));
84}
