38WebRTCResolver::WebRTCResolver(LibWebRTCSocketFactory& socketFactory, LibWebRTCResolverIdentifier identifier)
39 : m_socketFactory(socketFactory)
40 , m_identifier(identifier)
41{
42}
43
44void WebRTCResolver::setResolvedAddress(const Vector<RTCNetwork::IPAddress>& addresses)
45{
46 auto identifier = m_identifier;
47 auto& factory = m_socketFactory;
48
49 Vector<rtc::IPAddress> rtcAddresses;
50 rtcAddresses.reserveInitialCapacity(addresses.size());
51 for (auto& address : addresses)
52 rtcAddresses.uncheckedAppend(address.value);
53
54 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([&factory, identifier, rtcAddresses = WTFMove(rtcAddresses)]() {
55 auto* resolver = factory.resolver(identifier);
56 if (!resolver)
57 return;
58 resolver->setResolvedAddress(rtcAddresses);
59 });
60}
61
62void WebRTCResolver::resolvedAddressError(int error)
63{
64 auto identifier = m_identifier;
65 auto& factory = m_socketFactory;
66 WebCore::LibWebRTCProvider::callOnWebRTCNetworkThread([&factory, identifier, error]() {
67 auto* resolver = factory.resolver(identifier);
68 if (!resolver)
69 return;
70 resolver->setError(error);
71 });
72}
