57static NPIdentifier npIdentifierFromIdentifier(PropertyName propertyName)
58{
59 String name(propertyName.publicName());
60 // If the propertyName is Symbol.
61 if (name.isNull())
62 return nullptr;
63 return static_cast<NPIdentifier>(IdentifierRep::get(name.utf8().data()));
64}

121JSValue JSNPObject::callMethod(JSGlobalObject* lexicalGlobalObject, CallFrame* callFrame, NPIdentifier methodName)
122{
123 VM& vm = lexicalGlobalObject->vm();
124 auto scope = DECLARE_THROW_SCOPE(vm);
125
126 ASSERT_THIS_GC_OBJECT_INHERITS(info());
127 if (!m_npObject)
128 return throwInvalidAccessError(lexicalGlobalObject, scope);
129
130 // If the propertyName is symbol.
131 if (!methodName)
132 return jsUndefined();
133
134 size_t argumentCount = callFrame->argumentCount();
135 Vector<NPVariant, 8> arguments(argumentCount);
136
137 // Convert all arguments to NPVariants.
138 for (size_t i = 0; i < argumentCount; ++i)
139 m_objectMap->convertJSValueToNPVariant(lexicalGlobalObject, callFrame->uncheckedArgument(i), arguments[i]);
140
141 // Calling NPClass::invoke will call into plug-in code, and there's no telling what the plug-in can do.
142 // (including destroying the plug-in). Because of this, we make sure to keep the plug-in alive until
143 // the call has finished.
144 NPRuntimeObjectMap::PluginProtector protector(m_objectMap);
145
146 bool returnValue;
147 NPVariant result;
148 VOID_TO_NPVARIANT(result);
149
150 {
151 JSLock::DropAllLocks dropAllLocks(commonVM());
152 returnValue = m_npObject->_class->invoke(m_npObject, methodName, arguments.data(), argumentCount, &result);
153 NPRuntimeObjectMap::moveGlobalExceptionToExecState(lexicalGlobalObject);
154 }
155
156 // Release all arguments.
157 for (size_t i = 0; i < argumentCount; ++i)
158 releaseNPVariantValue(&arguments[i]);
159
160 if (!returnValue)
161 throwException(lexicalGlobalObject, scope, createError(lexicalGlobalObject, "Error calling method on NPObject."));
162
163 JSValue propertyValue = m_objectMap->convertNPVariantToJSValue(globalObject(), result);
164 releaseNPVariantValue(&result);
165 return propertyValue;
166}

295bool JSNPObject::getOwnPropertySlot(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, PropertySlot& slot)
296{
297 VM& vm = lexicalGlobalObject->vm();
298 auto scope = DECLARE_THROW_SCOPE(vm);
299
300 JSNPObject* thisObject = JSC::jsCast<JSNPObject*>(object);
301 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
302 if (!thisObject->m_npObject) {
303 throwInvalidAccessError(lexicalGlobalObject, scope);
304 return false;
305 }
306
307 NPIdentifier npIdentifier = npIdentifierFromIdentifier(propertyName);
308 // If the propertyName is symbol.
309 if (!npIdentifier)
310 return false;
311
312 // Calling NPClass::invoke will call into plug-in code, and there's no telling what the plug-in can do.
313 // (including destroying the plug-in). Because of this, we make sure to keep the plug-in alive until
314 // the call has finished.
315 NPRuntimeObjectMap::PluginProtector protector(thisObject->m_objectMap);
316
317 // First, check if the NPObject has a property with this name.
318 if (thisObject->m_npObject->_class->hasProperty && thisObject->m_npObject->_class->hasProperty(thisObject->m_npObject, npIdentifier)) {
319 slot.setCustom(thisObject, static_cast<unsigned>(JSC::PropertyAttribute::DontDelete), propertyGetter);
320 return true;
321 }
322
323 // Second, check if the NPObject has a method with this name.
324 if (thisObject->m_npObject->_class->hasMethod && thisObject->m_npObject->_class->hasMethod(thisObject->m_npObject, npIdentifier)) {
325 slot.setCustom(thisObject, JSC::PropertyAttribute::DontDelete | JSC::PropertyAttribute::ReadOnly, methodGetter);
326 return true;
327 }
328
329 return false;
330}

332bool JSNPObject::put(JSCell* cell, JSGlobalObject* lexicalGlobalObject, PropertyName propertyName, JSValue value, PutPropertySlot&)
333{
334 VM& vm = lexicalGlobalObject->vm();
335 auto scope = DECLARE_THROW_SCOPE(vm);
336
337 JSNPObject* thisObject = JSC::jsCast<JSNPObject*>(cell);
338 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
339 if (!thisObject->m_npObject) {
340 throwInvalidAccessError(lexicalGlobalObject, scope);
341 return false;
342 }
343
344 NPIdentifier npIdentifier = npIdentifierFromIdentifier(propertyName);
345 // If the propertyName is symbol.
346 if (!npIdentifier)
347 return false;
348
349 if (!thisObject->m_npObject->_class->hasProperty || !thisObject->m_npObject->_class->hasProperty(thisObject->m_npObject, npIdentifier)) {
350 // FIXME: Should we throw an exception here?
351 return false;
352 }
353
354 if (!thisObject->m_npObject->_class->setProperty)
355 return false;
356
357 NPVariant variant;
358 thisObject->m_objectMap->convertJSValueToNPVariant(lexicalGlobalObject, value, variant);
359
360 // Calling NPClass::setProperty will call into plug-in code, and there's no telling what the plug-in can do.
361 // (including destroying the plug-in). Because of this, we make sure to keep the plug-in alive until
362 // the call has finished.
363 NPRuntimeObjectMap::PluginProtector protector(thisObject->m_objectMap);
364
365 bool result = false;
366 {
367 JSLock::DropAllLocks dropAllLocks(commonVM());
368 result = thisObject->m_npObject->_class->setProperty(thisObject->m_npObject, npIdentifier, &variant);
369
370 NPRuntimeObjectMap::moveGlobalExceptionToExecState(lexicalGlobalObject);
371
372 // FIXME: Should we throw an exception if setProperty returns false?
373 }
374
375 releaseNPVariantValue(&variant);
376 return result;
377}

389bool JSNPObject::deleteProperty(JSGlobalObject* lexicalGlobalObject, NPIdentifier propertyName)
390{
391 VM& vm = lexicalGlobalObject->vm();
392 auto scope = DECLARE_THROW_SCOPE(vm);
393
394 ASSERT_THIS_GC_OBJECT_INHERITS(info());
395
396 // If the propertyName is symbol.
397 if (!propertyName)
398 return false;
399
400 if (!m_npObject) {
401 throwInvalidAccessError(lexicalGlobalObject, scope);
402 return false;
403 }
404
405 if (!m_npObject->_class->removeProperty) {
406 // FIXME: Should we throw an exception here?
407 return false;
408 }
409
410 // Calling NPClass::setProperty will call into plug-in code, and there's no telling what the plug-in can do.
411 // (including destroying the plug-in). Because of this, we make sure to keep the plug-in alive until
412 // the call has finished.
413 NPRuntimeObjectMap::PluginProtector protector(m_objectMap);
414
415 {
416 JSLock::DropAllLocks dropAllLocks(commonVM());
417
418 // FIXME: Should we throw an exception if removeProperty returns false?
419 if (!m_npObject->_class->removeProperty(m_npObject, propertyName))
420 return false;
421
422 NPRuntimeObjectMap::moveGlobalExceptionToExecState(lexicalGlobalObject);
423 }
424
425 return true;
426}

428void JSNPObject::getOwnPropertyNames(JSObject* object, JSGlobalObject* lexicalGlobalObject, PropertyNameArray& propertyNameArray, DontEnumPropertiesMode)
429{
430 VM& vm = lexicalGlobalObject->vm();
431 auto scope = DECLARE_THROW_SCOPE(vm);
432
433 JSNPObject* thisObject = jsCast<JSNPObject*>(object);
434 ASSERT_GC_OBJECT_INHERITS(thisObject, info());
435 if (!thisObject->m_npObject) {
436 throwInvalidAccessError(lexicalGlobalObject, scope);
437 return;
438 }
439
440 if (!NP_CLASS_STRUCT_VERSION_HAS_ENUM(thisObject->m_npObject->_class) || !thisObject->m_npObject->_class->enumerate)
441 return;
442
443 NPIdentifier* identifiers = 0;
444 uint32_t identifierCount = 0;
445
446 // Calling NPClass::enumerate will call into plug-in code, and there's no telling what the plug-in can do.
447 // (including destroying the plug-in). Because of this, we make sure to keep the plug-in alive until
448 // the call has finished.
449 NPRuntimeObjectMap::PluginProtector protector(thisObject->m_objectMap);
450
451 {
452 JSLock::DropAllLocks dropAllLocks(commonVM());
453
454 // FIXME: Should we throw an exception if enumerate returns false?
455 if (!thisObject->m_npObject->_class->enumerate(thisObject->m_npObject, &identifiers, &identifierCount))
456 return;
457
458 NPRuntimeObjectMap::moveGlobalExceptionToExecState(lexicalGlobalObject);
459 }
460
461 for (uint32_t i = 0; i < identifierCount; ++i) {
462 IdentifierRep* identifierRep = static_cast<IdentifierRep*>(identifiers[i]);
463
464 Identifier identifier;
465 if (identifierRep->isString()) {
466 const char* string = identifierRep->string();
467 int length = strlen(string);
468
469 identifier = Identifier::fromString(vm, String::fromUTF8WithLatin1Fallback(string, length));
470 } else
471 identifier = Identifier::from(vm, identifierRep->number());
472
473 propertyNameArray.add(identifier);
474 }
475
476 npnMemFree(identifiers);
477}

479JSC_DEFINE_CUSTOM_GETTER(propertyGetter, (JSGlobalObject* lexicalGlobalObject, EncodedJSValue thisValue, PropertyName propertyName))
480{
481 VM& vm = lexicalGlobalObject->vm();
482 auto scope = DECLARE_THROW_SCOPE(vm);
483
484 JSNPObject* thisObj = jsCast<JSNPObject*>(JSValue::decode(thisValue));
485 ASSERT_GC_OBJECT_INHERITS(thisObj, JSNPObject::info());
486
487 if (!thisObj->npObject())
488 return JSValue::encode(throwInvalidAccessError(lexicalGlobalObject, scope));
489
490 if (!thisObj->npObject()->_class->getProperty)
491 return JSValue::encode(jsUndefined());
492
493 NPVariant result;
494 VOID_TO_NPVARIANT(result);
495
496 // Calling NPClass::getProperty will call into plug-in code, and there's no telling what the plug-in can do.
497 // (including destroying the plug-in). Because of this, we make sure to keep the plug-in alive until
498 // the call has finished.
499 NPRuntimeObjectMap::PluginProtector protector(thisObj->objectMap());
500
501 bool returnValue;
502 {
503 JSLock::DropAllLocks dropAllLocks(commonVM());
504 NPIdentifier npIdentifier = npIdentifierFromIdentifier(propertyName);
505 // If the propertyName is symbol.
506 if (!npIdentifier)
507 return JSValue::encode(jsUndefined());
508
509 returnValue = thisObj->npObject()->_class->getProperty(thisObj->npObject(), npIdentifier, &result);
510
511 NPRuntimeObjectMap::moveGlobalExceptionToExecState(lexicalGlobalObject);
512 }
513
514 if (!returnValue)
515 return JSValue::encode(jsUndefined());
516
517 JSValue propertyValue = thisObj->objectMap()->convertNPVariantToJSValue(thisObj->globalObject(), result);
518 releaseNPVariantValue(&result);
519 return JSValue::encode(propertyValue);
520}

522JSC_DEFINE_CUSTOM_GETTER(methodGetter, (JSGlobalObject* lexicalGlobalObject, EncodedJSValue thisValue, PropertyName propertyName))
523{
524 VM& vm = lexicalGlobalObject->vm();
525 auto scope = DECLARE_THROW_SCOPE(vm);
526
527 JSNPObject* thisObj = jsCast<JSNPObject*>(JSValue::decode(thisValue));
528 ASSERT_GC_OBJECT_INHERITS(thisObj, JSNPObject::info());
529
530 if (!thisObj->npObject())
531 return JSValue::encode(throwInvalidAccessError(lexicalGlobalObject, scope));
532
533 NPIdentifier npIdentifier = npIdentifierFromIdentifier(propertyName);
534 // If the propertyName is symbol.
535 if (!npIdentifier)
536 return JSValue::encode(throwInvalidAccessError(lexicalGlobalObject, scope));
537
538 return JSValue::encode(JSNPMethod::create(thisObj->globalObject(), propertyName.publicName(), npIdentifier));
539}
