48NPJSObject* NPJSObject::create(VM& vm, NPRuntimeObjectMap* objectMap, JSObject* jsObject)
49{
50 // We should never have a JSNPObject inside an NPJSObject.
51 ASSERT(!jsObject->inherits<JSNPObject>(vm));
52
53 NPJSObject* npJSObject = toNPJSObject(createNPObject(0, npClass()));
54 npJSObject->initialize(vm, objectMap, jsObject);
55
56 return npJSObject;
57}

83static Identifier identifierFromIdentifierRep(JSGlobalObject* lexicalGlobalObject, IdentifierRep* identifierRep)
84{
85 VM& vm = lexicalGlobalObject->vm();
86 ASSERT(identifierRep->isString());
87
88 const char* string = identifierRep->string();
89 int length = strlen(string);
90
91 return Identifier::fromString(vm, String::fromUTF8WithLatin1Fallback(string, length));
92}

94bool NPJSObject::hasMethod(NPIdentifier methodName)
95{
96 IdentifierRep* identifierRep = static_cast<IdentifierRep*>(methodName);
97
98 if (!identifierRep->isString())
99 return false;
100
101 JSGlobalObject* lexicalGlobalObject = m_objectMap->globalObject();
102 if (!lexicalGlobalObject)
103 return false;
104
105 VM& vm = lexicalGlobalObject->vm();
106 JSLockHolder lock(vm);
107 auto scope = DECLARE_CATCH_SCOPE(vm);
108
109 JSValue value = m_jsObject->get(lexicalGlobalObject, identifierFromIdentifierRep(lexicalGlobalObject, identifierRep));
110 scope.clearException();
111
112 return value.isCallable(vm);
113}

115bool NPJSObject::invoke(NPIdentifier methodName, const NPVariant* arguments, uint32_t argumentCount, NPVariant* result)
116{
117 IdentifierRep* identifierRep = static_cast<IdentifierRep*>(methodName);
118
119 if (!identifierRep->isString())
120 return false;
121
122 JSGlobalObject* lexicalGlobalObject = m_objectMap->globalObject();
123 if (!lexicalGlobalObject)
124 return false;
125
126 JSLockHolder lock(lexicalGlobalObject);
127
128 JSValue function = m_jsObject->get(lexicalGlobalObject, identifierFromIdentifierRep(lexicalGlobalObject, identifierRep));
129 return invoke(lexicalGlobalObject, function, arguments, argumentCount, result);
130}

144bool NPJSObject::hasProperty(NPIdentifier identifier)
145{
146 IdentifierRep* identifierRep = static_cast<IdentifierRep*>(identifier);
147
148 JSGlobalObject* lexicalGlobalObject = m_objectMap->globalObject();
149 if (!lexicalGlobalObject)
150 return false;
151
152 VM& vm = lexicalGlobalObject->vm();
153 JSLockHolder lock(vm);
154 auto scope = DECLARE_CATCH_SCOPE(vm);
155
156 bool result;
157 if (identifierRep->isString())
158 result = m_jsObject->hasProperty(lexicalGlobalObject, identifierFromIdentifierRep(lexicalGlobalObject, identifierRep));
159 else
160 result = m_jsObject->hasProperty(lexicalGlobalObject, static_cast<uint32_t>(identifierRep->number()));
161
162 scope.clearException();
163 return result;
164}

166bool NPJSObject::getProperty(NPIdentifier propertyName, NPVariant* result)
167{
168 IdentifierRep* identifierRep = static_cast<IdentifierRep*>(propertyName);
169
170 JSGlobalObject* lexicalGlobalObject = m_objectMap->globalObject();
171 if (!lexicalGlobalObject)
172 return false;
173
174 VM& vm = lexicalGlobalObject->vm();
175 JSLockHolder lock(vm);
176 auto scope = DECLARE_CATCH_SCOPE(vm);
177
178 JSValue jsResult;
179 if (identifierRep->isString())
180 jsResult = m_jsObject->get(lexicalGlobalObject, identifierFromIdentifierRep(lexicalGlobalObject, identifierRep));
181 else
182 jsResult = m_jsObject->get(lexicalGlobalObject, static_cast<uint32_t>(identifierRep->number()));
183
184 m_objectMap->convertJSValueToNPVariant(lexicalGlobalObject, jsResult, *result);
185 scope.clearException();
186 return true;
187}

189bool NPJSObject::setProperty(NPIdentifier propertyName, const NPVariant* value)
190{
191 IdentifierRep* identifierRep = static_cast<IdentifierRep*>(propertyName);
192
193 JSGlobalObject* lexicalGlobalObject = m_objectMap->globalObject();
194 if (!lexicalGlobalObject)
195 return false;
196
197 VM& vm = lexicalGlobalObject->vm();
198 JSLockHolder lock(vm);
199 auto scope = DECLARE_CATCH_SCOPE(vm);
200
201 JSValue jsValue = m_objectMap->convertNPVariantToJSValue(m_objectMap->globalObject(), *value);
202 if (identifierRep->isString()) {
203 PutPropertySlot slot(m_jsObject.get());
204 m_jsObject->methodTable(vm)->put(m_jsObject.get(), lexicalGlobalObject, identifierFromIdentifierRep(lexicalGlobalObject, identifierRep), jsValue, slot);
205 } else
206 m_jsObject->methodTable(vm)->putByIndex(m_jsObject.get(), lexicalGlobalObject, identifierRep->number(), jsValue, false);
207 scope.clearException();
208
209 return true;
210}

212bool NPJSObject::removeProperty(NPIdentifier propertyName)
213{
214 IdentifierRep* identifierRep = static_cast<IdentifierRep*>(propertyName);
215
216 JSGlobalObject* lexicalGlobalObject = m_objectMap->globalObject();
217 if (!lexicalGlobalObject)
218 return false;
219
220 VM& vm = lexicalGlobalObject->vm();
221 JSLockHolder lock(vm);
222 auto scope = DECLARE_CATCH_SCOPE(vm);
223
224 if (identifierRep->isString()) {
225 Identifier identifier = identifierFromIdentifierRep(lexicalGlobalObject, identifierRep);
226
227 if (!m_jsObject->hasProperty(lexicalGlobalObject, identifier)) {
228 scope.clearException();
229 return false;
230 }
231
232 JSCell::deleteProperty(m_jsObject.get(), lexicalGlobalObject, identifier);
233 } else {
234 if (!m_jsObject->hasProperty(lexicalGlobalObject, static_cast<uint32_t>(identifierRep->number()))) {
235 scope.clearException();
236 return false;
237 }
238
239 m_jsObject->methodTable(vm)->deletePropertyByIndex(m_jsObject.get(), lexicalGlobalObject, identifierRep->number());
240 }
241
242 scope.clearException();
243 return true;
244}

246bool NPJSObject::enumerate(NPIdentifier** identifiers, uint32_t* identifierCount)
247{
248 JSGlobalObject* lexicalGlobalObject = m_objectMap->globalObject();
249 if (!lexicalGlobalObject)
250 return false;
251
252 VM& vm = lexicalGlobalObject->vm();
253 JSLockHolder lock(vm);
254
255 PropertyNameArray propertyNames(vm, PropertyNameMode::Strings, PrivateSymbolMode::Exclude);
256 m_jsObject->getPropertyNames(lexicalGlobalObject, propertyNames, DontEnumPropertiesMode::Exclude);
257
258 NPIdentifier* nameIdentifiers = npnMemNewArray<NPIdentifier>(propertyNames.size());
259
260 for (size_t i = 0; i < propertyNames.size(); ++i)
261 nameIdentifiers[i] = static_cast<NPIdentifier>(IdentifierRep::get(propertyNames[i].string().utf8().data()));
262
263 *identifiers = nameIdentifiers;
264 *identifierCount = propertyNames.size();
265
266 return true;
267}

269bool NPJSObject::construct(const NPVariant* arguments, uint32_t argumentCount, NPVariant* result)
270{
271 JSGlobalObject* lexicalGlobalObject = m_objectMap->globalObject();
272 if (!lexicalGlobalObject)
273 return false;
274
275 VM& vm = lexicalGlobalObject->vm();
276 JSLockHolder lock(vm);
277 auto scope = DECLARE_CATCH_SCOPE(vm);
278
279 auto constructData = getConstructData(vm, m_jsObject.get());
280 if (constructData.type == CallData::Type::None)
281 return false;
282
283 // Convert the passed in arguments.
284 MarkedArgumentBuffer argumentList;
285 for (uint32_t i = 0; i < argumentCount; ++i)
286 argumentList.append(m_objectMap->convertNPVariantToJSValue(m_objectMap->globalObject(), arguments[i]));
287 RELEASE_ASSERT(!argumentList.hasOverflowed());
288
289 JSValue value = JSC::construct(lexicalGlobalObject, m_jsObject.get(), constructData, argumentList);
290
291 // Convert and return the new object.
292 m_objectMap->convertJSValueToNPVariant(lexicalGlobalObject, value, *result);
293 scope.clearException();
294
295 return true;
296}

298bool NPJSObject::invoke(JSGlobalObject* lexicalGlobalObject, JSValue function, const NPVariant* arguments, uint32_t argumentCount, NPVariant* result)
299{
300 VM& vm = lexicalGlobalObject->vm();
301 auto scope = DECLARE_CATCH_SCOPE(vm);
302
303 auto callData = getCallData(vm, function);
304 if (callData.type == CallData::Type::None)
305 return false;
306
307 // Convert the passed in arguments.
308 MarkedArgumentBuffer argumentList;
309 for (uint32_t i = 0; i < argumentCount; ++i)
310 argumentList.append(m_objectMap->convertNPVariantToJSValue(lexicalGlobalObject, arguments[i]));
311 RELEASE_ASSERT(!argumentList.hasOverflowed());
312
313 JSValue value = JSC::call(lexicalGlobalObject, function, callData, m_jsObject.get(), argumentList);
314
315 if (UNLIKELY(scope.exception())) {
316 scope.clearException();
317 return false;
318 }
319
320 // Convert and return the result of the function call.
321 m_objectMap->convertJSValueToNPVariant(lexicalGlobalObject, value, *result);
322
323 if (UNLIKELY(scope.exception())) {
324 scope.clearException();
325 return false;
326 }
327
328 return true;
329}
