69static bool startsWithBlankLine(const char* bytes, unsigned length)
70{
71 return length > 0 && bytes[0] == '\n';
72}
73
74static int locationAfterFirstBlankLine(const char* bytes, unsigned length)
75{
76 for (unsigned i = 0; i < length - 4; i++) {
77 // Support for Acrobat. It sends "\n\n".
78 if (bytes[i] == '\n' && bytes[i + 1] == '\n')
79 return i + 2;
80
81 // Returns the position after 2 CRLF's or 1 CRLF if it is the first line.
82 if (bytes[i] == '\r' && bytes[i + 1] == '\n') {
83 i += 2;
84 if (i == 2)
85 return i;
86
87 if (bytes[i] == '\n') {
88 // Support for Director. It sends "\r\n\n" (3880387).
89 return i + 1;
90 }
91
92 if (bytes[i] == '\r' && bytes[i + 1] == '\n') {
93 // Support for Flash. It sends "\r\n\r\n" (3758113).
94 return i + 2;
95 }
96 }
97 }
98
99 return -1;
100}

102static const char* findEndOfLine(const char* bytes, unsigned length)
103{
104 // According to the HTTP specification EOL is defined as
105 // a CRLF pair. Unfortunately, some servers will use LF
106 // instead. Worse yet, some servers will use a combination
107 // of both (e.g. <header>CRLFLF<body>), so findEOL needs
108 // to be more forgiving. It will now accept CRLF, LF or
109 // CR.
110 //
111 // It returns 0 if EOLF is not found or it will return
112 // a pointer to the first terminating character.
113 for (unsigned i = 0; i < length; i++) {
114 if (bytes[i] == '\n')
115 return bytes + i;
116 if (bytes[i] == '\r') {
117 // Check to see if spanning buffer bounds
118 // (CRLF is across reads). If so, wait for
119 // next read.
120 if (i + 1 == length)
121 break;
122
123 return bytes + i;
124 }
125 }
126
127 return 0;
128}

130static String capitalizeRFC822HeaderFieldName(const String& name)
131{
132 bool capitalizeCharacter = true;
133 StringBuilder result;
134 for (unsigned i = 0; i < name.length(); i++) {
135 result.append(capitalizeCharacter ? toASCIIUpper(name[i]) : toASCIILower(name[i]));
136 if (name[i] == '-')
137 capitalizeCharacter = true;
138 else
139 capitalizeCharacter = false;
140 }
141 return result.toString();
142}

144static HTTPHeaderMap parseRFC822HeaderFields(const char* bytes, unsigned length)
145{
146 String lastHeaderKey;
147 HTTPHeaderMap headerFields;
148
149 // Loop over lines until we're past the header, or we can't find any more end-of-lines
150 while (const char* endOfLine = findEndOfLine(bytes, length)) {
151 const char* line = bytes;
152 int lineLength = endOfLine - bytes;
153
154 // Move bytes to the character after the terminator as returned by findEndOfLine.
155 bytes = endOfLine + 1;
156 if ((*endOfLine == '\r') && (*bytes == '\n'))
157 bytes++; // Safe since findEndOfLine won't return a spanning CRLF.
158
159 length -= (bytes - line);
160 if (!lineLength) {
161 // Blank line; we're at the end of the header
162 break;
163 }
164
165 if (*line == ' ' || *line == '\t') {
166 // Continuation of the previous header
167 if (lastHeaderKey.isNull()) {
168 // malformed header; ignore it and continue
169 continue;
170 }
171
172 // Merge the continuation of the previous header
173 String currentValue = headerFields.get(lastHeaderKey);
174 String newValue(line, lineLength);
175
176 headerFields.set(lastHeaderKey, currentValue + newValue);
177 } else {
178 // Brand new header
179 const char* colon = line;
180 while (*colon != ':' && colon != endOfLine)
181 colon++;
182
183 if (colon == endOfLine) {
184 // malformed header; ignore it and continue
185 continue;
186 }
187
188 lastHeaderKey = capitalizeRFC822HeaderFieldName(String(line, colon - line));
189 String value;
190
191 for (colon++; colon != endOfLine; colon++) {
192 if (*colon != ' ' && *colon != '\t')
193 break;
194 }
195 if (colon == endOfLine)
196 value = emptyString();
197 else
198 value = String(colon, endOfLine - colon);
199
200 String oldValue = headerFields.get(lastHeaderKey);
201 if (!oldValue.isNull())
202 value = oldValue + ", " + value;
203
204 headerFields.set(lastHeaderKey, value);
205 }
206 }
207
208 return headerFields;
209}

211static NPError parsePostBuffer(bool isFile, const char *buffer, uint32_t length, bool parseHeaders, HTTPHeaderMap& headerFields, Vector<uint8_t>& bodyData)
212{
213 RefPtr<SharedBuffer> fileContents;
214 const char* postBuffer = 0;
215 uint32_t postBufferSize = 0;
216
217 if (isFile) {
218 fileContents = SharedBuffer::createWithContentsOfFile(String::fromUTF8(buffer));
219 if (!fileContents)
220 return NPERR_FILE_NOT_FOUND;
221
222 postBuffer = fileContents->dataAsCharPtr();
223 postBufferSize = fileContents->size();
224
225 // FIXME: The NPAPI spec states that the file should be deleted here.
226 } else {
227 postBuffer = buffer;
228 postBufferSize = length;
229 }
230
231 if (parseHeaders) {
232 if (startsWithBlankLine(postBuffer, postBufferSize)) {
233 postBuffer++;
234 postBufferSize--;
235 } else {
236 int location = locationAfterFirstBlankLine(postBuffer, postBufferSize);
237 if (location != -1) {
238 // If the blank line is somewhere in the middle of the buffer, everything before is the header
239 headerFields = parseRFC822HeaderFields(postBuffer, location);
240 unsigned dataLength = postBufferSize - location;
241
242 // Sometimes plugins like to set Content-Length themselves when they post,
243 // but WebFoundation does not like that. So we will remove the header
244 // and instead truncate the data to the requested length.
245 String contentLength = headerFields.get(HTTPHeaderName::ContentLength);
246
247 if (!contentLength.isNull())
248 dataLength = std::min(parseIntegerAllowingTrailingJunk<unsigned>(contentLength).value_or(0), dataLength);
249 headerFields.remove(HTTPHeaderName::ContentLength);
250
251 postBuffer += location;
252 postBufferSize = dataLength;
253 }
254 }
255 }
256
257 ASSERT(bodyData.isEmpty());
258 bodyData.append(postBuffer, postBufferSize);
259
260 return NPERR_NO_ERROR;
261}

263static String makeURLString(const char* url)
264{
265 String urlString(url);
266
267 // Strip return characters.
268 urlString.replaceWithLiteral('\r', "");
269 urlString.replaceWithLiteral('\n', "");
270
271 return urlString;
272}

285static NPError NPN_PostURL(NPP npp, const char* url, const char* target, uint32_t len, const char* buf, NPBool file)
286{
287 HTTPHeaderMap headerFields;
288 Vector<uint8_t> postData;
289
290 // NPN_PostURL only allows headers if the post buffer points to a file.
291 bool parseHeaders = file;
292
293 NPError error = parsePostBuffer(file, buf, len, parseHeaders, headerFields, postData);
294 if (error != NPERR_NO_ERROR)
295 return error;
296
297 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
298 plugin->loadURL("POST", makeURLString(url), target, WTFMove(headerFields), postData, false, 0);
299 return NPERR_NO_ERROR;
300}

411static NPError NPN_GetValue(NPP npp, NPNVariable variable, void *value)
412{
413 switch (static_cast<unsigned>(variable)) {
414 case NPNVWindowNPObject: {
415 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
416 PluginDestructionProtector protector(plugin.get());
417
418 NPObject* windowNPObject = plugin->windowScriptNPObject();
419 if (!windowNPObject)
420 return NPERR_GENERIC_ERROR;
421
422 *(NPObject**)value = windowNPObject;
423 break;
424 }
425 case NPNVPluginElementNPObject: {
426 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
427 PluginDestructionProtector protector(plugin.get());
428
429 NPObject* pluginElementNPObject = plugin->pluginElementNPObject();
430 *(NPObject**)value = pluginElementNPObject;
431 break;
432 }
433 case NPNVprivateModeBool: {
434 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
435
436 *(NPBool*)value = plugin->isPrivateBrowsingEnabled();
437 break;
438 }
439
440 case NPNVmuteAudioBool: {
441 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
442 *(NPBool*)value = plugin->isMuted();
443 break;
444 }
445#if PLATFORM(COCOA)
446 case NPNVsupportsCoreGraphicsBool:
447 // Always claim to support the Core Graphics drawing model.
448 *(NPBool*)value = true;
449 break;
450
451 case WKNVSupportsCompositingCoreAnimationPluginsBool:
452 case NPNVsupportsCoreAnimationBool: {
453 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
454
455 *(NPBool*)value = plugin->isAcceleratedCompositingEnabled();
456 break;
457 }
458 case NPNVcontentsScaleFactor: {
459 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
460
461 *(double*)value = plugin->contentsScaleFactor();
462 break;
463 }
464 case NPNVsupportsCocoaBool:
465 // Always claim to support the Cocoa event model.
466 *(NPBool*)value = true;
467 break;
468
469 case NPNVsupportsUpdatedCocoaTextInputBool: {
470 // The plug-in is asking whether we support the updated Cocoa text input model.
471 // If we haven't yet delivered a key down event to the plug-in, we can opt into the updated
472 // model and say that we support it. Otherwise, we'll just fall back and say that we don't support it.
473 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
474
475 bool supportsUpdatedTextInput = !plugin->hasHandledAKeyDownEvent();
476 if (supportsUpdatedTextInput)
477 plugin->setPluginWantsLegacyCocoaTextInput(false);
478
479 *reinterpret_cast<NPBool*>(value) = supportsUpdatedTextInput;
480 break;
481 }
482
483 case WKNVCALayerRenderServerPort: {
484 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
485
486 *(mach_port_t*)value = plugin->compositingRenderServerPort().sendRight();
487 break;
488 }
489
490 case WKNVExpectsNonretainedLayer: {
491 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
492
493 // Asking for this will make us expect a non-retained layer from the plug-in.
494 plugin->setPluginReturnsNonretainedLayer(true);
495 *(NPBool*)value = true;
496 break;
497 }
498#elif PLATFORM(X11)
499 case NPNVxDisplay: {
500 if (!npp)
501 return NPERR_GENERIC_ERROR;
502 auto& display = PlatformDisplay::sharedDisplay();
503 if (display.type() != PlatformDisplay::Type::X11)
504 return NPERR_GENERIC_ERROR;
505 *reinterpret_cast<Display**>(value) = downcast<PlatformDisplayX11>(display).native();
506 break;
507 }
508 case NPNVSupportsXEmbedBool:
509 *static_cast<NPBool*>(value) = PlatformDisplay::sharedDisplay().type() == PlatformDisplay::Type::X11;
510 break;
511 case NPNVSupportsWindowless:
512 *static_cast<NPBool*>(value) = true;
513 break;
514
515 case NPNVToolkit: {
516 // Gtk based plugins need to be assured about the toolkit version.
517 const uint32_t expectedGtkToolKitVersion = 2;
518 *reinterpret_cast<uint32_t*>(value) = expectedGtkToolKitVersion;
519 break;
520 }
521
522 // TODO: implement NPNVnetscapeWindow once we want to support windowed plugins.
523#endif
524 default:
525 notImplemented();
526 return NPERR_GENERIC_ERROR;
527 }
528
529 return NPERR_NO_ERROR;
530}

626static NPUTF8 *NPN_UTF8FromIdentifier(NPIdentifier identifier)
627{
628 const char* string = static_cast<IdentifierRep*>(identifier)->string();
629 if (!string)
630 return 0;
631
632 uint32_t stringLength = strlen(string);
633 char* utf8String = npnMemNewArray<char>(stringLength + 1);
634 memcpy(utf8String, string, stringLength);
635 utf8String[stringLength] = '\0';
636
637 return utf8String;
638}

799static NPError copyCString(const CString& string, char** value, uint32_t* len)
800{
801 ASSERT(!string.isNull());
802 ASSERT(value);
803 ASSERT(len);
804
805 *value = npnMemNewArray<char>(string.length());
806 if (!*value)
807 return NPERR_GENERIC_ERROR;
808
809 memcpy(*value, string.data(), string.length());
810 *len = string.length();
811 return NPERR_NO_ERROR;
812}

814static NPError NPN_GetValueForURL(NPP npp, NPNURLVariable variable, const char* url, char** value, uint32_t* len)
815{
816 if (!value || !len)
817 return NPERR_GENERIC_ERROR;
818
819 switch (variable) {
820 case NPNURLVCookie: {
821 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
822 PluginDestructionProtector protector(plugin.get());
823
824 String cookies = plugin->cookiesForURL(makeURLString(url));
825 if (cookies.isNull())
826 return NPERR_GENERIC_ERROR;
827
828 return copyCString(cookies.utf8(), value, len);
829 }
830
831 case NPNURLVProxy: {
832 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
833 PluginDestructionProtector protector(plugin.get());
834
835 String proxies = plugin->proxiesForURL(makeURLString(url));
836 if (proxies.isNull())
837 return NPERR_GENERIC_ERROR;
838
839 return copyCString(proxies.utf8(), value, len);
840 }
841 default:
842 notImplemented();
843 return NPERR_GENERIC_ERROR;
844 }
845}

868static bool initializeProtectionSpace(const char* protocol, const char* host, int port, const char* scheme, const char* realm, ProtectionSpace& protectionSpace)
869{
870 ProtectionSpaceServerType serverType;
871 if (equalLettersIgnoringASCIICase(protocol, "http"))
872 serverType = ProtectionSpaceServerHTTP;
873 else if (equalLettersIgnoringASCIICase(protocol, "https"))
874 serverType = ProtectionSpaceServerHTTPS;
875 else {
876 // We only care about http and https.
877 return false;
878 }
879
880 ProtectionSpaceAuthenticationScheme authenticationScheme = ProtectionSpaceAuthenticationSchemeDefault;
881 if (serverType == ProtectionSpaceServerHTTP) {
882 if (equalLettersIgnoringASCIICase(scheme, "basic"))
883 authenticationScheme = ProtectionSpaceAuthenticationSchemeHTTPBasic;
884 else if (equalLettersIgnoringASCIICase(scheme, "digest"))
885 authenticationScheme = ProtectionSpaceAuthenticationSchemeHTTPDigest;
886 }
887
888 protectionSpace = ProtectionSpace(host, port, serverType, realm, authenticationScheme);
889 return true;
890}

892static NPError NPN_GetAuthenticationInfo(NPP npp, const char* protocol, const char* host, int32_t port, const char* scheme,
893 const char* realm, char** username, uint32_t* usernameLength, char** password, uint32_t* passwordLength)
894{
895 if (!protocol || !host || !scheme || !realm || !username || !usernameLength || !password || !passwordLength)
896 return NPERR_GENERIC_ERROR;
897
898 ProtectionSpace protectionSpace;
899 if (!initializeProtectionSpace(protocol, host, port, scheme, realm, protectionSpace))
900 return NPERR_GENERIC_ERROR;
901
902 RefPtr<NetscapePlugin> plugin = NetscapePlugin::fromNPP(npp);
903 String usernameString;
904 String passwordString;
905 if (!plugin->getAuthenticationInfo(protectionSpace, usernameString, passwordString))
906 return NPERR_GENERIC_ERROR;
907
908 NPError result = copyCString(usernameString.utf8(), username, usernameLength);
909 if (result != NPERR_NO_ERROR)
910 return result;
911
912 result = copyCString(passwordString.utf8(), password, passwordLength);
913 if (result != NPERR_NO_ERROR) {
914 npnMemFree(*username);
915 return result;
916 }
917
918 return NPERR_NO_ERROR;
919}
