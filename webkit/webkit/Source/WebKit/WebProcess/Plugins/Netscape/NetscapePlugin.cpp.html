121void NetscapePlugin::invalidate(const NPRect* invalidRect)
122{

124

127 else

163void NetscapePlugin::loadURL(const String& method, const String& urlString, const String& target, const HTTPHeaderMap& headerFields, const Vector<uint8_t>& httpBody,
164 bool sendNotification, void* notificationData)
165{
166 uint64_t requestID = ++m_nextRequestID;
167
168 controller()->loadURL(requestID, method, urlString, target, headerFields, httpBody, allowPopups());
169
170 if (target.isNull()) {
171 // The browser is going to send the data in a stream, create a plug-in stream.
172 auto pluginStream = NetscapePluginStream::create(*this, requestID, urlString, sendNotification, notificationData);
173 ASSERT(!m_streams.contains(requestID));
174
175 m_streams.set(requestID, WTFMove(pluginStream));
176 return;
177 }
178
179 if (sendNotification) {
180 // Eventually we are going to get a frameDidFinishLoading or frameDidFail call for this request.
181 // Keep track of the notification data so we can call NPP_URLNotify.
182 ASSERT(!m_pendingURLNotifications.contains(requestID));
183 m_pendingURLNotifications.set(requestID, std::make_pair(urlString, notificationData));
184 }
185}

187NPError NetscapePlugin::destroyStream(NPStream* stream, NPReason reason)
188{
189 NetscapePluginStream* pluginStream = 0;
190
191 for (StreamsMap::const_iterator it = m_streams.begin(), end = m_streams.end(); it != end; ++it) {
192 if (it->value->npStream() == stream) {
193 pluginStream = it->value.get();
194 break;
195 }
196 }
197
198 if (!pluginStream)
199 return NPERR_INVALID_INSTANCE_ERROR;
200
201 return pluginStream->destroy(reason);
202}

350uint32_t NetscapePlugin::scheduleTimer(unsigned interval, bool repeat, void (*timerFunc)(NPP, unsigned timerID))
351{
352 if (!timerFunc)
353 return 0;
354
355 // FIXME: Handle wrapping around.
356 unsigned timerID = ++m_nextTimerID;
357
358 auto timer = makeUnique<Timer>(this, timerID, interval, repeat, timerFunc);
359
360 // FIXME: Based on the plug-in visibility, figure out if we should throttle the timer, or if we should start it at all.
361 timer->start();
362 m_timers.set(timerID, WTFMove(timer));
363
364 return timerID;
365}
366
367void NetscapePlugin::unscheduleTimer(unsigned timerID)
368{
369 if (auto timer = m_timers.take(timerID))
370 timer->stop();
371}

398void NetscapePlugin::registerRedirect(NetscapePluginStream* stream, const URL& requestURL, int redirectResponseStatus, void* notificationData)
399{
400 // NPP_URLRedirectNotify may synchronously request this stream back out, so set it first
401 m_redirects.set(notificationData, std::make_pair(stream, requestURL.string()));
402 if (!NPP_URLRedirectNotify(requestURL.string().utf8().data(), redirectResponseStatus, notificationData)) {
403 m_redirects.take(notificationData);
404 controller()->continueStreamLoad(stream->streamID());
405 }
406}
407
408void NetscapePlugin::urlRedirectResponse(void* notifyData, bool allow)
409{
410 if (!m_redirects.contains(notifyData))
411 return;
412
413 auto redirect = m_redirects.take(notifyData);
414 if (!redirect.first)
415 return;
416
417 RefPtr<NetscapePluginStream> stream = redirect.first;
418 if (!allow) {
419 controller()->cancelStreamLoad(stream->streamID());
420 stream->stop(NPRES_USER_BREAK);
421 } else {
422 stream->setURL(redirect.second);
423 controller()->continueStreamLoad(stream->streamID());
424 }
425}

507void NetscapePlugin::callSetWindow()
508{
509 if (wantsPluginRelativeNPWindowCoordinates()) {
510 m_npWindow.x = 0;
511 m_npWindow.y = 0;
512 m_npWindow.clipRect.top = m_clipRect.y();
513 m_npWindow.clipRect.left = m_clipRect.x();
514 } else {
515 IntPoint pluginLocationInRootViewCoordinates = convertToRootView(IntPoint());
516 IntPoint clipRectInRootViewCoordinates = convertToRootView(m_clipRect.location());
517
518 m_npWindow.x = pluginLocationInRootViewCoordinates.x();
519 m_npWindow.y = pluginLocationInRootViewCoordinates.y();
520 m_npWindow.clipRect.top = clipRectInRootViewCoordinates.y();
521 m_npWindow.clipRect.left = clipRectInRootViewCoordinates.x();
522 }
523
524 m_npWindow.width = m_pluginSize.width();
525 m_npWindow.height = m_pluginSize.height();
526 m_npWindow.clipRect.right = m_npWindow.clipRect.left + m_clipRect.width();
527 m_npWindow.clipRect.bottom = m_npWindow.clipRect.top + m_clipRect.height();
528
529 NPP_SetWindow(&m_npWindow);
530 m_hasCalledSetWindow = true;
531}

547bool NetscapePlugin::shouldLoadSrcURL()
548{
549#if PLATFORM(X11)
550 // Flash crashes when NPP_GetValue is called for NPPVpluginCancelSrcStream in windowed mode.
551 if (m_isWindowed && m_pluginModule->pluginQuirks().contains(PluginQuirks::DoNotCancelSrcStreamInWindowedMode))
552 return true;
553#endif
554
555 // Check if we should cancel the load
556 NPBool cancelSrcStream = false;
557
558 if (NPP_GetValue(NPPVpluginCancelSrcStream, &cancelSrcStream) != NPERR_NO_ERROR)
559 return true;
560
561 return !cancelSrcStream;
562}

575bool NetscapePlugin::allowPopups() const
576{
577 if (m_pluginModule->pluginFuncs().version >= NPVERS_HAS_POPUPS_ENABLED_STATE) {
578 if (!m_popupEnabledStates.isEmpty())
579 return m_popupEnabledStates.last();
580 }
581

584 // support NPN_PushPopupEnabledState/NPN_PopPopupEnabledState.
585 return false;
586}

588#if PLUGIN_ARCHITECTURE(MAC)
589static bool isTransparentSilverlightBackgroundValue(const String& lowercaseBackgroundValue)
590{
591 // This checks if the background color value is transparent, according to
592 // the format documented at http://msdn.microsoft.com/en-us/library/cc838148(VS.95).aspx
593 if (lowercaseBackgroundValue.startsWith('#')) {
594 if (lowercaseBackgroundValue.length() == 5 && lowercaseBackgroundValue[1] != 'f') {
595 // An 8-bit RGB value with alpha transparency, in the form #ARGB.
596 return true;
597 }
598
599 if (lowercaseBackgroundValue.length() == 9 && !(lowercaseBackgroundValue[1] == 'f' && lowercaseBackgroundValue[2] == 'f')) {
600 // A 16-bit RGB value with alpha transparency, in the form #AARRGGBB.
601 return true;
602 }
603 } else if (lowercaseBackgroundValue.startsWith("sc#")) {
604 Vector<String> components = lowercaseBackgroundValue.substring(3).split(',');
605
606 // An ScRGB value with alpha transparency, in the form sc#A,R,G,B.
607 if (components.size() == 4) {
608 if (components[0].toDouble() < 1)
609 return true;
610 }
611 } else if (lowercaseBackgroundValue == "transparent")
612 return true;
613
614 // This is an opaque color.
615 return false;
616}
617#endif

619bool NetscapePlugin::initialize(const Parameters& parameters)
620{
621 uint16_t mode = parameters.isFullFramePlugin ? NP_FULL : NP_EMBED;
622
623 m_shouldUseManualLoader = parameters.shouldUseManualLoader;
624
625 CString mimeTypeCString = parameters.mimeType.utf8();
626
627 ASSERT(parameters.names.size() == parameters.values.size());
628
629 Vector<CString> paramNames;
630 Vector<CString> paramValues;
631 for (size_t i = 0; i < parameters.names.size(); ++i) {
632 String parameterName = parameters.names[i];
633
634#if PLUGIN_ARCHITECTURE(MAC)
635 if (m_pluginModule->pluginQuirks().contains(PluginQuirks::WantsLowercaseParameterNames))
636 parameterName = parameterName.convertToASCIILowercase();
637#endif
638
639 paramNames.append(parameterName.utf8());
640 paramValues.append(parameters.values[i].utf8());
641 }
642
643#if PLATFORM(X11)
644 if (equalLettersIgnoringASCIICase(parameters.mimeType, "application/x-shockwave-flash")) {
645 size_t wmodeIndex = parameters.names.find("wmode");
646 if (wmodeIndex != notFound) {
647 // Transparent window mode is not supported by X11 backend.
648 if (equalLettersIgnoringASCIICase(parameters.values[wmodeIndex], "transparent")
649 || (m_pluginModule->pluginQuirks().contains(PluginQuirks::ForceFlashWindowlessMode) && equalLettersIgnoringASCIICase(parameters.values[wmodeIndex], "window")))
650 paramValues[wmodeIndex] = "opaque";
651 } else if (m_pluginModule->pluginQuirks().contains(PluginQuirks::ForceFlashWindowlessMode)) {
652 paramNames.append("wmode");
653 paramValues.append("opaque");
654 }
655 }
656#endif
657
658 if (equalLettersIgnoringASCIICase(parameters.mimeType, "application/x-webkit-test-netscape")) {
659 paramNames.append("windowedPlugin");
660 paramValues.append("false");
661 }
662
663 // The strings that these pointers point to are kept alive by paramNames and paramValues.
664 Vector<const char*> names;
665 Vector<const char*> values;
666 for (size_t i = 0; i < paramNames.size(); ++i) {
667 names.append(paramNames[i].data());
668 values.append(paramValues[i].data());
669 }
670
671#if PLUGIN_ARCHITECTURE(MAC)
672 if (m_pluginModule->pluginQuirks().contains(PluginQuirks::MakeOpaqueUnlessTransparentSilverlightBackgroundAttributeExists)) {
673 for (size_t i = 0; i < parameters.names.size(); ++i) {
674 if (equalLettersIgnoringASCIICase(parameters.names[i], "background")) {
675 setIsTransparent(isTransparentSilverlightBackgroundValue(parameters.values[i].convertToASCIILowercase()));
676 break;
677 }
678 }
679 }
680
681 m_layerHostingMode = parameters.layerHostingMode;
682#endif
683
684 platformPreInitialize();
685
686 NetscapePlugin* previousNPPNewPlugin = currentNPPNewPlugin;
687
688 m_inNPPNew = true;
689 currentNPPNewPlugin = this;
690
691 NPError error = NPP_New(const_cast<char*>(mimeTypeCString.data()), mode, names.size(),
692 const_cast<char**>(names.data()), const_cast<char**>(values.data()), 0);
693
694 m_inNPPNew = false;
695 currentNPPNewPlugin = previousNPPNewPlugin;
696
697 if (error != NPERR_NO_ERROR)
698 return false;
699
700 m_isStarted = true;
701
702 // FIXME: This is not correct in all cases.
703 m_npWindow.type = NPWindowTypeDrawable;
704
705 if (!platformPostInitialize()) {
706 destroy();
707 return false;
708 }
709
710 // Load the src URL if needed.
711 if (!parameters.shouldUseManualLoader && !parameters.url.isEmpty() && shouldLoadSrcURL())
712 loadURL("GET", parameters.url.string(), String(), HTTPHeaderMap(), Vector<uint8_t>(), false, 0);
713
714 return true;
715}
716
717void NetscapePlugin::destroy()
718{
719 ASSERT(m_isStarted);
720
721 // Stop all streams.
722 stopAllStreams();
723
724#if !PLUGIN_ARCHITECTURE(MAC) && !PLUGIN_ARCHITECTURE(UNIX)
725 m_npWindow.window = 0;
726 callSetWindow();
727#endif
728
729 NPP_Destroy(0);
730
731 m_isStarted = false;
732
733 platformDestroy();
734

745RefPtr<ShareableBitmap> NetscapePlugin::snapshot()
746{
747 if (!supportsSnapshotting() || m_pluginSize.isEmpty())
748 return nullptr;
749
750 ASSERT(m_isStarted);
751
752 IntSize backingStoreSize = m_pluginSize;
753 backingStoreSize.scale(contentsScaleFactor());
754
755 auto bitmap = ShareableBitmap::createShareable(backingStoreSize, { });
756 auto context = bitmap->createGraphicsContext();
757 if (!context)
758 return nullptr;
759
760 // FIXME: We should really call applyDeviceScaleFactor instead of scale, but that ends up calling into WKSI
761 // which we currently don't have initiated in the plug-in process.
762 context->scale(contentsScaleFactor());
763
764 platformPaint(*context, IntRect(IntPoint(), m_pluginSize), true);
765
766 return bitmap;
767}

779void NetscapePlugin::geometryDidChange(const IntSize& pluginSize, const IntRect& clipRect, const AffineTransform& pluginToRootViewTransform)
780{
781 ASSERT(m_isStarted);
782
783 if (pluginSize == m_pluginSize && m_clipRect == clipRect && m_pluginToRootViewTransform == pluginToRootViewTransform) {
784 // Nothing to do.
785 return;
786 }
787
788 bool shouldCallSetWindow = true;
789
790 // If the plug-in doesn't want window relative coordinates, we don't need to call setWindow unless its size or clip rect changes.
791 if (m_hasCalledSetWindow && wantsPluginRelativeNPWindowCoordinates() && m_pluginSize == pluginSize && m_clipRect == clipRect)
792 shouldCallSetWindow = false;
793
794 m_pluginSize = pluginSize;
795 m_clipRect = clipRect;
796 m_pluginToRootViewTransform = pluginToRootViewTransform;
797
798#if PLUGIN_ARCHITECTURE(UNIX)
799 IntPoint frameRectLocationInWindowCoordinates = m_pluginToRootViewTransform.mapPoint(IntPoint());
800 m_frameRectInWindowCoordinates = IntRect(frameRectLocationInWindowCoordinates, m_pluginSize);
801#endif
802
803 platformGeometryDidChange();
804
805 if (!shouldCallSetWindow)
806 return;
807
808 callSetWindow();
809}

1059void NetscapePlugin::updateNPNPrivateMode()
1060{
1061 ASSERT(m_isStarted);
1062
1063 // From https://wiki.mozilla.org/Plugins:PrivateMode
1064 // When the browser turns private mode on or off it will call NPP_SetValue for "NPNVprivateModeBool"
1065 // (assigned enum value 18) with a pointer to an NPBool value on all applicable instances.
1066 // Plugins should check the boolean value pointed to, not the pointer itself.
1067 // The value will be true when private mode is on.
1068 NPBool value = m_privateBrowsingState || m_storageBlockingState;
1069 NPP_SetValue(NPNVprivateModeBool, &value);
