87bool NetscapePlugin::getScreenTransform(NPCoordinateSpace sourceSpace, AffineTransform& transform)
88{
89 ASSERT(transform.isIdentity());
90
91 switch (sourceSpace) {
92 case NPCoordinateSpacePlugin: {
93 transform.translate(m_windowFrameInScreenCoordinates.x(), m_windowFrameInScreenCoordinates.y());
94 transform.translate(m_viewFrameInWindowCoordinates.x(), m_viewFrameInWindowCoordinates.height() + m_viewFrameInWindowCoordinates.y());
95 transform.flipY();
96 transform *= m_pluginToRootViewTransform;
97 return true;
98 }
99
100 case NPCoordinateSpaceWindow: {
101 transform.translate(m_windowFrameInScreenCoordinates.x(), m_windowFrameInScreenCoordinates.y());
102 return true;
103 }
104
105 case NPCoordinateSpaceFlippedWindow: {
106 transform.translate(m_windowFrameInScreenCoordinates.x(), m_windowFrameInScreenCoordinates.height() + m_windowFrameInScreenCoordinates.y());
107 transform.flipY();
108 return true;
109 }
110
111 case NPCoordinateSpaceScreen: {
112 // Nothing to do.
113 return true;
114 }
115
116 case NPCoordinateSpaceFlippedScreen: {
117 double screenHeight = [(NSScreen *)[[NSScreen screens] objectAtIndex:0] frame].size.height;
118 transform.translate(0, screenHeight);
119 transform.flipY();
120 return true;
121 }
122
123 default:
124 return false;
125 }
126}

128NPBool NetscapePlugin::convertPoint(double sourceX, double sourceY, NPCoordinateSpace sourceSpace, double& destX, double& destY, NPCoordinateSpace destSpace)
129{
130 AffineTransform sourceTransform;
131 if (!getScreenTransform(sourceSpace, sourceTransform))
132 return false;
133
134 AffineTransform destTransform;
135 if (!getScreenTransform(destSpace, destTransform))
136 return false;
137
138 if (auto inverse = destTransform.inverse()) {
139 AffineTransform transform = inverse.value() * sourceTransform;
140
141 FloatPoint destinationPoint = transform.mapPoint(FloatPoint(sourceX, sourceY));
142
143 destX = destinationPoint.x();
144 destY = destinationPoint.y();
145 return true;
146 }
147 return false;
148}

182{
183 if (m_pluginModule->pluginQuirks().contains(PluginQuirks::LeakAllThrownNSExceptions)) {
184 // Patch -[NSException release] to not release the object.
185 static dispatch_once_t once;
186 dispatch_once(&once, ^{
187 auto exceptionClass = [NSException class];
188 auto releaseSelector = sel_registerName("release");
189 auto exceptionReleaseMethod = class_getInstanceMethod(exceptionClass, releaseSelector);
190 class_replaceMethod(exceptionClass, releaseSelector, reinterpret_cast<IMP>(NSException_release), method_getTypeEncoding(exceptionReleaseMethod));
191 });
192 }

248void NetscapePlugin::platformPaint(GraphicsContext& context, const IntRect& dirtyRect, bool isSnapshot)
249{
250 CGContextRef platformContext = context.platformContext();
251
252 switch (m_eventModel) {
253 case NPEventModelCocoa: {
254 // Don't send draw events when we're using the Core Animation drawing model.
255 if (!isSnapshot && m_drawingModel == NPDrawingModelCoreAnimation)
256 return;
257
258 NPCocoaEvent event = initializeEvent(NPCocoaEventDrawRect);
259
260 event.data.draw.context = platformContext;
261 event.data.draw.x = dirtyRect.x();
262 event.data.draw.y = dirtyRect.y();
263 event.data.draw.width = dirtyRect.width();
264 event.data.draw.height = dirtyRect.height();
265
266 NPP_HandleEvent(&event);
267 break;
268 }
269
270 default:
271 ASSERT_NOT_REACHED();
272 }
273}

275static uint32_t modifierFlags(const WebEvent& event)
276{
277 uint32_t modifiers = 0;
278
279 if (event.shiftKey())
280 modifiers |= NSEventModifierFlagShift;
281 if (event.controlKey())
282 modifiers |= NSEventModifierFlagControl;
283 if (event.altKey())
284 modifiers |= NSEventModifierFlagOption;
285 if (event.metaKey())
286 modifiers |= NSEventModifierFlagCommand;
287
288 return modifiers;
289}

307static void fillInCocoaEventFromMouseEvent(NPCocoaEvent& event, const WebMouseEvent& mouseEvent, const WebCore::IntPoint& eventPositionInPluginCoordinates)
308{
309 event.data.mouse.modifierFlags = modifierFlags(mouseEvent);
310 event.data.mouse.pluginX = eventPositionInPluginCoordinates.x();
311 event.data.mouse.pluginY = eventPositionInPluginCoordinates.y();
312 event.data.mouse.buttonNumber = buttonNumber(mouseEvent.button());
313 event.data.mouse.clickCount = mouseEvent.clickCount();
314 event.data.mouse.deltaX = mouseEvent.deltaX();
315 event.data.mouse.deltaY = mouseEvent.deltaY();
316 event.data.mouse.deltaZ = mouseEvent.deltaZ();
317}

319static NPCocoaEvent initializeMouseEvent(const WebMouseEvent& mouseEvent, const WebCore::IntPoint& eventPositionInPluginCoordinates)
320{
321 NPCocoaEventType eventType;
322
323 switch (mouseEvent.type()) {
324 case WebEvent::MouseDown:
325 eventType = NPCocoaEventMouseDown;
326 break;
327 case WebEvent::MouseUp:
328 eventType = NPCocoaEventMouseUp;
329 break;
330 case WebEvent::MouseMove:
331 if (mouseEvent.button() == WebMouseEvent::NoButton)
332 eventType = NPCocoaEventMouseMoved;
333 else
334 eventType = NPCocoaEventMouseDragged;
335 break;
336 default:
337 ASSERT_NOT_REACHED();
338 return NPCocoaEvent();
339 }
340
341 NPCocoaEvent event = initializeEvent(eventType);
342 fillInCocoaEventFromMouseEvent(event, mouseEvent, eventPositionInPluginCoordinates);
343 return event;
344}

346bool NetscapePlugin::platformHandleMouseEvent(const WebMouseEvent& mouseEvent)
347{
348 IntPoint eventPositionInPluginCoordinates;
349 if (!convertFromRootView(mouseEvent.position(), eventPositionInPluginCoordinates))
350 return true;
351
352 switch (m_eventModel) {
353 case NPEventModelCocoa: {
354 NPCocoaEvent event = initializeMouseEvent(mouseEvent, eventPositionInPluginCoordinates);
355
356 NPCocoaEvent* previousMouseEvent = m_currentMouseEvent;
357 m_currentMouseEvent = &event;
358
359 // Protect against NPP_HandleEvent causing the plug-in to be destroyed, since we
360 // access m_currentMouseEvent afterwards.
361 Ref<NetscapePlugin> protect(*this);
362
363 NPP_HandleEvent(&event);
364
365 m_currentMouseEvent = previousMouseEvent;
366
367 // Some plug-ins return false even if the mouse event has been handled.
368 // This leads to bugs such as <rdar://problem/9167611>. Work around this
369 // by always returning true.
370 return true;
371 }
372
373 default:
374 ASSERT_NOT_REACHED();
375 }
376
377 return false;
378}

487static NPCocoaEvent initializeKeyboardEvent(const WebKeyboardEvent& keyboardEvent)
488{
489 NPCocoaEventType eventType;
490
491 if (isFlagsChangedEvent(keyboardEvent))
492 eventType = NPCocoaEventFlagsChanged;
493 else {
494 switch (keyboardEvent.type()) {
495 case WebEvent::KeyDown:
496 eventType = NPCocoaEventKeyDown;
497 break;
498 case WebEvent::KeyUp:
499 eventType = NPCocoaEventKeyUp;
500 break;
501 default:
502 ASSERT_NOT_REACHED();
503 return NPCocoaEvent();
504 }
505 }
506
507 NPCocoaEvent event = initializeEvent(eventType);
508 event.data.key.modifierFlags = modifierFlags(keyboardEvent);
509 event.data.key.characters = convertToNPNNString(keyboardEvent.text());
510 event.data.key.charactersIgnoringModifiers = convertToNPNNString(keyboardEvent.unmodifiedText());
511 event.data.key.isARepeat = keyboardEvent.isAutoRepeat();
512 event.data.key.keyCode = keyboardEvent.nativeVirtualKeyCode();
513
514 return event;
515}

517bool NetscapePlugin::platformHandleKeyboardEvent(const WebKeyboardEvent& keyboardEvent)
518{
519 bool handled = false;
520
521 switch (m_eventModel) {
522 case NPEventModelCocoa: {
523 if (keyboardEvent.type() == WebEvent::KeyDown) {
524 m_hasHandledAKeyDownEvent = true;
525
526 if (!m_pluginWantsLegacyCocoaTextInput && m_isComplexTextInputEnabled && !keyboardEvent.isAutoRepeat()) {
527 // When complex text is enabled in the new model, the plug-in should never
528 // receive any key down or key up events until the composition is complete.
529 m_ignoreNextKeyUpEventCounter++;
530 return true;
531 }
532 } else if (keyboardEvent.type() == WebEvent::KeyUp && m_ignoreNextKeyUpEventCounter) {
533 m_ignoreNextKeyUpEventCounter--;
534 return true;
535 }
536
537 NPCocoaEvent event = initializeKeyboardEvent(keyboardEvent);
538 int16_t returnValue = NPP_HandleEvent(&event);
539 handled = returnValue;
540
541 if (!m_pluginWantsLegacyCocoaTextInput) {
542 if (event.type == NPCocoaEventKeyDown && returnValue == kNPEventStartIME) {
543 if (!keyboardEvent.isAutoRepeat())
544 m_ignoreNextKeyUpEventCounter++;
545 setComplexTextInputEnabled(true);
546 }
547 }
548
549 break;
550 }
551
552 default:
553 ASSERT_NOT_REACHED();
554 }
555
556 // Most plug-ins simply return true for all keyboard events, even those that aren't handled.
557 // This leads to bugs such as <rdar://problem/8740926>. We work around this by returning false
558 // if the keyboard event has the command modifier pressed.
559 // However, for command-A (the shortcurt for 'Select All' we will always return true, since we don't
560 // want the entire page to be selected if the focus is in a plug-in text field (see <rdar://problem/9309903>).
561 if (keyboardEvent.metaKey()) {
562 if (keyboardEvent.text() == "a")
563 return true;
564
565 return false;
566 }
567
568 return handled;
569}

643void NetscapePlugin::sendComplexTextInput(const String& textInput)
644{
645 if (!m_pluginWantsLegacyCocoaTextInput) {
646 // In the updated Cocoa text input spec, text input is disabled when the text input string has been sent
647 // by the UI process. Since the UI process has also updated its state, we can just reset the variable here
648 // instead of calling setComplexTextInputEnabled.
649 m_isComplexTextInputEnabled = false;
650
651 // The UI process can also disable text input by sending an empty input string. In this case, we don't want
652 // to send it to the plug-in.
653 if (textInput.isNull())
654 return;
655 }
656
657 switch (m_eventModel) {
658 case NPEventModelCocoa: {
659 NPCocoaEvent event = initializeEvent(NPCocoaEventTextInput);
660 event.data.text.text = convertToNPNNString(textInput);
661 NPP_HandleEvent(&event);
662 break;
663 }
664 default:
665 ASSERT_NOT_REACHED();
666 }
667}

682void NetscapePlugin::pluginFocusOrWindowFocusChanged()
683{
684 bool pluginHasFocusAndWindowHasFocus = m_pluginHasFocus && m_windowHasFocus;
685
686 controller()->pluginFocusOrWindowFocusChanged(pluginHasFocusAndWindowHasFocus);
687
688 // In the updated Cocoa text input spec, the plug-in will enable complex text input
689 // by returning kNPEventStartIME from it's NPCocoaEventKeyDown handler.
690 if (!m_pluginWantsLegacyCocoaTextInput)
691 return;
692
693 // In the old model, if the plug-in is focused, enable complex text input.
694 setComplexTextInputEnabled(pluginHasFocusAndWindowHasFocus);
695}

698{

707
708 controller()->setComplexTextInputState(complexTextInputState);
709}

716static void makeCGLPresentLayerOpaque(CALayer *pluginRootLayer)
717{
718 // We look for a layer that's the only sublayer of the root layer that is an instance
719 // of the CGLPresentLayer class which in turn is a subclass of CAOpenGLLayer and make
720 // it opaque if all these conditions hold.
721
722 NSArray *sublayers = [pluginRootLayer sublayers];
723 if ([sublayers count] != 1)
724 return;
725
726 Class cglPresentLayerClass = NSClassFromString(@"CGLPresentLayer");
727 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
728 if (![cglPresentLayerClass isSubclassOfClass:[CAOpenGLLayer class]])
729 return;
730 ALLOW_DEPRECATED_DECLARATIONS_END
731
732 CALayer *layer = [sublayers objectAtIndex:0];
733 if (![layer isKindOfClass:cglPresentLayerClass])
734 return;
735
736 [layer setOpaque:YES];
737}

739void NetscapePlugin::updatePluginLayer()
740{
741 if (m_drawingModel != NPDrawingModelCoreAnimation)
742 return;
743
744 void* value = 0;
745
746 // Get the Core Animation layer.
747 if (NPP_GetValue(NPPVpluginCoreAnimationLayer, &value) != NPERR_NO_ERROR)
748 return;
749
750 if (!value)
751 return;
752
753 ASSERT(!m_pluginLayer);
754
755 // The original Core Animation drawing model required that plug-ins pass a retained layer
756 // to the browser, which the browser would then adopt. However, the final spec changed this
757 // (See https://wiki.mozilla.org/NPAPI:CoreAnimationDrawingModel for more information)
758 // after a version of WebKit1 with the original implementation had shipped, but that now means
759 // that any plug-ins that expect the WebKit1 behavior would leak the CALayer.
760 // For plug-ins that we know return retained layers, we have the ReturnsRetainedCoreAnimationLayer
761 // plug-in quirk. Plug-ins can also check for whether the browser expects a non-retained layer to
762 // be returned by using NPN_GetValue and pass the WKNVExpectsNonretainedLayer parameter.
763 // https://bugs.webkit.org/show_bug.cgi?id=58282 describes the bug where WebKit expects retained layers.
764 if (m_pluginReturnsNonretainedLayer)
765 m_pluginLayer = (__bridge CALayer *)value;
766 else
767 m_pluginLayer = adoptNS((__bridge CALayer *)value);
768
769 if (m_pluginModule->pluginQuirks().contains(PluginQuirks::MakeOpaqueUnlessTransparentSilverlightBackgroundAttributeExists) &&
770 !m_isTransparent)
771 makeCGLPresentLayerOpaque(m_pluginLayer.get());
772}
