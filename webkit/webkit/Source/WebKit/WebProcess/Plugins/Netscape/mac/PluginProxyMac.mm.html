41static void makeRenderLayer(CALayer *pluginLayer, uint32_t layerHostingContextID)
42{
43 CALayer *renderLayer = [CALayer _web_renderLayerWithContextID:layerHostingContextID];
44 [renderLayer setFrame:[pluginLayer bounds]];
45 [renderLayer setAutoresizingMask:kCALayerWidthSizable | kCALayerHeightSizable];
46 [pluginLayer setSublayers:@[renderLayer]];
47}
48
49PlatformLayer* PluginProxy::pluginLayer()
50{
51 if (!m_pluginLayer && m_remoteLayerClientID) {
52 // Create a layer with flipped geometry and add the real plug-in layer as a sublayer
53 // so the coordinate system will match the event coordinate system.
54 m_pluginLayer = adoptNS([[CALayer alloc] init]);
55 [m_pluginLayer setGeometryFlipped:YES];
56 makeRenderLayer(m_pluginLayer.get(), m_remoteLayerClientID);
57 }
58
59 return m_pluginLayer.get();
60}

82void PluginProxy::setLayerHostingContextID(uint32_t layerHostingContextID)
83{
84 ASSERT(m_pluginLayer.get());
85
86 m_remoteLayerClientID = layerHostingContextID;
87 makeRenderLayer(m_pluginLayer.get(), m_remoteLayerClientID);
88}
