53static const String cssAlignmentValueForNSTextAlignment(NSTextAlignment alignment)
54{
55 switch (alignment) {
56 case NSTextAlignmentLeft:
57 return "left";
58 case NSTextAlignmentRight:
59 return "right";
60 case NSTextAlignmentCenter:
61 return "center";
62 case NSTextAlignmentJustified:
63 return "justify";
64 case NSTextAlignmentNatural:
65 return "-webkit-start";
66 }
67 ASSERT_NOT_REACHED();
68 return String();
69}

81Ref<Element> PDFPluginTextAnnotation::createAnnotationElement()
82{
83 Document& document = parent()->document();
84 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
85 PDFAnnotationTextWidget *textAnnotation = this->textAnnotation();
86 ALLOW_DEPRECATED_DECLARATIONS_END
87 bool isMultiline = textAnnotation.isMultiline;
88
89 auto element = document.createElement(isMultiline ? textareaTag : inputTag, false);
90 element->addEventListener(eventNames().keydownEvent, *eventListener(), false);
91
92 auto& styledElement = downcast<StyledElement>(element.get());
93
94 if (!textAnnotation)
95 return element;
96
97 // FIXME: Match font weight and style as well?
98 styledElement.setInlineStyleProperty(CSSPropertyColor, serializationForHTML(colorFromNSColor(textAnnotation.fontColor)));
99 styledElement.setInlineStyleProperty(CSSPropertyFontFamily, textAnnotation.font.familyName);
100 styledElement.setInlineStyleProperty(CSSPropertyTextAlign, cssAlignmentValueForNSTextAlignment(textAnnotation.alignment));
101
102 if (isMultiline)
103 downcast<HTMLTextAreaElement>(styledElement).setValue(textAnnotation.stringValue);
104 else
105 downcast<HTMLInputElement>(styledElement).setValue(textAnnotation.stringValue);
106
107 return element;
108}

110void PDFPluginTextAnnotation::updateGeometry()
111{
112 PDFPluginAnnotation::updateGeometry();
113
114 StyledElement* styledElement = static_cast<StyledElement*>(element());
115 styledElement->setInlineStyleProperty(CSSPropertyFontSize, textAnnotation().font.pointSize * pdfLayerController().contentScaleFactor, CSSUnitType::CSS_PX);
116}

129bool PDFPluginTextAnnotation::handleEvent(Event& event)
130{
131 if (PDFPluginAnnotation::handleEvent(event))
132 return true;
133
134 if (event.isKeyboardEvent() && event.type() == eventNames().keydownEvent) {
135 auto& keyboardEvent = downcast<KeyboardEvent>(event);
136
137 if (keyboardEvent.keyIdentifier() == "U+0009") {
138 if (keyboardEvent.ctrlKey() || keyboardEvent.metaKey() || keyboardEvent.altGraphKey())
139 return false;
140
141 if (keyboardEvent.shiftKey())
142 plugin()->focusPreviousAnnotation();
143 else
144 plugin()->focusNextAnnotation();
145
146 event.preventDefault();
147 return true;
148 }
149 }
150
151 return false;
152}
