42void Plugin::Parameters::encode(IPC::Encoder& encoder) const
43{
44 encoder << url.string();
45 encoder << names;
46 encoder << values;
47 encoder << mimeType;
48 encoder << isFullFramePlugin;
49 encoder << shouldUseManualLoader;
50#if PLATFORM(COCOA)
51 encoder << layerHostingMode;
52#endif
53}
54
55bool Plugin::Parameters::decode(IPC::Decoder& decoder, Parameters& parameters)
56{
57 String urlString;
58 if (!decoder.decode(urlString))
59 return false;
60 parameters.url = URL({ }, urlString);
61
62 if (!decoder.decode(parameters.names))
63 return false;
64 if (!decoder.decode(parameters.values))
65 return false;
66 if (!decoder.decode(parameters.mimeType))
67 return false;
68 if (!decoder.decode(parameters.isFullFramePlugin))
69 return false;
70 if (!decoder.decode(parameters.shouldUseManualLoader))
71 return false;
72#if PLATFORM(COCOA)
73 if (!decoder.decode(parameters.layerHostingMode))
74 return false;
75#endif
76 if (parameters.names.size() != parameters.values.size())
77 return false;
78
79 return true;
80}

89bool Plugin::initialize(PluginController& pluginController, const Parameters& parameters)
90{
91 ASSERT(!m_pluginController);
92
93 m_pluginController = makeWeakPtr(pluginController);
94
95 return initialize(parameters);
96}
97
98void Plugin::destroyPlugin()
99{
100 ASSERT(!m_isBeingDestroyed);
101 SetForScope<bool> scope { m_isBeingDestroyed, true };
102
103 destroy();
104
105 m_pluginController = nullptr;
106}
