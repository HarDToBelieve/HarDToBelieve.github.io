80bool PluginProxy::initialize(const Parameters& parameters)
81{
82 ASSERT(!m_connection);
83 m_connection = WebProcess::singleton().pluginProcessConnectionManager().getPluginProcessConnection(m_pluginProcessToken);
84
85 if (!m_connection)
86 return false;
87
88 // Add the plug-in proxy before creating the plug-in; it needs to be in the map because CreatePlugin
89 // can call back out to the plug-in proxy.
90 m_connection->addPluginProxy(this);
91
92 // Ask the plug-in process to create a plug-in.
93 m_pendingPluginCreationParameters = makeUnique<PluginCreationParameters>();
94
95 m_pendingPluginCreationParameters->pluginInstanceID = m_pluginInstanceID;
96 m_pendingPluginCreationParameters->windowNPObjectID = windowNPObjectID();
97 m_pendingPluginCreationParameters->parameters = parameters;
98 m_pendingPluginCreationParameters->userAgent = controller()->userAgent();
99 m_pendingPluginCreationParameters->contentsScaleFactor = contentsScaleFactor();
100 m_pendingPluginCreationParameters->isPrivateBrowsingEnabled = controller()->isPrivateBrowsingEnabled();
101 m_pendingPluginCreationParameters->isMuted = controller()->isMuted();
102 m_pendingPluginCreationParameters->artificialPluginInitializationDelayEnabled = controller()->artificialPluginInitializationDelayEnabled();
103 m_pendingPluginCreationParameters->isAcceleratedCompositingEnabled = controller()->isAcceleratedCompositingEnabled();
104
105 if (!canInitializeAsynchronously())
106 return initializeSynchronously();
107
108 // Remember that we tried to create this plug-in asynchronously in case we need to create it synchronously later.
109 m_waitingOnAsynchronousInitialization = true;
110 PluginCreationParameters creationParameters(*m_pendingPluginCreationParameters.get());
111 m_connection->connection()->send(Messages::WebProcessConnection::CreatePluginAsynchronously(creationParameters), m_pluginInstanceID);
112 return true;
113}
114
115bool PluginProxy::canInitializeAsynchronously() const
116{
117 return controller()->asynchronousPluginInitializationEnabled() && (m_connection->supportsAsynchronousPluginInitialization() || controller()->asynchronousPluginInitializationEnabledForAllPlugins());
118}
119
120bool PluginProxy::initializeSynchronously()
121{
122 ASSERT(m_pendingPluginCreationParameters);
123
124 m_pendingPluginCreationParameters->asynchronousCreationIncomplete = m_waitingOnAsynchronousInitialization;
125 bool result = false;
126 bool wantsWheelEvents = false;
127 uint32_t remoteLayerClientID = 0;
128
129 PluginCreationParameters parameters(*m_pendingPluginCreationParameters.get());
130
131 if (!m_connection->connection()->sendSync(Messages::WebProcessConnection::CreatePlugin(parameters), Messages::WebProcessConnection::CreatePlugin::Reply(result, wantsWheelEvents, remoteLayerClientID), 0) || !result)
132 didFailToCreatePluginInternal();
133 else
134 didCreatePluginInternal(wantsWheelEvents, remoteLayerClientID);
135
136 return result;
137}

139void PluginProxy::didCreatePlugin(bool wantsWheelEvents, uint32_t remoteLayerClientID, CompletionHandler<void()>&& completionHandler)
140{
141 // We might have tried to create the plug-in sychronously while waiting on the asynchronous reply,
142 // in which case we should ignore this message.
143 if (!m_waitingOnAsynchronousInitialization)
144 return completionHandler();
145
146 didCreatePluginInternal(wantsWheelEvents, remoteLayerClientID);
147 completionHandler();
148}
149
150void PluginProxy::didFailToCreatePlugin(CompletionHandler<void()>&& completionHandler)
151{
152 // We might have tried to create the plug-in sychronously while waiting on the asynchronous reply,
153 // in which case we should ignore this message.
154 if (!m_waitingOnAsynchronousInitialization)
155 return completionHandler();
156
157 didFailToCreatePluginInternal();
158 completionHandler();
159}
160
161void PluginProxy::didCreatePluginInternal(bool wantsWheelEvents, uint32_t remoteLayerClientID)
162{
163 m_wantsWheelEvents = wantsWheelEvents;
164 m_remoteLayerClientID = remoteLayerClientID;
165 m_isStarted = true;
166 controller()->didInitializePlugin();
167
168 // Whether synchronously or asynchronously, this plug-in was created and we shouldn't need to remember
169 // anything about how.
170 m_pendingPluginCreationParameters = nullptr;
171 m_waitingOnAsynchronousInitialization = false;
172}
173
174void PluginProxy::didFailToCreatePluginInternal()
175{
176 // Calling out to the connection and the controller could potentially cause the plug-in proxy to go away, so protect it here.
177 Ref<PluginProxy> protect(*this);
178
179 m_connection->removePluginProxy(this);
180 controller()->didFailToInitializePlugin();
181
182 // Whether synchronously or asynchronously, this plug-in failed to create and we shouldn't need to remember
183 // anything about how.
184 m_pendingPluginCreationParameters = nullptr;
185 m_waitingOnAsynchronousInitialization = false;
186}

188void PluginProxy::destroy()
189{
190 m_isStarted = false;
191
192 if (!m_connection)
193 return;
194
195 // Although this message is sent synchronously, the Plugin process replies immediately (before performing any tasks) so this is only waiting for
196 // confirmation that the Plugin process received the DestroyPlugin message.
197 m_connection->connection()->sendSync(Messages::WebProcessConnection::DestroyPlugin(m_pluginInstanceID, m_waitingOnAsynchronousInitialization), Messages::WebProcessConnection::DestroyPlugin::Reply(), 0, 1_s);
198 m_connection->removePluginProxy(this);
199}

201void PluginProxy::paint(GraphicsContext& graphicsContext, const IntRect& dirtyRect)
202{
203 if (!needsBackingStore() || !m_backingStore)
204 return;
205
206 if (!m_pluginBackingStoreContainsValidData) {
207 m_connection->connection()->sendSync(Messages::PluginControllerProxy::PaintEntirePlugin(), Messages::PluginControllerProxy::PaintEntirePlugin::Reply(), m_pluginInstanceID);
208
209 // Blit the plug-in backing store into our own backing store.
210 auto graphicsContext = m_backingStore->createGraphicsContext();
211 if (graphicsContext) {
212 graphicsContext->applyDeviceScaleFactor(contentsScaleFactor());
213 graphicsContext->setCompositeOperation(CompositeOperator::Copy);
214
215 m_pluginBackingStore->paint(*graphicsContext, contentsScaleFactor(), IntPoint(), pluginBounds());
216
217 m_pluginBackingStoreContainsValidData = true;
218 }
219 }
220
221 m_backingStore->paint(graphicsContext, contentsScaleFactor(), dirtyRect.location(), dirtyRect);
222
223 if (m_waitingForPaintInResponseToUpdate) {
224 m_waitingForPaintInResponseToUpdate = false;
225 m_connection->connection()->send(Messages::PluginControllerProxy::DidUpdate(), m_pluginInstanceID);
226 }
227}

229bool PluginProxy::supportsSnapshotting() const
230{
231 if (m_waitingOnAsynchronousInitialization)
232 return false;
233
234 bool isSupported = false;
235 if (m_connection && !m_connection->connection()->sendSync(Messages::PluginControllerProxy::SupportsSnapshotting(), Messages::PluginControllerProxy::SupportsSnapshotting::Reply(isSupported), m_pluginInstanceID))
236 return false;
237
238 return isSupported;

264void PluginProxy::geometryDidChange()
265{
266 ASSERT(m_isStarted);
267
268 ShareableBitmap::Handle pluginBackingStoreHandle;
269
270 if (updateBackingStore()) {
271 // Create a new plug-in backing store.
272 m_pluginBackingStore = ShareableBitmap::createShareable(m_backingStore->size(), { });
273 if (!m_pluginBackingStore)
274 return;
275
276 // Create a handle to the plug-in backing store so we can send it over.
277 if (!m_pluginBackingStore->createHandle(pluginBackingStoreHandle)) {
278 m_pluginBackingStore = nullptr;
279 return;
280 }
281
282 m_pluginBackingStoreContainsValidData = false;
283 }
284
285 m_connection->connection()->send(Messages::PluginControllerProxy::GeometryDidChange(m_pluginSize, m_clipRect, m_pluginToRootViewTransform, contentsScaleFactor(), pluginBackingStoreHandle), m_pluginInstanceID, IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
286}

288void PluginProxy::geometryDidChange(const IntSize& pluginSize, const IntRect& clipRect, const AffineTransform& pluginToRootViewTransform)
289{
290 if (pluginSize == m_pluginSize && m_clipRect == clipRect && m_pluginToRootViewTransform == pluginToRootViewTransform) {
291 // Nothing to do.
292 return;
293 }
294
295 m_pluginSize = pluginSize;
296 m_clipRect = clipRect;
297 m_pluginToRootViewTransform = pluginToRootViewTransform;
298
299 geometryDidChange();
300}

484NPObject* PluginProxy::pluginScriptableNPObject()
485{
486 // Sending the synchronous Messages::PluginControllerProxy::GetPluginScriptableNPObject message can cause us to dispatch an
487 // incoming synchronous message that ends up destroying the PluginProxy object.
488 PluginController::PluginDestructionProtector protector(controller());
489
490 uint64_t pluginScriptableNPObjectID = 0;
491
492 if (!m_connection->connection()->sendSync(Messages::PluginControllerProxy::GetPluginScriptableNPObject(), Messages::PluginControllerProxy::GetPluginScriptableNPObject::Reply(pluginScriptableNPObjectID), m_pluginInstanceID))
493 return 0;
494
495 if (!pluginScriptableNPObjectID)
496 return 0;
497
498 return m_connection->npRemoteObjectMap()->createNPObjectProxy(pluginScriptableNPObjectID, this);
499}

604bool PluginProxy::updateBackingStore()
605{
606 if (m_pluginSize.isEmpty() || !needsBackingStore())
607 return false;
608
609 IntSize backingStoreSize = m_pluginSize;
610 backingStoreSize.scale(contentsScaleFactor());
611
612 if (m_backingStore) {
613 if (m_backingStore->size() == backingStoreSize)
614 return false;
615 m_backingStore = nullptr; // Give malloc a chance to recycle our backing store.
616 }
617
618 m_backingStore = ShareableBitmap::create(backingStoreSize, { });
619 return !!m_backingStore;
620}

650void PluginProxy::evaluate(const NPVariantData& npObjectAsVariantData, const String& scriptString, bool allowPopups, CompletionHandler<void(bool returnValue, NPVariantData&& resultData)>&& completionHandler)
651{
652 PluginController::PluginDestructionProtector protector(controller());
653
654 NPVariant npObjectAsVariant = m_connection->npRemoteObjectMap()->npVariantDataToNPVariant(npObjectAsVariantData, this);
655 if (!NPVARIANT_IS_OBJECT(npObjectAsVariant) || !(NPVARIANT_TO_OBJECT(npObjectAsVariant)))
656 return completionHandler(false, { });
657
658 NPVariant result;
659 bool returnValue = controller()->evaluate(NPVARIANT_TO_OBJECT(npObjectAsVariant), scriptString, &result, allowPopups);
660 if (!returnValue)
661 return completionHandler(false, { });
662
663 // Convert the NPVariant to an NPVariantData.
664 NPVariantData resultData = m_connection->npRemoteObjectMap()->npVariantToNPVariantData(result, this);
665
666 // And release the result.
667 releaseNPVariantValue(&result);
668 releaseNPVariantValue(&npObjectAsVariant);
669
670 completionHandler(returnValue, WTFMove(resultData));
671}

715void PluginProxy::update(const IntRect& paintedRect)
716{
717 if (paintedRect == pluginBounds())
718 m_pluginBackingStoreContainsValidData = true;
719
720 if (m_backingStore) {
721 // Blit the plug-in backing store into our own backing store.
722 auto graphicsContext = m_backingStore->createGraphicsContext();
723 if (graphicsContext) {
724 graphicsContext->applyDeviceScaleFactor(contentsScaleFactor());
725 graphicsContext->setCompositeOperation(CompositeOperator::Copy);
726 m_pluginBackingStore->paint(*graphicsContext, contentsScaleFactor(), paintedRect.location(), paintedRect);
727 }
728 }
729
730 // Ask the controller to invalidate the rect for us.
731 m_waitingForPaintInResponseToUpdate = true;
732 controller()->invalidate(paintedRect);
733}
