73Vector<PluginInfo> WebPluginInfoProvider::pluginInfo(Page& page, std::optional<Vector<SupportedPluginIdentifier>>& supportedPluginIdentifiers)
74{
75#if ENABLE(NETSCAPE_PLUGIN_API)
76 populatePluginCache(page);
77
78 if (m_cachedSupportedPluginIdentifiers)
79 supportedPluginIdentifiers = *m_cachedSupportedPluginIdentifiers;
80
81 return page.mainFrame().arePluginsEnabled() ? m_cachedPlugins : m_cachedApplicationPlugins;
82#else
83 UNUSED_PARAM(page);
84 UNUSED_PARAM(supportedPluginIdentifiers);
85 return { };
86#endif // ENABLE(NETSCAPE_PLUGIN_API)
87}
88
89Vector<WebCore::PluginInfo> WebPluginInfoProvider::webVisiblePluginInfo(Page& page, const URL& url)
90{
91 std::optional<Vector<WebCore::SupportedPluginIdentifier>> supportedPluginIdentifiers;
92 auto plugins = pluginInfo(page, supportedPluginIdentifiers);
93
94 plugins.removeAllMatching([&] (auto& plugin) {
95 return supportedPluginIdentifiers && !isSupportedPlugin(*supportedPluginIdentifiers, url, plugin.bundleIdentifier);
96 });
97
98#if PLATFORM(MAC)
99 if (LegacySchemeRegistry::shouldTreatURLSchemeAsLocal(url.protocol().toString()))
100 return plugins;
101
102 for (int32_t i = plugins.size() - 1; i >= 0; --i) {
103 auto& info = plugins.at(i);
104
105 // Allow built-in plugins. Also tentatively allow plugins that the client might later selectively permit.
106 if (info.isApplicationPlugin || info.clientLoadPolicy == WebCore::PluginLoadClientPolicy::Ask)
107 continue;
108
109 if (info.clientLoadPolicy == WebCore::PluginLoadClientPolicy::Block)
110 plugins.remove(i);
111 }
112#endif
113 return plugins;
114}

116#if ENABLE(NETSCAPE_PLUGIN_API)
117void WebPluginInfoProvider::populatePluginCache(const WebCore::Page&)
118{
119 if (!m_pluginCacheIsPopulated) {
120#if PLATFORM(COCOA)
121 // Application plugins are not affected by enablePlugins setting, so we always need to scan plugins to get them.
122 bool shouldScanPlugins = true;
123#else
124 bool shouldScanPlugins = page.mainFrame().arePluginsEnabled();
125#endif
126 if (shouldScanPlugins) {
127 HangDetectionDisabler hangDetectionDisabler;
128 if (!WebProcess::singleton().parentProcessConnection()->sendSync(Messages::WebProcessProxy::GetPlugins(m_shouldRefreshPlugins),
129 Messages::WebProcessProxy::GetPlugins::Reply(m_cachedPlugins, m_cachedApplicationPlugins, m_cachedSupportedPluginIdentifiers), 0))
130 return;
131 }
132
133 m_shouldRefreshPlugins = false;
134 m_pluginCacheIsPopulated = true;
135 }
136}
137#endif
