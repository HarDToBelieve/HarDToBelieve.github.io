54void WebServiceWorkerFetchTaskClient::didReceiveRedirection(const WebCore::ResourceResponse& response)
55{
56 if (!m_connection)
57 return;
58 m_connection->send(Messages::ServiceWorkerFetchTask::DidReceiveRedirectResponse { response }, m_fetchIdentifier);
59
60 cleanup();
61}

63void WebServiceWorkerFetchTaskClient::didReceiveResponse(const ResourceResponse& response)
64{
65 if (!m_connection)
66 return;
67
68 if (m_needsContinueDidReceiveResponseMessage)
69 m_waitingForContinueDidReceiveResponseMessage = true;
70
71 m_connection->send(Messages::ServiceWorkerFetchTask::DidReceiveResponse { response, m_needsContinueDidReceiveResponseMessage }, m_fetchIdentifier);
72}

74void WebServiceWorkerFetchTaskClient::didReceiveData(Ref<SharedBuffer>&& buffer)
75{
76 if (!m_connection)
77 return;
78
79 if (m_waitingForContinueDidReceiveResponseMessage) {
80 if (!WTF::holds_alternative<Ref<SharedBuffer>>(m_responseData))
81 m_responseData = buffer->copy();
82 else
83 WTF::get<Ref<SharedBuffer>>(m_responseData)->append(WTFMove(buffer));
84 return;
85 }
86
87 m_connection->send(Messages::ServiceWorkerFetchTask::DidReceiveData { buffer.get(), static_cast<int64_t>(buffer->size()) }, m_fetchIdentifier);
88}

90void WebServiceWorkerFetchTaskClient::didReceiveFormDataAndFinish(Ref<FormData>&& formData)
91{
92 if (auto sharedBuffer = formData->asSharedBuffer()) {
93 didReceiveData(sharedBuffer.releaseNonNull());
94 didFinish();
95 return;
96 }
97
98 if (!m_connection)
99 return;
100
101 if (m_waitingForContinueDidReceiveResponseMessage) {
102 m_responseData = formData->isolatedCopy();
103 return;
104 }
105
106 // FIXME: We should send this form data to the other process and consume it there.
107 // For now and for the case of blobs, we read it there and send the data through IPC.
108 URL blobURL = formData->asBlobURL();
109 if (blobURL.isNull()) {
110 m_connection->send(Messages::ServiceWorkerFetchTask::DidReceiveFormData { IPC::FormDataReference { WTFMove(formData) } }, m_fetchIdentifier);
111 return;
112 }
113
114 callOnMainRunLoop([this, protectedThis = makeRef(*this), blobURL = blobURL.isolatedCopy()] () {
115 auto* serviceWorkerThreadProxy = SWContextManager::singleton().serviceWorkerThreadProxy(m_serviceWorkerIdentifier);
116 if (!serviceWorkerThreadProxy) {
117 didFail(internalError(blobURL));
118 return;
119 }
120
121 m_blobLoader.emplace(*this);
122 auto loader = serviceWorkerThreadProxy->createBlobLoader(*m_blobLoader, blobURL);
123 if (!loader) {
124 m_blobLoader = std::nullopt;
125 didFail(internalError(blobURL));
126 return;
127 }
128
129 m_blobLoader->loader = WTFMove(loader);
130 });
131}

141void WebServiceWorkerFetchTaskClient::didFinishBlobLoading()
142{
143 didFinish();
144
145 std::exchange(m_blobLoader, std::nullopt);

148void WebServiceWorkerFetchTaskClient::didFail(const ResourceError& error)
149{
150 if (!m_connection)
151 return;
152
153 if (m_waitingForContinueDidReceiveResponseMessage) {
154 RELEASE_LOG(ServiceWorker, "ServiceWorkerFrameLoaderClient::didFail while waiting, fetch identifier %llu", m_fetchIdentifier.toUInt64());
155
156 m_responseData = makeUniqueRef<ResourceError>(error.isolatedCopy());
157 return;
158 }
159
160 m_connection->send(Messages::ServiceWorkerFetchTask::DidFail { error }, m_fetchIdentifier);
161
162 cleanup();
163}
164
165void WebServiceWorkerFetchTaskClient::didFinish()
166{
167 if (!m_connection)
168 return;
169
170 if (m_waitingForContinueDidReceiveResponseMessage) {
171 RELEASE_LOG(ServiceWorker, "ServiceWorkerFrameLoaderClient::didFinish while waiting, fetch identifier %llu", m_fetchIdentifier.toUInt64());
172
173 m_didFinish = true;
174 return;
175 }
176
177 m_connection->send(Messages::ServiceWorkerFetchTask::DidFinish { }, m_fetchIdentifier);
178
179 cleanup();
180}

192void WebServiceWorkerFetchTaskClient::cancel()

194 m_connection = nullptr;

197void WebServiceWorkerFetchTaskClient::continueDidReceiveResponse()
198{
199 RELEASE_LOG(ServiceWorker, "ServiceWorkerFrameLoaderClient::continueDidReceiveResponse, has connection %d, didFinish %d, response type %ld", !!m_connection, m_didFinish, static_cast<long>(m_responseData.index()));
200
201 if (!m_connection)
202 return;
203
204 m_waitingForContinueDidReceiveResponseMessage = false;
205
206 switchOn(m_responseData, [this](std::nullptr_t&) {
207 if (m_didFinish)
208 didFinish();
209 }, [this](Ref<SharedBuffer>& buffer) {
210 didReceiveData(WTFMove(buffer));
211 if (m_didFinish)
212 didFinish();
213 }, [this](Ref<FormData>& formData) {
214 didReceiveFormDataAndFinish(WTFMove(formData));
215 }, [this](UniqueRef<ResourceError>& error) {
216 didFail(error.get());
217 });
218 m_responseData = nullptr;
219}
220
221void WebServiceWorkerFetchTaskClient::cleanup()
222{
223 m_connection = nullptr;
224
225 ensureOnMainRunLoop([serviceWorkerIdentifier = m_serviceWorkerIdentifier, serverConnectionIdentifier = m_serverConnectionIdentifier, fetchIdentifier = m_fetchIdentifier] {
226 if (auto* proxy = SWContextManager::singleton().serviceWorkerThreadProxy(serviceWorkerIdentifier))
227 proxy->removeFetch(serverConnectionIdentifier, fetchIdentifier);
228 });
229}
