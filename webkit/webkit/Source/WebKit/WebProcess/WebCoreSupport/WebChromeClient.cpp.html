262Page* WebChromeClient::createWindow(Frame& frame, const WindowFeatures& windowFeatures, const NavigationAction& navigationAction)
263{
264#if ENABLE(FULLSCREEN_API)
265 if (frame.document() && frame.document()->fullscreenManager().currentFullscreenElement())
266 frame.document()->fullscreenManager().cancelFullscreen();
267#endif
268
269 auto& webProcess = WebProcess::singleton();
270
271 NavigationActionData navigationActionData;
272 navigationActionData.navigationType = navigationAction.type();
273 navigationActionData.modifiers = InjectedBundleNavigationAction::modifiersForNavigationAction(navigationAction);
274 navigationActionData.mouseButton = InjectedBundleNavigationAction::mouseButtonForNavigationAction(navigationAction);
275 navigationActionData.syntheticClickType = InjectedBundleNavigationAction::syntheticClickTypeForNavigationAction(navigationAction);
276 navigationActionData.clickLocationInRootViewCoordinates = InjectedBundleNavigationAction::clickLocationInRootViewCoordinatesForNavigationAction(navigationAction);
277 navigationActionData.userGestureTokenIdentifier = webProcess.userGestureTokenIdentifier(navigationAction.userGestureToken());
278 navigationActionData.canHandleRequest = m_page.canHandleRequest(navigationAction.resourceRequest());
279 navigationActionData.shouldOpenExternalURLsPolicy = navigationAction.shouldOpenExternalURLsPolicy();
280 navigationActionData.downloadAttribute = navigationAction.downloadAttribute();
281 navigationActionData.privateClickMeasurement = navigationAction.privateClickMeasurement();
282
283 WebFrame* webFrame = WebFrame::fromCoreFrame(frame);
284
285 std::optional<PageIdentifier> newPageID;
286 std::optional<WebPageCreationParameters> parameters;
287 if (!webProcess.parentProcessConnection()->sendSync(Messages::WebPageProxy::CreateNewPage(webFrame->info(), webFrame->page()->webPageProxyIdentifier(), navigationAction.resourceRequest(), windowFeatures, navigationActionData), Messages::WebPageProxy::CreateNewPage::Reply(newPageID, parameters), m_page.identifier(), IPC::Timeout::infinity(), IPC::SendSyncOption::MaintainOrderingWithAsyncMessages))
288 return nullptr;
289
290 if (!newPageID)
291 return nullptr;
292 ASSERT(parameters);
293
294 parameters->oldPageID = m_page.identifier();
295
296 webProcess.createWebPage(*newPageID, WTFMove(*parameters));
297 return webProcess.webPage(*newPageID)->corePage();
298}

424void WebChromeClient::closeWindowSoon()
425{
426 // FIXME: This code assumes that the client will respond to a close page
427 // message by actually closing the page. Safari does this, but there is
428 // no guarantee that other applications will, which will leave this page
429 // half detached. This approach is an inherent limitation making parts of
430 // a close execute synchronously as part of window.close, but other parts
431 // later on.
432
433 m_page.corePage()->setGroupName(String());
434
435 auto& frame = m_page.mainWebFrame();
436 if (auto* coreFrame = frame.coreFrame())
437 coreFrame->loader().stopForUserCancel();
438
439 m_page.sendClose();
440}

634void WebChromeClient::contentsSizeChanged(Frame& frame, const IntSize& size) const
635{
636 FrameView* frameView = frame.view();
637
638 if (&frame.page()->mainFrame() != &frame)
639 return;
640
641 m_page.send(Messages::WebPageProxy::DidChangeContentSize(size));
642
643 m_page.drawingArea()->mainFrameContentSizeChanged(size);
644
645 if (frameView && !frameView->delegatesScrolling()) {
646 bool hasHorizontalScrollbar = frameView->horizontalScrollbar();
647 bool hasVerticalScrollbar = frameView->verticalScrollbar();
648
649 if (hasHorizontalScrollbar != m_cachedMainFrameHasHorizontalScrollbar || hasVerticalScrollbar != m_cachedMainFrameHasVerticalScrollbar) {
650 m_page.send(Messages::WebPageProxy::DidChangeScrollbarsForMainFrame(hasHorizontalScrollbar, hasVerticalScrollbar));
651
652 m_cachedMainFrameHasHorizontalScrollbar = hasHorizontalScrollbar;
653 m_cachedMainFrameHasVerticalScrollbar = hasVerticalScrollbar;
654 }
655 }

663bool WebChromeClient::shouldUnavailablePluginMessageBeButton(RenderEmbeddedObject::PluginUnavailabilityReason pluginUnavailabilityReason) const
664{
665 switch (pluginUnavailabilityReason) {
666 case RenderEmbeddedObject::PluginMissing:
667 // FIXME: <rdar://problem/8794397> We should only return true when there is a
668 // missingPluginButtonClicked callback defined on the Page UI client.
669 case RenderEmbeddedObject::InsecurePluginVersion:
670 return true;
671
672
673 case RenderEmbeddedObject::PluginCrashed:
674 case RenderEmbeddedObject::PluginBlockedByContentSecurityPolicy:
675 case RenderEmbeddedObject::UnsupportedPlugin:
676 case RenderEmbeddedObject::PluginTooSmall:
677 return false;
678 }
679
680 ASSERT_NOT_REACHED();
681 return false;
682}

684void WebChromeClient::unavailablePluginButtonClicked(Element& element, RenderEmbeddedObject::PluginUnavailabilityReason pluginUnavailabilityReason) const
685{
686#if ENABLE(NETSCAPE_PLUGIN_API)
687 ASSERT(element.hasTagName(objectTag) || element.hasTagName(embedTag) || element.hasTagName(appletTag));
688 ASSERT(pluginUnavailabilityReason == RenderEmbeddedObject::PluginMissing || pluginUnavailabilityReason == RenderEmbeddedObject::InsecurePluginVersion || pluginUnavailabilityReason);
689
690 auto& pluginElement = downcast<HTMLPlugInImageElement>(element);
691
692 String frameURLString = pluginElement.document().frame()->loader().documentLoader()->responseURL().string();

717static constexpr unsigned maxTitleLength = 1000; // Closest power of 10 above the W3C recommendation for Title length.
718
719void WebChromeClient::print(Frame& frame, const StringWithDirection& title)
720{
721 WebFrame* webFrame = WebFrame::fromCoreFrame(frame);
722 ASSERT(webFrame);
723
724#if PLATFORM(GTK) && HAVE(GTK_UNIX_PRINTING)
725 // When printing synchronously in GTK+ we need to make sure that we have a list of Printers before starting the print operation.
726 // Getting the list of printers is done synchronously by GTK+, but using a nested main loop that might process IPC messages
727 // comming from the UI process like EndPrinting. When the EndPriting message is received while the printer list is being populated,
728 // the print operation is finished unexpectely and the web process crashes, see https://bugs.webkit.org/show_bug.cgi?id=126979.
729 // The PrinterListGtk class gets the list of printers in the constructor so we just need to ensure there's an instance alive
730 // during the synchronous print operation.
731 RefPtr<PrinterListGtk> printerList = PrinterListGtk::getOrCreate();
732 if (!printerList) {
733 // PrinterListGtk::getOrCreate() returns nullptr when called while a printers enumeration is ongoing.
734 // This can happen if a synchronous print is started by a JavaScript and another one is inmeditaley started
735 // from a JavaScript event listener. The second print operation is handled by the nested main loop used by GTK+
736 // to enumerate the printers, and we end up here trying to get a reference of an object that is being constructed.
737 // It's very unlikely that the user wants to print twice in a row, and other browsers don't do two print operations
738 // in this particular case either. So, the safest solution is to return early here and ignore the second print.
739 // See https://bugs.webkit.org/show_bug.cgi?id=141035
740 return;
741 }
742#endif
743
744 WebCore::FloatSize pdfFirstPageSize;
745#if PLATFORM(COCOA)
746 if (auto* pluginView = WebPage::pluginViewForFrame(&frame)) {
747 if (auto* plugin = pluginView->plugin())
748 pdfFirstPageSize = plugin->pdfDocumentSizeForPrinting();
749 }
750#endif
751
752 auto truncatedTitle = truncateFromEnd(title, maxTitleLength);
753 m_page.sendSyncWithDelayedReply(Messages::WebPageProxy::PrintFrame(webFrame->frameID(), truncatedTitle.string, pdfFirstPageSize), Messages::WebPageProxy::PrintFrame::Reply());
754}

756void WebChromeClient::exceededDatabaseQuota(Frame& frame, const String& databaseName, DatabaseDetails details)
757{
758 WebFrame* webFrame = WebFrame::fromCoreFrame(frame);
759 ASSERT(webFrame);
760
761 auto& origin = frame.document()->securityOrigin();
762 auto& originData = origin.data();
763 auto& tracker = DatabaseTracker::singleton();
764 auto currentQuota = tracker.quota(originData);
765 auto currentOriginUsage = tracker.usage(originData);
766 uint64_t newQuota = 0;
767 auto securityOrigin = API::SecurityOrigin::create(SecurityOriginData::fromDatabaseIdentifier(originData.databaseIdentifier())->securityOrigin());
768 newQuota = m_page.injectedBundleUIClient().didExceedDatabaseQuota(&m_page, securityOrigin.ptr(), databaseName, details.displayName(), currentQuota, currentOriginUsage, details.currentUsage(), details.expectedUsage());
769
770 if (!newQuota)
771 m_page.sendSyncWithDelayedReply(Messages::WebPageProxy::ExceededDatabaseQuota(webFrame->frameID(), originData.databaseIdentifier(), databaseName, details.displayName(), currentQuota, currentOriginUsage, details.currentUsage(), details.expectedUsage()), Messages::WebPageProxy::ExceededDatabaseQuota::Reply(newQuota));
772
773 tracker.setQuota(originData, newQuota);
774}

781void WebChromeClient::reachedApplicationCacheOriginQuota(SecurityOrigin& origin, int64_t totalBytesNeeded)
782{
783 auto securityOrigin = API::SecurityOrigin::createFromString(origin.toString());
784 if (m_page.injectedBundleUIClient().didReachApplicationCacheOriginQuota(&m_page, securityOrigin.ptr(), totalBytesNeeded))
785 return;
786
787 auto& cacheStorage = m_page.corePage()->applicationCacheStorage();
788 int64_t currentQuota = 0;
789 if (!cacheStorage.calculateQuotaForOrigin(origin, currentQuota))
790 return;
791
792 uint64_t newQuota = 0;
793 m_page.sendSyncWithDelayedReply(Messages::WebPageProxy::ReachedApplicationCacheOriginQuota(origin.data().databaseIdentifier(), currentQuota, totalBytesNeeded), Messages::WebPageProxy::ReachedApplicationCacheOriginQuota::Reply(newQuota));
794
795 cacheStorage.storeUpdatedQuotaForOrigin(&origin, newQuota);
796}
