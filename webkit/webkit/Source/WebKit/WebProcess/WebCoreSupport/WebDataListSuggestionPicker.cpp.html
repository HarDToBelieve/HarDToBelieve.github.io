39WebDataListSuggestionPicker::WebDataListSuggestionPicker(WebPage& page, WebCore::DataListSuggestionsClient& client)
40 : m_client(client)
41 , m_page(page)
42{
43}
44
45void WebDataListSuggestionPicker::handleKeydownWithIdentifier(const String& key)
46{
47 WebProcess::singleton().parentProcessConnection()->send(Messages::WebPageProxy::HandleKeydownInDataList(key), m_page.identifier());
48}
49
50void WebDataListSuggestionPicker::didSelectOption(const String& selectedOption)
51{
52 m_client.didSelectDataListOption(selectedOption);
53}
54
55void WebDataListSuggestionPicker::didCloseSuggestions()
56{
57 m_client.didCloseSuggestions();
58}
59
60void WebDataListSuggestionPicker::close()
61{
62 WebProcess::singleton().parentProcessConnection()->send(Messages::WebPageProxy::EndDataListSuggestions(), m_page.identifier());
63}
64
65void WebDataListSuggestionPicker::displayWithActivationType(WebCore::DataListSuggestionActivationType type)
66{
67 auto suggestions = m_client.suggestions();
68 if (suggestions.isEmpty()) {
69 close();
70 return;
71 }
72
73 m_page.setActiveDataListSuggestionPicker(*this);
74
75 WebCore::DataListSuggestionInformation info { type, WTFMove(suggestions), m_client.elementRectInRootViewCoordinates() };
76 WebProcess::singleton().parentProcessConnection()->send(Messages::WebPageProxy::ShowDataListSuggestions(info), m_page.identifier());
77}
