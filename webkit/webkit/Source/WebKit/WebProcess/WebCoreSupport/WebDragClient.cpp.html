36void WebDragClient::willPerformDragDestinationAction(DragDestinationAction action, const DragData&)
37{
38 if (action == DragDestinationAction::Load)
39 m_page->willPerformLoadDragDestinationAction();
40 else
41 m_page->mayPerformUploadDragDestinationAction(); // Upload can happen from a drop event handler, so we should prepare early.
42}
