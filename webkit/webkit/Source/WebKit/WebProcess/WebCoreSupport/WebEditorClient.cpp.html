284void WebEditorClient::registerUndoStep(UndoStep& step)
285{
286 // FIXME: Add assertion that the command being reapplied is the same command that is
287 // being passed to us.
288 if (m_page->isInRedo())
289 return;
290
291 auto webStep = WebUndoStep::create(step);
292 auto stepID = webStep->stepID();
293
294 m_page->addWebUndoStep(stepID, WTFMove(webStep));
295 m_page->send(Messages::WebPageProxy::RegisterEditCommandForUndo(stepID, step.label()), IPC::SendOption::DispatchMessageEvenWhenWaitingForSyncReply);
296}

419static bool getActionTypeForKeyEvent(KeyboardEvent* event, WKInputFieldActionType& type)
420{
421 String key = event->keyIdentifier();
422 if (key == "Up")
423 type = WKInputFieldActionTypeMoveUp;
424 else if (key == "Down")
425 type = WKInputFieldActionTypeMoveDown;
426 else if (key == "U+001B")
427 type = WKInputFieldActionTypeCancel;
428 else if (key == "U+0009") {
429 if (event->shiftKey())
430 type = WKInputFieldActionTypeInsertBacktab;
431 else
432 type = WKInputFieldActionTypeInsertTab;
433 } else if (key == "Enter")
434 type = WKInputFieldActionTypeInsertNewline;
435 else
436 return false;
437
438 return true;
439}
440
441static API::InjectedBundle::FormClient::InputFieldAction toInputFieldAction(WKInputFieldActionType action)
442{
443 switch (action) {
444 case WKInputFieldActionTypeMoveUp:
445 return API::InjectedBundle::FormClient::InputFieldAction::MoveUp;
446 case WKInputFieldActionTypeMoveDown:
447 return API::InjectedBundle::FormClient::InputFieldAction::MoveDown;
448 case WKInputFieldActionTypeCancel:
449 return API::InjectedBundle::FormClient::InputFieldAction::Cancel;
450 case WKInputFieldActionTypeInsertTab:
451 return API::InjectedBundle::FormClient::InputFieldAction::InsertTab;
452 case WKInputFieldActionTypeInsertNewline:
453 return API::InjectedBundle::FormClient::InputFieldAction::InsertNewline;
454 case WKInputFieldActionTypeInsertDelete:
455 return API::InjectedBundle::FormClient::InputFieldAction::InsertDelete;
456 case WKInputFieldActionTypeInsertBacktab:
457 return API::InjectedBundle::FormClient::InputFieldAction::InsertBacktab;
458 }
459
460 ASSERT_NOT_REACHED();
461 return API::InjectedBundle::FormClient::InputFieldAction::Cancel;
462}

464bool WebEditorClient::doTextFieldCommandFromEvent(Element* element, KeyboardEvent* event)
465{
466 if (!is<HTMLInputElement>(*element))
467 return false;
468
469 WKInputFieldActionType actionType = static_cast<WKInputFieldActionType>(0);
470 if (!getActionTypeForKeyEvent(event, actionType))
471 return false;
472
473 WebFrame* webFrame = WebFrame::fromCoreFrame(*element->document().frame());
474 ASSERT(webFrame);
475
476 return m_page->injectedBundleFormClient().shouldPerformActionInTextField(m_page, downcast<HTMLInputElement>(element), toInputFieldAction(actionType), webFrame);
477}

537static uint64_t insertionPointFromCurrentSelection(const VisibleSelection& currentSelection)
538{
539 auto selectionStart = currentSelection.visibleStart();
540 auto range = makeSimpleRange(selectionStart, startOfParagraph(selectionStart));
541 return range ? characterCount(*range) : 0;
542}
