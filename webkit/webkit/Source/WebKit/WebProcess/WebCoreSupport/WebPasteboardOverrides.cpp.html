43void WebPasteboardOverrides::addOverride(const String& pasteboardName, const String& type, const Vector<uint8_t>& data)
44{
45 auto addResult = m_overridesMap.add(pasteboardName, nullptr);
46
47 if (addResult.isNewEntry) {
48 std::unique_ptr<HashMap<String, Vector<uint8_t>>> typeMap = makeUnique<HashMap<String, Vector<uint8_t>>>();
49 addResult.iterator->value = WTFMove(typeMap);
50 }
51
52 addResult.iterator->value->set(type, data);
53}
54
55void WebPasteboardOverrides::removeOverride(const String& pasteboardName, const String& type)
56{
57 auto it = m_overridesMap.find(pasteboardName);
58 if (it == m_overridesMap.end())
59 return;
60
61 ASSERT(it->value);
62
63 it->value->remove(type);
64
65 // If this was the last override for this pasteboard, remove its record completely.
66 if (it->value->isEmpty())
67 m_overridesMap.remove(it);
68}
69
70Vector<String> WebPasteboardOverrides::overriddenTypes(const String& pasteboardName)
71{
72 Vector<String> result;
73
74 auto it = m_overridesMap.find(pasteboardName);
75 if (it == m_overridesMap.end())
76 return result;
77
78 ASSERT(it->value);
79
80 for (String& type : it->value->keys())
81 result.append(type);
82
83 return result;
84}
85
86bool WebPasteboardOverrides::getDataForOverride(const String& pasteboardName, const String& type, Vector<uint8_t>& data) const
87{
88 auto pasteboardIterator = m_overridesMap.find(pasteboardName);
89 if (pasteboardIterator == m_overridesMap.end())
90 return false;
91
92 auto typeIterator = pasteboardIterator->value->find(type);
93 if (typeIterator == pasteboardIterator->value->end())
94 return false;
95
96 data = typeIterator->value;
97 return true;
98}
