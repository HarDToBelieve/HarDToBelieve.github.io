30#include <WebCore/PageIdentifier.h>

32#include <WebCore/ResourceLoadStatistics.h>

34#include <wtf/Forward.h>

36namespace WebKit {
37
38class WebPage;
39
40class WebResourceLoadObserver final : public WebCore::ResourceLoadObserver {
41public:
42 using TopFrameDomain = WebCore::RegistrableDomain;
43 using SubFrameDomain = WebCore::RegistrableDomain;
44
45 WebResourceLoadObserver(WebCore::ResourceLoadStatistics::IsEphemeral);
46 ~WebResourceLoadObserver();
47
48 void logSubresourceLoading(const WebCore::Frame*, const WebCore::ResourceRequest& newRequest, const WebCore::ResourceResponse& redirectResponse, FetchDestinationIsScriptLike) final;
49 void logWebSocketLoading(const URL& targetURL, const URL& mainFrameURL) final;
50 void logUserInteractionWithReducedTimeResolution(const WebCore::Document&) final;
51 void logFontLoad(const WebCore::Document&, const String& familyName, bool loadStatus) final;
52 void logCanvasRead(const WebCore::Document&) final;
53 void logCanvasWriteOrMeasure(const WebCore::Document&, const String& textWritten) final;
54 void logNavigatorAPIAccessed(const WebCore::Document&, const WebCore::ResourceLoadStatistics::NavigatorAPI) final;
55 void logScreenAPIAccessed(const WebCore::Document&, const WebCore::ResourceLoadStatistics::ScreenAPI) final;
56 void logSubresourceLoadingForTesting(const WebCore::RegistrableDomain& firstPartyDomain, const WebCore::RegistrableDomain& thirdPartyDomain, bool shouldScheduleNotification);
57
58#if !RELEASE_LOG_DISABLED
59 static void setShouldLogUserInteraction(bool);
60#endif
61
62 String statisticsForURL(const URL&) final;
63 void updateCentralStatisticsStore(CompletionHandler<void()>&&) final;
64 void clearState() final;
65
66 bool hasStatistics() const final { return !m_resourceStatisticsMap.isEmpty(); }
67
68 void setDomainsWithUserInteraction(HashSet<WebCore::RegistrableDomain>&& domains) final { m_domainsWithUserInteraction = WTFMove(domains); }
69 void setDomainsWithCrossPageStorageAccess(HashMap<TopFrameDomain, SubFrameDomain>&&, CompletionHandler<void()>&&) final;
70 bool hasHadUserInteraction(const WebCore::RegistrableDomain&) const final;
71 bool hasCrossPageStorageAccess(const SubFrameDomain&, const TopFrameDomain&) const final;
72
73private:
74 WebCore::ResourceLoadStatistics& ensureResourceStatisticsForRegistrableDomain(const WebCore::RegistrableDomain&);
75 void scheduleNotificationIfNeeded();
76
77 Vector<WebCore::ResourceLoadStatistics> takeStatistics();
78 void requestStorageAccessUnderOpener(const WebCore::RegistrableDomain& domainInNeedOfStorageAccess, WebPage& openerPage, WebCore::Document& openerDocument);
79
80 bool isEphemeral() const { return m_isEphemeral == WebCore::ResourceLoadStatistics::IsEphemeral::Yes; }
81
82 WebCore::ResourceLoadStatistics::IsEphemeral m_isEphemeral { WebCore::ResourceLoadStatistics::IsEphemeral::No };
83
84 HashMap<WebCore::RegistrableDomain, std::unique_ptr<WebCore::ResourceLoadStatistics>> m_resourceStatisticsMap;
85 HashMap<WebCore::RegistrableDomain, WTF::WallTime> m_lastReportedUserInteractionMap;
86
87 WebCore::Timer m_notificationTimer;
88
89 HashSet<WebCore::RegistrableDomain> m_domainsWithUserInteraction;
90 HashMap<TopFrameDomain, HashSet<SubFrameDomain>> m_domainsWithCrossPageStorageAccess;
91#if !RELEASE_LOG_DISABLED
92 uint64_t m_loggingCounter { 0 };
93 static bool shouldLogUserInteraction;
94#endif
95};
96
97} // namespace WebKit

99#endif // ENABLE(RESOURCE_LOAD_STATISTICS)
