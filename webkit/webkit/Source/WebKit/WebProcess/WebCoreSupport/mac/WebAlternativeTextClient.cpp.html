36WebAlternativeTextClient::WebAlternativeTextClient(WebPage* webPage)
37: m_page(webPage)
38{
39}
40
41WebAlternativeTextClient::~WebAlternativeTextClient()
42{
43#if USE(AUTOCORRECTION_PANEL)
44 m_page->send(Messages::WebPageProxy::DismissCorrectionPanel(ReasonForDismissingAlternativeTextIgnored));
45#endif
46}
47
48#if USE(AUTOCORRECTION_PANEL)
49void WebAlternativeTextClient::showCorrectionAlternative(AlternativeTextType type, const FloatRect& boundingBoxOfReplacedString, const String& replacedString, const String& replacementString, const Vector<String>& alternativeReplacementStrings)
50{
51 m_page->send(Messages::WebPageProxy::ShowCorrectionPanel(type, boundingBoxOfReplacedString, replacedString, replacementString, alternativeReplacementStrings));
52}
53
54void WebAlternativeTextClient::dismissAlternative(ReasonForDismissingAlternativeText reason)
55{
56 m_page->send(Messages::WebPageProxy::DismissCorrectionPanel(reason));
57}
58
59String WebAlternativeTextClient::dismissAlternativeSoon(ReasonForDismissingAlternativeText reason)
60{
61 String result;
62 m_page->sendSync(Messages::WebPageProxy::DismissCorrectionPanelSoon(reason), Messages::WebPageProxy::DismissCorrectionPanelSoon::Reply(result));
63 return result;
64}
65
66void WebAlternativeTextClient::recordAutocorrectionResponse(AutocorrectionResponse response, const String& replacedString, const String& replacementString)
67{
68 m_page->send(Messages::WebPageProxy::RecordAutocorrectionResponse(static_cast<int32_t>(response), replacedString, replacementString));
69}
70#endif
71
72void WebAlternativeTextClient::removeDictationAlternatives(WebCore::DictationContext dictationContext)
73{
74 m_page->send(Messages::WebPageProxy::RemoveDictationAlternatives(dictationContext));
75}
76
77void WebAlternativeTextClient::showDictationAlternativeUI(const WebCore::FloatRect& boundingBoxOfDictatedText, WebCore::DictationContext dictationContext)
78{
79 m_page->send(Messages::WebPageProxy::ShowDictationAlternativeUI(boundingBoxOfDictatedText, dictationContext));
80}
81
82Vector<String> WebAlternativeTextClient::dictationAlternatives(WebCore::DictationContext dictationContext)
83{
84 Vector<String> result;
85 m_page->sendSync(Messages::WebPageProxy::DictationAlternatives(dictationContext), Messages::WebPageProxy::DictationAlternatives::Reply(result));
86 return result;
87}
