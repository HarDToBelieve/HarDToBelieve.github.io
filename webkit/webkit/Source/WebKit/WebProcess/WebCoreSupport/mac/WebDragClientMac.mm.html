69static RefPtr<ShareableBitmap> convertDragImageToBitmap(DragImage image, const IntSize& size, Frame& frame)
70{
71 auto bitmap = ShareableBitmap::createShareable(size, { screenColorSpace(frame.mainFrame().view()) });
72 if (!bitmap)
73 return nullptr;
74
75 auto graphicsContext = bitmap->createGraphicsContext();
76 if (!graphicsContext)
77 return nullptr;
78
79 LocalCurrentGraphicsContext savedContext(*graphicsContext);
80#if USE(APPKIT)
81 [image drawInRect:NSMakeRect(0, 0, bitmap->size().width(), bitmap->size().height()) fromRect:NSZeroRect operation:NSCompositingOperationSourceOver fraction:1 respectFlipped:YES hints:nil];
82#else
83 CGContextDrawImage(graphicsContext->platformContext(), CGRectMake(0, 0, size.width(), size.height()), image);
84#endif
85
86 return bitmap;
87}

89void WebDragClient::startDrag(DragItem dragItem, DataTransfer&, Frame& frame)
90{
91 auto& image = dragItem.image;
92
93#if USE(APPKIT)
94 IntSize bitmapSize([image.get() size]);
95#else
96 IntSize bitmapSize(CGImageGetWidth(image.get().get()), CGImageGetHeight(image.get().get()));
97#endif
98 auto bitmap = convertDragImageToBitmap(image.get().get(), bitmapSize, frame);
99 ShareableBitmap::Handle handle;
100 if (!bitmap || !bitmap->createHandle(handle))
101 return;
102
103 m_page->willStartDrag();
104 m_page->send(Messages::WebPageProxy::StartDrag(dragItem, handle));
105}

116static WebCore::CachedImage* cachedImage(Element& element)
117{
118 auto* renderer = element.renderer();
119 if (!is<WebCore::RenderImage>(renderer))
120 return nullptr;
121 WebCore::CachedImage* image = downcast<WebCore::RenderImage>(*renderer).cachedImage();
122 if (!image || image->errorOccurred())
123 return nullptr;
124 return image;
125}

127void WebDragClient::declareAndWriteDragImage(const String& pasteboardName, Element& element, const URL& url, const String& label, Frame*)
128{
129 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
130 ASSERT(pasteboardName == String(NSDragPboard));
131 ALLOW_DEPRECATED_DECLARATIONS_END
132
133 WebCore::CachedImage* image = cachedImage(element);
134
135 String extension;
136 if (image) {
137 extension = image->image()->filenameExtension();
138 if (extension.isEmpty())
139 return;
140 }
141
142 String title = label;
143 if (title.isEmpty()) {
144 title = url.lastPathComponent().toString();
145 if (title.isEmpty())
146 title = WTF::userVisibleString(url);
147 }
148
149 auto archive = LegacyWebArchive::create(element);
150
151 NSURLResponse *response = image->response().nsURLResponse();
152
153 RefPtr<SharedBuffer> imageBuffer = image->image()->data();
154 size_t imageSize = imageBuffer->size();
155 SharedMemory::Handle imageHandle;
156
157 RefPtr<SharedMemory> sharedMemoryBuffer = SharedMemory::allocate(imageBuffer->size());
158 if (!sharedMemoryBuffer)
159 return;
160 memcpy(sharedMemoryBuffer->data(), imageBuffer->data(), imageSize);
161 sharedMemoryBuffer->createHandle(imageHandle, SharedMemory::Protection::ReadOnly);
162
163 RetainPtr<CFDataRef> data = archive ? archive->rawDataRepresentation() : 0;
164 SharedMemory::Handle archiveHandle;
165 size_t archiveSize = 0;
166 if (data) {
167 auto archiveBuffer = SharedBuffer::create((__bridge NSData *)data.get());
168 RefPtr<SharedMemory> archiveSharedMemoryBuffer = SharedMemory::allocate(archiveBuffer->size());
169 if (!archiveSharedMemoryBuffer)
170 return;
171 archiveSize = archiveBuffer->size();
172 memcpy(archiveSharedMemoryBuffer->data(), archiveBuffer->data(), archiveSize);
173 archiveSharedMemoryBuffer->createHandle(archiveHandle, SharedMemory::Protection::ReadOnly);
174 }
175
176 String filename = String([response suggestedFilename]);
177 if (m_page->isInspectorPage()) {
178 String downloadFilename = ResourceResponseBase::sanitizeSuggestedFilename(element.attributeWithoutSynchronization(HTMLNames::filenameAttr));
179 if (!downloadFilename.isEmpty())
180 filename = downloadFilename;
181 }
182
183 m_page->send(Messages::WebPageProxy::SetPromisedDataForImage(pasteboardName, SharedMemory::IPCHandle { WTFMove(imageHandle), imageSize }, filename, extension, title, String([[response URL] absoluteString]), WTF::userVisibleString(url), SharedMemory::IPCHandle { WTFMove(archiveHandle), archiveSize }, element.document().originIdentifierForPasteboard()));
184}
