66static void changeWordCase(WebPage* page, NSString *(*changeCase)(NSString *))
67{
68 Frame& frame = page->corePage()->focusController().focusedOrMainFrame();
69 if (!frame.editor().canEdit())
70 return;
71
72 frame.editor().command("selectWord").execute();
73
74 NSString *selectedString = frame.displayStringModifiedByEncoding(frame.editor().selectedText());
75 page->replaceSelectionWithText(&frame, changeCase(selectedString));
76}
