42namespace WebKit {
43using namespace WebCore;
44
45bool WebFrameNetworkingContext::localFileContentSniffingEnabled() const
46{
47 return frame() && frame()->settings().localFileContentSniffingEnabled();
48}
49
50SchedulePairHashSet* WebFrameNetworkingContext::scheduledRunLoopPairs() const
51{
52 if (!frame() || !frame()->page())
53 return nullptr;
54 return frame()->page()->scheduledRunLoopPairs();
55}
56
57RetainPtr<CFDataRef> WebFrameNetworkingContext::sourceApplicationAuditData() const
58{
59 return WebProcess::singleton().sourceApplicationAuditData();
60}
61
62String WebFrameNetworkingContext::sourceApplicationIdentifier() const
63{
64 return WebProcess::singleton().uiProcessBundleIdentifier();
65}
66
67ResourceError WebFrameNetworkingContext::blockedError(const ResourceRequest& request) const
68{
69 return frame()->loader().client().blockedError(request);
70}
71
72WebFrameLoaderClient* WebFrameNetworkingContext::webFrameLoaderClient() const
73{
74 if (!frame())
75 return nullptr;
76
77 return toWebFrameLoaderClient(frame()->loader().client());
78}
