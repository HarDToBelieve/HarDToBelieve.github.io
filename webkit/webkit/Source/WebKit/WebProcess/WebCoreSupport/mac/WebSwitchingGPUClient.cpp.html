36namespace WebKit {
37
38WebSwitchingGPUClient& WebSwitchingGPUClient::singleton()
39{
40 static NeverDestroyed<WebSwitchingGPUClient> client;
41 return client;
42}
43
44void WebSwitchingGPUClient::requestHighPerformanceGPU()
45{
46 m_requests++;
47 if (m_requests != 1)
48 return;
49 LOG(WebGL, "WebSwitchingGPUClient::requestHighPerformanceGPU() from WebProcess");
50 WebProcess::singleton().parentProcessConnection()->send(Messages::WebProcessProxy::RequestHighPerformanceGPU(), 0);
51}
52
53void WebSwitchingGPUClient::releaseHighPerformanceGPU()
54{
55 ASSERT(m_requests);
56 m_requests--;
57 if (m_requests)
58 return;
59 LOG(WebGL, "WebSwitchingGPUClient::releaseHighPerformanceGPU() from WebProcess");
60 WebProcess::singleton().parentProcessConnection()->send(Messages::WebProcessProxy::ReleaseHighPerformanceGPU(), 0);
61}
