29#include "RemoteObjectRegistryMessages.h"

33namespace WebKit {
34
35WebRemoteObjectRegistry::WebRemoteObjectRegistry(_WKRemoteObjectRegistry *remoteObjectRegistry, WebPage& page)
36 : RemoteObjectRegistry(remoteObjectRegistry)
37 , m_page(page)
38{
39 WebProcess::singleton().addMessageReceiver(Messages::RemoteObjectRegistry::messageReceiverName(), m_page.identifier(), *this);
40 page.setRemoteObjectRegistry(this);
41}
42
43WebRemoteObjectRegistry::~WebRemoteObjectRegistry()
44{
45 close();
46}
47
48void WebRemoteObjectRegistry::close()
49{
50 if (m_page.remoteObjectRegistry() == this) {
51 WebProcess::singleton().removeMessageReceiver(Messages::RemoteObjectRegistry::messageReceiverName(), m_page.identifier());
52 m_page.setRemoteObjectRegistry(nullptr);
53 }
54}
55
56IPC::MessageSender& WebRemoteObjectRegistry::messageSender()
57{
58 return m_page;
59}
60
61uint64_t WebRemoteObjectRegistry::messageDestinationID()
62{
63 return m_page.webPageProxyIdentifier().toUInt64();
64}
