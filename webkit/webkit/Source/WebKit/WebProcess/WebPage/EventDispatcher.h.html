51namespace WebKit {
52
53class WebPage;
54class WebWheelEvent;
55
56#if ENABLE(IOS_TOUCH_EVENTS)
57class WebTouchEvent;
58#endif

60class EventDispatcher : public IPC::Connection::WorkQueueMessageReceiver {
61public:
62 static Ref<EventDispatcher> create();
63 ~EventDispatcher();
64
65#if ENABLE(SCROLLING_THREAD)
66 void addScrollingTreeForPage(WebPage*);
67 void removeScrollingTreeForPage(WebPage*);
68#endif
69
70#if ENABLE(IOS_TOUCH_EVENTS)
71 using TouchEventQueue = Vector<std::pair<WebTouchEvent, CompletionHandler<void(bool)>>, 1>;
72 void takeQueuedTouchEventsForPage(const WebPage&, TouchEventQueue&);
73#endif
74
75 void initializeConnection(IPC::Connection*);
76
77 void notifyScrollingTreesDisplayWasRefreshed(WebCore::PlatformDisplayID);
78
79private:
80 EventDispatcher();
81
82 // IPC::Connection::WorkQueueMessageReceiver.
83 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) override;
84
85 // Message handlers
86 void wheelEvent(WebCore::PageIdentifier, const WebWheelEvent&, bool canRubberBandAtLeft, bool canRubberBandAtRight, bool canRubberBandAtTop, bool canRubberBandAtBottom);
87#if ENABLE(IOS_TOUCH_EVENTS)
88 void touchEvent(WebCore::PageIdentifier, const WebTouchEvent&, CompletionHandler<void(bool)>&&);
89 void touchEventWithoutCallback(WebCore::PageIdentifier, const WebTouchEvent&);
90#endif
91#if ENABLE(MAC_GESTURE_EVENTS)
92 void gestureEvent(WebCore::PageIdentifier, const WebGestureEvent&);
93#endif
94
95 // This is called on the main thread.
96 void dispatchWheelEvent(WebCore::PageIdentifier, const WebWheelEvent&, OptionSet<WebCore::WheelEventProcessingSteps>);
97 void dispatchWheelEventViaMainThread(WebCore::PageIdentifier, const WebWheelEvent&, OptionSet<WebCore::WheelEventProcessingSteps>);
98
99#if ENABLE(IOS_TOUCH_EVENTS)
100 void dispatchTouchEvents();
101#endif
102#if ENABLE(MAC_GESTURE_EVENTS)
103 void dispatchGestureEvent(WebCore::PageIdentifier, const WebGestureEvent&);
104#endif
105
106#if ENABLE(ASYNC_SCROLLING)
107 static void sendDidReceiveEvent(WebCore::PageIdentifier, WebEvent::Type, bool didHandleEvent);
108#endif
109
110#if PLATFORM(MAC)
111 void displayWasRefreshed(WebCore::PlatformDisplayID, const WebCore::DisplayUpdate&, bool sendToMainThread);
112#endif
113
114#if ENABLE(SCROLLING_THREAD)
115 void displayDidRefreshOnScrollingThread(WebCore::PlatformDisplayID);
116#endif
117
118 Ref<WorkQueue> m_queue;
119
120#if ENABLE(SCROLLING_THREAD)
121 Lock m_scrollingTreesLock;
122 HashMap<WebCore::PageIdentifier, RefPtr<WebCore::ThreadedScrollingTree>> m_scrollingTrees WTF_GUARDED_BY_LOCK(m_scrollingTreesLock);
123#endif
124 std::unique_ptr<WebCore::WheelEventDeltaFilter> m_recentWheelEventDeltaFilter;
125#if ENABLE(IOS_TOUCH_EVENTS)
126 Lock m_touchEventsLock;
127 HashMap<WebCore::PageIdentifier, TouchEventQueue> m_touchEvents WTF_GUARDED_BY_LOCK(m_touchEventsLock);
128#endif
129};
