53class FindController : private WebCore::PageOverlay::Client {
54 WTF_MAKE_FAST_ALLOCATED;
55 WTF_MAKE_NONCOPYABLE(FindController);
56
57public:
58 explicit FindController(WebPage*);
59 virtual ~FindController();
60
61 void findString(const String&, OptionSet<FindOptions>, unsigned maxMatchCount, CompletionHandler<void(bool)>&&);
62 void findStringMatches(const String&, OptionSet<FindOptions>, unsigned maxMatchCount);
63 void getImageForFindMatch(uint32_t matchIndex);
64 void selectFindMatch(uint32_t matchIndex);
65 void indicateFindMatch(uint32_t matchIndex);
66 void hideFindUI();
67 void countStringMatches(const String&, OptionSet<FindOptions>, unsigned maxMatchCount);
68 uint32_t replaceMatches(const Vector<uint32_t>& matchIndices, const String& replacementText, bool selectionOnly);
69
70 void hideFindIndicator();
71 void resetMatchIndex();
72 void showFindIndicatorInSelection();
73
74 bool isShowingOverlay() const { return m_isShowingFindIndicator && m_findPageOverlay; }
75
76 void deviceScaleFactorDidChange();
77 void didInvalidateDocumentMarkerRects();
78
79 void redraw();
80
81private:
82 // PageOverlay::Client.
83 void willMoveToPage(WebCore::PageOverlay&, WebCore::Page*) override;
84 void didMoveToPage(WebCore::PageOverlay&, WebCore::Page*) override;
85 bool mouseEvent(WebCore::PageOverlay&, const WebCore::PlatformMouseEvent&) override;
86 void drawRect(WebCore::PageOverlay&, WebCore::GraphicsContext&, const WebCore::IntRect& dirtyRect) override;
87
88 Vector<WebCore::FloatRect> rectsForTextMatchesInRect(WebCore::IntRect clipRect);
89 bool updateFindIndicator(WebCore::Frame& selectedFrame, bool isShowingOverlay, bool shouldAnimate = true);
90
91 enum class FindUIOriginator : uint8_t { FindString, FindStringMatches };
92 void updateFindUIAfterPageScroll(bool found, const String&, OptionSet<FindOptions>, unsigned maxMatchCount, WebCore::DidWrap, FindUIOriginator);
93
94 void willFindString();
95 void didFindString();
96 void didFailToFindString();
97 void didHideFindIndicator();
98
99 unsigned findIndicatorRadius() const;
100 bool shouldHideFindIndicatorOnScroll() const;
101 void didScrollAffectingFindIndicatorPosition();
102
103 WebPage* m_webPage;
104 WebCore::PageOverlay* m_findPageOverlay { nullptr };
105
106 // Whether the UI process is showing the find indicator. Note that this can be true even if
107 // the find indicator isn't showing, but it will never be false when it is showing.
108 bool m_isShowingFindIndicator { false };
109 WebCore::IntRect m_findIndicatorRect;
110 Vector<WebCore::SimpleRange> m_findMatches;
111 // Index value is -1 if not found or if number of matches exceeds provided maximum.
112 int m_foundStringMatchIndex { -1 };
113
114#if PLATFORM(IOS_FAMILY)
115 RefPtr<WebCore::PageOverlay> m_findIndicatorOverlay;
116 std::unique_ptr<FindIndicatorOverlayClientIOS> m_findIndicatorOverlayClient;
117#endif
118};
