26#pragma once
27
28#include <WebCore/GraphicsLayerCA.h>
29#include <WebCore/PlatformLayer.h>
30
31namespace WebKit {
32
33class RemoteLayerTreeContext;
34
35class GraphicsLayerCARemote final : public WebCore::GraphicsLayerCA {
36public:
37 GraphicsLayerCARemote(Type layerType, WebCore::GraphicsLayerClient&, RemoteLayerTreeContext&);
38 virtual ~GraphicsLayerCARemote();
39
40 bool filtersCanBeComposited(const WebCore::FilterOperations& filters) override;
41
42 void moveToContext(RemoteLayerTreeContext&);
43 void clearContext() { m_context = nullptr; }
44
45private:
46 bool isGraphicsLayerCARemote() const override { return true; }
47
48 Ref<WebCore::PlatformCALayer> createPlatformCALayer(WebCore::PlatformCALayer::LayerType, WebCore::PlatformCALayerClient* owner) override;
49 Ref<WebCore::PlatformCALayer> createPlatformCALayer(PlatformLayer*, WebCore::PlatformCALayerClient* owner) override;
50#if ENABLE(MODEL_ELEMENT)
51 Ref<WebCore::PlatformCALayer> createPlatformCALayer(Ref<WebCore::Model>, WebCore::PlatformCALayerClient* owner) override;
52#endif
53 Ref<WebCore::PlatformCAAnimation> createPlatformCAAnimation(WebCore::PlatformCAAnimation::AnimationType, const String& keyPath) override;
54
55 // PlatformCALayerRemote can't currently proxy directly composited image contents, so opt out of this optimization.
56 bool shouldDirectlyCompositeImage(WebCore::Image*) const override { return false; }
57
58 RemoteLayerTreeContext* m_context;
59};
60
61} // namespace WebKit

63SPECIALIZE_TYPE_TRAITS_GRAPHICSLAYER(WebKit::GraphicsLayerCARemote, isGraphicsLayerCARemote())
