133 using KeyframeValue = Variant<float, WebCore::Color, WebCore::FloatPoint3D, WebCore::TransformationMatrix, RefPtr<WebCore::FilterOperation>>;
134
135 struct Properties {
136 Properties()
137 : animationType(Basic)
138 , beginTime(0)
139 , duration(0)
140 , timeOffset(0)
141 , repeatCount(1)
142 , speed(1)
143 , fillMode(NoFillMode)
144 , valueFunction(NoValueFunction)
145 , autoReverses(false)
146 , removedOnCompletion(true)
147 , additive(false)
148 , reverseTimingFunctions(false)
149 , hasExplicitBeginTime(false)
150 {
151 }

168
169 bool autoReverses;
170 bool removedOnCompletion;
171 bool additive;
172 bool reverseTimingFunctions;
173 bool hasExplicitBeginTime;
174
175 // For basic animations, these vectors have two entries. For keyframe animations, two or more.

177 Vector<KeyframeValue> keyValues;
178 Vector<float> keyTimes;
179 Vector<RefPtr<WebCore::TimingFunction>> timingFunctions;

181 Vector<Properties> animations;
