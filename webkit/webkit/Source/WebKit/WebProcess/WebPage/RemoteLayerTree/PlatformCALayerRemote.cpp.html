48Ref<PlatformCALayerRemote> PlatformCALayerRemote::create(LayerType layerType, PlatformCALayerClient* owner, RemoteLayerTreeContext& context)
49{
50 RefPtr<PlatformCALayerRemote> layer;
51
52 if (layerType == LayerTypeTiledBackingLayer || layerType == LayerTypePageTiledBackingLayer)
53 layer = adoptRef(new PlatformCALayerRemoteTiledBacking(layerType, owner, context));
54 else
55 layer = adoptRef(new PlatformCALayerRemote(layerType, owner, context));
56
57 context.layerDidEnterContext(*layer, layerType);
58
59 return layer.releaseNonNull();
60}

100Ref<PlatformCALayer> PlatformCALayerRemote::clone(PlatformCALayerClient* owner) const
101{
102 auto clone = PlatformCALayerRemote::create(*this, owner, *m_context);
103
104 updateClonedLayerProperties(clone);
105
106 clone->setClonedLayer(this);
107 return WTFMove(clone);
108}

137void PlatformCALayerRemote::updateClonedLayerProperties(PlatformCALayerRemote& clone, bool copyContents) const
138{
139 clone.setPosition(position());
140 clone.setBounds(bounds());
141 clone.setAnchorPoint(anchorPoint());
142
143 if (m_properties.transform)
144 clone.setTransform(*m_properties.transform);
145
146 if (m_properties.sublayerTransform)
147 clone.setSublayerTransform(*m_properties.sublayerTransform);
148
149 if (copyContents)
150 clone.setContents(contents());
151
152 clone.setMasksToBounds(masksToBounds());
153 clone.setDoubleSided(isDoubleSided());
154 clone.setOpaque(isOpaque());
155 clone.setBackgroundColor(backgroundColor());
156 clone.setContentsScale(contentsScale());
157 clone.setCornerRadius(cornerRadius());
158
159 if (!m_properties.shapePath.isNull())
160 clone.setShapePath(m_properties.shapePath);
161
162 if (m_properties.shapeRoundedRect)
163 clone.setShapeRoundedRect(*m_properties.shapeRoundedRect);
164
165 if (m_properties.filters)
166 clone.copyFiltersFrom(*this);
167
168 clone.updateCustomAppearance(customAppearance());
169}

171void PlatformCALayerRemote::recursiveBuildTransaction(RemoteLayerTreeContext& context, RemoteLayerTreeTransaction& transaction)
172{
173 ASSERT(!m_properties.backingStore || owner());
174 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(&context == m_context);
175
176 if (m_properties.backingStore && (!owner() || !owner()->platformCALayerDrawsContent())) {
177 m_properties.backingStore = nullptr;
178 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::BackingStoreChanged);
179 }
180
181 if (m_properties.backingStore && m_properties.backingStoreAttached && m_properties.backingStore->display())
182 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::BackingStoreChanged);
183
184 if (m_properties.changedProperties) {
185 if (m_properties.changedProperties & RemoteLayerTreeTransaction::ChildrenChanged) {
186 m_properties.children.resize(m_children.size());
187 for (size_t i = 0; i < m_children.size(); ++i)
188 m_properties.children[i] = m_children[i]->layerID();
189 }
190
191 if (isPlatformCALayerRemoteCustom()) {
192 RemoteLayerTreePropertyApplier::applyPropertiesToLayer(platformLayer(), nullptr, m_properties, RemoteLayerBackingStore::LayerContentsType::CAMachPort);
193 didCommit();
194 return;
195 }
196
197 transaction.layerPropertiesChanged(*this);
198 }
199
200 for (size_t i = 0; i < m_children.size(); ++i) {
201 PlatformCALayerRemote& child = downcast<PlatformCALayerRemote>(*m_children[i]);
202 ASSERT(child.superlayer() == this);
203 child.recursiveBuildTransaction(context, transaction);
204 }
205
206 if (m_maskLayer)
207 m_maskLayer->recursiveBuildTransaction(context, transaction);
208}

210void PlatformCALayerRemote::didCommit()
211{
212 m_properties.addedAnimations.clear();
213 m_properties.keyPathsOfAnimationsToRemove.clear();
214 m_properties.resetChangedProperties();

217void PlatformCALayerRemote::ensureBackingStore()
218{
219 ASSERT(owner());
220
221 ASSERT(m_properties.backingStoreAttached);
222
223 if (!m_properties.backingStore)
224 m_properties.backingStore = makeUnique<RemoteLayerBackingStore>(this);
225
226 updateBackingStore();
227}
228
229void PlatformCALayerRemote::updateBackingStore()
230{
231 if (!m_properties.backingStore)
232 return;
233
234 ASSERT(m_properties.backingStoreAttached);
235
236 auto type = m_acceleratesDrawing ? RemoteLayerBackingStore::Type::IOSurface : RemoteLayerBackingStore::Type::Bitmap;
237#if ENABLE(CG_DISPLAY_LIST_BACKED_IMAGE_BUFFER)
238 if (m_context->useCGDisplayListsForDOMRendering())
239 type = RemoteLayerBackingStore::Type::CGDisplayList;
240#endif
241 m_properties.backingStore->ensureBackingStore(type, m_properties.bounds.size(), m_properties.contentsScale, m_wantsDeepColorBackingStore, m_properties.opaque);
242}

288void PlatformCALayerRemote::removeSublayer(PlatformCALayerRemote* layer)
289{

297void PlatformCALayerRemote::setSublayers(const PlatformCALayerList& list)
298{
299 removeAllSublayers();
300 m_children = list;
301
302 for (const auto& layer : list) {
303 layer->removeFromSuperlayer();
304 downcast<PlatformCALayerRemote>(*layer).m_superlayer = this;
305 }
306
307 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::ChildrenChanged);
308}
309
310void PlatformCALayerRemote::removeAllSublayers()
311{
312 PlatformCALayerList layersToRemove = m_children;
313 for (const auto& layer : layersToRemove)
314 layer->removeFromSuperlayer();
315 ASSERT(m_children.isEmpty());
316 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::ChildrenChanged);
317}

339void PlatformCALayerRemote::replaceSublayer(PlatformCALayer& reference, PlatformCALayer& layer)
340{
341 ASSERT(reference.superlayer() == this);
342 Ref<PlatformCALayer> protectedLayer(layer);
343
344 layer.removeFromSuperlayer();
345 size_t referenceIndex = m_children.find(&reference);
346 if (referenceIndex != notFound) {
347 m_children[referenceIndex]->removeFromSuperlayer();
348 m_children.insert(referenceIndex, &layer);
349 downcast<PlatformCALayerRemote>(layer).m_superlayer = this;
350 }
351
352 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::ChildrenChanged);
353}

355void PlatformCALayerRemote::adoptSublayers(PlatformCALayer& source)
356{
357 PlatformCALayerList layersToMove = downcast<PlatformCALayerRemote>(source).m_children;
358
359 if (const PlatformCALayerList* customLayers = source.customSublayers()) {
360 for (const auto& layer : *customLayers) {
361 size_t layerIndex = layersToMove.find(layer);
362 if (layerIndex != notFound)
363 layersToMove.remove(layerIndex);
364 }
365 }
366
367 setSublayers(layersToMove);
368}

370void PlatformCALayerRemote::addAnimationForKey(const String& key, PlatformCAAnimation& animation)
371{
372 auto addResult = m_animations.set(key, &animation);
373 bool appendToAddedAnimations = true;
374 if (!addResult.isNewEntry) {
375 // There is already an animation for this key. If the animation has not been sent to the UI
376 // process yet, we update the key properties before it happens. Otherwise, we just append it
377 // to the list of animations to be added: PlatformCAAnimationRemote::updateLayerAnimations
378 // will actually take care of replacing the existing animation.
379 for (auto& keyAnimationPair : m_properties.addedAnimations) {
380 if (keyAnimationPair.first == key) {
381 keyAnimationPair.second = downcast<PlatformCAAnimationRemote>(animation).properties();
382 appendToAddedAnimations = false;
383 break;
384 }
385 }
386 }
387 if (appendToAddedAnimations)
388 m_properties.addedAnimations.append(std::pair<String, PlatformCAAnimationRemote::Properties>(key, downcast<PlatformCAAnimationRemote>(animation).properties()));
389
390 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::AnimationsChanged);
391
392 if (m_context)
393 m_context->willStartAnimationOnLayer(*this);
394}
395
396void PlatformCALayerRemote::removeAnimationForKey(const String& key)
397{
398 if (m_animations.remove(key)) {
399 m_properties.addedAnimations.removeFirstMatching([&key](auto& pair) {
400 return pair.first == key;
401 });
402 }
403 m_properties.keyPathsOfAnimationsToRemove.add(key);
404 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::AnimationsChanged);
405}
406
407RefPtr<PlatformCAAnimation> PlatformCALayerRemote::animationForKey(const String& key)
408{
409 return m_animations.get(key);
410}

412static inline bool isEquivalentLayer(const PlatformCALayer* layer, GraphicsLayer::PlatformLayerID layerID)
413{
414 GraphicsLayer::PlatformLayerID newLayerID = layer ? layer->layerID() : 0;
415 return layerID == newLayerID;
416}

418void PlatformCALayerRemote::animationStarted(const String& key, MonotonicTime beginTime)
419{
420 auto it = m_animations.find(key);
421 if (it != m_animations.end())
422 downcast<PlatformCAAnimationRemote>(*it->value).didStart(currentTimeToMediaTime(beginTime));
423
424 if (m_owner)
425 m_owner->platformCALayerAnimationStarted(key, beginTime);
426}

434void PlatformCALayerRemote::setMask(PlatformCALayer* layer)
435{
436 if (isEquivalentLayer(layer, m_properties.maskLayerID))
437 return;
438
439 if (layer) {
440 m_maskLayer = downcast<PlatformCALayerRemote>(layer);
441 m_properties.maskLayerID = m_maskLayer->layerID();
442 } else {
443 m_maskLayer = nullptr;
444 m_properties.maskLayerID = 0;
445 }
446
447 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::MaskLayerChanged);
448}

450void PlatformCALayerRemote::setClonedLayer(const PlatformCALayer* layer)

453 return;
454
455 if (layer)
456 m_properties.clonedLayerID = layer->layerID();
457 else
458 m_properties.clonedLayerID = 0;
459
460 m_properties.notePropertiesChanged(RemoteLayerTreeTransaction::ClonedContentsChanged);
461}
