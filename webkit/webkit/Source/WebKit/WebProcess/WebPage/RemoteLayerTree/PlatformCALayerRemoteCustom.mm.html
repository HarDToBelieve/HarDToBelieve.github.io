46Ref<PlatformCALayerRemote> PlatformCALayerRemoteCustom::create(PlatformLayer *platformLayer, PlatformCALayerClient* owner, RemoteLayerTreeContext& context)
47{
48 auto layer = adoptRef(*new PlatformCALayerRemoteCustom(PlatformCALayerCocoa::layerTypeForPlatformLayer(platformLayer), platformLayer, owner, context));
49 context.layerDidEnterContext(layer.get(), layer->layerType());
50 return WTFMove(layer);
51}

53PlatformCALayerRemoteCustom::PlatformCALayerRemoteCustom(LayerType layerType, PlatformLayer * customLayer, PlatformCALayerClient* owner, RemoteLayerTreeContext& context)
54 : PlatformCALayerRemote(layerType, owner, context)
55{
56 switch (context.layerHostingMode()) {
57 case LayerHostingMode::InProcess:
58 m_layerHostingContext = LayerHostingContext::createForPort(WebProcess::singleton().compositingRenderServerPort());
59 break;
60#if HAVE(OUT_OF_PROCESS_LAYER_HOSTING)
61 case LayerHostingMode::OutOfProcess:
62 m_layerHostingContext = LayerHostingContext::createForExternalHostingProcess({
63#if PLATFORM(IOS_FAMILY)
64 context.canShowWhileLocked()
65#endif
66 });
67#if PLATFORM(IOS_FAMILY)
68 if (layerType == LayerTypeAVPlayerLayer) {
69 float scaleFactor = context.deviceScaleFactor();
70 // Set a scale factor here to make convertRect:toLayer:nil take scale factor into account. <rdar://problem/18316542>.
71 // This scale factor is inverted in the hosting process.
72 [customLayer setTransform:CATransform3DMakeScale(scaleFactor, scaleFactor, 1)];
73 }
74#endif
75 break;
76#endif
77 }
78
79 m_layerHostingContext->setRootLayer(customLayer);
80 [customLayer setValue:[NSValue valueWithPointer:this] forKey:platformCALayerPointer];
81
82 m_platformLayer = customLayer;
83 [customLayer web_disableAllActions];
84
85 m_providesContents = layerType == LayerTypeContentsProvidedLayer;
86
87 properties().position = FloatPoint3D(customLayer.position.x, customLayer.position.y, customLayer.zPosition);
88 properties().anchorPoint = FloatPoint3D(customLayer.anchorPoint.x, customLayer.anchorPoint.y, customLayer.anchorPointZ);
89 properties().bounds = customLayer.bounds;
90 properties().contentsRect = customLayer.contentsRect;
91}

93PlatformCALayerRemoteCustom::~PlatformCALayerRemoteCustom()
94{
95 [m_platformLayer setValue:nil forKey:platformCALayerPointer];
96}

110Ref<WebCore::PlatformCALayer> PlatformCALayerRemoteCustom::clone(PlatformCALayerClient* owner) const
111{
112 RetainPtr<CALayer> clonedLayer;
113 bool copyContents = true;
114
115 if (layerType() == LayerTypeAVPlayerLayer) {
116
117 if (PAL::isAVFoundationFrameworkAvailable() && [platformLayer() isKindOfClass:PAL::getAVPlayerLayerClass()]) {
118 clonedLayer = adoptNS([PAL::allocAVPlayerLayerInstance() init]);
119
120 AVPlayerLayer *destinationPlayerLayer = static_cast<AVPlayerLayer *>(clonedLayer.get());
121 AVPlayerLayer *sourcePlayerLayer = static_cast<AVPlayerLayer *>(platformLayer());
122 RunLoop::main().dispatch([destinationPlayerLayer, sourcePlayerLayer] {
123 [destinationPlayerLayer setPlayer:[sourcePlayerLayer player]];
124 });
125 } else {
126 // On iOS, the AVPlayerLayer is inside a WebVideoContainerLayer. This code needs to share logic with MediaPlayerPrivateAVFoundationObjC::createAVPlayerLayer().
127 clonedLayer = adoptNS([[CALayer alloc] init]);
128 }
129
130 copyContents = false;
131 } else if (layerType() == LayerTypeContentsProvidedLayer) {
132 clonedLayer = adoptNS([[CALayer alloc] init]);
133 // FIXME: currently copying WebGL contents breaks the original layer.
134 copyContents = false;
135 }
136
137 auto clone = adoptRef(*new PlatformCALayerRemoteCustom(layerType(), clonedLayer.get(), owner, *context()));
138 context()->layerDidEnterContext(clone.get(), clone->layerType());
139
140 updateClonedLayerProperties(clone.get(), copyContents);
141
142 clone->setClonedLayer(this);
143 return WTFMove(clone);
144}
