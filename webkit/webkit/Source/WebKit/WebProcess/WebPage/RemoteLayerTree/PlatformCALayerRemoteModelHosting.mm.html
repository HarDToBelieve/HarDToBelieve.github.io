37Ref<PlatformCALayerRemote> PlatformCALayerRemoteModelHosting::create(Ref<WebCore::Model> model, WebCore::PlatformCALayerClient* owner, RemoteLayerTreeContext& context)
38{
39 auto layer = adoptRef(*new PlatformCALayerRemoteModelHosting(model, owner, context));
40 context.layerDidEnterContext(layer.get(), layer->layerType());
41 return WTFMove(layer);
42}
43
44PlatformCALayerRemoteModelHosting::PlatformCALayerRemoteModelHosting(Ref<WebCore::Model> model, WebCore::PlatformCALayerClient* owner, RemoteLayerTreeContext& context)
45 : PlatformCALayerRemote(WebCore::PlatformCALayer::LayerTypeModelLayer, owner, context)
46 , m_model(model)
47{
48}
49
50PlatformCALayerRemoteModelHosting::~PlatformCALayerRemoteModelHosting() = default;
51
52Ref<WebCore::PlatformCALayer> PlatformCALayerRemoteModelHosting::clone(WebCore::PlatformCALayerClient* owner) const
53{
54 auto clone = adoptRef(*new PlatformCALayerRemoteModelHosting(m_model, owner, *context()));
55 context()->layerDidEnterContext(clone.get(), clone->layerType());
56
57 updateClonedLayerProperties(clone.get(), false);
58
59 clone->setClonedLayer(this);
60 return WTFMove(clone);
61}

63void PlatformCALayerRemoteModelHosting::populateCreationProperties(RemoteLayerTreeTransaction::LayerCreationProperties& properties, const RemoteLayerTreeContext& context, PlatformCALayer::LayerType type)
64{
65 PlatformCALayerRemote::populateCreationProperties(properties, context, type);
66 properties.model = m_model.ptr();
67}

69void PlatformCALayerRemoteModelHosting::dumpAdditionalProperties(TextStream& ts, OptionSet<PlatformLayerTreeAsTextFlags> flags)
70{
71 if (flags.contains(PlatformLayerTreeAsTextFlags::IncludeModels))
72 ts << indent << "(model data size " << m_model->data()->size() << ")\n";
73}
