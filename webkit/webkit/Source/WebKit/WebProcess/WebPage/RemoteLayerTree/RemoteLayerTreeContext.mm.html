44RemoteLayerTreeContext::RemoteLayerTreeContext(WebPage& webPage)
45 : m_webPage(webPage)
46 , m_currentTransaction(nullptr)

48}

50RemoteLayerTreeContext::~RemoteLayerTreeContext()
51{
52 for (auto& layer : m_livePlatformLayers.values())
53 layer->clearContext();
54
55 auto graphicsLayers = m_liveGraphicsLayers;
56 for (auto& layer : graphicsLayers)
57 layer->clearContext();
58}
59
60void RemoteLayerTreeContext::adoptLayersFromContext(RemoteLayerTreeContext& oldContext)
61{
62 auto& platformLayers = oldContext.m_livePlatformLayers;
63 while (!platformLayers.isEmpty())
64 platformLayers.begin()->value->moveToContext(*this);
65
66 auto& graphicsLayers = oldContext.m_liveGraphicsLayers;
67 while (!graphicsLayers.isEmpty())
68 (*graphicsLayers.begin())->moveToContext(*this);
69}

88void RemoteLayerTreeContext::layerDidEnterContext(PlatformCALayerRemote& layer, PlatformCALayer::LayerType type)
89{
90 GraphicsLayer::PlatformLayerID layerID = layer.layerID();
91
92 RemoteLayerTreeTransaction::LayerCreationProperties creationProperties;
93 layer.populateCreationProperties(creationProperties, *this, type);
94
95 m_createdLayers.add(layerID, WTFMove(creationProperties));
96 m_livePlatformLayers.add(layerID, &layer);
97}
98
99void RemoteLayerTreeContext::layerWillLeaveContext(PlatformCALayerRemote& layer)
100{
101 ASSERT(layer.layerID());
102 GraphicsLayer::PlatformLayerID layerID = layer.layerID();
103
104 m_createdLayers.remove(layerID);
105 m_livePlatformLayers.remove(layerID);
106
107 ASSERT(!m_destroyedLayers.contains(layerID));
108 m_destroyedLayers.append(layerID);
109
110 m_layersWithAnimations.remove(layerID);
111}

143void RemoteLayerTreeContext::buildTransaction(RemoteLayerTreeTransaction& transaction, PlatformCALayer& rootLayer)
144{
145 TraceScope tracingScope(BuildTransactionStart, BuildTransactionEnd);
146
147 PlatformCALayerRemote& rootLayerRemote = downcast<PlatformCALayerRemote>(rootLayer);
148 transaction.setRootLayerID(rootLayerRemote.layerID());
149
150 m_currentTransaction = &transaction;
151 rootLayerRemote.recursiveBuildTransaction(*this, transaction);
152 m_currentTransaction = nullptr;
153
154 transaction.setCreatedLayers(copyToVector(m_createdLayers.values()));
155 transaction.setDestroyedLayerIDs(WTFMove(m_destroyedLayers));
156
157 m_createdLayers.clear();
158}

171void RemoteLayerTreeContext::animationDidStart(WebCore::GraphicsLayer::PlatformLayerID layerID, const String& key, MonotonicTime startTime)
172{
173 auto it = m_layersWithAnimations.find(layerID);
174 if (it != m_layersWithAnimations.end())
175 it->value->animationStarted(key, startTime);
176}
177
178void RemoteLayerTreeContext::animationDidEnd(WebCore::GraphicsLayer::PlatformLayerID layerID, const String& key)
179{
180 auto it = m_layersWithAnimations.find(layerID);
181 if (it != m_layersWithAnimations.end())
182 it->value->animationEnded(key);
183}
