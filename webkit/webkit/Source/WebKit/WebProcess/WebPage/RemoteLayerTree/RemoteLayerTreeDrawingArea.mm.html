60RemoteLayerTreeDrawingArea::RemoteLayerTreeDrawingArea(WebPage& webPage, const WebPageCreationParameters& parameters)
61 : DrawingArea(DrawingAreaTypeRemoteLayerTree, parameters.drawingAreaIdentifier, webPage)
62 , m_remoteLayerTreeContext(makeUnique<RemoteLayerTreeContext>(webPage))
63 , m_rootLayer(GraphicsLayer::create(graphicsLayerFactory(), *this))
64 , m_updateRenderingTimer(*this, &RemoteLayerTreeDrawingArea::updateRendering)
65{
66 webPage.corePage()->settings().setForceCompositingMode(true);
67 m_rootLayer->setName(MAKE_STATIC_STRING_IMPL("drawing area root"));
68
69 m_commitQueue = adoptOSObject(dispatch_queue_create("com.apple.WebKit.WebContent.RemoteLayerTreeDrawingArea.CommitQueue", nullptr));
70
71 if (auto viewExposedRect = parameters.viewExposedRect)
72 setViewExposedRect(viewExposedRect);
73}

193void RemoteLayerTreeDrawingArea::setLayerTreeStateIsFrozen(bool isFrozen)
194{
195 if (m_isRenderingSuspended == isFrozen)
196 return;
197
198 tracePoint(isFrozen ? LayerTreeFreezeStart : LayerTreeFreezeEnd);
199
200 m_isRenderingSuspended = isFrozen;
201
202 if (!m_isRenderingSuspended && m_hasDeferredRenderingUpdate) {
203 m_hasDeferredRenderingUpdate = false;
204 startRenderingUpdateTimer();
205 }
206}
207
208void RemoteLayerTreeDrawingArea::forceRepaint()
209{
210 if (m_isRenderingSuspended)
211 return;
212
213 for (Frame* frame = &m_webPage.corePage()->mainFrame(); frame; frame = frame->tree().traverseNext()) {
214 FrameView* frameView = frame->view();
215 if (!frameView || !frameView->tiledBacking())
216 continue;
217
218 frameView->tiledBacking()->forceRepaint();
219 }
220
221 updateRendering();
222}

286void RemoteLayerTreeDrawingArea::addCommitHandlers()
287{
288 if (m_webPage.firstFlushAfterCommit())
289 return;
290
291 [CATransaction addCommitHandler:[retainedPage = makeRefPtr(&m_webPage)] {
292 if (Page* corePage = retainedPage->corePage()) {
293 if (Frame* coreFrame = retainedPage->mainFrame())
294 corePage->inspectorController().willComposite(*coreFrame);
295 }
296 } forPhase:kCATransactionPhasePreCommit];
297
298 [CATransaction addCommitHandler:[retainedPage = makeRefPtr(&m_webPage)] {
299 if (Page* corePage = retainedPage->corePage()) {
300 if (Frame* coreFrame = retainedPage->mainFrame())
301 corePage->inspectorController().didComposite(*coreFrame);
302 }
303 retainedPage->setFirstFlushAfterCommit(false);
304 } forPhase:kCATransactionPhasePostCommit];
305
306 m_webPage.setFirstFlushAfterCommit(true);
307}

309void RemoteLayerTreeDrawingArea::updateRendering()
310{
311 if (m_isRenderingSuspended) {
312 m_hasDeferredRenderingUpdate = true;
313 return;
314 }
315
316 if (m_waitingForBackingStoreSwap) {
317 m_deferredRenderingUpdateWhileWaitingForBackingStoreSwap = true;
318 return;
319 }
320
321 // This function is not reentrant, e.g. a rAF callback may force repaint.
322 if (m_inUpdateRendering)
323 return;
324
325 SetForScope<bool> change(m_inUpdateRendering, true);
326 m_webPage.updateRendering();
327
328 FloatRect visibleRect(FloatPoint(), m_viewSize);
329 if (auto exposedRect = m_webPage.mainFrameView()->viewExposedRect())
330 visibleRect.intersect(*exposedRect);
331
332 addCommitHandlers();
333
334 OptionSet<FinalizeRenderingUpdateFlags> flags;
335 if (m_nextRenderingUpdateRequiresSynchronousImageDecoding)
336 flags.add(FinalizeRenderingUpdateFlags::InvalidateImagesWithAsyncDecodes);
337
338 m_webPage.finalizeRenderingUpdate(flags);
339
340 // Because our view-relative overlay root layer is not attached to the FrameView's GraphicsLayer tree, we need to flush it manually.
341 if (m_viewOverlayRootLayer)
342 m_viewOverlayRootLayer->flushCompositingState(visibleRect);
343
344 RELEASE_ASSERT(!m_pendingBackingStoreFlusher || m_pendingBackingStoreFlusher->hasFlushed());
345
346 RemoteLayerBackingStoreCollection& backingStoreCollection = m_remoteLayerTreeContext->backingStoreCollection();
347 backingStoreCollection.willFlushLayers();
348
349 m_rootLayer->flushCompositingStateForThisLayerOnly();
350
351 // FIXME: Minimize these transactions if nothing changed.
352 RemoteLayerTreeTransaction layerTransaction;
353 layerTransaction.setTransactionID(takeNextTransactionID());
354 layerTransaction.setCallbackIDs(WTFMove(m_pendingCallbackIDs));
355
356 m_remoteLayerTreeContext->setNextRenderingUpdateRequiresSynchronousImageDecoding(m_nextRenderingUpdateRequiresSynchronousImageDecoding);
357 m_remoteLayerTreeContext->buildTransaction(layerTransaction, *downcast<GraphicsLayerCARemote>(m_rootLayer.get()).platformCALayer());
358 m_remoteLayerTreeContext->setNextRenderingUpdateRequiresSynchronousImageDecoding(false);
359
360 backingStoreCollection.willCommitLayerTree(layerTransaction);
361 m_webPage.willCommitLayerTree(layerTransaction);
362
363 layerTransaction.setNewlyReachedPaintingMilestones(m_pendingNewlyReachedPaintingMilestones);
364 m_pendingNewlyReachedPaintingMilestones = { };
365
366 layerTransaction.setActivityStateChangeID(m_activityStateChangeID);
367 m_activityStateChangeID = ActivityStateChangeAsynchronous;
368
369 RemoteScrollingCoordinatorTransaction scrollingTransaction;
370#if ENABLE(ASYNC_SCROLLING)
371 if (m_webPage.scrollingCoordinator())
372 downcast<RemoteScrollingCoordinator>(*m_webPage.scrollingCoordinator()).buildTransaction(scrollingTransaction);
373#endif
374
375 m_nextRenderingUpdateRequiresSynchronousImageDecoding = false;
376 m_waitingForBackingStoreSwap = true;
377
378 send(Messages::RemoteLayerTreeDrawingAreaProxy::WillCommitLayerTree(layerTransaction.transactionID()));
379
380 Messages::RemoteLayerTreeDrawingAreaProxy::CommitLayerTree message(layerTransaction, scrollingTransaction);
381 auto commitEncoder = makeUniqueRef<IPC::Encoder>(Messages::RemoteLayerTreeDrawingAreaProxy::CommitLayerTree::name(), m_identifier.toUInt64());
382 commitEncoder.get() << message.arguments();
383
384 // FIXME: Move all backing store flushing management to RemoteLayerBackingStoreCollection.
385 bool hadAnyChangedBackingStore = false;
386 Vector<std::unique_ptr<WebCore::ThreadSafeImageBufferFlusher>> flushers;
387 for (auto& layer : layerTransaction.changedLayers()) {
388 if (layer->properties().changedProperties & RemoteLayerTreeTransaction::BackingStoreChanged) {
389 hadAnyChangedBackingStore = true;
390 if (layer->properties().backingStore) {
391 if (auto pendingFlusher = layer->properties().backingStore->takePendingFlusher())
392 flushers.append(WTFMove(pendingFlusher));
393 }
394 }
395
396 layer->didCommit();
397 }
398
399 backingStoreCollection.didFlushLayers();
400
401 if (hadAnyChangedBackingStore)
402 backingStoreCollection.scheduleVolatilityTimer();
403
404 RefPtr<BackingStoreFlusher> backingStoreFlusher = BackingStoreFlusher::create(WebProcess::singleton().parentProcessConnection(), WTFMove(commitEncoder), WTFMove(flushers));
405 m_pendingBackingStoreFlusher = backingStoreFlusher;
406
407 auto pageID = m_webPage.identifier();
408 dispatch_async(m_commitQueue.get(), [backingStoreFlusher = WTFMove(backingStoreFlusher), pageID] {
409 backingStoreFlusher->flush();
410
411 MonotonicTime timestamp = MonotonicTime::now();
412 RunLoop::main().dispatch([pageID, timestamp] {
413 if (WebPage* webPage = WebProcess::singleton().webPage(pageID))
414 webPage->didFlushLayerTreeAtTime(timestamp);
415 });
416 });
417}

419void RemoteLayerTreeDrawingArea::didUpdate()
420{
421 // FIXME: This should use a counted replacement for setLayerTreeStateIsFrozen, but
422 // the callers of that function are not strictly paired.
423
424 m_waitingForBackingStoreSwap = false;
425
426 if (m_deferredRenderingUpdateWhileWaitingForBackingStoreSwap) {
427 triggerRenderingUpdate();
428 m_deferredRenderingUpdateWhileWaitingForBackingStoreSwap = false;
429 }
430
431 // This empty transaction serves to trigger CA's garbage collection of IOSurfaces. See <rdar://problem/16110687>
432 [CATransaction begin];
433 [CATransaction commit];
434
435 HashSet<RemoteLayerTreeDisplayRefreshMonitor*> monitorsToNotify = m_displayRefreshMonitors;
436 ASSERT(!m_displayRefreshMonitorsToNotify);
437 m_displayRefreshMonitorsToNotify = &monitorsToNotify;
438 while (!monitorsToNotify.isEmpty())
439 monitorsToNotify.takeAny()->didUpdateLayers();
440 m_displayRefreshMonitorsToNotify = nullptr;
441}

466void RemoteLayerTreeDrawingArea::BackingStoreFlusher::flush()
467{
468 ASSERT(!m_hasFlushed);
469
470 TraceScope tracingScope(BackingStoreFlushStart, BackingStoreFlushEnd);
471
472 for (auto& flusher : m_flushers)
473 flusher->flush();
474 m_hasFlushed = true;
475
476 ASSERT(m_commitEncoder);
477 m_connection->sendMessage(makeUniqueRefFromNonNullUniquePtr(WTFMove(m_commitEncoder)), { });
478}

480void RemoteLayerTreeDrawingArea::activityStateDidChange(OptionSet<WebCore::ActivityState::Flag>, ActivityStateChangeID activityStateChangeID, CompletionHandler<void()>&& callback)
481{
482 // FIXME: Should we suspend painting while not visible, like TiledCoreAnimationDrawingArea? Probably.
483
484 if (activityStateChangeID != ActivityStateChangeAsynchronous) {
485 m_nextRenderingUpdateRequiresSynchronousImageDecoding = true;
486 m_activityStateChangeID = activityStateChangeID;
487 startRenderingUpdateTimer();
488 }
489
490 // FIXME: We may want to match behavior in TiledCoreAnimationDrawingArea by firing these callbacks after the next compositing flush, rather than immediately after
491 // handling an activity state change.
492 callback();
493}

495void RemoteLayerTreeDrawingArea::addTransactionCallbackID(CallbackID callbackID)
496{
497 // Assume that if someone is listening for this transaction's completion, that they want it to
498 // be a "complete" paint (including images that would normally be asynchronously decoding).
499 m_nextRenderingUpdateRequiresSynchronousImageDecoding = true;
500
501 m_pendingCallbackIDs.append(static_cast<RemoteLayerTreeTransaction::TransactionCallbackID>(callbackID));
502 triggerRenderingUpdate();
503}
