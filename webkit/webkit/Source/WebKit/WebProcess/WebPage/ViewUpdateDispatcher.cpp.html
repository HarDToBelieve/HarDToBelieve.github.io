59void ViewUpdateDispatcher::visibleContentRectUpdate(WebCore::PageIdentifier pageID, const VisibleContentRectUpdateInfo& visibleContentRectUpdateInfo)
60{
61 bool updateListWasEmpty;
62 {
63 Locker locker { m_latestUpdateLock };
64 updateListWasEmpty = m_latestUpdate.isEmpty();
65 auto iterator = m_latestUpdate.find(pageID);
66 if (iterator == m_latestUpdate.end())
67 m_latestUpdate.set(pageID, makeUniqueRef<UpdateData>(visibleContentRectUpdateInfo, visibleContentRectUpdateInfo.timestamp()));
68 else
69 iterator->value.get().visibleContentRectUpdateInfo = visibleContentRectUpdateInfo;
70 }
71 if (updateListWasEmpty) {
72 RunLoop::main().dispatch([protectedThis = makeRef(*this)]() mutable {
73 protectedThis->dispatchVisibleContentRectUpdate();
74 });
75 }
76}
77
78void ViewUpdateDispatcher::dispatchVisibleContentRectUpdate()
79{
80 HashMap<WebCore::PageIdentifier, UniqueRef<UpdateData>> update;
81 {
82 Locker locker { m_latestUpdateLock };
83 update = std::exchange(m_latestUpdate, { });
84 }
85
86 for (auto& slot : update) {
87 if (WebPage* webPage = WebProcess::singleton().webPage(slot.key))
88 webPage->updateVisibleContentRects(slot.value.get().visibleContentRectUpdateInfo, slot.value.get().oldestTimestamp);
89 }
90}
