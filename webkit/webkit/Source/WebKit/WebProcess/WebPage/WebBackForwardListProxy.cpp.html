48// FIXME <rdar://problem/8819268>: This leaks all HistoryItems that go into these maps.
49// We need to clear up the life time of these objects.
50
51typedef HashMap<BackForwardItemIdentifier, RefPtr<HistoryItem>> IDToHistoryItemMap; // "ID" here is the item ID.
52static IDToHistoryItemMap& idToHistoryItemMap()
53{
54 static NeverDestroyed<IDToHistoryItemMap> map;
55 return map;
56}
57
58void WebBackForwardListProxy::addItemFromUIProcess(const BackForwardItemIdentifier& itemID, Ref<HistoryItem>&& item, PageIdentifier pageID, OverwriteExistingItem overwriteExistingItem)
59{
60 if (overwriteExistingItem == OverwriteExistingItem::No && idToHistoryItemMap().contains(itemID))
61 return;
62
63 idToHistoryItemMap().set(itemID, item.ptr());
64 clearCachedListCounts();
65}
66
67static void WK2NotifyHistoryItemChanged(HistoryItem& item)
68{
69 WebProcess::singleton().parentProcessConnection()->send(Messages::WebProcessProxy::UpdateBackForwardItem(toBackForwardListItemState(item)), 0);
70}
71
72HistoryItem* WebBackForwardListProxy::itemForID(const BackForwardItemIdentifier& itemID)
73{
74 return idToHistoryItemMap().get(itemID);
75}
76
77void WebBackForwardListProxy::removeItem(const BackForwardItemIdentifier& itemID)
78{
79 RefPtr<HistoryItem> item = idToHistoryItemMap().take(itemID);
80 if (!item)
81 return;
82
83 BackForwardCache::singleton().remove(*item);
84 WebCore::Page::clearPreviousItemFromAllPages(item.get());
85}

87WebBackForwardListProxy::WebBackForwardListProxy(WebPage& page)
88 : m_page(&page)
89{
90 // FIXME: This means that if we mix legacy WebKit and modern WebKit in the same process, we won't get both notifications.
91 WebCore::notifyHistoryItemChanged = WK2NotifyHistoryItemChanged;
92}
93
94void WebBackForwardListProxy::addItem(Ref<HistoryItem>&& item)
95{
96 if (!m_page)
97 return;
98
99 auto result = idToHistoryItemMap().add(item->identifier(), item.ptr());
100 ASSERT_UNUSED(result, result.isNewEntry);
101
102 LOG(BackForward, "(Back/Forward) WebProcess pid %i setting item %p for id %s with url %s", getCurrentProcessID(), item.ptr(), item->identifier().logString(), item->urlString().utf8().data());
103 clearCachedListCounts();
104 m_page->send(Messages::WebPageProxy::BackForwardAddItem(toBackForwardListItemState(item.get())));
105}
106
107void WebBackForwardListProxy::goToItem(HistoryItem& item)
108{
109 if (!m_page)
110 return;
111
112 WebBackForwardListCounts backForwardListCounts;
113 m_page->sendSync(Messages::WebPageProxy::BackForwardGoToItem(item.identifier()), Messages::WebPageProxy::BackForwardGoToItem::Reply(backForwardListCounts));
114 m_cachedBackForwardListCounts = backForwardListCounts;
115}
116
117RefPtr<HistoryItem> WebBackForwardListProxy::itemAtIndex(int itemIndex)
118{
119 if (!m_page)
120 return nullptr;
121
122 std::optional<BackForwardItemIdentifier> itemID;
123 if (!WebProcess::singleton().parentProcessConnection()->sendSync(Messages::WebPageProxy::BackForwardItemAtIndex(itemIndex), Messages::WebPageProxy::BackForwardItemAtIndex::Reply(itemID), m_page->identifier()))
124 return nullptr;
125
126 if (!itemID)
127 return nullptr;
128
129 return idToHistoryItemMap().get(*itemID);
130}

141
142const WebBackForwardListCounts& WebBackForwardListProxy::cacheListCountsIfNecessary() const
143{
144 if (!m_cachedBackForwardListCounts) {
145 WebBackForwardListCounts backForwardListCounts;
146 if (m_page)
147 WebProcess::singleton().parentProcessConnection()->sendSync(Messages::WebPageProxy::BackForwardListCounts(), Messages::WebPageProxy::BackForwardListCounts::Reply(backForwardListCounts), m_page->identifier());
148 m_cachedBackForwardListCounts = backForwardListCounts;
149 }
150 return *m_cachedBackForwardListCounts;
151}
152
153void WebBackForwardListProxy::clearCachedListCounts()
154{
155 m_cachedBackForwardListCounts = std::nullopt;
156}
157
158void WebBackForwardListProxy::close()
159{
160 ASSERT(m_page);
161 m_page = nullptr;
162 m_cachedBackForwardListCounts = WebBackForwardListCounts { };
163}
164
165void WebBackForwardListProxy::clear()
166{
167 m_cachedBackForwardListCounts = WebBackForwardListCounts { }; // Clearing the back/forward list will cause the counts to become 0.
168 m_page->send(Messages::WebPageProxy::BackForwardClear());
169}
