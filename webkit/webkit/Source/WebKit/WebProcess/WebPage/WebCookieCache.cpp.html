37bool WebCookieCache::isSupported()
38{
39#if HAVE(COOKIE_CHANGE_LISTENER_API)
40 // FIXME: This can eventually be removed, this is merely to ensure a smooth transition to the new API.
41 return inMemoryStorageSession().supportsCookieChangeListenerAPI();
42#else
43 return false;
44#endif
45}

47String WebCookieCache::cookiesForDOM(const URL& firstParty, const SameSiteInfo& sameSiteInfo, const URL& url, FrameIdentifier frameID, PageIdentifier pageID, IncludeSecureCookies includeSecureCookies)
48{
49 String host = url.host().toString();
50 if (!m_hostsWithInMemoryStorage.contains(host)) {
51 Vector<Cookie> cookies;
52 bool subscribeToCookieChangeNotifications = true;
53 if (!WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::NetworkConnectionToWebProcess::DomCookiesForHost(url.host().toString(), subscribeToCookieChangeNotifications), Messages::NetworkConnectionToWebProcess::DomCookiesForHost::Reply(cookies), 0))
54 return { };
55 pruneCacheIfNecessary();
56 m_hostsWithInMemoryStorage.add(host);
57 for (auto& cookie : cookies)
58 inMemoryStorageSession().setCookie(cookie);
59 }
60 return inMemoryStorageSession().cookiesForDOM(firstParty, sameSiteInfo, url, frameID, pageID, includeSecureCookies, ShouldAskITP::No, ShouldRelaxThirdPartyCookieBlocking::No).first;
61}
62
63void WebCookieCache::setCookiesFromDOM(const URL& firstParty, const SameSiteInfo& sameSiteInfo, const URL& url, FrameIdentifier frameID, PageIdentifier pageID, const String& cookieString, ShouldRelaxThirdPartyCookieBlocking shouldRelaxThirdPartyCookieBlocking)
64{
65 String host = url.host().toString();
66 if (m_hostsWithInMemoryStorage.contains(host))
67 inMemoryStorageSession().setCookiesFromDOM(firstParty, sameSiteInfo, url, frameID, pageID, ShouldAskITP::No, cookieString, shouldRelaxThirdPartyCookieBlocking);
68}

70void WebCookieCache::cookiesAdded(const String& host, const Vector<Cookie>& cookies)
71{
72 if (!m_hostsWithInMemoryStorage.contains(host))
73 return;
74
75 for (auto& cookie : cookies)
76 inMemoryStorageSession().setCookie(cookie);
77}
78
79void WebCookieCache::cookiesDeleted(const String& host, const Vector<WebCore::Cookie>& cookies)
80{
81 if (!m_hostsWithInMemoryStorage.contains(host))
82 return;
83
84 for (auto& cookie : cookies)
85 inMemoryStorageSession().deleteCookie(cookie);
86}
87
88void WebCookieCache::allCookiesDeleted()
89{
90 clear();
91}
92
93void WebCookieCache::clear()
94{
95#if HAVE(COOKIE_CHANGE_LISTENER_API)
96 if (!m_hostsWithInMemoryStorage.isEmpty())
97 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::UnsubscribeFromCookieChangeNotifications(m_hostsWithInMemoryStorage), 0);
98#endif
99 m_hostsWithInMemoryStorage.clear();
100 m_inMemoryStorageSession = nullptr;
101}
102
103void WebCookieCache::clearForHost(const String& host)
104{
105 String removedHost = m_hostsWithInMemoryStorage.take(host);
106 if (removedHost.isNull())
107 return;
108
109 inMemoryStorageSession().deleteCookiesForHostnames(Vector<String> { removedHost });
110#if HAVE(COOKIE_CHANGE_LISTENER_API)
111 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::UnsubscribeFromCookieChangeNotifications(HashSet<String> { removedHost }), 0);
112#endif
113}
114
115void WebCookieCache::pruneCacheIfNecessary()
116{
117 // We may want to raise this limit if we start using the cache for third-party iframes.
118 static const unsigned maxCachedHosts = 5;
119
120 while (m_hostsWithInMemoryStorage.size() >= maxCachedHosts)
121 clearForHost(*m_hostsWithInMemoryStorage.random());
122}
