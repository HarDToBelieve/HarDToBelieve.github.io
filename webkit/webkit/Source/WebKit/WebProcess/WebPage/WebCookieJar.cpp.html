55#if ENABLE(RESOURCE_LOAD_STATISTICS)
56static bool shouldBlockCookies(WebFrame* frame, const URL& firstPartyForCookies, const URL& resourceURL, ShouldAskITP& shouldAskITPInNetworkProcess)
57{
58 if (!WebCore::DeprecatedGlobalSettings::resourceLoadStatisticsEnabled())
59 return false;
60
61 RegistrableDomain firstPartyDomain { firstPartyForCookies };
62 if (firstPartyDomain.isEmpty())
63 return false;
64
65 RegistrableDomain resourceDomain { resourceURL };
66 if (resourceDomain.isEmpty())
67 return false;
68
69 if (firstPartyDomain == resourceDomain)
70 return false;
71
72 if (frame) {
73 if (frame->frameLoaderClient()->hasFrameSpecificStorageAccess())
74 return false;
75 if (auto* page = frame->page()) {
76 if (page->hasPageLevelStorageAccess(firstPartyDomain, resourceDomain))
77 return false;
78 if (auto* corePage = page->corePage()) {
79 if (corePage->shouldRelaxThirdPartyCookieBlocking() == WebCore::ShouldRelaxThirdPartyCookieBlocking::Yes)
80 return false;
81 }
82 }
83 }
84
85 // The WebContent process does not have enough information to deal with other policies than ThirdPartyCookieBlockingMode::All so we have to go to the NetworkProcess for all
86 // other policies and the request may end up getting blocked on NetworkProcess side.
87 if (WebProcess::singleton().thirdPartyCookieBlockingMode() != ThirdPartyCookieBlockingMode::All) {
88 shouldAskITPInNetworkProcess = ShouldAskITP::Yes;
89 return false;
90 }
91
92 return true;
93}

96bool WebCookieJar::isEligibleForCache(WebFrame& frame, const URL& firstPartyForCookies, const URL& resourceURL) const
97{
98 auto* page = frame.page() ? frame.page()->corePage() : nullptr;
99 if (!page || !page->settings().inProcessCookieCacheEnabled())
100 return false;
101
102 if (!m_cache.isSupported())
103 return false;
104
105 // For now, we only cache cookies for first-party content. Third-party cookie caching is a bit more complicated due to partitioning and storage access.
106 RegistrableDomain resourceDomain { resourceURL };
107 if (resourceDomain.isEmpty())
108 return false;
109
110 return frame.isMainFrame() || RegistrableDomain { firstPartyForCookies } == resourceDomain;
111}

113static WebCore::ShouldRelaxThirdPartyCookieBlocking shouldRelaxThirdPartyCookieBlocking(const WebFrame* frame)
114{
115 if (!frame)
116 return WebCore::ShouldRelaxThirdPartyCookieBlocking::No;
117 auto* page = frame->page();
118 if (!page)
119 return WebCore::ShouldRelaxThirdPartyCookieBlocking::No;
120 auto* corePage = page->corePage();
121 if (!corePage)
122 return WebCore::ShouldRelaxThirdPartyCookieBlocking::No;
123 return corePage->shouldRelaxThirdPartyCookieBlocking();
124}

126String WebCookieJar::cookies(WebCore::Document& document, const URL& url) const
127{
128 auto* webFrame = document.frame() ? WebFrame::fromCoreFrame(*document.frame()) : nullptr;
129 if (!webFrame || !webFrame->page())
130 return { };
131
132 ShouldAskITP shouldAskITPInNetworkProcess = ShouldAskITP::No;
133#if ENABLE(RESOURCE_LOAD_STATISTICS)
134 if (shouldBlockCookies(webFrame, document.firstPartyForCookies(), url, shouldAskITPInNetworkProcess))
135 return { };
136#endif
137
138 auto sameSiteInfo = CookieJar::sameSiteInfo(document, IsForDOMCookieAccess::Yes);
139 auto includeSecureCookies = CookieJar::shouldIncludeSecureCookies(document, url);
140 auto frameID = webFrame->frameID();
141 auto pageID = webFrame->page()->identifier();
142
143 if (isEligibleForCache(*webFrame, document.firstPartyForCookies(), url))
144 return m_cache.cookiesForDOM(document.firstPartyForCookies(), sameSiteInfo, url, frameID, pageID, includeSecureCookies);
145
146 String cookieString;
147 bool secureCookiesAccessed = false;
148 if (!WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::NetworkConnectionToWebProcess::CookiesForDOM(document.firstPartyForCookies(), sameSiteInfo, url, frameID, pageID, includeSecureCookies, shouldAskITPInNetworkProcess, shouldRelaxThirdPartyCookieBlocking(webFrame)), Messages::NetworkConnectionToWebProcess::CookiesForDOM::Reply(cookieString, secureCookiesAccessed), 0))
149 return { };
150
151 return cookieString;
152}

154void WebCookieJar::setCookies(WebCore::Document& document, const URL& url, const String& cookieString)
155{
156 auto* webFrame = document.frame() ? WebFrame::fromCoreFrame(*document.frame()) : nullptr;
157 if (!webFrame || !webFrame->page())
158 return;
159
160 ShouldAskITP shouldAskITPInNetworkProcess = ShouldAskITP::No;
161#if ENABLE(RESOURCE_LOAD_STATISTICS)
162 if (shouldBlockCookies(webFrame, document.firstPartyForCookies(), url, shouldAskITPInNetworkProcess))
163 return;
164#endif
165
166 auto sameSiteInfo = CookieJar::sameSiteInfo(document, IsForDOMCookieAccess::Yes);
167 auto frameID = webFrame->frameID();
168 auto pageID = webFrame->page()->identifier();
169
170 if (isEligibleForCache(*webFrame, document.firstPartyForCookies(), url))
171 m_cache.setCookiesFromDOM(document.firstPartyForCookies(), sameSiteInfo, url, frameID, pageID, cookieString, shouldRelaxThirdPartyCookieBlocking(webFrame));
172
173 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::SetCookiesFromDOM(document.firstPartyForCookies(), sameSiteInfo, url, frameID, pageID, shouldAskITPInNetworkProcess, cookieString, shouldRelaxThirdPartyCookieBlocking(webFrame)), 0);
174}

201bool WebCookieJar::cookiesEnabled(const Document& document) const
202{
203 auto* webFrame = document.frame() ? WebFrame::fromCoreFrame(*document.frame()) : nullptr;
204 if (!webFrame || !webFrame->page())
205 return false;
206
207#if ENABLE(RESOURCE_LOAD_STATISTICS)
208 ShouldAskITP dummy;
209 if (shouldBlockCookies(webFrame, document.firstPartyForCookies(), document.cookieURL(), dummy))
210 return false;
211#endif
212
213 return WebProcess::singleton().ensureNetworkProcessConnection().cookiesEnabled();
214}

216std::pair<String, WebCore::SecureCookiesAccessed> WebCookieJar::cookieRequestHeaderFieldValue(const URL& firstParty, const WebCore::SameSiteInfo& sameSiteInfo, const URL& url, std::optional<FrameIdentifier> frameID, std::optional<PageIdentifier> pageID, WebCore::IncludeSecureCookies includeSecureCookies) const
217{
218 ShouldAskITP shouldAskITPInNetworkProcess = ShouldAskITP::No;
219 auto* webFrame = frameID ? WebProcess::singleton().webFrame(*frameID) : nullptr;
220#if ENABLE(RESOURCE_LOAD_STATISTICS)
221 if (shouldBlockCookies(webFrame, firstParty, url, shouldAskITPInNetworkProcess))
222 return { };
223#endif
224
225 String cookieString;
226 bool secureCookiesAccessed = false;
227 if (!WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::NetworkConnectionToWebProcess::CookieRequestHeaderFieldValue(firstParty, sameSiteInfo, url, frameID, pageID, includeSecureCookies, shouldAskITPInNetworkProcess, shouldRelaxThirdPartyCookieBlocking(webFrame)), Messages::NetworkConnectionToWebProcess::CookieRequestHeaderFieldValue::Reply(cookieString, secureCookiesAccessed), 0))
228 return { };
229 return { cookieString, secureCookiesAccessed ? WebCore::SecureCookiesAccessed::Yes : WebCore::SecureCookiesAccessed::No };
230}

232bool WebCookieJar::getRawCookies(const WebCore::Document& document, const URL& url, Vector<WebCore::Cookie>& rawCookies) const
233{
234 auto* webFrame = document.frame() ? WebFrame::fromCoreFrame(*document.frame()) : nullptr;
235 ShouldAskITP shouldAskITPInNetworkProcess = ShouldAskITP::No;
236#if ENABLE(RESOURCE_LOAD_STATISTICS)
237 if (shouldBlockCookies(webFrame, document.firstPartyForCookies(), url, shouldAskITPInNetworkProcess))
238 return { };
239#endif
240
241 std::optional<FrameIdentifier> frameID = webFrame ? std::make_optional(webFrame->frameID()) : std::nullopt;
242 std::optional<PageIdentifier> pageID = webFrame && webFrame->page() ? std::make_optional(webFrame->page()->identifier()) : std::nullopt;
243 if (!WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::NetworkConnectionToWebProcess::GetRawCookies(document.firstPartyForCookies(), sameSiteInfo(document), url, frameID, pageID, shouldAskITPInNetworkProcess, shouldRelaxThirdPartyCookieBlocking(webFrame)), Messages::NetworkConnectionToWebProcess::GetRawCookies::Reply(rawCookies), 0))
244 return false;
245 return true;
246}
