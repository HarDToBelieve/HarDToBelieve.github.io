32namespace WebKit {
33
34Ref<WebOpenPanelResultListener> WebOpenPanelResultListener::create(WebPage& page, Ref<WebCore::FileChooser>&& fileChooser)
35{
36 return adoptRef(*new WebOpenPanelResultListener(page, WTFMove(fileChooser)));
37}
38
39WebOpenPanelResultListener::WebOpenPanelResultListener(WebPage& page, Ref<WebCore::FileChooser>&& fileChooser)
40 : m_page(&page)
41 , m_fileChooser(WTFMove(fileChooser))
42{
43}
44
45WebOpenPanelResultListener::~WebOpenPanelResultListener()
46{
47}
48
49void WebOpenPanelResultListener::didChooseFiles(const Vector<String>& files, const Vector<String>& replacementFiles)
50{
51 m_fileChooser->chooseFiles(files, replacementFiles);
52}
53
54#if PLATFORM(IOS_FAMILY)
55void WebOpenPanelResultListener::didChooseFilesWithDisplayStringAndIcon(const Vector<String>& files, const String& displayString, WebCore::Icon* displayIcon)
56{
57 m_fileChooser->chooseMediaFiles(files, displayString, displayIcon);
58}
59#endif
