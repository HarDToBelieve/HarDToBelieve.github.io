40namespace WebKit {
41
42class WebFrame;
43class WebPage;
44
45class WebPageOverlay : public API::ObjectImpl<API::Object::Type::BundlePageOverlay>, private WebCore::PageOverlay::Client {
46public:
47 struct ActionContext;
48
49 class Client {
50 public:
51 virtual ~Client() { }
52
53 virtual void willMoveToPage(WebPageOverlay&, WebPage*) = 0;
54 virtual void didMoveToPage(WebPageOverlay&, WebPage*) = 0;
55 virtual void drawRect(WebPageOverlay&, WebCore::GraphicsContext&, const WebCore::IntRect& dirtyRect) = 0;
56 virtual bool mouseEvent(WebPageOverlay&, const WebCore::PlatformMouseEvent&) = 0;
57 virtual void didScrollFrame(WebPageOverlay&, WebFrame*) { }
58
59#if PLATFORM(MAC)
60 virtual std::optional<ActionContext> actionContextForResultAtPoint(WebPageOverlay&, WebCore::FloatPoint) { return std::nullopt; }
61 virtual void dataDetectorsDidPresentUI(WebPageOverlay&) { }
62 virtual void dataDetectorsDidChangeUI(WebPageOverlay&) { }
63 virtual void dataDetectorsDidHideUI(WebPageOverlay&) { }
64#endif
65
66 virtual bool copyAccessibilityAttributeStringValueForPoint(WebPageOverlay&, String /* attribute */, WebCore::FloatPoint /* parameter */, String& /* value */) { return false; }
67 virtual bool copyAccessibilityAttributeBoolValueForPoint(WebPageOverlay&, String /* attribute */, WebCore::FloatPoint /* parameter */, bool& /* value */) { return false; }
68 virtual Vector<String> copyAccessibilityAttributeNames(WebPageOverlay&, bool /* parameterizedNames */) { return Vector<String>(); }
69 };
70
71 static Ref<WebPageOverlay> create(std::unique_ptr<Client>, WebCore::PageOverlay::OverlayType = WebCore::PageOverlay::OverlayType::View);
72 static WebPageOverlay* fromCoreOverlay(WebCore::PageOverlay&);
73 virtual ~WebPageOverlay();
74
75 void setNeedsDisplay(const WebCore::IntRect& dirtyRect);
76 void setNeedsDisplay();
77
78 void clear();
79
80 WebCore::PageOverlay* coreOverlay() const { return m_overlay.get(); }
81 Client& client() const { return *m_client; }
82
83#if PLATFORM(MAC)
84 struct ActionContext {
85 RetainPtr<DDActionContext> context;
86 WebCore::SimpleRange range;
87 };
88 std::optional<ActionContext> actionContextForResultAtPoint(WebCore::FloatPoint);
89 void dataDetectorsDidPresentUI();
90 void dataDetectorsDidChangeUI();
91 void dataDetectorsDidHideUI();
92#endif
93
94private:
95 WebPageOverlay(std::unique_ptr<Client>, WebCore::PageOverlay::OverlayType);
96
97 // WebCore::PageOverlay::Client
98 void willMoveToPage(WebCore::PageOverlay&, WebCore::Page*) override;
99 void didMoveToPage(WebCore::PageOverlay&, WebCore::Page*) override;
100 void drawRect(WebCore::PageOverlay&, WebCore::GraphicsContext&, const WebCore::IntRect& dirtyRect) override;
101 bool mouseEvent(WebCore::PageOverlay&, const WebCore::PlatformMouseEvent&) override;
102 void didScrollFrame(WebCore::PageOverlay&, WebCore::Frame&) override;
103
104 bool copyAccessibilityAttributeStringValueForPoint(WebCore::PageOverlay&, String /* attribute */, WebCore::FloatPoint /* parameter */, String& value) override;
105 bool copyAccessibilityAttributeBoolValueForPoint(WebCore::PageOverlay&, String /* attribute */, WebCore::FloatPoint /* parameter */, bool& value) override;
106 Vector<String> copyAccessibilityAttributeNames(WebCore::PageOverlay&, bool /* parameterizedNames */) override;
107
108 RefPtr<WebCore::PageOverlay> m_overlay;
109 std::unique_ptr<Client> m_client;
110};
