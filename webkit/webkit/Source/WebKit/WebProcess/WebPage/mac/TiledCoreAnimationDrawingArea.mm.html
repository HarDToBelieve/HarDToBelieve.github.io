77TiledCoreAnimationDrawingArea::TiledCoreAnimationDrawingArea(WebPage& webPage, const WebPageCreationParameters& parameters)
78 : DrawingArea(DrawingAreaTypeTiledCoreAnimation, parameters.drawingAreaIdentifier, webPage)
79 , m_isPaintingSuspended(!(parameters.activityState & ActivityState::IsVisible))
80{
81 m_webPage.corePage()->settings().setForceCompositingMode(true);
82
83 m_hostingLayer = [CALayer layer];
84 [m_hostingLayer setDelegate:[WebActionDisablingCALayerDelegate shared]];
85 [m_hostingLayer setFrame:m_webPage.bounds()];
86 [m_hostingLayer setOpaque:YES];
87 [m_hostingLayer setGeometryFlipped:YES];
88
89 m_renderUpdateRunLoopObserver = makeUnique<RunLoopObserver>(static_cast<CFIndex>(RunLoopObserver::WellKnownRunLoopOrders::LayerFlush), [this]() {
90 this->updateRenderingRunLoopCallback();
91 });
92
93 updateLayerHostingContext();
94
95 setColorSpace(parameters.colorSpace);
96
97 if (auto viewExposedRect = parameters.viewExposedRect)
98 setViewExposedRect(viewExposedRect);
99
100 if (!parameters.isProcessSwap)
101 sendEnterAcceleratedCompositingModeIfNeeded();
102}

162void TiledCoreAnimationDrawingArea::forceRepaint()
163{
164 if (m_layerTreeStateIsFrozen)
165 return;
166
167 for (Frame* frame = &m_webPage.corePage()->mainFrame(); frame; frame = frame->tree().traverseNext()) {
168 FrameView* frameView = frame->view();
169 if (!frameView || !frameView->tiledBacking())
170 continue;
171
172 frameView->tiledBacking()->forceRepaint();
173 }
174
175 updateRendering();
176 [CATransaction flush];
177 [CATransaction synchronize];
178}

275void TiledCoreAnimationDrawingArea::scaleViewToFitDocumentIfNeeded()
276{
277 const int maximumDocumentWidthForScaling = 1440;
278 const float minimumViewScale = 0.1;
279
280 if (!m_shouldScaleViewToFitDocument)
281 return;
282
283 LOG(Resize, "TiledCoreAnimationDrawingArea %p scaleViewToFitDocumentIfNeeded", this);
284 m_webPage.layoutIfNeeded();
285
286 if (!m_webPage.mainFrameView() || !m_webPage.mainFrameView()->renderView())
287 return;
288
289 int viewWidth = m_webPage.size().width();
290 int documentWidth = m_webPage.mainFrameView()->renderView()->unscaledDocumentRect().width();
291
292 bool documentWidthChanged = m_lastDocumentSizeForScaleToFit.width() != documentWidth;
293 bool viewWidthChanged = m_lastViewSizeForScaleToFit.width() != viewWidth;
294
295 LOG(Resize, " documentWidthChanged=%d, viewWidthChanged=%d", documentWidthChanged, viewWidthChanged);
296
297 if (!documentWidthChanged && !viewWidthChanged)
298 return;
299
300 // The view is now bigger than the document, so we'll re-evaluate whether we have to scale.
301 if (m_isScalingViewToFitDocument && viewWidth >= m_lastDocumentSizeForScaleToFit.width())
302 m_isScalingViewToFitDocument = false;
303
304 // Our current understanding of the document width is still up to date, and we're in scaling mode.
305 // Update the viewScale without doing an extra layout to re-determine the document width.
306 if (m_isScalingViewToFitDocument) {
307 if (!documentWidthChanged) {
308 m_lastViewSizeForScaleToFit = m_webPage.size();
309 float viewScale = (float)viewWidth / (float)m_lastDocumentSizeForScaleToFit.width();
310 if (viewScale < minimumViewScale) {
311 viewScale = minimumViewScale;
312 documentWidth = std::ceil(viewWidth / viewScale);
313 }
314 IntSize fixedLayoutSize(documentWidth, std::ceil((m_webPage.size().height() - m_webPage.corePage()->topContentInset()) / viewScale));
315 m_webPage.setFixedLayoutSize(fixedLayoutSize);
316 m_webPage.scaleView(viewScale);
317
318 LOG(Resize, " using fixed layout at %dx%d. document width %d unchanged, scaled to %.4f to fit view width %d", fixedLayoutSize.width(), fixedLayoutSize.height(), documentWidth, viewScale, viewWidth);
319 return;
320 }
321
322 IntSize fixedLayoutSize = m_webPage.fixedLayoutSize();
323 if (documentWidth > fixedLayoutSize.width()) {
324 LOG(Resize, " page laid out wider than fixed layout width. Not attempting to re-scale");
325 return;
326 }
327 }
328
329 LOG(Resize, " doing unconstrained layout");
330
331 // Lay out at the view size.
332 m_webPage.setUseFixedLayout(false);
333 m_webPage.layoutIfNeeded();
334
335 if (!m_webPage.mainFrameView() || !m_webPage.mainFrameView()->renderView())
336 return;
337
338 IntSize documentSize = m_webPage.mainFrameView()->renderView()->unscaledDocumentRect().size();
339 m_lastViewSizeForScaleToFit = m_webPage.size();
340 m_lastDocumentSizeForScaleToFit = documentSize;
341
342 documentWidth = documentSize.width();
343
344 float viewScale = 1;
345
346 LOG(Resize, " unscaled document size %dx%d. need to scale down: %d", documentSize.width(), documentSize.height(), documentWidth && documentWidth < maximumDocumentWidthForScaling && viewWidth < documentWidth);
347
348 // Avoid scaling down documents that don't fit in a certain width, to allow
349 // sites that want horizontal scrollbars to continue to have them.
350 if (documentWidth && documentWidth < maximumDocumentWidthForScaling && viewWidth < documentWidth) {
351 // If the document doesn't fit in the view, scale it down but lay out at the view size.
352 m_isScalingViewToFitDocument = true;
353 m_webPage.setUseFixedLayout(true);
354 viewScale = (float)viewWidth / (float)documentWidth;
355 if (viewScale < minimumViewScale) {
356 viewScale = minimumViewScale;
357 documentWidth = std::ceil(viewWidth / viewScale);
358 }
359 IntSize fixedLayoutSize(documentWidth, std::ceil((m_webPage.size().height() - m_webPage.corePage()->topContentInset()) / viewScale));
360 m_webPage.setFixedLayoutSize(fixedLayoutSize);
361
362 LOG(Resize, " using fixed layout at %dx%d. document width %d, scaled to %.4f to fit view width %d", fixedLayoutSize.width(), fixedLayoutSize.height(), documentWidth, viewScale, viewWidth);
363 }
364
365 m_webPage.scaleView(viewScale);
366}

368void TiledCoreAnimationDrawingArea::dispatchAfterEnsuringUpdatedScrollPosition(WTF::Function<void ()>&& function)
369{
370#if ENABLE(ASYNC_SCROLLING)
371 if (!m_webPage.corePage()->scrollingCoordinator()) {
372 function();
373 return;
374 }
375
376 m_webPage.ref();
377 m_webPage.corePage()->scrollingCoordinator()->commitTreeStateIfNeeded();
378
379 if (!m_layerTreeStateIsFrozen)
380 invalidateRenderingUpdateRunLoopObserver();
381
382 // It is possible for the drawing area to be destroyed before the bound block
383 // is invoked, so grab a reference to the web page here so we can access the drawing area through it.
384 // (The web page is already kept alive by dispatchAfterEnsuringUpdatedScrollPosition).
385 WebPage* webPage = &m_webPage;
386
387 ScrollingThread::dispatchBarrier([this, webPage, function = WTFMove(function)] {
388 DrawingArea* drawingArea = webPage->drawingArea();
389 if (!drawingArea)
390 return;
391
392 function();
393
394 if (!m_layerTreeStateIsFrozen)
395 scheduleRenderingUpdateRunLoopObserver();
396
397 webPage->deref();
398 });
399#else
400 function();
401#endif
402}

419void TiledCoreAnimationDrawingArea::addCommitHandlers()
420{
421 if (m_webPage.firstFlushAfterCommit())
422 return;
423
424 [CATransaction addCommitHandler:[retainedPage = makeRefPtr(&m_webPage)] {
425 if (Page* corePage = retainedPage->corePage()) {
426 if (Frame* coreFrame = retainedPage->mainFrame())
427 corePage->inspectorController().willComposite(*coreFrame);
428 }
429 } forPhase:kCATransactionPhasePreCommit];
430
431 [CATransaction addCommitHandler:[retainedPage = makeRefPtr(&m_webPage)] {
432 if (Page* corePage = retainedPage->corePage()) {
433 if (Frame* coreFrame = retainedPage->mainFrame())
434 corePage->inspectorController().didComposite(*coreFrame);
435 }
436 if (auto drawingArea = static_cast<TiledCoreAnimationDrawingArea*>(retainedPage->drawingArea()))
437 drawingArea->sendPendingNewlyReachedPaintingMilestones();
438 retainedPage->setFirstFlushAfterCommit(false);
439 } forPhase:kCATransactionPhasePostCommit];
440
441 m_webPage.setFirstFlushAfterCommit(true);
442}

444void TiledCoreAnimationDrawingArea::updateRendering(UpdateRenderingType flushType)
445{
446 if (layerTreeStateIsFrozen())
447 return;
448
449 @autoreleasepool {
450 scaleViewToFitDocumentIfNeeded();
451
452 m_webPage.updateRendering();
453 m_webPage.flushPendingThemeColorChange();
454 m_webPage.flushPendingPageExtendedBackgroundColorChange();
455 m_webPage.flushPendingSampledPageTopColorChange();
456 m_webPage.flushPendingEditorStateUpdate();
457 m_webPage.flushPendingIntrinsicContentSizeUpdate();
458
459 if (m_pendingRootLayer) {
460 setRootCompositingLayer(m_pendingRootLayer.get());
461 m_pendingRootLayer = nullptr;
462 }
463
464 FloatRect visibleRect = [m_hostingLayer frame];
465 if (auto exposedRect = m_webPage.mainFrameView()->viewExposedRect())
466 visibleRect.intersect(*exposedRect);
467
468 // Because our view-relative overlay root layer is not attached to the main GraphicsLayer tree, we need to flush it manually.
469 if (m_viewOverlayRootLayer)
470 m_viewOverlayRootLayer->flushCompositingState(visibleRect);
471
472 addCommitHandlers();
473
474 OptionSet<FinalizeRenderingUpdateFlags> flags;
475 if (flushType == UpdateRenderingType::Normal)
476 flags.add(FinalizeRenderingUpdateFlags::ApplyScrollingTreeLayerPositions);
477
478 m_webPage.finalizeRenderingUpdate(flags);
479
480 // If we have an active transient zoom, we want the zoom to win over any changes
481 // that WebCore makes to the relevant layers, so re-apply our changes after flushing.
482 if (m_transientZoomScale != 1)
483 applyTransientZoomToLayers(m_transientZoomScale, m_transientZoomOrigin);
484
485 if (!m_pendingCallbackIDs.isEmpty()) {
486 send(Messages::DrawingAreaProxy::DispatchPresentationCallbacksAfterFlushingLayers(m_pendingCallbackIDs));
487 m_pendingCallbackIDs.clear();
488 }
489
490 sendDidFirstLayerFlushIfNeeded();
491 m_webPage.didUpdateRendering();
492 handleActivityStateChangeCallbacksIfNeeded();
493 invalidateRenderingUpdateRunLoopObserver();
494 }
495}

497void TiledCoreAnimationDrawingArea::handleActivityStateChangeCallbacks()
498{
499 if (!m_shouldHandleActivityStateChangeCallbacks)
500 return;
501 m_shouldHandleActivityStateChangeCallbacks = false;
502
503 if (m_activityStateChangeID != ActivityStateChangeAsynchronous)
504 m_webPage.send(Messages::WebPageProxy::DidUpdateActivityState());
505
506 for (auto& callback : std::exchange(m_nextActivityStateChangeCallbacks, { }))
507 callback();
508
509 m_activityStateChangeID = ActivityStateChangeAsynchronous;
510}

512void TiledCoreAnimationDrawingArea::handleActivityStateChangeCallbacksIfNeeded()
513{
514 if (!m_shouldHandleActivityStateChangeCallbacks)
515 return;
516
517 // If there is no active transaction, likely there is no layer change or change is committed,
518 // perform the callbacks immediately, which may unblock UI process.
519 if (![CATransaction currentState]) {
520 handleActivityStateChangeCallbacks();
521 return;
522 }
523
524 [CATransaction addCommitHandler:[weakThis = makeWeakPtr(*this)] {
525 if (!weakThis)
526 return;
527
528 auto protectedPage = makeRef(weakThis->m_webPage);
529 auto* drawingArea = static_cast<TiledCoreAnimationDrawingArea*>(protectedPage->drawingArea());
530 ASSERT(weakThis.get() == drawingArea);
531 if (drawingArea != weakThis.get())
532 return;
533
534 drawingArea->handleActivityStateChangeCallbacks();
535 } forPhase:kCATransactionPhasePostCommit];
536}

538void TiledCoreAnimationDrawingArea::activityStateDidChange(OptionSet<ActivityState::Flag> changed, ActivityStateChangeID activityStateChangeID, CompletionHandler<void()>&& nextActivityStateChangeCallback)
539{
540 m_nextActivityStateChangeCallbacks.append(WTFMove(nextActivityStateChangeCallback));
541 m_activityStateChangeID = std::max(m_activityStateChangeID, activityStateChangeID);
542
543 if (changed & ActivityState::IsVisible) {
544 if (m_webPage.isVisible())
545 resumePainting();
546 else
547 suspendPainting();
548 }
549
550 if (m_activityStateChangeID != ActivityStateChangeAsynchronous || !m_nextActivityStateChangeCallbacks.isEmpty()) {
551 m_shouldHandleActivityStateChangeCallbacks = true;
552 triggerRenderingUpdate();
553 }
554}

598void TiledCoreAnimationDrawingArea::updateGeometry(const IntSize& viewSize, bool flushSynchronously, const WTF::MachSendRight& fencePort)
599{
600 m_inUpdateGeometry = true;
601
602 IntSize size = viewSize;
603 IntSize contentSize = IntSize(-1, -1);
604
605 if (!m_webPage.minimumSizeForAutoLayout().width() || m_webPage.autoSizingShouldExpandToViewHeight() || (!m_webPage.sizeToContentAutoSizeMaximumSize().width() && !m_webPage.sizeToContentAutoSizeMaximumSize().height()))
606 m_webPage.setSize(size);
607
608 FrameView* frameView = m_webPage.mainFrameView();
609
610 if (m_webPage.autoSizingShouldExpandToViewHeight() && frameView)
611 frameView->setAutoSizeFixedMinimumHeight(viewSize.height());
612
613 m_webPage.layoutIfNeeded();
614
615 if (frameView && (m_webPage.minimumSizeForAutoLayout().width() || (m_webPage.sizeToContentAutoSizeMaximumSize().width() && m_webPage.sizeToContentAutoSizeMaximumSize().height()))) {
616 contentSize = frameView->autoSizingIntrinsicContentSize();
617 size = contentSize;
618 }
619
620 updateRendering();
621
622 [CATransaction begin];
623 [CATransaction setDisableActions:YES];
624
625 [m_hostingLayer setFrame:CGRectMake(0, 0, viewSize.width(), viewSize.height())];
626
627 [CATransaction commit];
628
629 if (flushSynchronously)
630 [CATransaction flush];
631
632 send(Messages::DrawingAreaProxy::DidUpdateGeometry());
633
634 m_inUpdateGeometry = false;
635
636 m_layerHostingContext->setFencePort(fencePort.sendRight());
637}

664void TiledCoreAnimationDrawingArea::updateLayerHostingContext()
665{
666 RetainPtr<CGColorSpaceRef> colorSpace;
667
668 // Invalidate the old context.
669 if (m_layerHostingContext) {
670 colorSpace = m_layerHostingContext->colorSpace();
671 m_layerHostingContext->invalidate();
672 m_layerHostingContext = nullptr;
673 }
674
675 // Create a new context and set it up.
676 switch (m_webPage.layerHostingMode()) {
677 case LayerHostingMode::InProcess:
678 m_layerHostingContext = LayerHostingContext::createForPort(WebProcess::singleton().compositingRenderServerPort());
679 break;

783void TiledCoreAnimationDrawingArea::applyTransientZoomToLayers(double scale, FloatPoint origin)
784{
785 // FIXME: Scrollbars should stay in-place and change height while zooming.
786
787 if (!m_hostingLayer)
788 return;
789
790 TransformationMatrix transform;
791 transform.translate(origin.x(), origin.y());
792 transform.scale(scale);
793
794 PlatformCALayer* zoomLayer = layerForTransientZoom();
795 zoomLayer->setTransform(transform);
796 zoomLayer->setAnchorPoint(FloatPoint3D());
797 zoomLayer->setPosition(FloatPoint3D());
798
799 if (PlatformCALayer* shadowLayer = shadowLayerForTransientZoom()) {
800 FrameView& frameView = *m_webPage.mainFrameView();
801 shadowLayer->setBounds(shadowLayerBoundsForFrame(frameView, scale));
802 shadowLayer->setPosition(shadowLayerPositionForFrame(frameView, origin));
803 }
804
805 m_transientZoomScale = scale;
806 m_transientZoomOrigin = origin;
807}

809void TiledCoreAnimationDrawingArea::adjustTransientZoom(double scale, FloatPoint origin)
810{
811 scale *= m_webPage.viewScaleFactor();
812
813 applyTransientZoomToLayers(scale, origin);
814
815 double currentPageScale = m_webPage.totalScaleFactor();
816 if (scale > currentPageScale)
817 return;
818
819 FrameView* frameView = m_webPage.mainFrameView();
820 FloatRect tileCoverageRect = frameView->visibleContentRectIncludingScrollbars();
821 tileCoverageRect.moveBy(-origin);
822 tileCoverageRect.scale(currentPageScale / scale);
823 frameView->renderView()->layer()->backing()->tiledBacking()->prepopulateRect(tileCoverageRect);
824}

839void TiledCoreAnimationDrawingArea::commitTransientZoom(double scale, FloatPoint origin)
840{
841 scale *= m_webPage.viewScaleFactor();
842
843 FrameView& frameView = *m_webPage.mainFrameView();
844 FloatRect visibleContentRect = frameView.visibleContentRectIncludingScrollbars();
845
846 FloatPoint constrainedOrigin = visibleContentRect.location();
847 constrainedOrigin.moveBy(-origin);
848
849 IntSize scaledTotalContentsSize = frameView.totalContentsSize();
850 scaledTotalContentsSize.scale(scale / m_webPage.totalScaleFactor());
851
852 // Scaling may have exposed the overhang area, so we need to constrain the final
853 // layer position exactly like scrolling will once it's committed, to ensure that
854 // scrolling doesn't make the view jump.
855 constrainedOrigin = ScrollableArea::constrainScrollPositionForOverhang(roundedIntRect(visibleContentRect), scaledTotalContentsSize, roundedIntPoint(constrainedOrigin), frameView.scrollOrigin(), frameView.headerHeight(), frameView.footerHeight());
856 constrainedOrigin.moveBy(-visibleContentRect.location());
857 constrainedOrigin = -constrainedOrigin;
858
859 if (m_transientZoomScale == scale && roundedIntPoint(m_transientZoomOrigin) == roundedIntPoint(constrainedOrigin)) {
860 // We're already at the right scale and position, so we don't need to animate.
861 applyTransientZoomToPage(scale, origin);
862 return;
863 }
864
865 TransformationMatrix transform;
866 transform.translate(constrainedOrigin.x(), constrainedOrigin.y());
867 transform.scale(scale);
868
869 RetainPtr<CABasicAnimation> renderViewAnimationCA = transientZoomSnapAnimationForKeyPath("transform");
870 auto renderViewAnimation = PlatformCAAnimationCocoa::create(renderViewAnimationCA.get());
871 renderViewAnimation->setToValue(transform);
872
873 RetainPtr<CALayer> shadowCALayer;
874 if (PlatformCALayer* shadowLayer = shadowLayerForTransientZoom())
875 shadowCALayer = shadowLayer->platformLayer();
876
877 RefPtr<PlatformCALayer> zoomLayer = layerForTransientZoom();
878 RefPtr<WebPage> page = &m_webPage;
879
880 [CATransaction begin];
881 [CATransaction setCompletionBlock:[zoomLayer, shadowCALayer, page, scale, origin] () {
882 zoomLayer->removeAnimationForKey("transientZoomCommit");
883 if (shadowCALayer)
884 [shadowCALayer removeAllAnimations];
885
886 if (TiledCoreAnimationDrawingArea* drawingArea = static_cast<TiledCoreAnimationDrawingArea*>(page->drawingArea()))
887 drawingArea->applyTransientZoomToPage(scale, origin);
888 }];
889
890 zoomLayer->addAnimationForKey("transientZoomCommit", renderViewAnimation.get());
891
892 if (shadowCALayer) {
893 FloatRect shadowBounds = shadowLayerBoundsForFrame(frameView, scale);
894 RetainPtr<CGPathRef> shadowPath = adoptCF(CGPathCreateWithRect(shadowBounds, NULL));
895
896 RetainPtr<CABasicAnimation> shadowBoundsAnimation = transientZoomSnapAnimationForKeyPath("bounds");
897 [shadowBoundsAnimation setToValue:[NSValue valueWithRect:shadowBounds]];
898 RetainPtr<CABasicAnimation> shadowPositionAnimation = transientZoomSnapAnimationForKeyPath("position");
899 [shadowPositionAnimation setToValue:[NSValue valueWithPoint:shadowLayerPositionForFrame(frameView, constrainedOrigin)]];
900 RetainPtr<CABasicAnimation> shadowPathAnimation = transientZoomSnapAnimationForKeyPath("shadowPath");
901 [shadowPathAnimation setToValue:(__bridge id)shadowPath.get()];
902
903 [shadowCALayer addAnimation:shadowBoundsAnimation.get() forKey:@"transientZoomCommitShadowBounds"];
904 [shadowCALayer addAnimation:shadowPositionAnimation.get() forKey:@"transientZoomCommitShadowPosition"];
905 [shadowCALayer addAnimation:shadowPathAnimation.get() forKey:@"transientZoomCommitShadowPath"];
906 }
907
908 [CATransaction commit];
909}

911void TiledCoreAnimationDrawingArea::applyTransientZoomToPage(double scale, FloatPoint origin)
912{
913 // If the page scale is already the target scale, setPageScaleFactor() will short-circuit
914 // and not apply the transform, so we can't depend on it to do so.
915 TransformationMatrix finalTransform;
916 finalTransform.scale(scale);
917 layerForTransientZoom()->setTransform(finalTransform);
918
919 FrameView& frameView = *m_webPage.mainFrameView();
920
921 if (PlatformCALayer* shadowLayer = shadowLayerForTransientZoom()) {
922 shadowLayer->setBounds(shadowLayerBoundsForFrame(frameView, 1));
923 shadowLayer->setPosition(shadowLayerPositionForFrame(frameView, FloatPoint()));
924 }
925
926 FloatPoint unscrolledOrigin(origin);
927 FloatRect unobscuredContentRect = frameView.unobscuredContentRectIncludingScrollbars();
928 unscrolledOrigin.moveBy(-unobscuredContentRect.location());
929 m_webPage.scalePage(scale / m_webPage.viewScaleFactor(), roundedIntPoint(-unscrolledOrigin));
930 m_transientZoomScale = 1;
931 updateRendering(UpdateRenderingType::TransientZoom);
932}
