77ALLOW_DEPRECATED_IMPLEMENTATIONS_BEGIN
78- (NSArray *)accessibilityAttributeNames
79ALLOW_DEPRECATED_IMPLEMENTATIONS_END
80{
81 if (!m_attributeNames)
82 m_attributeNames = adoptNS([[NSArray alloc] initWithObjects:
83 NSAccessibilityRoleAttribute, NSAccessibilityRoleDescriptionAttribute, NSAccessibilityFocusedAttribute,
84 NSAccessibilityParentAttribute, NSAccessibilityWindowAttribute, NSAccessibilityTopLevelUIElementAttribute,
85 NSAccessibilityPositionAttribute, NSAccessibilitySizeAttribute, NSAccessibilityChildrenAttribute, NSAccessibilityPrimaryScreenHeightAttribute, nil]);
86
87 return m_attributeNames.get();
88}

90ALLOW_DEPRECATED_IMPLEMENTATIONS_BEGIN
91- (NSArray *)accessibilityParameterizedAttributeNames
92ALLOW_DEPRECATED_IMPLEMENTATIONS_END
93{
94#if ENABLE(ACCESSIBILITY_ISOLATED_TREE)
95 if (id cachedNames = self.cachedParameterizedAttributeNames)
96 return cachedNames;
97#endif
98
99 auto names = ax::retrieveValueFromMainThread<RetainPtr<id>>([PROTECTED_SELF] () -> RetainPtr<id> {
100 auto page = protectedSelf->m_page;
101 if (!page)
102 return @[];
103
104 auto corePage = page->corePage();
105 if (!corePage)
106 return @[];
107
108 return createNSArray(corePage->pageOverlayController().copyAccessibilityAttributesNames(true));
109 });
110
111#if ENABLE(ACCESSIBILITY_ISOLATED_TREE)
112 self.cachedParameterizedAttributeNames = names.get();
113#endif
114
115 return names.autorelease();
116}

156ALLOW_DEPRECATED_IMPLEMENTATIONS_BEGIN
157- (id)accessibilityAttributeValue:(NSString *)attribute
158ALLOW_DEPRECATED_IMPLEMENTATIONS_END
159{
160 if (!WebCore::AXObjectCache::accessibilityEnabled())
161 WebCore::AXObjectCache::enableAccessibility();
162
163 if ([attribute isEqualToString:NSAccessibilityParentAttribute])
164 return m_parent.get();
165
166 if ([attribute isEqualToString:NSAccessibilityWindowAttribute])
167 return [m_parent accessibilityAttributeValue:NSAccessibilityWindowAttribute];
168
169 if ([attribute isEqualToString:NSAccessibilityTopLevelUIElementAttribute])
170 return [m_parent accessibilityAttributeValue:NSAccessibilityTopLevelUIElementAttribute];
171
172 if ([attribute isEqualToString:NSAccessibilityRoleAttribute])
173 return NSAccessibilityGroupRole;
174
175 if ([attribute isEqualToString:NSAccessibilityRoleDescriptionAttribute])
176 return NSAccessibilityRoleDescription(NSAccessibilityGroupRole, nil);
177
178 if ([attribute isEqualToString:NSAccessibilityFocusedAttribute])
179 return @NO;
180
181 if (!m_pageID)
182 return nil;
183
184 if ([attribute isEqualToString:NSAccessibilityPositionAttribute])
185 return [self accessibilityAttributePositionValue];
186
187 if ([attribute isEqualToString:NSAccessibilityPrimaryScreenHeightAttribute])
188 return @(WebCore::screenRectForPrimaryScreen().size().height());
189
190 if ([attribute isEqualToString:NSAccessibilitySizeAttribute])
191 return [self accessibilityAttributeSizeValue];
192
193 if ([attribute isEqualToString:NSAccessibilityChildrenAttribute])
194 return [self accessibilityChildren];
195
196 return nil;
197}

217- (id)accessibilityDataDetectorValue:(NSString *)attribute point:(WebCore::FloatPoint&)point
218{
219 return ax::retrieveValueFromMainThread<RetainPtr<id>>([&attribute, &point, PROTECTED_SELF] () -> RetainPtr<id> {
220 if (!protectedSelf->m_page)
221 return nil;
222 id value = nil;
223 if ([attribute isEqualToString:@"AXDataDetectorExistsAtPoint"] || [attribute isEqualToString:@"AXDidShowDataDetectorMenuAtPoint"]) {
224 bool boolValue;
225 if (protectedSelf->m_page->corePage()->pageOverlayController().copyAccessibilityAttributeBoolValueForPoint(attribute, point, boolValue))
226 value = [NSNumber numberWithBool:boolValue];
227 }
228 if ([attribute isEqualToString:@"AXDataDetectorTypeAtPoint"]) {
229 String stringValue;
230 if (protectedSelf->m_page->corePage()->pageOverlayController().copyAccessibilityAttributeStringValueForPoint(attribute, point, stringValue))
231 value = [NSString stringWithString:stringValue];
232 }
233 return value;
234 }).autorelease();
235}

237ALLOW_DEPRECATED_IMPLEMENTATIONS_BEGIN
238- (id)accessibilityAttributeValue:(NSString *)attribute forParameter:(id)parameter
239ALLOW_DEPRECATED_IMPLEMENTATIONS_END
240{
241 WebCore::FloatPoint pageOverlayPoint;
242 if ([parameter isKindOfClass:[NSValue class]] && !strcmp([(NSValue *)parameter objCType], @encode(NSPoint)))
243 pageOverlayPoint = [self convertScreenPointToRootView:[(NSValue *)parameter pointValue]];
244 else
245 return nil;
246
247 if ([attribute isEqualToString:@"AXDataDetectorExistsAtPoint"] || [attribute isEqualToString:@"AXDidShowDataDetectorMenuAtPoint"] || [attribute isEqualToString:@"AXDataDetectorTypeAtPoint"])
248 return [self accessibilityDataDetectorValue:attribute point:pageOverlayPoint];
249
250 return nil;
251}

258ALLOW_DEPRECATED_DECLARATIONS_BEGIN
259- (id)accessibilityHitTest:(NSPoint)point
260{
261 auto convertedPoint = ax::retrieveValueFromMainThread<WebCore::IntPoint>([&point, PROTECTED_SELF] () -> WebCore::IntPoint {
262 if (!protectedSelf->m_page)
263 return WebCore::IntPoint(point);
264
265 auto convertedPoint = protectedSelf->m_page->screenToRootView(WebCore::IntPoint(point));
266
267 // Some plugins may be able to figure out the scroll position and inset on their own.
268 bool applyContentOffset = true;
269
270 // Isolated tree frames have the offset encoded into them so we don't need to undo here.
271#if ENABLE(ACCESSIBILITY_ISOLATED_TREE)
272 applyContentOffset = !WebCore::AXObjectCache::isIsolatedTreeEnabled();
273#endif
274 if (auto pluginView = WebKit::WebPage::pluginViewForFrame(protectedSelf->m_page->mainFrame()))
275 applyContentOffset = !pluginView->plugin()->pluginHandlesContentOffsetForAccessibilityHitTest();
276
277 if (!applyContentOffset)
278 return convertedPoint;
279
280 if (auto* frameView = protectedSelf->m_page->mainFrameView())
281 convertedPoint.moveBy(frameView->scrollPosition());
282 if (auto* page = protectedSelf->m_page->corePage())
283 convertedPoint.move(0, -page->topContentInset());
284 return convertedPoint;
285 });
286
287 return [[self accessibilityRootObjectWrapper] accessibilityHitTest:convertedPoint];
288}
