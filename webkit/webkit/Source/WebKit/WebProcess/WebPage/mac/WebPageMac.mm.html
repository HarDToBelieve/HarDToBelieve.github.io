143void WebPage::getPlatformEditorState(Frame& frame, EditorState& result) const
144{
145 getPlatformEditorStateCommon(frame, result);
146
147 if (result.isMissingPostLayoutData)
148 return;
149
150 auto& selection = frame.selection().selection();
151 auto selectedRange = selection.toNormalizedRange();
152 if (!selectedRange)
153 return;
154
155 auto& postLayoutData = result.postLayoutData();
156 VisiblePosition selectionStart = selection.visibleStart();
157 auto selectionStartBoundary = makeBoundaryPoint(selectionStart);
158 auto selectionEnd = makeBoundaryPoint(selection.visibleEnd());
159 auto paragraphStart = makeBoundaryPoint(startOfParagraph(selectionStart));
160
161 if (!selectionStartBoundary || !selectionEnd || !paragraphStart)
162 return;
163
164 auto contextRangeForCandidateRequest = frame.editor().contextRangeForCandidateRequest();
165
166 postLayoutData.candidateRequestStartPosition = characterCount({ *paragraphStart, *selectionStartBoundary });
167 postLayoutData.selectedTextLength = characterCount({ *selectionStartBoundary, *selectionEnd });
168 postLayoutData.paragraphContextForCandidateRequest = contextRangeForCandidateRequest ? plainText(*contextRangeForCandidateRequest) : String();
169 postLayoutData.stringForCandidateRequest = frame.editor().stringForCandidateRequest();
170 postLayoutData.canEnableAutomaticSpellingCorrection = frame.editor().canEnableAutomaticSpellingCorrection();
171
172 auto quads = RenderObject::absoluteTextQuads(*selectedRange);
173 if (!quads.isEmpty())
174 postLayoutData.selectionBoundingRect = frame.view()->contentsToWindow(quads[0].enclosingBoundingBox());
175 else if (selection.isCaret()) {
176 // Quads will be empty at the start of a paragraph.
177 postLayoutData.selectionBoundingRect = frame.view()->contentsToWindow(frame.selection().absoluteCaretBounds());
178 }
179}

211static String commandNameForSelectorName(const String& selectorName)
212{
213 // Map selectors into Editor command names.
214 // This is not needed for any selectors that have the same name as the Editor command.
215 static constexpr std::pair<ComparableASCIILiteral, ASCIILiteral> selectorExceptions[] = {
216 { "insertNewlineIgnoringFieldEditor:", "InsertNewline"_s },
217 { "insertParagraphSeparator:", "InsertNewline"_s },
218 { "insertTabIgnoringFieldEditor:", "InsertTab"_s },
219 { "pageDown:", "MovePageDown"_s },
220 { "pageDownAndModifySelection:", "MovePageDownAndModifySelection"_s },
221 { "pageUp:", "MovePageUp"_s },
222 { "pageUpAndModifySelection:", "MovePageUpAndModifySelection"_s },
223 };
224 static constexpr SortedArrayMap map { selectorExceptions };
225 if (auto commandName = map.tryGet(selectorName))
226 return *commandName;
227
228 // Remove the trailing colon.
229 // No need to capitalize the command name since Editor command names are not case sensitive.
230 size_t selectorNameLength = selectorName.length();
231 if (selectorNameLength < 2 || selectorName[selectorNameLength - 1] != ':')
232 return String();
233 return selectorName.left(selectorNameLength - 1);
234}

244bool WebPage::executeKeypressCommandsInternal(const Vector<WebCore::KeypressCommand>& commands, KeyboardEvent* event)
245{
246 Frame& frame = event ? *frameForEvent(event) : m_page->focusController().focusedOrMainFrame();
247 ASSERT(frame.page() == corePage());
248
249 bool eventWasHandled = false;
250 for (size_t i = 0; i < commands.size(); ++i) {
251 if (commands[i].commandName == "insertText:") {
252 if (frame.editor().hasComposition()) {
253 eventWasHandled = true;
254 frame.editor().confirmComposition(commands[i].text);
255 } else {
256 if (!frame.editor().canEdit())
257 continue;
258
259 // An insertText: might be handled by other responders in the chain if we don't handle it.
260 // One example is space bar that results in scrolling down the page.
261 eventWasHandled |= frame.editor().insertText(commands[i].text, event);
262 }
263 } else {
264 Editor::Command command = frame.editor().command(commandNameForSelectorName(commands[i].commandName));
265 if (command.isSupported()) {
266 bool commandExecutedByEditor = command.execute(event);
267 eventWasHandled |= commandExecutedByEditor;
268 if (!commandExecutedByEditor) {
269 bool performedNonEditingBehavior = event->underlyingPlatformEvent()->type() == PlatformEvent::RawKeyDown && performNonEditingBehaviorForSelector(commands[i].commandName, event);
270 eventWasHandled |= performedNonEditingBehavior;
271 }
272 } else {
273 bool commandWasHandledByUIProcess = false;
274 WebProcess::singleton().parentProcessConnection()->sendSync(Messages::WebPageProxy::ExecuteSavedCommandBySelector(commands[i].commandName),
275 Messages::WebPageProxy::ExecuteSavedCommandBySelector::Reply(commandWasHandledByUIProcess), m_identifier);
276 eventWasHandled |= commandWasHandledByUIProcess;
277 }
278 }
279 }
280 return eventWasHandled;
281}

283bool WebPage::handleEditingKeyboardEvent(KeyboardEvent& event)
284{
285 auto* frame = frameForEvent(&event);
286
287 auto* platformEvent = event.underlyingPlatformEvent();
288 if (!platformEvent)
289 return false;
290 auto& commands = event.keypressCommands();
291
292 ASSERT(!platformEvent->macEvent()); // Cannot have a native event in WebProcess.
293
294 // Don't handle Esc while handling keydown event, we need to dispatch a keypress first.
295 if (platformEvent->type() != PlatformEvent::Char && platformEvent->windowsVirtualKeyCode() == VK_ESCAPE && commands.size() == 1 && commandNameForSelectorName(commands[0].commandName) == "cancelOperation")
296 return false;
297
298 if (handleKeyEventByRelinquishingFocusToChrome(event))
299 return true;
300
301 bool eventWasHandled = false;
302
303 // Are there commands that could just cause text insertion if executed via Editor?
304 // WebKit doesn't have enough information about mode to decide how they should be treated, so we leave it upon WebCore
305 // to either handle them immediately (e.g. Tab that changes focus) or let a keypress event be generated
306 // (e.g. Tab that inserts a Tab character, or Enter).
307 bool haveTextInsertionCommands = false;
308 for (auto& command : commands) {
309 if (frame->editor().command(commandNameForSelectorName(command.commandName)).isTextInsertion())
310 haveTextInsertionCommands = true;
311 }
312 // If there are no text insertion commands, default keydown handler is the right time to execute the commands.
313 // Keypress (Char event) handler is the latest opportunity to execute.
314 if (!haveTextInsertionCommands || platformEvent->type() == PlatformEvent::Char) {
315 eventWasHandled = executeKeypressCommandsInternal(commands, &event);
316 commands.clear();
317 }
318
319 return eventWasHandled;
320}

330void WebPage::attributedSubstringForCharacterRangeAsync(const EditingRange& editingRange, CompletionHandler<void(const WebCore::AttributedString&, const EditingRange&)>&& completionHandler)
331{
332 Frame& frame = m_page->focusController().focusedOrMainFrame();
333
334 const VisibleSelection& selection = frame.selection().selection();
335 if (selection.isNone() || !selection.isContentEditable() || selection.isInPasswordField()) {
336 completionHandler({ }, { });
337 return;
338 }
339
340 auto range = EditingRange::toRange(frame, editingRange);
341 if (!range) {
342 completionHandler({ }, { });
343 return;
344 }
345
346 auto attributedString = editingAttributedString(*range, IncludeImages::No).string;
347
348 // WebCore::editingAttributedStringFromRange() insists on inserting a trailing
349 // whitespace at the end of the string which breaks the ATOK input method. <rdar://problem/5400551>
350 // To work around this we truncate the resultant string to the correct length.
351 if ([attributedString length] > editingRange.length) {
352 ASSERT([attributedString length] == editingRange.length + 1);
353 ASSERT([[attributedString string] characterAtIndex:editingRange.length] == '\n' || [[attributedString string] characterAtIndex:editingRange.length] == ' ');
354 attributedString = [attributedString attributedSubstringFromRange:NSMakeRange(0, editingRange.length)];
355 }
356
357 EditingRange rangeToSend(editingRange.location, [attributedString length]);
358 ASSERT(rangeToSend.isValid());
359 if (!rangeToSend.isValid()) {
360 // Send an empty EditingRange as a last resort for <rdar://problem/27078089>.
361 completionHandler({ WTFMove(attributedString), nil }, EditingRange());
362 return;
363 }
364
365 completionHandler({ WTFMove(attributedString), nil }, rangeToSend);
366}

368void WebPage::fontAtSelection(CompletionHandler<void(const FontInfo&, double, bool)>&& completionHandler)
369{
370 bool selectionHasMultipleFonts = false;
371 auto& frame = m_page->focusController().focusedOrMainFrame();
372
373 if (frame.selection().selection().isNone()) {
374 completionHandler({ }, 0, false);
375 return;
376 }
377
378 auto font = frame.editor().fontForSelection(selectionHasMultipleFonts);
379 if (!font) {
380 completionHandler({ }, 0, false);
381 return;
382 }
383
384 auto ctFont = font->getCTFont();
385 if (!ctFont) {
386 completionHandler({ }, 0, false);
387 return;
388 }
389
390 auto fontDescriptor = adoptCF(CTFontCopyFontDescriptor(ctFont));
391 if (!fontDescriptor) {
392 completionHandler({ }, 0, false);
393 return;
394 }
395
396 completionHandler({ adoptCF(CTFontDescriptorCopyAttributes(fontDescriptor.get())) }, CTFontGetSize(ctFont), selectionHasMultipleFonts);
397}

403DictionaryPopupInfo WebPage::dictionaryPopupInfoForSelectionInPDFPlugin(PDFSelection *selection, PDFPlugin& pdfPlugin, NSDictionary *options, WebCore::TextIndicatorPresentationTransition presentationTransition)
404{
405 DictionaryPopupInfo dictionaryPopupInfo;
406 if (!selection.string.length)
407 return dictionaryPopupInfo;
408
409 NSRect rangeRect = pdfPlugin.rectForSelectionInRootView(selection);
410
411 NSAttributedString *nsAttributedString = selection.attributedString;
412
413 RetainPtr<NSMutableAttributedString> scaledNSAttributedString = adoptNS([[NSMutableAttributedString alloc] initWithString:[nsAttributedString string]]);
414
415 NSFontManager *fontManager = [NSFontManager sharedFontManager];
416
417 CGFloat scaleFactor = pdfPlugin.scaleFactor();
418
419 __block CGFloat maxAscender = 0;
420 __block CGFloat maxDescender = 0;
421 [nsAttributedString enumerateAttributesInRange:NSMakeRange(0, [nsAttributedString length]) options:0 usingBlock:^(NSDictionary *attributes, NSRange range, BOOL *stop) {
422 RetainPtr<NSMutableDictionary> scaledAttributes = adoptNS([attributes mutableCopy]);
423
424 NSFont *font = [scaledAttributes objectForKey:NSFontAttributeName];
425 if (font) {
426 maxAscender = std::max(maxAscender, font.ascender * scaleFactor);
427 maxDescender = std::min(maxDescender, font.descender * scaleFactor);
428 font = [fontManager convertFont:font toSize:[font pointSize] * scaleFactor];
429 [scaledAttributes setObject:font forKey:NSFontAttributeName];
430 }
431
432 [scaledNSAttributedString addAttributes:scaledAttributes.get() range:range];
433 }];
434
435 rangeRect.size.height = nsAttributedString.size.height * scaleFactor;
436 rangeRect.size.width = nsAttributedString.size.width * scaleFactor;
437
438 TextIndicatorData dataForSelection;
439 dataForSelection.selectionRectInRootViewCoordinates = rangeRect;
440 dataForSelection.textBoundingRectInRootViewCoordinates = rangeRect;
441 dataForSelection.contentImageScaleFactor = scaleFactor;
442 dataForSelection.presentationTransition = presentationTransition;
443
444 dictionaryPopupInfo.origin = rangeRect.origin;
445 dictionaryPopupInfo.options = options;
446 dictionaryPopupInfo.textIndicator = dataForSelection;
447 dictionaryPopupInfo.attributedString = scaledNSAttributedString;
448
449 return dictionaryPopupInfo;
450}

454bool WebPage::performNonEditingBehaviorForSelector(const String& selector, KeyboardEvent* event)
455{
456 // First give accessibility a chance to handle the event.
457 Frame* frame = frameForEvent(event);
458 frame->eventHandler().handleKeyboardSelectionMovementForAccessibility(*event);
459 if (event->defaultHandled())
460 return true;
461
462 // FIXME: All these selectors have corresponding Editor commands, but the commands only work in editable content.
463 // Should such non-editing behaviors be implemented in Editor or EventHandler::defaultArrowEventHandler() perhaps?
464
465 bool didPerformAction = false;
466
467 if (!frame->settings().eventHandlerDrivenSmoothKeyboardScrollingEnabled()) {
468 if (selector == "moveUp:")
469 didPerformAction = scroll(m_page.get(), ScrollUp, ScrollByLine);
470 else if (selector == "moveToBeginningOfParagraph:")
471 didPerformAction = scroll(m_page.get(), ScrollUp, ScrollByPage);
472 else if (selector == "moveToBeginningOfDocument:") {
473 didPerformAction = scroll(m_page.get(), ScrollUp, ScrollByDocument);
474 didPerformAction |= scroll(m_page.get(), ScrollLeft, ScrollByDocument);
475 } else if (selector == "moveDown:")
476 didPerformAction = scroll(m_page.get(), ScrollDown, ScrollByLine);
477 else if (selector == "moveToEndOfParagraph:")
478 didPerformAction = scroll(m_page.get(), ScrollDown, ScrollByPage);
479 else if (selector == "moveToEndOfDocument:") {
480 didPerformAction = scroll(m_page.get(), ScrollDown, ScrollByDocument);
481 didPerformAction |= scroll(m_page.get(), ScrollLeft, ScrollByDocument);
482 } else if (selector == "moveLeft:")
483 didPerformAction = scroll(m_page.get(), ScrollLeft, ScrollByLine);
484 else if (selector == "moveWordLeft:")
485 didPerformAction = scroll(m_page.get(), ScrollLeft, ScrollByPage);
486 else if (selector == "moveRight:")
487 didPerformAction = scroll(m_page.get(), ScrollRight, ScrollByLine);
488 else if (selector == "moveWordRight:")
489 didPerformAction = scroll(m_page.get(), ScrollRight, ScrollByPage);
490 }
491
492 if (selector == "moveToLeftEndOfLine:")
493 didPerformAction = m_userInterfaceLayoutDirection == WebCore::UserInterfaceLayoutDirection::LTR ? m_page->backForward().goBack() : m_page->backForward().goForward();
494 else if (selector == "moveToRightEndOfLine:")
495 didPerformAction = m_userInterfaceLayoutDirection == WebCore::UserInterfaceLayoutDirection::LTR ? m_page->backForward().goForward() : m_page->backForward().goBack();
496
497 return didPerformAction;
498}

575 completionHandler(SharedMemory::IPCHandle { WTFMove(handle), size });

606void WebPage::acceptsFirstMouse(int eventNumber, const WebKit::WebMouseEvent& event, CompletionHandler<void(bool)>&& completionHandler)
607{
608 if (WebProcess::singleton().parentProcessConnection()->inSendSync()) {
609 // In case we're already inside a sendSync message, it's possible that the page is in a
610 // transitionary state, so any hit-testing could cause crashes so we just return early in that case.
611 return completionHandler(false);
612 }
613
614 auto& frame = m_page->focusController().focusedOrMainFrame();
615
616 constexpr OptionSet<HitTestRequest::Type> hitType { HitTestRequest::Type::ReadOnly, HitTestRequest::Type::Active, HitTestRequest::Type::AllowChildFrameContent };
617 HitTestResult hitResult = frame.eventHandler().hitTestResultAtPoint(frame.view()->windowToContents(event.position()), hitType);
618 frame.eventHandler().setActivationEventNumber(eventNumber);
619 bool result = false;

689static void drawPDFPage(PDFDocument *pdfDocument, CFIndex pageIndex, CGContextRef context, CGFloat pageSetupScaleFactor, CGSize paperSize)
690{
691 CGContextSaveGState(context);
692
693 CGContextScaleCTM(context, pageSetupScaleFactor, pageSetupScaleFactor);
694
695 PDFPage *pdfPage = [pdfDocument pageAtIndex:pageIndex];
696 NSRect cropBox = [pdfPage boundsForBox:kPDFDisplayBoxCropBox];
697 if (NSIsEmptyRect(cropBox))
698 cropBox = [pdfPage boundsForBox:kPDFDisplayBoxMediaBox];
699 else
700 cropBox = NSIntersectionRect(cropBox, [pdfPage boundsForBox:kPDFDisplayBoxMediaBox]);
701
702 // Always auto-rotate PDF content regardless of the paper orientation.
703 NSInteger rotation = [pdfPage rotation];
704 if (rotation == 90 || rotation == 270)
705 std::swap(cropBox.size.width, cropBox.size.height);
706
707 bool shouldRotate = (paperSize.width < paperSize.height) != (cropBox.size.width < cropBox.size.height);
708 if (shouldRotate)
709 std::swap(cropBox.size.width, cropBox.size.height);
710
711 // Center.
712 CGFloat widthDifference = paperSize.width / pageSetupScaleFactor - cropBox.size.width;
713 CGFloat heightDifference = paperSize.height / pageSetupScaleFactor - cropBox.size.height;
714 if (widthDifference || heightDifference)
715 CGContextTranslateCTM(context, roundCGFloat(widthDifference / 2), roundCGFloat(heightDifference / 2));
716
717 if (shouldRotate) {
718 CGContextRotateCTM(context, static_cast<CGFloat>(piOverTwoDouble));
719 CGContextTranslateCTM(context, 0, -cropBox.size.width);
720 }
721
722 [NSGraphicsContext saveGraphicsState];
723 [NSGraphicsContext setCurrentContext:[NSGraphicsContext graphicsContextWithCGContext:context flipped:NO]];
724 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
725 [pdfPage drawWithBox:kPDFDisplayBoxCropBox];
726 ALLOW_DEPRECATED_DECLARATIONS_END
727 [NSGraphicsContext restoreGraphicsState];
728
729 CGAffineTransform transform = CGContextGetCTM(context);
730
731 for (PDFAnnotation *annotation in [pdfPage annotations]) {
732 if (![annotation isKindOfClass:getPDFAnnotationLinkClass()])
733 continue;
734
735 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
736 PDFAnnotationLink *linkAnnotation = (PDFAnnotationLink *)annotation;
737 ALLOW_DEPRECATED_DECLARATIONS_END
738 NSURL *url = [linkAnnotation URL];
739 if (!url)
740 continue;
741
742 CGRect urlRect = NSRectToCGRect([linkAnnotation bounds]);
743 CGRect transformedRect = CGRectApplyAffineTransform(urlRect, transform);
744 CGPDFContextSetURLForRect(context, (CFURLRef)url, transformedRect);
745 }
746
747 CGContextRestoreGState(context);
748}

750void WebPage::drawPDFDocument(CGContextRef context, PDFDocument *pdfDocument, const PrintInfo& printInfo, const WebCore::IntRect& rect)
751{
752 NSUInteger pageCount = [pdfDocument pageCount];
753 IntSize paperSize(ceilf(printInfo.availablePaperWidth), ceilf(printInfo.availablePaperHeight));
754 IntRect pageRect(IntPoint(), paperSize);
755 for (NSUInteger i = 0; i < pageCount; ++i) {
756 if (pageRect.intersects(rect)) {
757 CGContextSaveGState(context);
758
759 CGContextTranslateCTM(context, pageRect.x() - rect.x(), pageRect.y() - rect.y());
760 drawPDFPage(pdfDocument, i, context, printInfo.pageSetupScaleFactor, paperSize);
761
762 CGContextRestoreGState(context);
763 }
764 pageRect.move(0, pageRect.height());
765 }
766}

768void WebPage::drawPagesToPDFFromPDFDocument(CGContextRef context, PDFDocument *pdfDocument, const PrintInfo& printInfo, uint32_t first, uint32_t count)
769{
770 NSUInteger pageCount = [pdfDocument pageCount];
771 for (uint32_t page = first; page < first + count; ++page) {
772 if (page >= pageCount)
773 break;
774
775 RetainPtr<CFDictionaryRef> pageInfo = adoptCF(CFDictionaryCreateMutable(0, 0, &kCFTypeDictionaryKeyCallBacks, &kCFTypeDictionaryValueCallBacks));
776
777 CGPDFContextBeginPage(context, pageInfo.get());
778 drawPDFPage(pdfDocument, page, context, printInfo.pageSetupScaleFactor, CGSizeMake(printInfo.availablePaperWidth, printInfo.availablePaperHeight));
779 CGPDFContextEndPage(context);
780 }

854void WebPage::performImmediateActionHitTestAtLocation(WebCore::FloatPoint locationInViewCoordinates)
855{
856 layoutIfNeeded();
857
858 auto& mainFrame = corePage()->mainFrame();
859 if (!mainFrame.view() || !mainFrame.view()->renderView()) {
860 send(Messages::WebPageProxy::DidPerformImmediateActionHitTest(WebHitTestResultData(), false, UserData()));
861 return;
862 }
863
864 auto locationInContentCoordinates = mainFrame.view()->rootViewToContents(roundedIntPoint(locationInViewCoordinates));
865 auto hitTestResult = mainFrame.eventHandler().hitTestResultAtPoint(locationInContentCoordinates, {
866 HitTestRequest::Type::ReadOnly,
867 HitTestRequest::Type::Active,
868 HitTestRequest::Type::DisallowUserAgentShadowContentExceptForImageOverlays,
869 HitTestRequest::Type::AllowChildFrameContent,
870 });
871
872 bool immediateActionHitTestPreventsDefault = false;
873 Element* element = hitTestResult.targetElement();
874
875 mainFrame.eventHandler().setImmediateActionStage(ImmediateActionStage::PerformedHitTest);
876 if (element)
877 immediateActionHitTestPreventsDefault = element->dispatchMouseForceWillBegin();
878
879 WebHitTestResultData immediateActionResult(hitTestResult, { });
880
881 auto selectionRange = corePage()->focusController().focusedOrMainFrame().selection().selection().firstRange();
882
883 auto indicatorOptions = [&](const SimpleRange& range) {
884 OptionSet<TextIndicatorOption> options { TextIndicatorOption::UseBoundingRectAndPaintAllContentForComplexRanges };
885 if (HTMLElement::isInsideImageOverlay(range))
886 options.add({ TextIndicatorOption::PaintAllContent, TextIndicatorOption::PaintBackgrounds });
887 return options;
888 };
889
890 URL absoluteLinkURL = hitTestResult.absoluteLinkURL();
891 if (auto urlElement = makeRefPtr(hitTestResult.URLElement()); !absoluteLinkURL.isEmpty() && urlElement) {
892 auto elementRange = makeRangeSelectingNodeContents(*urlElement);
893 immediateActionResult.linkTextIndicator = TextIndicator::createWithRange(elementRange, indicatorOptions(elementRange), TextIndicatorPresentationTransition::FadeIn);
894 }
895
896 if (auto lookupResult = lookupTextAtLocation(locationInViewCoordinates)) {
897 auto [lookupRange, options] = WTFMove(*lookupResult);
898 immediateActionResult.lookupText = plainText(lookupRange);
899 if (auto* node = hitTestResult.innerNode()) {
900 if (auto* frame = node->document().frame())
901 immediateActionResult.dictionaryPopupInfo = dictionaryPopupInfoForRange(*frame, lookupRange, options, TextIndicatorPresentationTransition::FadeIn);
902 }
903 }
904
905 bool pageOverlayDidOverrideDataDetectors = false;
906 for (auto& overlay : corePage()->pageOverlayController().pageOverlays()) {
907 auto webOverlay = WebPageOverlay::fromCoreOverlay(*overlay);
908 if (!webOverlay)
909 continue;
910
911 auto actionContext = webOverlay->actionContextForResultAtPoint(locationInContentCoordinates);
912 if (!actionContext)
913 continue;
914
915 auto view = actionContext->range.start.document().view();
916 if (!view)
917 continue;
918
919 pageOverlayDidOverrideDataDetectors = true;
920 immediateActionResult.detectedDataActionContext = actionContext->context.get();
921 immediateActionResult.detectedDataBoundingBox = view->contentsToWindow(enclosingIntRect(unitedBoundingBoxes(RenderObject::absoluteTextQuads(actionContext->range))));
922 immediateActionResult.detectedDataTextIndicator = TextIndicator::createWithRange(actionContext->range, indicatorOptions(actionContext->range), TextIndicatorPresentationTransition::FadeIn);
923 immediateActionResult.detectedDataOriginatingPageOverlay = overlay->pageOverlayID();
924 break;
925 }
926
927 // FIXME: Avoid scanning if we will just throw away the result (e.g. we're over a link).
928 if (!pageOverlayDidOverrideDataDetectors && hitTestResult.innerNode() && (hitTestResult.innerNode()->isTextNode() || hitTestResult.isOverTextInsideFormControlElement())) {
929 if (auto result = DataDetection::detectItemAroundHitTestResult(hitTestResult)) {
930 immediateActionResult.detectedDataActionContext = WTFMove(result->actionContext);
931 immediateActionResult.detectedDataBoundingBox = result->boundingBox;
932 immediateActionResult.detectedDataTextIndicator = TextIndicator::createWithRange(result->range, indicatorOptions(result->range), TextIndicatorPresentationTransition::FadeIn);
933 }
934 }
935
936#if ENABLE(PDFKIT_PLUGIN)
937 if (is<HTMLPlugInImageElement>(element)) {
938 if (auto* pluginView = static_cast<PluginView*>(downcast<HTMLPlugInImageElement>(*element).pluginWidget())) {
939 if (is<PDFPlugin>(pluginView->plugin())) {
940 // FIXME: We don't have API to identify images inside PDFs based on position.
941 auto& plugIn = downcast<PDFPlugin>(*pluginView->plugin());
942 auto lookupResult = plugIn.lookupTextAtLocation(locationInViewCoordinates, immediateActionResult);
943 auto lookupText = std::get<String>(lookupResult);
944 if (!lookupText.isEmpty()) {
945 // FIXME (144030): Focus does not seem to get set to the PDF when invoking the menu.
946 auto& document = element->document();
947 if (is<PluginDocument>(document))
948 downcast<PluginDocument>(document).setFocusedElement(element);
949
950 auto selection = std::get<PDFSelection *>(lookupResult);
951 auto options = std::get<NSDictionary *>(lookupResult);
952
953 immediateActionResult.lookupText = lookupText;
954 immediateActionResult.isTextNode = true;
955 immediateActionResult.isSelected = true;
956 immediateActionResult.dictionaryPopupInfo = dictionaryPopupInfoForSelectionInPDFPlugin(selection, plugIn, options, TextIndicatorPresentationTransition::FadeIn);
957 }
958 }
959 }
960 }
961#endif
962
963 RefPtr<API::Object> userData;
964 injectedBundleContextMenuClient().prepareForImmediateAction(*this, hitTestResult, userData);
965
966 send(Messages::WebPageProxy::DidPerformImmediateActionHitTest(immediateActionResult, immediateActionHitTestPreventsDefault, UserData(WebProcess::singleton().transformObjectsToHandles(userData.get()).get())));
967}
