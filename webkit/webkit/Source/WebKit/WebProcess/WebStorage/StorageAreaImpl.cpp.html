40Ref<StorageAreaImpl> StorageAreaImpl::create(StorageAreaMap& storageAreaMap)
41{
42 return adoptRef(*new StorageAreaImpl(storageAreaMap));
43}
44
45StorageAreaImpl::StorageAreaImpl(StorageAreaMap& storageAreaMap)
46 : m_identifier(Identifier::generate())
47 , m_storageAreaMap(makeWeakPtr(storageAreaMap))
48{
49 storageAreaMap.incrementUseCount();
50}
51
52StorageAreaImpl::~StorageAreaImpl()
53{
54 if (m_storageAreaMap)
55 m_storageAreaMap->decrementUseCount();
56}
57
58unsigned StorageAreaImpl::length()
59{
60 return m_storageAreaMap ? m_storageAreaMap->length() : 0;
61}
62
63String StorageAreaImpl::key(unsigned index)
64{
65 return m_storageAreaMap ? m_storageAreaMap->key(index) : nullString();
66}
67
68String StorageAreaImpl::item(const String& key)
69{
70 return m_storageAreaMap ? m_storageAreaMap->item(key) : nullString();
71}
72
73void StorageAreaImpl::setItem(Frame* sourceFrame, const String& key, const String& value, bool& quotaException)
74{
75 ASSERT(!value.isNull());
76
77 if (m_storageAreaMap)
78 m_storageAreaMap->setItem(sourceFrame, this, key, value, quotaException);
79}
80
81void StorageAreaImpl::removeItem(Frame* sourceFrame, const String& key)
82{
83 if (m_storageAreaMap)
84 m_storageAreaMap->removeItem(sourceFrame, this, key);
85}
86
87void StorageAreaImpl::clear(Frame* sourceFrame)
88{
89 if (m_storageAreaMap)
90 m_storageAreaMap->clear(sourceFrame, this);
91}
92
93bool StorageAreaImpl::contains(const String& key)
94{
95 if (m_storageAreaMap)
96 return m_storageAreaMap->contains(key);
97
98 return false;
99}
100
101StorageType StorageAreaImpl::storageType() const
102{
103 if (m_storageAreaMap)
104 return m_storageAreaMap->type();
105
106 // We probably need an Invalid type.
107 return StorageType::Local;
108}
