81void StorageAreaMap::setItem(Frame* sourceFrame, StorageAreaImpl* sourceArea, const String& key, const String& value, bool& quotaException)
82{
83 auto& map = ensureMap();
84 ASSERT(!map.isShared());
85
86 String oldValue;
87 quotaException = false;
88 map.setItem(key, value, oldValue, quotaException);
89 if (quotaException)
90 return;
91
92 if (oldValue == value)
93 return;
94
95 m_pendingValueChanges.add(key);
96
97 if (m_mapID)
98 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::StorageManagerSet::SetItem(*m_mapID, sourceArea->identifier(), m_currentSeed, key, value, sourceFrame->document()->url().string()), 0);
99 else
100 RELEASE_LOG_ERROR(Storage, "StorageAreaMap::setItem failed because storage map ID is invalid");
101}

103void StorageAreaMap::removeItem(WebCore::Frame* sourceFrame, StorageAreaImpl* sourceArea, const String& key)
104{
105 auto& map = ensureMap();
106 ASSERT(!map.isShared());
107
108 String oldValue;
109 map.removeItem(key, oldValue);
110
111 if (oldValue.isNull())
112 return;
113
114 m_pendingValueChanges.add(key);
115
116 if (m_mapID)
117 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::StorageManagerSet::RemoveItem(*m_mapID, sourceArea->identifier(), m_currentSeed, key, sourceFrame->document()->url().string()), 0);
118 else
119 RELEASE_LOG_ERROR(Storage, "StorageAreaMap::removeItem failed because storage map ID is invalid");
120}

122void StorageAreaMap::clear(WebCore::Frame* sourceFrame, StorageAreaImpl* sourceArea)
123{
124 connect();
125
126 resetValues();
127
128 m_hasPendingClear = true;
129 m_map = makeUnique<StorageMap>(m_quotaInBytes);
130
131 if (m_mapID)
132 WebProcess::singleton().ensureNetworkProcessConnection().connection().send(Messages::StorageManagerSet::Clear(*m_mapID, sourceArea->identifier(), m_currentSeed, sourceFrame->document()->url().string()), 0);
133 else
134 RELEASE_LOG_ERROR(Storage, "StorageAreaMap::clear failed because storage map ID is invalid");
135}

142void StorageAreaMap::resetValues()

151StorageMap& StorageAreaMap::ensureMap()
152{
153 connect();
154
155 if (!m_map) {
156 m_map = makeUnique<StorageMap>(m_quotaInBytes);
157
158 if (m_mapID) {
159 // We need to use a IPC::UnboundedSynchronousIPCScope to prevent UIProcess hangs in case we receive a synchronous IPC from the UIProcess while we're waiting for a response
160 // from our StorageManagerSet::GetValues() IPC. This IPC may be very slow because it may need to fetch the values from disk and there may be a lot of data.
161 IPC::UnboundedSynchronousIPCScope unboundedSynchronousIPCScope;
162 HashMap<String, String> values;
163 WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::StorageManagerSet::GetValues(*m_mapID), Messages::StorageManagerSet::GetValues::Reply(values), 0);
164 m_map->importItems(WTFMove(values));
165 } else
166 RELEASE_LOG_ERROR(Storage, "StorageAreaMap::ensureMap failed to load from network process because storage map ID is invalid");
167 }
168 return *m_map;
169}

171void StorageAreaMap::didSetItem(uint64_t mapSeed, const String& key, bool quotaError)
172{
173 if (m_currentSeed != mapSeed)
174 return;
175
176 ASSERT(m_pendingValueChanges.contains(key));
177
178 if (quotaError) {
179 resetValues();
180 return;
181 }
182
183 m_pendingValueChanges.remove(key);
184}

186void StorageAreaMap::didRemoveItem(uint64_t mapSeed, const String& key)
187{
188 if (m_currentSeed != mapSeed)
189 return;
190
191 ASSERT(m_pendingValueChanges.contains(key));
192 m_pendingValueChanges.remove(key);
193}

204bool StorageAreaMap::shouldApplyChangeForKey(const String& key) const
205{
206 // We have not yet loaded anything from this storage map.
207 if (!m_map)
208 return false;
209
210 // Check if this storage area is currently waiting for the storage manager to update the given key.
211 // If that is the case, we don't want to apply any changes made by other storage areas, since
212 // our change was made last.
213 if (m_pendingValueChanges.contains(key))
214 return false;
215
216 return true;
217}
218
219void StorageAreaMap::applyChange(const String& key, const String& newValue)
220{
221 ASSERT(!m_map || !m_map->isShared());
222
223 // There is at least one clear pending we don't want to apply any changes until we get the corresponding DidClear messages.
224 if (m_hasPendingClear)
225 return;
226
227 if (!key) {
228 // A null key means clear.
229 auto newMap = makeUnique<StorageMap>(m_quotaInBytes);
230
231 // Any changes that were made locally after the clear must still be kept around in the new map.
232 for (auto& change : m_pendingValueChanges) {
233 auto& key = change.key;
234 String value = m_map->getItem(key);
235 if (!value) {
236 // This change must have been a pending remove, ignore it.
237 continue;
238 }
239
240 String oldValue;
241 newMap->setItemIgnoringQuota(key, oldValue);
242 }
243
244 m_map = WTFMove(newMap);
245 return;
246 }
247
248 if (!shouldApplyChangeForKey(key))
249 return;
250
251 if (!newValue) {
252 // A null new value means that the item should be removed.
253 String oldValue;
254 m_map->removeItem(key, oldValue);
255 return;
256 }
257
258 m_map->setItemIgnoringQuota(key, newValue);
259}

261void StorageAreaMap::dispatchStorageEvent(const std::optional<StorageAreaImplIdentifier>& storageAreaImplID, const String& key, const String& oldValue, const String& newValue, const String& urlString)
262{
263 if (!storageAreaImplID) {
264 // This storage event originates from another process so we need to apply the change to our storage area map.
265 applyChange(key, newValue);
266 }
267
268 if (type() == StorageType::Session)
269 dispatchSessionStorageEvent(storageAreaImplID, key, oldValue, newValue, urlString);
270 else
271 dispatchLocalStorageEvent(storageAreaImplID, key, oldValue, newValue, urlString);
272}

279static Vector<RefPtr<Frame>> framesForEventDispatching(Page& page, SecurityOrigin& origin, StorageType storageType, const std::optional<StorageAreaImplIdentifier>& storageAreaImplID)
280{
281 Vector<RefPtr<Frame>> frames;
282 page.forEachDocument([&](auto& document) {
283 if (!document.securityOrigin().equal(&origin))
284 return;
285
286 auto* window = document.domWindow();
287 if (!window)
288 return;
289
290 Storage* storage = nullptr;
291 switch (storageType) {
292 case StorageType::Session:
293 storage = window->optionalSessionStorage();
294 break;
295 case StorageType::Local:
296 case StorageType::TransientLocal:
297 storage = window->optionalLocalStorage();
298 break;
299 }
300
301 if (!storage)
302 return;
303
304 auto& storageArea = static_cast<StorageAreaImpl&>(storage->area());
305 if (storageArea.identifier() == storageAreaImplID) {
306 // This is the storage area that caused the event to be dispatched.
307 return;
308 }
309
310 if (auto* frame = document.frame())
311 frames.append(frame);
312 });
313 return frames;
314}

316void StorageAreaMap::dispatchSessionStorageEvent(const std::optional<StorageAreaImplIdentifier>& storageAreaImplID, const String& key, const String& oldValue, const String& newValue, const String& urlString)
317{
318 // Namespace IDs for session storage namespaces are equivalent to web page IDs
319 // so we can get the right page here.
320 auto* webPage = WebProcess::singleton().webPage(m_namespace.sessionStoragePageID());
321 if (!webPage)
322 return;
323
324 auto* page = webPage->corePage();
325 if (!page)
326 return;
327
328 auto frames = framesForEventDispatching(*page, m_securityOrigin, StorageType::Session, storageAreaImplID);
329 StorageEventDispatcher::dispatchSessionStorageEventsToFrames(*page, frames, key, oldValue, newValue, urlString, m_securityOrigin->data());
330}

332void StorageAreaMap::dispatchLocalStorageEvent(const std::optional<StorageAreaImplIdentifier>& storageAreaImplID, const String& key, const String& oldValue, const String& newValue, const String& urlString)
333{
334 ASSERT(isLocalStorage(type()));
335
336 Vector<RefPtr<Frame>> frames;
337
338 // Namespace IDs for local storage namespaces are equivalent to web page group IDs.
339 auto& pageGroup = *WebProcess::singleton().webPageGroup(m_namespace.pageGroupID())->corePageGroup();
340 for (auto& page : pageGroup.pages())
341 frames.appendVector(framesForEventDispatching(page, m_securityOrigin, StorageType::Local, storageAreaImplID));
342
343 StorageEventDispatcher::dispatchLocalStorageEventsToFrames(pageGroup, frames, key, oldValue, newValue, urlString, m_securityOrigin->data());
344}

346void StorageAreaMap::connect()
347{
348 if (m_mapID)
349 return;
350
351 switch (m_type) {
352 case StorageType::Local:
353 case StorageType::TransientLocal:
354 if (SecurityOrigin* topLevelOrigin = m_namespace.topLevelOrigin())
355 WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::StorageManagerSet::ConnectToTransientLocalStorageArea(WebProcess::singleton().sessionID(), m_namespace.storageNamespaceID(), topLevelOrigin->data(), m_securityOrigin->data()), Messages::StorageManagerSet::ConnectToTransientLocalStorageArea::Reply(m_mapID), 0);
356 else
357 WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::StorageManagerSet::ConnectToLocalStorageArea(WebProcess::singleton().sessionID(), m_namespace.storageNamespaceID(), m_securityOrigin->data()), Messages::StorageManagerSet::ConnectToLocalStorageArea::Reply(m_mapID), 0);
358 break;
359 case StorageType::Session:
360 WebProcess::singleton().ensureNetworkProcessConnection().connection().sendSync(Messages::StorageManagerSet::ConnectToSessionStorageArea(WebProcess::singleton().sessionID(), m_namespace.storageNamespaceID(), m_securityOrigin->data()), Messages::StorageManagerSet::ConnectToSessionStorageArea::Reply(m_mapID), 0);
361 }
362
363 if (m_mapID)
364 WebProcess::singleton().registerStorageAreaMap(*this);
365}
366
367void StorageAreaMap::disconnect()
368{
369 if (!m_mapID)
370 return;
371
372 resetValues();
373 WebProcess::singleton().unregisterStorageAreaMap(*this);
374
375 if (auto* networkProcessConnection = WebProcess::singleton().existingNetworkProcessConnection())
376 networkProcessConnection->connection().send(Messages::StorageManagerSet::DisconnectFromStorageArea(*m_mapID), 0);
377
378 m_mapID = { };
379}
