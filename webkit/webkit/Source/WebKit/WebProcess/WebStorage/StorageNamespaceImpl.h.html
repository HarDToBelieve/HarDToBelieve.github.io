38namespace WebKit {
39
40class StorageAreaMap;
41class WebPage;
42
43class StorageNamespaceImpl final : public WebCore::StorageNamespace {
44public:
45 using Identifier = StorageNamespaceIdentifier;
46
47 static Ref<StorageNamespaceImpl> createSessionStorageNamespace(Identifier, WebCore::PageIdentifier, unsigned quotaInBytes);
48 static Ref<StorageNamespaceImpl> createLocalStorageNamespace(Identifier, unsigned quotaInBytes);
49 static Ref<StorageNamespaceImpl> createTransientLocalStorageNamespace(Identifier, WebCore::SecurityOrigin& topLevelOrigin, uint64_t quotaInBytes);
50
51 virtual ~StorageNamespaceImpl();
52
53 WebCore::StorageType storageType() const { return m_storageType; }
54 Identifier storageNamespaceID() const { return m_storageNamespaceID; }
55 WebCore::PageIdentifier sessionStoragePageID() const;
56 uint64_t pageGroupID() const;
57 WebCore::SecurityOrigin* topLevelOrigin() const { return m_topLevelOrigin.get(); }
58 unsigned quotaInBytes() const { return m_quotaInBytes; }
59 PAL::SessionID sessionID() const override;
60
61 void destroyStorageAreaMap(StorageAreaMap&);
62
63 void setSessionIDForTesting(PAL::SessionID) override;
64
65private:
66 StorageNamespaceImpl(WebCore::StorageType, Identifier, const std::optional<WebCore::PageIdentifier>&, WebCore::SecurityOrigin* topLevelOrigin, unsigned quotaInBytes);
67
68 Ref<WebCore::StorageArea> storageArea(const WebCore::SecurityOriginData&) override;
69 uint64_t storageAreaMapCountForTesting() const final { return m_storageAreaMaps.size(); }
70
71 // FIXME: This is only valid for session storage and should probably be moved to a subclass.
72 Ref<WebCore::StorageNamespace> copy(WebCore::Page&) override;
73
74 const WebCore::StorageType m_storageType;
75 const Identifier m_storageNamespaceID;
76 std::optional<WebCore::PageIdentifier> m_sessionPageID;
77
78 // Only used for transient local storage namespaces.
79 const RefPtr<WebCore::SecurityOrigin> m_topLevelOrigin;
80
81 const unsigned m_quotaInBytes;
82
83 HashMap<WebCore::SecurityOriginData, std::unique_ptr<StorageAreaMap>> m_storageAreaMaps;
84};

86} // namespace WebKit
