36namespace WebKit {
37using namespace WebCore;
38
39static HashMap<StorageNamespaceIdentifier, WebStorageNamespaceProvider*>& storageNamespaceProviders()
40{
41 static NeverDestroyed<HashMap<StorageNamespaceIdentifier, WebStorageNamespaceProvider*>> storageNamespaceProviders;
42
43 return storageNamespaceProviders;
44}
45
46Ref<WebStorageNamespaceProvider> WebStorageNamespaceProvider::getOrCreate(WebPageGroupProxy& pageGroup)
47{
48 RefPtr<WebStorageNamespaceProvider> storageNamespaceProvider;
49 auto* result = storageNamespaceProviders().ensure(pageGroup.localStorageNamespaceIdentifier(), [&]() {
50 storageNamespaceProvider = adoptRef(*new WebStorageNamespaceProvider(pageGroup.localStorageNamespaceIdentifier()));
51 return storageNamespaceProvider.get();
52 }).iterator->value;
53 return *result;
54}
55
56WebStorageNamespaceProvider::WebStorageNamespaceProvider(StorageNamespaceIdentifier localStorageNamespaceIdentifier)
57 : m_localStorageNamespaceIdentifier(localStorageNamespaceIdentifier)
58{
59}
60
61WebStorageNamespaceProvider::~WebStorageNamespaceProvider()
62{
63 ASSERT(storageNamespaceProviders().contains(m_localStorageNamespaceIdentifier));
64
65 storageNamespaceProviders().remove(m_localStorageNamespaceIdentifier);
66}
67
68Ref<WebCore::StorageNamespace> WebStorageNamespaceProvider::createSessionStorageNamespace(Page& page, unsigned quota)
69{
70 auto& webPage = WebPage::fromCorePage(page);
71 return StorageNamespaceImpl::createSessionStorageNamespace(webPage.sessionStorageNamespaceIdentifier(), webPage.identifier(), quota);
72}
73
74Ref<WebCore::StorageNamespace> WebStorageNamespaceProvider::createLocalStorageNamespace(unsigned quota, PAL::SessionID sessionID)
75{
76 ASSERT_UNUSED(sessionID, sessionID == WebProcess::singleton().sessionID());
77 return StorageNamespaceImpl::createLocalStorageNamespace(m_localStorageNamespaceIdentifier, quota);
78}
79
80Ref<WebCore::StorageNamespace> WebStorageNamespaceProvider::createTransientLocalStorageNamespace(WebCore::SecurityOrigin& topLevelOrigin, unsigned quota, PAL::SessionID sessionID)
81{
82 ASSERT_UNUSED(sessionID, sessionID == WebProcess::singleton().sessionID());
83 return StorageNamespaceImpl::createTransientLocalStorageNamespace(m_localStorageNamespaceIdentifier, topLevelOrigin, quota);
84}
