40AudioSessionRoutingArbitrator::AudioSessionRoutingArbitrator(WebProcess& process)
41 : m_process(process)
42 , m_observer([this] (AudioSession& session) { session.setRoutingArbitrationClient(makeWeakPtr(this)); })
43{
44 AudioSession::addAudioSessionChangedObserver(m_observer);
45}
46
47AudioSessionRoutingArbitrator::~AudioSessionRoutingArbitrator() = default;
48
49const char* AudioSessionRoutingArbitrator::supplementName()
50{
51 return "AudioSessionRoutingArbitrator";
52}
53
54void AudioSessionRoutingArbitrator::beginRoutingArbitrationWithCategory(AudioSession::CategoryType category, CompletionHandler<void(RoutingArbitrationError, DefaultRouteChanged)>&& callback)
55{
56 m_process.parentProcessConnection()->sendWithAsyncReply(Messages::AudioSessionRoutingArbitratorProxy::BeginRoutingArbitrationWithCategory(category), WTFMove(callback), AudioSessionRoutingArbitratorProxy::destinationId());
57}
58
59void AudioSessionRoutingArbitrator::leaveRoutingAbritration()
60{
61 m_process.parentProcessConnection()->send(Messages::AudioSessionRoutingArbitratorProxy::EndRoutingArbitration(), AudioSessionRoutingArbitratorProxy::destinationId());
62}
