38LaunchServicesDatabaseManager& LaunchServicesDatabaseManager::singleton()
39{
40 static NeverDestroyed<LaunchServicesDatabaseManager> manager;
41 return manager.get();
42}
43
44void LaunchServicesDatabaseManager::handleEvent(xpc_object_t message)
45{
46 String messageName = xpc_dictionary_get_string(message, XPCEndpoint::xpcMessageNameKey);
47 if (messageName.isEmpty())
48 return;
49 if (messageName == LaunchServicesDatabaseXPCConstants::xpcUpdateLaunchServicesDatabaseMessageName) {
50#if HAVE(LSDATABASECONTEXT)
51 auto database = xpc_dictionary_get_value(message, LaunchServicesDatabaseXPCConstants::xpcLaunchServicesDatabaseKey);
52
53 if (database) {
54 auto context = [NSClassFromString(@"LSDatabaseContext") sharedDatabaseContext];
55 if (![context respondsToSelector:@selector(observeDatabaseChange4WebKit:)])
56 return;
57 [context observeDatabaseChange4WebKit:database];
58 }
59#endif
60 m_semaphore.signal();
61 m_hasReceivedLaunchServicesDatabase = true;
62 }
63}
64
65void LaunchServicesDatabaseManager::didConnect()
66{
67 auto message = adoptOSObject(xpc_dictionary_create(nullptr, nullptr, 0));
68 xpc_dictionary_set_string(message.get(), XPCEndpoint::xpcMessageNameKey, LaunchServicesDatabaseXPCConstants::xpcRequestLaunchServicesDatabaseUpdateMessageName);
69
70 auto connection = this->connection();
71 if (!connection)
72 return;
73
74 xpc_connection_send_message(connection.get(), message.get());
75}
76
77bool LaunchServicesDatabaseManager::waitForDatabaseUpdate(Seconds timeout)
78{
79 if (m_hasReceivedLaunchServicesDatabase)
80 return true;
81 return m_semaphore.waitFor(timeout);
82}
