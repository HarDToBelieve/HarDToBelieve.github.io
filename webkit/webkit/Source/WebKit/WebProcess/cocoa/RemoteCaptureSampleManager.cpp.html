52void RemoteCaptureSampleManager::stopListeningForIPC()
53{
54 if (m_isRegisteredToParentProcessConnection)
55 WebProcess::singleton().parentProcessConnection()->removeThreadMessageReceiver(Messages::RemoteCaptureSampleManager::messageReceiverName());
56 setConnection(nullptr);
57}

59void RemoteCaptureSampleManager::setConnection(IPC::Connection* connection)
60{
61 if (m_connection == connection)
62 return;
63
64 auto* parentConnection = WebProcess::singleton().parentProcessConnection();
65 if (connection == parentConnection) {
66 if (!m_isRegisteredToParentProcessConnection) {
67 m_isRegisteredToParentProcessConnection = true;
68 parentConnection->addThreadMessageReceiver(Messages::RemoteCaptureSampleManager::messageReceiverName(), this);
69 }
70 return;
71 }
72 if (m_connection)
73 m_connection->removeThreadMessageReceiver(Messages::RemoteCaptureSampleManager::messageReceiverName());
74
75 m_connection = WTFMove(connection);
76
77 if (m_connection)
78 m_connection->addThreadMessageReceiver(Messages::RemoteCaptureSampleManager::messageReceiverName(), this);
79}

81void RemoteCaptureSampleManager::addSource(Ref<RemoteRealtimeAudioSource>&& source)
82{
83 ASSERT(WTF::isMainRunLoop());
84 setConnection(source->connection());
85
86 m_queue->dispatch([this, protectedThis = makeRef(*this), source = WTFMove(source)]() mutable {
87 auto identifier = source->identifier();
88
89 ASSERT(!m_audioSources.contains(identifier));
90 m_audioSources.add(identifier, makeUnique<RemoteAudio>(WTFMove(source)));
91 });
92}

107void RemoteCaptureSampleManager::removeSource(WebCore::RealtimeMediaSourceIdentifier identifier)
108{
109 ASSERT(WTF::isMainRunLoop());
110 m_queue->dispatch([this, protectedThis = makeRef(*this), identifier] {
111 ASSERT(m_audioSources.contains(identifier) || m_videoSources.contains(identifier));
112 if (!m_audioSources.remove(identifier))
113 m_videoSources.remove(identifier);
114 });
115}

128void RemoteCaptureSampleManager::audioStorageChanged(WebCore::RealtimeMediaSourceIdentifier identifier, const SharedMemory::IPCHandle& ipcHandle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames, IPC::Semaphore&& semaphore, const MediaTime& mediaTime, size_t frameChunkSize)
129{
130 ASSERT(!WTF::isMainRunLoop());
131
132 auto iterator = m_audioSources.find(identifier);
133 if (iterator == m_audioSources.end()) {
134 RELEASE_LOG_ERROR(WebRTC, "Unable to find source %llu for storageChanged", identifier.toUInt64());
135 return;
136 }
137 iterator->value->setStorage(ipcHandle.handle, description, numberOfFrames, WTFMove(semaphore), mediaTime, frameChunkSize);
138}

162void RemoteCaptureSampleManager::RemoteAudio::stopThread()
163{
164 if (!m_thread)
165 return;
166
167 m_shouldStopThread = true;
168 m_semaphore.signal();
169 m_thread->waitForCompletion();
170 m_thread = nullptr;
171}

173void RemoteCaptureSampleManager::RemoteAudio::startThread()
174{
175 ASSERT(!m_thread);
176 m_shouldStopThread = false;
177 auto threadLoop = [this]() mutable {
178 m_readOffset = 0;
179 do {
180 // If wait fails, the semaphore on the other side was probably destroyed and we should just exit here and wait to launch a new thread.
181 if (!m_semaphore.wait())
182 break;
183 if (m_shouldStopThread)
184 break;
185
186 auto currentTime = m_startTime + MediaTime { m_readOffset, static_cast<uint32_t>(m_description.sampleRate()) };
187 m_ringBuffer->fetch(m_buffer->list(), m_frameChunkSize, m_readOffset);
188 m_readOffset += m_frameChunkSize;
189
190 m_source->remoteAudioSamplesAvailable(currentTime, *m_buffer, m_description, m_frameChunkSize);
191 } while (!m_shouldStopThread);
192 };
193 m_thread = Thread::create("RemoteAudioSourceProviderManager::RemoteAudio thread", WTFMove(threadLoop), ThreadType::Audio, Thread::QOS::UserInteractive);
194}

196void RemoteCaptureSampleManager::RemoteAudio::setStorage(const SharedMemory::Handle& handle, const WebCore::CAAudioStreamDescription& description, uint64_t numberOfFrames, IPC::Semaphore&& semaphore, const MediaTime& mediaTime, size_t frameChunkSize)
197{
198 stopThread();
199
200 if (!numberOfFrames) {
201 m_ringBuffer = nullptr;
202 m_buffer = nullptr;
203 return;
204 }
205
206 m_semaphore = WTFMove(semaphore);
207 m_description = description;
208 m_startTime = mediaTime;
209 m_frameChunkSize = frameChunkSize;
210
211 m_ringBuffer = CARingBuffer::adoptStorage(makeUniqueRef<ReadOnlySharedRingBufferStorage>(handle), description, numberOfFrames).moveToUniquePtr();
212 m_buffer = makeUnique<WebAudioBufferList>(description, numberOfFrames);
213 m_buffer->setSampleCount(m_frameChunkSize);
214
215 startThread();
216}

223void RemoteCaptureSampleManager::RemoteVideo::videoSampleAvailable(RemoteVideoSample&& remoteSample)
224{
225 if (!m_imageTransferSession || m_imageTransferSession->pixelFormat() != remoteSample.videoFormat())
226 m_imageTransferSession = ImageTransferSessionVT::create(remoteSample.videoFormat());
227
228 if (!m_imageTransferSession) {
229 ASSERT_NOT_REACHED();
230 return;
231 }
232
233 auto sampleRef = m_imageTransferSession->createMediaSample(remoteSample);
234 if (!sampleRef) {
235 ASSERT_NOT_REACHED();
236 return;
237 }
238
239 m_source->videoSampleAvailable(*sampleRef, remoteSample.size());
240}
