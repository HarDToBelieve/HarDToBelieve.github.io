46RemoteRealtimeMediaSourceProxy::~RemoteRealtimeMediaSourceProxy()
47{
48 failApplyConstraintCallbacks("Source terminated"_s);
49}
50
51IPC::Connection* RemoteRealtimeMediaSourceProxy::connection()
52{
53 ASSERT(isMainRunLoop());
54#if ENABLE(GPU_PROCESS)
55 if (m_shouldCaptureInGPUProcess)
56 return &WebProcess::singleton().ensureGPUProcessConnection().connection();
57#endif
58 return WebProcess::singleton().parentProcessConnection();
59}
60
61void RemoteRealtimeMediaSourceProxy::startProducingData()
62{
63 connection()->send(Messages::UserMediaCaptureManagerProxy::StartProducingData { m_identifier }, 0);
64}
65
66void RemoteRealtimeMediaSourceProxy::stopProducingData()
67{
68 connection()->send(Messages::UserMediaCaptureManagerProxy::StopProducingData { m_identifier }, 0);
69}
70
71void RemoteRealtimeMediaSourceProxy::createRemoteMediaSource(const String& deviceIDHashSalt, CreateCallback&& callback)
72{
73 connection()->sendWithAsyncReply(Messages::UserMediaCaptureManagerProxy::CreateMediaSourceForCaptureDeviceWithConstraints(identifier(), m_device, deviceIDHashSalt, m_constraints), WTFMove(callback));
74}
75
76void RemoteRealtimeMediaSourceProxy::applyConstraints(const MediaConstraints& constraints, RealtimeMediaSource::ApplyConstraintsHandler&& completionHandler)
77{
78 m_pendingApplyConstraintsCallbacks.append(WTFMove(completionHandler));
79 // FIXME: Use sendAsyncWithReply.
80 connection()->send(Messages::UserMediaCaptureManagerProxy::ApplyConstraints { m_identifier, constraints }, 0);
81}
82
83void RemoteRealtimeMediaSourceProxy::applyConstraintsSucceeded()
84{
85 auto callback = m_pendingApplyConstraintsCallbacks.takeFirst();
86 callback({ });
87}
88
89void RemoteRealtimeMediaSourceProxy::applyConstraintsFailed(String&& failedConstraint, String&& errorMessage)
90{
91 auto callback = m_pendingApplyConstraintsCallbacks.takeFirst();
92 callback(RealtimeMediaSource::ApplyConstraintsError { WTFMove(failedConstraint), WTFMove(errorMessage) });
93}
94
95void RemoteRealtimeMediaSourceProxy::failApplyConstraintCallbacks(const String& errorMessage)
96{
97 auto callbacks = WTFMove(m_pendingApplyConstraintsCallbacks);
98 while (!callbacks.isEmpty())
99 callbacks.takeFirst()(RealtimeMediaSource::ApplyConstraintsError { { }, errorMessage });
100}
101
102void RemoteRealtimeMediaSourceProxy::hasEnded()
103{
104 connection()->send(Messages::UserMediaCaptureManagerProxy::End { m_identifier }, 0);
105}
106
107void RemoteRealtimeMediaSourceProxy::whenReady(CompletionHandler<void(String)>&& callback)
108{
109 if (m_isReady)
110 return callback(WTFMove(m_errorMessage));
111 m_callback = WTFMove(callback);
112}
113
114void RemoteRealtimeMediaSourceProxy::setAsReady()
115{
116 ASSERT(!m_isReady);
117 m_isReady = true;
118 if (m_callback)
119 m_callback({ });
120}
121
122void RemoteRealtimeMediaSourceProxy::didFail(String&& errorMessage)
123{
124 m_isReady = true;
125 m_errorMessage = WTFMove(errorMessage);
126 if (m_callback)
127 m_callback(m_errorMessage);
128}
