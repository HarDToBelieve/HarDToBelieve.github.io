50class UserMediaCaptureManager : public WebProcessSupplement, public IPC::MessageReceiver {
51 WTF_MAKE_FAST_ALLOCATED;
52public:
53 explicit UserMediaCaptureManager(WebProcess&);
54 ~UserMediaCaptureManager();
55
56 static const char* supplementName();
57
58 void didReceiveMessageFromGPUProcess(IPC::Connection& connection, IPC::Decoder& decoder) { didReceiveMessage(connection, decoder); }
59 void setupCaptureProcesses(bool shouldCaptureAudioInUIProcess, bool shouldCaptureAudioInGPUProcess, bool shouldCaptureVideoInUIProcess, bool shouldCaptureVideoInGPUProcess, bool shouldCaptureDisplayInUIProcess);
60
61 void addAudioSource(Ref<RemoteRealtimeAudioSource>&&);
62 void removeAudioSource(WebCore::RealtimeMediaSourceIdentifier);
63
64 void addVideoSource(Ref<RemoteRealtimeVideoSource>&&);
65 void removeVideoSource(WebCore::RealtimeMediaSourceIdentifier);
66
67 RemoteCaptureSampleManager& remoteCaptureSampleManager() { return m_remoteCaptureSampleManager; }
68
69private:
70 // WebCore::RealtimeMediaSource factories
71 class AudioFactory : public WebCore::AudioCaptureFactory {
72 public:
73 explicit AudioFactory(UserMediaCaptureManager& manager) : m_manager(manager) { }
74 void setShouldCaptureInGPUProcess(bool);
75
76 private:
77 WebCore::CaptureSourceOrError createAudioCaptureSource(const WebCore::CaptureDevice&, String&& hashSalt, const WebCore::MediaConstraints*) final;
78 WebCore::CaptureDeviceManager& audioCaptureDeviceManager() final { return m_manager.m_noOpCaptureDeviceManager; }
79 const Vector<WebCore::CaptureDevice>& speakerDevices() const final { return m_speakerDevices; }
80
81 UserMediaCaptureManager& m_manager;
82 bool m_shouldCaptureInGPUProcess { false };
83 Vector<WebCore::CaptureDevice> m_speakerDevices;
84 };
85 class VideoFactory : public WebCore::VideoCaptureFactory {
86 public:
87 explicit VideoFactory(UserMediaCaptureManager& manager) : m_manager(manager) { }
88 void setShouldCaptureInGPUProcess(bool value) { m_shouldCaptureInGPUProcess = value; }
89
90 private:
91 WebCore::CaptureSourceOrError createVideoCaptureSource(const WebCore::CaptureDevice&, String&& hashSalt, const WebCore::MediaConstraints*) final;
92 WebCore::CaptureDeviceManager& videoCaptureDeviceManager() final { return m_manager.m_noOpCaptureDeviceManager; }
93
94 UserMediaCaptureManager& m_manager;
95 bool m_shouldCaptureInGPUProcess { false };
96 };
97 class DisplayFactory : public WebCore::DisplayCaptureFactory {
98 public:
99 explicit DisplayFactory(UserMediaCaptureManager& manager) : m_manager(manager) { }
100
101 private:
102 WebCore::CaptureSourceOrError createDisplayCaptureSource(const WebCore::CaptureDevice&, const WebCore::MediaConstraints*) final;
103 WebCore::CaptureDeviceManager& displayCaptureDeviceManager() final { return m_manager.m_noOpCaptureDeviceManager; }
104
105 UserMediaCaptureManager& m_manager;
106 };
107
108 class NoOpCaptureDeviceManager : public WebCore::CaptureDeviceManager {
109 public:
110 NoOpCaptureDeviceManager() = default;
111
112 private:
113 const Vector<WebCore::CaptureDevice>& captureDevices() final
114 {
115 ASSERT_NOT_REACHED();
116 return m_emptyDevices;
117 }
118 Vector<WebCore::CaptureDevice> m_emptyDevices;
119 };
120
121 // IPC::MessageReceiver
122 void didReceiveMessage(IPC::Connection&, IPC::Decoder&) final;
123
124 // Messages::UserMediaCaptureManager
125 void captureFailed(WebCore::RealtimeMediaSourceIdentifier);
126 void sourceStopped(WebCore::RealtimeMediaSourceIdentifier);
127 void sourceMutedChanged(WebCore::RealtimeMediaSourceIdentifier, bool muted);
128 void sourceSettingsChanged(WebCore::RealtimeMediaSourceIdentifier, WebCore::RealtimeMediaSourceSettings&&);
129 void applyConstraintsSucceeded(WebCore::RealtimeMediaSourceIdentifier, WebCore::RealtimeMediaSourceSettings&&);
130 void applyConstraintsFailed(WebCore::RealtimeMediaSourceIdentifier, String&&, String&&);
131
132 HashMap<WebCore::RealtimeMediaSourceIdentifier, Ref<RemoteRealtimeAudioSource>> m_audioSources;
133 HashMap<WebCore::RealtimeMediaSourceIdentifier, Ref<RemoteRealtimeVideoSource>> m_videoSources;
134 WebProcess& m_process;
135 NoOpCaptureDeviceManager m_noOpCaptureDeviceManager;
136 AudioFactory m_audioFactory;
137 VideoFactory m_videoFactory;
138 DisplayFactory m_displayFactory;
139 RemoteCaptureSampleManager m_remoteCaptureSampleManager;
140};
