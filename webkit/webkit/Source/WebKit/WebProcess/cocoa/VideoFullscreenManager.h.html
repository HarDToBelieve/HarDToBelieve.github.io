87
88 WebCore::HTMLMediaElementEnums::VideoFullscreenMode fullscreenMode() const { return m_fullscreenMode; }

91 bool fullscreenStandby() const { return m_fullscreenStandby; }

97private:

99 void hasVideoChanged(bool) override;
100 void videoDimensionsChanged(const WebCore::FloatSize&) override;

102 VideoFullscreenInterfaceContext(VideoFullscreenManager&, PlaybackSessionContextIdentifier);

109 WebCore::HTMLMediaElementEnums::VideoFullscreenMode m_fullscreenMode { WebCore::HTMLMediaElementEnums::VideoFullscreenModeNone };
110 bool m_fullscreenStandby { false };
111 bool m_isFullscreen { false };
