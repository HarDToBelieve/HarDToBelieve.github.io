247 // If we have an extension handle for MobileGestalt, it means the MobileGestalt cache is invalid.
248 // In this case, we perform a set of MobileGestalt queries while having access to the daemon,
249 // which will populate the MobileGestalt in-memory cache with correct values.
250 // The set of queries below was determined by finding all possible queries that have cachable
251 // values, and would reach out to the daemon for the answer. That way, the in-memory cache
252 // should be identical to a valid MobileGestalt cache after having queried all of these values.
253#if PLATFORM(IOS_FAMILY) && !PLATFORM(MACCATALYST)
254 MGGetFloat32Answer(kMGQMainScreenScale, 0);
255 MGGetSInt32Answer(kMGQMainScreenPitch, 0);
256 MGGetSInt32Answer(kMGQMainScreenClass, MGScreenClassPad2);
257 MGGetBoolAnswer(kMGQAppleInternalInstallCapability);
258 MGGetBoolAnswer(kMGQiPadCapability);
259 auto deviceName = adoptCF(MGCopyAnswer(kMGQDeviceName, nullptr));
260 MGGetSInt32Answer(kMGQDeviceClassNumber, MGDeviceClassInvalid);
261 MGGetBoolAnswer(kMGQHasExtendedColorDisplay);
262 MGGetFloat32Answer(kMGQDeviceCornerRadius, 0);
263 MGGetBoolAnswer(kMGQSupportsForceTouch);
264
265 auto answer = adoptCF(MGCopyAnswer(kMGQBluetoothCapability, nullptr));
266 answer = MGCopyAnswer(kMGQDeviceProximityCapability, nullptr);
267 answer = MGCopyAnswer(kMGQDeviceSupportsARKit, nullptr);
268 answer = MGCopyAnswer(kMGQTimeSyncCapability, nullptr);
269 answer = MGCopyAnswer(kMGQWAPICapability, nullptr);
270 answer = MGCopyAnswer(kMGQMainDisplayRotation, nullptr);
271#endif
272 ok = extension->revoke();
273 ASSERT_UNUSED(ok, ok);
274 }

504void WebProcess::updateProcessName(IsInProcessInitialization isInProcessInitialization)
505{
506#if PLATFORM(MAC)
507 RetainPtr<NSString> applicationName;
508 switch (m_processType) {
509 case ProcessType::Inspector:
510 applicationName = [NSString stringWithFormat:WEB_UI_STRING("%@ Web Inspector", "Visible name of Web Inspector's web process. The argument is the application name."), (NSString *)m_uiProcessName];
511 break;
512 case ProcessType::ServiceWorker:
513 applicationName = [NSString stringWithFormat:WEB_UI_STRING("%@ Service Worker (%@)", "Visible name of Service Worker process. The argument is the application name."), (NSString *)m_uiProcessName, (NSString *)m_registrableDomain.string()];
514 break;
515 case ProcessType::PrewarmedWebContent:
516 applicationName = [NSString stringWithFormat:WEB_UI_STRING("%@ Web Content (Prewarmed)", "Visible name of the web process. The argument is the application name."), (NSString *)m_uiProcessName];
517 break;
518 case ProcessType::CachedWebContent:
519 applicationName = [NSString stringWithFormat:WEB_UI_STRING("%@ Web Content (Cached)", "Visible name of the web process. The argument is the application name."), (NSString *)m_uiProcessName];
520 break;
521 case ProcessType::WebContent:
522 applicationName = [NSString stringWithFormat:WEB_UI_STRING("%@ Web Content", "Visible name of the web process. The argument is the application name."), (NSString *)m_uiProcessName];
523 break;
524 }
525
526#if ENABLE(SET_WEBCONTENT_PROCESS_INFORMATION_IN_NETWORK_PROCESS)
527 // During WebProcess initialization, we are still able to talk to LaunchServices to set the process name so there is no need to go
528 // via the NetworkProcess. Prewarmed WebProcesses also do not have a network process connection until they are actually used by
529 // a page.
530 if (isInProcessInitialization == IsInProcessInitialization::No) {
531 audit_token_t auditToken = { 0 };
532 mach_msg_type_number_t info_size = TASK_AUDIT_TOKEN_COUNT;
533 kern_return_t kr = task_info(mach_task_self(), TASK_AUDIT_TOKEN, reinterpret_cast<integer_t *>(&auditToken), &info_size);
534 if (kr != KERN_SUCCESS) {
535 WEBPROCESS_RELEASE_LOG_ERROR(Process, "updateProcessName: Unable to get audit token for self. Error: %{public}s (%x)", mach_error_string(kr), kr);
536 return;
537 }
538 String displayName = applicationName.get();
539 ensureNetworkProcessConnection().connection().send(Messages::NetworkConnectionToWebProcess::UpdateActivePages(displayName, Vector<String>(), auditToken), 0);
540 return;
541 }
542#endif // ENABLE(SET_WEBCONTENT_PROCESS_INFORMATION_IN_NETWORK_PROCESS)
543
544 // Note that it is important for _RegisterApplication() to have been called before setting the display name.

590#if USE(OS_STATE)
591void WebProcess::registerWithStateDumper()
592{
593 os_state_add_handler(dispatch_get_main_queue(), ^(os_state_hints_t hints) {
594
595 @autoreleasepool {
596 os_state_data_t os_state = nil;
597
598 // Only gather state on faults and sysdiagnose. It's overkill for
599 // general error messages.
600 if (hints->osh_api == OS_STATE_API_ERROR)
601 return os_state;
602
603 // Create a dictionary to contain the collected state. This
604 // dictionary will be serialized and passed back to os_state.
605 auto stateDict = adoptNS([[NSMutableDictionary alloc] init]);
606
607 {
608 auto memoryUsageStats = adoptNS([[NSMutableDictionary alloc] init]);
609 for (auto& it : PerformanceLogging::memoryUsageStatistics(ShouldIncludeExpensiveComputations::Yes)) {
610 auto keyString = adoptNS([[NSString alloc] initWithUTF8String:it.key]);
611 [memoryUsageStats setObject:@(it.value) forKey:keyString.get()];
612 }
613 [stateDict setObject:memoryUsageStats.get() forKey:@"Memory Usage Stats"];
614 }
615
616 {
617 auto jsObjectCounts = adoptNS([[NSMutableDictionary alloc] init]);
618 for (auto& it : PerformanceLogging::javaScriptObjectCounts()) {
619 auto keyString = adoptNS([[NSString alloc] initWithUTF8String:it.key]);
620 [jsObjectCounts setObject:@(it.value) forKey:keyString.get()];
621 }
622 [stateDict setObject:jsObjectCounts.get() forKey:@"JavaScript Object Counts"];
623 }
624
625 auto pageLoadTimes = createNSArray(m_pageMap.values(), [] (auto& page) -> id {
626 if (page->usesEphemeralSession())
627 return nil;
628
629 return [NSDate dateWithTimeIntervalSince1970:page->loadCommitTime().secondsSinceEpoch().seconds()];
630 });
631
632 // Adding an empty array to the process state may provide an
633 // indication of the existance of private sessions, which we'd like
634 // to hide, so don't add empty arrays.
635 if ([pageLoadTimes count])
636 [stateDict setObject:pageLoadTimes.get() forKey:@"Page Load Times"];
637
638 // --- Possibly add other state here as other entries in the dictionary. ---
639
640 // Submitting an empty process state object may provide an
641 // indication of the existance of private sessions, which we'd like
642 // to hide, so don't return empty dictionaries.
643 if (![stateDict count])
644 return os_state;
645
646 // Serialize the accumulated process state so that we can put the
647 // result in an os_state_data_t structure.
648 NSError* error = nil;
649 NSData* data = [NSPropertyListSerialization dataWithPropertyList:stateDict.get() format:NSPropertyListBinaryFormat_v1_0 options:0 error:&error];
650
651 if (!data) {
652 ASSERT(data);
653 return os_state;
654 }
655
656 size_t neededSize = OS_STATE_DATA_SIZE_NEEDED(data.length);
657 os_state = (os_state_data_t)malloc(neededSize);
658 if (os_state) {
659 memset(os_state, 0, neededSize);
660 os_state->osd_type = OS_STATE_DATA_SERIALIZED_NSCF_OBJECT;
661 os_state->osd_data_size = data.length;
662 strlcpy(os_state->osd_title, "WebContent state", sizeof(os_state->osd_title));
663 memcpy(os_state->osd_data, data.bytes, data.length);
664 }
665
666 return os_state;
667 }
668 });
669}

869void WebProcess::updateCPULimit()
870{
871#if PLATFORM(MAC)
872 std::optional<double> cpuLimit;
873 if (m_processType == ProcessType::ServiceWorker)
874 cpuLimit = serviceWorkerCPULimit;
875 else {
876 // Use the largest limit among all pages in this process.
877 for (auto& page : m_pageMap.values()) {
878 auto pageCPULimit = page->cpuLimit();
879 if (!pageCPULimit) {
880 cpuLimit = std::nullopt;
881 break;
882 }
883 if (!cpuLimit || pageCPULimit > cpuLimit.value())
884 cpuLimit = pageCPULimit;
885 }
886 }
887
888 if (m_cpuLimit == cpuLimit)
889 return;
890
891 m_cpuLimit = cpuLimit;
892 updateCPUMonitorState(CPUMonitorUpdateReason::LimitHasChanged);
893#endif
894}

896void WebProcess::updateCPUMonitorState(CPUMonitorUpdateReason reason)
897{
898#if PLATFORM(MAC)
899 if (!m_cpuLimit) {
900 if (m_cpuMonitor)
901 m_cpuMonitor->setCPULimit(std::nullopt);
902 return;
903 }
904
905 if (!m_cpuMonitor) {
906 m_cpuMonitor = makeUnique<CPUMonitor>(cpuMonitoringInterval, [this](double cpuUsage) {
907 if (m_processType == ProcessType::ServiceWorker)
908 WEBPROCESS_RELEASE_LOG_ERROR(ProcessSuspension, "updateCPUMonitorState: Service worker process exceeded CPU limit of %.1f%% (was using %.1f%%)", m_cpuLimit.value() * 100, cpuUsage * 100);
909 else
910 WEBPROCESS_RELEASE_LOG_ERROR(ProcessSuspension, "updateCPUMonitorState: WebProcess exceeded CPU limit of %.1f%% (was using %.1f%%) hasVisiblePages? %d", m_cpuLimit.value() * 100, cpuUsage * 100, hasVisibleWebPage());
911 parentProcessConnection()->send(Messages::WebProcessProxy::DidExceedCPULimit(), 0);
912 });
913 } else if (reason == CPUMonitorUpdateReason::VisibilityHasChanged) {
914 // If the visibility has changed, stop the CPU monitor before setting its limit. This is needed because the CPU usage can vary wildly based on visibility and we would
915 // not want to report that a process has exceeded its background CPU limit even though most of the CPU time was used while the process was visible.
916 m_cpuMonitor->setCPULimit(std::nullopt);
917 }
918 m_cpuMonitor->setCPULimit(m_cpuLimit);
919#else
920 UNUSED_PARAM(reason);
921#endif
922}

1201static void setPreferenceValue(const String& domain, const String& key, id value)
1202{
1203 if (domain.isEmpty()) {
1204 CFPreferencesSetValue(key.createCFString().get(), (__bridge CFPropertyListRef)value, kCFPreferencesAnyApplication, kCFPreferencesCurrentUser, kCFPreferencesAnyHost);
1205#if ASSERT_ENABLED
1206 id valueAfterSetting = [[NSUserDefaults standardUserDefaults] objectForKey:key];
1207 ASSERT(valueAfterSetting == value || [valueAfterSetting isEqual:value] || key == "AppleLanguages");
1208#endif
1209 } else
1210 CFPreferencesSetValue(key.createCFString().get(), (__bridge CFPropertyListRef)value, domain.createCFString().get(), kCFPreferencesCurrentUser, kCFPreferencesAnyHost);
1211
1212 if (key == "AppleLanguages") {
1213 // We need to set AppleLanguages for the volatile domain, similarly to what we do in XPCServiceMain.mm.
1214 NSDictionary *existingArguments = [[NSUserDefaults standardUserDefaults] volatileDomainForName:NSArgumentDomain];
1215 RetainPtr<NSMutableDictionary> newArguments = adoptNS([existingArguments mutableCopy]);
1216 [newArguments setValue:value forKey:@"AppleLanguages"];
1217 [[NSUserDefaults standardUserDefaults] setVolatileDomain:newArguments.get() forName:NSArgumentDomain];
1218
1219 WTF::languageDidChange();
1220 }
1221
1222 if (domain == String(kAXSAccessibilityPreferenceDomain)) {
1223#if HAVE(UPDATE_WEB_ACCESSIBILITY_SETTINGS) && ENABLE(CFPREFS_DIRECT_MODE)
1224 if (_AXSUpdateWebAccessibilitySettingsPtr())
1225 _AXSUpdateWebAccessibilitySettingsPtr()();
1226#elif PLATFORM(IOS_FAMILY)
1227 // If the update method is not available, to update the cache inside AccessibilitySupport,
1228 // these methods need to be called directly.

1230 _AXSSetReduceMotionEnabled([(NSNumber *)value boolValue]);

1232 _AXSSetDarkenSystemColors([(NSNumber *)value boolValue]);
1233#endif
1234 }
1235
1236#if USE(APPKIT)
1237 auto cfKey = key.createCFString();
1238 if (CFEqual(cfKey.get(), kAXInterfaceReduceMotionKey) || CFEqual(cfKey.get(), kAXInterfaceIncreaseContrastKey) || key == invertColorsPreferenceKey())

1243void WebProcess::notifyPreferencesChanged(const String& domain, const String& key, const std::optional<String>& encodedValue)
1244{
1245 if (!encodedValue) {
1246 setPreferenceValue(domain, key, nil);
1247 dispatchSimulatedNotificationsForPreferenceChange(key);
1248 return;
1249 }
1250 auto encodedData = adoptNS([[NSData alloc] initWithBase64EncodedString:*encodedValue options:0]);
1251 if (!encodedData)
1252 return;
1253 NSError *err = nil;
1254 auto classes = [NSSet setWithArray:@[[NSString class], [NSNumber class], [NSDate class], [NSDictionary class], [NSArray class], [NSData class]]];
1255 id object = [NSKeyedUnarchiver unarchivedObjectOfClasses:classes fromData:encodedData.get() error:&err];
1256 ASSERT(!err);
1257 if (err)
1258 return;
1259 setPreferenceValue(domain, key, object);
1260 dispatchSimulatedNotificationsForPreferenceChange(key);
