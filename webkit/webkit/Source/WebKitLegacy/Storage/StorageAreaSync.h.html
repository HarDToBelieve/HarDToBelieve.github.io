41class StorageAreaImpl;
42
43class StorageAreaSync : public ThreadSafeRefCounted<StorageAreaSync, WTF::DestructionThread::Main> {
44public:
45 static Ref<StorageAreaSync> create(RefPtr<WebCore::StorageSyncManager>&&, Ref<StorageAreaImpl>&&, const String& databaseIdentifier);
46 ~StorageAreaSync();
47
48 void scheduleFinalSync();
49 void blockUntilImportComplete();
50
51 void scheduleItemForSync(const String& key, const String& value);
52 void scheduleClear();
53 void scheduleCloseDatabase();
54
55 void scheduleSync();
56
57private:
58 StorageAreaSync(RefPtr<WebCore::StorageSyncManager>&&, Ref<StorageAreaImpl>&&, const String& databaseIdentifier);
59
60 WebCore::Timer m_syncTimer;
61 HashMap<String, String> m_changedItems;
62 bool m_itemsCleared;
63
64 bool m_finalSyncScheduled;
65
66 RefPtr<StorageAreaImpl> m_storageArea;
67 RefPtr<WebCore::StorageSyncManager> m_syncManager;
68
69 // The database handle will only ever be opened and used on the background thread.

79private:

85 void syncTimerFired();

88
89 const String m_databaseIdentifier;
90
91 Lock m_syncLock;
92 HashMap<String, String> m_itemsPendingSync;
93 bool m_clearItemsWhileSyncing;
94 bool m_syncScheduled;
95 bool m_syncInProgress;
96 bool m_databaseOpenFailed;
97
98 bool m_syncCloseDatabase;
99
100 mutable Lock m_importLock;
101 Condition m_importCondition;
102 bool m_importComplete WTF_GUARDED_BY_LOCK(m_importLock);
103 void markImported();
104 void migrateItemTableIfNeeded();
105};
