26#include "StorageThread.h"

28#include <wtf/AutodrainedPool.h>
29#include <wtf/HashSet.h>
30#include <wtf/MainThread.h>
31#include <wtf/NeverDestroyed.h>
32
33namespace WebCore {
34
35static HashSet<StorageThread*>& activeStorageThreads()
36{
37 ASSERT(isMainThread());
38 static NeverDestroyed<HashSet<StorageThread*>> threads;
39 return threads;
40}
41
42StorageThread::StorageThread(Type type)
43 : m_type(type)
44{
45 ASSERT(isMainThread());
46}
47
48StorageThread::~StorageThread()
49{
50 ASSERT(isMainThread());
51 ASSERT(!m_thread);
52}
53
54void StorageThread::start()
55{
56 ASSERT(isMainThread());
57 if (!m_thread) {
58 if (m_type == Type::LocalStorage) {
59 m_thread = Thread::create("LocalStorage", [this] {
60 threadEntryPoint();
61 });
62 } else {
63 ASSERT(m_type == Type::IndexedDB);
64 m_thread = Thread::create("IndexedDB", [this] {
65 threadEntryPoint();
66 });
67 }
68 }
69 activeStorageThreads().add(this);
70}
71
72void StorageThread::threadEntryPoint()
73{
74 ASSERT(!isMainThread());
75
76 while (auto function = m_queue.waitForMessage()) {
77 AutodrainedPool pool;
78 (*function)();
79 }
80}
81
82void StorageThread::dispatch(Function<void ()>&& function)
83{
84 ASSERT(isMainThread());
85 ASSERT(!m_queue.killed() && m_thread);
86 m_queue.append(makeUnique<Function<void ()>>(WTFMove(function)));
87}
88
89void StorageThread::terminate()
90{
91 ASSERT(isMainThread());
92 ASSERT(!m_queue.killed() && m_thread);
93 activeStorageThreads().remove(this);
94 // Even in weird, exceptional cases, don't wait on a nonexistent thread to terminate.
95 if (!m_thread)
96 return;
97
98 m_queue.append(makeUnique<Function<void ()>>([this] {
99 performTerminate();
100 }));
101 m_thread->waitForCompletion();
102 ASSERT(m_queue.killed());
103 m_thread = nullptr;
104}
105
106void StorageThread::performTerminate()
107{
108 ASSERT(!isMainThread());
109 m_queue.kill();
110}
111
112void StorageThread::releaseFastMallocFreeMemoryInAllThreads()
113{
114 for (auto& thread : activeStorageThreads())
115 thread->dispatch(&WTF::releaseFastMallocFreeMemory);
116}
