45WebCore::IDBClient::IDBConnectionToServer& WebDatabaseProvider::idbConnectionToServerForSession(PAL::SessionID sessionID)
46{
47 return m_idbServerMap.ensure(sessionID, [&sessionID] {
48 return sessionID.isEphemeral() ? InProcessIDBServer::create(sessionID) : InProcessIDBServer::create(sessionID, indexedDatabaseDirectoryPath());
49 }).iterator->value->connectionToServer();
50}
51
52void WebDatabaseProvider::deleteAllDatabases()
53{
54 for (auto& server : m_idbServerMap.values())
55 server->closeAndDeleteDatabasesModifiedSince(-WallTime::infinity());
56}
