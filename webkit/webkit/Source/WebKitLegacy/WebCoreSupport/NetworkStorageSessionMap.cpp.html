36static std::unique_ptr<WebCore::NetworkStorageSession>& defaultNetworkStorageSession()
37{
38 ASSERT(isMainThread());
39 static NeverDestroyed<std::unique_ptr<WebCore::NetworkStorageSession>> session;
40 return session;
41}

43static HashMap<PAL::SessionID, std::unique_ptr<WebCore::NetworkStorageSession>>& globalSessionMap()
44{
45 static NeverDestroyed<HashMap<PAL::SessionID, std::unique_ptr<WebCore::NetworkStorageSession>>> map;
46 return map;
47}

49WebCore::NetworkStorageSession* NetworkStorageSessionMap::storageSession(PAL::SessionID sessionID)
50{
51 if (sessionID == PAL::SessionID::defaultSessionID())
52 return &defaultStorageSession();
53 return globalSessionMap().get(sessionID);
54}
55
56WebCore::NetworkStorageSession& NetworkStorageSessionMap::defaultStorageSession()
57{
58 if (!defaultNetworkStorageSession())
59 defaultNetworkStorageSession() = makeUnique<WebCore::NetworkStorageSession>(PAL::SessionID::defaultSessionID());
60 return *defaultNetworkStorageSession();
61}

63void NetworkStorageSessionMap::switchToNewTestingSession()
64{
65#if PLATFORM(COCOA) || USE(CFURLCONNECTION)
66 // Session name should be short enough for shared memory region name to be under the limit, otherwise sandbox rules won't work (see <rdar://problem/13642852>).
67 auto session = WebCore::createPrivateStorageSession(makeString("WebKit Test-", getCurrentProcessID()).createCFString().get());
68
69 RetainPtr<CFHTTPCookieStorageRef> cookieStorage;
70 if (WebCore::NetworkStorageSession::processMayUseCookieAPI()) {
71 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
72 if (session)
73 cookieStorage = adoptCF(_CFURLStorageSessionCopyCookieStorage(kCFAllocatorDefault, session.get()));
74 }
75
76 defaultNetworkStorageSession() = makeUnique<WebCore::NetworkStorageSession>(PAL::SessionID::defaultSessionID(), WTFMove(session), WTFMove(cookieStorage));
77#endif
78}

80void NetworkStorageSessionMap::ensureSession(PAL::SessionID sessionID, const String& identifierBase)
81{
82#if PLATFORM(COCOA) || USE(CFURLCONNECTION)
83 auto addResult = globalSessionMap().add(sessionID, nullptr);
84 if (!addResult.isNewEntry)
85 return;
86
87 auto identifier = makeString(identifierBase, ".PrivateBrowsing.", createCanonicalUUIDString()).createCFString();
88
89 RetainPtr<CFURLStorageSessionRef> storageSession;
90 if (sessionID.isEphemeral())
91 storageSession = WebCore::createPrivateStorageSession(identifier.get());
92 else
93 storageSession = WebCore::NetworkStorageSession::createCFStorageSessionForIdentifier(identifier.get());
94
95 RetainPtr<CFHTTPCookieStorageRef> cookieStorage;
96 if (WebCore::NetworkStorageSession::processMayUseCookieAPI()) {
97 ASSERT(hasProcessPrivilege(ProcessPrivilege::CanAccessRawCookies));
98 if (storageSession)
99 cookieStorage = adoptCF(_CFURLStorageSessionCopyCookieStorage(kCFAllocatorDefault, storageSession.get()));
100 }
101
102 addResult.iterator->value = makeUnique<WebCore::NetworkStorageSession>(sessionID, WTFMove(storageSession), WTFMove(cookieStorage));
103
104#elif USE(CURL)
105 globalSessionMap().ensure(sessionID, [sessionID] {
106 return makeUnique<WebCore::NetworkStorageSession>(sessionID);
107 });
108#endif
109}
110
111void NetworkStorageSessionMap::destroySession(PAL::SessionID sessionID)
112{
113 globalSessionMap().remove(sessionID);
114}
