64WebResourceLoadScheduler::HostInformation* WebResourceLoadScheduler::hostForURL(const URL& url, CreateHostPolicy createHostPolicy)
65{
66 if (!url.protocolIsInHTTPFamily())
67 return m_nonHTTPProtocolHost;
68
69 m_hosts.checkConsistency();
70 String hostName = url.host().toString();
71 HostInformation* host = m_hosts.get(hostName);
72 if (!host && createHostPolicy == CreateIfNotFound) {
73 host = new HostInformation(hostName, maxRequestsInFlightPerHost);
74 m_hosts.add(hostName, host);
75 }
76 return host;
77}

92void WebResourceLoadScheduler::loadResource(Frame& frame, CachedResource& resource, ResourceRequest&& request, const ResourceLoaderOptions& options, CompletionHandler<void(RefPtr<WebCore::SubresourceLoader>&&)>&& completionHandler)
93{
94 SubresourceLoader::create(frame, resource, WTFMove(request), options, [this, completionHandler = WTFMove(completionHandler)] (RefPtr<WebCore::SubresourceLoader>&& loader) mutable {
95 if (loader)
96 scheduleLoad(loader.get());
97#if PLATFORM(IOS_FAMILY)
98 // Since we defer loader initialization until scheduling on iOS, the frame
99 // load delegate that would be called in SubresourceLoader::create() on
100 // other ports might be called in scheduleLoad() instead. Our contract to
101 // callers of this method is that a null loader is returned if the load was
102 // cancelled by a frame load delegate.
103 if (!loader || loader->reachedTerminalState())
104 return completionHandler(nullptr);
105#endif
106 completionHandler(WTFMove(loader));
107 });
108}

134void WebResourceLoadScheduler::scheduleLoad(ResourceLoader* resourceLoader)
135{
136 ASSERT(resourceLoader);
137
138#if PLATFORM(IOS_FAMILY)
139 // If there's a web archive resource for this URL, we don't need to schedule the load since it will never touch the network.
140 if (!isSuspendingPendingRequests() && resourceLoader->documentLoader()->archiveResourceForURL(resourceLoader->iOSOriginalRequest().url())) {
141 resourceLoader->startLoading();
142 return;
143 }
144#else
145 if (resourceLoader->documentLoader()->archiveResourceForURL(resourceLoader->request().url())) {
146 resourceLoader->start();
147 return;
148 }
149#endif
150
151#if PLATFORM(IOS_FAMILY)
152 HostInformation* host = hostForURL(resourceLoader->iOSOriginalRequest().url(), CreateIfNotFound);
153#else
154 HostInformation* host = hostForURL(resourceLoader->url(), CreateIfNotFound);
155#endif
156
157 ResourceLoadPriority priority = resourceLoader->request().priority();
158
159 bool hadRequests = host->hasRequests();
160 host->schedule(resourceLoader, priority);
161
162#if PLATFORM(COCOA)
163 if (ResourceRequest::resourcePrioritiesEnabled() && !isSuspendingPendingRequests()) {
164 // Serve all requests at once to keep the pipeline full at the network layer.
165 // FIXME: Does this code do anything useful, given that we also set maxRequestsInFlightPerHost to effectively unlimited on these platforms?
166 servePendingRequests(host, ResourceLoadPriority::VeryLow);
167 return;
168 }
169#endif
170
171#if PLATFORM(IOS_FAMILY)
172 if ((priority > ResourceLoadPriority::Low || !resourceLoader->iOSOriginalRequest().url().protocolIsInHTTPFamily() || (priority == ResourceLoadPriority::Low && !hadRequests)) && !isSuspendingPendingRequests()) {
173 // Try to request important resources immediately.
174 servePendingRequests(host, priority);
175 return;
176 }
177#else
178 if (priority > ResourceLoadPriority::Low || !resourceLoader->url().protocolIsInHTTPFamily() || (priority == ResourceLoadPriority::Low && !hadRequests)) {
179 // Try to request important resources immediately.
180 servePendingRequests(host, priority);
181 return;
182 }
183#endif
184
185 // Handle asynchronously so early low priority requests don't
186 // get scheduled before later high priority ones.
187 scheduleServePendingRequests();
188}

190void WebResourceLoadScheduler::remove(ResourceLoader* resourceLoader)
191{
192 ASSERT(resourceLoader);
193
194 HostInformation* host = hostForURL(resourceLoader->url());
195 if (host)
196 host->remove(resourceLoader);
197#if PLATFORM(IOS_FAMILY)
198 // ResourceLoader::url() doesn't start returning the correct value until the load starts. If we get canceled before that, we need to look for originalRequest url instead.
199 // FIXME: ResourceLoader::url() should be made to return a sensible value at all times.
200 if (!resourceLoader->iOSOriginalRequest().isNull()) {
201 HostInformation* originalHost = hostForURL(resourceLoader->iOSOriginalRequest().url());
202 if (originalHost && originalHost != host)
203 originalHost->remove(resourceLoader);
204 }
205#endif
206 scheduleServePendingRequests();
207}

226void WebResourceLoadScheduler::crossOriginRedirectReceived(ResourceLoader* resourceLoader, const URL& redirectURL)
227{
228 HostInformation* oldHost = hostForURL(resourceLoader->url());
229 ASSERT(oldHost);
230 if (!oldHost)
231 return;
232
233 HostInformation* newHost = hostForURL(redirectURL, CreateIfNotFound);
234
235 if (oldHost->name() == newHost->name())
236 return;
237
238 newHost->addLoadInProgress(resourceLoader);
239 oldHost->remove(resourceLoader);
240}

242void WebResourceLoadScheduler::servePendingRequests(ResourceLoadPriority minimumPriority)
243{
244 if (isSuspendingPendingRequests())
245 return;
246
247 m_requestTimer.stop();
248
249 servePendingRequests(m_nonHTTPProtocolHost, minimumPriority);
250
251 for (auto* host : copyToVector(m_hosts.values())) {
252 if (host->hasRequests())
253 servePendingRequests(host, minimumPriority);
254 else
255 delete m_hosts.take(host->name());
256 }
257}
258
259void WebResourceLoadScheduler::servePendingRequests(HostInformation* host, ResourceLoadPriority minimumPriority)
260{
261 auto priority = ResourceLoadPriority::Highest;
262 while (true) {
263 auto& requestsPending = host->requestsPending(priority);
264 while (!requestsPending.isEmpty()) {
265 RefPtr<ResourceLoader> resourceLoader = requestsPending.first();
266
267 // For named hosts - which are only http(s) hosts - we should always enforce the connection limit.
268 // For non-named hosts - everything but http(s) - we should only enforce the limit if the document isn't done parsing
269 // and we don't know all stylesheets yet.
270 Document* document = resourceLoader->frameLoader() ? resourceLoader->frameLoader()->frame().document() : 0;
271 bool shouldLimitRequests = !host->name().isNull() || (document && (document->parsing() || !document->haveStylesheetsLoaded()));
272 if (shouldLimitRequests && host->limitRequests(priority))
273 return;
274
275 requestsPending.removeFirst();
276 host->addLoadInProgress(resourceLoader.get());
277#if PLATFORM(IOS_FAMILY)
278 if (!IOSApplication::isWebProcess()) {
279 resourceLoader->startLoading();
280 return;
281 }
282#endif
283 resourceLoader->start();
284 }
285 if (priority == minimumPriority)
286 return;
287 --priority;
288 }
289}

291void WebResourceLoadScheduler::suspendPendingRequests()
292{
293 ++m_suspendPendingRequestsCount;
294}

296void WebResourceLoadScheduler::resumePendingRequests()
297{
298 ASSERT(m_suspendPendingRequestsCount);
299 --m_suspendPendingRequestsCount;
300 if (m_suspendPendingRequestsCount)
301 return;
302 if (!m_hosts.isEmpty() || m_nonHTTPProtocolHost->hasRequests())
303 scheduleServePendingRequests();
304}

306void WebResourceLoadScheduler::scheduleServePendingRequests()
307{
308 if (!m_requestTimer.isActive())
309 m_requestTimer.startOneShot(0_s);
310}

328unsigned WebResourceLoadScheduler::HostInformation::priorityToIndex(ResourceLoadPriority priority)
329{
330 switch (priority) {
331 case ResourceLoadPriority::VeryLow:
332 return 0;
333 case ResourceLoadPriority::Low:
334 return 1;
335 case ResourceLoadPriority::Medium:
336 return 2;
337 case ResourceLoadPriority::High:
338 return 3;
339 case ResourceLoadPriority::VeryHigh:
340 return 4;
341 }
342 ASSERT_NOT_REACHED();
343 return 0;
344}

346void WebResourceLoadScheduler::HostInformation::schedule(ResourceLoader* resourceLoader, ResourceLoadPriority priority)
347{
348 m_requestsPending[priorityToIndex(priority)].append(resourceLoader);
349}

351void WebResourceLoadScheduler::HostInformation::addLoadInProgress(ResourceLoader* resourceLoader)
352{
353 m_requestsLoading.add(resourceLoader);
354}
355
356void WebResourceLoadScheduler::HostInformation::remove(ResourceLoader* resourceLoader)
357{
358 if (m_requestsLoading.remove(resourceLoader))
359 return;
360
361 for (auto& requestQueue : m_requestsPending) {
362 for (auto it = requestQueue.begin(), end = requestQueue.end(); it != end; ++it) {
363 if (*it == resourceLoader) {
364 requestQueue.remove(it);
365 return;
366 }
367 }
368 }
369}

371bool WebResourceLoadScheduler::HostInformation::hasRequests() const
372{
373 if (!m_requestsLoading.isEmpty())
374 return true;
375 for (auto& requestQueue : m_requestsPending) {
376 if (!requestQueue.isEmpty())
377 return true;
378 }
379 return false;
380}
381
382bool WebResourceLoadScheduler::HostInformation::limitRequests(ResourceLoadPriority priority) const
383{
384 if (priority == ResourceLoadPriority::VeryLow && !m_requestsLoading.isEmpty())
385 return true;
386 return m_requestsLoading.size() >= (webResourceLoadScheduler().isSerialLoadingEnabled() ? 1 : m_maxRequestsInFlight);
387}
