62static const char* inspectorStartsAttachedSetting = "inspectorStartsAttached";
63static const char* inspectorAttachDisabledSetting = "inspectorAttachDisabled";
64
65static inline RetainPtr<CFStringRef> createKeyForPreferences(const String& key)
66{
67 return adoptCF(CFStringCreateWithFormat(0, 0, CFSTR("WebKit Web Inspector Setting - %@"), key.createCFString().get()));
68}
69
70static void populateSetting(const String& key, String* setting)
71{
72 RetainPtr<CFStringRef> preferencesKey = createKeyForPreferences(key);
73 RetainPtr<CFPropertyListRef> value = adoptCF(CFPreferencesCopyAppValue(preferencesKey.get(), kCFPreferencesCurrentApplication));
74
75 if (!value)
76 return;
77
78 CFTypeID type = CFGetTypeID(value.get());
79 if (type == CFStringGetTypeID())
80 *setting = static_cast<String>(static_cast<CFStringRef>(value.get()));
81 else if (type == CFBooleanGetTypeID())
82 *setting = static_cast<bool>(CFBooleanGetValue(static_cast<CFBooleanRef>(value.get()))) ? "true" : "false";
83 else
84 *setting = emptyString();
85}
86
87static void storeSetting(const String& key, const String& setting)
88{
89 CFPreferencesSetAppValue(createKeyForPreferences(key).get(), setting.createCFString().get(), kCFPreferencesCurrentApplication);
90}
91
92static void deleteSetting(const String& key)
93{
94 CFPreferencesSetAppValue(createKeyForPreferences(key).get(), nullptr, kCFPreferencesCurrentApplication);
95}

97void WebInspectorClient::sendMessageToFrontend(const String& message)
98{
99 m_frontendClient->frontendAPIDispatcher().dispatchMessageAsync(message);
100}
101
102bool WebInspectorClient::inspectorAttachDisabled()
103{
104 String value;
105 populateSetting(inspectorAttachDisabledSetting, &value);
106 if (value.isEmpty())
107 return false;
108 return value == "true";
109}
110
111void WebInspectorClient::setInspectorAttachDisabled(bool disabled)
112{
113 storeSetting(inspectorAttachDisabledSetting, disabled ? "true" : "false");
114}
115
116void WebInspectorClient::deleteInspectorStartsAttached()
117{
118 deleteSetting(inspectorAttachDisabledSetting);
119}
120
121bool WebInspectorClient::inspectorStartsAttached()
122{
123 String value;
124 populateSetting(inspectorStartsAttachedSetting, &value);
125 if (value.isEmpty())
126 return true;
127 return value == "true";
128}
129
130void WebInspectorClient::setInspectorStartsAttached(bool attached)
131{
132 storeSetting(inspectorStartsAttachedSetting, attached ? "true" : "false");
133}
134
135void WebInspectorClient::deleteInspectorAttachDisabled()
136{
137 deleteSetting(inspectorStartsAttachedSetting);
138}

140std::unique_ptr<WebCore::InspectorFrontendClientLocal::Settings> WebInspectorClient::createFrontendSettings()
141{
142 class InspectorFrontendSettingsCF : public WebCore::InspectorFrontendClientLocal::Settings {
143 public:
144 virtual ~InspectorFrontendSettingsCF() { }
145 virtual String getProperty(const String& name)
146 {
147 String value;
148 populateSetting(name, &value);
149 return value;
150 }
151
152 virtual void setProperty(const String& name, const String& value)
153 {
154 storeSetting(name, value);
155 }
156
157 virtual void deleteProperty(const String& name)
158 {
159 deleteSetting(name);
160 }
161 };
162 return makeUnique<InspectorFrontendSettingsCF>();
163}
