57+ (WebVisiblePosition *)_wrapVisiblePosition:(VisiblePosition)visiblePosition
58{
59 auto vp = adoptNS([[WebVisiblePosition alloc] init]);
60 VisiblePosition *copy = new VisiblePosition(visiblePosition);
61 vp->_internal = reinterpret_cast<WebObjectInternal *>(copy);
62 return vp.autorelease();
63}

65// Returns nil if visible position is null.

67{
68 return (visiblePosition.isNotNull() ? [WebVisiblePosition _wrapVisiblePosition:visiblePosition] : nil);
69}

98- (NSComparisonResult)compare:(WebVisiblePosition *)other
99{
100 VisiblePosition myVP = [self _visiblePosition];
101 VisiblePosition otherVP = [other _visiblePosition];
102
103 if (myVP == otherVP)
104 return NSOrderedSame;
105 else if (myVP < otherVP)
106 return NSOrderedAscending;
107 else
108 return NSOrderedDescending;
109}

136- (BOOL)directionIsDownstream:(WebTextAdjustmentDirection)direction
137{
138 if (direction == WebTextAdjustmentBackward)
139 return NO;
140
141 if (direction == WebTextAdjustmentForward)
142 return YES;
143
144
145 if ([self textDirection] == TextDirection::LTR)
146 return (direction == WebTextAdjustmentRight);
147 return (direction == WebTextAdjustmentLeft);
148}

150- (WebVisiblePosition *)positionByMovingInDirection:(WebTextAdjustmentDirection)direction amount:(UInt32)amount withAffinityDownstream:(BOOL)affinityDownstream
151{
152 auto vp = [self _visiblePosition];
153 vp.setAffinity(affinityDownstream ? Affinity::Downstream : Affinity::Upstream);
154
155 switch (direction) {
156 case WebTextAdjustmentForward: {
157 for (UInt32 i = 0; i < amount; i++)
158 vp = vp.next();
159 break;
160 }
161 case WebTextAdjustmentBackward: {
162 for (UInt32 i = 0; i < amount; i++)
163 vp = vp.previous();
164 break;
165 }
166 case WebTextAdjustmentRight: {
167 for (UInt32 i = 0; i < amount; i++)
168 vp = vp.right();
169 break;
170 }
171 case WebTextAdjustmentLeft: {
172 for (UInt32 i = 0; i < amount; i++)
173 vp = vp.left();
174 break;
175 }
176 case WebTextAdjustmentUp: {
177 int xOffset = vp.lineDirectionPointForBlockDirectionNavigation();
178 for (UInt32 i = 0; i < amount; i++)
179 vp = previousLinePosition(vp, xOffset);
180 break;
181 }
182 case WebTextAdjustmentDown: {
183 int xOffset = vp.lineDirectionPointForBlockDirectionNavigation();
184 for (UInt32 i = 0; i < amount; i++)
185 vp = nextLinePosition(vp, xOffset);
186 break;
187 }
188 default: {
189 ASSERT_NOT_REACHED();
190 break;
191 }
192 }
193 return [WebVisiblePosition _wrapVisiblePositionIfValid:vp];
194}

201static inline TextGranularity toTextGranularity(WebTextGranularity webGranularity)
202{
203 TextGranularity granularity;
204
205 switch (webGranularity) {
206 case WebTextGranularityCharacter:
207 granularity = TextGranularity::CharacterGranularity;
208 break;
209
210 case WebTextGranularityWord:
211 granularity = TextGranularity::WordGranularity;
212 break;
213
214 case WebTextGranularitySentence:
215 granularity = TextGranularity::SentenceGranularity;
216 break;
217
218 case WebTextGranularityLine:
219 granularity = TextGranularity::LineGranularity;
220 break;
221
222 case WebTextGranularityParagraph:
223 granularity = TextGranularity::ParagraphGranularity;
224 break;
225
226 case WebTextGranularityAll:
227 granularity = TextGranularity::DocumentGranularity;
228 break;
229
230 default:
231 ASSERT_NOT_REACHED();
232 break;
233 }
234
235 return granularity;
236}

238static inline SelectionDirection toSelectionDirection(WebTextAdjustmentDirection direction)
239{
240 SelectionDirection result;
241
242 switch (direction) {
243 case WebTextAdjustmentForward:
244 result = SelectionDirection::Forward;
245 break;
246
247 case WebTextAdjustmentBackward:
248 result = SelectionDirection::Backward;
249 break;
250
251 case WebTextAdjustmentRight:
252 result = SelectionDirection::Right;
253 break;
254
255 case WebTextAdjustmentLeft:
256 result = SelectionDirection::Left;
257 break;
258
259 case WebTextAdjustmentUp:
260 result = SelectionDirection::Left;
261 break;
262
263 case WebTextAdjustmentDown:
264 result = SelectionDirection::Right;
265 break;
266 }
267
268 return result;
269}

297- (WebVisiblePosition *)positionAtStartOrEndOfWord
298{
299 // Ripped from WebCore::Frame::moveSelectionToStartOrEndOfCurrentWord
300
301 // Note: this is the iOS notion, not the unicode notion.
302 // Here, a word starts with non-whitespace or at the start of a line and
303 // ends at the next whitespace, or at the end of a line.
304
305 // Selection rule: If the selection is before the first character or
306 // just after the first character of a word that is longer than one
307 // character, move to the start of the word. Otherwise, move to the
308 // end of the word.
309
310 VisiblePosition pos = [self _visiblePosition];
311 VisiblePosition originalPos(pos);
312
313 UChar32 ch = pos.characterAfter();
314 bool isComplex = requiresContextForWordBoundary(ch);
315 if (isComplex) {
316 // for complex layout, find word around insertion point
317 VisiblePosition visibleWordStart = startOfWord(pos);
318 Position wordStart = visibleWordStart.deepEquivalent();
319
320 // place caret in front of word if pos is within first 2 characters of word (see Selection Rule above)
321 if (wordStart.deprecatedEditingOffset() + 1 >= pos.deepEquivalent().deprecatedEditingOffset()) {
322 pos = wordStart;
323 } else {
324 // calculate end of word to insert caret after word
325 VisiblePosition visibleWordEnd = endOfWord(pos);
326 Position wordEnd = visibleWordEnd.deepEquivalent();
327 pos = wordEnd;
328 }
329 } else {
330 UChar32 c = pos.characterAfter();
331 CFCharacterSetRef set = CFCharacterSetGetPredefined(kCFCharacterSetWhitespaceAndNewline);
332 if (c == 0 || CFCharacterSetIsLongCharacterMember(set, c)) {
333 // search backward for a non-space
334 while (1) {
335 if (pos.isNull() || isStartOfLine(pos))
336 break;
337 c = pos.characterBefore();
338 if (!CFCharacterSetIsLongCharacterMember(set, c))
339 break;
340 pos = pos.previous(CannotCrossEditingBoundary); // stay in editable content
341 }
342 }
343 else {
344 // See how far the selection extends into the current word.
345 // Follow the rule stated above.
346 int index = 0;
347 while (1) {
348 if (pos.isNull() || isStartOfLine(pos))
349 break;
350 c = pos.characterBefore();
351 if (CFCharacterSetIsLongCharacterMember(set, c))
352 break;
353 pos = pos.previous(CannotCrossEditingBoundary); // stay in editable content
354 index++;
355 if (index > 1)
356 break;
357 }
358 if (index > 1) {
359 // search forward for a space
360 pos = originalPos;
361 while (1) {
362 if (pos.isNull() || isEndOfLine(pos))
363 break;
364 c = pos.characterAfter();
365 if (CFCharacterSetIsLongCharacterMember(set, c))
366 break;
367 pos = pos.next(CannotCrossEditingBoundary); // stay in editable content
368 }
369 }
370 }
371 }
372
373 return [WebVisiblePosition _wrapVisiblePositionIfValid:pos];
374}

387- (BOOL)atAlphaNumericBoundaryInDirection:(WebTextAdjustmentDirection)direction
388{
389 static CFCharacterSetRef set = CFCharacterSetGetPredefined(kCFCharacterSetAlphaNumeric);
390 VisiblePosition pos = [self _visiblePosition];
391 UChar32 charBefore = pos.characterBefore();
392 UChar32 charAfter = pos.characterAfter();
393 bool before = CFCharacterSetIsCharacterMember(set, charBefore);
394 bool after = CFCharacterSetIsCharacterMember(set, charAfter);
395 return [self directionIsDownstream:direction] ? (before && !after) : (!before && after);
396}

398- (DOMRange *)enclosingRangeWithDictationPhraseAlternatives:(NSArray **)alternatives
399{
400 ASSERT(alternatives);
401 if (!alternatives)
402 return nil;
403
404 // *alternatives should not already point to an array.
405 ASSERT(!*alternatives);
406 *alternatives = nil;
407
408 auto position = [self _visiblePosition];
409 auto* node = position.deepEquivalent().anchorNode();
410 if (!node)
411 return nil;
412
413 unsigned offset = position.deepEquivalent().deprecatedEditingOffset();
414 auto& document = node->document();
415 for (auto marker : document.markers().markersFor(*node, DocumentMarker::DictationPhraseWithAlternatives)) {
416 if (marker->startOffset() <= offset && marker->endOffset() >= offset) {
417 *alternatives = createNSArray(WTF::get<Vector<String>>(marker->data())).autorelease();
418 return kit(makeSimpleRange(*node, *marker));
419 }
420 }
421 return nil;
422}

424- (DOMRange *)enclosingRangeWithCorrectionIndicator
425{
426 auto position = [self _visiblePosition];
427 auto* node = position.deepEquivalent().anchorNode();
428 if (!node)
429 return nil;
430
431 unsigned offset = position.deepEquivalent().deprecatedEditingOffset();
432 auto& document = node->document();
433 for (auto marker : document.markers().markersFor(*node, DocumentMarker::Spelling)) {
434 if (marker->startOffset() <= offset && marker->endOffset() >= offset)
435 return kit(makeSimpleRange(*node, *marker));
436 }
437 return nil;
438}

475+ (DOMRange *)rangeForFirstPosition:(WebVisiblePosition *)first second:(WebVisiblePosition *)second
476{
477 auto firstPosition = [first _visiblePosition];
478 auto secondPosition = [second _visiblePosition];
479 if (secondPosition < firstPosition)
480 std::swap(firstPosition, secondPosition);
481 return kit(makeSimpleRange(firstPosition, secondPosition));
482}
