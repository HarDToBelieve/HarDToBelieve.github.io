258- (void)_doPostLoadOrUnlockTasks
259{
260 if (!_document)
261 return;
262
263 [self _updateTitleForDocumentIfAvailable];
264 [self _evaluateJSForDocument:_document];
265
266 // Any remaining work on the document should be done before this call to -layout,
267 // which will hand ownership of the document to the delegate (UIWebPDFView) and
268 // release the placeholder's document.
269 [self layout];
270}

272- (void)finishedLoadingWithDataSource:(WebDataSource *)dataSource
273{
274 [self dataSourceUpdated:dataSource];
275
276 _didFinishLoad = YES;
277 auto provider = adoptCF(CGDataProviderCreateWithCFData((CFDataRef)[dataSource data]));
278 if (!provider)
279 return;
280
281 _document = CGPDFDocumentCreateWithProvider(provider.get());
282
283 [self _doPostLoadOrUnlockTasks];

298- (void)_evaluateJSForDocument:(CGPDFDocumentRef)pdfDocument
299{
300 if (!pdfDocument || !CGPDFDocumentIsUnlocked(pdfDocument))
301 return;
302
303 NSArray *scripts = allScriptsInPDFDocument(pdfDocument);
304
305 if ([scripts count]) {
306 JSGlobalContextRef ctx = JSGlobalContextCreate(0);
307 JSObjectRef jsPDFDoc = makeJSPDFDoc(ctx, _dataSource);
308 for (NSString *script in scripts)
309 JSEvaluateScript(ctx, OpaqueJSString::tryCreate(script).get(), jsPDFDoc, nullptr, 0, nullptr);
310 JSGlobalContextRelease(ctx);
311 }
312}

314- (void)_updateTitleForURL:(NSURL *)URL
315{
316 NSString *titleFromURL = [URL lastPathComponent];
317 if (![titleFromURL length] || [titleFromURL isEqualToString:@"/"])
318 titleFromURL = [[URL _web_hostString] _webkit_decodeHostName];
319
320 [self setTitle:titleFromURL];
321 [[self _frame] _dispatchDidReceiveTitle:titleFromURL];
322}
323
324- (void)_updateTitleForDocumentIfAvailable
325{
326 if (!_document || !CGPDFDocumentIsUnlocked(_document))
327 return;
328
329 RetainPtr<CFStringRef> title;
330
331 CGPDFDictionaryRef info = CGPDFDocumentGetInfo(_document);
332 CGPDFStringRef value;
333 if (CGPDFDictionaryGetString(info, "Title", &value))
334 title = adoptCF(CGPDFStringCopyTextString(value));
335
336 if (title && CFStringGetLength(title.get())) {
337 [self setTitle:(NSString *)title.get()];
338 [[self _frame] _dispatchDidReceiveTitle:(NSString *)title.get()];
339 }
340}

342- (CGRect)_getPDFPageBounds:(CGPDFPageRef)page
343{
344 if (!page)
345 return CGRectZero;
346
347 CGRect bounds = CGPDFPageGetBoxRect(page, kCGPDFCropBox);
348 CGFloat rotation = CGPDFPageGetRotationAngle(page) * (M_PI / 180.0f);
349 if (rotation != 0)
350 bounds = CGRectApplyAffineTransform(bounds, CGAffineTransformMakeRotation(rotation));
351
352 bounds.origin.x = roundf(bounds.origin.x);
353 bounds.origin.y = roundf(bounds.origin.y);
354 bounds.size.width = roundf(bounds.size.width);
355 bounds.size.height = roundf(bounds.size.height);
356
357 return bounds;
358}

360- (CGSize)_computePageRects:(CGPDFDocumentRef)pdfDocument
361{
362 // Do we even need to compute the page rects?
363 if (self.pageRects)
364 return [self bounds].size;
365
366 if (!pdfDocument)
367 return CGSizeZero;
368
369 if (!CGPDFDocumentIsUnlocked(pdfDocument))
370 return CGSizeZero;
371
372 size_t pageCount = CGPDFDocumentGetNumberOfPages(pdfDocument);
373 if (!pageCount)
374 return CGSizeZero;
375
376 NSMutableArray *pageRects = [NSMutableArray array];
377 if (!pageRects)
378 return CGSizeZero;
379
380 NSMutableArray *pageYOrigins = [NSMutableArray array];
381 if (!pageYOrigins)
382 return CGSizeZero;
383
384 // Temporary vector to avoid getting the page rects twice.
385 WTF::Vector<CGRect> pageCropBoxes;
386
387 // First we need to determine the width of the widest page.
388 CGFloat maxPageWidth = 0.0f;
389
390 // CG uses page numbers instead of page indices, so 1 based.
391 for (size_t i = 1; i <= pageCount; i++) {
392 CGPDFPageRef page = CGPDFDocumentGetPage(pdfDocument, i);
393 if (!page) {
394 // So if there is a missing page, then effectively the document ends here.
395 // See <rdar://problem/10428152> iOS Mail crashes when opening a PDF
396 pageCount = i - 1;
397 break;
398 }
399
400 CGRect pageRect = [self _getPDFPageBounds:page];
401
402 maxPageWidth = std::max<CGFloat>(maxPageWidth, pageRect.size.width);
403
404 pageCropBoxes.append(pageRect);
405 }
406
407 if (!pageCount)
408 return CGSizeZero;
409
410 CGFloat scalingFactor = _containerSize.width / maxPageWidth;
411 if (_containerSize.width < FLT_EPSILON)
412 scalingFactor = 1.0;
413
414 // Including the margins, what is the desired width of the content?
415 CGFloat desiredWidth = (maxPageWidth + (PAGE_WIDTH_INSET * 2.0f)) * scalingFactor;
416 CGFloat desiredHeight = PAGE_HEIGHT_INSET;
417
418 for (size_t i = 0; i < pageCount; i++) {
419
420 CGRect pageRect = pageCropBoxes[i];
421
422 // Apply our scaling to this page's bounds.
423 pageRect.size.width = roundf(pageRect.size.width * scalingFactor);
424 pageRect.size.height = roundf(pageRect.size.height * scalingFactor);
425
426 // Center this page horizontally and update the starting vertical offset.
427 pageRect.origin.x = roundf((desiredWidth - pageRect.size.width) / 2.0f);
428 pageRect.origin.y = desiredHeight;
429
430 // Save this page's rect and minimum y-offset.
431 [pageRects addObject:[NSValue _web_valueWithCGRect:pageRect]];
432 [pageYOrigins addObject:[NSNumber numberWithFloat:CGRectGetMinY(pageRect)]];
433
434 // Update the next starting vertical offset.
435 desiredHeight += pageRect.size.height + PAGE_HEIGHT_INSET;
436 }
437
438 // Save the resulting page rects array and minimum y-offsets array.
439 self.pageRects = pageRects;
440 self.pageYOrigins = pageYOrigins;
441
442 // Determine the result desired bounds.
443 return CGSizeMake(roundf(desiredWidth), roundf(desiredHeight));
444}

451- (CGRect)rectForPageNumber:(NSUInteger)pageNumber
452{
453 // Page number is 1-based, not 0 based.
454 if ((!pageNumber) || (pageNumber > [_pageRects count]))
455 return CGRectNull;
456
457 return [[_pageRects objectAtIndex:pageNumber - 1] _web_CGRectValue];
458}

460- (void)simulateClickOnLinkToURL:(NSURL *)URL
461{
462 if (!URL)
463 return;
464
465 auto event = MouseEvent::create(eventNames().clickEvent, Event::CanBubble::Yes, Event::IsCancelable::Yes, Event::IsComposed::Yes,
466 MonotonicTime::now(), nullptr, 1, { }, { }, { }, { }, 0, 0, nullptr, 0, 0, MouseEvent::IsSimulated::Yes);
467
468 // Call to the frame loader because this is where our security checks are made.
469 Frame* frame = core([_dataSource webFrame]);
470 FrameLoadRequest frameLoadRequest { *frame->document(), frame->document()->securityOrigin(), { URL }, { }, InitiatedByMainFrame::Unknown };
471frameLoadRequest.setReferrerPolicy(ReferrerPolicy::NoReferrer);
472 frame->loader().loadFrameRequest(WTFMove(frameLoadRequest), event.ptr(), nullptr);
473}
