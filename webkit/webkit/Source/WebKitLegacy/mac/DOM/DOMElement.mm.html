49static inline WebCore::Element& unwrap(DOMElement& wrapper)
50{
51 ASSERT(wrapper._internal);
52 return downcast<WebCore::Element>(reinterpret_cast<WebCore::Node&>(*wrapper._internal));
53}
54
55WebCore::Element* core(DOMElement *wrapper)
56{
57 return wrapper ? &unwrap(*wrapper) : nullptr;
58}
59
60DOMElement *kit(WebCore::Element* value)
61{
62 WebCoreThreadViolationCheckRoundOne();
63 return static_cast<DOMElement*>(kit(static_cast<WebCore::Node*>(value)));
64}
