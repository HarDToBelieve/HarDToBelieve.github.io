207WebCore::HTMLDocument* core(DOMHTMLDocument *wrapper)
208{
209 return wrapper ? reinterpret_cast<WebCore::HTMLDocument*>(wrapper->_internal) : 0;
210}
211
212DOMHTMLDocument *kit(WebCore::HTMLDocument* value)
213{
214 WebCoreThreadViolationCheckRoundOne();
215 return static_cast<DOMHTMLDocument*>(kit(static_cast<WebCore::Node*>(value)));
216}
