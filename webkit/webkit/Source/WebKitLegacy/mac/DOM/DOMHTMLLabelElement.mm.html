44@implementation DOMHTMLLabelElement

46- (DOMHTMLFormElement *)form

58- (void)setHtmlFor:(NSString *)newHtmlFor

62}

64- (DOMHTMLElement *)control

66 WebCore::JSMainThreadNullState state;

76- (void)setAccessKey:(NSString *)newAccessKey
77{
78 WebCore::JSMainThreadNullState state;

80}
