124WebCore::HTMLOptionElement* core(DOMHTMLOptionElement *wrapper)
125{
126 return wrapper ? reinterpret_cast<WebCore::HTMLOptionElement*>(wrapper->_internal) : 0;
127}

129DOMHTMLOptionElement *kit(WebCore::HTMLOptionElement* value)
130{
131 WebCoreThreadViolationCheckRoundOne();
132 return static_cast<DOMHTMLOptionElement*>(kit(static_cast<WebCore::Node*>(value)));
133}
