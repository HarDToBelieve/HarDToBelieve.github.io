48- (void)dealloc
49{

56}

110DOMHTMLOptionsCollection *kit(WebCore::HTMLOptionsCollection* value)
111{
112 WebCoreThreadViolationCheckRoundOne();
113 if (!value)
114 return nil;
115 if (DOMHTMLOptionsCollection *wrapper = getDOMWrapper(value))
116 return retainPtr(wrapper).autorelease();
117 auto wrapper = adoptNS([[DOMHTMLOptionsCollection alloc] _init]);
118 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
119 value->ref();
120 addDOMWrapper(wrapper.get(), value);
121 return wrapper.autorelease();
122}
