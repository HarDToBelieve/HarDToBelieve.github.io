40#define IMPL static_cast<WebCore::HTMLStyleElement*>(reinterpret_cast<WebCore::Node*>(_internal))
41
42@implementation DOMHTMLStyleElement
43
44- (BOOL)disabled
45{
46 WebCore::JSMainThreadNullState state;
47 return IMPL->disabled();
48}
49
50- (void)setDisabled:(BOOL)newDisabled
51{
52 WebCore::JSMainThreadNullState state;
53 IMPL->setDisabled(newDisabled);
54}
55
56- (NSString *)media
57{
58 WebCore::JSMainThreadNullState state;
59 return IMPL->getAttribute(WebCore::HTMLNames::mediaAttr);
60}
61
62- (void)setMedia:(NSString *)newMedia
63{
64 WebCore::JSMainThreadNullState state;
65 IMPL->setAttributeWithoutSynchronization(WebCore::HTMLNames::mediaAttr, newMedia);
66}
67
68- (NSString *)type
69{
70 WebCore::JSMainThreadNullState state;
71 return IMPL->getAttribute(WebCore::HTMLNames::typeAttr);
72}
73
74- (void)setType:(NSString *)newType
75{
76 WebCore::JSMainThreadNullState state;
77 IMPL->setAttributeWithoutSynchronization(WebCore::HTMLNames::typeAttr, newType);
78}
79
80- (DOMStyleSheet *)sheet
81{
82 WebCore::JSMainThreadNullState state;
83 return kit(WTF::getPtr(IMPL->sheet()));
84}
85
86- (NSString *)nonce
87{
88 WebCore::JSMainThreadNullState state;
89 return IMPL->getAttribute(WebCore::HTMLNames::nonceAttr);
90}
91
92- (void)setNonce:(NSString *)newNonce
93{
94 WebCore::JSMainThreadNullState state;
95 IMPL->setAttributeWithoutSynchronization(WebCore::HTMLNames::nonceAttr, newNonce);
96}
97
98@end
99
100WebCore::HTMLStyleElement* core(DOMHTMLStyleElement *wrapper)
101{
102 return wrapper ? reinterpret_cast<WebCore::HTMLStyleElement*>(wrapper->_internal) : 0;
103}

105#undef IMPL
