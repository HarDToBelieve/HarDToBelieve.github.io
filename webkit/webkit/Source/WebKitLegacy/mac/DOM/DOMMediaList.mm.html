43@implementation DOMMediaList
44
45- (void)dealloc
46{
47 if (WebCoreObjCScheduleDeallocateOnMainThread([DOMMediaList class], self))
48 return;
49
50 if (_internal)
51 IMPL->deref();
52 [super dealloc];
53}

93DOMMediaList *kit(WebCore::MediaList* value)
94{
95 WebCoreThreadViolationCheckRoundOne();
96 if (!value)
97 return nil;
98 if (DOMMediaList *wrapper = getDOMWrapper(value))
99 return retainPtr(wrapper).autorelease();
100 auto wrapper = adoptNS([[DOMMediaList alloc] _init]);
101 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
102 value->ref();
103 addDOMWrapper(wrapper.get(), value);
104 return wrapper.autorelease();
105}
