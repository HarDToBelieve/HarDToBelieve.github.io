48static inline Node& unwrap(DOMNode& wrapper)
49{
50 ASSERT(wrapper._internal);
51 return reinterpret_cast<Node&>(*wrapper._internal);
52}
53
54Node* core(DOMNode *wrapper)
55{
56 return wrapper ? &unwrap(*wrapper) : nullptr;
57}
58
59DOMNode *kit(Node* value)
60{
61 WebCoreThreadViolationCheckRoundOne();
62 if (!value)
63 return nil;
64 if (DOMNode *wrapper = getDOMWrapper(value))
65 return retainPtr(wrapper).autorelease();
66 RetainPtr<DOMNode> wrapper = adoptNS([[kitClass(value) alloc] _init]);
67 if (!wrapper)
68 return nil;
69 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
70 value->ref();
71 addDOMWrapper(wrapper.get(), value);
72 return wrapper.autorelease();
73}
