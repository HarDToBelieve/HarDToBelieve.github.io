45@implementation DOMNodeIterator

47- (void)dealloc
48{
49 if (WebCoreObjCScheduleDeallocateOnMainThread([DOMNodeIterator class], self))
50 return;
51
52 if (_internal) {
53 [self detach];
54 IMPL->deref();
55 };
56 [super dealloc];
57}

124DOMNodeIterator *kit(WebCore::NodeIterator* value)
125{
126 WebCoreThreadViolationCheckRoundOne();
127 if (!value)
128 return nil;
129 if (DOMNodeIterator *wrapper = getDOMWrapper(value))
130 return retainPtr(wrapper).autorelease();
131 auto wrapper = adoptNS([[DOMNodeIterator alloc] _init]);
132 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
133 value->ref();
134 addDOMWrapper(wrapper.get(), value);
135 return wrapper.autorelease();
136}
