108DOMStyleSheet *kit(WebCore::StyleSheet* value)
109{
110 WebCoreThreadViolationCheckRoundOne();
111 if (!value)
112 return nil;
113 if (DOMStyleSheet *wrapper = getDOMWrapper(value))
114 return retainPtr(wrapper).autorelease();
115 RetainPtr<DOMStyleSheet> wrapper = adoptNS([[kitClass(value) alloc] _init]);
116 if (!wrapper)
117 return nil;
118 wrapper->_internal = reinterpret_cast<DOMObjectInternal*>(value);
119 value->ref();
120 addDOMWrapper(wrapper.get(), value);
121 return wrapper.autorelease();
122}
