79- (void)move:(UInt32)amount inDirection:(WebTextAdjustmentDirection)direction
80{
81 auto& range = *core(self);
82
83 WebCore::FrameSelection frameSelection;
84 frameSelection.setSelection(makeSimpleRange(range));
85
86 WebCore::TextGranularity granularity = WebCore::TextGranularity::CharacterGranularity;
87 // Until WebKit supports vertical layout, "down" is equivalent to "forward by a line" and
88 // "up" is equivalent to "backward by a line".
89 if (direction == WebTextAdjustmentDown) {
90 direction = WebTextAdjustmentForward;
91 granularity = WebCore::TextGranularity::LineGranularity;
92 } else if (direction == WebTextAdjustmentUp) {
93 direction = WebTextAdjustmentBackward;
94 granularity = WebCore::TextGranularity::LineGranularity;
95 }
96
97 for (UInt32 i = 0; i < amount; i++)
98 frameSelection.modify(WebCore::FrameSelection::AlterationMove, (WebCore::SelectionDirection)direction, granularity);
99
100 Position start = frameSelection.selection().start().parentAnchoredEquivalent();
101 Position end = frameSelection.selection().end().parentAnchoredEquivalent();
102 if (start.containerNode())
103 range.setStart(*start.containerNode(), start.offsetInContainerNode());
104 if (end.containerNode())
105 range.setEnd(*end.containerNode(), end.offsetInContainerNode());
106}
107
108- (void)extend:(UInt32)amount inDirection:(WebTextAdjustmentDirection)direction
109{
110 auto& range = *core(self);
111
112 WebCore::FrameSelection frameSelection;
113 frameSelection.setSelection(makeSimpleRange(range));
114
115 for (UInt32 i = 0; i < amount; i++)
116 frameSelection.modify(WebCore::FrameSelection::AlterationExtend, (WebCore::SelectionDirection)direction, WebCore::TextGranularity::CharacterGranularity);
117
118 Position start = frameSelection.selection().start().parentAnchoredEquivalent();
119 Position end = frameSelection.selection().end().parentAnchoredEquivalent();
120 if (start.containerNode())
121 range.setStart(*start.containerNode(), start.offsetInContainerNode());
122 if (end.containerNode())
123 range.setEnd(*end.containerNode(), end.offsetInContainerNode());
124}

126// FIXME: Refactor to share code with intersectingNodesWithDeprecatedZeroOffsetStartQuirk.
127static WebCore::Node* firstNodeAfter(const WebCore::BoundaryPoint& point)
128{
129 if (point.container->isCharacterDataNode())
130 return point.container.ptr();
131 if (auto child = point.container->traverseToChildAt(point.offset))
132 return child;
133 if (!point.offset)
134 return point.container.ptr();
135 return WebCore::NodeTraversal::nextSkippingChildren(point.container);
136}

138- (DOMNode *)firstNode
139{
140 return kit(firstNodeAfter(makeSimpleRange(*core(self)).start));
141}

150- (NSArray *)boundingBoxes
151{
152 NSArray *rects = nil;
153 NSRect bounds = [self boundingBox];
154 if (!NSIsEmptyRect(bounds)) {
155 if ([self isKindOfClass:[DOMElement class]]) {
156 DOMDocument *document = [self ownerDocument];
157 DOMCSSStyleDeclaration *style = [document getComputedStyle:(DOMElement *)self pseudoElement:@""];
158 if ([[style getPropertyValue:@"display"] isEqualToString:@"inline"])
159 rects = [self lineBoxRects];
160 } else if ([self isKindOfClass:[DOMText class]])
161 rects = [self lineBoxRects];
162 }
163
164 if (![rects count])
165 rects = @[[NSValue valueWithRect:bounds]];
166
167 return rects;
168}
169
170- (NSArray *)absoluteQuads
171{
172 NSArray *quads = nil;
173 NSRect bounds = [self boundingBox];
174 if (!NSIsEmptyRect(bounds)) {
175 if ([self isKindOfClass:[DOMElement class]]) {
176 DOMDocument *document = [self ownerDocument];
177 DOMCSSStyleDeclaration *style = [document getComputedStyle:(DOMElement *)self pseudoElement:@""];
178 if ([[style getPropertyValue:@"display"] isEqualToString:@"inline"])
179 quads = [self lineBoxQuads];
180 } else if ([self isKindOfClass:[DOMText class]])
181 quads = [self lineBoxQuads];
182 }
183
184 if (![quads count])
185 quads = @[adoptNS([[WKQuadObject alloc] initWithQuad:[self absoluteQuad]]).get()];
186
187 return quads;
188}

190- (NSArray *)borderRadii
191{
192 RenderObject* renderer = core(self)->renderer();
193
194 if (is<RenderBox>(renderer)) {
195 WebCore::RoundedRect::Radii radii = downcast<RenderBox>(*renderer).borderRadii();
196 return @[[NSValue valueWithSize:(WebCore::FloatSize)radii.topLeft()],
197 [NSValue valueWithSize:(WebCore::FloatSize)radii.topRight()],
198 [NSValue valueWithSize:(WebCore::FloatSize)radii.bottomLeft()],
199 [NSValue valueWithSize:(WebCore::FloatSize)radii.bottomRight()]];
200 }
201 NSValue *emptyValue = [NSValue valueWithSize:CGSizeZero];
202 return @[emptyValue, emptyValue, emptyValue, emptyValue];
203}

205- (BOOL)containsOnlyInlineObjects
206{
207 RenderObject* renderer = core(self)->renderer();
208 return renderer
209 && renderer->childrenInline()
210 && (is<RenderBlock>(*renderer) && !downcast<RenderBlock>(*renderer).inlineContinuation())
211 && !renderer->isTable();
212}

218}

226- (CGFloat)textHeight
227{
228 RenderObject* renderer = core(self)->renderer();
229 if (is<RenderText>(renderer))
230 return downcast<RenderText>(*renderer).style().computedLineHeight();
231
232 return CGFLOAT_MAX;
233}
234
235- (DOMNode *)findExplodedTextNodeAtPoint:(CGPoint)point
236{
237 auto* renderer = core(self)->renderer();
238 if (!is<WebCore::RenderBlockFlow>(renderer))
239 return nil;
240
241 auto* renderText = downcast<WebCore::RenderBlockFlow>(*renderer).findClosestTextAtAbsolutePoint(point);
242 if (renderText && renderText->textNode())
243 return kit(renderText->textNode());
244
245 return nil;
246}

256@implementation DOMHTMLElement (DOMUIKitComplexityExtensions)
257
258- (int)structuralComplexityContribution
259{
260 int result = 0;
261 RenderObject * renderer = core(self)->renderer();
262 if (renderer) {
263 if (renderer->isFloatingOrOutOfFlowPositioned() ||
264 renderer->isWidget()) {
265 result = INT_MAX;
266 } else if (!renderer->firstChildSlow()) {
267 result = 0;
268 } else if (is<WebCore::RenderBlockFlow>(*renderer) || (is<RenderBlock>(*renderer) && downcast<RenderBlock>(*renderer).inlineContinuation())) {
269 BOOL noCost = NO;
270 if (is<RenderBox>(*renderer)) {
271 RenderBox& asBox = renderer->enclosingBox();
272 RenderObject* parent = asBox.parent();
273 RenderBox* parentRenderBox = is<RenderBox>(parent) ? downcast<RenderBox>(parent) : nullptr;
274 if (parentRenderBox && asBox.width() == parentRenderBox->width()) {
275 noCost = YES;
276 }
277 }
278 result = (noCost ? 0 : 1);
279 } else if (renderer->hasTransform()) {
280 result = INT_MAX;
281 }
282 }
283 return result;
284}

328@implementation DOMHTMLAreaElement (DOMUIKitExtensions)
329
330- (CGRect)boundingBoxWithOwner:(DOMNode *)owner
331{
332 // ignores transforms
333 return owner ? snappedIntRect(core(self)->computeRect(core(owner)->renderer())) : CGRectZero;
334}
335
336- (WKQuad)absoluteQuadWithOwner:(DOMNode *)owner
337{
338 if (owner) {
339 WebCore::IntRect rect = snappedIntRect(core(self)->computeRect(core(owner)->renderer()));
340 WKQuad quad;
341 quad.p1 = CGPointMake(rect.x(), rect.y());
342 quad.p2 = CGPointMake(rect.maxX(), rect.y());
343 quad.p3 = CGPointMake(rect.maxX(), rect.maxY());
344 quad.p4 = CGPointMake(rect.x(), rect.maxY());
345 return quad;
346 }
347
348 WKQuad zeroQuad = { CGPointZero, CGPointZero, CGPointZero, CGPointZero };
349 return zeroQuad;
350}
351
352- (NSArray *)boundingBoxesWithOwner:(DOMNode *)owner
353{
354 return @[[NSValue valueWithRect:[self boundingBoxWithOwner:owner]]];
355}

357- (NSArray *)absoluteQuadsWithOwner:(DOMNode *)owner
358{

378@implementation DOMHTMLImageElement (WebDOMHTMLImageElementOperationsPrivate)
379
380- (NSData *)dataRepresentation:(BOOL)rawImageData
381{
382 auto* cachedImage = core(self)->cachedImage();
383 if (!cachedImage)
384 return nil;
385 auto* image = cachedImage->image();
386 if (!image)
387 return nil;
388 auto* data = rawImageData ? cachedImage->resourceBuffer() : image->data();
389 if (!data)
390 return nil;
391 return data->createNSData().autorelease();
392}
393
394- (NSString *)mimeType
395{
396 auto* cachedImage = core(self)->cachedImage();
397 if (!cachedImage || !cachedImage->image())
398 return nil;
399 return cachedImage->response().mimeType();
400}
