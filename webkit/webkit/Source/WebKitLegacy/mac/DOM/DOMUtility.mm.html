76static WebScriptObject *createDOMWrapper(JSC::JSObject& jsWrapper)
77{
78 JSC::VM& vm = jsWrapper.vm();
79 #define WRAP(className) \
80 if (auto* wrapped = WebCore::JS##className::toWrapped(vm, &jsWrapper)) \
81 return kit(wrapped);
82
83 WRAP(CSSRule)
84 WRAP(CSSRuleList)
85 WRAP(CSSStyleDeclaration)
86 WRAP(DeprecatedCSSOMValue)
87 WRAP(DeprecatedCSSOMCounter)
88 WRAP(DOMImplementation)
89 WRAP(Event)
90 WRAP(HTMLOptionsCollection)
91 WRAP(MediaList)
92 WRAP(NamedNodeMap)
93 WRAP(Node)
94 WRAP(NodeIterator)
95 WRAP(NodeList)
96 WRAP(DeprecatedCSSOMRGBColor)
97 WRAP(Range)
98 WRAP(DeprecatedCSSOMRect)
99 WRAP(StyleSheet)
100 WRAP(StyleSheetList)
101 WRAP(TreeWalker)
102 WRAP(WindowProxy)
103 WRAP(XPathExpression)
104 WRAP(XPathResult)
105
106 // This must be after HTMLOptionsCollection, because JSHTMLCollection is a base class of
107 // JSHTMLOptionsCollection, and HTMLCollection is *not* a base class of HTMLOptionsCollection.
108 WRAP(HTMLCollection)
109
110 #undef WRAP
111
112 return nil;
113}
114
115static void disconnectWindowWrapper(WebScriptObject *windowWrapper)
116{
117 ASSERT([windowWrapper isKindOfClass:[DOMAbstractView class]]);
118 [(DOMAbstractView *)windowWrapper _disconnectFrame];
119}
120
121void initializeDOMWrapperHooks()
122{
123 WebCore::initializeDOMWrapperHooks(createDOMWrapper, disconnectWindowWrapper);
124}
