34NSString * const DOMException = @"DOMException";
35NSString * const DOMRangeException = @"DOMRangeException";
36NSString * const DOMEventException = @"DOMEventException";
37NSString * const DOMXPathException = @"DOMXPathException";
38
39static NO_RETURN void raiseDOMErrorException(WebCore::ExceptionCode ec)
40{
41 ASSERT(ec);
42
43 auto description = WebCore::DOMException::description(ec);
44
45 RetainPtr<NSString> reason;
46 if (description.name)
47 reason = adoptNS([[NSString alloc] initWithFormat:@"*** %s: %@ %d", description.name.characters(), DOMException, description.legacyCode]);
48 else
49 reason = adoptNS([[NSString alloc] initWithFormat:@"*** %@ %d", DOMException, description.legacyCode]);
50
51 auto userInfo = @{ DOMException: @(description.legacyCode) };
52 auto exception = [NSException exceptionWithName:DOMException reason:reason.get() userInfo:userInfo];
53 [exception raise];
54
55 RELEASE_ASSERT_NOT_REACHED();
56}
57
58void raiseTypeErrorException()
59{
60 raiseDOMErrorException(WebCore::TypeError);
61}
62
63void raiseNotSupportedErrorException()
64{
65 raiseDOMErrorException(WebCore::NotSupportedError);
66}
67
68void raiseDOMErrorException(WebCore::Exception&& exception)
69{
70 raiseDOMErrorException(exception.code());
71}
