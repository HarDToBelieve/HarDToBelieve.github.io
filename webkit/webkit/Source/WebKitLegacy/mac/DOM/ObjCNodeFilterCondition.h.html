36class Node;

38class ObjCNodeFilterCondition final : public NodeFilterCondition {
39public:
40 static Ref<ObjCNodeFilterCondition> create(id <DOMNodeFilter> filter)
41 {
42 return adoptRef(*new ObjCNodeFilterCondition(filter));
43 }
44
45 unsigned short acceptNode(Node&) const override;
46
47private:
48 ObjCNodeFilterCondition(id <DOMNodeFilter> filter)
49 : m_filter(filter)
50 {
51 }
52
53 RetainPtr<id <DOMNodeFilter> > m_filter;
54};
