37/*
38 @discussion Notifications sent when history is modified.
39 @constant WebHistoryItemsAddedNotification Posted from addItems:. This
40 notification comes with a userInfo dictionary that contains the array of
41 items added. The key for the array is WebHistoryItemsKey.
42 @constant WebHistoryItemsRemovedNotification Posted from removeItems:.
43 This notification comes with a userInfo dictionary that contains the array of
44 items removed. The key for the array is WebHistoryItemsKey.
45 @constant WebHistoryAllItemsRemovedNotification Posted from removeAllItems
46 @constant WebHistoryLoadedNotification Posted from loadFromURL:error:.
47*/
48extern NSString *WebHistoryItemsAddedNotification WEBKIT_DEPRECATED_MAC(10_3, 10_14);
49extern NSString *WebHistoryItemsRemovedNotification WEBKIT_DEPRECATED_MAC(10_3, 10_14);
50extern NSString *WebHistoryAllItemsRemovedNotification WEBKIT_DEPRECATED_MAC(10_3, 10_14);
51extern NSString *WebHistoryLoadedNotification WEBKIT_DEPRECATED_MAC(10_3, 10_14);
52extern NSString *WebHistorySavedNotification WEBKIT_DEPRECATED_MAC(10_3, 10_14);
53
54extern NSString *WebHistoryItemsKey WEBKIT_DEPRECATED_MAC(10_3, 10_14);

56/*!
57 @class WebHistory
58 @discussion WebHistory is used to track pages that have been loaded
59 by WebKit.
60*/
61WEBKIT_CLASS_DEPRECATED_MAC(10_3, 10_14)
62@interface WebHistory : NSObject {
63@package
64 WebHistoryPrivate *_historyPrivate;
65}
66
67/*!
68 @method optionalSharedHistory
69 @abstract Returns a shared WebHistory instance initialized with the default history file.
70 @result A WebHistory object.
71*/
72+ (WebHistory *)optionalSharedHistory;
73
74/*!
75 @method setOptionalSharedHistory:
76 @param history The history to use for the global WebHistory.
77*/
78+ (void)setOptionalSharedHistory:(WebHistory *)history;
79
80/*!
81 @method loadFromURL:error:
82 @param URL The URL to use to initialize the WebHistory.
83 @param error Set to nil or an NSError instance if an error occurred.
84 @abstract The designated initializer for WebHistory.
85 @result Returns YES if successful, NO otherwise.
86*/
87- (BOOL)loadFromURL:(NSURL *)URL error:(NSError **)error;
88
89/*!
90 @method saveToURL:error:
91 @discussion Save history to URL. It is the client's responsibility to call this at appropriate times.
92 @param URL The URL to use to save the WebHistory.
93 @param error Set to nil or an NSError instance if an error occurred.
94 @result Returns YES if successful, NO otherwise.
95*/
96- (BOOL)saveToURL:(NSURL *)URL error:(NSError **)error;
97
98/*!
99 @method addItems:
100 @param newItems An array of WebHistoryItems to add to the WebHistory.
101*/
102- (void)addItems:(NSArray *)newItems;
103
104/*!
105 @method removeItems:
106 @param items An array of WebHistoryItems to remove from the WebHistory.
107*/
108- (void)removeItems:(NSArray *)items;
109
110/*!
111 @method removeAllItems
112*/
113- (void)removeAllItems;
114
115/*!
116 @property orderedLastVisitedDays
117 @abstract An array of NSCalendarDates for which history items exist in the WebHistory.
118 @discussion An array of NSCalendarDates, each one representing a unique day that contains one
119 or more history items, ordered from most recent to oldest.
120*/
121@property (nonatomic, readonly, copy) NSArray *orderedLastVisitedDays;
122
123#if !TARGET_OS_IPHONE
124/*!
125 @method orderedItemsLastVisitedOnDay:
126 @discussion Get an array of WebHistoryItem that were last visited on the day represented by the
127 specified NSCalendarDate, ordered from most recent to oldest.
128 @param calendarDate A date identifying the unique day of interest.
129 @result Returns an array of WebHistoryItems last visited on the indicated day.
130*/
131#pragma clang diagnostic push
132#pragma clang diagnostic ignored "-Wdeprecated-declarations"
133- (NSArray *)orderedItemsLastVisitedOnDay:(NSCalendarDate *)calendarDate;
134#pragma clang diagnostic pop

138 @method itemForURL:

143- (WebHistoryItem *)itemForURL:(NSURL *)URL;

149@property (nonatomic) int historyItemLimit;

155@property (nonatomic) int historyAgeInDaysLimit;

157@end
