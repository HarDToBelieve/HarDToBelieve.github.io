161{

163 RetainPtr<WebHistoryItem> copy = adoptNS([[[self class] alloc] initWithWebCoreHistoryItem:core(_private)->copy()]);

166
167 historyItemWrappers().set(core(copy->_private), copy.get());
168
169 return copy.leakRef();

227- (NSString *)description
228{
229 HistoryItem* coreItem = core(_private);
230 NSMutableString *result = [NSMutableString stringWithFormat:@"%@ %@", [super description], (NSString*)coreItem->urlString()];
231 if (!coreItem->target().isEmpty()) {
232 NSString *target = coreItem->target();
233 [result appendFormat:@" in \"%@\"", target];
234 }
235 if (coreItem->isTargetItem()) {
236 [result appendString:@" *target*"];
237 }
238 if (coreItem->formData()) {
239 [result appendString:@" *POST*"];
240 }
241
242 if (coreItem->children().size()) {
243 const auto& children = coreItem->children();
244 int currPos = [result length];
245 unsigned size = children.size();
246 for (unsigned i = 0; i < size; ++i) {
247 WebHistoryItem *child = kit(const_cast<HistoryItem*>(children[i].ptr()));
248 [result appendString:@"\n"];
249 [result appendString:[child description]];
250 }
251 // shift all the contents over. A bit slow, but hey, this is for debugging.
252 NSRange replRange = { static_cast<NSUInteger>(currPos), [result length] - currPos };
253 [result replaceOccurrencesOfString:@"\n" withString:@"\n " options:0 range:replRange];
254 }
255
256 return result;
257}

272 return retainPtr(wrapper).autorelease();
273 return adoptNS([[WebHistoryItem alloc] initWithWebCoreHistoryItem:*item]).autorelease();

290- (id)initWithWebCoreHistoryItem:(Ref<HistoryItem>&&)item
291{
292 WebCoreThreadViolationCheckRoundOne();
293
294 // Need to tell WebCore what function to call for the
295 // "History Item has Changed" notification - no harm in doing this
296 // everytime a WebHistoryItem is created
297 // Note: We also do this in [WebFrameView initWithFrame:] where we do
298 // other "init before WebKit is used" type things
299 // FIXME: This means that if we mix legacy WebKit and modern WebKit in the same process, we won't get both notifications.
300 WebCore::notifyHistoryItemChanged = WKNotifyHistoryItemChanged;
301
302 if (!(self = [super init]))
303 return nil;
304
305 _private = [[WebHistoryItemPrivate alloc] init];
306 _private->_historyItem = WTFMove(item);
307
308 ASSERT(!historyItemWrappers().get(core(_private)));
309 historyItemWrappers().set(core(_private), self);
310 return self;
311}

323- (id)initFromDictionaryRepresentation:(NSDictionary *)dict
324{
325 NSString *URLString = [dict _webkit_stringForKey:@""];
326 NSString *title = [dict _webkit_stringForKey:titleKey];
327
328 // Do an existence check to avoid calling doubleValue on a nil string. Leave
329 // time interval at 0 if there's no value in dict.
330 NSString *timeIntervalString = [dict _webkit_stringForKey:lastVisitedTimeIntervalKey];
331 NSTimeInterval lastVisited = timeIntervalString == nil ? 0 : [timeIntervalString doubleValue];
332
333 self = [self initWithURLString:URLString title:title displayTitle:[dict _webkit_stringForKey:displayTitleKey] lastVisitedTimeInterval:lastVisited];
334
335 // Check if we've read a broken URL from the file that has non-Latin1 chars. If so, try to convert
336 // as if it was from user typing.
337 if (![URLString canBeConvertedToEncoding:NSISOLatin1StringEncoding]) {
338 NSURL *tempURL = [NSURL _webkit_URLWithUserTypedString:URLString];
339 ASSERT(tempURL);
340 NSString *newURLString = [tempURL _web_originalDataAsString];
341 core(_private)->setURLString(newURLString);
342 core(_private)->setOriginalURLString(newURLString);
343 }
344
345 if ([dict _webkit_boolForKey:lastVisitWasFailureKey])
346 core(_private)->setLastVisitWasFailure(true);
347
348 if (NSArray *redirectURLs = [dict _webkit_arrayForKey:redirectURLsKey])
349 _private->_redirectURLs = makeUnique<Vector<String>>(makeVector<String>(redirectURLs));
350
351 for (id childDict in [[dict objectForKey:childrenKey] reverseObjectEnumerator]) {
352 auto child = adoptNS([[WebHistoryItem alloc] initFromDictionaryRepresentation:childDict]);
353 core(_private)->addChildItem(*core(child->_private));
354 }
355
356#if PLATFORM(IOS_FAMILY)
357 NSNumber *scaleValue = [dict objectForKey:scaleKey];
358 NSNumber *scaleIsInitialValue = [dict objectForKey:scaleIsInitialKey];
359 if (scaleValue && scaleIsInitialValue)
360 core(_private)->setScale([scaleValue floatValue], [scaleIsInitialValue boolValue]);
361
362 if (id viewportArguments = [dict objectForKey:@"WebViewportArguments"])
363 [self _setViewportArguments:viewportArguments];
364
365 NSNumber *scrollPointXValue = [dict objectForKey:scrollPointXKey];
366 NSNumber *scrollPointYValue = [dict objectForKey:scrollPointYKey];
367 if (scrollPointXValue && scrollPointYValue)
368 core(_private)->setScrollPosition(IntPoint([scrollPointXValue intValue], [scrollPointYValue intValue]));
369#endif
370
371 return self;
372}

401- (NSDictionary *)dictionaryRepresentationIncludingChildren:(BOOL)includesChildren
402#else
403- (NSDictionary *)dictionaryRepresentation
404#endif
405{
406 NSMutableDictionary *dict = [NSMutableDictionary dictionaryWithCapacity:8];
407
408 HistoryItem* coreItem = core(_private);
409
410 if (!coreItem->urlString().isEmpty())
411 [dict setObject:(NSString*)coreItem->urlString() forKey:@""];
412 if (!coreItem->title().isEmpty())
413 [dict setObject:(NSString*)coreItem->title() forKey:titleKey];
414 if (!coreItem->alternateTitle().isEmpty())
415 [dict setObject:(NSString*)coreItem->alternateTitle() forKey:displayTitleKey];
416 if (_private->_lastVisitedTime) {
417 // Store as a string to maintain backward compatibility. (See 3245793)
418 [dict setObject:[NSString stringWithFormat:@"%.1lf", _private->_lastVisitedTime]
419 forKey:lastVisitedTimeIntervalKey];
420 }
421 if (coreItem->lastVisitWasFailure())
422 [dict setObject:@YES forKey:lastVisitWasFailureKey];
423 if (auto redirectURLs = _private->_redirectURLs.get())
424 [dict setObject:createNSArray(*redirectURLs).get() forKey:redirectURLsKey];
425
426#if PLATFORM(IOS_FAMILY)
427 if (includesChildren && coreItem->children().size()) {
428#else
429 if (coreItem->children().size()) {
430#endif
431 const auto& children = coreItem->children();
432 NSMutableArray *childDicts = [NSMutableArray arrayWithCapacity:children.size()];
433
434 for (int i = children.size() - 1; i >= 0; i--)
435 [childDicts addObject:[kit(const_cast<HistoryItem*>(children[i].ptr())) dictionaryRepresentation]];
436 [dict setObject: childDicts forKey:childrenKey];
437 }
438
439#if PLATFORM(IOS_FAMILY)
440 [dict setObject:[NSNumber numberWithFloat:core(_private)->scale()] forKey:scaleKey];
441 [dict setObject:[NSNumber numberWithBool:core(_private)->scaleIsInitial()] forKey:scaleIsInitialKey];
442
443 NSDictionary *viewportArguments = [self _viewportArguments];
444 if (viewportArguments)
445 [dict setObject:viewportArguments forKey:@"WebViewportArguments"];
446
447 IntPoint scrollPosition = core(_private)->scrollPosition();
448 [dict setObject:@(scrollPosition.x()) forKey:scrollPointXKey];
449 [dict setObject:@(scrollPosition.y()) forKey:scrollPointYKey];
450#endif
451
452 return dict;
453}

531{

534 [argumentsDictionary setObject:[NSNumber numberWithFloat:viewportArguments.zoom] forKey:WebViewportInitialScaleKey];
535 [argumentsDictionary setObject:[NSNumber numberWithFloat:viewportArguments.minZoom] forKey:WebViewportMinimumScaleKey];
536 [argumentsDictionary setObject:[NSNumber numberWithFloat:viewportArguments.maxZoom] forKey:WebViewportMaximumScaleKey];
537 [argumentsDictionary setObject:[NSNumber numberWithFloat:viewportArguments.width] forKey:WebViewportWidthKey];
538 [argumentsDictionary setObject:[NSNumber numberWithFloat:viewportArguments.height] forKey:WebViewportHeightKey];
539 [argumentsDictionary setObject:[NSNumber numberWithFloat:viewportArguments.userZoom] forKey:WebViewportUserScalableKey];
540 [argumentsDictionary setObject:[NSNumber numberWithFloat:viewportArguments.shrinkToFit] forKey:WebViewportShrinkToFitKey];
541 return argumentsDictionary;
542}
