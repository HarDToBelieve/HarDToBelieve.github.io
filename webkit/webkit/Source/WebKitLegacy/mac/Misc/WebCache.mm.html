76 return @[

120+ (void)empty
121{
122 // Toggling the cache model like this forces the cache to evict all its in-memory resources.
123 WebCacheModel cacheModel = [WebView _cacheModel];
124 [WebView _setCacheModel:WebCacheModelDocumentViewer];
125 [WebView _setCacheModel:cacheModel];
126
127 // Empty the application cache.
128 webApplicationCacheStorage().empty();
129
130 // Empty the Cross-Origin Preflight cache
131 WebCore::CrossOriginPreflightResultCache::singleton().clear();
132}

136+ (void)emptyInMemoryResources
137{
138 // This method gets called from MobileSafari after it calls [WebView
139 // _close]. [WebView _close] schedules its work on the WebThread. So we
140 // schedule this method on the WebThread as well so as to pick up all the
141 // dead resources left behind after closing the WebViews
142 WebThreadRun(^{
143 // Toggling the cache model like this forces the cache to evict all its in-memory resources.
144 WebCacheModel cacheModel = [WebView _cacheModel];
145 [WebView _setCacheModel:WebCacheModelDocumentViewer];
146 [WebView _setCacheModel:cacheModel];
147
148 WebCore::MemoryCache::singleton().pruneLiveResources(true);
149 });
150}

152+ (void)sizeOfDeadResources:(int *)resources
153{
154 WebCore::MemoryCache::Statistics stats = WebCore::MemoryCache::singleton().getStatistics();
155 if (resources) {
156 *resources = (stats.images.size - stats.images.liveSize)
157 + (stats.cssStyleSheets.size - stats.cssStyleSheets.liveSize)
158#if ENABLE(XSLT)
159 + (stats.xslStyleSheets.size - stats.xslStyleSheets.liveSize)
160#endif
161 + (stats.scripts.size - stats.scripts.liveSize);
162 }
163}

165+ (CGImageRef)imageForURL:(NSURL *)url
166{
167 if (!url)
168 return nullptr;
169
170 WebCore::ResourceRequest request(url);
171 WebCore::CachedResource* cachedResource = WebCore::MemoryCache::singleton().resourceForRequest(request, PAL::SessionID::defaultSessionID());
172 if (!is<WebCore::CachedImage>(cachedResource))
173 return nullptr;
174
175 auto& cachedImage = downcast<WebCore::CachedImage>(*cachedResource);
176 if (!cachedImage.hasImage())
177 return nullptr;
178
179 auto nativeImage = cachedImage.image()->nativeImage();
180 if (!nativeImage)
181 return nullptr;
182
183 return nativeImage->platformImage().get();
184}
