52static RetainPtr<CFMutableDictionaryRef>& lookupTable()
53{
54 static NeverDestroyed<RetainPtr<CFMutableDictionaryRef>> lookupTable;
55 return lookupTable;
56}
57
58static void addLookupKey(NSString *key, SEL selector)
59{
60 CFDictionaryAddValue(lookupTable().get(), (__bridge CFStringRef)key, selector);
61}
62
63static void cacheValueForKey(const void *key, const void *value, void *self)
64{
65 // calling objectForKey will cache the value in our _cache dictionary
66 [(__bridge WebElementDictionary *)self objectForKey:(__bridge NSString *)key];
67}

80+ (void)initializeLookupTable
81{
82 if (lookupTable())
83 return;
84
85 lookupTable() = adoptCF(CFDictionaryCreateMutable(kCFAllocatorDefault, 0, &kCFCopyStringDictionaryKeyCallBacks, NULL));
86
87 addLookupKey(WebElementDOMNodeKey, @selector(_domNode));
88 addLookupKey(WebElementFrameKey, @selector(_webFrame));
89 addLookupKey(WebElementImageAltStringKey, @selector(_altDisplayString));
90#if !PLATFORM(IOS_FAMILY)
91 addLookupKey(WebElementImageKey, @selector(_image));
92 addLookupKey(WebElementImageRectKey, @selector(_imageRect));
93#endif
94 addLookupKey(WebElementImageURLKey, @selector(_absoluteImageURL));
95 addLookupKey(WebElementIsSelectedKey, @selector(_isSelected));
96 addLookupKey(WebElementMediaURLKey, @selector(_absoluteMediaURL));
97 addLookupKey(WebElementSpellingToolTipKey, @selector(_spellingToolTip));
98 addLookupKey(WebElementTitleKey, @selector(_title));
99 addLookupKey(WebElementLinkURLKey, @selector(_absoluteLinkURL));
100 addLookupKey(WebElementLinkTargetFrameKey, @selector(_targetWebFrame));
101 addLookupKey(WebElementLinkTitleKey, @selector(_titleDisplayString));
102 addLookupKey(WebElementLinkLabelKey, @selector(_textContent));
103 addLookupKey(WebElementLinkIsLiveKey, @selector(_isLiveLink));
104 addLookupKey(WebElementIsContentEditableKey, @selector(_isContentEditable));
105 addLookupKey(WebElementIsInScrollBarKey, @selector(_isInScrollBar));
106}

108- (id)initWithHitTestResult:(const HitTestResult&)result
109{
110 [[self class] initializeLookupTable];
111 self = [super init];
112 if (!self)
113 return nil;
114 _result = new HitTestResult(result);
115 return self;
116}

118- (void)dealloc
119{
120 if (WebCoreObjCScheduleDeallocateOnMainThread([WebElementDictionary class], self))
121 return;
122
123 delete _result;
124 [_cache release];
125 [_nilValues release];
126 [super dealloc];
127}

129- (void)_fillCache
130{
131 CFDictionaryApplyFunction(lookupTable().get(), cacheValueForKey, (__bridge void*)self);
132 _cacheComplete = YES;
133}
134
135- (NSUInteger)count
136{
137 if (!_cacheComplete)
138 [self _fillCache];
139 return [_cache count];
140}
141
142- (NSEnumerator *)keyEnumerator
143{
144 if (!_cacheComplete)
145 [self _fillCache];
146 return [_cache keyEnumerator];
147}
148
149- (id)objectForKey:(id)key
150{
151 id value = [_cache objectForKey:key];
152 if (value || _cacheComplete || [_nilValues containsObject:key])
153 return value;
154
155 SEL selector = static_cast<SEL>(const_cast<void*>(CFDictionaryGetValue(lookupTable().get(), (__bridge CFTypeRef)key)));
156 if (!selector)
157 return nil;
158 value = [self performSelector:selector];
159
160 NSUInteger lookupTableCount = CFDictionaryGetCount(lookupTable().get());
161 if (value) {
162 if (!_cache)
163 _cache = [[NSMutableDictionary alloc] initWithCapacity:lookupTableCount];
164 [_cache setObject:value forKey:key];
165 } else {
166 if (!_nilValues)
167 _nilValues = [[NSMutableSet alloc] initWithCapacity:lookupTableCount];
168 [_nilValues addObject:key];
169 }
170
171 _cacheComplete = ([_cache count] + [_nilValues count]) == lookupTableCount;
172
173 return value;
174}

186// String's NSString* operator converts null Strings to empty NSStrings for compatibility
187// with AppKit. We need to work around that here.
188static NSString* NSStringOrNil(String coreString)
189{
190 if (coreString.isNull())
191 return nil;
192 return coreString;
193}
