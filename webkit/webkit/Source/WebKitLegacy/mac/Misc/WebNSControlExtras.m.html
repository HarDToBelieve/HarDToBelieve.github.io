31#import "WebNSControlExtras.h"
32
33@implementation NSControl (WebExtras)
34
35- (void)sizeToFitAndAdjustWindowHeight
36{
37 NSRect frame = [self frame];
38
39 NSSize bestSize = [[self cell] cellSizeForBounds:NSMakeRect(0.0f, 0.0f, frame.size.width, 10000.0f)];
40
41 float heightDelta = bestSize.height - frame.size.height;
42
43 frame.size.height += heightDelta;
44 frame.origin.y -= heightDelta;
45 [self setFrame:frame];
46
47 NSWindow *window = [self window];
48 NSRect windowFrame = [window frame];
49
50 windowFrame.size.height += heightDelta;
51 [window setFrame:windowFrame display:NO];
52}
53
54@end
