38- (NSString *)_web_capitalizeRFC822HeaderFieldName
39{
40 CFStringRef name = (__bridge CFStringRef)self;
41 NSString *result = nil;
42
43 CFIndex len = CFStringGetLength(name);
44 char* charPtr = nullptr;
45 UniChar* uniCharPtr = nullptr;
46 Boolean useUniCharPtr = FALSE;
47 Boolean shouldCapitalize = TRUE;
48 Boolean somethingChanged = FALSE;
49
50 for (CFIndex i = 0; i < len; i ++) {
51 UniChar ch = CFStringGetCharacterAtIndex(name, i);
52 Boolean replace = FALSE;
53 if (shouldCapitalize && ch >= 'a' && ch <= 'z') {
54 ch = ch + 'A' - 'a';
55 replace = TRUE;
56 } else if (!shouldCapitalize && ch >= 'A' && ch <= 'Z') {
57 ch = ch + 'a' - 'A';
58 replace = TRUE;
59 }
60 if (replace) {
61 if (!somethingChanged) {
62 somethingChanged = TRUE;
63 if (CFStringGetBytes(name, CFRangeMake(0, len), kCFStringEncodingISOLatin1, 0, FALSE, NULL, 0, NULL) == len) {
64 // Can be encoded in ISOLatin1
65 useUniCharPtr = FALSE;
66 charPtr = static_cast<char*>(CFAllocatorAllocate(kCFAllocatorDefault, len + 1, 0));
67 CFStringGetCString(name, charPtr, len+1, kCFStringEncodingISOLatin1);
68 } else {
69 useUniCharPtr = TRUE;
70 uniCharPtr = static_cast<UniChar*>(CFAllocatorAllocate(kCFAllocatorDefault, len * sizeof(UniChar), 0));
71 CFStringGetCharacters(name, CFRangeMake(0, len), uniCharPtr);
72 }
73 }
74 if (useUniCharPtr)
75 uniCharPtr[i] = ch;
76 else
77 charPtr[i] = ch;
78 }
79 if (ch == '-')
80 shouldCapitalize = TRUE;
81 else
82 shouldCapitalize = FALSE;
83 }
84 if (somethingChanged) {
85 if (useUniCharPtr)
86 result = adoptCF(CFStringCreateWithCharactersNoCopy(kCFAllocatorDefault, uniCharPtr, len, nullptr)).bridgingAutorelease();
87 else
88 result = adoptCF(CFStringCreateWithCStringNoCopy(kCFAllocatorDefault, charPtr, kCFStringEncodingISOLatin1, nullptr)).bridgingAutorelease();
89 } else
90 result = self;
91
92 return result;
93}

99- (NSString *)_webkit_guessedMIMETypeForXML
100{
101 NSUInteger length = [self length];
102 const UInt8* bytes = static_cast<const UInt8*>([self bytes]);
103
104#define CHANNEL_TAG_LENGTH 7
105
106 const char* p = reinterpret_cast<const char*>(bytes);
107 int remaining = std::min<NSUInteger>(length, WEB_GUESS_MIME_TYPE_PEEK_LENGTH) - (CHANNEL_TAG_LENGTH - 1);
108
109 BOOL foundRDF = false;
110
111 while (remaining > 0) {
112 // Look for a "<".
113 const char* hit = static_cast<const char*>(memchr(p, '<', remaining));
114 if (!hit)
115 break;
116
117 // We are trying to identify RSS or Atom. RSS has a top-level
118 // element of either <rss> or <rdf>. However, there are
119 // non-RSS RDF files, so in the case of <rdf> we further look
120 // for a <channel> element. In the case of an Atom file, a
121 // top-level <feed> element is all we need to see. Only tags
122 // starting with <? or <! can precede the root element. We
123 // bail if we don't find an <rss>, <feed> or <rdf> element
124 // right after those.
125
126 if (foundRDF) {
127 if (!strncasecmp(hit, "<channel", strlen("<channel")))
128 return @"application/rss+xml";
129 } else if (!strncasecmp(hit, "<rdf", strlen("<rdf")))
130 foundRDF = TRUE;
131 else if (!strncasecmp(hit, "<rss", strlen("<rss")))
132 return @"application/rss+xml";
133 else if (!strncasecmp(hit, "<feed", strlen("<feed")))
134 return @"application/atom+xml";
135 else if (strncasecmp(hit, "<?", strlen("<?")) && strncasecmp(hit, "<!", strlen("<!")))
136 return nil;
137
138 // Skip the "<" and continue.
139 remaining -= (hit + 1) - p;
140 p = hit + 1;
141 }
142
143 return nil;
144}

146- (NSString *)_webkit_guessedMIMEType
147{
148#define JPEG_MAGIC_NUMBER_LENGTH 4
149#define SCRIPT_TAG_LENGTH 7
150#define TEXT_HTML_LENGTH 9
151#define VCARD_HEADER_LENGTH 11
152#define VCAL_HEADER_LENGTH 15
153
154 NSString *MIMEType = [self _webkit_guessedMIMETypeForXML];
155 if ([MIMEType length])
156 return MIMEType;
157
158 NSUInteger length = [self length];
159 const char* bytes = static_cast<const char*>([self bytes]);
160
161 const char* p = bytes;
162 int remaining = std::min<NSUInteger>(length, WEB_GUESS_MIME_TYPE_PEEK_LENGTH) - (SCRIPT_TAG_LENGTH - 1);
163 while (remaining > 0) {
164 // Look for a "<".
165 const char* hit = static_cast<const char*>(memchr(p, '<', remaining));
166 if (!hit)
167 break;
168
169 // If we found a "<", look for "<html>" or "<a " or "<script".
170 if (!strncasecmp(hit, "<html>", strlen("<html>"))
171 || !strncasecmp(hit, "<a ", strlen("<a "))
172 || !strncasecmp(hit, "<script", strlen("<script"))
173 || !strncasecmp(hit, "<title>", strlen("<title>"))) {
174 return @"text/html";
175 }
176
177 // Skip the "<" and continue.
178 remaining -= (hit + 1) - p;
179 p = hit + 1;
180 }
181
182 // Test for a broken server which has sent the content type as part of the content.
183 // This code could be improved to look for other mime types.
184 p = bytes;
185 remaining = std::min<NSUInteger>(length, WEB_GUESS_MIME_TYPE_PEEK_LENGTH) - (TEXT_HTML_LENGTH - 1);
186 while (remaining > 0) {
187 // Look for a "t" or "T".
188 const char* hit = nullptr;
189 const char* lowerhit = static_cast<const char*>(memchr(p, 't', remaining));
190 const char* upperhit = static_cast<const char*>(memchr(p, 'T', remaining));
191 if (!lowerhit && !upperhit)
192 break;
193
194 if (!lowerhit)
195 hit = upperhit;
196 else if (!upperhit)
197 hit = lowerhit;
198 else
199 hit = std::min<const char*>(lowerhit, upperhit);
200
201 // If we found a "t/T", look for "text/html".
202 if (!strncasecmp(hit, "text/html", TEXT_HTML_LENGTH))
203 return @"text/html";
204
205 // Skip the "t/T" and continue.
206 remaining -= (hit + 1) - p;
207 p = hit + 1;
208 }
209
210 if ((length >= VCARD_HEADER_LENGTH) && !strncmp(bytes, "BEGIN:VCARD", VCARD_HEADER_LENGTH))
211 return @"text/vcard";
212 if ((length >= VCAL_HEADER_LENGTH) && !strncmp(bytes, "BEGIN:VCALENDAR", VCAL_HEADER_LENGTH))
213 return @"text/calendar";
214
215 // Test for plain text.
216 NSUInteger i;
217 for (i = 0; i < length; ++i) {
218 char c = bytes[i];
219 if ((c < 0x20 || c > 0x7E) && (c != '\t' && c != '\r' && c != '\n'))
220 break;
221 }
222 if (i == length) {
223 // Didn't encounter any bad characters, looks like plain text.
224 return @"text/plain";
225 }
226
227 // Looks like this is a binary file.
228
229 // Sniff for the JPEG magic number.
230 if ((length >= JPEG_MAGIC_NUMBER_LENGTH) && !strncmp(bytes, "\xFF\xD8\xFF\xE0", JPEG_MAGIC_NUMBER_LENGTH))
231 return @"image/jpeg";
232
233#undef JPEG_MAGIC_NUMBER_LENGTH
234#undef SCRIPT_TAG_LENGTH
235#undef TEXT_HTML_LENGTH
236#undef VCARD_HEADER_LENGTH
237#undef VCAL_HEADER_LENGTH
238
239 return nil;
240}

246- (BOOL)_web_isCaseInsensitiveEqualToCString:(const char *)string
247{
248 ASSERT(string);
249
250 const char* bytes = static_cast<const char*>([self bytes]);
251 return !strncasecmp(bytes, string, [self length]);
252}

254static const UInt8 *_findEOL(const UInt8 *bytes, CFIndex len)
255{
256 // According to the HTTP specification EOL is defined as
257 // a CRLF pair. Unfortunately, some servers will use LF
258 // instead. Worse yet, some servers will use a combination
259 // of both (e.g. <header>CRLFLF<body>), so findEOL needs
260 // to be more forgiving. It will now accept CRLF, LF, or
261 // CR.
262 //
263 // It returns NULL if EOL is not found or it will return
264 // a pointer to the first terminating character.
265 for (CFIndex i = 0; i < len; i++) {
266 UInt8 c = bytes[i];
267 if ('\n' == c)
268 return bytes + i;
269 if ('\r' == c) {
270 // Check to see if spanning buffer bounds
271 // (CRLF is across reads). If so, wait for
272 // next read.
273 if (i + 1 == len)
274 break;
275
276 return bytes + i;
277 }
278 }
279
280 return nullptr;
281}

283- (NSMutableDictionary *)_webkit_parseRFC822HeaderFields
284{
285 NSMutableDictionary *headerFields = [NSMutableDictionary dictionary];
286
287 const UInt8* bytes = static_cast<const UInt8*>([self bytes]);
288 NSUInteger length = [self length];
289 RetainPtr<NSString> lastKey;
290 const UInt8 *eol;
291
292 // Loop over lines until we're past the header, or we can't find any more end-of-lines
293 while ((eol = _findEOL(bytes, length))) {
294 const UInt8 *line = bytes;
295 SInt32 lineLength = eol - bytes;
296
297 // Move bytes to the character after the terminator as returned by _findEOL.
298 bytes = eol + 1;
299 if (('\r' == *eol) && ('\n' == *bytes))
300 bytes++; // Safe since _findEOL won't return a spanning CRLF.
301
302 length -= (bytes - line);
303 if (!lineLength) {
304 // Blank line; we're at the end of the header
305 break;
306 }
307 if (*line == ' ' || *line == '\t') {
308 // Continuation of the previous header
309 if (!lastKey) {
310 // malformed header; ignore it and continue
311 continue;
312 }
313 // Merge the continuation of the previous header
314 NSString *currentValue = [headerFields objectForKey:lastKey.get()];
315 auto newValue = adoptNS([[NSString alloc] initWithBytes:line length:lineLength encoding:NSISOLatin1StringEncoding]);
316 ASSERT(currentValue);
317 ASSERT(newValue);
318 [headerFields setObject:[currentValue stringByAppendingString:newValue.get()] forKey:lastKey.get()];
319 } else {
320 // Brand new header
321 const UInt8* colon;
322 for (colon = line; *colon != ':' && colon != eol; colon++) { }
323 if (colon == eol) {
324 // malformed header; ignore it and continue
325 continue;
326 }
327 lastKey = adoptNS([[NSString alloc] initWithBytes:line length:colon - line encoding:NSISOLatin1StringEncoding]);
328 lastKey = [lastKey _web_capitalizeRFC822HeaderFieldName];
329 for (colon++; colon != eol; colon++) {
330 if (*colon != ' ' && *colon != '\t')
331 break;
332 }
333 auto value = adoptNS([[NSString alloc] initWithBytes:colon length:eol - colon encoding:NSISOLatin1StringEncoding]);
334 if (NSString *oldValue = [headerFields objectForKey:lastKey.get()])
335 value = adoptNS([[NSString alloc] initWithFormat:@"%@, %@", oldValue, value.get()]);
336 [headerFields setObject:value.get() forKey:lastKey.get()];
337 }
338 }
339
340 return headerFields;
341}

348- (NSInteger)_web_locationAfterFirstBlankLine
349{
350 const char *bytes = (const char *)[self bytes];
351 NSUInteger length = [self length];
352
353 unsigned i;
354 for (i = 0; i < length - 4; i++) {
355
356 // Support for Acrobat. It sends "\n\n".
357 if (bytes[i] == '\n' && bytes[i+1] == '\n')
358 return i+2;
359
360 // Returns the position after 2 CRLF's or 1 CRLF if it is the first line.
361 if (bytes[i] == '\r' && bytes[i+1] == '\n') {
362 i += 2;
363 if (i == 2)
364 return i;
365 if (bytes[i] == '\n') {
366 // Support for Director. It sends "\r\n\n" (3880387).
367 return i+1;
368 }
369 if (bytes[i] == '\r' && bytes[i+1] == '\n') {
370 // Support for Flash. It sends "\r\n\r\n" (3758113).
371 return i+2;
372 }
373 }
374 }
375 return NSNotFound;
376}
