29#import "WebNSObjectExtras.h"

31#import <wtf/Assertions.h>
32#import <wtf/RetainPtr.h>
33
34@interface WebMainThreadInvoker : NSProxy
35{
36 id target;
37 RetainPtr<id> exception;
38}
39@end
40
41@interface NSInvocation (WebMainThreadInvoker)
42- (void)_webkit_invokeAndHandleException:(WebMainThreadInvoker *)exceptionHandler;
43@end
44
45static bool returnTypeIsObject(NSInvocation *invocation)
46{
47 // Could use either _C_ID or NSObjCObjectType, but it seems that neither is
48 // both available and non-deprecated on all versions of Mac OS X we support.
49 return strchr([[invocation methodSignature] methodReturnType], '@');
50}
51
52@implementation WebMainThreadInvoker
53
54- (id)initWithTarget:(id)passedTarget
55{
56 target = passedTarget;
57 return self;
58}
59
60- (void)forwardInvocation:(NSInvocation *)invocation
61{
62 [invocation setTarget:target];
63 [invocation performSelectorOnMainThread:@selector(_webkit_invokeAndHandleException:) withObject:self waitUntilDone:YES];
64 if (exception) {
65 auto exceptionToThrow = std::exchange(exception, nil);
66 @throw exceptionToThrow.get();
67 } else if (returnTypeIsObject(invocation)) {
68 // _webkit_invokeAndHandleException retained the return value on the main thread.
69 // Now autorelease it on the calling thread.
70 id returnValue;
71 [invocation getReturnValue:&returnValue];
72 adoptNS(returnValue).autorelease();
73 }
74}
75
76- (NSMethodSignature *)methodSignatureForSelector:(SEL)selector
77{
78 return [target methodSignatureForSelector:selector];
79}
80
81- (void)handleException:(id)passedException
82{
83 ASSERT(!exception);
84 exception = passedException;
85}
86
87@end
88
89@implementation NSInvocation (WebMainThreadInvoker)
90
91- (void)_webkit_invokeAndHandleException:(WebMainThreadInvoker *)exceptionHandler
92{
93 @try {
94 [self invoke];
95 } @catch (id exception) {
96 [exceptionHandler handleException:exception];
97 return;
98 }
99 if (returnTypeIsObject(self)) {
100 // Retain the return value on the main thread.
101 // -[WebMainThreadInvoker forwardInvocation:] will autorelease it on the calling thread.
102 id value;
103 [self getReturnValue:&value];
104 [value retain];
105 }
106}
107
108@end
109
110@implementation NSObject (WebNSObjectExtras)
111
112+ (id)_webkit_invokeOnMainThread
113{
114 return adoptNS([[WebMainThreadInvoker alloc] initWithTarget:self]).autorelease();
115}
116
117- (id)_webkit_invokeOnMainThread
118{
119 return adoptNS([[WebMainThreadInvoker alloc] initWithTarget:self]).autorelease();
120}
