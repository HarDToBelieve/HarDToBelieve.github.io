29#import "WebNSURLRequestExtras.h"
30
31#import "WebNSURLExtras.h"
32
33#define WebContentType (@"Content-Type")
34#define WebUserAgent (@"User-Agent")
35#define WebReferrer (@"Referer")
36
37@implementation NSMutableURLRequest (WebNSURLRequestExtras)
38
39- (void)_web_setHTTPContentType:(NSString *)contentType
40{
41 [self setValue:contentType forHTTPHeaderField:WebContentType];
42}
43
44- (void)_web_setHTTPReferrer:(NSString *)referrer
45{
46 // Do not set the referrer to a string that refers to a file URL.
47 // That is a potential security hole.
48 if ([referrer _webkit_isFileURL])
49 return;
50
51 // Don't allow empty Referer: headers; some servers refuse them
52 if ([referrer length] == 0)
53 return;
54
55 [self setValue:referrer forHTTPHeaderField:WebReferrer];
56}
57
58- (void)_web_setHTTPUserAgent:(NSString *)userAgent
59{
60 [self setValue:userAgent forHTTPHeaderField:WebUserAgent];
61}
62
63@end
