39static WebCore::FontCascade& fontFromNSFont(NSFont *font)
40{
41 static NeverDestroyed<RetainPtr<NSFont>> currentNSFont;
42 static NeverDestroyed<WebCore::FontCascade> currentFont;
43 if ([font isEqual:currentNSFont.get().get()])
44 return currentFont;
45 currentNSFont.get() = font;
46 currentFont.get() = WebCore::FontCascade(WebCore::FontPlatformData((__bridge CTFontRef)font, [font pointSize]));
47 return currentFont;
48}

50@implementation WebStringTruncator
51
52+ (void)initialize
53{
54 JSC::initialize();
55 WTF::initializeMainThread();
56 WebCore::populateJITOperations();
57}
58
59+ (NSString *)centerTruncateString:(NSString *)string toWidth:(float)maxWidth
60{
61 static NeverDestroyed<RetainPtr<NSFont>> menuFont = [NSFont menuFontOfSize:0];
62
63 ASSERT(menuFont.get());
64 if (!menuFont.get())
65 return nil;
66
67 return WebCore::StringTruncator::centerTruncate(string, maxWidth, fontFromNSFont(menuFont.get().get()));
68}
69
70+ (NSString *)centerTruncateString:(NSString *)string toWidth:(float)maxWidth withFont:(NSFont *)font
71{
72 if (!font)
73 return nil;
74
75 return WebCore::StringTruncator::centerTruncate(string, maxWidth, fontFromNSFont(font));
76}
77
78+ (NSString *)rightTruncateString:(NSString *)string toWidth:(float)maxWidth withFont:(NSFont *)font
79{
80 if (!font)
81 return nil;
82
83 return WebCore::StringTruncator::rightTruncate(string, maxWidth, fontFromNSFont(font));
84}
85
86+ (float)widthOfString:(NSString *)string font:(NSFont *)font
87{
88 if (!font)
89 return 0;
90
91 return WebCore::StringTruncator::width(string, fontFromNSFont(font));
92}

94@end
