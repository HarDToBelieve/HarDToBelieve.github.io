71NetscapePluginHostProxy* NetscapePluginHostManager::hostForPlugin(const WTF::String& pluginPath, cpu_type_t pluginArchitecture, const String& bundleIdentifier)
72{
73 PluginHostMap::AddResult result = m_pluginHosts.add(pluginPath, nullptr);
74
75 // The package was already in the map, just return it.
76 if (!result.isNewEntry)
77 return result.iterator->value;
78
79 mach_port_t clientPort = MACH_PORT_NULL;
80 if (mach_port_allocate(mach_task_self(), MACH_PORT_RIGHT_RECEIVE, &clientPort) != KERN_SUCCESS) {
81 m_pluginHosts.remove(result.iterator);
82 return nullptr;
83 }
84
85 mach_port_t pluginHostPort = MACH_PORT_NULL;
86 ProcessSerialNumber pluginHostPSN;
87 if (!spawnPluginHost(pluginPath, pluginArchitecture, clientPort, pluginHostPort, pluginHostPSN)) {
88 mach_port_mod_refs(mach_task_self(), clientPort, MACH_PORT_RIGHT_RECEIVE, -1);
89 m_pluginHosts.remove(result.iterator);
90 return nullptr;
91 }
92
93 // Since Flash NPObjects add methods dynamically, we don't want to cache when a property/method doesn't exist
94 // on an object because it could be added later.
95 bool shouldCacheMissingPropertiesAndMethods = bundleIdentifier != "com.macromedia.Flash Player.plugin";
96
97 NetscapePluginHostProxy* hostProxy = new NetscapePluginHostProxy(clientPort, pluginHostPort, pluginHostPSN, shouldCacheMissingPropertiesAndMethods);
98
99 result.iterator->value = hostProxy;
100
101 return hostProxy;
102}

104static NSString *preferredBundleLocalizationName()
105{
106 // FIXME: Any use of this function to pass localizations to another
107 // process is likely not completely right, since it only considers
108 // one localization.
109 NSArray *preferredLocalizations = [[NSBundle mainBundle] preferredLocalizations];
110 if (!preferredLocalizations || ![preferredLocalizations count])
111 return @"en_US";
112
113 NSString *language = [preferredLocalizations objectAtIndex:0];
114
115 // FIXME: <rdar://problem/18083880> Replace use of Script Manager
116 // to canonicalize locales with a custom Web-specific table
117 LangCode languageCode;
118 RegionCode regionCode;
119
120 Boolean success = CFLocaleGetLanguageRegionEncodingForLocaleIdentifier((__bridge CFStringRef)language, &languageCode, &regionCode, nullptr, nullptr);
121 if (!success)
122 return @"en_US";
123
124 return adoptCF(CFLocaleCreateCanonicalLocaleIdentifierFromScriptManagerCodes(0, languageCode, regionCode)).bridgingAutorelease();
125}

127bool NetscapePluginHostManager::spawnPluginHost(const String& pluginPath, cpu_type_t pluginArchitecture, mach_port_t clientPort, mach_port_t& pluginHostPort, ProcessSerialNumber& pluginHostPSN)
128{
129 if (m_pluginVendorPort == MACH_PORT_NULL) {
130 if (!initializeVendorPort())
131 return false;
132 }
133
134 if (!CARenderServerStart())
135 return MACH_PORT_NULL;
136
137 mach_port_t renderServerPort = CARenderServerGetPort();
138 if (!MACH_PORT_VALID(renderServerPort))
139 return false;
140
141 NSString *pluginHostAppPath = [[NSBundle bundleForClass:[WebNetscapePluginPackage class]] pathForAuxiliaryExecutable:pluginHostAppName];
142 NSString *pluginHostAppExecutablePath = [[NSBundle bundleWithPath:pluginHostAppPath] executablePath];
143
144 NSDictionary *launchProperties = @{
145 @"pluginHostPath": pluginHostAppExecutablePath,
146 @"cpuType": @(pluginArchitecture),
147 @"localization": preferredBundleLocalizationName(),
148 };
149 NSData *data = [NSPropertyListSerialization dataWithPropertyList:launchProperties format:NSPropertyListBinaryFormat_v1_0 options:0 error:nullptr];
150 ASSERT(data);
151
152 kern_return_t kr = _WKPASpawnPluginHost(m_pluginVendorPort, reinterpret_cast<uint8_t*>(const_cast<void*>([data bytes])), [data length], &pluginHostPort);
153
154 if (kr == MACH_SEND_INVALID_DEST) {
155 // The plug-in vendor port has gone away for some reason. Try to reinitialize it.
156 m_pluginVendorPort = MACH_PORT_NULL;
157 if (!initializeVendorPort())
158 return false;
159
160 // And spawn the plug-in host again.
161 kr = _WKPASpawnPluginHost(m_pluginVendorPort, reinterpret_cast<uint8_t*>(const_cast<void*>([data bytes])), [data length], &pluginHostPort);
162 }
163
164 if (kr != KERN_SUCCESS) {
165 // FIXME: Check for invalid dest and try to re-spawn the plug-in agent.
166 LOG_ERROR("Failed to spawn plug-in host, error %x", kr);
167 return false;
168 }
169
170 NSString *visibleName = [NSString stringWithFormat:UI_STRING_INTERNAL("%@ (%@ Internet plug-in)",
171 "visible name of the plug-in host process. The first argument is the plug-in name "
172 "and the second argument is the application name."),
173 [[(NSString*)pluginPath lastPathComponent] stringByDeletingPathExtension], [[NSProcessInfo processInfo] processName]];
174
175 auto hostProperties = @{
176 @"visibleName": visibleName,
177 @"bundlePath": (NSString *)pluginPath,
178 };
179 data = [NSPropertyListSerialization dataWithPropertyList:hostProperties format:NSPropertyListBinaryFormat_v1_0 options:0 error:NULL];
180 ASSERT(data);
181
182 ProcessSerialNumber psn;
183
184 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
185 GetCurrentProcess(&psn);
186 ALLOW_DEPRECATED_DECLARATIONS_END
187
188 ASSERT(MACH_PORT_VALID(clientPort));
189 kr = _WKPHCheckInWithPluginHost(pluginHostPort, static_cast<uint8_t*>(const_cast<void*>([data bytes])), [data length], clientPort, psn.highLongOfPSN, psn.lowLongOfPSN, renderServerPort,
190 &pluginHostPSN.highLongOfPSN, &pluginHostPSN.lowLongOfPSN);
191
192 if (kr != KERN_SUCCESS) {
193 LOG_ERROR("Failed to check in with plug-in host, error %x", kr);
194 deallocateSendRightSafely(pluginHostPort);
195
196 return false;
197 }
198
199 return true;
200}

202bool NetscapePluginHostManager::initializeVendorPort()
203{
204 ASSERT(m_pluginVendorPort == MACH_PORT_NULL);
205
206 // Get the plug-in agent port.
207 mach_port_t pluginAgentPort = MACH_PORT_NULL;
208 if (bootstrap_look_up(bootstrap_port, "com.apple.WebKit.PluginAgent", &pluginAgentPort) != KERN_SUCCESS) {
209 LOG_ERROR("Failed to look up the plug-in agent port");
210 return false;
211 }
212
213 NSData *appNameData = [[[NSProcessInfo processInfo] processName] dataUsingEncoding:NSUTF8StringEncoding];
214
215 // Tell the plug-in agent that we exist.
216 if (_WKPACheckInApplication(pluginAgentPort, static_cast<uint8_t*>(const_cast<void*>([appNameData bytes])), [appNameData length], &m_pluginVendorPort) != KERN_SUCCESS)
217 return false;
218
219 // FIXME: Should we add a notification for when the vendor port dies?
220
221 return true;
222}

224void NetscapePluginHostManager::pluginHostDied(NetscapePluginHostProxy* pluginHost)
225{
226 PluginHostMap::iterator end = m_pluginHosts.end();
227
228 // This has O(n) complexity but the number of active plug-in hosts is very small so it shouldn't matter.
229 for (PluginHostMap::iterator it = m_pluginHosts.begin(); it != end; ++it) {
230 if (it->value == pluginHost) {
231 m_pluginHosts.remove(it);
232 return;
233 }
234 }

237RefPtr<NetscapePluginInstanceProxy> NetscapePluginHostManager::instantiatePlugin(const String& pluginPath, cpu_type_t pluginArchitecture, const String& bundleIdentifier, WebHostedNetscapePluginView *pluginView, NSString *mimeType, NSArray *attributeKeys, NSArray *attributeValues, NSString *userAgent, NSURL *sourceURL, bool fullFrame, bool isPrivateBrowsingEnabled, bool isAcceleratedCompositingEnabled, bool hostLayersInWindowServer)
238{
239 NetscapePluginHostProxy* hostProxy = hostForPlugin(pluginPath, pluginArchitecture, bundleIdentifier);
240 if (!hostProxy)
241 return nullptr;
242
243 RetainPtr<NSMutableDictionary> properties = adoptNS([[NSMutableDictionary alloc] init]);
244
245 if (mimeType)
246 [properties.get() setObject:mimeType forKey:@"mimeType"];
247
248 ASSERT_ARG(userAgent, userAgent);
249 [properties.get() setObject:userAgent forKey:@"userAgent"];
250
251 ASSERT_ARG(attributeKeys, attributeKeys);
252 [properties.get() setObject:attributeKeys forKey:@"attributeKeys"];
253
254 ASSERT_ARG(attributeValues, attributeValues);
255 [properties.get() setObject:attributeValues forKey:@"attributeValues"];
256
257 if (sourceURL)
258 [properties.get() setObject:[sourceURL absoluteString] forKey:@"sourceURL"];
259
260 [properties.get() setObject:[NSNumber numberWithBool:fullFrame] forKey:@"fullFrame"];
261 [properties.get() setObject:[NSNumber numberWithBool:isPrivateBrowsingEnabled] forKey:@"privateBrowsingEnabled"];
262 [properties.get() setObject:[NSNumber numberWithBool:isAcceleratedCompositingEnabled] forKey:@"acceleratedCompositingEnabled"];
263 [properties.get() setObject:[NSNumber numberWithBool:hostLayersInWindowServer] forKey:@"hostLayersInWindowServer"];
264
265 NSData *data = [NSPropertyListSerialization dataWithPropertyList:properties.get() format:NSPropertyListBinaryFormat_v1_0 options:0 error:nullptr];
266 ASSERT(data);
267
268 auto instance = NetscapePluginInstanceProxy::create(hostProxy, pluginView, fullFrame);
269 uint32_t requestID = instance->nextRequestID();
270 kern_return_t kr = _WKPHInstantiatePlugin(hostProxy->port(), requestID, static_cast<uint8_t*>(const_cast<void*>([data bytes])), [data length], instance->pluginID());
271 if (kr == MACH_SEND_INVALID_DEST) {
272 // Invalidate the instance.
273 instance->invalidate();
274
275 // The plug-in host must have died, but we haven't received the death notification yet.
276 pluginHostDied(hostProxy);
277
278 // Try to spawn it again.
279 hostProxy = hostForPlugin(pluginPath, pluginArchitecture, bundleIdentifier);
280
281 // Create a new instance.
282 instance = NetscapePluginInstanceProxy::create(hostProxy, pluginView, fullFrame);
283 requestID = instance->nextRequestID();
284 _WKPHInstantiatePlugin(hostProxy->port(), requestID, static_cast<uint8_t*>(const_cast<void*>([data bytes])), [data length], instance->pluginID());
285 }
286
287 auto reply = instance->waitForReply<NetscapePluginInstanceProxy::InstantiatePluginReply>(requestID);
288 if (!reply || reply->m_resultCode != KERN_SUCCESS) {
289 instance->cleanup();
290 return nullptr;
291 }
292
293 instance->setRenderContextID(reply->m_renderContextID);
294 instance->setRendererType(reply->m_rendererType);
295
296 return WTFMove(instance);
297}

299void NetscapePluginHostManager::createPropertyListFile(const String& pluginPath, cpu_type_t pluginArchitecture, const String& bundleIdentifier)
300{
301 NetscapePluginHostProxy* hostProxy = hostForPlugin(pluginPath, pluginArchitecture, bundleIdentifier);

304
305 _WKPHCreatePluginMIMETypesPreferences(hostProxy->port());
306}

308void NetscapePluginHostManager::didCreateWindow()
309{
310 // See if any of our hosts are in full-screen mode.
311 PluginHostMap::iterator end = m_pluginHosts.end();
312 for (PluginHostMap::iterator it = m_pluginHosts.begin(); it != end; ++it) {
313 NetscapePluginHostProxy* hostProxy = it->value;
314
315 if (!hostProxy->isMenuBarVisible()) {
316 // Make ourselves the front process.
317 ProcessSerialNumber psn;
318 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
319 GetCurrentProcess(&psn);
320 SetFrontProcess(&psn);
321 ALLOW_DEPRECATED_DECLARATIONS_END
322 return;
323 }
324 }
