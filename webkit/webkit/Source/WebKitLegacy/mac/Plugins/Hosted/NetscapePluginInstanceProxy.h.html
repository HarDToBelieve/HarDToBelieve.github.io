252 template <typename T>
253 std::unique_ptr<T> waitForReply(uint32_t requestID)
254 {
255 Ref<NetscapePluginInstanceProxy> protect(*this); // Plug-in host may crash while we are waiting for reply, releasing all instances to the instance proxy.
256
257 willCallPluginFunction();
258 m_waitingForReply = true;
259
260 auto reply = processRequestsAndWaitForReply(requestID);
261 if (reply)
262 ASSERT(reply->m_type == T::ReplyType);
263
264 m_waitingForReply = false;
265
266 bool stopped = false;
267 didCallPluginFunction(stopped);
268 if (stopped) {
269 // The instance proxy may have been deleted from didCallPluginFunction(), so a null reply needs to be returned.
270 return nullptr;
271 }
272
273 return std::unique_ptr<T>(static_cast<T*>(reply.release()));
274 }

345 // All NPRuntime functions will return false when destroying a plug-in. This is necessary because there may be unhandled messages waiting,
346 // and spinning in processRequests() will unexpectedly execute them from inside destroy(). That's not a good time to execute arbitrary JavaScript,
347 // since both loading and rendering data structures may be in inconsistent state.
348 // This suppresses calls from all plug-ins, even those in different pages, since JS might affect the frame with plug-in that's being stopped.
349 //
350 // FIXME: Plug-ins can execute arbitrary JS from destroy() in same process case, and other browsers also support that.
351 // A better fix may be to make sure that unrelated messages are postponed until after destroy() returns.
352 // Another possible fix may be to send destroy message at a time when internal structures are consistent.
353 //
354 // FIXME: We lack similar message suppression in other cases - resize() is also triggered by layout, so executing arbitrary JS is also problematic.
355 static bool m_inDestroy;
