145JSValue ProxyInstance::invoke(JSC::JSGlobalObject* lexicalGlobalObject, JSC::CallFrame* callFrame, InvokeType type, uint64_t identifier, const ArgList& args)
146{
147 if (!m_instanceProxy)
148 return jsUndefined();
149
150 RetainPtr<NSData> arguments(m_instanceProxy->marshalValues(lexicalGlobalObject, args));
151
152 uint32_t requestID = m_instanceProxy->nextRequestID();
153
154 for (unsigned i = 0; i < args.size(); i++)
155 m_instanceProxy->retainLocalObject(args.at(i));
156
157 if (_WKPHNPObjectInvoke(m_instanceProxy->hostProxy()->port(), m_instanceProxy->pluginID(), requestID, m_objectID, type, identifier, static_cast<char*>(const_cast<void*>([arguments.get() bytes])), [arguments.get() length]) != KERN_SUCCESS) {
158 if (m_instanceProxy) {
159 for (unsigned i = 0; i < args.size(); i++)
160 m_instanceProxy->releaseLocalObject(args.at(i));
161 }
162 return jsUndefined();
163 }
164
165 auto reply = waitForReply<NetscapePluginInstanceProxy::BooleanAndDataReply>(requestID);
166 NetscapePluginInstanceProxy::moveGlobalExceptionToExecState(lexicalGlobalObject);
167
168 if (m_instanceProxy) {
169 for (unsigned i = 0; i < args.size(); i++)
170 m_instanceProxy->releaseLocalObject(args.at(i));
171 }
172
173 if (!reply || !reply->m_returnValue)
174 return jsUndefined();
175
176 return m_instanceProxy->demarshalValue(lexicalGlobalObject, reinterpret_cast<char*>(const_cast<unsigned char*>(CFDataGetBytePtr(reply->m_result.get()))), CFDataGetLength(reply->m_result.get()));
177}

179class ProxyRuntimeMethod final : public RuntimeMethod {
180public:
181 using Base = RuntimeMethod;
182
183 static ProxyRuntimeMethod* create(JSGlobalObject* lexicalGlobalObject, JSGlobalObject* globalObject, const String& name, Bindings::Method* method)
184 {
185 VM& vm = globalObject->vm();
186 // FIXME: deprecatedGetDOMStructure uses the prototype off of the wrong global object
187 // lexicalGlobalObject-vm() is also likely wrong.
188 Structure* domStructure = deprecatedGetDOMStructure<ProxyRuntimeMethod>(lexicalGlobalObject);
189 ProxyRuntimeMethod* runtimeMethod = new (allocateCell<ProxyRuntimeMethod>(vm.heap)) ProxyRuntimeMethod(vm, domStructure, method);
190 runtimeMethod->finishCreation(vm, name);
191 return runtimeMethod;
192 }
193
194 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
195 {
196 return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), &s_info);
197 }
198
199 DECLARE_INFO;
200
201private:
202 ProxyRuntimeMethod(VM& vm, Structure* structure, Bindings::Method* method)
203 : RuntimeMethod(vm, structure, method)
204 {
205 }
206
207 void finishCreation(VM& vm, const String& name)
208 {
209 Base::finishCreation(vm, name);
210 ASSERT(inherits(vm, info()));
211 }
212};

222JSValue ProxyInstance::invokeMethod(JSGlobalObject* lexicalGlobalObject, JSC::CallFrame* callFrame, JSC::RuntimeMethod* runtimeMethod)
223{
224 VM& vm = lexicalGlobalObject->vm();
225 auto scope = DECLARE_THROW_SCOPE(vm);
226
227 if (!asObject(runtimeMethod)->inherits<ProxyRuntimeMethod>(vm))
228 return throwTypeError(lexicalGlobalObject, scope, "Attempt to invoke non-plug-in method on plug-in object."_s);
229
230 ProxyMethod* method = static_cast<ProxyMethod*>(runtimeMethod->method());
231 ASSERT(method);
232
233 return invoke(lexicalGlobalObject, callFrame, Invoke, method->serverIdentifier(), ArgList(callFrame));
234}

236bool ProxyInstance::supportsInvokeDefaultMethod() const
237{
238 if (!m_instanceProxy)
239 return false;
240
241 uint32_t requestID = m_instanceProxy->nextRequestID();
242
243 if (_WKPHNPObjectHasInvokeDefaultMethod(m_instanceProxy->hostProxy()->port(), m_instanceProxy->pluginID(), requestID, m_objectID) != KERN_SUCCESS)
244 return false;
245
246 auto reply = waitForReply<NetscapePluginInstanceProxy::BooleanReply>(requestID);
247 return reply && reply->m_result;
248}

306void ProxyInstance::getPropertyNames(JSGlobalObject* lexicalGlobalObject, PropertyNameArray& nameArray)
307{
308 if (!m_instanceProxy)
309 return;
310
311 uint32_t requestID = m_instanceProxy->nextRequestID();
312
313 if (_WKPHNPObjectEnumerate(m_instanceProxy->hostProxy()->port(), m_instanceProxy->pluginID(), requestID, m_objectID) != KERN_SUCCESS)
314 return;
315
316 auto reply = waitForReply<NetscapePluginInstanceProxy::BooleanAndDataReply>(requestID);
317 NetscapePluginInstanceProxy::moveGlobalExceptionToExecState(lexicalGlobalObject);
318 if (!reply || !reply->m_returnValue)
319 return;
320
321 NSArray *array = [NSPropertyListSerialization propertyListWithData:(__bridge NSData *)reply->m_result.get() options:NSPropertyListImmutable format:nullptr error:nullptr];
322
323 VM& vm = lexicalGlobalObject->vm();
324 for (NSNumber *number in array) {
325 IdentifierRep* identifier = reinterpret_cast<IdentifierRep*>([number longLongValue]);
326 if (!IdentifierRep::isValid(identifier))
327 continue;
328
329 if (identifier->isString()) {
330 const char* str = identifier->string();
331 nameArray.add(Identifier::fromString(vm, String::fromUTF8WithLatin1Fallback(str, strlen(str))));
332 } else
333 nameArray.add(Identifier::from(vm, identifier->number()));
334 }
335}

337Method* ProxyInstance::methodNamed(PropertyName propertyName)
338{
339 String name(propertyName.publicName());
340 if (name.isNull())
341 return nullptr;
342
343 if (!m_instanceProxy)
344 return nullptr;
345
346 // If we already have an entry in the map, use it.
347 auto existingMapEntry = m_methods.find(name.impl());
348 if (existingMapEntry != m_methods.end())
349 return existingMapEntry->value.get();
350
351 uint64_t methodName = reinterpret_cast<uint64_t>(_NPN_GetStringIdentifier(name.ascii().data()));
352 uint32_t requestID = m_instanceProxy->nextRequestID();
353
354 if (_WKPHNPObjectHasMethod(m_instanceProxy->hostProxy()->port(), m_instanceProxy->pluginID(), requestID, m_objectID, methodName) != KERN_SUCCESS)
355 return nullptr;
356
357 auto reply = waitForReply<NetscapePluginInstanceProxy::BooleanReply>(requestID);
358 if (!reply)
359 return nullptr;
360
361 if (!reply->m_result && !m_instanceProxy->hostProxy()->shouldCacheMissingPropertiesAndMethods())
362 return nullptr;
363
364 // Add a new entry to the map unless an entry was added while we were in waitForReply.
365 auto mapAddResult = m_methods.add(name.impl(), nullptr);
366 if (mapAddResult.isNewEntry && reply->m_result)
367 mapAddResult.iterator->value = makeUnique<ProxyMethod>(methodName);
368
369 return mapAddResult.iterator->value.get();
370}

373{
374 String name(propertyName.publicName());

383 if (existingMapEntry != m_fields.end())
384 return existingMapEntry->value.get();
385
386 uint64_t identifier = reinterpret_cast<uint64_t>(_NPN_GetStringIdentifier(name.ascii().data()));
387 uint32_t requestID = m_instanceProxy->nextRequestID();

389 if (_WKPHNPObjectHasProperty(m_instanceProxy->hostProxy()->port(), m_instanceProxy->pluginID(), requestID, m_objectID, identifier) != KERN_SUCCESS)
390 return nullptr;
391
392 auto reply = waitForReply<NetscapePluginInstanceProxy::BooleanReply>(requestID);

407{
408 if (!m_instanceProxy)

415 return jsUndefined();
416

419 if (!reply || !reply->m_returnValue)
420 return jsUndefined();

423}

425bool ProxyInstance::setFieldValue(JSGlobalObject* lexicalGlobalObject, const Field* field, JSValue value) const
426{
427 if (!m_instanceProxy)
428 return false;
429
430 uint64_t serverIdentifier = static_cast<const ProxyField*>(field)->serverIdentifier();
431 uint32_t requestID = m_instanceProxy->nextRequestID();
432
433 data_t valueData;
434 mach_msg_type_number_t valueLength;
435
436 m_instanceProxy->marshalValue(lexicalGlobalObject, value, valueData, valueLength);
437 m_instanceProxy->retainLocalObject(value);
438 kern_return_t kr = _WKPHNPObjectSetProperty(m_instanceProxy->hostProxy()->port(), m_instanceProxy->pluginID(), requestID, m_objectID, serverIdentifier, valueData, valueLength);
439 mig_deallocate(reinterpret_cast<vm_address_t>(valueData), valueLength);
440 if (m_instanceProxy)
441 m_instanceProxy->releaseLocalObject(value);
442 if (kr != KERN_SUCCESS)
443 return false;
444
445 waitForReply<NetscapePluginInstanceProxy::BooleanReply>(requestID);
446 NetscapePluginInstanceProxy::moveGlobalExceptionToExecState(lexicalGlobalObject);
447 return true;
448}

450void ProxyInstance::invalidate()
451{
452 ASSERT(m_instanceProxy);
453
454 if (NetscapePluginHostProxy* hostProxy = m_instanceProxy->hostProxy())
455 _WKPHNPObjectRelease(hostProxy->port(), m_instanceProxy->pluginID(), m_objectID);
456 m_instanceProxy = nullptr;
457}
