62class SoftwareCARenderer {
63 WTF_MAKE_FAST_ALLOCATED;
64public:
65 explicit SoftwareCARenderer(uint32_t contextID)
66 : m_context { [CAContext localContext] }
67 , m_renderContext { CARenderCGNew(0) }
68 {
69 [m_context setLayer:[CALayer _web_renderLayerWithContextID:contextID]];
70 }
71
72 ~SoftwareCARenderer()
73 {
74 CARenderCGDestroy(m_renderContext);
75 }
76
77 void render(CGContextRef context, CGRect rect)
78 {
79 CARenderUpdate* update = CARenderUpdateBegin(nullptr, 0, CACurrentMediaTime(), nullptr, 0, &rect);
80 CARenderUpdateAddContext(update, [m_context renderContext]);
81 CARenderUpdateAddRect(update, &rect);
82 CARenderCGRender(m_renderContext, update, context);
83 CARenderUpdateFinish(update);
84 }
85
86private:
87 RetainPtr<CAContext> m_context;
88 CARenderCGContext* m_renderContext;
89};

157- (BOOL)createPlugin
158{
159 ASSERT(!_proxy);
160
161 NSString *userAgent = [[self webView] userAgentForURL:_baseURL.get()];
162 BOOL acceleratedCompositingEnabled = false;
163 acceleratedCompositingEnabled = [[[self webView] preferences] acceleratedCompositingEnabled];
164 _hostsLayersInWindowServer = [self windowHostsLayersInWindowServer];
165
166 _proxy = NetscapePluginHostManager::singleton().instantiatePlugin([_pluginPackage.get() path], [_pluginPackage.get() pluginHostArchitecture], [_pluginPackage.get() bundleIdentifier], self, _MIMEType.get(), _attributeKeys.get(), _attributeValues.get(), userAgent, _sourceURL.get(),
167 _mode == NP_FULL, _isPrivateBrowsingEnabled, acceleratedCompositingEnabled, _hostsLayersInWindowServer);
168 if (!_proxy)
169 return NO;
170
171 if (_proxy->rendererType() == UseSoftwareRenderer)
172 _softwareRenderer = makeUnique<SoftwareCARenderer>(_proxy->renderContextID());
173 else
174 [self createPluginLayer];
175
176 // Update the window frame.
177 _proxy->windowFrameChanged([[self window] frame]);
178
179 return YES;
180}

182- (void)createPluginLayer
183{
184 BOOL acceleratedCompositingEnabled = false;
185 acceleratedCompositingEnabled = [[[self webView] preferences] acceleratedCompositingEnabled];
186
187 _pluginLayer = [CALayer _web_renderLayerWithContextID:_proxy->renderContextID()];
188
189 if (acceleratedCompositingEnabled && _proxy->rendererType() == UseAcceleratedCompositing) {
190 // FIXME: This code can be shared between WebHostedNetscapePluginView and WebNetscapePluginView.
191 // Since this layer isn't going to be inserted into a view, we need to create another layer and flip its geometry
192 // in order to get the coordinate system right.
193 RetainPtr<CALayer> realPluginLayer = WTFMove(_pluginLayer);
194
195 _pluginLayer = adoptNS([[CALayer alloc] init]);
196 _pluginLayer.get().bounds = realPluginLayer.get().bounds;
197 _pluginLayer.get().geometryFlipped = YES;
198
199 realPluginLayer.get().autoresizingMask = kCALayerWidthSizable | kCALayerHeightSizable;
200 [_pluginLayer.get() addSublayer:realPluginLayer.get()];
201
202 // Eagerly enter compositing mode, since we know we'll need it. This avoids firing setNeedsStyleRecalc()
203 // for iframes that contain composited plugins at bad times. https://bugs.webkit.org/show_bug.cgi?id=39033
204 core([self webFrame])->view()->enterCompositingMode();
205 [self element]->invalidateStyleAndLayerComposition();
206 } else
207 self.wantsLayer = YES;
208}

210- (void)setHostsLayersInWindowServer:(bool)hostsLayersInWindowServer
211{
212 if (_proxy->rendererType() == UseSoftwareRenderer)
213 return;
214
215 RetainPtr<CALayer> currentSuperlayer = [_pluginLayer superlayer];
216
217 _hostsLayersInWindowServer = hostsLayersInWindowServer;
218
219 [self createPluginLayer];
220 [self setLayer:currentSuperlayer.get()];
221}

256- (void)updateAndSetWindow
257{
258 if (!_proxy)
259 return;
260
261 // The base coordinates of a window and it's contentView happen to be the equal at a userSpaceScaleFactor
262 // of 1. For non-1.0 scale factors this assumption is false.
263 NSView *windowContentView = [[self window] contentView];
264 NSRect boundsInWindow = [self convertRect:[self bounds] toView:windowContentView];
265
266 NSRect visibleRectInWindow;
267
268 // Core Animation plug-ins need to be updated (with a 0,0,0,0 clipRect) when
269 // moved to a background tab. We don't do this for Core Graphics plug-ins as
270 // older versions of Flash have historical WebKit-specific code that isn't
271 // compatible with this behavior.
272 BOOL shouldClipOutPlugin = _pluginLayer && [self shouldClipOutPlugin];
273 if (!shouldClipOutPlugin)
274 visibleRectInWindow = [self actualVisibleRectInWindow];
275 else
276 visibleRectInWindow = NSZeroRect;
277
278 // Flip Y to convert NSWindow coordinates to top-left-based window coordinates.
279 float borderViewHeight = [[self currentWindow] frame].size.height;
280 boundsInWindow.origin.y = borderViewHeight - NSMaxY(boundsInWindow);
281
282 if (!shouldClipOutPlugin)
283 visibleRectInWindow.origin.y = borderViewHeight - NSMaxY(visibleRectInWindow);
284
285 _previousSize = boundsInWindow.size;
286
287 _proxy->resize(boundsInWindow, visibleRectInWindow);
288
289 bool shouldHostLayersInWindowServer = [self windowHostsLayersInWindowServer];
290 if (_hostsLayersInWindowServer != shouldHostLayersInWindowServer)
291 _proxy->setShouldHostLayersInWindowServer(shouldHostLayersInWindowServer);
292}

343- (void)addWindowObservers
344{
345 [super addWindowObservers];
346
347 ASSERT([self window]);
348
349 NSWindow *window = [self window];
350
351 NSNotificationCenter *notificationCenter = [NSNotificationCenter defaultCenter];
352 [notificationCenter addObserver:self selector:@selector(windowFrameDidChange:)
353 name:NSWindowDidMoveNotification object:window];
354 [notificationCenter addObserver:self selector:@selector(windowFrameDidChange:)
355 name:NSWindowDidResizeNotification object:window];
356
357 if (_proxy)
358 _proxy->windowFrameChanged([window frame]);
359 [self updateAndSetWindow];
360}

399{
400 if (_isStarted && _proxy)
401 _proxy->mouseEvent(self, event, NPCocoaEventMouseExited);
402
403 // Set cursor back to arrow cursor. Because NSCursor doesn't know about changes that the plugin made, we could get confused about what we think the
404 // current cursor is otherwise. Therefore we have no choice but to unconditionally reset the cursor when the mouse exits the plugin.
405 // FIXME: This should be job of plugin host, see <rdar://problem/7654434>.

458- (void)pluginHostDied
459{
460 if (is<RenderEmbeddedObject>(_element->renderer()))
461 downcast<RenderEmbeddedObject>(*_element->renderer()).setPluginUnavailabilityReason(RenderEmbeddedObject::PluginCrashed);
462
463 _pluginLayer = nil;
464 _proxy = nullptr;
465
466 // No need for us to be layer backed anymore
467 self.wantsLayer = NO;
468
469 [self invalidatePluginContentRect:[self bounds]];
470}

472- (void)drawRect:(NSRect)rect
473{
474 if (_cachedSnapshot) {
475 NSRect sourceRect = { NSZeroPoint, [_cachedSnapshot.get() size] };
476 [_cachedSnapshot.get() drawInRect:[self bounds] fromRect:sourceRect operation:NSCompositingOperationSourceOver fraction:1];
477 return;
478 }
479
480 if (_proxy) {
481 if (_softwareRenderer) {
482 if ([NSGraphicsContext currentContextDrawingToScreen]) {
483 _softwareRenderer->render([[NSGraphicsContext currentContext] CGContext], NSRectToCGRect(rect));
484 _proxy->didDraw();
485 } else
486 _proxy->print([[NSGraphicsContext currentContext] CGContext], [self bounds].size.width, [self bounds].size.height);
487 } else if (_snapshotting && [self supportsSnapshotting]) {
488 _proxy->snapshot([[NSGraphicsContext currentContext] CGContext], [self bounds].size.width, [self bounds].size.height);
489 }
490
491 return;
492 }
493}
