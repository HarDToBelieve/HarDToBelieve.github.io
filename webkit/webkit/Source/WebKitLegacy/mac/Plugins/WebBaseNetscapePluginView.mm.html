129{

132

220- (void)removeTrackingRect
221{
222 if (_trackingTag) {
223 [self removeTrackingRect:_trackingTag];
224 _trackingTag = 0;
225
226 // Do the following after setting trackingTag to 0 so we don't re-enter.
227
228 // Balance the retain in resetTrackingRect. Use autorelease in case we hold
229 // the last reference to the window during tear-down, to avoid crashing AppKit.
230 [[self window] autorelease];
231 }
232}

235{
236 [self removeTrackingRect];
237 if (_isStarted) {
238 // Retain the window so that removeTrackingRect can work after the window is closed.
239 [[self window] retain];
240 _trackingTag = [self addTrackingRect:[self bounds] owner:self userData:nil assumeInside:NO];
241 }

341- (void)start
342{
343 ASSERT([self currentWindow]);
344
345 if (_isStarted)
346 return;
347
348 if (_triedAndFailedToCreatePlugin)
349 return;
350
351 ASSERT([self webView]);
352
353 if (![[[self webView] preferences] arePlugInsEnabled])
354 return;
355
356 Frame* frame = core([self webFrame]);
357 if (!frame)
358 return;
359 Page* page = frame->page();
360 if (!page)
361 return;
362
363 bool wasDeferring = page->defersLoading();
364 if (!wasDeferring)
365 page->setDefersLoading(true);
366
367 BOOL result = [self createPlugin];
368
369 if (!wasDeferring)
370 page->setDefersLoading(false);
371
372 if (!result) {
373 _triedAndFailedToCreatePlugin = YES;
374 return;
375 }
376
377 _isStarted = YES;
378
379 [[self webView] addPluginInstanceView:self];
380
381 if ([self currentWindow])
382 [self updateAndSetWindow];
383
384 if ([self window]) {
385 [self addWindowObservers];
386 if ([[self window] isKeyWindow]) {
387 [self sendActivateEvent:YES];
388 }
389 [self restartTimers];
390 }
391
392 [self resetTrackingRect];
393
394 [self loadStream];
395}

397- (void)stop
398{
399 if (![self shouldStop])
400 return;
401
402 [self removeTrackingRect];
403
404 if (!_isStarted)
405 return;
406
407 _isStarted = NO;
408
409 [[self webView] removePluginInstanceView:self];
410
411 // Stop the timers
412 [self stopTimers];
413
414 // Stop notifications and callbacks.
415 [self removeWindowObservers];
416
417 [self destroyPlugin];
418}

453- (void)viewWillMoveToWindow:(NSWindow *)newWindow
454{
455 // We must remove the tracking rect before we move to the new window.
456 // Once we move to the new window, it will be too late.
457 [self removeTrackingRect];
458 [self removeWindowObservers];
459
460 // Workaround for: <rdar://problem/3822871> resignFirstResponder is not sent to first responder view when it is removed from the window
461 [self setHasFocus:NO];
462
463 if (!newWindow) {
464 if ([[self webView] hostWindow]) {
465 // View will be moved out of the actual window but it still has a host window.
466 [self stopTimers];
467 } else {
468 // View will have no associated windows.
469 [self stop];
470
471 // Stop observing WebPreferencesChangedInternalNotification -- we only need to observe this when installed in the view hierarchy.
472 // When not in the view hierarchy, -viewWillMoveToWindow: and -viewDidMoveToWindow will start/stop the plugin as needed.
473 [[NSNotificationCenter defaultCenter] removeObserver:self name:WebPreferencesChangedInternalNotification object:nil];
474 }
475 }
476}

478- (void)viewWillMoveToSuperview:(NSView *)newSuperview
479{
480 if (!newSuperview) {
481 // Stop the plug-in when it is removed from its superview. It is not sufficient to do this in -viewWillMoveToWindow:nil, because
482 // the WebView might still has a hostWindow at that point, which prevents the plug-in from being destroyed.
483 // There is no need to start the plug-in when moving into a superview. -viewDidMoveToWindow takes care of that.
484 [self stop];
485
486 // Stop observing WebPreferencesChangedInternalNotification -- we only need to observe this when installed in the view hierarchy.
487 // When not in the view hierarchy, -viewWillMoveToWindow: and -viewDidMoveToWindow will start/stop the plugin as needed.
488 [[NSNotificationCenter defaultCenter] removeObserver:self name:WebPreferencesChangedInternalNotification object:nil];
489 }
490}
491
492- (void)viewDidMoveToWindow
493{
494 [self resetTrackingRect];
495
496 if ([self window]) {
497 // While in the view hierarchy, observe WebPreferencesChangedInternalNotification so that we can start/stop depending
498 // on whether plugins are enabled.
499 [[NSNotificationCenter defaultCenter] addObserver:self
500 selector:@selector(preferencesHaveChanged:)
501 name:WebPreferencesChangedInternalNotification
502 object:nil];
503
504 _isPrivateBrowsingEnabled = [[[self webView] preferences] privateBrowsingEnabled];
505
506 // View moved to an actual window. Start it if not already started.
507 [self start];
508
509 // Starting the plug-in can result in it removing itself from the window so we need to ensure that we're still in
510 // place before doing anything that requires a window.
511 if ([self window]) {
512 [self restartTimers];
513 [self addWindowObservers];
514 }
515 } else if ([[self webView] hostWindow]) {
516 // View moved out of an actual window, but still has a host window.
517 // Call setWindow to explicitly "clip out" the plug-in from sight.
518 // FIXME: It would be nice to do this where we call stopNullEvents in viewWillMoveToWindow.
519 [self updateAndSetWindow];
520 }
521}
522
523- (void)viewWillMoveToHostWindow:(NSWindow *)hostWindow
524{
525 if (!hostWindow && ![self window]) {
526 // View will have no associated windows.
527 [self stop];
528
529 // Remove WebPreferencesChangedInternalNotification observer -- we will observe once again when we move back into the window
530 [[NSNotificationCenter defaultCenter] removeObserver:self name:WebPreferencesChangedInternalNotification object:nil];
531 }
532}
533
534- (void)viewDidMoveToHostWindow
535{
536 if ([[self webView] hostWindow]) {
537 // View now has an associated window. Start it if not already started.
538 [self start];
539 }

583- (void)preferencesHaveChanged:(NSNotification *)notification
584{
585 WebPreferences *preferences = [[self webView] preferences];
586
587 if ([notification object] != preferences)
588 return;
589
590 BOOL arePlugInsEnabled = [preferences arePlugInsEnabled];
591 if (_isStarted != arePlugInsEnabled) {
592 if (arePlugInsEnabled) {
593 if ([self currentWindow]) {
594 [self start];
595 }
596 } else {
597 [self stop];
598 [self invalidatePluginContentRect:[self bounds]];
599 }
600 }
601
602 BOOL isPrivateBrowsingEnabled = [preferences privateBrowsingEnabled];
603 if (isPrivateBrowsingEnabled != _isPrivateBrowsingEnabled) {
604 _isPrivateBrowsingEnabled = isPrivateBrowsingEnabled;
605 [self privateBrowsingModeDidChange];
606 }
607}

609ALLOW_DEPRECATED_IMPLEMENTATIONS_BEGIN
610- (void)renewGState
611ALLOW_DEPRECATED_IMPLEMENTATIONS_END
612{
613 [super renewGState];
614
615 // -renewGState is called whenever the view's geometry changes. It's a little hacky to override this method, but
616 // much safer than walking up the view hierarchy and observing frame/bounds changed notifications, since you don't
617 // have to track subsequent changes to the view hierarchy and add/remove notification observers.
618 // NSOpenGLView uses the exact same technique to reshape its OpenGL surface.
619
620 // All of the work this method does may safely be skipped if the view is not in a window. When the view
621 // is moved back into a window, everything should be set up correctly.
622 if (![self window])
623 return;
624
625 [self updateAndSetWindow];
626
627 [self resetTrackingRect];
628
629 // Check to see if the plugin view is completely obscured (scrolled out of view, for example).
630 // For performance reasons, we send null events at a lower rate to plugins which are obscured.
631 BOOL oldIsObscured = _isCompletelyObscured;
632 _isCompletelyObscured = NSIsEmptyRect([self visibleRect]);
633 if (_isCompletelyObscured != oldIsObscured)
634 [self restartTimers];
635}

707- (BOOL)convertFromX:(double)sourceX andY:(double)sourceY space:(NPCoordinateSpace)sourceSpace
708 toX:(double *)destX andY:(double *)destY space:(NPCoordinateSpace)destSpace
709{
710 // Nothing to do
711 if (sourceSpace == destSpace) {
712 if (destX)
713 *destX = sourceX;
714 if (destY)
715 *destY = sourceY;
716 return YES;
717 }
718
719 NSPoint sourcePoint = NSMakePoint(sourceX, sourceY);
720
721 NSPoint sourcePointInScreenSpace;
722
723 // First convert to screen space
724 switch (sourceSpace) {
725 case NPCoordinateSpacePlugin:
726 sourcePointInScreenSpace = [self convertPoint:sourcePoint toView:nil];
727 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
728 sourcePointInScreenSpace = [[self currentWindow] convertBaseToScreen:sourcePointInScreenSpace];
729 ALLOW_DEPRECATED_DECLARATIONS_END
730 break;
731
732 case NPCoordinateSpaceWindow:
733 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
734 sourcePointInScreenSpace = [[self currentWindow] convertBaseToScreen:sourcePoint];
735 ALLOW_DEPRECATED_DECLARATIONS_END
736 break;
737
738 case NPCoordinateSpaceFlippedWindow:
739 sourcePoint.y = [[self currentWindow] frame].size.height - sourcePoint.y;
740 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
741 sourcePointInScreenSpace = [[self currentWindow] convertBaseToScreen:sourcePoint];
742 ALLOW_DEPRECATED_DECLARATIONS_END
743 break;
744
745 case NPCoordinateSpaceScreen:
746 sourcePointInScreenSpace = sourcePoint;
747 break;
748
749 case NPCoordinateSpaceFlippedScreen:
750 sourcePoint.y = [(NSScreen *)[[NSScreen screens] objectAtIndex:0] frame].size.height - sourcePoint.y;
751 sourcePointInScreenSpace = sourcePoint;
752 break;
753 default:
754 return FALSE;
755 }
756
757 NSPoint destPoint;
758
759 // Then convert back to the destination space
760 switch (destSpace) {
761 case NPCoordinateSpacePlugin:
762 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
763 destPoint = [[self currentWindow] convertScreenToBase:sourcePointInScreenSpace];
764 ALLOW_DEPRECATED_DECLARATIONS_END
765 destPoint = [self convertPoint:destPoint fromView:nil];
766 break;
767
768 case NPCoordinateSpaceWindow:
769 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
770 destPoint = [[self currentWindow] convertScreenToBase:sourcePointInScreenSpace];
771 ALLOW_DEPRECATED_DECLARATIONS_END
772 break;
773
774 case NPCoordinateSpaceFlippedWindow:
775 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
776 destPoint = [[self currentWindow] convertScreenToBase:sourcePointInScreenSpace];
777 ALLOW_DEPRECATED_DECLARATIONS_END
778 destPoint.y = [[self currentWindow] frame].size.height - destPoint.y;
779 break;
780
781 case NPCoordinateSpaceScreen:
782 destPoint = sourcePointInScreenSpace;
783 break;
784
785 case NPCoordinateSpaceFlippedScreen:
786 destPoint = sourcePointInScreenSpace;
787 destPoint.y = [(NSScreen *)[[NSScreen screens] objectAtIndex:0] frame].size.height - destPoint.y;
788 break;
789
790 default:
791 return FALSE;
792 }
793
794 if (destX)
795 *destX = destPoint.x;
796 if (destY)
797 *destY = destPoint.y;
798
799 return TRUE;
800}

803- (void)invalidatePluginContentRect:(NSRect)rect
804{
805 auto* renderer = _element->renderer();
806 if (!is<RenderEmbeddedObject>(renderer))
807 return;
808 auto& object = downcast<RenderEmbeddedObject>(*renderer);
809 IntRect contentRect(rect);
810 contentRect.move(object.borderLeft() + object.paddingLeft(), object.borderTop() + object.paddingTop());
811 object.repaintRectangle(contentRect);
812}

815{

840bool getAuthenticationInfo(const char* protocolStr, const char* hostStr, int32_t port, const char* schemeStr, const char* realmStr, CString& username, CString& password)
841{
842 if (!equalLettersIgnoringASCIICase(protocolStr, "http") && !equalLettersIgnoringASCIICase(protocolStr, "https"))
843 return false;
844
845 NSString *host = [NSString stringWithUTF8String:hostStr];
846 if (!hostStr)
847 return false;
848
849 NSString *protocol = [NSString stringWithUTF8String:protocolStr];
850 if (!protocol)
851 return false;
852
853 NSString *realm = [NSString stringWithUTF8String:realmStr];
854 if (!realm)
855 return NPERR_GENERIC_ERROR;
856
857 NSString *authenticationMethod = NSURLAuthenticationMethodDefault;
858 if (equalLettersIgnoringASCIICase(protocolStr, "http")) {
859 if (equalLettersIgnoringASCIICase(schemeStr, "basic"))
860 authenticationMethod = NSURLAuthenticationMethodHTTPBasic;
861 else if (equalLettersIgnoringASCIICase(schemeStr, "digest"))
862 authenticationMethod = NSURLAuthenticationMethodHTTPDigest;
863 }
864
865 RetainPtr<NSURLProtectionSpace> protectionSpace = adoptNS([[NSURLProtectionSpace alloc] initWithHost:host port:port protocol:protocol realm:realm authenticationMethod:authenticationMethod]);
866
867 NSURLCredential *credential = NetworkStorageSessionMap::defaultStorageSession().credentialStorage().get(emptyString(), ProtectionSpace(protectionSpace.get())).nsCredential();
868 if (!credential)
869 credential = [[NSURLCredentialStorage sharedCredentialStorage] defaultCredentialForProtectionSpace:protectionSpace.get()];
870 if (!credential)
871 return false;
872
873 if (![credential hasPassword])
874 return false;
875
876 username = [[credential user] UTF8String];
877 password = [[credential password] UTF8String];
878
879 return true;
880}

882void sendUserChangeNotifications()

884 auto consoleConnectionChangeNotifyProc = [](CGSNotificationType type, CGSNotificationData, CGSByteCount, CGSNotificationArg) {

889 notificationName = LoginWindowDidSwitchFromUserNotification;

893 };
894
895 CGSRegisterNotifyProc(consoleConnectionChangeNotifyProc, kCGSessionConsoleConnect, nullptr);
896 CGSRegisterNotifyProc(consoleConnectionChangeNotifyProc, kCGSessionConsoleDisconnect, nullptr);
897}
