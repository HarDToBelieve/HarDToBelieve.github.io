68+ (WebBasePluginPackage *)pluginWithPath:(NSString *)pluginPath
69{
70
71 auto pluginPackage = adoptNS([[WebPluginPackage alloc] initWithPath:pluginPath]);
72
73 if (!pluginPackage) {
74#if ENABLE(NETSCAPE_PLUGIN_API)
75 pluginPackage = adoptNS([[WebNetscapePluginPackage alloc] initWithPath:pluginPath]);
76#else
77 return nil;
78#endif
79 }
80
81 return pluginPackage.autorelease();
82}

84- (id)initWithPath:(NSString *)pluginPath
85{
86 if (!(self = [super init]))
87 return nil;
88
89 path = [pluginPath stringByResolvingSymlinksInPath];
90 cfBundle = adoptCF(CFBundleCreate(kCFAllocatorDefault, (CFURLRef)[NSURL fileURLWithPath:path]));
91
92 if (!cfBundle) {
93 [self release];
94 return nil;
95 }
96
97 return self;
98}

134- (BOOL)getPluginInfoFromPLists
135{
136 if (!cfBundle)
137 return NO;
138
139 NSDictionary *MIMETypes = [self _objectForInfoDictionaryKey:WebPluginMIMETypesKey];
140 if (!MIMETypes)
141 return NO;
142
143 NSEnumerator *keyEnumerator = [MIMETypes keyEnumerator];
144 NSDictionary *MIMEDictionary;
145 NSString *MIME;
146
147 while ((MIME = [keyEnumerator nextObject]) != nil) {
148 MIMEDictionary = [MIMETypes objectForKey:MIME];
149
150 // FIXME: Consider storing disabled MIME types.
151 NSNumber *isEnabled = [MIMEDictionary objectForKey:WebPluginTypeEnabledKey];
152 if (isEnabled && [isEnabled boolValue] == NO)
153 continue;
154
155 WebCore::MimeClassInfo mimeClassInfo;
156
157 NSArray *extensions = [[MIMEDictionary objectForKey:WebPluginExtensionsKey] _web_lowercaseStrings];
158 for (NSString *extension in extensions) {
159 // The DivX plug-in lists multiple extensions in a comma separated string instead of using
160 // multiple array elements in the property list. Work around this here by splitting the
161 // extension string into components.
162 for (NSString *component in [extension componentsSeparatedByString:@","])
163 mimeClassInfo.extensions.append(component);
164 }
165
166 mimeClassInfo.type = String(MIME).convertToASCIILowercase();
167 mimeClassInfo.desc = [MIMEDictionary objectForKey:WebPluginTypeDescriptionKey];
168
169 pluginInfo.mimes.append(mimeClassInfo);
170 }
171
172 NSString *filename = [(NSString *)path lastPathComponent];
173 pluginInfo.file = filename;
174
175 NSString *theName = [self _objectForInfoDictionaryKey:WebPluginNameKey];
176 if (!theName)
177 theName = filename;
178 pluginInfo.name = theName;
179
180 NSString *description = [self _objectForInfoDictionaryKey:WebPluginDescriptionKey];
181 if (!description)
182 description = filename;
183 pluginInfo.desc = description;
184
185 pluginInfo.isApplicationPlugin = false;
186 pluginInfo.clientLoadPolicy = WebCore::PluginLoadClientPolicy::Undefined;
187#if PLATFORM(MAC)
188 pluginInfo.bundleIdentifier = self.bundleIdentifier;
189 pluginInfo.versionString = self.bundleVersion;
190#endif
191
192 return YES;
193}

269static inline void swapIntsInHeader(uint32_t* rawData, size_t length)
270{
271 for (size_t i = 0; i < length; ++i)
272 rawData[i] = OSSwapInt32(rawData[i]);
273}

275- (BOOL)isNativeLibraryData:(NSData *)data
276{
277 NSUInteger sizeInBytes = [data length];
278 Vector<uint32_t, 128> rawData((sizeInBytes + 3) / 4);
279 memcpy(rawData.data(), [data bytes], sizeInBytes);
280
281 unsigned numArchs = 0;
282 struct fat_arch singleArch = { 0, 0, 0, 0, 0 };
283 struct fat_arch* archs = 0;
284
285 if (sizeInBytes >= sizeof(struct mach_header_64)) {
286 uint32_t magic = *rawData.data();
287
288 if (magic == MH_MAGIC || magic == MH_CIGAM) {
289 // We have a 32-bit thin binary
290 struct mach_header* header = (struct mach_header*)rawData.data();
291
292 // Check if we need to swap the bytes
293 if (magic == MH_CIGAM)
294 swapIntsInHeader(rawData.data(), rawData.size());
295
296 singleArch.cputype = header->cputype;
297 singleArch.cpusubtype = header->cpusubtype;
298
299 archs = &singleArch;
300 numArchs = 1;
301 } else if (magic == MH_MAGIC_64 || magic == MH_CIGAM_64) {
302 // We have a 64-bit thin binary
303 struct mach_header_64* header = (struct mach_header_64*)rawData.data();
304
305 // Check if we need to swap the bytes
306 if (magic == MH_CIGAM_64)
307 swapIntsInHeader(rawData.data(), rawData.size());
308
309 singleArch.cputype = header->cputype;
310 singleArch.cpusubtype = header->cpusubtype;
311
312 archs = &singleArch;
313 numArchs = 1;
314 } else if (magic == FAT_MAGIC || magic == FAT_CIGAM) {
315 // We have a fat (universal) binary
316
317 // Check if we need to swap the bytes
318 if (magic == FAT_CIGAM)
319 swapIntsInHeader(rawData.data(), rawData.size());
320
321 COMPILE_ASSERT(sizeof(struct fat_header) % sizeof(uint32_t) == 0, struct_fat_header_must_be_integral_size_of_uint32_t);
322 archs = reinterpret_cast<struct fat_arch*>(rawData.data() + sizeof(struct fat_header) / sizeof(uint32_t));
323 numArchs = reinterpret_cast<struct fat_header*>(rawData.data())->nfat_arch;
324
325 unsigned maxArchs = (sizeInBytes - sizeof(struct fat_header)) / sizeof(struct fat_arch);
326 if (numArchs > maxArchs)
327 numArchs = maxArchs;
328 }
329 }
330
331 if (!archs || !numArchs)
332 return NO;
333
334 const NXArchInfo* localArch = NXGetLocalArchInfo();
335 if (!localArch)
336 return NO;
337
338 cpu_type_t cputype = localArch->cputype;
339 cpu_subtype_t cpusubtype = localArch->cpusubtype;
340
341#ifdef __x86_64__
342 // NXGetLocalArchInfo returns CPU_TYPE_X86 even when running in 64-bit.
343 // See <rdar://problem/4996965> for more information.
344 cputype = CPU_TYPE_X86_64;
345#endif
346
347 return NXFindBestFatArch(cputype, cpusubtype, archs, numArchs) != 0;
348}

378- (String)bundleVersion
379{
380 CFDictionaryRef infoDictionary = CFBundleGetInfoDictionary(cfBundle.get());
381 if (!infoDictionary)
382 return String();
383
384 CFTypeRef bundleVersionString = CFDictionaryGetValue(infoDictionary, kCFBundleVersionKey);
385 if (!bundleVersionString || CFGetTypeID(bundleVersionString) != CFStringGetTypeID())
386 return String();
387
388 return reinterpret_cast<CFStringRef>(bundleVersionString);
389}

395- (NSArray *)_web_lowercaseStrings
396{
397 NSMutableArray *lowercaseStrings = [NSMutableArray arrayWithCapacity:[self count]];
398 NSEnumerator *strings = [self objectEnumerator];
399 NSString *string;
400
401 while ((string = [strings nextObject]) != nil) {
402 if ([string isKindOfClass:[NSString class]])
403 [lowercaseStrings addObject:[string lowercaseString]];
404 }
405
406 return lowercaseStrings;
407}
