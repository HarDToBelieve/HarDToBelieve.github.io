64class PluginStopDeferrer {
65public:
66 PluginStopDeferrer(WebNetscapePluginView* pluginView)
67 : m_pluginView(pluginView)
68 {
69 ASSERT(m_pluginView);
70
71 [m_pluginView.get() willCallPlugInFunction];
72 }
73
74 ~PluginStopDeferrer()
75 {
76 ASSERT(m_pluginView);
77 [m_pluginView.get() didCallPlugInFunction];
78 }
79
80private:
81 RetainPtr<WebNetscapePluginView> m_pluginView;
82};

96NPReason WebNetscapePluginStream::reasonForError(NSError *error)
97{
98 if (!error)
99 return NPRES_DONE;
100
101 if ([[error domain] isEqualToString:NSURLErrorDomain] && [error code] == NSURLErrorCancelled)
102 return NPRES_USER_BREAK;
103
104 return NPRES_NETWORK_ERR;
105}

107NSError *WebNetscapePluginStream::pluginCancelledConnectionError() const
108{
109 return adoptNS([[NSError alloc] _initWithPluginErrorCode:WebKitErrorPlugInCancelledConnection
110 contentURL:m_responseURL ? m_responseURL.get() : (NSURL *)m_requestURL
111 pluginPageURL:nil
112 pluginName:[[m_pluginView.get() pluginPackage] pluginInfo].name
113 MIMEType:(NSString *)String::fromUTF8(m_mimeType.data(), m_mimeType.length())]).autorelease();

116NSError *WebNetscapePluginStream::errorForReason(NPReason reason) const
117{
118 if (reason == NPRES_DONE)
119 return nil;
120
121 if (reason == NPRES_USER_BREAK)
122 return [NSError _webKitErrorWithDomain:NSURLErrorDomain
123 code:NSURLErrorCancelled
124 URL:m_responseURL ? m_responseURL.get() : (NSURL *)m_requestURL];
125
126 return pluginCancelledConnectionError();
127}

147WebNetscapePluginStream::WebNetscapePluginStream(NSURLRequest *request, NPP plugin, bool sendNotification, void* notifyData)
148 : m_requestURL([request URL])
149 , m_plugin(0)
150 , m_transferMode(0)
151 , m_offset(0)
152 , m_fileDescriptor(-1)
153 , m_sendNotification(sendNotification)
154 , m_notifyData(notifyData)
155 , m_headers(0)
156 , m_reason(NPRES_BASE)
157 , m_isTerminated(false)
158 , m_newStreamSuccessful(false)
159 , m_frameLoader(0)
160 , m_request(adoptNS([request mutableCopy]))
161 , m_pluginFuncs(0)
162 , m_deliverDataTimer(*this, &WebNetscapePluginStream::deliverData)
163{
164 memset(&m_stream, 0, sizeof(NPStream));
165
166 WebNetscapePluginView *view = (WebNetscapePluginView *)plugin->ndata;
167
168 // This check has already been done by the plug-in view.
169 ASSERT(core([view webFrame])->document()->securityOrigin().canDisplay([request URL]));
170
171 ASSERT([request URL]);
172 ASSERT(plugin);
173
174 setPlugin(plugin);
175
176 streams().add(&m_stream, plugin);
177
178 String referrer = SecurityPolicy::generateReferrerHeader(core([view webFrame])->document()->referrerPolicy(), [request URL], core([view webFrame])->loader().outgoingReferrer());
179 if (referrer.isEmpty())
180 [m_request.get() _web_setHTTPReferrer:nil];
181 else
182 [m_request.get() _web_setHTTPReferrer:referrer];
183}

185WebNetscapePluginStream::~WebNetscapePluginStream()
186{
187 ASSERT(!m_plugin);
188 ASSERT(m_isTerminated);
189 ASSERT(!m_stream.ndata);
190
191 // The stream file should have been deleted, and the path freed, in -_destroyStream
192 ASSERT(!m_path);
193 ASSERT(m_fileDescriptor == -1);
194
195 free((void *)m_stream.url);
196 free(m_headers);
197
198 streams().remove(&m_stream);
199}

201void WebNetscapePluginStream::setPlugin(NPP plugin)
202{
203 if (plugin) {
204 m_plugin = plugin;
205 m_pluginView = static_cast<WebNetscapePluginView *>(m_plugin->ndata);
206
207 WebNetscapePluginPackage *pluginPackage = [m_pluginView.get() pluginPackage];
208
209 m_pluginFuncs = [pluginPackage pluginFuncs];
210 } else {
211 WebNetscapePluginView *view = m_pluginView.get();
212 m_plugin = 0;
213 m_pluginFuncs = 0;
214
215 [view disconnectStream:this];
216 m_pluginView = 0;
217 }

220void WebNetscapePluginStream::startStream(NSURL *url, long long expectedContentLength, NSDate *lastModifiedDate, const String& mimeType, NSData *headers)
221{
222 ASSERT(!m_isTerminated);
223
224 m_responseURL = url;
225 m_mimeType = mimeType.utf8();
226
227 free((void *)m_stream.url);
228 m_stream.url = strdup([m_responseURL.get() _web_URLCString]);
229
230 m_stream.ndata = this;
231 m_stream.end = expectedContentLength > 0 ? (uint32_t)expectedContentLength : 0;
232 m_stream.lastmodified = (uint32_t)[lastModifiedDate timeIntervalSince1970];
233 m_stream.notifyData = m_notifyData;
234
235 if (headers) {
236 unsigned len = [headers length];
237 m_headers = (char*) malloc(len + 1);
238 [headers getBytes:m_headers length:len];
239 m_headers[len] = 0;
240 m_stream.headers = m_headers;
241 }
242
243 m_transferMode = NP_NORMAL;
244 m_offset = 0;
245 m_reason = WEB_REASON_NONE;
246 // FIXME: If WebNetscapePluginStream called our initializer we wouldn't have to do this here.
247 m_fileDescriptor = -1;
248
249 // FIXME: Need a way to check if stream is seekable
250
251 NPError npErr;
252 {
253 PluginStopDeferrer deferrer(m_pluginView.get());
254 npErr = m_pluginFuncs->newstream(m_plugin, m_mimeType.mutableData(), &m_stream, NO, &m_transferMode);
255 }
256
257 LOG(Plugins, "NPP_NewStream URL=%@ MIME=%s error=%d", m_responseURL.get(), m_mimeType.data(), npErr);
258
259 if (npErr != NPERR_NO_ERROR) {
260 LOG_ERROR("NPP_NewStream failed with error: %d responseURL: %@", npErr, m_responseURL.get());
261 // Calling cancelLoadWithError: cancels the load, but doesn't call NPP_DestroyStream.
262 cancelLoadWithError(pluginCancelledConnectionError());
263 return;
264 }
265
266 m_newStreamSuccessful = true;
267
268 switch (m_transferMode) {
269 case NP_NORMAL:
270 LOG(Plugins, "Stream type: NP_NORMAL");
271 break;
272 case NP_ASFILEONLY:
273 LOG(Plugins, "Stream type: NP_ASFILEONLY");
274 break;
275 case NP_ASFILE:
276 LOG(Plugins, "Stream type: NP_ASFILE");
277 break;
278 case NP_SEEK:
279 LOG_ERROR("Stream type: NP_SEEK not yet supported");
280 cancelLoadAndDestroyStreamWithError(pluginCancelledConnectionError());
281 break;
282 default:
283 LOG_ERROR("unknown stream type");
284 }
285}

312void WebNetscapePluginStream::didReceiveResponse(NetscapePlugInStreamLoader*, const ResourceResponse& response)
313{
314 NSURLResponse *r = response.nsURLResponse();
315
316 NSMutableData *theHeaders = nil;
317 long long expectedContentLength = [r expectedContentLength];
318
319 if ([r isKindOfClass:[NSHTTPURLResponse class]]) {
320 NSHTTPURLResponse *httpResponse = (NSHTTPURLResponse *)r;
321 theHeaders = [NSMutableData dataWithCapacity:1024];
322
323 // FIXME: it would be nice to be able to get the raw HTTP header block.
324 // This includes the HTTP version, the real status text,
325 // all headers in their original order and including duplicates,
326 // and all original bytes verbatim, rather than sent through Unicode translation.
327 // Unfortunately NSHTTPURLResponse doesn't provide access at that low a level.
328
329 [theHeaders appendBytes:"HTTP " length:5];
330 char statusStr[10];
331 long statusCode = [httpResponse statusCode];
332 snprintf(statusStr, sizeof(statusStr), "%ld", statusCode);
333 [theHeaders appendBytes:statusStr length:strlen(statusStr)];
334 [theHeaders appendBytes:" OK\n" length:4];
335
336 // HACK: pass the headers through as UTF-8.
337 // This is not the intended behavior; we're supposed to pass original bytes verbatim.
338 // But we don't have the original bytes, we have NSStrings built by the URL loading system.
339 // It hopefully shouldn't matter, since RFC2616/RFC822 require ASCII-only headers,
340 // but surely someone out there is using non-ASCII characters, and hopefully UTF-8 is adequate here.
341 // It seems better than NSASCIIStringEncoding, which will lose information if non-ASCII is used.
342
343 NSDictionary *headerDict = [httpResponse allHeaderFields];
344 NSArray *keys = [[headerDict allKeys] sortedArrayUsingSelector:@selector(caseInsensitiveCompare:)];
345 NSEnumerator *i = [keys objectEnumerator];
346 NSString *k;
347 while ((k = [i nextObject]) != nil) {
348 NSString *v = [headerDict objectForKey:k];
349 [theHeaders appendData:[k dataUsingEncoding:NSUTF8StringEncoding]];
350 [theHeaders appendBytes:": " length:2];
351 [theHeaders appendData:[v dataUsingEncoding:NSUTF8StringEncoding]];
352 [theHeaders appendBytes:"\n" length:1];
353 }
354
355 // If the content is encoded (most likely compressed), then don't send its length to the plugin,
356 // which is only interested in the decoded length, not yet known at the moment.
357 // <rdar://problem/4470599> tracks a request for -[NSURLResponse expectedContentLength] to incorporate this logic.
358 NSString *contentEncoding = (NSString *)[[(NSHTTPURLResponse *)r allHeaderFields] objectForKey:@"Content-Encoding"];
359 if (contentEncoding && ![contentEncoding isEqualToString:@"identity"])
360 expectedContentLength = -1;
361
362 // startStreamResponseURL:... will null-terminate.
363 }
364
365 startStream([r URL], expectedContentLength, [r _lastModifiedDate], response.mimeType(), theHeaders);
366}

373bool WebNetscapePluginStream::wantsAllStreams() const
374{

376 return false;

391void WebNetscapePluginStream::destroyStream()
392{
393 if (m_isTerminated)
394 return;
395
396 Ref<WebNetscapePluginStream> protect(*this);
397
398 ASSERT(m_reason != WEB_REASON_NONE);
399 ASSERT([m_deliveryData.get() length] == 0);
400
401 m_deliverDataTimer.stop();
402
403 if (m_stream.ndata) {
404 if (m_reason == NPRES_DONE && (m_transferMode == NP_ASFILE || m_transferMode == NP_ASFILEONLY)) {
405 ASSERT(m_fileDescriptor == -1);
406 ASSERT(m_path);
407
408 PluginStopDeferrer deferrer(m_pluginView.get());
409 m_pluginFuncs->asfile(m_plugin, &m_stream, [m_path.get() fileSystemRepresentation]);
410 LOG(Plugins, "NPP_StreamAsFile responseURL=%@ path=%s", m_responseURL.get(), m_path.get());
411 }
412
413 if (m_path) {
414 // Delete the file after calling NPP_StreamAsFile(), instead of in -dealloc/-finalize. It should be OK
415 // to delete the file here -- NPP_StreamAsFile() is always called immediately before NPP_DestroyStream()
416 // (the stream destruction function), so there can be no expectation that a plugin will read the stream
417 // file asynchronously after NPP_StreamAsFile() is called.
418 unlink([m_path.get() fileSystemRepresentation]);
419 m_path = 0;
420
421 if (m_isTerminated)
422 return;
423 }
424
425 if (m_fileDescriptor != -1) {
426 // The file may still be open if we are destroying the stream before it completed loading.
427 close(m_fileDescriptor);
428 m_fileDescriptor = -1;
429 }
430
431 if (m_newStreamSuccessful) {
432 PluginStopDeferrer deferrer(m_pluginView.get());
433#if !LOG_DISABLED
434 NPError npErr =
435#endif
436 m_pluginFuncs->destroystream(m_plugin, &m_stream, m_reason);
437 LOG(Plugins, "NPP_DestroyStream responseURL=%@ error=%d", m_responseURL.get(), npErr);
438 }
439
440 free(m_headers);
441 m_headers = NULL;
442 m_stream.headers = NULL;
443
444 m_stream.ndata = 0;
445
446 if (m_isTerminated)
447 return;
448 }
449
450 if (m_sendNotification) {
451 // NPP_URLNotify expects the request URL, not the response URL.
452 PluginStopDeferrer deferrer(m_pluginView.get());
453 m_pluginFuncs->urlnotify(m_plugin, m_requestURL.string().utf8().data(), m_reason, m_notifyData);
454 LOG(Plugins, "NPP_URLNotify requestURL=%@ reason=%d", (NSURL *)m_requestURL, m_reason);
455 }
456
457 m_isTerminated = true;
458
459 setPlugin(0);
460}

462void WebNetscapePluginStream::destroyStreamWithReason(NPReason reason)
463{
464 m_reason = reason;
465 if (m_reason != NPRES_DONE) {
466 // Stop any pending data from being streamed.
467 [m_deliveryData.get() setLength:0];
468 } else if ([m_deliveryData.get() length] > 0) {
469 // There is more data to be streamed, don't destroy the stream now.
470 return;
471 }
472
473 Ref<WebNetscapePluginStream> protect(*this);
474 destroyStream();
475 ASSERT(!m_stream.ndata);
476}

509void WebNetscapePluginStream::deliverData()
510{
511 if (!m_stream.ndata || [m_deliveryData.get() length] == 0)
512 return;
513
514 Ref<WebNetscapePluginStream> protect(*this);
515
516 int32_t totalBytes = [m_deliveryData.get() length];
517 int32_t totalBytesDelivered = 0;
518
519 while (totalBytesDelivered < totalBytes) {
520 PluginStopDeferrer deferrer(m_pluginView.get());
521 int32_t deliveryBytes = m_pluginFuncs->writeready(m_plugin, &m_stream);
522 LOG(Plugins, "NPP_WriteReady responseURL=%@ bytes=%d", m_responseURL.get(), deliveryBytes);
523
524 if (m_isTerminated)
525 return;
526
527 if (deliveryBytes <= 0) {
528 // Plug-in can't receive anymore data right now. Send it later.
529 if (!m_deliverDataTimer.isActive())
530 m_deliverDataTimer.startOneShot(0_s);
531 break;
532 } else {
533 deliveryBytes = std::min(deliveryBytes, totalBytes - totalBytesDelivered);
534 NSData *subdata = [m_deliveryData.get() subdataWithRange:NSMakeRange(totalBytesDelivered, deliveryBytes)];
535 PluginStopDeferrer deferrer(m_pluginView.get());
536 deliveryBytes = m_pluginFuncs->write(m_plugin, &m_stream, m_offset, [subdata length], const_cast<void*>([subdata bytes]));
537 if (deliveryBytes < 0) {
538 // Netscape documentation says that a negative result from NPP_Write means cancel the load.
539 cancelLoadAndDestroyStreamWithError(pluginCancelledConnectionError());
540 return;
541 }
542 deliveryBytes = std::min<int32_t>(deliveryBytes, [subdata length]);
543 m_offset += deliveryBytes;
544 totalBytesDelivered += deliveryBytes;
545 LOG(Plugins, "NPP_Write responseURL=%@ bytes=%d total-delivered=%d/%d", m_responseURL.get(), deliveryBytes, m_offset, m_stream.end);
546 }
547 }
548
549 if (totalBytesDelivered > 0) {
550 if (totalBytesDelivered < totalBytes) {
551 auto newDeliveryData = adoptNS([[NSMutableData alloc] initWithCapacity:totalBytes - totalBytesDelivered]);
552 [newDeliveryData appendBytes:static_cast<char*>(const_cast<void*>([m_deliveryData.get() bytes])) + totalBytesDelivered length:totalBytes - totalBytesDelivered];
553
554 m_deliveryData = WTFMove(newDeliveryData);
555 } else {
556 [m_deliveryData.get() setLength:0];
557 if (m_reason != WEB_REASON_NONE)
558 destroyStream();
559 }
560 }
561}

563void WebNetscapePluginStream::deliverDataToFile(NSData *data)
564{
565 if (m_fileDescriptor == -1 && !m_path) {
566 NSString *temporaryFileMask = [NSTemporaryDirectory() stringByAppendingPathComponent:@"WebKitPlugInStreamXXXXXX"];
567 char *temporaryFileName = strdup([temporaryFileMask fileSystemRepresentation]);
568 m_fileDescriptor = mkstemp(temporaryFileName);
569 if (m_fileDescriptor == -1) {
570 LOG_ERROR("Can't create a temporary file.");
571 // This is not a network error, but the only error codes are "network error" and "user break".
572 destroyStreamWithReason(NPRES_NETWORK_ERR);
573 free(temporaryFileName);
574 return;
575 }
576
577 m_path = [NSString stringWithUTF8String:temporaryFileName];
578 free(temporaryFileName);
579 }
580
581 int dataLength = [data length];
582 if (!dataLength)
583 return;
584
585 int byteCount = write(m_fileDescriptor, [data bytes], dataLength);
586 if (byteCount != dataLength) {
587 // This happens only rarely, when we are out of disk space or have a disk I/O error.
588 LOG_ERROR("error writing to temporary file, errno %d", errno);
589 close(m_fileDescriptor);
590 m_fileDescriptor = -1;
591
592 // This is not a network error, but the only error codes are "network error" and "user break".
593 destroyStreamWithReason(NPRES_NETWORK_ERR);
594 m_path = 0;
595 }
596}

598void WebNetscapePluginStream::didFinishLoading(NetscapePlugInStreamLoader*)
599{
600 if (!m_stream.ndata)
601 return;
602
603 if (m_transferMode == NP_ASFILE || m_transferMode == NP_ASFILEONLY) {
604 // Fake the delivery of an empty data to ensure that the file has been created
605 deliverDataToFile([NSData data]);
606 if (m_fileDescriptor != -1)
607 close(m_fileDescriptor);
608 m_fileDescriptor = -1;
609 }
610
611 destroyStreamWithReason(NPRES_DONE);
612}

614void WebNetscapePluginStream::didReceiveData(NetscapePlugInStreamLoader*, const uint8_t* bytes, int length)
615{
616 auto data = adoptNS([[NSData alloc] initWithBytesNoCopy:(void*)bytes length:length freeWhenDone:NO]);
617
618 ASSERT([data length] > 0);
619
620 if (m_transferMode != NP_ASFILEONLY) {
621 if (!m_deliveryData)
622 m_deliveryData = adoptNS([[NSMutableData alloc] initWithCapacity:[data length]]);
623 [m_deliveryData.get() appendData:data.get()];
624 deliverData();
625 }
626 if (m_transferMode == NP_ASFILE || m_transferMode == NP_ASFILEONLY)
627 deliverDataToFile(data.get());
628}
