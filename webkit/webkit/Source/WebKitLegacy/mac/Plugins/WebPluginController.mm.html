151 return nil;
152 }
153

259- (void)startAllPlugins
260{
261 if (_started)
262 return;
263
264 if ([_views count] > 0)
265 LOG(Plugins, "starting WebKit plugins : %@", [_views description]);
266
267 int count = [_views count];
268 for (int i = 0; i < count; i++) {
269 id aView = [_views objectAtIndex:i];
270 if ([aView respondsToSelector:@selector(webPlugInStart)]) {
271 JSC::JSLock::DropAllLocks dropAllLocks(WebCore::commonVM());
272 [aView webPlugInStart];
273 } else if ([aView respondsToSelector:@selector(pluginStart)]) {
274 JSC::JSLock::DropAllLocks dropAllLocks(WebCore::commonVM());
275 [aView pluginStart];
276 }
277 }
278 _started = YES;
279}

326- (void)addPlugin:(NSView *)view
327{
328 if (!_documentView) {
329 LOG_ERROR("can't add a plug-in to a defunct WebPluginController");
330 return;
331 }
332
333 if (![_views containsObject:view]) {
334 [_views addObject:view];
335#if !PLATFORM(IOS_FAMILY)
336 [[_documentView _webView] addPluginInstanceView:view];
337#endif
338
339#if !PLATFORM(IOS_FAMILY)
340 BOOL oldDefersCallbacks = [[self webView] defersCallbacks];
341 if (!oldDefersCallbacks)
342 [[self webView] setDefersCallbacks:YES];
343
344 if (isKindOfClass(view, @"WmvPlugin"))
345 installFlip4MacPlugInWorkaroundIfNecessary();
346#endif
347
348 LOG(Plugins, "initializing plug-in %@", view);
349 if ([view respondsToSelector:@selector(webPlugInInitialize)]) {
350 JSC::JSLock::DropAllLocks dropAllLocks(WebCore::commonVM());
351 [view webPlugInInitialize];
352 } else if ([view respondsToSelector:@selector(pluginInitialize)]) {
353 JSC::JSLock::DropAllLocks dropAllLocks(WebCore::commonVM());
354 [view pluginInitialize];
355 }
356
357#if !PLATFORM(IOS_FAMILY)
358 if (!oldDefersCallbacks)
359 [[self webView] setDefersCallbacks:NO];
360#endif
361
362 if (_started) {
363 LOG(Plugins, "starting plug-in %@", view);
364 if ([view respondsToSelector:@selector(webPlugInStart)]) {
365 JSC::JSLock::DropAllLocks dropAllLocks(WebCore::commonVM());
366 [view webPlugInStart];
367 } else if ([view respondsToSelector:@selector(pluginStart)]) {
368 JSC::JSLock::DropAllLocks dropAllLocks(WebCore::commonVM());
369 [view pluginStart];
370 }
371
372 if ([view respondsToSelector:@selector(setContainingWindow:)]) {
373 JSC::JSLock::DropAllLocks dropAllLocks(WebCore::commonVM());
374 [view setContainingWindow:[_documentView window]];
375 }
376 }
377 }

473- (void)webPlugInContainerLoadRequest:(NSURLRequest *)request inFrame:(NSString *)target
474{
475 if (!request) {
476 LOG_ERROR("nil URL passed");
477 return;
478 }
479 if (!_documentView) {
480 LOG_ERROR("could not load URL %@ because plug-in has already been destroyed", request);
481 return;
482 }
483 WebFrame *frame = [_dataSource webFrame];
484 if (!frame) {
485 LOG_ERROR("could not load URL %@ because plug-in has already been stopped", request);
486 return;
487 }
488 if (!target) {
489 target = @"_top";
490 }
491 NSString *JSString = [[request URL] _webkit_scriptIfJavaScriptURL];
492 if (JSString) {
493 if ([frame findFrameNamed:target] != frame) {
494 LOG_ERROR("JavaScript requests can only be made on the frame that contains the plug-in");
495 return;
496 }
497 [frame _stringByEvaluatingJavaScriptFromString:JSString];
498 } else {
499 if (!request) {
500 LOG_ERROR("could not load URL %@", [request URL]);
501 return;
502 }
503 WebCore::FrameLoadRequest frameLoadRequest { *core(frame), request };
504 frameLoadRequest.setFrameName(target);
505 frameLoadRequest.setShouldCheckNewWindowPolicy(true);
506 core(frame)->loader().load(WTFMove(frameLoadRequest));
507 }
508}

620// Existing versions of the Flip4Mac WebKit plug-in have an object lifetime bug related to an NSAlert that is
621// used to notify the user about updates to the plug-in. This bug can result in Safari crashing if the page
622// containing the plug-in navigates while the alert is displayed (<rdar://problem/7313430>).
623//
624// The gist of the bug is thus: Flip4Mac sets an instance of the TSUpdateCheck class as the modal delegate of the
625// NSAlert instance. This TSUpdateCheck instance itself has a delegate. The delegate is set to the WmvPlugin
626// instance which is the NSView subclass that is exposed to WebKit as the plug-in view. Since this relationship
627// is that of delegates the TSUpdateCheck does not retain the WmvPlugin. This leads to a bug if the WmvPlugin
628// instance is destroyed before the TSUpdateCheck instance as the TSUpdateCheck instance will be left with a
629// pointer to a stale object. This will happen if a page containing the Flip4Mac plug-in triggers a navigation
630// while the update sheet is visible as the WmvPlugin instance is removed from the view hierarchy and there are
631// no other references to keep the object alive.
632//
633// We work around this bug by patching the following two messages:
634//
635// 1) -[NSAlert beginSheetModalForWindow:modalDelegate:didEndSelector:contextInfo:]
636// 2) -[TSUpdateCheck alertDidEnd:returnCode:contextInfo:]
637//
638// Our override of 1) detects whether it is Flip4Mac's update sheet triggering the alert by checking whether the
639// modal delegate is an instance of TSUpdateCheck. If it is, it retains the modal delegate's delegate.
640//
641// Our override of 2) then autoreleases the delegate, balancing the retain we added in 1).
642//
643// These two overrides have the effect of ensuring that the WmvPlugin instance will always outlive the TSUpdateCheck
644// instance, preventing the TSUpdateCheck instance from accessing a stale delegate pointer and crashing the application.
645
646
647typedef void (*beginSheetModalForWindowIMP)(id, SEL, NSWindow *, id, SEL, void*);
648static beginSheetModalForWindowIMP original_NSAlert_beginSheetModalForWindow_modalDelegate_didEndSelector_contextInfo_;
649
650typedef void (*alertDidEndIMP)(id, SEL, NSAlert *, NSInteger, void*);
651static alertDidEndIMP original_TSUpdateCheck_alertDidEnd_returnCode_contextInfo_;
652
653@class TSUpdateCheck;
654
655static void WebKit_TSUpdateCheck_alertDidEnd_returnCode_contextInfo_(id object, SEL selector, NSAlert *alert, NSInteger returnCode, void* contextInfo)
656{
657 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
658 [[(TSUpdateCheck *)object delegate] autorelease];
659 ALLOW_DEPRECATED_DECLARATIONS_END
660
661 original_TSUpdateCheck_alertDidEnd_returnCode_contextInfo_(object, selector, alert, returnCode, contextInfo);
662}
663
664static void WebKit_NSAlert_beginSheetModalForWindow_modalDelegate_didEndSelector_contextInfo_(id object, SEL selector, NSWindow *window, id modalDelegate, SEL didEndSelector, void* contextInfo)
665{
666 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
667 if (isKindOfClass(modalDelegate, @"TSUpdateCheck"))
668 [[(TSUpdateCheck *)modalDelegate delegate] retain];
669 ALLOW_DEPRECATED_DECLARATIONS_END
670
671 original_NSAlert_beginSheetModalForWindow_modalDelegate_didEndSelector_contextInfo_(object, selector, window, modalDelegate, didEndSelector, contextInfo);
672}

674static void installFlip4MacPlugInWorkaroundIfNecessary()
675{
676 static bool hasInstalledFlip4MacPlugInWorkaround;
677 if (!hasInstalledFlip4MacPlugInWorkaround) {
678 Class TSUpdateCheck = objc_lookUpClass("TSUpdateCheck");
679 if (!TSUpdateCheck)
680 return;
681
682IGNORE_WARNINGS_BEGIN("undeclared-selector")
683 Method methodToPatch = class_getInstanceMethod(TSUpdateCheck, @selector(alertDidEnd:returnCode:contextInfo:));
684IGNORE_WARNINGS_END
685 if (!methodToPatch)
686 return;
687
688 IMP originalMethod = method_setImplementation(methodToPatch, reinterpret_cast<IMP>(WebKit_TSUpdateCheck_alertDidEnd_returnCode_contextInfo_));
689 original_TSUpdateCheck_alertDidEnd_returnCode_contextInfo_ = reinterpret_cast<alertDidEndIMP>(originalMethod);
690
691 methodToPatch = class_getInstanceMethod(objc_getRequiredClass("NSAlert"), @selector(beginSheetModalForWindow:modalDelegate:didEndSelector:contextInfo:));
692 originalMethod = method_setImplementation(methodToPatch, reinterpret_cast<IMP>(WebKit_NSAlert_beginSheetModalForWindow_modalDelegate_didEndSelector_contextInfo_));
693 original_NSAlert_beginSheetModalForWindow_modalDelegate_didEndSelector_contextInfo_ = reinterpret_cast<beginSheetModalForWindowIMP>(originalMethod);
694
695 hasInstalledFlip4MacPlugInWorkaround = true;
696 }
