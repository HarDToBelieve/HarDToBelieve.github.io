71{
72 auto& database = sharedDatabase();
73 if (!database) {

92static void checkCandidate(WebBasePluginPackage * __strong *currentPlugin, WebBasePluginPackage * __strong *candidatePlugin)
93{
94 if (!*currentPlugin) {
95 *currentPlugin = *candidatePlugin;
96 return;
97 }
98
99 if ([*currentPlugin bundleIdentifier] == [*candidatePlugin bundleIdentifier] && [*candidatePlugin versionNumber] > [*currentPlugin versionNumber])
100 *currentPlugin = *candidatePlugin;
101}
102
103struct PluginPackageCandidates {
104 PluginPackageCandidates()
105 : webPlugin(nil)
106 , netscapePlugin(nil)
107 {
108 }
109
110 void update(WebBasePluginPackage *plugin)
111 {
112 if ([plugin isKindOfClass:[WebPluginPackage class]]) {
113 checkCandidate(&webPlugin, &plugin);
114 return;
115 }
116
117#if ENABLE(NETSCAPE_PLUGIN_API)
118 if([plugin isKindOfClass:[WebNetscapePluginPackage class]]) {
119 checkCandidate(&netscapePlugin, &plugin);
120 return;
121 }
122#endif
123 ASSERT_NOT_REACHED();
124 }
125
126 WebBasePluginPackage *bestCandidate()
127 {
128 // Allow other plug-ins to win over QT because if the user has installed a plug-in that can handle a type
129 // that the QT plug-in can handle, they probably intended to override QT.
130 if (webPlugin && ![webPlugin isQuickTimePlugIn])
131 return webPlugin;
132
133 if (netscapePlugin && ![netscapePlugin isQuickTimePlugIn])
134 return netscapePlugin;
135
136 if (webPlugin)
137 return webPlugin;
138 if (netscapePlugin)
139 return netscapePlugin;
140
141 return nil;
142 }
143
144 WebBasePluginPackage *webPlugin;
145 WebBasePluginPackage *netscapePlugin;
146};

148- (WebBasePluginPackage *)pluginForMIMEType:(NSString *)MIMEType
149{
150 PluginPackageCandidates candidates;
151
152 MIMEType = [MIMEType lowercaseString];
153 NSEnumerator *pluginEnumerator = [plugins objectEnumerator];
154
155 while (WebBasePluginPackage *plugin = [pluginEnumerator nextObject]) {
156 if ([plugin supportsMIMEType:MIMEType])
157 candidates.update(plugin);
158 }
159
160 return candidates.bestCandidate();
161}

163- (WebBasePluginPackage *)pluginForExtension:(NSString *)extension
164{
165 PluginPackageCandidates candidates;
166
167 extension = [extension lowercaseString];
168 NSEnumerator *pluginEnumerator = [plugins objectEnumerator];
169
170 while (WebBasePluginPackage *plugin = [pluginEnumerator nextObject]) {
171 if ([plugin supportsExtension:extension])
172 candidates.update(plugin);
173 }
174
175 WebBasePluginPackage *plugin = candidates.bestCandidate();
176
177 if (!plugin) {
178 // If no plug-in was found from the extension, attempt to map from the extension to a MIME type
179 // and find the a plug-in from the MIME type. This is done in case the plug-in has not fully specified
180 // an extension <-> MIME type mapping.
181 NSString *MIMEType = [[NSURLFileTypeMappings sharedMappings] MIMETypeForExtension:extension];
182 if ([MIMEType length] > 0)
183 plugin = [self pluginForMIMEType:MIMEType];
184 }
185 return plugin;
186}

199+ (void)setAdditionalWebPlugInPaths:(NSArray *)additionalPaths
200{
201 if (additionalPaths == additionalWebPlugInPaths())
202 return;
203
204 additionalWebPlugInPaths() = adoptNS([additionalPaths copy]);
205
206 // One might be tempted to add additionalWebPlugInPaths to the global WebPluginDatabase here.
207 // For backward compatibility with earlier versions of the +setAdditionalWebPlugInPaths: SPI,
208 // we need to save a copy of the additional paths and not cause a refresh of the plugin DB
209 // at this time.
210 // See Radars 4608487 and 4609047.
211}
212
213- (void)setPlugInPaths:(NSArray *)newPaths
214{
215 if (plugInPaths == newPaths)
216 return;
217
218 [plugInPaths release];
219 plugInPaths = [newPaths copy];
220}

222- (void)close
223{
224 NSEnumerator *pluginEnumerator = [[self plugins] objectEnumerator];
225 WebBasePluginPackage *plugin;
226 while ((plugin = [pluginEnumerator nextObject]) != nil)
227 [self _removePlugin:plugin];
228 [plugins release];
229 plugins = nil;
230}

253- (void)refresh
254{
255 // This method does a bit of autoreleasing, so create an autorelease pool to ensure that calling
256 // -refresh multiple times does not bloat the default pool.
257 @autoreleasepool {
258 // Create map from plug-in path to WebBasePluginPackage
259 if (!plugins)
260 plugins = [[NSMutableDictionary alloc] initWithCapacity:12];
261
262 // Find all plug-ins on disk
263 NSMutableSet *newPlugins = [self _scanForNewPlugins];
264
265 // Find plug-ins to remove from database (i.e., plug-ins that no longer exist on disk)
266 NSMutableSet *pluginsToRemove = [NSMutableSet set];
267 NSEnumerator *pluginEnumerator = [plugins objectEnumerator];
268 WebBasePluginPackage *plugin;
269 while ((plugin = [pluginEnumerator nextObject]) != nil) {
270 // Any plug-ins that were removed from disk since the last refresh should be removed from
271 // the database.
272 if (![newPlugins containsObject:plugin])
273 [pluginsToRemove addObject:plugin];
274
275 // Remove every member of 'plugins' from 'newPlugins'. After this loop exits, 'newPlugins'
276 // will be the set of new plug-ins that should be added to the database.
277 [newPlugins removeObject:plugin];
278 }
279
280#if !LOG_DISABLED
281 if ([newPlugins count] > 0)
282 LOG(Plugins, "New plugins:\n%@", newPlugins);
283 if ([pluginsToRemove count] > 0)
284 LOG(Plugins, "Removed plugins:\n%@", pluginsToRemove);
285#endif
286
287 // Remove plugins from database
288 pluginEnumerator = [pluginsToRemove objectEnumerator];
289 while ((plugin = [pluginEnumerator nextObject]) != nil)
290 [self _removePlugin:plugin];
291
292 // Add new plugins to database
293 pluginEnumerator = [newPlugins objectEnumerator];
294 while ((plugin = [pluginEnumerator nextObject]) != nil)
295 [self _addPlugin:plugin];
296
297 // Build a list of MIME types.
298 auto MIMETypes = adoptNS([[NSMutableSet alloc] init]);
299 pluginEnumerator = [plugins objectEnumerator];
300 while ((plugin = [pluginEnumerator nextObject])) {
301 const auto& pluginInfo = [plugin pluginInfo];
302 for (size_t i = 0; i < pluginInfo.mimes.size(); ++i)
303 [MIMETypes addObject:pluginInfo.mimes[i].type];
304 }
305
306 // Register plug-in views and representations.
307 NSEnumerator *MIMEEnumerator = [MIMETypes objectEnumerator];
308 NSString *MIMEType;
309 while ((MIMEType = [MIMEEnumerator nextObject]) != nil) {
310 [registeredMIMETypes addObject:MIMEType];
311
312 if ([WebView canShowMIMETypeAsHTML:MIMEType]) {
313 // Don't allow plug-ins to override our core HTML types.
314 continue;
315 }
316 plugin = [self pluginForMIMEType:MIMEType];
317 if ([plugin isJavaPlugIn]) {
318 // Don't register the Java plug-in for a document view since Java files should be downloaded when not embedded.
319 continue;
320 }
321 if ([plugin isQuickTimePlugIn] && [[WebFrameView _viewTypesAllowImageTypeOmission:NO] objectForKey:MIMEType]) {
322 // Don't allow the QT plug-in to override any types because it claims many that we can handle ourselves.
323 continue;
324 }
325
326 if (self == sharedDatabase())
327 [WebView _registerPluginMIMEType:MIMEType];
328 }
329 }
330}

347- (void)removePluginInstanceViewsFor:(WebFrame*)webFrame
348{
349 // This handles handles the case where a frame or view is being destroyed and the plugin needs to be removed from the list first
350
351 if( [pluginInstanceViews count] == 0 )
352 return;
353
354 NSView <WebDocumentView> *documentView = [[webFrame frameView] documentView];
355 if ([documentView isKindOfClass:[WebHTMLView class]]) {
356 for (NSView *subview in [documentView subviews]) {
357#if ENABLE(NETSCAPE_PLUGIN_API)
358 if ([subview isKindOfClass:[WebBaseNetscapePluginView class]] || [WebPluginController isPlugInView:subview])
359#else
360 if ([WebPluginController isPlugInView:subview])
361#endif
362 [pluginInstanceViews removeObject:subview];
363 }
364 }
365}

367- (void)destroyAllPluginInstanceViews
368{
369 NSView *view;
370 NSArray *pli = [pluginInstanceViews allObjects];
371 NSEnumerator *enumerator = [pli objectEnumerator];
372 while ((view = [enumerator nextObject]) != nil) {
373#if ENABLE(NETSCAPE_PLUGIN_API)
374 if ([view isKindOfClass:[WebBaseNetscapePluginView class]]) {
375 ASSERT([view respondsToSelector:@selector(stop)]);
376 [view performSelector:@selector(stop)];
377 } else
378#endif
379 if ([WebPluginController isPlugInView:view]) {
380 ASSERT([[view superview] isKindOfClass:[WebHTMLView class]]);
381 ASSERT([[view superview] respondsToSelector:@selector(_destroyAllWebPlugins)]);
382 // this will actually destroy all plugin instances for a webHTMLView and remove them from this list
383 [[view superview] performSelector:@selector(_destroyAllWebPlugins)];
384 }
385 }
386}

392+ (NSArray *)_defaultPlugInPaths
393{
394#if !PLATFORM(IOS_FAMILY)
395 // Plug-ins are found in order of precedence.
396 // If there are duplicates, the first found plug-in is used.
397 // For example, if there is a QuickTime.plugin in the users's home directory
398 // that is used instead of the /Library/Internet Plug-ins version.
399 // The purpose is to allow non-admin users to update their plug-ins.
400 return @[
401 [NSHomeDirectory() stringByAppendingPathComponent:@"Library/Internet Plug-Ins"],
402 @"/Library/Internet Plug-Ins",
403 [[NSBundle mainBundle] builtInPlugInsPath],
404 ];
405#else
406 // iOS plug-ins are all located in /System/Library/Internet Plug-Ins
407#if !PLATFORM(IOS_FAMILY_SIMULATOR)
408 NSArray *systemLibrary = NSSearchPathForDirectoriesInDomains(NSLibraryDirectory, NSSystemDomainMask, YES);
409 if (!systemLibrary || [systemLibrary count] == 0)
410 return nil;
411 NSString *systemDir = (NSString*)[systemLibrary objectAtIndex:0];
412#else
413 NSString* platformRootDir = [NSString stringWithUTF8String:WebKitPlatformSystemRootDirectory()];
414 NSString *systemDir = [platformRootDir stringByAppendingPathComponent:@"System/Library"];
415#endif
416 return @[[systemDir stringByAppendingPathComponent:@"Internet Plug-Ins"]];
417#endif
418}

420- (NSArray *)_plugInPaths
421{
422 if (self == sharedDatabase() && additionalWebPlugInPaths()) {
423 // Add additionalWebPlugInPaths to the global WebPluginDatabase. We do this here for
424 // backward compatibility with earlier versions of the +setAdditionalWebPlugInPaths: SPI,
425 // which simply saved a copy of the additional paths and did not cause the plugin DB to
426 // refresh. See Radars 4608487 and 4609047.
427 auto modifiedPlugInPaths = adoptNS([plugInPaths mutableCopy]);
428 [modifiedPlugInPaths addObjectsFromArray:additionalWebPlugInPaths().get()];
429 return modifiedPlugInPaths.autorelease();
430 }
431 return plugInPaths;
432}

434- (void)_addPlugin:(WebBasePluginPackage *)plugin
435{
436 ASSERT(plugin);
437 NSString *pluginPath = [plugin path];
438 ASSERT(pluginPath);
439 [plugins setObject:plugin forKey:pluginPath];
440 [plugin wasAddedToPluginDatabase:self];
441}
442
443- (void)_removePlugin:(WebBasePluginPackage *)plugin
444{
445 ASSERT(plugin);
446
447 // Unregister plug-in's MIME type registrations
448 const auto& pluginInfo = [plugin pluginInfo];
449 for (size_t i = 0; i < pluginInfo.mimes.size(); ++i) {
450 NSString *MIMEType = pluginInfo.mimes[i].type;
451
452 if ([registeredMIMETypes containsObject:MIMEType]) {
453 if (self == sharedDatabase())
454 [WebView _unregisterPluginMIMEType:MIMEType];
455 [registeredMIMETypes removeObject:MIMEType];
456 }
457 }
458
459 // Remove plug-in from database
460 NSString *pluginPath = [plugin path];
461 ASSERT(pluginPath);
462 auto protectedPlugin = retainPtr(plugin);
463 [plugins removeObjectForKey:pluginPath];
464 [plugin wasRemovedFromPluginDatabase:self];
465}

467- (NSMutableSet *)_scanForNewPlugins
468{
469 NSMutableSet *newPlugins = [NSMutableSet set];
470 NSEnumerator *directoryEnumerator = [[self _plugInPaths] objectEnumerator];
471 auto uniqueFilenames = adoptNS([[NSMutableSet alloc] init]);
472 NSFileManager *fileManager = [NSFileManager defaultManager];
473 NSString *pluginDirectory;
474 while ((pluginDirectory = [directoryEnumerator nextObject]) != nil) {
475 // Get contents of each plug-in directory
476 NSEnumerator *filenameEnumerator = [[fileManager contentsOfDirectoryAtPath:pluginDirectory error:NULL] objectEnumerator];
477 NSString *filename;
478 while ((filename = [filenameEnumerator nextObject]) != nil) {
479 // Unique plug-ins by filename
480 if ([uniqueFilenames containsObject:filename])
481 continue;
482 [uniqueFilenames addObject:filename];
483
484 // Create a plug-in package for this path
485 NSString *pluginPath = [pluginDirectory stringByAppendingPathComponent:filename];
486 WebBasePluginPackage *pluginPackage = [plugins objectForKey:pluginPath];
487 if (!pluginPackage)
488 pluginPackage = [WebBasePluginPackage pluginWithPath:pluginPath];
489 if (pluginPackage)
490 [newPlugins addObject:pluginPackage];
491 }
492 }
493
494 return newPlugins;
495}
