48{
49 static WebDatabaseManagerClient* sharedClient = new WebDatabaseManagerClient();

53#if PLATFORM(IOS_FAMILY)
54static void onNewDatabaseOriginAdded(CFNotificationCenterRef, void* observer, CFStringRef, const void*, CFDictionaryRef)
55{
56 ASSERT(observer);
57
58 WebDatabaseManagerClient* client = reinterpret_cast<WebDatabaseManagerClient*>(observer);
59 client->newDatabaseOriginWasAdded();
60}
61
62static void onDatabaseDeleted(CFNotificationCenterRef, void* observer, CFStringRef, const void*, CFDictionaryRef)
63{
64 ASSERT(observer);
65
66 WebDatabaseManagerClient* client = reinterpret_cast<WebDatabaseManagerClient*>(observer);
67 client->databaseWasDeleted();
68}
69
70static void onDatabaseOriginDeleted(CFNotificationCenterRef, void* observer, CFStringRef, const void*, CFDictionaryRef)
71{
72 ASSERT(observer);
73
74 WebDatabaseManagerClient* client = reinterpret_cast<WebDatabaseManagerClient*>(observer);
75 client->databaseOriginWasDeleted();
76}
77#endif

79WebDatabaseManagerClient::WebDatabaseManagerClient()
80{
81#if PLATFORM(IOS_FAMILY)
82 CFNotificationCenterRef center = CFNotificationCenterGetDarwinNotifyCenter();
83 CFNotificationCenterAddObserver(center, this, onNewDatabaseOriginAdded, WebDatabaseOriginWasAddedNotification, 0, CFNotificationSuspensionBehaviorDeliverImmediately);
84 CFNotificationCenterAddObserver(center, this, onDatabaseDeleted, WebDatabaseWasDeletedNotification, 0, CFNotificationSuspensionBehaviorDeliverImmediately);
85 CFNotificationCenterAddObserver(center, this, onDatabaseOriginDeleted, WebDatabaseOriginWasDeletedNotification, 0, CFNotificationSuspensionBehaviorDeliverImmediately);
86#endif
87}
88
89WebDatabaseManagerClient::~WebDatabaseManagerClient()
90{
91#if PLATFORM(IOS_FAMILY)
92 CFNotificationCenterRemoveObserver(CFNotificationCenterGetDarwinNotifyCenter(), this, 0, 0);
93#endif
94}

96class DidModifyOriginData {
97 WTF_MAKE_FAST_ALLOCATED;
98public:
99 static void dispatchToMainThread(WebDatabaseManagerClient& client, const SecurityOriginData& origin)
100 {
101 auto context = makeUnique<DidModifyOriginData>(client, origin);
102 callOnMainThread([context = WTFMove(context)] {
103 context->client.dispatchDidModifyOrigin(context->origin);
104 });
105 }
106
107 DidModifyOriginData(WebDatabaseManagerClient& client, const SecurityOriginData& origin)
108 : client(client)
109 , origin(origin.isolatedCopy())
110 {
111 }
112
113private:
114 WebDatabaseManagerClient& client;
115 SecurityOriginData origin;
116};
117
118void WebDatabaseManagerClient::dispatchDidModifyOrigin(const SecurityOriginData& origin)
119{
120 if (!isMainThread()) {
121 DidModifyOriginData::dispatchToMainThread(*this, origin);
122 return;
123 }
124
125 auto webSecurityOrigin = adoptNS([[WebSecurityOrigin alloc] _initWithWebCoreSecurityOrigin:origin.securityOrigin().ptr()]);
126
127 [[NSNotificationCenter defaultCenter] postNotificationName:WebDatabaseDidModifyOriginNotification object:webSecurityOrigin.get()];
128}
129
130void WebDatabaseManagerClient::dispatchDidModifyDatabase(const SecurityOriginData& origin, const String& databaseIdentifier)
131{
132 if (!isMainThread()) {
133 DidModifyOriginData::dispatchToMainThread(*this, origin);
134 return;
135 }
136
137 auto webSecurityOrigin = adoptNS([[WebSecurityOrigin alloc] _initWithWebCoreSecurityOrigin:origin.securityOrigin().ptr()]);
138 auto userInfo = adoptNS([[NSDictionary alloc] initWithObjectsAndKeys:(NSString *)databaseIdentifier, WebDatabaseIdentifierKey, nil]);
139 [[NSNotificationCenter defaultCenter] postNotificationName:WebDatabaseDidModifyDatabaseNotification object:webSecurityOrigin.get() userInfo:userInfo.get()];
140}

144void WebDatabaseManagerClient::dispatchDidAddNewOrigin()
145{
146 m_isHandlingNewDatabaseOriginNotification = true;
147 // Send a notification to all apps that a new origin has been added, so other apps with opened database can refresh their origin maps.
148 CFNotificationCenterPostNotification(CFNotificationCenterGetDarwinNotifyCenter(), WebDatabaseOriginWasAddedNotification, nullptr, nullptr, true);
149}
150
151void WebDatabaseManagerClient::dispatchDidDeleteDatabase()
152{
153 m_isHandlingDeleteDatabaseNotification = true;
154 // Send a notification to all apps that a database has been deleted, so other apps with the deleted database open will close it properly.
155 CFNotificationCenterPostNotification(CFNotificationCenterGetDarwinNotifyCenter(), WebDatabaseWasDeletedNotification, nullptr, nullptr, true);
156}
157
158void WebDatabaseManagerClient::dispatchDidDeleteDatabaseOrigin()
159{
160 m_isHandlingDeleteDatabaseOriginNotification = true;
161 // Send a notification to all apps that an origin has been deleted, so other apps can update their origin maps.
162 CFNotificationCenterPostNotification(CFNotificationCenterGetDarwinNotifyCenter(), WebDatabaseOriginWasDeletedNotification, nullptr, nullptr, true);
163}

165void WebDatabaseManagerClient::newDatabaseOriginWasAdded()
166{
167 // Locks the WebThread for the rest of the run loop.
168 WebThreadLock();
169
170 // If this is the process that added the new origin, its quota map should have been updated
171 // and does not need to be invalidated.
172 if (m_isHandlingNewDatabaseOriginNotification) {
173 m_isHandlingNewDatabaseOriginNotification = false;
174 return;
175 }
176
177 databaseOriginsDidChange();
178}
179
180void WebDatabaseManagerClient::databaseWasDeleted()
181{
182 // Locks the WebThread for the rest of the run loop.
183 WebThreadLock();
184
185 // If this is the process that added the new origin, its quota map should have been updated
186 // and does not need to be invalidated.
187 if (m_isHandlingDeleteDatabaseNotification) {
188 m_isHandlingDeleteDatabaseNotification = false;
189 return;
190 }
191
192 DatabaseTracker::singleton().removeDeletedOpenedDatabases();
193}
194
195void WebDatabaseManagerClient::databaseOriginWasDeleted()
196{
197 // Locks the WebThread for the rest of the run loop.
198 WebThreadLock();
199
200 // If this is the process that added the new origin, its quota map should have been updated
201 // and does not need to be invalidated.
202 if (m_isHandlingDeleteDatabaseOriginNotification) {
203 m_isHandlingDeleteDatabaseOriginNotification = false;
204 return;
205 }
206
207 databaseOriginsDidChange();
208}
209
210void WebDatabaseManagerClient::databaseOriginsDidChange()
211{
212 // Send a notification (within the app) about the origins change. If an app needs to update its UI when the origins change
213 // (such as Safari Settings), it can listen for that notification.
214 CFNotificationCenterPostNotification(CFNotificationCenterGetLocalCenter(), WebDatabaseOriginsDidChangeNotification, 0, 0, true);
215}
