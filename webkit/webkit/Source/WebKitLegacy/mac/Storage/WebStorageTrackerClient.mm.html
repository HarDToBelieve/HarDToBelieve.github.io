52void WebStorageTrackerClient::dispatchDidModifyOrigin(SecurityOrigin* origin)
53{
54 RetainPtr<WebSecurityOrigin> webSecurityOrigin = adoptNS([[WebSecurityOrigin alloc] _initWithWebCoreSecurityOrigin:origin]);
55
56 [[NSNotificationCenter defaultCenter] postNotificationName:WebStorageDidModifyOriginNotification
57 object:webSecurityOrigin.get()];
58}
59
60void WebStorageTrackerClient::dispatchDidModifyOrigin(const String& originIdentifier)
61{
62 auto origin = SecurityOriginData::fromDatabaseIdentifier(originIdentifier);
63
64 if (!origin) {
65 ASSERT_NOT_REACHED();
66 return;
67 }
68
69 if (isMainThread()) {
70 dispatchDidModifyOrigin(origin->securityOrigin().ptr());
71 return;
72 }
73
74 callOnMainThread([origin = origin->securityOrigin()->isolatedCopy()]() mutable {
75 WebStorageTrackerClient::sharedWebStorageTrackerClient()->dispatchDidModifyOrigin(origin.ptr());
76 });
77}
