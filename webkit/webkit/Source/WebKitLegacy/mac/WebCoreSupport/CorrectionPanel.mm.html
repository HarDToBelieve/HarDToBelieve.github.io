35static inline NSCorrectionIndicatorType correctionIndicatorType(AlternativeTextType alternativeTextType)
36{
37 switch (alternativeTextType) {
38 case AlternativeTextTypeCorrection:
39 return NSCorrectionIndicatorTypeDefault;
40 case AlternativeTextTypeReversion:
41 return NSCorrectionIndicatorTypeReversion;
42 case AlternativeTextTypeSpellingSuggestions:
43 return NSCorrectionIndicatorTypeGuesses;
44 case AlternativeTextTypeDictationAlternatives:
45 ASSERT_NOT_REACHED();
46 break;
47 }
48 ASSERT_NOT_REACHED();
49 return NSCorrectionIndicatorTypeDefault;
50}

52CorrectionPanel::CorrectionPanel()
53 : m_wasDismissedExternally(false)
54 , m_reasonForDismissing(ReasonForDismissingAlternativeTextIgnored)
55{
56}
57
58CorrectionPanel::~CorrectionPanel()
59{
60 dismissInternal(ReasonForDismissingAlternativeTextIgnored, false);
61}
62
63void CorrectionPanel::show(WebView* view, AlternativeTextType type, const FloatRect& boundingBoxOfReplacedString, const String& replacedString, const String& replacementString, const Vector<String>& alternativeReplacementStrings)
64{
65 dismissInternal(ReasonForDismissingAlternativeTextIgnored, false);
66
67 if (!view)
68 return;
69
70 NSString* replacedStringAsNSString = replacedString;
71 NSString* replacementStringAsNSString = replacementString;
72 m_view = view;
73 NSCorrectionIndicatorType indicatorType = correctionIndicatorType(type);
74
75 RetainPtr<NSArray> alternativeStrings;
76 if (!alternativeReplacementStrings.isEmpty())
77 alternativeStrings = createNSArray(alternativeReplacementStrings);
78
79 [[NSSpellChecker sharedSpellChecker] showCorrectionIndicatorOfType:indicatorType primaryString:replacementStringAsNSString alternativeStrings:alternativeStrings.get() forStringInRect:[view _convertRectFromRootView:boundingBoxOfReplacedString] view:m_view.get() completionHandler:^(NSString* acceptedString) {
80 handleAcceptedReplacement(acceptedString, replacedStringAsNSString, replacementStringAsNSString, indicatorType);
81 }];
82}
83
84String CorrectionPanel::dismiss(ReasonForDismissingAlternativeText reason)
85{
86 return dismissInternal(reason, true);
87}
88
89String CorrectionPanel::dismissInternal(ReasonForDismissingAlternativeText reason, bool dismissingExternally)
90{
91 if (!isShowing())
92 return String();
93
94 m_wasDismissedExternally = dismissingExternally;
95 m_reasonForDismissing = reason;
96 m_resultForDismissal.clear();
97 [[NSSpellChecker sharedSpellChecker] dismissCorrectionIndicatorForView:m_view.get()];
98 return m_resultForDismissal.get();
99}

101void CorrectionPanel::recordAutocorrectionResponse(NSInteger spellCheckerDocumentTag, NSCorrectionResponse response, const String& replacedString, const String& replacementString)
102{
103 [[NSSpellChecker sharedSpellChecker] recordResponse:response toCorrection:replacementString forWord:replacedString language:nil inSpellDocumentWithTag:spellCheckerDocumentTag];
104}
105
106void CorrectionPanel::handleAcceptedReplacement(NSString* acceptedReplacement, NSString* replaced, NSString* proposedReplacement, NSCorrectionIndicatorType correctionIndicatorType)
107{
108 if (!m_view)
109 return;
110
111 NSInteger documentTag = [m_view.get() spellCheckerDocumentTag];
112
113 switch (correctionIndicatorType) {
114 case NSCorrectionIndicatorTypeDefault:
115 if (acceptedReplacement)
116 recordAutocorrectionResponse(documentTag, NSCorrectionResponseAccepted, replaced, acceptedReplacement);
117 else {
118 if (!m_wasDismissedExternally || m_reasonForDismissing == ReasonForDismissingAlternativeTextCancelled)
119 recordAutocorrectionResponse(documentTag, NSCorrectionResponseRejected, replaced, proposedReplacement);
120 else
121 recordAutocorrectionResponse(documentTag, NSCorrectionResponseIgnored, replaced, proposedReplacement);
122 }
123 break;
124 case NSCorrectionIndicatorTypeReversion:
125 if (acceptedReplacement)
126 recordAutocorrectionResponse(documentTag, NSCorrectionResponseReverted, replaced, acceptedReplacement);
127 break;
128 case NSCorrectionIndicatorTypeGuesses:
129 if (acceptedReplacement)
130 recordAutocorrectionResponse(documentTag, NSCorrectionResponseAccepted, replaced, acceptedReplacement);
131 break;
132 }
133
134 [m_view.get() handleAcceptedAlternativeText:acceptedReplacement];
135 m_view.clear();
136 if (acceptedReplacement)
137 m_resultForDismissal = adoptNS([acceptedReplacement copy]);
138}
