57void PopupMenuMac::populate()
58{
59 if (m_popup)
60 clear();
61 else {
62 m_popup = adoptNS([[NSPopUpButtonCell alloc] initTextCell:@"" pullsDown:!m_client->shouldPopOver()]);
63 [m_popup setUsesItemFromMenu:NO];
64 [m_popup setAutoenablesItems:NO];
65 }
66
67 // For pullDown menus the first item is hidden.
68 if (!m_client->shouldPopOver())
69 [m_popup addItemWithTitle:@""];
70
71 TextDirection menuTextDirection = m_client->menuStyle().textDirection();
72 [m_popup setUserInterfaceLayoutDirection:menuTextDirection == TextDirection::LTR ? NSUserInterfaceLayoutDirectionLeftToRight : NSUserInterfaceLayoutDirectionRightToLeft];
73
74 ASSERT(m_client);
75 int size = m_client->listSize();
76
77 for (int i = 0; i < size; i++) {
78 if (m_client->itemIsSeparator(i)) {
79 [[m_popup menu] addItem:[NSMenuItem separatorItem]];
80 continue;
81 }
82
83 PopupMenuStyle style = m_client->itemStyle(i);
84 RetainPtr<NSMutableDictionary> attributes = adoptNS([[NSMutableDictionary alloc] init]);
85 if (style.font() != FontCascade()) {
86 RetainPtr<CTFontRef> font = style.font().primaryFont().getCTFont();
87 if (!font) {
88 CGFloat size = style.font().primaryFont().platformData().size();
89 font = adoptCF(CTFontCreateUIFontForLanguage(isFontWeightBold(style.font().weight()) ? kCTFontUIFontEmphasizedSystem : kCTFontUIFontSystem, size, nullptr));
90 }
91 [attributes setObject:toNSFont(font.get()) forKey:NSFontAttributeName];
92 }
93
94 RetainPtr<NSMutableParagraphStyle> paragraphStyle = adoptNS([[NSParagraphStyle defaultParagraphStyle] mutableCopy]);
95 [paragraphStyle setAlignment:menuTextDirection == TextDirection::LTR ? NSTextAlignmentLeft : NSTextAlignmentRight];
96 NSWritingDirection writingDirection = style.textDirection() == TextDirection::LTR ? NSWritingDirectionLeftToRight : NSWritingDirectionRightToLeft;
97 [paragraphStyle setBaseWritingDirection:writingDirection];
98 if (style.hasTextDirectionOverride()) {
99 RetainPtr<NSNumber> writingDirectionValue = adoptNS([[NSNumber alloc] initWithInteger:writingDirection + NSWritingDirectionOverride]);
100 RetainPtr<NSArray> writingDirectionArray = adoptNS([[NSArray alloc] initWithObjects:writingDirectionValue.get(), nil]);
101 [attributes setObject:writingDirectionArray.get() forKey:NSWritingDirectionAttributeName];
102 }
103 [attributes setObject:paragraphStyle.get() forKey:NSParagraphStyleAttributeName];
104
105 // FIXME: Add support for styling the foreground and background colors.
106 // FIXME: Find a way to customize text color when an item is highlighted.
107 RetainPtr<NSAttributedString> string = adoptNS([[NSAttributedString alloc] initWithString:m_client->itemText(i) attributes:attributes.get()]);
108
109 [m_popup addItemWithTitle:@""];
110 NSMenuItem *menuItem = [m_popup lastItem];
111 [menuItem setAttributedTitle:string.get()];
112 // We set the title as well as the attributed title here. The attributed title will be displayed in the menu,
113 // but typeahead will use the non-attributed string that doesn't contain any leading or trailing whitespace.
114 [menuItem setTitle:[[string string] stringByTrimmingCharactersInSet:[NSCharacterSet whitespaceCharacterSet]]];
115 [menuItem setEnabled:m_client->itemIsEnabled(i)];
116 [menuItem setToolTip:m_client->itemToolTip(i)];
117
118 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
119 // Allow the accessible text of the item to be overridden if necessary.
120 if (AXObjectCache::accessibilityEnabled()) {
121 NSString *accessibilityOverride = m_client->itemAccessibilityText(i);
122 if ([accessibilityOverride length])
123 [menuItem accessibilitySetOverrideValue:accessibilityOverride forAttribute:NSAccessibilityDescriptionAttribute];
124 }
125 ALLOW_DEPRECATED_DECLARATIONS_END
126 }
127}

129void PopupMenuMac::show(const IntRect& r, FrameView* v, int selectedIndex)
130{
131 populate();
132 int numItems = [m_popup numberOfItems];
133 if (numItems <= 0) {
134 if (m_client)
135 m_client->popupDidHide();
136 return;
137 }
138 ASSERT(numItems > selectedIndex);
139
140 // Workaround for crazy bug where a selectedIndex of -1 for a menu with only 1 item will cause a blank menu.
141 if (selectedIndex == -1 && numItems == 2 && !m_client->shouldPopOver() && ![[m_popup itemAtIndex:1] isEnabled])
142 selectedIndex = 0;
143
144 NSView* view = v->documentView();
145
146 TextDirection textDirection = m_client->menuStyle().textDirection();
147
148 [m_popup attachPopUpWithFrame:r inView:view];
149 [m_popup selectItemAtIndex:selectedIndex];
150 [m_popup setUserInterfaceLayoutDirection:textDirection == TextDirection::LTR ? NSUserInterfaceLayoutDirectionLeftToRight : NSUserInterfaceLayoutDirectionRightToLeft];
151
152 NSMenu *menu = [m_popup menu];
153 [menu setUserInterfaceLayoutDirection:textDirection == TextDirection::LTR ? NSUserInterfaceLayoutDirectionLeftToRight : NSUserInterfaceLayoutDirectionRightToLeft];
154
155 NSPoint location;
156 CTFontRef font = m_client->menuStyle().font().primaryFont().getCTFont();
157
158 // These values were borrowed from AppKit to match their placement of the menu.
159 const int popOverHorizontalAdjust = -13;
160 const int popUnderHorizontalAdjust = 6;
161 const int popUnderVerticalAdjust = 6;
162 if (m_client->shouldPopOver()) {
163 NSRect titleFrame = [m_popup titleRectForBounds:r];
164 if (titleFrame.size.width <= 0 || titleFrame.size.height <= 0)
165 titleFrame = r;
166 float vertOffset = roundf((NSMaxY(r) - NSMaxY(titleFrame)) + NSHeight(titleFrame));
167 // Adjust for fonts other than the system font.
168 auto defaultFont = adoptCF(CTFontCreateUIFontForLanguage(kCTFontUIFontSystem, CTFontGetSize(font), nil));
169 vertOffset += CTFontGetDescent(font) - CTFontGetDescent(defaultFont.get());
170 vertOffset = fminf(NSHeight(r), vertOffset);
171 if (textDirection == TextDirection::LTR)
172 location = NSMakePoint(NSMinX(r) + popOverHorizontalAdjust, NSMaxY(r) - vertOffset);
173 else
174 location = NSMakePoint(NSMaxX(r) - popOverHorizontalAdjust, NSMaxY(r) - vertOffset);
175 } else {
176 if (textDirection == TextDirection::LTR)
177 location = NSMakePoint(NSMinX(r) + popUnderHorizontalAdjust, NSMaxY(r) + popUnderVerticalAdjust);
178 else
179 location = NSMakePoint(NSMaxX(r) - popUnderHorizontalAdjust, NSMaxY(r) + popUnderVerticalAdjust);
180 }
181 // Save the current event that triggered the popup, so we can clean up our event
182 // state after the NSMenu goes away.
183 Ref<Frame> frame(v->frame());
184 RetainPtr<NSEvent> event = frame->eventHandler().currentNSEvent();
185
186 Ref<PopupMenuMac> protector(*this);
187
188 RetainPtr<NSView> dummyView = adoptNS([[NSView alloc] initWithFrame:r]);
189 [dummyView.get() setUserInterfaceLayoutDirection:textDirection == TextDirection::LTR ? NSUserInterfaceLayoutDirectionLeftToRight : NSUserInterfaceLayoutDirectionRightToLeft];
190 [view addSubview:dummyView.get()];
191 location = [dummyView convertPoint:location fromView:view];
192
193 if (Page* page = frame->page()) {
194 WebView* webView = kit(page);
195 BEGIN_BLOCK_OBJC_EXCEPTIONS
196 CallUIDelegate(webView, @selector(webView:willPopupMenu:), menu);
197 END_BLOCK_OBJC_EXCEPTIONS
198 }
199
200 NSControlSize controlSize;
201 switch (m_client->menuStyle().menuSize()) {
202 case PopupMenuStyle::PopupMenuSizeNormal:
203 controlSize = NSControlSizeRegular;
204 break;
205 case PopupMenuStyle::PopupMenuSizeSmall:
206 controlSize = NSControlSizeSmall;
207 break;
208 case PopupMenuStyle::PopupMenuSizeMini:
209 controlSize = NSControlSizeMini;
210 break;
211#if HAVE(LARGE_CONTROL_SIZE)
212 case PopupMenuStyle::PopupMenuSizeLarge:
213 controlSize = NSControlSizeLarge;
214 break;
215#endif
216 }
217
218 PAL::popUpMenu(menu, location, roundf(NSWidth(r)), dummyView.get(), selectedIndex, toNSFont(font), controlSize, !m_client->menuStyle().hasDefaultAppearance());
219
220 [m_popup dismissPopUp];
221 [dummyView removeFromSuperview];
222
223 if (!m_client)
224 return;
225
226 int newIndex = [m_popup indexOfSelectedItem];
227 m_client->popupDidHide();
228
229 // Adjust newIndex for hidden first item.
230 if (!m_client->shouldPopOver())
231 newIndex--;
232
233 if (selectedIndex != newIndex && newIndex >= 0)
234 m_client->valueChanged(newIndex);
235
236 // Give the frame a chance to fix up its event state, since the popup eats all the
237 // events during tracking.
238 frame->eventHandler().sendFakeEventsAfterWidgetTracking(event.get());
239}

241void PopupMenuMac::hide()
242{
243 [[m_popup menu] cancelTracking];
244 if (m_client)
245 m_client->popupDidHide();
246}
