55String WebAlternativeTextClient::dismissAlternativeSoon(ReasonForDismissingAlternativeText reason)

60static inline NSCorrectionResponse toCorrectionResponse(AutocorrectionResponse response)
61{
62 switch (response) {
63 case WebCore::AutocorrectionResponse::Reverted:
64 return NSCorrectionResponseReverted;
65 case WebCore::AutocorrectionResponse::Edited:
66 return NSCorrectionResponseEdited;
67 case WebCore::AutocorrectionResponse::Accepted:
68 return NSCorrectionResponseAccepted;
69 }
70
71 ASSERT_NOT_REACHED();
72 return NSCorrectionResponseAccepted;
73}
74
75void WebAlternativeTextClient::recordAutocorrectionResponse(AutocorrectionResponse response, const String& replacedString, const String& replacementString)
76{
77 CorrectionPanel::recordAutocorrectionResponse([m_webView spellCheckerDocumentTag], toCorrectionResponse(response), replacedString, replacementString);
78}
