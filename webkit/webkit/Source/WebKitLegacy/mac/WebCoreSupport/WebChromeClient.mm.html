207void WebChromeClient::takeFocus(FocusDirection direction)
208{
209#if !PLATFORM(IOS_FAMILY)
210 if (direction == FocusDirection::Forward) {
211 // Since we're trying to move focus out of m_webView, and because
212 // m_webView may contain subviews within it, we ask it for the next key
213 // view of the last view in its key view loop. This makes m_webView
214 // behave as if it had no subviews, which is the behavior we want.
215 NSView *lastView = [m_webView _findLastViewInKeyViewLoop];
216 // avoid triggering assertions if the WebView is the only thing in the key loop
217 if ([m_webView _becomingFirstResponderFromOutside] && m_webView == [lastView nextValidKeyView])
218 return;
219 [[m_webView window] selectKeyViewFollowingView:lastView];
220 } else {
221 // avoid triggering assertions if the WebView is the only thing in the key loop
222 if ([m_webView _becomingFirstResponderFromOutside] && m_webView == [m_webView previousValidKeyView])
223 return;
224 [[m_webView window] selectKeyViewPrecedingView:m_webView];
225 }
226#endif
227}

245Page* WebChromeClient::createWindow(Frame& frame, const WindowFeatures& features, const NavigationAction&)
246{
247 id delegate = [m_webView UIDelegate];
248 WebView *newWebView;
249
250#if ENABLE(FULLSCREEN_API)
251 if (frame.document() && frame.document()->fullscreenManager().currentFullscreenElement())
252 frame.document()->fullscreenManager().cancelFullscreen();
253#endif
254
255 if ([delegate respondsToSelector:@selector(webView:createWebViewWithRequest:windowFeatures:)]) {
256 auto dictFeatures = adoptNS([[NSMutableDictionary alloc] initWithObjectsAndKeys:
257 @(features.menuBarVisible), @"menuBarVisible",
258 @(features.statusBarVisible), @"statusBarVisible",
259 @(features.toolBarVisible), @"toolBarVisible",
260 @(features.scrollbarsVisible), @"scrollbarsVisible",
261 @(features.resizable), @"resizable",
262 @(features.fullscreen), @"fullscreen",
263 @(features.dialog), @"dialog",
264 nil]);
265
266 if (features.x)
267 [dictFeatures setObject:@(*features.x) forKey:@"x"];
268 if (features.y)
269 [dictFeatures setObject:@(*features.y) forKey:@"y"];
270 if (features.width)
271 [dictFeatures setObject:@(*features.width) forKey:@"width"];
272 if (features.height)
273 [dictFeatures setObject:@(*features.height) forKey:@"height"];
274
275 newWebView = CallUIDelegate(m_webView, @selector(webView:createWebViewWithRequest:windowFeatures:), nil, dictFeatures.get());
276 } else if (features.dialog && [delegate respondsToSelector:@selector(webView:createWebViewModalDialogWithRequest:)]) {
277 newWebView = CallUIDelegate(m_webView, @selector(webView:createWebViewModalDialogWithRequest:), nil);
278 } else {
279 newWebView = CallUIDelegate(m_webView, @selector(webView:createWebViewWithRequest:), nil);
280 }
281
282#if USE(PLUGIN_HOST_PROCESS) && ENABLE(NETSCAPE_PLUGIN_API)
283 if (newWebView)
284 WebKit::NetscapePluginHostManager::singleton().didCreateWindow();
285#endif
286
287 return core(newWebView);
288}

353{
354 switch (source) {
355 case MessageSource::XML:
356 return WebConsoleMessageXMLMessageSource;
357 case MessageSource::JS:
358 return WebConsoleMessageJSMessageSource;
359 case MessageSource::Network:
360 return WebConsoleMessageNetworkMessageSource;
361 case MessageSource::ConsoleAPI:
362 return WebConsoleMessageConsoleAPIMessageSource;
363 case MessageSource::Storage:
364 return WebConsoleMessageStorageMessageSource;
365 case MessageSource::AppCache:
366 return WebConsoleMessageAppCacheMessageSource;
367 case MessageSource::Rendering:
368 return WebConsoleMessageRenderingMessageSource;
369 case MessageSource::CSS:
370 return WebConsoleMessageCSSMessageSource;
371 case MessageSource::Security:
372 return WebConsoleMessageSecurityMessageSource;
373 case MessageSource::ContentBlocker:
374 return WebConsoleMessageContentBlockerMessageSource;
375 case MessageSource::Media:
376 return WebConsoleMessageMediaMessageSource;
377 case MessageSource::MediaSource:
378 return WebConsoleMessageMediaSourceMessageSource;
379 case MessageSource::WebRTC:
380 return WebConsoleMessageWebRTCMessageSource;
381 case MessageSource::ITPDebug:
382 return WebConsoleMessageITPDebugMessageSource;
383 case MessageSource::PrivateClickMeasurement:
384 return WebConsoleMessagePrivateClickMeasurementMessageSource;
385 case MessageSource::PaymentRequest:
386 return WebConsoleMessagePaymentRequestMessageSource;

390 ASSERT_NOT_REACHED();
391 return @"";

412void WebChromeClient::addMessageToConsole(MessageSource source, MessageLevel level, const String& message, unsigned lineNumber, unsigned columnNumber, const String& sourceURL)
413{
414#if !PLATFORM(IOS_FAMILY)
415 id delegate = [m_webView UIDelegate];
416#else
417 if (![m_webView _allowsMessaging])
418 return;
419
420 id delegate = [m_webView _UIKitDelegate];
421 // No delegate means nothing to send this data to so bail.
422 if (!delegate)
423 return;
424#endif
425
426 BOOL respondsToNewSelector = NO;
427
428 SEL selector = @selector(webView:addMessageToConsole:withSource:);
429 if ([delegate respondsToSelector:selector])
430 respondsToNewSelector = YES;
431 else {
432 // The old selector only takes JSMessageSource messages.
433 if (source != MessageSource::JS)
434 return;
435 selector = @selector(webView:addMessageToConsole:);
436 if (![delegate respondsToSelector:selector])
437 return;
438 }
439
440 NSString *messageSource = stringForMessageSource(source);
441 auto dictionary = @{
442 @"message": (NSString *)message,
443 @"lineNumber": @(lineNumber),
444 @"columnNumber": @(columnNumber),
445 @"sourceURL": (NSString *)sourceURL,
446 @"MessageSource": messageSource,
447 @"MessageLevel": stringForMessageLevel(level),
448 };
449
450#if PLATFORM(IOS_FAMILY)
451 [[[m_webView _UIKitDelegateForwarder] asyncForwarder] webView:m_webView addMessageToConsole:dictionary withSource:messageSource];
452#else
453 if (respondsToNewSelector)
454 CallUIDelegate(m_webView, selector, dictionary, messageSource);
455 else
456 CallUIDelegate(m_webView, selector, dictionary);
457#endif
458}

470void WebChromeClient::closeWindowSoon()
471{
472 // We need to remove the parent WebView from WebViewSets here, before it actually
473 // closes, to make sure that JavaScript code that executes before it closes
474 // can't find it. Otherwise, window.open will select a closed WebView instead of
475 // opening a new one <rdar://problem/3572585>.
476
477 // We also need to stop the load to prevent further parsing or JavaScript execution
478 // after the window has torn down <rdar://problem/4161660>.
479
480 // FIXME: This code assumes that the UI delegate will respond to a webViewClose
481 // message by actually closing the WebView. Safari guarantees this behavior, but other apps might not.
482 // This approach is an inherent limitation of not making a close execute immediately
483 // after a call to window.close.
484
485 [m_webView setGroupName:nil];
486 [m_webView stopLoading:nil];
487 [m_webView performSelector:@selector(_closeWindow) withObject:nil afterDelay:0.0];
488}

522bool WebChromeClient::runJavaScriptPrompt(Frame& frame, const String& prompt, const String& defaultText, String& result)
523{
524 id delegate = [m_webView UIDelegate];
525 SEL selector = @selector(webView:runJavaScriptTextInputPanelWithPrompt:defaultText:initiatedByFrame:);
526 NSString *defaultString = defaultText;
527 if ([delegate respondsToSelector:selector]) {
528 result = (NSString *)CallUIDelegate(m_webView, selector, prompt, defaultString, kit(&frame));
529 return !result.isNull();
530 }
531
532 // Call the old version of the delegate method if it is implemented.
533 selector = @selector(webView:runJavaScriptTextInputPanelWithPrompt:defaultText:);
534 if ([delegate respondsToSelector:selector]) {
535 result = (NSString *)CallUIDelegate(m_webView, selector, prompt, defaultString);
536 return !result.isNull();
537 }
538
539 result = [[WebDefaultUIDelegate sharedUIDelegate] webView:m_webView runJavaScriptTextInputPanelWithPrompt:prompt defaultText:defaultString initiatedByFrame:kit(&frame)];
540 return !result.isNull();
541}

609void WebChromeClient::scrollRectIntoView(const IntRect& r) const
610{
611 // FIXME: This scrolling behavior should be under the control of the embedding client,
612 // perhaps in a delegate method, rather than something WebKit does unconditionally.
613 NSView *coordinateView = [[[m_webView mainFrame] frameView] documentView];
614 NSRect rect = r;
615 for (NSView *view = m_webView; view; view = [view superview]) {
616 if ([view isKindOfClass:[NSClipView class]]) {
617 NSClipView *clipView = (NSClipView *)view;
618 NSView *documentView = [clipView documentView];
619 [documentView scrollRectToVisible:[documentView convertRect:rect fromView:coordinateView]];
620 }
621 }
622}
