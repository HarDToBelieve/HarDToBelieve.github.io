131bool WebContextMenuClient::clientFloatRectForNode(Node& node, FloatRect& rect) const
132{
133 RenderObject* renderer = node.renderer();
134 if (!renderer) {
135 // This method shouldn't be called in cases where the controlled node hasn't rendered.
136 ASSERT_NOT_REACHED();
137 return false;
138 }
139
140 if (!is<RenderBox>(*renderer))
141 return false;
142 auto& renderBox = downcast<RenderBox>(*renderer);
143
144 LayoutRect layoutRect = renderBox.clientBoxRect();
145 FloatQuad floatQuad = renderBox.localToAbsoluteQuad(FloatQuad(layoutRect));
146 rect = floatQuad.boundingBox();
147
148 return true;
149}

167WebCore::FloatRect WebContextMenuClient::screenRectForCurrentSharingServicePickerItem(WebSharingServicePickerController &)
168{
169 Page* page = [m_webView page];
170 if (!page)
171 return NSZeroRect;
172
173 Node* node = page->contextMenuController().context().hitTestResult().innerNode();
174 if (!node)
175 return NSZeroRect;
176
177 FrameView* frameView = node->document().view();
178 if (!frameView) {
179 // This method shouldn't be called in cases where the controlled node isn't in a rendered view.
180 ASSERT_NOT_REACHED();
181 return NSZeroRect;
182 }
183
184 FloatRect rect;
185 if (!clientFloatRectForNode(*node, rect))
186 return NSZeroRect;
187
188 // FIXME: https://webkit.org/b/132915
189 // Ideally we'd like to convert the content rect to screen coordinates without the lossy float -> int conversion.
190 // Creating a rounded int rect works well in practice, but might still lead to off-by-one-pixel problems in edge cases.
191 IntRect intRect = roundedIntRect(rect);
192 return frameView->contentsToScreen(intRect);
193}

195RetainPtr<NSImage> WebContextMenuClient::imageForCurrentSharingServicePickerItem(WebSharingServicePickerController &)
196{
197 auto page = [m_webView page];
198 if (!page)
199 return nil;
200
201 auto node = makeRefPtr(page->contextMenuController().context().hitTestResult().innerNode());
202 if (!node)
203 return nil;
204
205 auto frameView = makeRefPtr(node->document().view());
206 if (!frameView) {
207 // This method shouldn't be called in cases where the controlled node isn't in a rendered view.
208 ASSERT_NOT_REACHED();
209 return nil;
210 }
211
212 FloatRect rect;
213 if (!clientFloatRectForNode(*node, rect))
214 return nil;
215
216 // This is effectively a snapshot, and will be painted in an unaccelerated fashion in line with FrameSnapshotting.
217 auto buffer = ImageBuffer::create(rect.size(), RenderingMode::Unaccelerated, 1, DestinationColorSpace::SRGB(), PixelFormat::BGRA8);
218 if (!buffer)
219 return nil;
220
221 auto oldSelection = frameView->frame().selection().selection();
222 frameView->frame().selection().setSelection(*makeRangeSelectingNode(*node), FrameSelection::DoNotSetFocus);
223
224 auto oldPaintBehavior = frameView->paintBehavior();
225 frameView->setPaintBehavior(PaintBehavior::SelectionOnly);
226
227 buffer->context().translate(-toFloatSize(rect.location()));
228 frameView->paintContents(buffer->context(), roundedIntRect(rect));
229
230 frameView->frame().selection().setSelection(oldSelection);
231 frameView->setPaintBehavior(oldPaintBehavior);
232
233 auto image = ImageBuffer::sinkIntoImage(WTFMove(buffer));
234 if (!image)
235 return nil;
236
237 return image->snapshotNSImage();
238}

242NSMenu *WebContextMenuClient::contextMenuForEvent(NSEvent *event, NSView *view, bool& isServicesMenu)
243{
244 isServicesMenu = false;
245
246 Page* page = [m_webView page];
247 if (!page)
248 return nil;
249
250#if ENABLE(SERVICE_CONTROLS)
251 if (Image* image = page->contextMenuController().context().controlledImage()) {
252 ASSERT(page->contextMenuController().context().hitTestResult().innerNode());
253
254 RetainPtr<NSItemProvider> itemProvider = adoptNS([[NSItemProvider alloc] initWithItem:image->snapshotNSImage().get() typeIdentifier:@"public.image"]);
255
256 bool isContentEditable = page->contextMenuController().context().hitTestResult().innerNode()->isContentEditable();
257 m_sharingServicePickerController = adoptNS([[WebSharingServicePickerController alloc] initWithItems:@[ itemProvider.get() ] includeEditorServices:isContentEditable client:this style:NSSharingServicePickerStyleRollover]);
258
259 isServicesMenu = true;
260 return [m_sharingServicePickerController menu];
261 }
262#endif
263
264 return [view menuForEvent:event];
265}

267void WebContextMenuClient::showContextMenu()
268{
269 Page* page = [m_webView page];
270 if (!page)
271 return;
272 Frame* frame = page->contextMenuController().hitTestResult().innerNodeFrame();
273 if (!frame)
274 return;
275 FrameView* frameView = frame->view();
276 if (!frameView)
277 return;
278
279 NSView* view = frameView->documentView();
280 IntPoint point = frameView->contentsToWindow(page->contextMenuController().hitTestResult().roundedPointInInnerNodeFrame());
281 NSEvent* event = [NSEvent mouseEventWithType:NSEventTypeRightMouseDown location:point modifierFlags:0 timestamp:0 windowNumber:[[view window] windowNumber] context:0 eventNumber:0 clickCount:1 pressure:1];
282
283 // Show the contextual menu for this event.
284 bool isServicesMenu;
285 if (NSMenu *menu = contextMenuForEvent(event, view, isServicesMenu)) {
286 if (isServicesMenu)
287 [menu popUpMenuPositioningItem:nil atLocation:[view convertPoint:point toView:nil] inView:view];
288 else
289 [NSMenu popUpContextMenu:menu withEvent:event forView:view];
290 }
