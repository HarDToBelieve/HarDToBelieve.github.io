33using namespace WebCore;
34
35WebDeviceOrientationClient::WebDeviceOrientationClient(WebView* webView)
36 : m_webView(webView)
37 , m_controller(0)
38{
39}
40
41void WebDeviceOrientationClient::setController(DeviceOrientationController* controller)
42{
43 // This is called by the Page constructor before our WebView has the provider set up.
44 // MemoryCache the controller for later use.
45 m_controller = controller;
46}
47
48void WebDeviceOrientationClient::startUpdating()
49{
50 [getProvider() startUpdating];
51}
52
53void WebDeviceOrientationClient::stopUpdating()
54{
55 [getProvider() stopUpdating];
56}
57
58DeviceOrientationData* WebDeviceOrientationClient::lastOrientation() const
59{
60 return core([getProvider() lastOrientation]);
61}
62
63void WebDeviceOrientationClient::deviceOrientationControllerDestroyed()
64{
65 delete this;
66}
67
68id<WebDeviceOrientationProvider> WebDeviceOrientationClient::getProvider() const
69{
70 if (!m_provider) {
71 m_provider = [m_webView _deviceOrientationProvider];
72 if ([m_provider respondsToSelector:@selector(setController:)])
73 wtfObjCMsgSend<void>(m_provider, @selector(setController:), m_controller);
74 }
75 return m_provider;
76}
