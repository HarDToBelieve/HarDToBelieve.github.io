128{
129 RefPtr<UndoStep> m_step;
130}
131
132+ (WebUndoStep *)stepWithUndoStep:(Ref<UndoStep>&&)step;

137@implementation WebUndoStep
138
139+ (void)initialize
140{
141#if !PLATFORM(IOS_FAMILY)
142 JSC::initialize();
143 WTF::initializeMainThread();
144 WebCore::populateJITOperations();
145#endif
146}

148- (id)initWithUndoStep:(Ref<UndoStep>&&)step
149{
150 self = [super init];
151 if (!self)
152 return nil;
153 m_step = WTFMove(step);
154 return self;
155}
156
157- (void)dealloc
158{
159 if (WebCoreObjCScheduleDeallocateOnMainThread([WebUndoStep class], self))
160 return;
161
162 [super dealloc];
163}
164
165+ (WebUndoStep *)stepWithUndoStep:(Ref<UndoStep>&&)step
166{
167 return adoptNS([[WebUndoStep alloc] initWithUndoStep:WTFMove(step)]).autorelease();
168}
169
170- (UndoStep&)step
171{
172 return *m_step;
173}

186- (void)undoEditing:(id)arg
187{
188 ASSERT([arg isKindOfClass:[WebUndoStep class]]);
189 [arg step].unapply();

352void WebEditorClient::respondToChangedSelection(Frame* frame)
353{
354 if (frame->editor().isGettingDictionaryPopupInfo())
355 return;
356
357 NSView<WebDocumentView> *documentView = [[kit(frame) frameView] documentView];
358 if ([documentView isKindOfClass:[WebHTMLView class]]) {
359 [(WebHTMLView *)documentView _selectionChanged];
360 [m_webView updateTouchBar];
361 m_lastEditorStateWasContentEditable = [(WebHTMLView *)documentView _isEditable] ? EditorStateIsContentEditable::Yes : EditorStateIsContentEditable::No;
362 }
363
364#if !PLATFORM(IOS_FAMILY)
365 // FIXME: This quirk is needed due to <rdar://problem/5009625> - We can phase it out once Aperture can adopt the new behavior on their end
366 if (!WebKitLinkedOnOrAfter(WEBKIT_FIRST_VERSION_WITHOUT_APERTURE_QUIRK) && [[[NSBundle mainBundle] bundleIdentifier] isEqualToString:@"com.apple.Aperture"])
367 return;
368
369 [[NSNotificationCenter defaultCenter] postNotificationName:WebViewDidChangeSelectionNotification object:m_webView];
370#else
371 // Selection can be changed while deallocating down the WebView / Frame / Editor. Do not post in that case because it's already too late
372 // for the NSInvocation to retain the WebView.
373 if (![m_webView _isClosing])
374 WebThreadPostNotification(WebViewDidChangeSelectionNotification, m_webView, nil);
375#endif
376
377#if PLATFORM(MAC)
378 if (frame->editor().canEdit())
379 requestCandidatesForSelection(frame->selection().selection());
380#endif
381}

434static NSDictionary *attributesForAttributedStringConversion()
435{
436 // This function needs to be kept in sync with identically named one in WebCore, which is used on newer OS versions.
437 auto excludedElements = adoptNS([[NSMutableArray alloc] initWithObjects:
438 // Omit style since we want style to be inline so the fragment can be easily inserted.
439 @"style",
440 // Omit xml so the result is not XHTML.
441 @"xml",
442 // Omit tags that will get stripped when converted to a fragment anyway.
443 @"doctype", @"html", @"head", @"body",
444 // Omit deprecated tags.
445 @"applet", @"basefont", @"center", @"dir", @"font", @"menu", @"s", @"strike", @"u",
446#if !ENABLE(ATTACHMENT_ELEMENT)
447 // Omit object so no file attachments are part of the fragment.
448 @"object",
449#endif
450 nil]);
451
452#if ENABLE(ATTACHMENT_ELEMENT)
453 if (!RuntimeEnabledFeatures::sharedFeatures().attachmentElementEnabled())
454 [excludedElements addObject:@"object"];
455#endif
456
457 return @{ NSExcludedElementsDocumentAttribute: excludedElements.get() };
458}

460void _WebCreateFragment(Document& document, NSAttributedString *string, FragmentAndResources& result)
461{
462 static NeverDestroyed<RetainPtr<NSDictionary>> documentAttributes = attributesForAttributedStringConversion();
463 NSArray *subresources;
464 DOMDocumentFragment* fragment = [string _documentFromRange:NSMakeRange(0, [string length])
465 document:kit(&document) documentAttributes:documentAttributes.get().get() subresources:&subresources];
466 result.fragment = core(fragment);
467 for (WebResource* resource in subresources)
468 result.resources.append([resource _coreResource].get());
469}

578void WebEditorClient::registerUndoOrRedoStep(UndoStep& step, bool isRedo)
579{
580 NSUndoManager *undoManager = [m_webView undoManager];
581
582#if PLATFORM(IOS_FAMILY)
583 // While we are undoing, we shouldn't be asked to register another Undo operation, we shouldn't even be touching the DOM.
584 // But just in case this happens, return to avoid putting the undo manager into an inconsistent state.
585 // Same for being asked to register a Redo operation in the midst of another Redo.
586 if (([undoManager isUndoing] && !isRedo) || ([undoManager isRedoing] && isRedo))
587 return;
588#endif
589
590 NSString *actionName = step.label();
591 [undoManager registerUndoWithTarget:m_undoTarget.get() selector:(isRedo ? @selector(redoEditing:) : @selector(undoEditing:)) object:[WebUndoStep stepWithUndoStep:step]];
592 if (actionName)
593 [undoManager setActionName:actionName];
594 m_haveUndoRedoOperations = YES;
595}

597void WebEditorClient::updateEditorStateAfterLayoutIfEditabilityChanged()
598{
599 // FIXME: We should update EditorStateIsContentEditable to track whether the state is richly
600 // editable or plainttext-only.
601 if (m_lastEditorStateWasContentEditable == EditorStateIsContentEditable::Unset)
602 return;
603
604 Frame* frame = core([m_webView _selectedOrMainFrame]);
605 if (!frame)
606 return;
607
608 NSView<WebDocumentView> *documentView = [[kit(frame) frameView] documentView];
609 if (![documentView isKindOfClass:[WebHTMLView class]])
610 return;
611
612 EditorStateIsContentEditable editorStateIsContentEditable = [(WebHTMLView *)documentView _isEditable] ? EditorStateIsContentEditable::Yes : EditorStateIsContentEditable::No;
613 if (m_lastEditorStateWasContentEditable != editorStateIsContentEditable)
614 [m_webView updateTouchBar];
615}

627void WebEditorClient::clearUndoRedoOperations()
628{
629 if (m_haveUndoRedoOperations) {
630 // workaround for <rdar://problem/4645507> NSUndoManager dies
631 // with uncaught exception when undo items cleared while
632 // groups are open
633 NSUndoManager *undoManager = [m_webView undoManager];
634 int groupingLevel = [undoManager groupingLevel];
635 for (int i = 0; i < groupingLevel; ++i)
636 [undoManager endUndoGrouping];
637
638 [undoManager removeAllActionsWithTarget:m_undoTarget.get()];
639
640 for (int i = 0; i < groupingLevel; ++i)
641 [undoManager beginUndoGrouping];
642
643 m_haveUndoRedoOperations = NO;
644 }
645}

743static SEL selectorForKeyEvent(KeyboardEvent* event)
744{
745 // FIXME: This helper function is for the auto-fill code so we can pass a selector to the form delegate.
746 // Eventually, we should move all of the auto-fill code down to WebKit and remove the need for this function by
747 // not relying on the selector in the new implementation.
748 // The key identifiers are from <http://www.w3.org/TR/DOM-Level-3-Events/keyset.html#KeySet-Set>
749 const String& key = event->keyIdentifier();
750 if (key == "Up")
751 return @selector(moveUp:);
752 if (key == "Down")
753 return @selector(moveDown:);
754IGNORE_WARNINGS_BEGIN("undeclared-selector")
755 if (key == "U+001B")
756 return @selector(cancel:);
757 if (key == "U+0009") {
758 if (event->shiftKey())
759 return @selector(insertBacktab:);
760 return @selector(insertTab:);
761 }
762 if (key == "Enter")
763 return @selector(insertNewline:);
764IGNORE_WARNINGS_END
765 return 0;
766}

952static Vector<TextCheckingResult> core(NSArray *incomingResults, OptionSet<TextCheckingType> checkingTypes)
953{
954 Vector<TextCheckingResult> results;
955
956 for (NSTextCheckingResult *incomingResult in incomingResults) {
957 NSTextCheckingType resultType = [incomingResult resultType];
958 ASSERT(incomingResult.range.location != NSNotFound);
959 ASSERT(incomingResult.range.length > 0);
960 auto resultRange = incomingResult.range;
961 if (resultType == NSTextCheckingTypeSpelling && checkingTypes.contains(TextCheckingType::Spelling)) {
962 TextCheckingResult result;
963 result.type = TextCheckingType::Spelling;
964 result.range = resultRange;
965 results.append(result);
966 } else if (resultType == NSTextCheckingTypeGrammar && checkingTypes.contains(TextCheckingType::Grammar)) {
967 TextCheckingResult result;
968 NSArray *details = [incomingResult grammarDetails];
969 result.type = TextCheckingType::Grammar;
970 result.range = resultRange;
971 for (NSDictionary *incomingDetail in details) {
972 ASSERT(incomingDetail);
973 GrammarDetail detail;
974 NSValue *detailRangeAsNSValue = [incomingDetail objectForKey:NSGrammarRange];
975 ASSERT(detailRangeAsNSValue);
976 NSRange detailNSRange = [detailRangeAsNSValue rangeValue];
977 ASSERT(detailNSRange.location != NSNotFound);
978 ASSERT(detailNSRange.length > 0);
979 detail.range = detailNSRange;
980 detail.userDescription = [incomingDetail objectForKey:NSGrammarUserDescription];
981 NSArray *guesses = [incomingDetail objectForKey:NSGrammarCorrections];
982 for (NSString *guess in guesses)
983 detail.guesses.append(String(guess));
984 result.details.append(detail);
985 }
986 results.append(result);
987 } else if (resultType == NSTextCheckingTypeLink && checkingTypes.contains(TextCheckingType::Link)) {
988 TextCheckingResult result;
989 result.type = TextCheckingType::Link;
990 result.range = resultRange;
991 result.replacement = [[incomingResult URL] absoluteString];
992 results.append(result);
993 } else if (resultType == NSTextCheckingTypeQuote && checkingTypes.contains(TextCheckingType::Quote)) {
994 TextCheckingResult result;
995 result.type = TextCheckingType::Quote;
996 result.range = resultRange;
997 result.replacement = [incomingResult replacementString];
998 results.append(result);
999 } else if (resultType == NSTextCheckingTypeDash && checkingTypes.contains(TextCheckingType::Dash)) {
1000 TextCheckingResult result;
1001 result.type = TextCheckingType::Dash;
1002 result.range = resultRange;
1003 result.replacement = [incomingResult replacementString];
1004 results.append(result);
1005 } else if (resultType == NSTextCheckingTypeReplacement && checkingTypes.contains(TextCheckingType::Replacement)) {
1006 TextCheckingResult result;
1007 result.type = TextCheckingType::Replacement;
1008 result.range = resultRange;
1009 result.replacement = [incomingResult replacementString];
1010 results.append(result);
1011 } else if (resultType == NSTextCheckingTypeCorrection && checkingTypes.contains(TextCheckingType::Correction)) {
1012 TextCheckingResult result;
1013 result.type = TextCheckingType::Correction;
1014 result.range = resultRange;
1015 result.replacement = [incomingResult replacementString];
1016 results.append(result);
1017 }
1018 }
1019
1020 return results;
1021}
