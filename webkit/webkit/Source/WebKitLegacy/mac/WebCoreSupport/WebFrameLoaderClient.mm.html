239#if PLATFORM(IOS_FAMILY)
240bool WebFrameLoaderClient::forceLayoutOnRestoreFromBackForwardCache()
241{
242 NSView <WebDocumentView> *view = [m_webFrame->_private->webFrameView documentView];
243 // This gets called to lay out a page restored from the back/forward cache.
244 // To work around timing problems with UIKit, restore fixed
245 // layout settings here.
246 WebView* webView = getWebView(m_webFrame.get());
247 bool isMainFrame = [webView mainFrame] == m_webFrame.get();
248 auto* coreFrame = core(m_webFrame.get());
249 if (isMainFrame && coreFrame->view()) {
250 WebCore::IntSize newSize([webView _fixedLayoutSize]);
251 coreFrame->view()->setFixedLayoutSize(newSize);
252 coreFrame->view()->setUseFixedLayout(!newSize.isEmpty());
253 }
254 [view setNeedsLayout:YES];
255 [view layout];
256 return true;
257}
258#endif

301void WebFrameLoaderClient::convertMainResourceLoadToDownload(WebCore::DocumentLoader* documentLoader, const WebCore::ResourceRequest& request, const WebCore::ResourceResponse& response)
302{
303 WebView *webView = getWebView(m_webFrame.get());
304 auto* mainResourceLoader = documentLoader->mainResourceLoader();
305
306 if (!mainResourceLoader) {
307 // The resource has already been cached, or the conversion is being attmpted when not calling SubresourceLoader::didReceiveResponse().
308 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
309 auto webDownload = adoptNS([[WebDownload alloc] initWithRequest:request.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::UpdateHTTPBody) delegate:[webView downloadDelegate]]);
310 ALLOW_DEPRECATED_DECLARATIONS_END
311 webDownload.autorelease();
312 return;
313 }
314
315 auto* handle = mainResourceLoader->handle();
316
317ALLOW_DEPRECATED_DECLARATIONS_BEGIN
318 [WebDownload _downloadWithLoadingConnection:handle->connection() request:request.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::UpdateHTTPBody) response:response.nsURLResponse() delegate:[webView downloadDelegate] proxy:nil];
319ALLOW_DEPRECATED_DECLARATIONS_END
320}

359void WebFrameLoaderClient::dispatchWillSendRequest(WebCore::DocumentLoader* loader, unsigned long identifier, WebCore::ResourceRequest& request, const WebCore::ResourceResponse& redirectResponse)
360{
361 WebView *webView = getWebView(m_webFrame.get());
362 WebResourceDelegateImplementationCache* implementations = WebViewGetResourceLoadDelegateImplementations(webView);
363
364 if (redirectResponse.isNull())
365 static_cast<WebDocumentLoaderMac*>(loader)->increaseLoadCount(identifier);
366
367 NSURLRequest *currentURLRequest = request.nsURLRequest(WebCore::HTTPBodyUpdatePolicy::UpdateHTTPBody);
368
369#if PLATFORM(MAC)
370 if (WebCore::MacApplication::isAppleMail() && loader->substituteData().isValid()) {
371 // Mail.app checks for this property to detect data / archive loads.
372 [NSURLProtocol setProperty:@"" forKey:@"WebDataRequest" inRequest:(NSMutableURLRequest *)currentURLRequest];
373 }
374#endif
375
376 NSURLRequest *newURLRequest = currentURLRequest;
377#if PLATFORM(IOS_FAMILY)
378 if (implementations->webThreadWillSendRequestFunc) {
379 newURLRequest = (NSURLRequest *)CallResourceLoadDelegateInWebThread(implementations->webThreadWillSendRequestFunc, webView, @selector(webThreadWebView:resource:willSendRequest:redirectResponse:fromDataSource:), [webView _objectForIdentifier:identifier], currentURLRequest, redirectResponse.nsURLResponse(), dataSource(loader));
380 } else
381#endif
382 if (implementations->willSendRequestFunc)
383 newURLRequest = (NSURLRequest *)CallResourceLoadDelegate(implementations->willSendRequestFunc, webView, @selector(webView:resource:willSendRequest:redirectResponse:fromDataSource:), [webView _objectForIdentifier:identifier], currentURLRequest, redirectResponse.nsURLResponse(), dataSource(loader));
384
385 if (newURLRequest != currentURLRequest)
386 request.updateFromDelegatePreservingOldProperties(WebCore::ResourceRequest(newURLRequest));
387}

1403void WebFrameLoaderClient::transitionToCommittedForNewPage()
1404{
1405 WebDataSource *dataSource = [m_webFrame.get() _dataSource];
1406
1407#if PLATFORM(IOS_FAMILY)
1408 bool willProduceHTMLView;
1409 // Fast path that skips initialization of objc class objects.
1410 if ([dataSource _documentLoader]->responseMIMEType() == "text/html")
1411 willProduceHTMLView = true;
1412 else
1413 willProduceHTMLView = [m_webFrame->_private->webFrameView _viewClassForMIMEType:[dataSource _responseMIMEType]] == [WebHTMLView class];
1414#else
1415 // FIXME (Viewless): I assume we want the equivalent of this optimization for viewless mode too.
1416 bool willProduceHTMLView = [m_webFrame->_private->webFrameView _viewClassForMIMEType:[dataSource _responseMIMEType]] == [WebHTMLView class];
1417#endif
1418 bool canSkipCreation = core(m_webFrame.get())->loader().stateMachine().committingFirstRealLoad() && willProduceHTMLView;
1419 if (canSkipCreation) {
1420 [[m_webFrame->_private->webFrameView documentView] setDataSource:dataSource];
1421 return;
1422 }
1423
1424#if !PLATFORM(IOS_FAMILY)
1425 // Don't suppress scrollbars before the view creation if we're making the view for a non-HTML view.
1426 if (!willProduceHTMLView)
1427 [[m_webFrame->_private->webFrameView _scrollView] setScrollBarsSuppressed:NO repaintOnUnsuppress:NO];
1428
1429 // clean up webkit plugin instances before WebHTMLView gets freed.
1430 WebView *webView = getWebView(m_webFrame.get());
1431 [webView removePluginInstanceViewsFor:(m_webFrame.get())];
1432
1433 RetainPtr<NSView <WebDocumentView>> documentView = [m_webFrame->_private->webFrameView _makeDocumentViewForDataSource:dataSource];
1434#else
1435 RetainPtr<NSView <WebDocumentView>> documentView;
1436
1437 // Fast path that skips initialization of objc class objects.
1438 if (willProduceHTMLView) {
1439 documentView = adoptNS([[WebHTMLView alloc] initWithFrame:[m_webFrame->_private->webFrameView bounds]]);
1440 [m_webFrame->_private->webFrameView _setDocumentView:documentView.get()];
1441 } else
1442 documentView = [m_webFrame->_private->webFrameView _makeDocumentViewForDataSource:dataSource];
1443#endif
1444 if (!documentView)
1445 return;
1446
1447 // FIXME: Could we skip some of this work for a top-level view that is not a WebHTMLView?
1448
1449 // If we own the view, delete the old one - otherwise the render m_frame will take care of deleting the view.
1450 auto* coreFrame = core(m_webFrame.get());
1451 auto* page = coreFrame->page();
1452 bool isMainFrame = coreFrame->isMainFrame();
1453 if (isMainFrame && coreFrame->view())
1454 coreFrame->view()->setParentVisible(false);
1455 coreFrame->setView(nullptr);
1456 auto coreView = WebCore::FrameView::create(*coreFrame);
1457 coreFrame->setView(coreView.copyRef());
1458
1459#if PLATFORM(IOS_FAMILY)
1460 page->setDelegatesScaling(true);
1461#endif
1462
1463 [m_webFrame.get() _updateBackgroundAndUpdatesWhileOffscreen];
1464 [m_webFrame->_private->webFrameView _install];
1465
1466 if (isMainFrame) {
1467#if PLATFORM(IOS_FAMILY)
1468 coreView->setDelegatesScrolling(true);
1469#endif
1470 coreView->setParentVisible(true);
1471 }
1472
1473 // Call setDataSource on the document view after it has been placed in the view hierarchy.
1474 // This what we for the top-level view, so should do this for views in subframes as well.
1475 [documentView setDataSource:dataSource];
1476
1477 // The following is a no-op for WebHTMLRepresentation, but for custom document types
1478 // like the ones that Safari uses for bookmarks it is the only way the DocumentLoader
1479 // will get the proper title.
1480 if (auto documentLoader = [dataSource _documentLoader])
1481 documentLoader->setTitle({ [dataSource pageTitle], WebCore::TextDirection::LTR });
1482
1483 if (auto* ownerElement = coreFrame->ownerElement())
1484 coreFrame->view()->setCanHaveScrollbars(ownerElement->scrollingMode() != WebCore::ScrollbarAlwaysOff);
1485
1486 // If the document view implicitly became first responder, make sure to set the focused frame properly.
1487 if ([[documentView window] firstResponder] == documentView) {
1488 page->focusController().setFocusedFrame(coreFrame);
1489 page->focusController().setFocused(true);
1490 }
1491}

1659
1660 if ([m_webFrame->_private->webFrameView _viewClassForMIMEType:type])

1870RefPtr<WebCore::Widget> WebFrameLoaderClient::createPlugin(const WebCore::IntSize& size, WebCore::HTMLPlugInElement& element, const URL& url,
1871 const Vector<String>& paramNames, const Vector<String>& paramValues, const String& mimeType, bool loadManually)
1872{
1873 BEGIN_BLOCK_OBJC_EXCEPTIONS
1874
1875 ASSERT(paramNames.size() == paramValues.size());
1876
1877 int errorCode = 0;
1878
1879 WebView *webView = getWebView(m_webFrame.get());
1880 auto* document = core(m_webFrame.get())->document();
1881 NSURL *baseURL = document->baseURL();
1882 NSURL *pluginURL = url;
1883 auto attributeKeys = createNSArray(paramNames);
1884
1885#if PLATFORM(MAC)
1886 SEL selector = @selector(webView:plugInViewWithArguments:);
1887 if ([[webView UIDelegate] respondsToSelector:selector]) {
1888 NSDictionary *attributes = [NSDictionary dictionaryWithObjects:createNSArray(paramValues).get() forKeys:attributeKeys.get()];
1889 auto arguments = adoptNS([[NSDictionary alloc] initWithObjectsAndKeys:
1890 attributes, WebPlugInAttributesKey,
1891 @(loadManually ? WebPlugInModeFull : WebPlugInModeEmbed), WebPlugInModeKey,
1892 [NSNumber numberWithBool:!loadManually], WebPlugInShouldLoadMainResourceKey,
1893 kit(&element), WebPlugInContainingElementKey,
1894 // FIXME: We should be passing base URL, see <https://bugs.webkit.org/show_bug.cgi?id=35215>.
1895 pluginURL, WebPlugInBaseURLKey, // pluginURL might be nil, so add it last
1896 nil]);
1897
1898 NSView *view = CallUIDelegate(webView, selector, arguments.get());
1899
1900 if (view)
1901 return adoptRef(*new PluginWidget(view));
1902 }
1903#endif
1904
1905 NSString *MIMEType;
1906 WebBasePluginPackage *pluginPackage;
1907 if (mimeType.isEmpty()) {
1908 MIMEType = nil;
1909 pluginPackage = nil;
1910 } else {
1911 MIMEType = mimeType;
1912 pluginPackage = [webView _pluginForMIMEType:mimeType];
1913 }
1914
1915 NSString *extension = [[pluginURL path] pathExtension];
1916
1917#if PLATFORM(IOS_FAMILY)

1919#else

1921#endif
1922 {
1923 pluginPackage = [webView _pluginForExtension:extension];
1924 if (pluginPackage) {
1925 NSString *newMIMEType = [pluginPackage MIMETypeForExtension:extension];

1927 MIMEType = newMIMEType;
1928 }
1929 }
1930
1931 NSView *view = nil;
1932
1933 if (pluginPackage) {
1934 if (shouldBlockPlugin(pluginPackage)) {
1935 errorCode = WebKitErrorBlockedPlugInVersion;
1936 if (is<WebCore::RenderEmbeddedObject>(element.renderer()))
1937 downcast<WebCore::RenderEmbeddedObject>(*element.renderer()).setPluginUnavailabilityReason(WebCore::RenderEmbeddedObject::InsecurePluginVersion);
1938 } else {
1939 if ([pluginPackage isKindOfClass:[WebPluginPackage class]])
1940 view = pluginView(m_webFrame.get(), (WebPluginPackage *)pluginPackage, attributeKeys.get(), createNSArray(paramValues).get(), baseURL, kit(&element), loadManually);
1941#if ENABLE(NETSCAPE_PLUGIN_API)
1942 else if ([pluginPackage isKindOfClass:[WebNetscapePluginPackage class]] && [webView.preferences _boolValueForKey:@"WebKitNPAPIPlugInsEnabledForTestingInWebKitLegacy"]) {
1943 auto pluginView = adoptNS([[NETSCAPE_PLUGIN_VIEW alloc]
1944 initWithFrame:NSMakeRect(0, 0, size.width(), size.height())
1945 pluginPackage:(WebNetscapePluginPackage *)pluginPackage
1946 URL:pluginURL
1947 baseURL:baseURL
1948 MIMEType:MIMEType
1949 attributeKeys:attributeKeys.get()
1950 attributeValues:createNSArray(paramValues).get()
1951 loadManually:loadManually
1952 element:&element]);
1953
1954 return adoptRef(new NetscapePluginWidget(pluginView.get()));
1955 }
1956#endif
1957 }
1958 } else
1959 errorCode = WebKitErrorCannotFindPlugIn;
1960
1961 if (!errorCode && !view)
1962 errorCode = WebKitErrorCannotLoadPlugIn;
1963
1964 if (errorCode && m_webFrame) {
1965 WebResourceDelegateImplementationCache* implementations = WebViewGetResourceLoadDelegateImplementations(webView);
1966 if (implementations->plugInFailedWithErrorFunc) {
1967 URL pluginPageURL = document->completeURL(WebCore::stripLeadingAndTrailingHTMLSpaces(parameterValue(paramNames, paramValues, "pluginspage")));
1968 if (!pluginPageURL.protocolIsInHTTPFamily())
1969 pluginPageURL = URL();
1970 NSString *pluginName = pluginPackage ? (NSString *)[pluginPackage pluginInfo].name : nil;
1971
1972 auto error = adoptNS([[NSError alloc] _initWithPluginErrorCode:errorCode
1973 contentURL:pluginURL pluginPageURL:pluginPageURL pluginName:pluginName MIMEType:MIMEType]);
1974 CallResourceLoadDelegate(implementations->plugInFailedWithErrorFunc, [m_webFrame.get() webView],
1975 @selector(webView:plugInFailedWithError:dataSource:), error.get(), [m_webFrame.get() _dataSource]);
1976 }
1977
1978 return nullptr;
1979 }
1980
1981 ASSERT(view);
1982#if PLATFORM(IOS_FAMILY)
1983 return adoptRef(new PluginWidgetIOS(view));
1984#else
1985 return adoptRef(new PluginWidget(view));
1986#endif
1987
1988 END_BLOCK_OBJC_EXCEPTIONS
1989
1990 return nullptr;
1991}
