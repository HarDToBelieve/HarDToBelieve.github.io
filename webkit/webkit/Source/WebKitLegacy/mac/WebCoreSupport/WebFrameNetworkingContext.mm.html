47NetworkStorageSession& WebFrameNetworkingContext::ensurePrivateBrowsingSession()
48{
49 ASSERT(isMainThread());
50 NetworkStorageSessionMap::ensureSession(PAL::SessionID::legacyPrivateSessionID(), [[NSBundle mainBundle] bundleIdentifier]);
51 return *NetworkStorageSessionMap::storageSession(PAL::SessionID::legacyPrivateSessionID());
52}
53
54void WebFrameNetworkingContext::destroyPrivateBrowsingSession()
55{
56 ASSERT(isMainThread());
57 NetworkStorageSessionMap::destroySession(PAL::SessionID::legacyPrivateSessionID());
58}

60bool WebFrameNetworkingContext::localFileContentSniffingEnabled() const

65SchedulePairHashSet* WebFrameNetworkingContext::scheduledRunLoopPairs() const
66{
67 if (!frame() || !frame()->page())
68 return nullptr;
69 return frame()->page()->scheduledRunLoopPairs();
70}

72RetainPtr<CFDataRef> WebFrameNetworkingContext::sourceApplicationAuditData() const
73{
74 if (!frame() || !frame()->page())
75 return nullptr;
76
77 WebView *webview = kit(frame()->page());
78 if (!webview)
79 return nullptr;
80
81 return (__bridge CFDataRef)webview._sourceApplicationAuditData;
82}

94NetworkStorageSession* WebFrameNetworkingContext::storageSession() const
95{
96 ASSERT(isMainThread());
97 if (frame() && frame()->page() && frame()->page()->sessionID().isEphemeral()) {
98 if (auto* session = NetworkStorageSessionMap::storageSession(PAL::SessionID::legacyPrivateSessionID()))
99 return session;
100 // Some requests may still be coming shortly before WebCore updates the session ID and after WebKit destroys the private browsing session.
101 LOG_ERROR("Invalid session ID. Please file a bug unless you just disabled private browsing, in which case it's an expected race.");
102 }
103 return &NetworkStorageSessionMap::defaultStorageSession();
104}
