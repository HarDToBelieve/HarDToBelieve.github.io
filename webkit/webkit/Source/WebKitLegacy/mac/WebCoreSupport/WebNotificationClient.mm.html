50static uint64_t generateNotificationID()
51{
52 static uint64_t uniqueNotificationID = 1;
53 return uniqueNotificationID++;
54}

61bool WebNotificationClient::show(Notification* notification)
62{
63 if (![m_webView _notificationProvider])
64 return false;
65
66 uint64_t notificationID = generateNotificationID();
67 RetainPtr<WebNotification> webNotification = adoptNS([[WebNotification alloc] initWithCoreNotification:notification notificationID:notificationID]);
68 m_notificationMap.set(notification, webNotification);
69
70 auto it = m_notificationContextMap.add(notification->scriptExecutionContext(), Vector<RetainPtr<WebNotification>>()).iterator;
71 it->value.append(webNotification);
72
73 [[m_webView _notificationProvider] showNotification:webNotification.get() fromWebView:m_webView];
74 return true;
75}

77void WebNotificationClient::cancel(Notification* notification)
78{
79 WebNotification *webNotification = m_notificationMap.get(notification).get();
80 if (!webNotification)
81 return;
82
83 [[m_webView _notificationProvider] cancelNotification:webNotification];
84}

86void WebNotificationClient::clearNotifications(ScriptExecutionContext* context)
87{
88 auto it = m_notificationContextMap.find(context);
89 if (it == m_notificationContextMap.end())
90 return;
91
92 auto finalizedNotificationIDs = createNSArray(it->value, [&] (auto& notification) {
93 auto& coreNotification = *core(notification.get());
94 coreNotification.finalize();
95 m_notificationMap.remove(&coreNotification);
96 return @([notification notificationID]);
97 });
98
99 [[m_webView _notificationProvider] clearNotifications:finalizedNotificationIDs.get()];
100 m_notificationContextMap.remove(it);
101}

103void WebNotificationClient::notificationObjectDestroyed(Notification* notification)
104{
105 RetainPtr<WebNotification> webNotification = m_notificationMap.take(notification);
106 if (!webNotification)
107 return;
108
109 auto it = m_notificationContextMap.find(notification->scriptExecutionContext());
110 ASSERT(it != m_notificationContextMap.end());
111 size_t index = it->value.find(webNotification);
112 ASSERT(index != notFound);
113 it->value.remove(index);
114 if (it->value.isEmpty())
115 m_notificationContextMap.remove(it);
116
117 [[m_webView _notificationProvider] notificationDestroyed:webNotification.get()];
118}

125void WebNotificationClient::requestPermission(ScriptExecutionContext& context, WebNotificationPolicyListener *listener)
126{
127 SEL selector = @selector(webView:decidePolicyForNotificationRequestFromOrigin:listener:);
128 if (![[m_webView UIDelegate] respondsToSelector:selector])
129 return;
130
131 m_everRequestedPermission = true;
132
133 auto webOrigin = adoptNS([[WebSecurityOrigin alloc] _initWithWebCoreSecurityOrigin:context.securityOrigin()]);
134
135 CallUIDelegate(m_webView, selector, webOrigin.get(), listener);
136}

138void WebNotificationClient::requestPermission(ScriptExecutionContext& context, PermissionHandler&& permissionHandler)
139{
140 BEGIN_BLOCK_OBJC_EXCEPTIONS
141 auto listener = adoptNS([[WebNotificationPolicyListener alloc] initWithPermissionHandler:WTFMove(permissionHandler)]);
142 requestPermission(context, listener.get());
143 END_BLOCK_OBJC_EXCEPTIONS
144}

146NotificationClient::Permission WebNotificationClient::checkPermission(ScriptExecutionContext* context)
147{
148 if (!context || !context->isDocument())
149 return NotificationClient::Permission::Denied;
150 if (![[m_webView preferences] notificationsEnabled])
151 return NotificationClient::Permission::Denied;
152 auto webOrigin = adoptNS([[WebSecurityOrigin alloc] _initWithWebCoreSecurityOrigin:context->securityOrigin()]);
153 WebNotificationPermission permission = [[m_webView _notificationProvider] policyForOrigin:webOrigin.get()];
154 switch (permission) {
155 case WebNotificationPermissionAllowed:
156 return NotificationClient::Permission::Granted;
157 case WebNotificationPermissionDenied:
158 return NotificationClient::Permission::Denied;
159 case WebNotificationPermissionNotAllowed:
160 return NotificationClient::Permission::Default;
161 default:
162 return NotificationClient::Permission::Default;
163 }
164}

173- (id)initWithPermissionHandler:(NotificationClient::PermissionHandler&&)permissionHandler
174{
175 if (!(self = [super init]))
176 return nil;
177
178 _permissionHandler = WTFMove(permissionHandler);
179 return self;
180}
181
182- (void)allow
183{
184 if (_permissionHandler)
185 _permissionHandler(NotificationClient::Permission::Granted);
186}
187
188- (void)deny
189{
190 if (_permissionHandler)
191 _permissionHandler(NotificationClient::Permission::Denied);
192}
