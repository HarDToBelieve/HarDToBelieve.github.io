26#import "WebOpenPanelResultListener.h"
27
28#import <WebCore/FileChooser.h>
29#import <wtf/RefPtr.h>
30#import <wtf/cocoa/VectorCocoa.h>
31
32#if PLATFORM(IOS_FAMILY)
33#import <WebCore/Icon.h>
34#endif
35
36using namespace WebCore;
37
38@implementation WebOpenPanelResultListener
39
40- (id)initWithChooser:(FileChooser&)chooser
41{
42 self = [super init];
43 if (!self)
44 return nil;
45 _chooser = &chooser;
46 return self;
47}
48
49- (void)cancel
50{
51 _chooser = nullptr;
52}
53
54- (void)chooseFilename:(NSString *)filename
55{
56 ASSERT(_chooser);
57 if (!_chooser)
58 return;
59 _chooser->chooseFile(filename);
60 _chooser = nullptr;
61}
62
63- (void)chooseFilenames:(NSArray *)filenames
64{
65 ASSERT(_chooser);
66 if (!_chooser)
67 return;
68 _chooser->chooseFiles(makeVector<String>(filenames));
69 _chooser = nullptr;
70}
71
72#if PLATFORM(IOS_FAMILY)
73
74- (void)chooseFilename:(NSString *)filename displayString:(NSString *)displayString iconImage:(CGImageRef)imageRef
75{
76 [self chooseFilenames:@[filename] displayString:displayString iconImage:imageRef];
77}
78
79- (void)chooseFilenames:(NSArray *)filenames displayString:(NSString *)displayString iconImage:(CGImageRef)imageRef
80{
81 ASSERT(_chooser);
82 if (!_chooser)
83 return;
84
85 _chooser->chooseMediaFiles(makeVector<String>(filenames), displayString, Icon::createIconForImage(imageRef).get());
86 _chooser = nullptr;
87}
88
89#endif
90
91@end
