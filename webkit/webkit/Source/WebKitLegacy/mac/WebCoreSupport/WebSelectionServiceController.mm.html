43void WebSelectionServiceController::handleSelectionServiceClick(WebCore::FrameSelection& selection, const Vector<String>& /*telephoneNumbers*/, const WebCore::IntPoint& point)
44{
45 Page* page = [m_webView page];
46 if (!page)
47 return;
48
49 auto range = selection.selection().firstRange();
50 if (!range)
51 return;
52
53 auto attributedSelection = attributedString(*range).string;
54 if (!attributedSelection)
55 return;
56
57 auto items = @[ attributedSelection.get() ];
58 bool isEditable = selection.selection().isContentEditable();
59
60 m_sharingServicePickerController = adoptNS([[WebSharingServicePickerController alloc] initWithItems:items includeEditorServices:isEditable client:this style:NSSharingServicePickerStyleTextSelection]);
61
62 auto menu = adoptNS([[m_sharingServicePickerController menu] copy]);
63 [menu setShowsStateColumn:YES];
64 [menu popUpMenuPositioningItem:nil atLocation:[m_webView convertPoint:point toView:nil] inView:m_webView];
65}

67static bool hasCompatibleServicesForItems(NSArray *items)
68{
69 return [NSSharingService sharingServicesForItems:items mask:NSSharingServiceMaskViewer | NSSharingServiceMaskEditor].count;
70}
71
72bool WebSelectionServiceController::hasRelevantSelectionServices(bool isTextOnly) const
73{
74 RetainPtr<NSAttributedString> attributedString = adoptNS([[NSAttributedString alloc] initWithString:@"a"]);
75
76 bool hasSelectionServices = hasCompatibleServicesForItems(@[ attributedString.get() ]);
77 if (isTextOnly && hasSelectionServices)
78 return true;
79
80 auto attachment = adoptNS([[NSTextAttachment alloc] init]);
81 auto image = adoptNS([[NSImage alloc] init]);
82 auto cell = adoptNS([[NSTextAttachmentCell alloc] initImageCell:image.get()]);
83 [attachment setAttachmentCell:cell.get()];
84 NSMutableAttributedString *attributedStringWithRichContent = (NSMutableAttributedString *)[NSMutableAttributedString attributedStringWithAttachment:attachment.get()];
85 [attributedStringWithRichContent appendAttributedString:attributedString.get()];
86
87 return hasCompatibleServicesForItems(@[ attributedStringWithRichContent ]);
88}
