39static bool s_shouldTrackVisitedLinks;
40
41static HashSet<WebVisitedLinkStore*>& visitedLinkStores()
42{
43 static NeverDestroyed<HashSet<WebVisitedLinkStore*>> visitedLinkStores;
44
45 return visitedLinkStores;
46}

49Ref<WebVisitedLinkStore> WebVisitedLinkStore::create()
50{
51 return adoptRef(*new WebVisitedLinkStore);
52}
53
54WebVisitedLinkStore::WebVisitedLinkStore()
55 : m_visitedLinksPopulated(false)
56{
57 visitedLinkStores().add(this);
58}
59
60WebVisitedLinkStore::~WebVisitedLinkStore()
61{
62 visitedLinkStores().remove(this);
63}

65void WebVisitedLinkStore::setShouldTrackVisitedLinks(bool shouldTrackVisitedLinks)
66{
67 if (s_shouldTrackVisitedLinks == shouldTrackVisitedLinks)
68 return;
69 s_shouldTrackVisitedLinks = shouldTrackVisitedLinks;
70 if (!s_shouldTrackVisitedLinks)
71 removeAllVisitedLinks();
72}
73
74void WebVisitedLinkStore::removeAllVisitedLinks()
75{
76 for (auto& visitedLinkStore : visitedLinkStores())
77 visitedLinkStore->removeVisitedLinkHashes();
78}

80void WebVisitedLinkStore::addVisitedLink(NSString *urlString)
81{
82 if (!s_shouldTrackVisitedLinks)
83 return;
84
85 size_t length = urlString.length;
86
87 if (auto characters = CFStringGetCharactersPtr((__bridge CFStringRef)urlString)) {
88 addVisitedLinkHash(computeSharedStringHash(reinterpret_cast<const UChar*>(characters), length));
89 return;
90 }
91
92 Vector<UniChar, 512> buffer(length);
93 [urlString getCharacters:buffer.data()];
94
95 addVisitedLinkHash(computeSharedStringHash(reinterpret_cast<const UChar*>(buffer.data()), length));
96}
97
98void WebVisitedLinkStore::removeVisitedLink(NSString *urlString)
99{
100 auto linkHash = computeSharedStringHash(urlString);
101
102 ASSERT(m_visitedLinkHashes.contains(linkHash));
103 m_visitedLinkHashes.remove(linkHash);
104
105 invalidateStylesForLink(linkHash);
106}

108bool WebVisitedLinkStore::isLinkVisited(Page& page, SharedStringHash linkHash, const URL& baseURL, const AtomString& attributeURL)
109{
110 populateVisitedLinksIfNeeded(page);
111
112 return m_visitedLinkHashes.contains(linkHash);
113}
114
115void WebVisitedLinkStore::addVisitedLink(Page& sourcePage, SharedStringHash linkHash)
116{
117 if (!s_shouldTrackVisitedLinks)
118 return;
119
120 addVisitedLinkHash(linkHash);
121}
122
123void WebVisitedLinkStore::populateVisitedLinksIfNeeded(Page& page)
124{
125 if (m_visitedLinksPopulated)
126 return;
127
128 m_visitedLinksPopulated = true;
129
130 WebView *webView = kit(&page);
131 ASSERT(webView);
132
133 if (webView.historyDelegate) {
134 WebHistoryDelegateImplementationCache* implementations = WebViewGetHistoryDelegateImplementations(webView);
135
136 if (implementations->populateVisitedLinksFunc)
137 CallHistoryDelegate(implementations->populateVisitedLinksFunc, webView, @selector(populateVisitedLinksForWebView:));
138
139 return;
140 }
141
142 BEGIN_BLOCK_OBJC_EXCEPTIONS
143 [[WebHistory optionalSharedHistory] _addVisitedLinksToVisitedLinkStore:*this];
144 END_BLOCK_OBJC_EXCEPTIONS
145}
146
147void WebVisitedLinkStore::addVisitedLinkHash(SharedStringHash linkHash)
148{
149 ASSERT(s_shouldTrackVisitedLinks);
150
151 m_visitedLinkHashes.add(linkHash);
152
153 invalidateStylesForLink(linkHash);
154}
155
156void WebVisitedLinkStore::removeVisitedLinkHashes()
157{
158 m_visitedLinksPopulated = false;
159 if (m_visitedLinkHashes.isEmpty())
160 return;
161 m_visitedLinkHashes.clear();
162
163 invalidateStylesForAllLinks();
164}
