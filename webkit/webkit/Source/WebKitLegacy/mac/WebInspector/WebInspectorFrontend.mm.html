26#import "WebInspectorFrontend.h"
27
28#import "WebInspectorClient.h"
29#import <WebCore/InspectorFrontendClient.h>
30
31using namespace WebCore;
32
33@implementation WebInspectorFrontend
34
35- (id)initWithFrontendClient:(NakedPtr<WebInspectorFrontendClient>)frontendClient
36{
37 if (!(self = [super init]))
38 return nil;
39
40 m_frontendClient = frontendClient;
41 return self;
42}
43
44- (void)attach
45{
46 m_frontendClient->attachWindow(InspectorFrontendClient::DockSide::Bottom);
47}
48
49- (void)detach
50{
51 m_frontendClient->detachWindow();
52}
53
54- (void)close
55{
56 m_frontendClient->closeWindow();
57}
58
59- (BOOL)isDebuggingEnabled
60{
61 return m_frontendClient->isDebuggingEnabled();
62}
63
64- (void)setDebuggingEnabled:(BOOL)enabled
65{
66 m_frontendClient->setDebuggingEnabled(enabled);
67}
68
69- (BOOL)isProfilingJavaScript
70{
71 return m_frontendClient->isProfilingJavaScript();
72}
73
74- (void)startProfilingJavaScript
75{
76 m_frontendClient->startProfilingJavaScript();
77}
78
79- (void)stopProfilingJavaScript
80{
81 m_frontendClient->stopProfilingJavaScript();
82}
83
84- (BOOL)isTimelineProfilingEnabled
85{
86 return m_frontendClient->isTimelineProfilingEnabled();
87}
88
89- (void)setTimelineProfilingEnabled:(BOOL)enabled
90{
91 m_frontendClient->setTimelineProfilingEnabled(enabled);
92}
93
94- (void)showConsole
95{
96 m_frontendClient->showConsole();
97}
98
99@end
