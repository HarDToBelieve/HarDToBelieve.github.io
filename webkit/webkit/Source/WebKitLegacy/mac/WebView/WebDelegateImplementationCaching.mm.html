76// We use these functions to call the delegates and block exceptions. These functions are
77// declared inside a WebView category to get direct access to the delegate data memebers,
78// preventing more ObjC message dispatch and compensating for the expense of the @try/@catch.
79
80typedef float (*ObjCMsgSendFPRet)(id, SEL, ...);
81#if !PLATFORM(IOS_FAMILY)
82static const ObjCMsgSendFPRet objc_msgSend_float_return = reinterpret_cast<ObjCMsgSendFPRet>(objc_msgSend);
83#endif

85static inline id CallDelegate(WebView *self, id delegate, SEL selector)
86{
87#if !PLATFORM(IOS_FAMILY)
88 if (!delegate || ![delegate respondsToSelector:selector])
89 return nil;
90 @try {
91 return wtfObjCMsgSend<id>(delegate, selector, self);
92 } @catch(id exception) {
93 ReportDiscardedDelegateException(selector, exception);
94 }
95 return nil;
96#else
97 if (!delegate || ![delegate respondsToSelector:selector])
98 return nil;
99
100 NSInvocation *invocation = WebThreadMakeNSInvocation(delegate, selector);
101 [invocation setArgument:&self atIndex:2];
102
103 id returnValue;
104 @try {
105 WebThreadCallDelegate(invocation);
106 if ([[invocation methodSignature] methodReturnLength] == 0) {
107 return nil;
108 }
109 [invocation getReturnValue:&returnValue];
110 return returnValue;
111 } @catch(id exception) {
112 ReportDiscardedDelegateException(selector, exception);
113 }
114 return nil;
115#endif
116}
