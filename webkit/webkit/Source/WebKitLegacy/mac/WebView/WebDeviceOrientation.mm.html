26#import "WebDeviceOrientationInternal.h"

28using namespace WebCore;

30@implementation WebDeviceOrientationInternal
31
32- (id)initWithCoreDeviceOrientation:(RefPtr<DeviceOrientationData>&&)coreDeviceOrientation
33{
34 self = [super init];
35 if (!self)
36 return nil;
37 m_orientation = WTFMove(coreDeviceOrientation);
38 return self;
39}
40
41@end
42
43@implementation WebDeviceOrientation (Internal)
44
45- (id)initWithCoreDeviceOrientation:(RefPtr<WebCore::DeviceOrientationData>&&)coreDeviceOrientation
46{
47 self = [super init];
48 if (!self)
49 return nil;
50 m_internal = [[WebDeviceOrientationInternal alloc] initWithCoreDeviceOrientation:WTFMove(coreDeviceOrientation)];
51 return self;
52}
53
54@end
55
56@implementation WebDeviceOrientation
57
58DeviceOrientationData* core(WebDeviceOrientation* orientation)
59{
60 return orientation ? orientation->m_internal->m_orientation.get() : 0;
61}
62
63static std::optional<double> convert(bool canProvide, double value)
64{
65 if (!canProvide)
66 return std::nullopt;
67 return value;
68}
69
70- (id)initWithCanProvideAlpha:(bool)canProvideAlpha alpha:(double)alpha canProvideBeta:(bool)canProvideBeta beta:(double)beta canProvideGamma:(bool)canProvideGamma gamma:(double)gamma
71{
72 self = [super init];
73 if (!self)
74 return nil;
75#if PLATFORM(IOS_FAMILY)
76 // We don't use this API, but make sure that it compiles with the new
77 // compass parameters.
78 m_internal = [[WebDeviceOrientationInternal alloc] initWithCoreDeviceOrientation:DeviceOrientationData::create(convert(canProvideAlpha, alpha), convert(canProvideBeta, beta), convert(canProvideGamma, gamma), std::nullopt, std::nullopt)];
79#else
80 m_internal = [[WebDeviceOrientationInternal alloc] initWithCoreDeviceOrientation:DeviceOrientationData::create(convert(canProvideAlpha, alpha), convert(canProvideBeta, beta), convert(canProvideGamma, gamma), std::nullopt)];
81#endif
82 return self;
83}
84
85- (void)dealloc
86{
87 [m_internal release];
88 [super dealloc];
89}
90
91@end
