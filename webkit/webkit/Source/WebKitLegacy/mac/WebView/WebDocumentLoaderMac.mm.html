36WebDocumentLoaderMac::WebDocumentLoaderMac(const ResourceRequest& request, const SubstituteData& substituteData)
37 : DocumentLoader(request, substituteData)
38 , m_dataSource(nil)
39 , m_isDataSourceRetained(false)
40{
41}

43static inline bool needsDataLoadWorkaround(WebView *webView)
44{
45#if !PLATFORM(IOS_FAMILY)
46 static bool needsWorkaround = !WebKitLinkedOnOrAfter(WEBKIT_FIRST_VERSION_WITHOUT_ADOBE_INSTALLER_QUIRK)
47 && [[[NSBundle mainBundle] bundleIdentifier] isEqualToString:@"com.adobe.Installers.Setup"];
48 return needsWorkaround;
49#else
50 return NO;
51#endif
52}

54void WebDocumentLoaderMac::setDataSource(WebDataSource *dataSource, WebView *webView)
55{
56 ASSERT(!m_dataSource);
57 ASSERT(!m_isDataSourceRetained);
58
59 m_dataSource = dataSource;
60 retainDataSource();
61
62 m_resourceLoadDelegate = [webView resourceLoadDelegate];
63 m_downloadDelegate = [webView downloadDelegate];
64
65 // Some clients run the run loop in a way that prevents the data load timer
66 // from firing. We work around that issue here. See <rdar://problem/5266289>
67 // and <rdar://problem/5049509>.
68 if (needsDataLoadWorkaround(webView))
69 m_deferMainResourceDataLoad = false;
70}

84void WebDocumentLoaderMac::detachFromFrame()
85{
86 DocumentLoader::detachFromFrame();
87
88 if (m_loadingResources.isEmpty())
89 releaseDataSource();
90
91 // FIXME: What prevents the data source from getting deallocated while the

95void WebDocumentLoaderMac::increaseLoadCount(unsigned long identifier)
96{
97 ASSERT(m_dataSource);
98
99 if (m_loadingResources.contains(identifier))
100 return;
101 m_loadingResources.add(identifier);
102
103 retainDataSource();
104}
105
106void WebDocumentLoaderMac::decreaseLoadCount(unsigned long identifier)
107{
108 HashSet<unsigned long>::iterator it = m_loadingResources.find(identifier);
109
110 // It is valid for a load to be cancelled before it's started.
111 if (it == m_loadingResources.end())
112 return;
113
114 m_loadingResources.remove(it);
115
116 if (m_loadingResources.isEmpty()) {
117 m_resourceLoadDelegate = 0;
118 m_downloadDelegate = 0;
119 if (!frame())
120 releaseDataSource();
121 }
122}
123
124void WebDocumentLoaderMac::retainDataSource()
125{
126 if (m_isDataSourceRetained || !m_dataSource)
127 return;
128 m_isDataSourceRetained = true;
129 CFRetain(m_dataSource);
130}
131
132void WebDocumentLoaderMac::releaseDataSource()
133{
134 if (!m_isDataSourceRetained)
135 return;
136 ASSERT(m_dataSource);
137 m_isDataSourceRetained = false;
138 CFRelease(m_dataSource);
139}
140
141void WebDocumentLoaderMac::detachDataSource()
142{
143 ASSERT(!m_isDataSourceRetained);
144 m_dataSource = nil;
145}
