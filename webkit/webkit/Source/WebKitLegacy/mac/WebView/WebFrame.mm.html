297+ (Ref<WebCore::Frame>)_createFrameWithPage:(WebCore::Page*)page frameName:(const String&)name frameView:(WebFrameView *)frameView ownerElement:(WebCore::HTMLFrameOwnerElement*)ownerElement
298{
299 WebView *webView = kit(page);
300
301 RetainPtr<WebFrame> frame = adoptNS([[self alloc] _initWithWebFrameView:frameView webView:webView]);
302 auto coreFrame = WebCore::Frame::create(page, ownerElement, makeUniqueRef<WebFrameLoaderClient>(frame.get()));
303 frame->_private->coreFrame = coreFrame.ptr();
304
305 coreFrame.get().tree().setName(name);
306 if (ownerElement) {
307 ASSERT(ownerElement->document().frame());
308 ownerElement->document().frame()->tree().appendChild(coreFrame.get());
309 }
310
311 coreFrame.get().init();
312
313 [webView _setZoomMultiplier:[webView _realZoomMultiplier] isTextOnly:[webView _realZoomMultiplierIsTextOnly]];
314
315 return coreFrame;
316}

357- (void)_attachScriptDebugger
358{
359 auto& windowProxy = _private->coreFrame->windowProxy();
360
361 // Calling ScriptController::globalObject() would create a window proxy, and dispatch corresponding callbacks, which may be premature
362 // if the script debugger is attached before a document is created. These calls use the debuggerWorld(), we will need to pass a world
363 // to be able to debug isolated worlds.
364 if (!windowProxy.existingJSWindowProxy(WebCore::debuggerWorld()))
365 return;
366
367 auto* globalObject = windowProxy.globalObject(WebCore::debuggerWorld());
368 if (!globalObject)
369 return;
370
371 if (_private->scriptDebugger) {
372 ASSERT(_private->scriptDebugger.get() == globalObject->debugger());
373 return;
374 }
375
376 _private->scriptDebugger = makeUnique<WebScriptDebugger>(globalObject);
377}

384- (id)_initWithWebFrameView:(WebFrameView *)fv webView:(WebView *)v
385{
386 self = [super init];
387 if (!self)
388 return nil;
389
390 _private = [[WebFramePrivate alloc] init];
391
392 // Set includedInWebKitStatistics before calling WebFrameView _setWebFrame, since
393 // it calls WebFrame _isIncludedInWebKitStatistics.
394 if ((_private->includedInWebKitStatistics = [[v class] shouldIncludeInWebKitStatistics]))
395 ++WebFrameCount;
396
397 if (fv) {
398 [_private setWebFrameView:fv];
399 [fv _setWebFrame:self];
400 }
401
402 _private->shouldCreateRenderers = YES;
403
404 return self;
405}

418- (void)_updateBackgroundAndUpdatesWhileOffscreen
419{
420 WebView *webView = getWebView(self);
421 BOOL drawsBackground = [webView drawsBackground];
422#if !PLATFORM(IOS_FAMILY)
423 NSColor *backgroundColor = [webView backgroundColor];
424#else
425 CGColorRef backgroundColor = [webView backgroundColor];
426#endif
427
428 auto coreFrame = _private->coreFrame;
429 for (auto frame = coreFrame; frame; frame = frame->tree().traverseNext(coreFrame)) {
430 // Don't call setDrawsBackground:YES here because it may be NO because of a load
431 // in progress; WebFrameLoaderClient keeps it set to NO during the load process.
432 WebFrame *webFrame = kit(frame);
433 if (!drawsBackground)
434 [[[webFrame frameView] _scrollView] setDrawsBackground:NO];
435#if !PLATFORM(IOS_FAMILY)
436 [[[webFrame frameView] _scrollView] setBackgroundColor:backgroundColor];
437#endif
438
439 if (auto* view = frame->view()) {
440 view->setTransparent(!drawsBackground);
441#if !PLATFORM(IOS_FAMILY)
442 ALLOW_DEPRECATED_DECLARATIONS_BEGIN
443 WebCore::Color color = WebCore::colorFromNSColor([backgroundColor colorUsingColorSpaceName:NSDeviceRGBColorSpace]);
444 ALLOW_DEPRECATED_DECLARATIONS_END
445#else
446 WebCore::Color color = WebCore::Color(backgroundColor);
447#endif
448 view->setBaseBackgroundColor(color);
449 view->setShouldUpdateWhileOffscreen([webView shouldUpdateWhileOffscreen]);
450 }
451 }

578- (OptionSet<WebCore::PaintBehavior>)_paintBehaviorForDestinationContext:(CGContextRef)context
579{
580#if PLATFORM(MAC)
581 // -currentContextDrawingToScreen returns YES for bitmap contexts.
582 BOOL isPrinting = ![NSGraphicsContext currentContextDrawingToScreen];
583 if (isPrinting)
584 return OptionSet<WebCore::PaintBehavior>(WebCore::PaintBehavior::FlattenCompositingLayers) | WebCore::PaintBehavior::Snapshotting;
585#endif
586
587 if (CGContextGetType(context) != kCGContextTypeBitmap)
588 return WebCore::PaintBehavior::Normal;
589
590 // If we're drawing into a bitmap, we could be snapshotting or drawing into a layer-backed view.
591 if (WebHTMLView *documentView = [self _webHTMLDocumentView]) {
592#if PLATFORM(IOS_FAMILY)
593 return [[documentView window] isInSnapshottingPaint] ? WebCore::PaintBehavior::Snapshotting : WebCore::PaintBehavior::Normal;
594#endif
595#if PLATFORM(MAC)
596 // Even if we are layer-backed, we may be painting into an offscreen context.
597 // We can be sure it's an offscreen context if we are not parented in a window, so exclude that case.
598 // This does not cover all cases, such as a parented view being painted into an offscreen context.
599 if ([documentView _web_isDrawingIntoLayer] && documentView.window)
600 return WebCore::PaintBehavior::Normal;
601#endif
602 }
603
604 return OptionSet<WebCore::PaintBehavior>(WebCore::PaintBehavior::FlattenCompositingLayers) | WebCore::PaintBehavior::Snapshotting;
605}

607- (void)_drawRect:(NSRect)rect contentsOnly:(BOOL)contentsOnly
608{
609#if !PLATFORM(IOS_FAMILY)
610 ASSERT([[NSGraphicsContext currentContext] isFlipped]);
611
612 CGContextRef ctx = [[NSGraphicsContext currentContext] CGContext];
613#else
614 CGContextRef ctx = WKGetCurrentGraphicsContext();
615#endif
616 WebCore::GraphicsContextCG context(ctx);
617
618#if PLATFORM(IOS_FAMILY)
619 WebCore::Frame *frame = core(self);
620 if (WebCore::Page* page = frame->page())
621 context.setIsAcceleratedContext(page->settings().acceleratedDrawingEnabled());
622#elif PLATFORM(MAC)
623 if (WebHTMLView *htmlDocumentView = [self _webHTMLDocumentView])
624 context.setIsAcceleratedContext([htmlDocumentView _web_isDrawingIntoAcceleratedLayer]);
625#endif
626
627 auto* view = _private->coreFrame->view();
628
629 OptionSet<WebCore::PaintBehavior> oldBehavior = view->paintBehavior();
630 OptionSet<WebCore::PaintBehavior> paintBehavior = oldBehavior;
631
632 if (auto* parentFrame = _private->coreFrame->tree().parent()) {
633 // For subframes, we need to inherit the paint behavior from our parent
634 if (auto* parentView = parentFrame ? parentFrame->view() : nullptr) {
635 if (parentView->paintBehavior().contains(WebCore::PaintBehavior::FlattenCompositingLayers))
636 paintBehavior.add(WebCore::PaintBehavior::FlattenCompositingLayers);
637
638 if (parentView->paintBehavior().contains(WebCore::PaintBehavior::Snapshotting))
639 paintBehavior.add(WebCore::PaintBehavior::Snapshotting);
640
641 if (parentView->paintBehavior().contains(WebCore::PaintBehavior::TileFirstPaint))
642 paintBehavior.add(WebCore::PaintBehavior::TileFirstPaint);
643 }
644 } else
645 paintBehavior.add([self _paintBehaviorForDestinationContext:ctx]);
646
647 view->setPaintBehavior(paintBehavior);
648
649 if (contentsOnly)
650 view->paintContents(context, WebCore::enclosingIntRect(rect));
651 else
652 view->paint(context, WebCore::enclosingIntRect(rect));
653
654 view->setPaintBehavior(oldBehavior);
655}

675- (NSString *)_stringByEvaluatingJavaScriptFromString:(NSString *)string forceUserGesture:(BOOL)forceUserGesture
676{
677 if (!string)
678 return @"";
679
680 RELEASE_ASSERT(isMainThread());
681
682 ASSERT(_private->coreFrame->document());
683 RetainPtr<WebFrame> protect(self); // Executing arbitrary JavaScript can destroy the frame.
684
685#if PLATFORM(IOS_FAMILY)
686 ASSERT(WebThreadIsLockedOrDisabled());
687 JSC::JSGlobalObject* lexicalGlobalObject = _private->coreFrame->script().globalObject(WebCore::mainThreadNormalWorld());
688 JSC::JSLockHolder jscLock(lexicalGlobalObject);
689#endif
690
691 JSC::JSValue result = _private->coreFrame->script().executeScriptIgnoringException(string, forceUserGesture);
692
693 if (!_private->coreFrame) // In case the script removed our frame from the page.
694 return @"";
695
696 // This bizarre set of rules matches behavior from WebKit for Safari 2.0.

812- (std::optional<WebCore::SimpleRange>)_convertToDOMRange:(NSRange)range rangeIsRelativeTo:(WebRangeIsRelativeTo)rangeIsRelativeTo
813{
814 if (range.location == NSNotFound)
815 return std::nullopt;
816
817 if (rangeIsRelativeTo == WebRangeIsRelativeTo::EditableRoot) {
818 // Our critical assumption is that this code path is only called by input methods that
819 // concentrate on a given area containing the selection
820 // We have to do this because of text fields and textareas. The DOM for those is not
821 // directly in the document DOM, so serialization is problematic. Our solution is
822 // to use the root editable element of the selection start as the positional base.
823 // That fits with AppKit's idea of an input context.
824 auto* element = _private->coreFrame->selection().rootEditableElementOrDocumentElement();
825 if (!element)
826 return std::nullopt;
827 return resolveCharacterRange(makeRangeSelectingNodeContents(*element), range);
828 }
829
830 ASSERT(rangeIsRelativeTo == WebRangeIsRelativeTo::Paragraph);
831
832 auto paragraphStart = makeBoundaryPoint(startOfParagraph(_private->coreFrame->selection().selection().visibleStart()));
833 if (!paragraphStart)
834 return std::nullopt;
835
836 auto scopeEnd = makeBoundaryPointAfterNodeContents(paragraphStart->container->treeScope().rootNode());
837 return WebCore::resolveCharacterRange({ WTFMove(*paragraphStart), WTFMove(scopeEnd) }, range);
838}

1695// Collects the ranges and metadata for all of the mars voltas in the root editable element.
1696- (void)getDictationResultRanges:(NSArray **)outRanges andMetadatas:(NSArray **)outMetadatas
1697{
1698 ASSERT(outRanges);
1699 if (!outRanges)
1700 return;
1701
1702 // *outRanges should not already point to an array.
1703 ASSERT(!(*outRanges));
1704 *outRanges = nil;
1705
1706 ASSERT(outMetadatas);
1707 if (!outMetadatas)
1708 return;
1709
1710 // *metadata should not already point to an array.
1711 ASSERT(!(*outMetadatas));
1712 *outMetadatas = nil;
1713
1714 NSMutableArray *ranges = [NSMutableArray array];
1715 NSMutableArray *metadatas = [NSMutableArray array];
1716
1717 auto* frame = core(self);
1718 auto* document = frame->document();
1719
1720 auto& selection = frame->selection().selection();
1721 auto root = selection.isNone() ? frame->document()->bodyOrFrameset() : selection.rootEditableElement();
1722
1723 RetainPtr<DOMRange> previousDOMRange;
1724 id previousMetadata = nil;
1725
1726 for (WebCore::Node* node = root; node; node = WebCore::NodeTraversal::next(*node)) {
1727 auto markers = document->markers().markersFor(*node);
1728 for (auto* marker : markers) {
1729 if (marker->type() != WebCore::DocumentMarker::DictationResult)
1730 continue;
1731
1732 id metadata = WTF::get<RetainPtr<id>>(marker->data()).get();
1733
1734 // All result markers should have metadata.
1735 ASSERT(metadata);
1736 if (!metadata)
1737 continue;
1738
1739 DOMRange *domRange = kit(makeSimpleRange(*node, *marker));
1740
1741 if (metadata != previousMetadata) {
1742 [metadatas addObject:metadata];
1743 [ranges addObject:domRange];
1744 previousMetadata = metadata;
1745 previousDOMRange = domRange;
1746 } else {
1747 // It is possible for a DocumentMarker to be split by editing. Adjacent markers with the
1748 // the same metadata are for the same result. So combine their ranges.
1749 ASSERT(previousDOMRange == [ranges lastObject]);
1750 [ranges removeLastObject];
1751 DOMNode *startContainer = [domRange startContainer];
1752 int startOffset = [domRange startOffset];
1753 [previousDOMRange setEnd:startContainer offset:startOffset];
1754 [ranges addObject:previousDOMRange.get()];
1755 }
1756 }
1757 }
1758
1759 *outRanges = ranges;
1760 *outMetadatas = metadatas;
1761
1762 return;
1763}

1969// Determines whether whitespace needs to be added around aString to preserve proper spacing and
1970// punctuation when it's inserted into the receiver's text over charRange. Returns by reference
1971// in beforeString and afterString any whitespace that should be added, unless either or both are
1972// nil. Both are returned as nil if aString is nil or if smart insertion and deletion are disabled.
1973- (void)_smartInsertForString:(NSString *)pasteString replacingRange:(DOMRange *)rangeToReplace beforeString:(NSString **)beforeString afterString:(NSString **)afterString
1974{
1975 // give back nil pointers in case of early returns
1976 if (beforeString)
1977 *beforeString = nil;
1978 if (afterString)
1979 *afterString = nil;
1980
1981 auto range = makeSimpleRange(core(rangeToReplace));
1982 if (!range)
1983 return;
1984
1985 auto start = WebCore::VisiblePosition { makeContainerOffsetPosition(range->start) };
1986 auto end = WebCore::VisiblePosition { makeContainerOffsetPosition(range->end) };
1987
1988 bool addLeadingSpace = start.deepEquivalent().leadingWhitespacePosition(WebCore::VisiblePosition::defaultAffinity, true).isNull() && !isStartOfParagraph(start);
1989 if (addLeadingSpace) {
1990 if (UChar previousCharacter = start.previous().characterAfter())
1991 addLeadingSpace = !WebCore::isCharacterSmartReplaceExempt(previousCharacter, true);
1992 }
1993
1994 bool addTrailingSpace = end.deepEquivalent().trailingWhitespacePosition(WebCore::VisiblePosition::defaultAffinity, true).isNull() && !isEndOfParagraph(end);
1995 if (addTrailingSpace) {
1996 if (UChar followingCharacter = end.characterAfter())
1997 addTrailingSpace = !WebCore::isCharacterSmartReplaceExempt(followingCharacter, false);
1998 }
1999
2000 // inspect source
2001 bool hasWhitespaceAtStart = false;
2002 bool hasWhitespaceAtEnd = false;
2003 unsigned pasteLength = [pasteString length];
2004 if (pasteLength > 0) {
2005 NSCharacterSet *whiteSet = [NSCharacterSet whitespaceAndNewlineCharacterSet];
2006
2007 if ([whiteSet characterIsMember:[pasteString characterAtIndex:0]]) {
2008 hasWhitespaceAtStart = YES;
2009 }
2010 if ([whiteSet characterIsMember:[pasteString characterAtIndex:(pasteLength - 1)]]) {
2011 hasWhitespaceAtEnd = YES;
2012 }
2013 }
2014
2015 // issue the verdict
2016 if (beforeString && addLeadingSpace && !hasWhitespaceAtStart)
2017 *beforeString = @" ";
2018 if (afterString && addTrailingSpace && !hasWhitespaceAtEnd)
2019 *afterString = @" ";
2020}
