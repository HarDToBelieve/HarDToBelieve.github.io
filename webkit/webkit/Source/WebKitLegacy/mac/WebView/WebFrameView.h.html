29#import <Foundation/Foundation.h>
30#import <WebKitLegacy/WebKitAvailability.h>
31
32#if !TARGET_OS_IPHONE
33#import <AppKit/AppKit.h>
34#else
35#import <WebKitLegacy/WAKAppKitStubs.h>
36#import <WebKitLegacy/WAKView.h>
37#endif
38
39@class WebDataSource;
40@class WebFrame;
41@class WebFrameViewPrivate;
42
43@protocol WebDocumentView;
44
45/*!
46 @class WebFrameView
47*/
48WEBKIT_CLASS_DEPRECATED_MAC(10_3, 10_14)
49@interface WebFrameView : NSView
50{
51@package
52 WebFrameViewPrivate *_private;
53}
54
55/*!
56 @property webFrame
57 @abstract The WebFrame associated with this WebFrameView
58*/
59@property (nonatomic, readonly, strong) WebFrame *webFrame;
60
61/*!
62 @property documentView
63 @abstract The WebFrameView's document subview
64 @discussion The subview that renders the WebFrameView's contents
65*/
66@property (nonatomic, readonly, strong) NSView<WebDocumentView> *documentView;
67
68/*!
69 @property allowsScrolling
70 @abstract Whether the WebFrameView allows its document to be scrolled
71*/
72@property (nonatomic) BOOL allowsScrolling;
73
74#if !TARGET_OS_IPHONE
75/*!
76 @property canPrintHeadersAndFooters
77 @abstract Whether this frame can print headers and footers
78*/
79@property (nonatomic, readonly) BOOL canPrintHeadersAndFooters;
80
81/*!
82 @method printOperationWithPrintInfo
83 @abstract Creates a print operation set up to print this frame
84 @result A newly created print operation object
85*/
86- (NSPrintOperation *)printOperationWithPrintInfo:(NSPrintInfo *)printInfo;
87#endif
88
89/*!
90 @property documentViewShouldHandlePrint
91 @abstract Called by the host application before it initializes and runs a print operation.
92 @discussion If NO is returned, the host application will abort its print operation and call -printDocumentView on the
93 WebFrameView. The document view is then expected to run its own print operation. If YES is returned, the host
94 application's print operation will continue as normal.
95*/
96@property (nonatomic, readonly) BOOL documentViewShouldHandlePrint;
97
98/*!
99 @method printDocumentView
100 @abstract Called by the host application when the WebFrameView returns YES from -documentViewShouldHandlePrint.
101*/
102- (void)printDocumentView;
103
104@end
