26#import "WebGeolocationPosition.h"
27
28#import "WebGeolocationPositionInternal.h"
29#import <WebCore/GeolocationPositionData.h>
30#import <wtf/RefPtr.h>
31
32using namespace WebCore;
33
34@interface WebGeolocationPositionInternal : NSObject
35{
36@public
37 GeolocationPositionData _position;
38}
39
40- (id)initWithCoreGeolocationPosition:(GeolocationPositionData&&)coreGeolocationPosition;
41@end
42
43@implementation WebGeolocationPositionInternal
44
45- (id)initWithCoreGeolocationPosition:(GeolocationPositionData&&)coreGeolocationPosition
46{
47 self = [super init];
48 if (!self)
49 return nil;
50 _position = WTFMove(coreGeolocationPosition);
51 return self;
52}
53
54@end
55
56@implementation WebGeolocationPosition
57
58std::optional<GeolocationPositionData> core(WebGeolocationPosition *position)
59{
60 if (!position)
61 return std::nullopt;
62 return position->_internal->_position;
63}
64
65- (id)initWithTimestamp:(double)timestamp latitude:(double)latitude longitude:(double)longitude accuracy:(double)accuracy
66{
67 self = [super init];
68 if (!self)
69 return nil;
70 _internal = [[WebGeolocationPositionInternal alloc] initWithCoreGeolocationPosition:GeolocationPositionData { timestamp, latitude, longitude, accuracy }];
71 return self;
72}
73
74- (id)initWithGeolocationPosition:(GeolocationPositionData&&)coreGeolocationPosition
75{
76 self = [super init];
77 if (!self)
78 return nil;
79 _internal = [[WebGeolocationPositionInternal alloc] initWithCoreGeolocationPosition:WTFMove(coreGeolocationPosition)];
80 return self;
81}
82
83- (void)dealloc
84{
85 [_internal release];
86 [super dealloc];
87}
88
89@end
