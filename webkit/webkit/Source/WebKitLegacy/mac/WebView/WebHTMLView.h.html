39/*!
40 @class WebHTMLView
41 @discussion A document view of WebFrameView that displays HTML content.
42 WebHTMLView is a NSControl because it hosts NSCells that are painted by WebCore's Aqua theme
43 renderer (and those cells must be hosted by an enclosing NSControl in order to paint properly).
44*/
45#if !TARGET_OS_IPHONE
46@interface WebHTMLView : NSControl <WebDocumentView, WebDocumentSearching>
47#else
48@interface WebHTMLView : NSView <WebDocumentView, WebDocumentSearching>
49#endif
50{
51@private
52 WebHTMLViewPrivate *_private;
53}
