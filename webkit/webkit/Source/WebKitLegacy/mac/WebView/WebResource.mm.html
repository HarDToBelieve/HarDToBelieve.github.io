116- (instancetype)initWithCoder:(NSCoder *)decoder
117{
118 WebCoreThreadViolationCheckRoundTwo();
119
120 self = [super init];
121 if (!self)
122 return nil;
123
124 NSData *data = nil;
125 NSURL *url = nil;
126 NSString *mimeType = nil, *textEncoding = nil, *frameName = nil;
127 NSURLResponse *response = nil;
128
129 @try {
130 id object = [decoder decodeObjectForKey:WebResourceDataKey];
131 if ([object isKindOfClass:[NSData class]])
132 data = object;
133 object = [decoder decodeObjectForKey:WebResourceURLKey];
134 if ([object isKindOfClass:[NSURL class]])
135 url = object;
136 object = [decoder decodeObjectForKey:WebResourceMIMETypeKey];
137 if ([object isKindOfClass:[NSString class]])
138 mimeType = object;
139 object = [decoder decodeObjectForKey:WebResourceTextEncodingNameKey];
140 if ([object isKindOfClass:[NSString class]])
141 textEncoding = object;
142 object = [decoder decodeObjectForKey:WebResourceFrameNameKey];
143 if ([object isKindOfClass:[NSString class]])
144 frameName = object;
145 object = [decoder decodeObjectForKey:WebResourceResponseKey];
146 if ([object isKindOfClass:[NSURLResponse class]])
147 response = object;
148 } @catch(id) {
149 [self release];
150 return nil;
151 }
152
153 auto coreResource = ArchiveResource::create(SharedBuffer::create(data), url, mimeType, textEncoding, frameName, response);
154 if (!coreResource) {
155 [self release];
156 return nil;
157 }
158
159 _private = [[WebResourcePrivate alloc] initWithCoreResource:coreResource.releaseNonNull()];
160 return self;
161}

163- (void)encodeWithCoder:(NSCoder *)encoder
164{
165 auto* resource = _private->coreResource.get();
166
167 NSData *data = nil;
168 NSURL *url = nil;
169 NSString *mimeType = nil, *textEncoding = nil, *frameName = nil;
170 NSURLResponse *response = nil;
171
172 if (resource) {
173 data = resource->data().createNSData().get();
174 url = resource->url();
175 mimeType = resource->mimeType();
176 textEncoding = resource->textEncoding();
177 frameName = resource->frameName();
178 response = resource->response().nsURLResponse();
179 }
180 [encoder encodeObject:data forKey:WebResourceDataKey];
181 [encoder encodeObject:url forKey:WebResourceURLKey];
182 [encoder encodeObject:mimeType forKey:WebResourceMIMETypeKey];
183 [encoder encodeObject:textEncoding forKey:WebResourceTextEncodingNameKey];
184 [encoder encodeObject:frameName forKey:WebResourceFrameNameKey];
185 [encoder encodeObject:response forKey:WebResourceResponseKey];
186}

286- (id)_initWithData:(NSData *)data
287 URL:(NSURL *)URL
288 MIMEType:(NSString *)MIMEType
289 textEncodingName:(NSString *)textEncodingName
290 frameName:(NSString *)frameName
291 response:(NSURLResponse *)response
292 copyData:(BOOL)copyData
293{
294 WebCoreThreadViolationCheckRoundTwo();
295
296 self = [super init];
297 if (!self)
298 return nil;
299
300 if (!data || !URL || !MIMEType) {
301 [self release];
302 return nil;
303 }
304
305 auto coreResource = ArchiveResource::create(SharedBuffer::create(copyData ? adoptNS([data copy]).get() : data), URL, MIMEType, textEncodingName, frameName, response);
306 if (!coreResource) {
307 [self release];
308 return nil;
309 }
310
311 _private = [[WebResourcePrivate alloc] initWithCoreResource:coreResource.releaseNonNull()];
312 return self;
313}
314
315- (id)_initWithData:(NSData *)data URL:(NSURL *)URL response:(NSURLResponse *)response
316{
317 // Pass NO for copyData since the data doesn't need to be copied since we know that callers will no longer modify it.
318 // Copying it will also cause a performance regression.
319 return [self _initWithData:data
320 URL:URL
321 MIMEType:[response MIMEType]
322 textEncodingName:[response textEncodingName]
323 frameName:nil
324 response:response
325 copyData:NO];
326}

350- (NSURLResponse *)_response
351{
352 WebCoreThreadViolationCheckRoundTwo();
353
354 NSURLResponse *response = nil;
355 if (_private->coreResource)
356 response = _private->coreResource->response().nsURLResponse();
357 return response ? response : adoptNS([[NSURLResponse alloc] init]).autorelease();
358}
359
360- (NSString *)_stringValue
361{
362 WebCoreThreadViolationCheckRoundTwo();
363
364 WebCore::TextEncoding encoding;
365 if (_private->coreResource)
366 encoding = _private->coreResource->textEncoding();
367 if (!encoding.isValid())
368 encoding = WindowsLatin1Encoding();
369
370 SharedBuffer* coreData = _private->coreResource ? &_private->coreResource->data() : nullptr;
371 return encoding.decode(reinterpret_cast<const char*>(coreData ? coreData->data() : nullptr), coreData ? coreData->size() : 0);
372}
