34@interface WebScriptWorldPrivate : NSObject {
35@public
36 RefPtr<WebCore::DOMWrapperWorld> world;
37}
38@end

40@implementation WebScriptWorldPrivate
41@end
42
43typedef HashMap<WebCore::DOMWrapperWorld*, WebScriptWorld*> WorldMap;
44static WorldMap& allWorlds()
45{
46 static WorldMap& map = *new WorldMap;
47 return map;
48}
49
50@implementation WebScriptWorld
51
52- (id)initWithWorld:(Ref<WebCore::DOMWrapperWorld>&&)world
53{
54 self = [super init];
55 if (!self)
56 return nil;
57
58 _private = [[WebScriptWorldPrivate alloc] init];
59 _private->world = WTFMove(world);
60
61 ASSERT_ARG(world, !allWorlds().contains(_private->world.get()));
62 allWorlds().add(_private->world.get(), self);
63
64 return self;
65}
66
67- (id)init
68{
69 return [self initWithWorld:WebCore::ScriptController::createWorld("WebScriptWorld"_s, WebCore::ScriptController::WorldType::User)];
70}
71
72- (void)unregisterWorld
73{
74 _private->world->clearWrappers();
75}
76
77- (void)dealloc
78{
79 ASSERT(allWorlds().contains(_private->world.get()));
80 allWorlds().remove(_private->world.get());
81
82 [_private release];
83 _private = nil;
84 [super dealloc];
85}

87+ (WebScriptWorld *)standardWorld
88{
89 static WebScriptWorld *world = [[WebScriptWorld alloc] initWithWorld:WebCore::mainThreadNormalWorld()];
90 return world;
91}
92
93+ (WebScriptWorld *)world
94{
95 return adoptNS([[self alloc] init]).autorelease();
96}

98+ (WebScriptWorld *)scriptWorldForGlobalContext:(JSGlobalContextRef)context
99{
100 return [self findOrCreateWorld:WebCore::currentWorld(*toJS(context))];
101}

103#if JSC_OBJC_API_ENABLED
104+ (WebScriptWorld *)scriptWorldForJavaScriptContext:(JSContext *)context

107}

110@end

112@implementation WebScriptWorld (WebInternal)
113
114WebCore::DOMWrapperWorld* core(WebScriptWorld *world)
115{
116 return world ? world->_private->world.get() : 0;
117}
118
119+ (WebScriptWorld *)findOrCreateWorld:(WebCore::DOMWrapperWorld&)world
120{
121 if (&world == &WebCore::mainThreadNormalWorld())
122 return [self standardWorld];
123
124 if (WebScriptWorld *existingWorld = allWorlds().get(&world))
125 return existingWorld;
126
127 return adoptNS([[self alloc] initWithWorld:world]).autorelease();
128}

130@end
