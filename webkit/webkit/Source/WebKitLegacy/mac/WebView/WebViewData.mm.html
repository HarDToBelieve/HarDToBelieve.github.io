67static CFRunLoopRef currentRunLoop()
68{
69#if PLATFORM(IOS_FAMILY)
70 // A race condition during WebView deallocation can lead to a crash if the layer sync run loop
71 // observer is added to the main run loop <rdar://problem/9798550>. However, for responsiveness,
72 // we still allow this, see <rdar://problem/7403328>. Since the race condition and subsequent
73 // crash are especially troublesome for iBooks, we never allow the observer to be added to the
74 // main run loop in iBooks.
75 if (WebCore::CocoaApplication::isIBooks())
76 return WebThreadRunLoop();
77#endif
78 return CFRunLoopGetCurrent();
79}

99WebViewLayerFlushScheduler::WebViewLayerFlushScheduler(LayerFlushController* flushController)
100 : m_flushController(flushController)
101{
102 m_runLoopObserver = makeUnique<WebCore::RunLoopObserver>(static_cast<CFIndex>(WebCore::RunLoopObserver::WellKnownRunLoopOrders::LayerFlush), [this]() {
103 this->layerFlushCallback();
104 });
105}

109}

111void WebViewLayerFlushScheduler::schedule()
112{
113 if (m_insideCallback)
114 m_rescheduledInsideCallback = true;
115
116 m_runLoopObserver->schedule(currentRunLoop());
117}
118
119void WebViewLayerFlushScheduler::invalidate()
120{
121 m_runLoopObserver->invalidate();
122}
123
124void WebViewLayerFlushScheduler::layerFlushCallback()
125{
126#if PLATFORM(IOS_FAMILY)
127 // Normally the layer flush callback happens before the web lock auto-unlock observer runs.
128 // However if the flush is rescheduled from the callback it may get pushed past it, to the next cycle.
129 WebThreadLock();
130#endif
131
132 @autoreleasepool {
133 RefPtr<LayerFlushController> protector = m_flushController;
134
135 SetForScope<bool> insideCallbackScope(m_insideCallback, true);
136 m_rescheduledInsideCallback = false;
137
138 if (m_flushController->flushLayers() && !m_rescheduledInsideCallback)
139 invalidate();
140 }
141}

157- (void)startObserving:(NSWindow *)window
158{
159 // An NSView derived object such as WebView cannot observe these notifications, because NSView itself observes them.
160 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(_windowVisibilityChanged:)
161 name:@"NSWindowDidOrderOffScreenNotification" object:window];
162 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(_windowVisibilityChanged:)
163 name:@"_NSWindowDidBecomeVisible" object:window];
164}
165
166- (void)stopObserving:(NSWindow *)window
167{
168 [[NSNotificationCenter defaultCenter] removeObserver:self name:@"NSWindowDidOrderOffScreenNotification" object:window];
169 [[NSNotificationCenter defaultCenter] removeObserver:self name:@"_NSWindowDidBecomeVisible" object:window];
170}

174 [_view _windowVisibilityChanged:notification];
