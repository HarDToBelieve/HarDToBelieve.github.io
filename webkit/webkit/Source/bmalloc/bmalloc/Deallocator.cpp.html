38namespace bmalloc {
39
40Deallocator::Deallocator(Heap& heap)
41 : m_heap(heap)
42{
43 BASSERT(!Environment::get()->isDebugHeapEnabled());
44}
45
46Deallocator::~Deallocator()
47{
48 scavenge();
49}
50
51void Deallocator::scavenge()
52{
53 UniqueLockHolder lock(Heap::mutex());
54
55 processObjectLog(lock);
56 m_heap.deallocateLineCache(lock, lineCache(lock));
57}
58
59void Deallocator::processObjectLog(UniqueLockHolder& lock)
60{
61 for (Object object : m_objectLog)
62 m_heap.derefSmallLine(lock, object, lineCache(lock));
63 m_objectLog.clear();
64}
65
66void Deallocator::deallocateSlowCase(void* object)
67{
68 if (!object)
69 return;
70
71 if (m_heap.isLarge(object)) {
72 UniqueLockHolder lock(Heap::mutex());
73 m_heap.deallocateLarge(lock, object);
74 return;
75 }
76
77 if (m_objectLog.size() == m_objectLog.capacity()) {
78 UniqueLockHolder lock(Heap::mutex());
79 processObjectLog(lock);
80 }
81
82 m_objectLog.push(object);
83}
