50struct Callback {
51 Callback() { }
52
53 Callback(void (*function)(void*), void *argument)
54 : function(function)
55 , argument(argument)
56 {
57 }
58
59 void (*function)(void*) { nullptr };
60 void* argument { nullptr };
61};

79// This is exactly 32GB because inside JSC, indexed accesses for arrays, typed arrays, etc,
80// use unsigned 32-bit ints as indices. The items those indices access are 8 bytes or less
81// in size. 2^32 * 8 = 32GB. This means if an access on a caged type happens to go out of
82// bounds, the access is guaranteed to land somewhere else in the cage or inside the runway.
83// If this were less than 32GB, those OOB accesses could reach outside of the cage.
84constexpr size_t gigacageRunway = 32llu * bmalloc::Sizes::GB;
85
86bool disablePrimitiveGigacageRequested = false;

92size_t runwaySize(Kind kind)
93{
94 switch (kind) {
95 case Kind::Primitive:
96 return gigacageRunway;
97 case Kind::JSValue:
98 return 0;
99 case Kind::NumberOfKinds:
100 RELEASE_BASSERT_NOT_REACHED();
101 }
102 return 0;
103}

107void ensureGigacage()
108{
109 static std::once_flag onceFlag;
110 std::call_once(
111 onceFlag,
112 [] {
113 RELEASE_BASSERT(!g_gigacageConfig.ensureGigacageHasBeenCalled);
114 g_gigacageConfig.ensureGigacageHasBeenCalled = true;
115
116 if (!shouldBeEnabled())
117 return;
118
119#if BENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)
120 // We might only get page size alignment, but that's also the minimum
121 // alignment we need for freezing the Config.
122 RELEASE_BASSERT(!(reinterpret_cast<size_t>(&g_gigacageConfig) & (vmPageSize() - 1)));
123#endif
124
125 Kind shuffledKinds[NumberOfKinds];
126 for (unsigned i = 0; i < NumberOfKinds; ++i)
127 shuffledKinds[i] = static_cast<Kind>(i);
128
129 // We just go ahead and assume that 64 bits is enough randomness. That's trivially true right
130 // now, but would stop being true if we went crazy with gigacages. Based on my math, 21 is the
131 // largest value of n so that n! <= 2^64.
132 static_assert(NumberOfKinds <= 21, "too many kinds");
133 uint64_t random;
134 cryptoRandom(reinterpret_cast<unsigned char*>(&random), sizeof(random));
135 for (unsigned i = NumberOfKinds; i--;) {
136 unsigned limit = i + 1;
137 unsigned j = static_cast<unsigned>(random % limit);
138 random /= limit;
139 std::swap(shuffledKinds[i], shuffledKinds[j]);
140 }
141
142 auto alignTo = [] (Kind kind, size_t totalSize) -> size_t {
143 return roundUpToMultipleOf(alignment(kind), totalSize);
144 };
145 auto bump = [] (Kind kind, size_t totalSize) -> size_t {
146 return totalSize + maxSize(kind);
147 };
148
149 size_t totalSize = 0;
150 size_t maxAlignment = 0;
151
152 for (Kind kind : shuffledKinds) {
153 totalSize = bump(kind, alignTo(kind, totalSize));
154 totalSize += runwaySize(kind);
155 maxAlignment = std::max(maxAlignment, alignment(kind));
156 }
157
158 // FIXME: Randomize where this goes.
159 // https://bugs.webkit.org/show_bug.cgi?id=175245
160 void* base = tryVMAllocate(maxAlignment, totalSize, VMTag::JSGigacage);
161 if (!base) {
162 if (GIGACAGE_ALLOCATION_CAN_FAIL)
163 return;
164 fprintf(stderr, "FATAL: Could not allocate gigacage memory with maxAlignment = %lu, totalSize = %lu.\n", maxAlignment, totalSize);
165 fprintf(stderr, "(Make sure you have not set a virtual memory limit.)\n");
166 BCRASH();
167 }
168
169 size_t nextCage = 0;
170 for (Kind kind : shuffledKinds) {
171 nextCage = alignTo(kind, nextCage);
172 g_gigacageConfig.setBasePtr(kind, reinterpret_cast<char*>(base) + nextCage);
173 nextCage = bump(kind, nextCage);
174 if (runwaySize(kind) > 0) {
175 char* runway = reinterpret_cast<char*>(base) + nextCage;
176 // Make OOB accesses into the runway crash.
177 vmRevokePermissions(runway, runwaySize(kind));
178 nextCage += runwaySize(kind);
179 }
180 }
181
182 g_gigacageConfig.start = base;
183 g_gigacageConfig.totalSize = totalSize;
184 vmDeallocatePhysicalPages(base, totalSize);
185 g_gigacageConfig.isEnabled = true;
186 });
187}

189void disablePrimitiveGigacage()
190{
191 if (g_gigacageConfig.disablingPrimitiveGigacageIsForbidden)
192 fprintf(stderr, "FATAL: Disabling Primitive gigacage is forbidden, but we don't want that in this process.\n");
193
194 RELEASE_BASSERT(!g_gigacageConfig.disablingPrimitiveGigacageIsForbidden);
195
196 ensureGigacage();
197 disablePrimitiveGigacageRequested = true;
198 if (!g_gigacageConfig.basePtrs[Primitive]) {
199 // It was never enabled. That means that we never even saved any callbacks. Or, we had already disabled
200 // it before, and already called the callbacks.
201 return;
202 }
203
204 PrimitiveDisableCallbacks& callbacks = *PrimitiveDisableCallbacks::get();
205 UniqueLockHolder lock(PrimitiveDisableCallbacks::mutex());
206 for (Callback& callback : callbacks.callbacks)
207 callback.function(callback.argument);
208 callbacks.callbacks.shrink(0);
209}
210
211void addPrimitiveDisableCallback(void (*function)(void*), void* argument)
212{
213 ensureGigacage();
214 if (!g_gigacageConfig.basePtrs[Primitive]) {
215 // It was already disabled or we were never able to enable it.
216 function(argument);
217 return;
218 }
219
220 PrimitiveDisableCallbacks& callbacks = *PrimitiveDisableCallbacks::get();
221 UniqueLockHolder lock(PrimitiveDisableCallbacks::mutex());
222 callbacks.callbacks.push(Callback(function, argument));
223}
224
225void removePrimitiveDisableCallback(void (*function)(void*), void* argument)
226{
227 PrimitiveDisableCallbacks& callbacks = *PrimitiveDisableCallbacks::get();
228 UniqueLockHolder lock(PrimitiveDisableCallbacks::mutex());
229 for (size_t i = 0; i < callbacks.callbacks.size(); ++i) {
230 if (callbacks.callbacks[i].function == function
231 && callbacks.callbacks[i].argument == argument) {
232 callbacks.callbacks[i] = callbacks.callbacks.last();
233 callbacks.callbacks.pop();
234 return;
235 }
236 }
237}

239static bool verifyGigacageIsEnabled()
240{
241 bool isEnabled = g_gigacageConfig.isEnabled;
242 for (size_t i = 0; i < NumberOfKinds; ++i)
243 isEnabled = isEnabled && g_gigacageConfig.basePtrs[i];
244 isEnabled = isEnabled && g_gigacageConfig.start;
245 isEnabled = isEnabled && g_gigacageConfig.totalSize;
246 return isEnabled;
247}
248
249void forbidDisablingPrimitiveGigacage()
250{
251 ensureGigacage();
252 RELEASE_BASSERT(g_gigacageConfig.shouldBeEnabledHasBeenCalled
253 && (GIGACAGE_ALLOCATION_CAN_FAIL || !g_gigacageConfig.shouldBeEnabled || verifyGigacageIsEnabled()));
254
255 if (!g_gigacageConfig.disablingPrimitiveGigacageIsForbidden)
256 g_gigacageConfig.disablingPrimitiveGigacageIsForbidden = true;
257 RELEASE_BASSERT(disablingPrimitiveGigacageIsForbidden());
258}
259
260bool shouldBeEnabled()
261{
262 static std::once_flag onceFlag;
263 std::call_once(
264 onceFlag,
265 [] {
266 RELEASE_BASSERT(!g_gigacageConfig.shouldBeEnabledHasBeenCalled);
267 g_gigacageConfig.shouldBeEnabledHasBeenCalled = true;
268
269 bool debugHeapEnabled = Environment::get()->isDebugHeapEnabled();
270 if (debugHeapEnabled)
271 return;
272
273 if (!gigacageEnabledForProcess())
274 return;
275
276 if (char* gigacageEnabled = getenv("GIGACAGE_ENABLED")) {
277 if (!strcasecmp(gigacageEnabled, "no") || !strcasecmp(gigacageEnabled, "false") || !strcasecmp(gigacageEnabled, "0")) {
278 fprintf(stderr, "Warning: disabling gigacage because GIGACAGE_ENABLED=%s!\n", gigacageEnabled);
279 return;
280 } else if (strcasecmp(gigacageEnabled, "yes") && strcasecmp(gigacageEnabled, "true") && strcasecmp(gigacageEnabled, "1"))
281 fprintf(stderr, "Warning: invalid argument to GIGACAGE_ENABLED: %s\n", gigacageEnabled);
282 }
283
284 g_gigacageConfig.shouldBeEnabled = true;
285 });
286 return g_gigacageConfig.shouldBeEnabled;
287}
