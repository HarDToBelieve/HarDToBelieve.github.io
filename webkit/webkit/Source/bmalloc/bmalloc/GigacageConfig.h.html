44namespace Gigacage {
45
46struct Config {
47 void* basePtr(Kind kind) const
48 {
49 RELEASE_BASSERT(kind < NumberOfKinds);
50 return basePtrs[kind];
51 }
52
53 void setBasePtr(Kind kind, void* ptr)
54 {
55 RELEASE_BASSERT(kind < NumberOfKinds);
56 basePtrs[kind] = ptr;
57 }
58
59 // All the fields in this struct should be chosen such that their
60 // initial value is 0 / null / falsy because Config is instantiated
61 // as a global singleton.
62
63 bool isPermanentlyFrozen; // Will be set by the client if the Config gets frozen.
64 bool isEnabled;
65 bool disablingPrimitiveGigacageIsForbidden;
66 bool shouldBeEnabled;
67
68 // We would like to just put the std::once_flag for these functions
69 // here, but we can't because std::once_flag has a implicitly-deleted
70 // default constructor. So, we use a boolean instead.
71 bool shouldBeEnabledHasBeenCalled;
72 bool ensureGigacageHasBeenCalled;
73
74 void* start;
75 size_t totalSize;
76 void* basePtrs[NumberOfKinds];
77};

79#if BENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)

81constexpr size_t startSlotOfGigacageConfig = 0;
82constexpr size_t startOffsetOfGigacageConfig = startSlotOfGigacageConfig * sizeof(WebConfig::Slot);
83
84constexpr size_t reservedSlotsForGigacageConfig = 6;
85constexpr size_t reservedBytesForGigacageConfig = reservedSlotsForGigacageConfig * sizeof(WebConfig::Slot);
86
87constexpr size_t alignmentOfGigacageConfig = std::alignment_of<Gigacage::Config>::value;
88
89static_assert(sizeof(Gigacage::Config) <= reservedBytesForGigacageConfig);
90static_assert(bmalloc::roundUpToMultipleOf<alignmentOfGigacageConfig>(startOffsetOfGigacageConfig) == startOffsetOfGigacageConfig);
91
92#define g_gigacageConfig (*bmalloc::bitwise_cast<Gigacage::Config*>(&WebConfig::g_config[Gigacage::startSlotOfGigacageConfig]))
93
