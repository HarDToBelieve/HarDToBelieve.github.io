156inline void Heap::allocateSmallBumpRanges(
157 UniqueLockHolder& lock, size_t sizeClass,
158 BumpAllocator& allocator, BumpRangeCache& rangeCache,
159 LineCache& lineCache, FailureAction action)
160{
161 if (sizeClass < bmalloc::sizeClass(smallLineSize))
162 return allocateSmallBumpRangesByMetadata(lock, sizeClass, allocator, rangeCache, lineCache, action);
163 return allocateSmallBumpRangesByObject(lock, sizeClass, allocator, rangeCache, lineCache, action);
164}
165
166inline void Heap::derefSmallLine(UniqueLockHolder& lock, Object object, LineCache& lineCache)
167{
168 if (!object.line()->deref(lock))
169 return;
170 deallocateSmallLine(lock, object, lineCache);
171}
172
173inline bool Heap::isLarge(void* object)
174{
175 return m_objectTypes.get(Object(object).chunk()) == ObjectType::Large;
176}
