49static constexpr bool verbose = false;

51struct PrintTime {
52 PrintTime(const char* str)
53 : string(str)
54 { }
55
56 ~PrintTime()
57 {
58 if (!printed)
59 print();
60 }
61 void print()
62 {
63 if (verbose) {
64 fprintf(stderr, "%s %lfms\n", string, static_cast<double>(std::chrono::duration_cast<std::chrono::microseconds>(std::chrono::steady_clock::now() - start).count()) / 1000);
65 printed = true;
66 }
67 }
68 const char* string;
69 std::chrono::steady_clock::time_point start { std::chrono::steady_clock::now() };
70 bool printed { false };
71};

75Scavenger::Scavenger(const LockHolder&)
76{
77 BASSERT(!Environment::get()->isDebugHeapEnabled());
78
79#if BOS(DARWIN)
80 auto queue = dispatch_queue_create("WebKit Malloc Memory Pressure Handler", DISPATCH_QUEUE_SERIAL);
81 m_pressureHandlerDispatchSource = dispatch_source_create(DISPATCH_SOURCE_TYPE_MEMORYPRESSURE, 0, DISPATCH_MEMORYPRESSURE_CRITICAL, queue);
82 dispatch_source_set_event_handler(m_pressureHandlerDispatchSource, ^{
83 scavenge();
84 });
85 dispatch_resume(m_pressureHandlerDispatchSource);
86 dispatch_release(queue);
87#endif
88#if BUSE(PARTIAL_SCAVENGE)
89 m_waitTime = std::chrono::milliseconds(m_isInMiniMode ? 200 : 2000);
90#else
91 m_waitTime = std::chrono::milliseconds(10);
92#endif
93
94 m_thread = std::thread(&threadEntryPoint, this);
95}

98{
99 LockHolder lock(mutex());
100 run(lock);
101}

103void Scavenger::run(const LockHolder&)
104{
105 m_state = State::Run;
106 m_condition.notify_all();
107}

109void Scavenger::runSoon()
110{
111 LockHolder lock(mutex());
112 runSoon(lock);
113}
114
115void Scavenger::runSoon(const LockHolder&)
116{
117 if (willRunSoon())
118 return;
119 m_state = State::RunSoon;
120 m_condition.notify_all();
121}

129void Scavenger::scheduleIfUnderMemoryPressure(size_t bytes)
130{
131 LockHolder lock(mutex());
132 scheduleIfUnderMemoryPressure(lock, bytes);
133}
134
135void Scavenger::scheduleIfUnderMemoryPressure(const LockHolder& lock, size_t bytes)
136{
137 m_scavengerBytes += bytes;
138 if (m_scavengerBytes < scavengerBytesPerMemoryPressureCheck)
139 return;
140
141 m_scavengerBytes = 0;
142
143 if (willRun())
144 return;
145
146 if (!isUnderMemoryPressure())
147 return;
148
149 m_isProbablyGrowing = false;
150 run(lock);
151}
152
153void Scavenger::schedule(size_t bytes)
154{
155 LockHolder lock(mutex());
156 scheduleIfUnderMemoryPressure(lock, bytes);
157
158 if (willRunSoon())
159 return;
160
161 m_isProbablyGrowing = false;
162 runSoon(lock);
163}

165inline void dumpStats()
166{
167 auto dump = [] (auto* string, auto size) {
168 fprintf(stderr, "%s %zuMB\n", string, static_cast<size_t>(size) / 1024 / 1024);
169 };
170
171#if BOS(DARWIN)
172 task_vm_info_data_t vmInfo;
173 mach_msg_type_number_t vmSize = TASK_VM_INFO_COUNT;
174 if (KERN_SUCCESS == task_info(mach_task_self(), TASK_VM_INFO, (task_info_t)(&vmInfo), &vmSize)) {
175 dump("phys_footprint", vmInfo.phys_footprint);
176 dump("internal+compressed", vmInfo.internal + vmInfo.compressed);
177 }
178#endif
179
180 dump("bmalloc-freeable", Scavenger::get()->freeableMemory());
181 dump("bmalloc-footprint", Scavenger::get()->footprint());
182}

198void Scavenger::enableMiniMode()
199{
200 m_isInMiniMode = true; // We just store to this racily. The scavenger thread will eventually pick up the right value.
201 if (m_state == State::RunSoon)
202 run();

205void Scavenger::scavenge()
206{
207 if (!m_isEnabled)
208 return;
209
210 UniqueLockHolder lock(m_scavengingMutex);
211
212 if (verbose) {
213 fprintf(stderr, "--------------------------------\n");
214 fprintf(stderr, "--before scavenging--\n");
215 dumpStats();
216 }
217
218 {
219 BulkDecommit decommitter;
220
221 {
222 PrintTime printTime("\nfull scavenge under lock time");
223#if !BUSE(PARTIAL_SCAVENGE)
224 size_t deferredDecommits = 0;
225#endif
226 UniqueLockHolder lock(Heap::mutex());
227 for (unsigned i = numHeaps; i--;) {
228 if (!isActiveHeapKind(static_cast<HeapKind>(i)))
229 continue;
230#if BUSE(PARTIAL_SCAVENGE)
231 PerProcess<PerHeapKind<Heap>>::get()->at(i).scavenge(lock, decommitter);
232#else
233 PerProcess<PerHeapKind<Heap>>::get()->at(i).scavenge(lock, decommitter, deferredDecommits);
234#endif
235 }
236 decommitter.processEager();
237
238#if !BUSE(PARTIAL_SCAVENGE)
239 if (deferredDecommits)
240 m_state = State::RunSoon;
241#endif
242 }
243
244 {
245 PrintTime printTime("full scavenge lazy decommit time");
246 decommitter.processLazy();
247 }
248
249 {
250 PrintTime printTime("full scavenge mark all as eligible time");
251 LockHolder lock(Heap::mutex());
252 for (unsigned i = numHeaps; i--;) {
253 if (!isActiveHeapKind(static_cast<HeapKind>(i)))
254 continue;
255 PerProcess<PerHeapKind<Heap>>::get()->at(i).markAllLargeAsEligibile(lock);
256 }
257 }
258 }
259
260 {
261 RELEASE_BASSERT(!m_deferredDecommits.size());
262 AllIsoHeaps::get()->forEach(
263 [&] (IsoHeapImplBase& heap) {
264 heap.scavenge(m_deferredDecommits);
265 });
266 IsoHeapImplBase::finishScavenging(m_deferredDecommits);
267 m_deferredDecommits.shrink(0);
268 }
269
270 if (verbose) {
271 fprintf(stderr, "--after scavenging--\n");
272 dumpStats();
273 fprintf(stderr, "--------------------------------\n");
274 }
275
276 {
277 UniqueLockHolder lock(mutex());
278 m_lastFullScavengeTime = std::chrono::steady_clock::now();
279 }
280}

282#if BUSE(PARTIAL_SCAVENGE)
283void Scavenger::partialScavenge()
284{
285 if (!m_isEnabled)
286 return;
287
288 UniqueLockHolder lock(m_scavengingMutex);
289
290 if (verbose) {
291 fprintf(stderr, "--------------------------------\n");
292 fprintf(stderr, "--before partial scavenging--\n");
293 dumpStats();
294 }
295
296 {
297 BulkDecommit decommitter;
298 {
299 PrintTime printTime("\npartialScavenge under lock time");
300 UniqueLockHolder lock(Heap::mutex());
301 for (unsigned i = numHeaps; i--;) {
302 if (!isActiveHeapKind(static_cast<HeapKind>(i)))
303 continue;
304 Heap& heap = PerProcess<PerHeapKind<Heap>>::get()->at(i);
305 size_t freeableMemory = heap.freeableMemory(lock);
306 if (freeableMemory < 4 * MB)
307 continue;
308 heap.scavengeToHighWatermark(lock, decommitter);
309 }
310
311 decommitter.processEager();
312 }
313
314 {
315 PrintTime printTime("partialScavenge lazy decommit time");
316 decommitter.processLazy();
317 }
318
319 {
320 PrintTime printTime("partialScavenge mark all as eligible time");
321 LockHolder lock(Heap::mutex());
322 for (unsigned i = numHeaps; i--;) {
323 if (!isActiveHeapKind(static_cast<HeapKind>(i)))
324 continue;
325 Heap& heap = PerProcess<PerHeapKind<Heap>>::get()->at(i);
326 heap.markAllLargeAsEligibile(lock);
327 }
328 }
329 }
330
331 {
332 RELEASE_BASSERT(!m_deferredDecommits.size());
333 AllIsoHeaps::get()->forEach(
334 [&] (IsoHeapImplBase& heap) {
335 heap.scavengeToHighWatermark(m_deferredDecommits);
336 });
337 IsoHeapImplBase::finishScavenging(m_deferredDecommits);
338 m_deferredDecommits.shrink(0);
339 }
340
341 if (verbose) {
342 fprintf(stderr, "--after partial scavenging--\n");
343 dumpStats();
344 fprintf(stderr, "--------------------------------\n");
345 }
346
347 {
348 UniqueLockHolder lock(mutex());
349 m_lastPartialScavengeTime = std::chrono::steady_clock::now();
350 }
351}
352#endif

354size_t Scavenger::freeableMemory()
355{
356 size_t result = 0;
357 {
358 UniqueLockHolder lock(Heap::mutex());
359 for (unsigned i = numHeaps; i--;) {
360 if (!isActiveHeapKind(static_cast<HeapKind>(i)))
361 continue;
362 result += PerProcess<PerHeapKind<Heap>>::get()->at(i).freeableMemory(lock);
363 }
364 }
365
366 AllIsoHeaps::get()->forEach(
367 [&] (IsoHeapImplBase& heap) {
368 result += heap.freeableMemory();
369 });
370
371 return result;
372}

374size_t Scavenger::footprint()
375{
376 RELEASE_BASSERT(!Environment::get()->isDebugHeapEnabled());
377
378 size_t result = 0;
379 for (unsigned i = numHeaps; i--;) {
380 if (!isActiveHeapKind(static_cast<HeapKind>(i)))
381 continue;
382 result += PerProcess<PerHeapKind<Heap>>::get()->at(i).footprint();
383 }
384
385 AllIsoHeaps::get()->forEach(
386 [&] (IsoHeapImplBase& heap) {
387 result += heap.footprint();
388 });
389
390 return result;
391}

398void Scavenger::threadRunLoop()
399{
400 setSelfQOSClass();
401#if BOS(DARWIN)
402 setThreadName("JavaScriptCore bmalloc scavenger");
403#else
404 setThreadName("BMScavenger");
405#endif
406
407 // This loop ratchets downward from most active to least active state. While
408 // we ratchet downward, any other thread may reset our state.
409
410 // We require any state change while we are sleeping to signal to our
411 // condition variable and wake us up.
412
413 while (true) {
414 if (m_state == State::Sleep) {
415 UniqueLockHolder lock(mutex());
416 m_condition.wait(lock, [&]() { return m_state != State::Sleep; });
417 }
418
419 if (m_state == State::RunSoon) {
420 UniqueLockHolder lock(mutex());
421 m_condition.wait_for(lock, m_waitTime, [&]() { return m_state != State::RunSoon; });
422 }
423
424 m_state = State::Sleep;
425
426 setSelfQOSClass();
427
428 if (verbose) {
429 fprintf(stderr, "--------------------------------\n");
430 fprintf(stderr, "considering running scavenger\n");
431 dumpStats();
432 fprintf(stderr, "--------------------------------\n");
433 }
434
435#if BUSE(PARTIAL_SCAVENGE)
436 enum class ScavengeMode {
437 None,
438 Partial,
439 Full
440 };
441
442 size_t freeableMemory = this->freeableMemory();
443
444 ScavengeMode scavengeMode = [&] {
445 auto timeSinceLastFullScavenge = this->timeSinceLastFullScavenge();
446 auto timeSinceLastPartialScavenge = this->timeSinceLastPartialScavenge();
447 auto timeSinceLastScavenge = std::min(timeSinceLastPartialScavenge, timeSinceLastFullScavenge);
448
449 if (isUnderMemoryPressure() && freeableMemory > 1 * MB && timeSinceLastScavenge > std::chrono::milliseconds(5))
450 return ScavengeMode::Full;
451
452 if (!m_isProbablyGrowing) {
453 if (timeSinceLastFullScavenge < std::chrono::milliseconds(1000) && !m_isInMiniMode)
454 return ScavengeMode::Partial;
455 return ScavengeMode::Full;
456 }
457
458 if (m_isInMiniMode) {
459 if (timeSinceLastFullScavenge < std::chrono::milliseconds(200))
460 return ScavengeMode::Partial;
461 return ScavengeMode::Full;
462 }
463
464#if BCPU(X86_64)
465 auto partialScavengeInterval = std::chrono::milliseconds(12000);
466#else
467 auto partialScavengeInterval = std::chrono::milliseconds(8000);
468#endif
469 if (timeSinceLastScavenge < partialScavengeInterval) {
470 // Rate limit partial scavenges.
471 return ScavengeMode::None;
472 }
473 if (freeableMemory < 25 * MB)
474 return ScavengeMode::None;
475 if (5 * freeableMemory < footprint())
476 return ScavengeMode::None;
477 return ScavengeMode::Partial;
478 }();
479
480 m_isProbablyGrowing = false;
481
482 switch (scavengeMode) {
483 case ScavengeMode::None: {
484 runSoon();
485 break;
486 }
487 case ScavengeMode::Partial: {
488 partialScavenge();
489 runSoon();
490 break;
491 }
492 case ScavengeMode::Full: {
493 scavenge();
494 break;
495 }
496 }
497#else
498 std::chrono::steady_clock::time_point start { std::chrono::steady_clock::now() };
499
500 scavenge();
501
502 auto timeSpentScavenging = std::chrono::steady_clock::now() - start;
503
504 if (verbose) {
505 fprintf(stderr, "time spent scavenging %lfms\n",
506 static_cast<double>(std::chrono::duration_cast<std::chrono::microseconds>(timeSpentScavenging).count()) / 1000);
507 }
508
509 // FIXME: We need to investigate mini-mode's adjustment.
510 // https://bugs.webkit.org/show_bug.cgi?id=203987
511 if (!m_isInMiniMode) {
512 timeSpentScavenging *= 150;
513 std::chrono::milliseconds newWaitTime = std::chrono::duration_cast<std::chrono::milliseconds>(timeSpentScavenging);
514 m_waitTime = std::min(std::max(newWaitTime, std::chrono::milliseconds(100)), std::chrono::milliseconds(10000));
515 }
516
517 if (verbose)
518 fprintf(stderr, "new wait time %lldms\n", static_cast<long long int>(m_waitTime.count()));
519#endif
520 }
521}

523void Scavenger::setThreadName(const char* name)
524{
525 BUNUSED(name);
526#if BOS(DARWIN) || BPLATFORM(PLAYSTATION)
527 pthread_setname_np(name);
528#elif BOS(LINUX)
529 // Truncate the given name since Linux limits the size of the thread name 16 including null terminator.
530 std::array<char, 16> buf;
531 strncpy(buf.data(), name, buf.size() - 1);
532 buf[buf.size() - 1] = '\0';
533 pthread_setname_np(pthread_self(), buf.data());
534#endif
535}
