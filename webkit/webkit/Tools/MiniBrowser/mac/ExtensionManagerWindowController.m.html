26#import "ExtensionManagerWindowController.h"
27
28#import "AppDelegate.h"
29#import <WebKit/WKUserContentControllerPrivate.h>
30#import <WebKit/_WKUserContentExtensionStore.h>
31
32@implementation ExtensionManagerWindowController
33
34- (instancetype)init
35{
36 self = [self initWithWindowNibName:@"ExtensionManagerWindowController"];
37 if (self) {
38 NSArray* installedContentExtensions = [[NSUserDefaults standardUserDefaults] arrayForKey:@"InstalledContentExtensions"];
39 if (installedContentExtensions) {
40 for (NSString *identifier in installedContentExtensions) {
41 [[_WKUserContentExtensionStore defaultStore] lookupContentExtensionForIdentifier:identifier completionHandler:^(_WKUserContentFilter *extension, NSError *error)
42 {
43 if (error) {
44 NSLog(@"Extension store got out of sync with system defaults.");
45 return;
46 }
47
48 BrowserAppDelegate* appDelegate = [[NSApplication sharedApplication] browserAppDelegate];
49 [appDelegate.userContentContoller _addUserContentFilter:extension];
50 }];
51 }
52
53 }
54 }
55 return self;
56}
57
58- (void)windowDidLoad
59{
60 [super windowDidLoad];
61
62 NSArray* installedContentExtensions = [[NSUserDefaults standardUserDefaults] arrayForKey:@"InstalledContentExtensions"];
63 if (installedContentExtensions) {
64 for (NSString *extension in installedContentExtensions)
65 [arrayController addObject:extension];
66 }
67}
68
69- (IBAction)add:(id)sender
70{
71 NSOpenPanel *openPanel = [[NSOpenPanel openPanel] retain];
72#pragma clang diagnostic push
73#pragma clang diagnostic ignored "-Wdeprecated-declarations"
74 openPanel.allowedFileTypes = @[ @"public.json" ];
75#pragma clang diagnostic pop
76
77 [openPanel beginSheetModalForWindow:self.window completionHandler:^(NSInteger result)
78 {
79 if (result != NSModalResponseOK)
80 return;
81
82 NSURL *url = [openPanel.URLs objectAtIndex:0];
83 NSString *identifier = url.lastPathComponent;
84 NSString *jsonString = [[NSString alloc] initWithContentsOfURL:url encoding:NSUTF8StringEncoding error:nil];
85
86 [[_WKUserContentExtensionStore defaultStore] compileContentExtensionForIdentifier:identifier encodedContentExtension:jsonString completionHandler:^(_WKUserContentFilter *extension, NSError *error)
87 {
88
89 if (error) {
90 NSAlert *alert = [NSAlert alertWithError:error];
91 [alert runModal];
92 return;
93 }
94
95 NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
96
97 NSArray* installedContentExtensions = [defaults arrayForKey:@"InstalledContentExtensions"];
98 NSMutableArray *mutableInstalledContentExtensions;
99 if (installedContentExtensions)
100 mutableInstalledContentExtensions = [installedContentExtensions mutableCopy];
101 else
102 mutableInstalledContentExtensions = [[NSMutableArray alloc] init];
103
104 [mutableInstalledContentExtensions addObject:identifier];
105 [defaults setObject:mutableInstalledContentExtensions forKey:@"InstalledContentExtensions"];
106 [mutableInstalledContentExtensions release];
107
108 [arrayController addObject:identifier];
109
110 BrowserAppDelegate* appDelegate = [[NSApplication sharedApplication] browserAppDelegate];
111 [appDelegate.userContentContoller _addUserContentFilter:extension];
112 }];
113 }];
114
115}
116
117- (IBAction)remove:(id)sender
118{
119 NSUInteger index = [arrayController selectionIndex];
120
121 NSString *identifierToRemove = arrayController.arrangedObjects[index];
122
123 [[_WKUserContentExtensionStore defaultStore] removeContentExtensionForIdentifier:identifierToRemove completionHandler:^(NSError *error)
124 {
125 if (error) {
126 NSAlert *alert = [NSAlert alertWithError:error];
127 [alert runModal];
128 return;
129 }
130
131 NSUserDefaults *defaults = [NSUserDefaults standardUserDefaults];
132
133 NSMutableArray *installedContentExtensions = [[defaults arrayForKey:@"InstalledContentExtensions"] mutableCopy];
134 [installedContentExtensions removeObject:identifierToRemove];
135 [defaults setObject:installedContentExtensions forKey:@"InstalledContentExtensions"];
136 [installedContentExtensions release];
137
138 [arrayController removeObjectAtArrangedObjectIndex:index];
139 BrowserAppDelegate* appDelegate = [[NSApplication sharedApplication] browserAppDelegate];
140 [appDelegate.userContentContoller _removeUserContentFilter:identifierToRemove];
141 }];
142}
