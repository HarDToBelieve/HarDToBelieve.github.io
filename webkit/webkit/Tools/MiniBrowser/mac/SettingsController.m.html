90- (instancetype)initWithMenu:(NSMenu *)menu
91{
92 self = [super init];
93 if (!self)
94 return nil;
95
96 _menu = [menu retain];
97
98 NSArray *onByDefaultPrefs = @[
99 UseWebKit2ByDefaultPreferenceKey,
100 AcceleratedDrawingEnabledPreferenceKey,
101 SubpixelAntialiasedLayerTextEnabledPreferenceKey,
102 LargeImageAsyncDecodingEnabledPreferenceKey,
103 AnimatedImageAsyncDecodingEnabledPreferenceKey,
104 WebViewFillsWindowKey,
105 ResourceLoadStatisticsEnabledPreferenceKey,
106 ];
107
108 NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
109 for (NSString *prefName in onByDefaultPrefs) {
110 if (![userDefaults objectForKey:prefName])
111 [userDefaults setBool:YES forKey:prefName];
112 }
113
114 [self _populateMenu];
115
116 return self;
117}

308- (void)buildUserAgentsMenu:(NSMenu *)menu
309{
310 NSDictionary* defaultUAInfo = @{
311 @"label" : @"Default",
312 @"identifier" : @"default",
313 };
314
315 NSMenuItem *menuItem = [[NSMenuItem alloc] initWithTitle:@"Default" action:@selector(changeCutomUserAgent:) keyEquivalent:@""];
316 [menuItem setTarget:self];
317 [menuItem setRepresentedObject:defaultUAInfo];
318
319 [menu addItem:menuItem];
320 [menuItem release];
321
322 [menu addItem:[NSMenuItem separatorItem]];
323
324 for (NSDictionary *userAgentData in [[self class] userAgentData]) {
325 NSString *name = userAgentData[@"label"];
326
327 if ([name isEqualToString:@"-"]) {
328 [menu addItem:[NSMenuItem separatorItem]];
329 continue;
330 }
331
332 menuItem = [[NSMenuItem alloc] initWithTitle:name action:@selector(changeCutomUserAgent:) keyEquivalent:@""];
333 [menuItem setTarget:self];
334 [menuItem setRepresentedObject:userAgentData];
335 [menu addItem:menuItem];
336 [menuItem release];
337 }

342- (BOOL)validateMenuItem:(NSMenuItem *)menuItem
343#pragma GCC diagnostic pop
344{
345 SEL action = [menuItem action];
346
347 if (action == @selector(toggleUseWebKit2ByDefault:))
348 [menuItem setState:[self useWebKit2ByDefault] ? NSControlStateValueOn : NSControlStateValueOff];
349 else if (action == @selector(toggleCreateEditorByDefault:))
350 [menuItem setState:[self createEditorByDefault] ? NSControlStateValueOn : NSControlStateValueOff];
351 else if (action == @selector(toggleUseTransparentWindows:))
352 [menuItem setState:[self useTransparentWindows] ? NSControlStateValueOn : NSControlStateValueOff];
353 else if (action == @selector(toggleUsePaginatedMode:))
354 [menuItem setState:[self usePaginatedMode] ? NSControlStateValueOn : NSControlStateValueOff];
355 else if (action == @selector(toggleShowLayerBorders:))
356 [menuItem setState:[self layerBordersVisible] ? NSControlStateValueOn : NSControlStateValueOff];
357 else if (action == @selector(toggleLegacyLineLayoutVisualCoverageEnabled:))
358 [menuItem setState:[self legacyLineLayoutVisualCoverageEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
359 else if (action == @selector(toggleIncrementalRenderingSuppressed:))
360 [menuItem setState:[self incrementalRenderingSuppressed] ? NSControlStateValueOn : NSControlStateValueOff];
361 else if (action == @selector(toggleAcceleratedDrawingEnabled:))
362 [menuItem setState:[self acceleratedDrawingEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
363 else if (action == @selector(toggleDisplayListDrawingEnabled:))
364 [menuItem setState:[self displayListDrawingEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
365 else if (action == @selector(toggleSubpixelAntialiasedLayerTextEnabled:))
366 [menuItem setState:[self subpixelAntialiasedLayerTextEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
367 else if (action == @selector(toggleResourceLoadStatisticsEnabled:))
368 [menuItem setState:[self resourceLoadStatisticsEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
369 else if (action == @selector(toggleLargeImageAsyncDecodingEnabled:))
370 [menuItem setState:[self largeImageAsyncDecodingEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
371 else if (action == @selector(toggleAnimatedImageAsyncDecodingEnabled:))
372 [menuItem setState:[self animatedImageAsyncDecodingEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
373 else if (action == @selector(toggleAppleColorFilterEnabled:))
374 [menuItem setState:[self appleColorFilterEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
375 else if (action == @selector(togglePunchOutWhiteBackgroundsInDarkMode:))
376 [menuItem setState:[self punchOutWhiteBackgroundsInDarkMode] ? NSControlStateValueOn : NSControlStateValueOff];
377 else if (action == @selector(toggleUseSystemAppearance:))
378 [menuItem setState:[self useSystemAppearance] ? NSControlStateValueOn : NSControlStateValueOff];
379 else if (action == @selector(toggleReserveSpaceForBanners:))
380 [menuItem setState:[self isSpaceReservedForBanners] ? NSControlStateValueOn : NSControlStateValueOff];
381 else if (action == @selector(toggleShowTiledScrollingIndicator:))
382 [menuItem setState:[self tiledScrollingIndicatorVisible] ? NSControlStateValueOn : NSControlStateValueOff];
383 else if (action == @selector(toggleShowResourceUsageOverlay:))
384 [menuItem setState:[self resourceUsageOverlayVisible] ? NSControlStateValueOn : NSControlStateValueOff];
385 else if (action == @selector(toggleLoadsAllSiteIcons:))
386 [menuItem setState:[self loadsAllSiteIcons] ? NSControlStateValueOn : NSControlStateValueOff];
387 else if (action == @selector(toggleUsesGameControllerFramework:))
388 [menuItem setState:[self usesGameControllerFramework] ? NSControlStateValueOn : NSControlStateValueOff];
389 else if (action == @selector(toggleNetworkCacheSpeculativeRevalidationDisabled:))
390 [menuItem setState:[self networkCacheSpeculativeRevalidationDisabled] ? NSControlStateValueOn : NSControlStateValueOff];
391 else if (action == @selector(toggleProcessSwapOnWindowOpenWithOpener:))
392 [menuItem setState:[self processSwapOnWindowOpenWithOpenerEnabled] ? NSControlStateValueOn : NSControlStateValueOff];
393 else if (action == @selector(toggleUseUISideCompositing:))
394 [menuItem setState:[self useUISideCompositing] ? NSControlStateValueOn : NSControlStateValueOff];
395 else if (action == @selector(togglePerWindowWebProcessesDisabled:))
396 [menuItem setState:[self perWindowWebProcessesDisabled] ? NSControlStateValueOn : NSControlStateValueOff];
397 else if (action == @selector(toggleDebugOverlay:))
398 [menuItem setState:[self debugOverlayVisible:menuItem] ? NSControlStateValueOn : NSControlStateValueOff];
399 else if (action == @selector(changeCutomUserAgent:)) {
400
401 NSString *savedUAIdentifier = [[NSUserDefaults standardUserDefaults] stringForKey:CustomUserAgentPreferenceKey];
402 NSDictionary *userAgentDict = [menuItem representedObject];
403 if (userAgentDict) {
404 BOOL isCurrentUA = [userAgentDict[@"identifier"] isEqualToString:savedUAIdentifier];
405 [menuItem setState:isCurrentUA ? NSControlStateValueOn : NSControlStateValueOff];
406 } else
407 [menuItem setState:NSControlStateValueOff];
408 }
409
410 WKPreferences *defaultPreferences = [[NSApplication sharedApplication] browserAppDelegate].defaultPreferences;
411 if (menuItem.tag == ExperimentalFeatureTag) {
412 _WKExperimentalFeature *feature = menuItem.representedObject;
413 [menuItem setState:[defaultPreferences _isEnabledForExperimentalFeature:feature] ? NSControlStateValueOn : NSControlStateValueOff];
414 }
415 if (menuItem.tag == InternalDebugFeatureTag) {
416 _WKInternalDebugFeature *feature = menuItem.representedObject;
417 [menuItem setState:[defaultPreferences _isEnabledForInternalDebugFeature:feature] ? NSControlStateValueOn : NSControlStateValueOff];
418 }
419
420 return YES;
421}

481- (void)togglePerWindowWebProcessesDisabled:(id)sender
482{
483 NSAlert *alert = [[NSAlert alloc] init];
484 [alert setMessageText:self.perWindowWebProcessesDisabled ? @"Are you sure you want to switch to per-window web processes?" : @"Are you sure you want to switch to a single web process?"];
485 [alert setInformativeText:@"This requires quitting and relaunching MiniBrowser. I'll do the quitting. You will have to do the relaunching."];
486 [alert addButtonWithTitle:@"Switch and Quit"];
487 [alert addButtonWithTitle:@"Cancel"];
488
489 NSModalResponse response = [alert runModal];
490 [alert release];
491
492 if (response != NSAlertFirstButtonReturn)
493 return;
494
495 [self _toggleBooleanDefault:PerWindowWebProcessesDisabledKey];
496 [NSApp terminate:self];
497}
