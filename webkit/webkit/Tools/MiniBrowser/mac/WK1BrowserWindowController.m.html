43- (void)awakeFromNib
44{
45 _webView = [[WebView alloc] initWithFrame:[containerView bounds] frameName:nil groupName:@"MiniBrowser"];
46 [_webView setAutoresizingMask:(NSViewWidthSizable | NSViewHeightSizable)];
47
48 _webView.editable = self.isEditable;
49
50 [_webView setFrameLoadDelegate:self];
51 [_webView setUIDelegate:self];
52 [_webView setResourceLoadDelegate:self];
53 [_webView setPolicyDelegate:self];
54
55 [[WebPreferences standardPreferences] setFullScreenEnabled:YES];
56 [[WebPreferences standardPreferences] setDeveloperExtrasEnabled:YES];
57 [[WebPreferences standardPreferences] setImageControlsEnabled:YES];
58 [[WebPreferences standardPreferences] setServiceControlsEnabled:YES];
59 [[WebPreferences standardPreferences] setJavaScriptCanOpenWindowsAutomatically:YES];
60
61 [_webView _listenForLayoutMilestones:WebDidFirstLayout | WebDidFirstVisuallyNonEmptyLayout | WebDidHitRelevantRepaintedObjectsAreaThreshold];
62
63 SettingsController *settingsController = [[NSApplication sharedApplication] browserAppDelegate].settingsController;
64 if (settingsController.customUserAgent)
65 _webView.customUserAgent = settingsController.customUserAgent;
66
67 [self didChangeSettings];
68 [[NSNotificationCenter defaultCenter] addObserver:self selector:@selector(userAgentDidChange:) name:kUserAgentChangedNotificationName object:nil];
69
70 [containerView addSubview:_webView];
71}
72
73- (void)dealloc
74{
75 [[NSNotificationCenter defaultCenter] removeObserver:self];
76 [_webView setFrameLoadDelegate:nil];
77 [_webView setUIDelegate:nil];
78 [_webView setResourceLoadDelegate:nil];
79 [_webView release];
80
81 [super dealloc];
82}

140static BOOL areEssentiallyEqual(double a, double b)
141{
142 double tolerance = 0.001;
143 return (fabs(a - b) <= tolerance);
144}

146#pragma GCC diagnostic push
147#pragma GCC diagnostic ignored "-Wdeprecated-implementations"
148- (BOOL)validateMenuItem:(NSMenuItem *)menuItem
149#pragma GCC diagnostic pop
150{
151 SEL action = [menuItem action];
152
153 if (action == @selector(saveAsPDF:))
154 return NO;
155 if (action == @selector(saveAsWebArchive:))
156 return NO;
157
158 if (action == @selector(zoomIn:))
159 return [self canZoomIn];
160 if (action == @selector(zoomOut:))
161 return [self canZoomOut];
162 if (action == @selector(resetZoom:))
163 return [self canResetZoom];
164
165 if (action == @selector(showHideWebView:))
166 [menuItem setTitle:[_webView isHidden] ? @"Show Web View" : @"Hide Web View"];
167 else if (action == @selector(removeReinsertWebView:))
168 [menuItem setTitle:[_webView window] ? @"Remove Web View" : @"Insert Web View"];
169 else if (action == @selector(toggleFullWindowWebView:))
170 [menuItem setTitle:[self webViewFillsWindow] ? @"Inset Web View" : @"Fit Web View to Window"];
171 else if (action == @selector(toggleZoomMode:))
172 [menuItem setState:_zoomTextOnly ? NSControlStateValueOn : NSControlStateValueOff];
173 else if (action == @selector(toggleEditable:))
174 [menuItem setState:self.isEditable ? NSControlStateValueOn : NSControlStateValueOff];
175 else if (action == @selector(showHideWebInspector:))
176 [menuItem setTitle:_webView.inspector.isOpen ? @"Close Web Inspector" : @"Show Web Inspector"];
177 else if (action == @selector(toggleAlwaysShowsHorizontalScroller:))
178 menuItem.state = _webView.alwaysShowHorizontalScroller ? NSControlStateValueOn : NSControlStateValueOff;
179 else if (action == @selector(toggleAlwaysShowsVerticalScroller:))
180 menuItem.state = _webView.alwaysShowVerticalScroller ? NSControlStateValueOn : NSControlStateValueOff;
181
182 if (action == @selector(setPageScale:))
183 [menuItem setState:areEssentiallyEqual([_webView _viewScaleFactor], [self pageScaleForMenuItemTag:[menuItem tag]])];
184
185 return YES;
186}

188- (BOOL)validateUserInterfaceItem:(id <NSValidatedUserInterfaceItem>)item
189{
190 SEL action = [item action];
191
192 if (action == @selector(goBack:))
193 return [_webView canGoBack];
194
195 if (action == @selector(goForward:))
196 return [_webView canGoForward];
197
198 if (action == @selector(showCertificate:))
199 return NO;
200
201 return YES;
202}

323- (void)didChangeSettings
324{
325 SettingsController *settings = [[NSApplication sharedApplication] browserAppDelegate].settingsController;
326
327 [[WebPreferences standardPreferences] setShowDebugBorders:settings.layerBordersVisible];
328 [[WebPreferences standardPreferences] setLegacyLineLayoutVisualCoverageEnabled:settings.legacyLineLayoutVisualCoverageEnabled];
329 [[WebPreferences standardPreferences] setShowRepaintCounter:settings.layerBordersVisible];
330 [[WebPreferences standardPreferences] setSuppressesIncrementalRendering:settings.incrementalRenderingSuppressed];
331 [[WebPreferences standardPreferences] setAcceleratedDrawingEnabled:settings.acceleratedDrawingEnabled];
332 [[WebPreferences standardPreferences] setSubpixelAntialiasedLayerTextEnabled:settings.subpixelAntialiasedLayerTextEnabled];
333 [[WebPreferences standardPreferences] setResourceLoadStatisticsEnabled:settings.resourceLoadStatisticsEnabled];
334 [[WebPreferences standardPreferences] setLargeImageAsyncDecodingEnabled:settings.largeImageAsyncDecodingEnabled];
335 [[WebPreferences standardPreferences] setAnimatedImageAsyncDecodingEnabled:settings.animatedImageAsyncDecodingEnabled];
336 [[WebPreferences standardPreferences] setColorFilterEnabled:settings.appleColorFilterEnabled];
337 [[WebPreferences standardPreferences] setPunchOutWhiteBackgroundsInDarkMode:settings.punchOutWhiteBackgroundsInDarkMode];
338
339 _webView._useSystemAppearance = settings.useSystemAppearance;
340
341 [self setWebViewFillsWindow:settings.webViewFillsWindow];
342
343 BOOL useTransparentWindows = settings.useTransparentWindows;
344 if (useTransparentWindows != !self.window.isOpaque) {
345 [self.window setOpaque:!useTransparentWindows];
346 [self.window setBackgroundColor:[NSColor clearColor]];
347 [self.window setHasShadow:!useTransparentWindows];
348
349 [_webView setBackgroundColor:useTransparentWindows ? [NSColor clearColor] : [NSColor whiteColor]];
350
351 [self.window display];
352 }
353
354 BOOL usePaginatedMode = settings.usePaginatedMode;
355 if (usePaginatedMode != (_webView._paginationMode != WebPaginationModeUnpaginated)) {
356 if (usePaginatedMode) {
357 [_webView _setPaginationMode:WebPaginationModeLeftToRight];
358 [_webView _setPageLength:_webView.bounds.size.width / 2];
359 [_webView _setGapBetweenPages:10];
360 } else
361 [_webView _setPaginationMode:WebPaginationModeUnpaginated];
362 }
363}

392- (void)updateTitle:(NSString *)title
393{
394 if (!title) {
395 NSURL *url = _webView.mainFrame.dataSource.request.URL;
396 title = url.lastPathComponent ?: url._web_userVisibleString;
397 }
398
399 [self.window setTitle:[title stringByAppendingFormat:@" [WK1]%@", _webView.editable ? @" [Editable]" : @""]];
400}

410- (void)webView:(WebView *)sender didCommitLoadForFrame:(WebFrame *)frame
411{

413 return;
414

444{

446 [alert addButtonWithTitle:@"OK"];

455- (BOOL)webView:(WebView *)sender runBeforeUnloadConfirmPanelWithMessage:(NSString *)message initiatedByFrame:(WebFrame *)frame
456{
457 NSAlert *alert = [[NSAlert alloc] init];
458
459 alert.messageText = [NSString stringWithFormat:@"JavaScript before unload dialog from %@.", frame.dataSource.request.URL.absoluteString];
460 alert.informativeText = message;
461
462 [alert addButtonWithTitle:@"Leave Page"];
463 [alert addButtonWithTitle:@"Stay On Page"];
464
465 NSModalResponse response = [alert runModal];
466
467 [alert release];
468
469 return response == NSAlertFirstButtonReturn;
470}
