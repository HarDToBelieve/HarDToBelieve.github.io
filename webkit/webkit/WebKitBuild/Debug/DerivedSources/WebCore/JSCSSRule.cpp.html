451 spaces.m_subspaceForCSSRule = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSCSSRule);
452 auto* space = spaces.m_subspaceForCSSRule.get();
453IGNORE_WARNINGS_BEGIN("unreachable-code")
454IGNORE_WARNINGS_BEGIN("tautological-compare")
455 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSCSSRule::visitOutputConstraints;
456 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
457 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
458 clientData.outputConstraintSpaces().append(space);
459IGNORE_WARNINGS_END
460IGNORE_WARNINGS_END
461 return space;
462}
