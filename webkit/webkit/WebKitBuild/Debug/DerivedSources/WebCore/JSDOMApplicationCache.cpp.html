582 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSDOMApplicationCache>)
583 spaces.m_subspaceForDOMApplicationCache = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSDOMApplicationCache);
584 else
585 spaces.m_subspaceForDOMApplicationCache = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSDOMApplicationCache);
586 auto* space = spaces.m_subspaceForDOMApplicationCache.get();
587IGNORE_WARNINGS_BEGIN("unreachable-code")
588IGNORE_WARNINGS_BEGIN("tautological-compare")
589 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSDOMApplicationCache::visitOutputConstraints;
590 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
591 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)

594IGNORE_WARNINGS_END
