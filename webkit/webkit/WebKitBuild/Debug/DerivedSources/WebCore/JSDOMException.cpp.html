447 auto& spaces = clientData.subspaces();
448 if (auto* space = spaces.m_subspaceForDOMException.get())
449 return space;
450 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSDOMException> || !JSDOMException::needsDestruction);
451 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSDOMException>)
452 spaces.m_subspaceForDOMException = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSDOMException);
453 else
454 spaces.m_subspaceForDOMException = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSDOMException);
455 auto* space = spaces.m_subspaceForDOMException.get();
456IGNORE_WARNINGS_BEGIN("unreachable-code")
457IGNORE_WARNINGS_BEGIN("tautological-compare")
458 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSDOMException::visitOutputConstraints;
459 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
460 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
461 clientData.outputConstraintSpaces().append(space);
462IGNORE_WARNINGS_END
463IGNORE_WARNINGS_END
464 return space;
465}
