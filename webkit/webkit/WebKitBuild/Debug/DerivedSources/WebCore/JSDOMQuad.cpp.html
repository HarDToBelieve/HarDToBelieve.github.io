466 // attribute to DOMQuad.
467 static_assert(!std::is_polymorphic<DOMQuad>::value, "DOMQuad is polymorphic but the IDL claims it is not");
468 return createWrapper<DOMQuad>(globalObject, WTFMove(impl));
