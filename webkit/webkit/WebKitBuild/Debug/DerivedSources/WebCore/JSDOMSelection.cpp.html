778JSC::IsoSubspace* JSDOMSelection::subspaceForImpl(JSC::VM& vm)
779{
780 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
781 auto& spaces = clientData.subspaces();
782 if (auto* space = spaces.m_subspaceForDOMSelection.get())
783 return space;
784 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSDOMSelection> || !JSDOMSelection::needsDestruction);
785 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSDOMSelection>)
786 spaces.m_subspaceForDOMSelection = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSDOMSelection);
787 else
788 spaces.m_subspaceForDOMSelection = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSDOMSelection);
789 auto* space = spaces.m_subspaceForDOMSelection.get();
790IGNORE_WARNINGS_BEGIN("unreachable-code")
791IGNORE_WARNINGS_BEGIN("tautological-compare")
792 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSDOMSelection::visitOutputConstraints;
793 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
794 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
795 clientData.outputConstraintSpaces().append(space);
796IGNORE_WARNINGS_END
797IGNORE_WARNINGS_END
798 return space;
799}
