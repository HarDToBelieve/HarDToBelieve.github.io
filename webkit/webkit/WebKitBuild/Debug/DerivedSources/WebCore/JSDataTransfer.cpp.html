478 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSDataTransfer> || !JSDataTransfer::needsDestruction);
479 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSDataTransfer>)

482 spaces.m_subspaceForDataTransfer = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSDataTransfer);
483 auto* space = spaces.m_subspaceForDataTransfer.get();
484IGNORE_WARNINGS_BEGIN("unreachable-code")
485IGNORE_WARNINGS_BEGIN("tautological-compare")
486 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSDataTransfer::visitOutputConstraints;

488 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)

490IGNORE_WARNINGS_END
491IGNORE_WARNINGS_END
492 return space;
493}
