496 auto* root = WTF::getPtr(jsHistory->wrapped().window());
497 if (!root)
498 return false;
499 if (UNLIKELY(reason))
500 *reason = "Reachable from Window";
501 return visitor.containsOpaqueRoot(root);
502}
