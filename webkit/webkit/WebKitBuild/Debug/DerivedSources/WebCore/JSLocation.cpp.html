700JSC::IsoSubspace* JSLocation::subspaceForImpl(JSC::VM& vm)
701{
702 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
703 auto& spaces = clientData.subspaces();
704 if (auto* space = spaces.m_subspaceForLocation.get())
705 return space;
706 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSLocation> || !JSLocation::needsDestruction);
707 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSLocation>)
708 spaces.m_subspaceForLocation = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSLocation);
709 else
710 spaces.m_subspaceForLocation = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSLocation);
711 auto* space = spaces.m_subspaceForLocation.get();
712IGNORE_WARNINGS_BEGIN("unreachable-code")
713IGNORE_WARNINGS_BEGIN("tautological-compare")
714 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSLocation::visitOutputConstraints;
715 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
716 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
717 clientData.outputConstraintSpaces().append(space);
718IGNORE_WARNINGS_END
719IGNORE_WARNINGS_END
720 return space;
721}
