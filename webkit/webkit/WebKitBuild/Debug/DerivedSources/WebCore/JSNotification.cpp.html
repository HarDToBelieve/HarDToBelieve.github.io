697JSC::IsoSubspace* JSNotification::subspaceForImpl(JSC::VM& vm)
698{
699 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
700 auto& spaces = clientData.subspaces();
701 if (auto* space = spaces.m_subspaceForNotification.get())
702 return space;
703 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSNotification> || !JSNotification::needsDestruction);
704 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSNotification>)
705 spaces.m_subspaceForNotification = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSNotification);
706 else
707 spaces.m_subspaceForNotification = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSNotification);
708 auto* space = spaces.m_subspaceForNotification.get();
709IGNORE_WARNINGS_BEGIN("unreachable-code")
710IGNORE_WARNINGS_BEGIN("tautological-compare")
711 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSNotification::visitOutputConstraints;
712 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
713 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
714 clientData.outputConstraintSpaces().append(space);
715IGNORE_WARNINGS_END
716IGNORE_WARNINGS_END
717 return space;
718}

729bool JSNotificationOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, AbstractSlotVisitor& visitor, const char** reason)
730{
731 auto* jsNotification = jsCast<JSNotification*>(handle.slot()->asCell());
732 auto& wrapped = jsNotification->wrapped();
733 if (!wrapped.isContextStopped() && wrapped.hasPendingActivity()) {
734 if (UNLIKELY(reason))
735 *reason = "ActiveDOMObject with pending activity";
736 return true;
737 }
738 if (jsNotification->wrapped().isFiringEventListeners()) {
739 if (UNLIKELY(reason))
740 *reason = "EventTarget firing event listeners";
741 return true;
742 }
743 UNUSED_PARAM(visitor);
744 UNUSED_PARAM(reason);
745 return false;
746}
