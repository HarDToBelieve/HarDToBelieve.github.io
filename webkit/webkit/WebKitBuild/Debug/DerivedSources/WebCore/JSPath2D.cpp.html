648JSC::IsoSubspace* JSPath2D::subspaceForImpl(JSC::VM& vm)
649{
650 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
651 auto& spaces = clientData.subspaces();
652 if (auto* space = spaces.m_subspaceForPath2D.get())
653 return space;
654 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSPath2D> || !JSPath2D::needsDestruction);
655 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSPath2D>)
656 spaces.m_subspaceForPath2D = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSPath2D);
657 else
658 spaces.m_subspaceForPath2D = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSPath2D);
659 auto* space = spaces.m_subspaceForPath2D.get();
660IGNORE_WARNINGS_BEGIN("unreachable-code")
661IGNORE_WARNINGS_BEGIN("tautological-compare")
662 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSPath2D::visitOutputConstraints;
663 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
664 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
665 clientData.outputConstraintSpaces().append(space);
666IGNORE_WARNINGS_END
667IGNORE_WARNINGS_END
668 return space;
669}
