563JSC::IsoSubspace* JSPerformance::subspaceForImpl(JSC::VM& vm)
564{
565 auto& clientData = *static_cast<JSVMClientData*>(vm.clientData);
566 auto& spaces = clientData.subspaces();
567 if (auto* space = spaces.m_subspaceForPerformance.get())
568 return space;
569 static_assert(std::is_base_of_v<JSC::JSDestructibleObject, JSPerformance> || !JSPerformance::needsDestruction);
570 if constexpr (std::is_base_of_v<JSC::JSDestructibleObject, JSPerformance>)
571 spaces.m_subspaceForPerformance = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.destructibleObjectHeapCellType.get(), JSPerformance);
572 else
573 spaces.m_subspaceForPerformance = makeUnique<IsoSubspace> ISO_SUBSPACE_INIT(vm.heap, vm.cellHeapCellType.get(), JSPerformance);
574 auto* space = spaces.m_subspaceForPerformance.get();
575IGNORE_WARNINGS_BEGIN("unreachable-code")
576IGNORE_WARNINGS_BEGIN("tautological-compare")
577 void (*myVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSPerformance::visitOutputConstraints;
578 void (*jsCellVisitOutputConstraint)(JSC::JSCell*, JSC::SlotVisitor&) = JSC::JSCell::visitOutputConstraints;
579 if (myVisitOutputConstraint != jsCellVisitOutputConstraint)
580 clientData.outputConstraintSpaces().append(space);
581IGNORE_WARNINGS_END
582IGNORE_WARNINGS_END
583 return space;
584}

595bool JSPerformanceOwner::isReachableFromOpaqueRoots(JSC::Handle<JSC::Unknown> handle, void*, AbstractSlotVisitor& visitor, const char** reason)
596{
597 auto* jsPerformance = jsCast<JSPerformance*>(handle.slot()->asCell());
598 if (jsPerformance->wrapped().isFiringEventListeners()) {
599 if (UNLIKELY(reason))
600 *reason = "EventTarget firing event listeners";
601 return true;
602 }
603 ScriptExecutionContext* root = WTF::getPtr(jsPerformance->wrapped().scriptExecutionContext());

605 return false;
606 if (UNLIKELY(reason))
607 *reason = "Reachable from ScriptExecutionContext";
608 return visitor.containsOpaqueRoot(root);
