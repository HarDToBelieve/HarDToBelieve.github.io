67void DrawingArea::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
68{
69#if USE(COORDINATED_GRAPHICS) || USE(TEXTURE_MAPPER)
70 if (decoder.messageName() == Messages::DrawingArea::UpdateBackingStoreState::name())
71 return IPC::handleMessage<Messages::DrawingArea::UpdateBackingStoreState>(decoder, this, &DrawingArea::updateBackingStoreState);
72#endif
73 if (decoder.messageName() == Messages::DrawingArea::DidUpdate::name())
74 return IPC::handleMessage<Messages::DrawingArea::DidUpdate>(decoder, this, &DrawingArea::didUpdate);
75#if PLATFORM(COCOA)
76 if (decoder.messageName() == Messages::DrawingArea::UpdateGeometry::name())
77 return IPC::handleMessage<Messages::DrawingArea::UpdateGeometry>(decoder, this, &DrawingArea::updateGeometry);
78#endif
79#if PLATFORM(COCOA)
80 if (decoder.messageName() == Messages::DrawingArea::SetDeviceScaleFactor::name())
81 return IPC::handleMessage<Messages::DrawingArea::SetDeviceScaleFactor>(decoder, this, &DrawingArea::setDeviceScaleFactor);
82#endif
83#if PLATFORM(COCOA)
84 if (decoder.messageName() == Messages::DrawingArea::SetColorSpace::name())
85 return IPC::handleMessage<Messages::DrawingArea::SetColorSpace>(decoder, this, &DrawingArea::setColorSpace);
86#endif
87#if PLATFORM(COCOA)
88 if (decoder.messageName() == Messages::DrawingArea::SetViewExposedRect::name())
89 return IPC::handleMessage<Messages::DrawingArea::SetViewExposedRect>(decoder, this, &DrawingArea::setViewExposedRect);
90#endif
91#if PLATFORM(COCOA)
92 if (decoder.messageName() == Messages::DrawingArea::AcceleratedAnimationDidStart::name())
93 return IPC::handleMessage<Messages::DrawingArea::AcceleratedAnimationDidStart>(decoder, this, &DrawingArea::acceleratedAnimationDidStart);
94#endif
95#if PLATFORM(COCOA)
96 if (decoder.messageName() == Messages::DrawingArea::AcceleratedAnimationDidEnd::name())
97 return IPC::handleMessage<Messages::DrawingArea::AcceleratedAnimationDidEnd>(decoder, this, &DrawingArea::acceleratedAnimationDidEnd);
98#endif
99#if PLATFORM(COCOA)
100 if (decoder.messageName() == Messages::DrawingArea::AddTransactionCallbackID::name())
101 return IPC::handleMessage<Messages::DrawingArea::AddTransactionCallbackID>(decoder, this, &DrawingArea::addTransactionCallbackID);
102#endif
103#if PLATFORM(COCOA) || PLATFORM(GTK)
104 if (decoder.messageName() == Messages::DrawingArea::AdjustTransientZoom::name())
105 return IPC::handleMessage<Messages::DrawingArea::AdjustTransientZoom>(decoder, this, &DrawingArea::adjustTransientZoom);
106#endif
107#if PLATFORM(COCOA) || PLATFORM(GTK)
108 if (decoder.messageName() == Messages::DrawingArea::CommitTransientZoom::name())
109 return IPC::handleMessage<Messages::DrawingArea::CommitTransientZoom>(decoder, this, &DrawingArea::commitTransientZoom);
110#endif
111 UNUSED_PARAM(connection);
112 UNUSED_PARAM(decoder);
113#if ENABLE(IPC_TESTING_API)
114 if (connection.ignoreInvalidMessageForTesting())
115 return;
116#endif // ENABLE(IPC_TESTING_API)
117 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
118}
