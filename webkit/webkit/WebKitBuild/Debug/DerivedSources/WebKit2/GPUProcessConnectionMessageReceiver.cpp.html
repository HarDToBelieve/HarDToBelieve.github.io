47void BeginRoutingArbitrationWithCategory::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError&&, WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged&&)>&& completionHandler)
48{
49 std::optional<WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError> error;
50 decoder >> error;
51 if (!error) {
52 ASSERT_NOT_REACHED();
53 cancelReply(WTFMove(completionHandler));
54 return;
55 }
56 std::optional<WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged> defaultRouteChanged;
57 decoder >> defaultRouteChanged;
58 if (!defaultRouteChanged) {
59 ASSERT_NOT_REACHED();
60 cancelReply(WTFMove(completionHandler));
61 return;
62 }
63 completionHandler(WTFMove(*error), WTFMove(*defaultRouteChanged));
64}
65
66void BeginRoutingArbitrationWithCategory::cancelReply(CompletionHandler<void(WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError&&, WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged&&)>&& completionHandler)
67{
68 completionHandler(IPC::AsyncReplyError<WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError>::create(), IPC::AsyncReplyError<WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged>::create());
69}
70
71void BeginRoutingArbitrationWithCategory::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, WebKit::AudioSessionRoutingArbitratorProxy::RoutingArbitrationError error, WebKit::AudioSessionRoutingArbitratorProxy::DefaultRouteChanged defaultRouteChanged)
72{
73 encoder.get() << error;
74 encoder.get() << defaultRouteChanged;
75 connection.sendSyncReply(WTFMove(encoder));
76}

86void GPUProcessConnection::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
87{
88 auto protectedThis = makeRef(*this);
89 if (decoder.messageName() == Messages::GPUProcessConnection::DidReceiveRemoteCommand::name())
90 return IPC::handleMessage<Messages::GPUProcessConnection::DidReceiveRemoteCommand>(decoder, this, &GPUProcessConnection::didReceiveRemoteCommand);
91#if ENABLE(ROUTING_ARBITRATION)
92 if (decoder.messageName() == Messages::GPUProcessConnection::BeginRoutingArbitrationWithCategory::name())
93 return IPC::handleMessageAsync<Messages::GPUProcessConnection::BeginRoutingArbitrationWithCategory>(connection, decoder, this, &GPUProcessConnection::beginRoutingArbitrationWithCategory);
94#endif
95#if ENABLE(ROUTING_ARBITRATION)
96 if (decoder.messageName() == Messages::GPUProcessConnection::EndRoutingArbitration::name())
97 return IPC::handleMessage<Messages::GPUProcessConnection::EndRoutingArbitration>(decoder, this, &GPUProcessConnection::endRoutingArbitration);
98#endif
99 if (dispatchMessage(connection, decoder))
100 return;
101 UNUSED_PARAM(connection);
102 UNUSED_PARAM(decoder);
103#if ENABLE(IPC_TESTING_API)
104 if (connection.ignoreInvalidMessageForTesting())
105 return;
106#endif // ENABLE(IPC_TESTING_API)
107 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
108}
109
110bool GPUProcessConnection::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
111{
112 auto protectedThis = makeRef(*this);
113 if (dispatchSyncMessage(connection, decoder, replyEncoder))
114 return true;
115 UNUSED_PARAM(connection);
116 UNUSED_PARAM(decoder);
117 UNUSED_PARAM(replyEncoder);
118#if ENABLE(IPC_TESTING_API)
119 if (connection.ignoreInvalidMessageForTesting())
120 return false;
121#endif // ENABLE(IPC_TESTING_API)
122 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
123 return false;
124}
