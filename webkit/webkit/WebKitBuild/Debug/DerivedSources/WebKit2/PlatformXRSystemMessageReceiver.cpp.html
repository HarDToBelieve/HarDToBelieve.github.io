41void EnumerateImmersiveXRDevices::callReply(IPC::Decoder& decoder, CompletionHandler<void(Vector<WebKit::XRDeviceInfo>&&)>&& completionHandler)
42{
43 std::optional<Vector<WebKit::XRDeviceInfo>> devicesInfos;
44 decoder >> devicesInfos;
45 if (!devicesInfos) {
46 ASSERT_NOT_REACHED();
47 cancelReply(WTFMove(completionHandler));
48 return;
49 }
50 completionHandler(WTFMove(*devicesInfos));
51}
52
53void EnumerateImmersiveXRDevices::cancelReply(CompletionHandler<void(Vector<WebKit::XRDeviceInfo>&&)>&& completionHandler)
54{
55 completionHandler(IPC::AsyncReplyError<Vector<WebKit::XRDeviceInfo>>::create());
56}
57
58void EnumerateImmersiveXRDevices::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const Vector<WebKit::XRDeviceInfo>& devicesInfos)
59{
60 encoder.get() << devicesInfos;
61 connection.sendSyncReply(WTFMove(encoder));
62}

64void RequestFrame::callReply(IPC::Decoder& decoder, CompletionHandler<void(PlatformXR::Device::FrameData&&)>&& completionHandler)
65{

76void RequestFrame::cancelReply(CompletionHandler<void(PlatformXR::Device::FrameData&&)>&& completionHandler)

78 completionHandler(IPC::AsyncReplyError<PlatformXR::Device::FrameData>::create());
79}

93void PlatformXRSystem::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
94{
95 if (decoder.messageName() == Messages::PlatformXRSystem::EnumerateImmersiveXRDevices::name())
96 return IPC::handleMessageAsync<Messages::PlatformXRSystem::EnumerateImmersiveXRDevices>(connection, decoder, this, &PlatformXRSystem::enumerateImmersiveXRDevices);
97 if (decoder.messageName() == Messages::PlatformXRSystem::InitializeTrackingAndRendering::name())
98 return IPC::handleMessage<Messages::PlatformXRSystem::InitializeTrackingAndRendering>(decoder, this, &PlatformXRSystem::initializeTrackingAndRendering);
99 if (decoder.messageName() == Messages::PlatformXRSystem::ShutDownTrackingAndRendering::name())
100 return IPC::handleMessage<Messages::PlatformXRSystem::ShutDownTrackingAndRendering>(decoder, this, &PlatformXRSystem::shutDownTrackingAndRendering);
101 if (decoder.messageName() == Messages::PlatformXRSystem::RequestFrame::name())
102 return IPC::handleMessageAsync<Messages::PlatformXRSystem::RequestFrame>(connection, decoder, this, &PlatformXRSystem::requestFrame);
103 if (decoder.messageName() == Messages::PlatformXRSystem::SubmitFrame::name())
104 return IPC::handleMessage<Messages::PlatformXRSystem::SubmitFrame>(decoder, this, &PlatformXRSystem::submitFrame);
105 UNUSED_PARAM(connection);
106 UNUSED_PARAM(decoder);
107#if ENABLE(IPC_TESTING_API)
108 if (connection.ignoreInvalidMessageForTesting())
109 return;
110#endif // ENABLE(IPC_TESTING_API)
111 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
112}
