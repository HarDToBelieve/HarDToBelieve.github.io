34namespace WebKit {

36void PlatformXRSystemProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
37{
38 if (decoder.messageName() == Messages::PlatformXRSystemProxy::SessionDidEnd::name())
39 return IPC::handleMessage<Messages::PlatformXRSystemProxy::SessionDidEnd>(decoder, this, &PlatformXRSystemProxy::sessionDidEnd);
40 UNUSED_PARAM(connection);
41 UNUSED_PARAM(decoder);
42#if ENABLE(IPC_TESTING_API)
43 if (connection.ignoreInvalidMessageForTesting())
44 return;
45#endif // ENABLE(IPC_TESTING_API)
46 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
47}
