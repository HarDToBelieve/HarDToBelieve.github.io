38void PlaybackSessionManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
39{
40 auto protectedThis = makeRef(*this);
41 if (decoder.messageName() == Messages::PlaybackSessionManager::Play::name())
42 return IPC::handleMessage<Messages::PlaybackSessionManager::Play>(decoder, this, &PlaybackSessionManager::play);
43 if (decoder.messageName() == Messages::PlaybackSessionManager::Pause::name())
44 return IPC::handleMessage<Messages::PlaybackSessionManager::Pause>(decoder, this, &PlaybackSessionManager::pause);
45 if (decoder.messageName() == Messages::PlaybackSessionManager::TogglePlayState::name())
46 return IPC::handleMessage<Messages::PlaybackSessionManager::TogglePlayState>(decoder, this, &PlaybackSessionManager::togglePlayState);
47 if (decoder.messageName() == Messages::PlaybackSessionManager::BeginScrubbing::name())
48 return IPC::handleMessage<Messages::PlaybackSessionManager::BeginScrubbing>(decoder, this, &PlaybackSessionManager::beginScrubbing);
49 if (decoder.messageName() == Messages::PlaybackSessionManager::EndScrubbing::name())
50 return IPC::handleMessage<Messages::PlaybackSessionManager::EndScrubbing>(decoder, this, &PlaybackSessionManager::endScrubbing);
51 if (decoder.messageName() == Messages::PlaybackSessionManager::SeekToTime::name())
52 return IPC::handleMessage<Messages::PlaybackSessionManager::SeekToTime>(decoder, this, &PlaybackSessionManager::seekToTime);
53 if (decoder.messageName() == Messages::PlaybackSessionManager::FastSeek::name())
54 return IPC::handleMessage<Messages::PlaybackSessionManager::FastSeek>(decoder, this, &PlaybackSessionManager::fastSeek);
55 if (decoder.messageName() == Messages::PlaybackSessionManager::BeginScanningForward::name())
56 return IPC::handleMessage<Messages::PlaybackSessionManager::BeginScanningForward>(decoder, this, &PlaybackSessionManager::beginScanningForward);
57 if (decoder.messageName() == Messages::PlaybackSessionManager::BeginScanningBackward::name())
58 return IPC::handleMessage<Messages::PlaybackSessionManager::BeginScanningBackward>(decoder, this, &PlaybackSessionManager::beginScanningBackward);
59 if (decoder.messageName() == Messages::PlaybackSessionManager::EndScanning::name())
60 return IPC::handleMessage<Messages::PlaybackSessionManager::EndScanning>(decoder, this, &PlaybackSessionManager::endScanning);
61 if (decoder.messageName() == Messages::PlaybackSessionManager::SetDefaultPlaybackRate::name())
62 return IPC::handleMessage<Messages::PlaybackSessionManager::SetDefaultPlaybackRate>(decoder, this, &PlaybackSessionManager::setDefaultPlaybackRate);
63 if (decoder.messageName() == Messages::PlaybackSessionManager::SetPlaybackRate::name())
64 return IPC::handleMessage<Messages::PlaybackSessionManager::SetPlaybackRate>(decoder, this, &PlaybackSessionManager::setPlaybackRate);
65 if (decoder.messageName() == Messages::PlaybackSessionManager::SelectAudioMediaOption::name())
66 return IPC::handleMessage<Messages::PlaybackSessionManager::SelectAudioMediaOption>(decoder, this, &PlaybackSessionManager::selectAudioMediaOption);
67 if (decoder.messageName() == Messages::PlaybackSessionManager::SelectLegibleMediaOption::name())
68 return IPC::handleMessage<Messages::PlaybackSessionManager::SelectLegibleMediaOption>(decoder, this, &PlaybackSessionManager::selectLegibleMediaOption);
69 if (decoder.messageName() == Messages::PlaybackSessionManager::HandleControlledElementIDRequest::name())
70 return IPC::handleMessage<Messages::PlaybackSessionManager::HandleControlledElementIDRequest>(decoder, this, &PlaybackSessionManager::handleControlledElementIDRequest);
71 if (decoder.messageName() == Messages::PlaybackSessionManager::TogglePictureInPicture::name())
72 return IPC::handleMessage<Messages::PlaybackSessionManager::TogglePictureInPicture>(decoder, this, &PlaybackSessionManager::togglePictureInPicture);
73 if (decoder.messageName() == Messages::PlaybackSessionManager::ToggleMuted::name())
74 return IPC::handleMessage<Messages::PlaybackSessionManager::ToggleMuted>(decoder, this, &PlaybackSessionManager::toggleMuted);
75 if (decoder.messageName() == Messages::PlaybackSessionManager::SetMuted::name())
76 return IPC::handleMessage<Messages::PlaybackSessionManager::SetMuted>(decoder, this, &PlaybackSessionManager::setMuted);
77 if (decoder.messageName() == Messages::PlaybackSessionManager::SetVolume::name())
78 return IPC::handleMessage<Messages::PlaybackSessionManager::SetVolume>(decoder, this, &PlaybackSessionManager::setVolume);
79 if (decoder.messageName() == Messages::PlaybackSessionManager::SetPlayingOnSecondScreen::name())
80 return IPC::handleMessage<Messages::PlaybackSessionManager::SetPlayingOnSecondScreen>(decoder, this, &PlaybackSessionManager::setPlayingOnSecondScreen);
81 if (decoder.messageName() == Messages::PlaybackSessionManager::SendRemoteCommand::name())
82 return IPC::handleMessage<Messages::PlaybackSessionManager::SendRemoteCommand>(decoder, this, &PlaybackSessionManager::sendRemoteCommand);
83 UNUSED_PARAM(connection);
84 UNUSED_PARAM(decoder);
85#if ENABLE(IPC_TESTING_API)
86 if (connection.ignoreInvalidMessageForTesting())
87 return;
88#endif // ENABLE(IPC_TESTING_API)
89 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
90}
