146 if (decoder.messageName() == Messages::PluginControllerProxy::StreamDidReceiveResponse::name())
147 return IPC::handleMessage<Messages::PluginControllerProxy::StreamDidReceiveResponse>(decoder, this, &PluginControllerProxy::streamDidReceiveResponse);

149 return IPC::handleMessage<Messages::PluginControllerProxy::StreamDidReceiveData>(decoder, this, &PluginControllerProxy::streamDidReceiveData);

199bool PluginControllerProxy::didReceiveSyncPluginControllerProxyMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
200{
201 if (decoder.messageName() == Messages::PluginControllerProxy::HandleWheelEvent::name())
202 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::HandleWheelEvent>(connection, decoder, replyEncoder, this, &PluginControllerProxy::handleWheelEvent);
203 if (decoder.messageName() == Messages::PluginControllerProxy::HandleMouseEnterEvent::name())
204 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::HandleMouseEnterEvent>(connection, decoder, replyEncoder, this, &PluginControllerProxy::handleMouseEnterEvent);
205 if (decoder.messageName() == Messages::PluginControllerProxy::HandleMouseLeaveEvent::name())
206 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::HandleMouseLeaveEvent>(connection, decoder, replyEncoder, this, &PluginControllerProxy::handleMouseLeaveEvent);
207 if (decoder.messageName() == Messages::PluginControllerProxy::HandleKeyboardEvent::name())
208 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::HandleKeyboardEvent>(connection, decoder, replyEncoder, this, &PluginControllerProxy::handleKeyboardEvent);
209 if (decoder.messageName() == Messages::PluginControllerProxy::HandleEditingCommand::name())
210 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::HandleEditingCommand>(connection, decoder, replyEncoder, this, &PluginControllerProxy::handleEditingCommand);
211 if (decoder.messageName() == Messages::PluginControllerProxy::IsEditingCommandEnabled::name())
212 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::IsEditingCommandEnabled>(connection, decoder, replyEncoder, this, &PluginControllerProxy::isEditingCommandEnabled);
213 if (decoder.messageName() == Messages::PluginControllerProxy::HandlesPageScaleFactor::name())
214 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::HandlesPageScaleFactor>(connection, decoder, replyEncoder, this, &PluginControllerProxy::handlesPageScaleFactor);
215 if (decoder.messageName() == Messages::PluginControllerProxy::RequiresUnifiedScaleFactor::name())
216 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::RequiresUnifiedScaleFactor>(connection, decoder, replyEncoder, this, &PluginControllerProxy::requiresUnifiedScaleFactor);
217 if (decoder.messageName() == Messages::PluginControllerProxy::PaintEntirePlugin::name())
218 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::PaintEntirePlugin>(connection, decoder, replyEncoder, this, &PluginControllerProxy::paintEntirePlugin);
219 if (decoder.messageName() == Messages::PluginControllerProxy::GetPluginScriptableNPObject::name())
220 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::GetPluginScriptableNPObject>(connection, decoder, replyEncoder, this, &PluginControllerProxy::getPluginScriptableNPObject);
221 if (decoder.messageName() == Messages::PluginControllerProxy::SupportsSnapshotting::name())
222 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::SupportsSnapshotting>(connection, decoder, replyEncoder, this, &PluginControllerProxy::supportsSnapshotting);
223 if (decoder.messageName() == Messages::PluginControllerProxy::Snapshot::name())
224 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::Snapshot>(connection, decoder, replyEncoder, this, &PluginControllerProxy::snapshot);
225 if (decoder.messageName() == Messages::PluginControllerProxy::GetFormValue::name())
226 return IPC::handleMessageSynchronous<Messages::PluginControllerProxy::GetFormValue>(connection, decoder, replyEncoder, this, &PluginControllerProxy::getFormValue);
227 UNUSED_PARAM(connection);
228 UNUSED_PARAM(decoder);
229 UNUSED_PARAM(replyEncoder);
230#if ENABLE(IPC_TESTING_API)
231 if (connection.ignoreInvalidMessageForTesting())
232 return false;
233#endif // ENABLE(IPC_TESTING_API)
234 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
235 return false;
236}
