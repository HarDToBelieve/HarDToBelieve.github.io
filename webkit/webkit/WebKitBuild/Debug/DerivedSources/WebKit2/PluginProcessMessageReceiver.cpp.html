40void PluginProcess::didReceivePluginProcessMessage(IPC::Connection& connection, IPC::Decoder& decoder)
41{
42 if (decoder.messageName() == Messages::PluginProcess::InitializePluginProcess::name())
43 return IPC::handleMessage<Messages::PluginProcess::InitializePluginProcess>(decoder, this, &PluginProcess::initializePluginProcess);
44 if (decoder.messageName() == Messages::PluginProcess::CreateWebProcessConnection::name())
45 return IPC::handleMessage<Messages::PluginProcess::CreateWebProcessConnection>(decoder, this, &PluginProcess::createWebProcessConnection);
46 if (decoder.messageName() == Messages::PluginProcess::GetSitesWithData::name())
47 return IPC::handleMessage<Messages::PluginProcess::GetSitesWithData>(decoder, this, &PluginProcess::getSitesWithData);
48 if (decoder.messageName() == Messages::PluginProcess::DeleteWebsiteData::name())
49 return IPC::handleMessage<Messages::PluginProcess::DeleteWebsiteData>(decoder, this, &PluginProcess::deleteWebsiteData);
50 if (decoder.messageName() == Messages::PluginProcess::DeleteWebsiteDataForHostNames::name())
51 return IPC::handleMessage<Messages::PluginProcess::DeleteWebsiteDataForHostNames>(decoder, this, &PluginProcess::deleteWebsiteDataForHostNames);
52#if PLATFORM(COCOA)
53 if (decoder.messageName() == Messages::PluginProcess::SetQOS::name())
54 return IPC::handleMessage<Messages::PluginProcess::SetQOS>(decoder, this, &PluginProcess::setQOS);
55#endif
56 UNUSED_PARAM(connection);
57 UNUSED_PARAM(decoder);
58#if ENABLE(IPC_TESTING_API)
59 if (connection.ignoreInvalidMessageForTesting())
60 return;
61#endif // ENABLE(IPC_TESTING_API)
62 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
63}
