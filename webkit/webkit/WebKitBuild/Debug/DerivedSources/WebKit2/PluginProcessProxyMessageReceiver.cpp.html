89void PluginProcessProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
90{
91 auto protectedThis = makeRef(*this);
92 if (decoder.messageName() == Messages::PluginProcessProxy::DidCreateWebProcessConnection::name())
93 return IPC::handleMessage<Messages::PluginProcessProxy::DidCreateWebProcessConnection>(decoder, this, &PluginProcessProxy::didCreateWebProcessConnection);
94 if (decoder.messageName() == Messages::PluginProcessProxy::DidGetSitesWithData::name())
95 return IPC::handleMessage<Messages::PluginProcessProxy::DidGetSitesWithData>(decoder, this, &PluginProcessProxy::didGetSitesWithData);
96 if (decoder.messageName() == Messages::PluginProcessProxy::DidDeleteWebsiteData::name())
97 return IPC::handleMessage<Messages::PluginProcessProxy::DidDeleteWebsiteData>(decoder, this, &PluginProcessProxy::didDeleteWebsiteData);
98 if (decoder.messageName() == Messages::PluginProcessProxy::DidDeleteWebsiteDataForHostNames::name())
99 return IPC::handleMessage<Messages::PluginProcessProxy::DidDeleteWebsiteDataForHostNames>(decoder, this, &PluginProcessProxy::didDeleteWebsiteDataForHostNames);
100#if PLATFORM(COCOA)
101 if (decoder.messageName() == Messages::PluginProcessProxy::SetModalWindowIsShowing::name())
102 return IPC::handleMessage<Messages::PluginProcessProxy::SetModalWindowIsShowing>(decoder, this, &PluginProcessProxy::setModalWindowIsShowing);
103#endif
104#if PLATFORM(COCOA)
105 if (decoder.messageName() == Messages::PluginProcessProxy::SetFullscreenWindowIsShowing::name())
106 return IPC::handleMessage<Messages::PluginProcessProxy::SetFullscreenWindowIsShowing>(decoder, this, &PluginProcessProxy::setFullscreenWindowIsShowing);
107#endif
108 UNUSED_PARAM(connection);
109 UNUSED_PARAM(decoder);
110#if ENABLE(IPC_TESTING_API)
111 if (connection.ignoreInvalidMessageForTesting())
112 return;
113#endif // ENABLE(IPC_TESTING_API)
114 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
115}
116
117bool PluginProcessProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
118{
119 auto protectedThis = makeRef(*this);
120#if PLATFORM(COCOA)
121 if (decoder.messageName() == Messages::PluginProcessProxy::LaunchProcess::name())
122 return IPC::handleMessageSynchronous<Messages::PluginProcessProxy::LaunchProcess>(connection, decoder, replyEncoder, this, &PluginProcessProxy::launchProcess);
123#endif
124#if PLATFORM(COCOA)
125 if (decoder.messageName() == Messages::PluginProcessProxy::LaunchApplicationAtURL::name())
126 return IPC::handleMessageSynchronous<Messages::PluginProcessProxy::LaunchApplicationAtURL>(connection, decoder, replyEncoder, this, &PluginProcessProxy::launchApplicationAtURL);
127#endif
128#if PLATFORM(COCOA)
129 if (decoder.messageName() == Messages::PluginProcessProxy::OpenURL::name())
130 return IPC::handleMessageSynchronous<Messages::PluginProcessProxy::OpenURL>(connection, decoder, replyEncoder, this, &PluginProcessProxy::openURL);
131#endif
132#if PLATFORM(COCOA)
133 if (decoder.messageName() == Messages::PluginProcessProxy::OpenFile::name())
134 return IPC::handleMessageSynchronous<Messages::PluginProcessProxy::OpenFile>(connection, decoder, replyEncoder, this, &PluginProcessProxy::openFile);
135#endif
136 UNUSED_PARAM(connection);
137 UNUSED_PARAM(decoder);
138 UNUSED_PARAM(replyEncoder);
139#if ENABLE(IPC_TESTING_API)
140 if (connection.ignoreInvalidMessageForTesting())
141 return false;
142#endif // ENABLE(IPC_TESTING_API)
143 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
144 return false;
145}
