104void PluginProxy::didReceivePluginProxyMessage(IPC::Connection& connection, IPC::Decoder& decoder)
105{
106 auto protectedThis = makeRef(*this);
107 if (decoder.messageName() == Messages::PluginProxy::LoadURL::name())
108 return IPC::handleMessage<Messages::PluginProxy::LoadURL>(decoder, this, &PluginProxy::loadURL);
109 if (decoder.messageName() == Messages::PluginProxy::Update::name())
110 return IPC::handleMessage<Messages::PluginProxy::Update>(decoder, this, &PluginProxy::update);
111 if (decoder.messageName() == Messages::PluginProxy::SetCookiesForURL::name())
112 return IPC::handleMessage<Messages::PluginProxy::SetCookiesForURL>(decoder, this, &PluginProxy::setCookiesForURL);
113 if (decoder.messageName() == Messages::PluginProxy::CancelStreamLoad::name())
114 return IPC::handleMessage<Messages::PluginProxy::CancelStreamLoad>(decoder, this, &PluginProxy::cancelStreamLoad);
115 if (decoder.messageName() == Messages::PluginProxy::ContinueStreamLoad::name())
116 return IPC::handleMessage<Messages::PluginProxy::ContinueStreamLoad>(decoder, this, &PluginProxy::continueStreamLoad);
117 if (decoder.messageName() == Messages::PluginProxy::CancelManualStreamLoad::name())
118 return IPC::handleMessage<Messages::PluginProxy::CancelManualStreamLoad>(decoder, this, &PluginProxy::cancelManualStreamLoad);
119 if (decoder.messageName() == Messages::PluginProxy::SetStatusbarText::name())
120 return IPC::handleMessage<Messages::PluginProxy::SetStatusbarText>(decoder, this, &PluginProxy::setStatusbarText);
121#if PLATFORM(COCOA)
122 if (decoder.messageName() == Messages::PluginProxy::PluginFocusOrWindowFocusChanged::name())
123 return IPC::handleMessage<Messages::PluginProxy::PluginFocusOrWindowFocusChanged>(decoder, this, &PluginProxy::pluginFocusOrWindowFocusChanged);
124#endif
125#if PLATFORM(COCOA)
126 if (decoder.messageName() == Messages::PluginProxy::SetComplexTextInputState::name())
127 return IPC::handleMessage<Messages::PluginProxy::SetComplexTextInputState>(decoder, this, &PluginProxy::setComplexTextInputState);
128#endif
129#if PLATFORM(COCOA)
130 if (decoder.messageName() == Messages::PluginProxy::SetLayerHostingContextID::name())
131 return IPC::handleMessage<Messages::PluginProxy::SetLayerHostingContextID>(decoder, this, &PluginProxy::setLayerHostingContextID);
132#endif
133#if PLATFORM(X11)
134 if (decoder.messageName() == Messages::PluginProxy::WindowedPluginGeometryDidChange::name())
135 return IPC::handleMessage<Messages::PluginProxy::WindowedPluginGeometryDidChange>(decoder, this, &PluginProxy::windowedPluginGeometryDidChange);
136#endif
137#if PLATFORM(X11)
138 if (decoder.messageName() == Messages::PluginProxy::WindowedPluginVisibilityDidChange::name())
139 return IPC::handleMessage<Messages::PluginProxy::WindowedPluginVisibilityDidChange>(decoder, this, &PluginProxy::windowedPluginVisibilityDidChange);
140#endif
141 if (decoder.messageName() == Messages::PluginProxy::SetPluginIsPlayingAudio::name())
142 return IPC::handleMessage<Messages::PluginProxy::SetPluginIsPlayingAudio>(decoder, this, &PluginProxy::setPluginIsPlayingAudio);
143 UNUSED_PARAM(connection);
144 UNUSED_PARAM(decoder);
145#if ENABLE(IPC_TESTING_API)
146 if (connection.ignoreInvalidMessageForTesting())
147 return;
148#endif // ENABLE(IPC_TESTING_API)
149 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
150}

152bool PluginProxy::didReceiveSyncPluginProxyMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
153{
154 auto protectedThis = makeRef(*this);
155 if (decoder.messageName() == Messages::PluginProxy::ProxiesForURL::name())
156 return IPC::handleMessageSynchronous<Messages::PluginProxy::ProxiesForURL>(connection, decoder, replyEncoder, this, &PluginProxy::proxiesForURL);
157 if (decoder.messageName() == Messages::PluginProxy::CookiesForURL::name())
158 return IPC::handleMessageSynchronous<Messages::PluginProxy::CookiesForURL>(connection, decoder, replyEncoder, this, &PluginProxy::cookiesForURL);
159 if (decoder.messageName() == Messages::PluginProxy::GetAuthenticationInfo::name())
160 return IPC::handleMessageSynchronous<Messages::PluginProxy::GetAuthenticationInfo>(connection, decoder, replyEncoder, this, &PluginProxy::getAuthenticationInfo);
161 if (decoder.messageName() == Messages::PluginProxy::GetPluginElementNPObject::name())
162 return IPC::handleMessageSynchronous<Messages::PluginProxy::GetPluginElementNPObject>(connection, decoder, replyEncoder, this, &PluginProxy::getPluginElementNPObject);
163 if (decoder.messageName() == Messages::PluginProxy::Evaluate::name())
164 return IPC::handleMessageSynchronous<Messages::PluginProxy::Evaluate>(connection, decoder, replyEncoder, this, &PluginProxy::evaluate);
165#if PLATFORM(X11)
166 if (decoder.messageName() == Messages::PluginProxy::CreatePluginContainer::name())
167 return IPC::handleMessageSynchronous<Messages::PluginProxy::CreatePluginContainer>(connection, decoder, replyEncoder, this, &PluginProxy::createPluginContainer);
168#endif
169 if (decoder.messageName() == Messages::PluginProxy::DidCreatePlugin::name())
170 return IPC::handleMessageSynchronous<Messages::PluginProxy::DidCreatePlugin>(connection, decoder, replyEncoder, this, &PluginProxy::didCreatePlugin);
171 if (decoder.messageName() == Messages::PluginProxy::DidFailToCreatePlugin::name())
172 return IPC::handleMessageSynchronous<Messages::PluginProxy::DidFailToCreatePlugin>(connection, decoder, replyEncoder, this, &PluginProxy::didFailToCreatePlugin);
173 UNUSED_PARAM(connection);
174 UNUSED_PARAM(decoder);
175 UNUSED_PARAM(replyEncoder);
176#if ENABLE(IPC_TESTING_API)
177 if (connection.ignoreInvalidMessageForTesting())
178 return false;
179#endif // ENABLE(IPC_TESTING_API)
180 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
181 return false;
182}
