71void StartAudioDestination::callReply(IPC::Decoder& decoder, CompletionHandler<void(bool&&)>&& completionHandler)
72{
73 std::optional<bool> isPlaying;
74 decoder >> isPlaying;
75 if (!isPlaying) {
76 ASSERT_NOT_REACHED();
77 cancelReply(WTFMove(completionHandler));
78 return;
79 }
80 completionHandler(WTFMove(*isPlaying));
81}
82
83void StartAudioDestination::cancelReply(CompletionHandler<void(bool&&)>&& completionHandler)
84{
85 completionHandler(IPC::AsyncReplyError<bool>::create());
86}

123void RemoteAudioDestinationManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
124{
125 if (decoder.messageName() == Messages::RemoteAudioDestinationManager::DeleteAudioDestination::name())
126 return IPC::handleMessageAsync<Messages::RemoteAudioDestinationManager::DeleteAudioDestination>(connection, decoder, this, &RemoteAudioDestinationManager::deleteAudioDestination);
127 if (decoder.messageName() == Messages::RemoteAudioDestinationManager::StartAudioDestination::name())
128 return IPC::handleMessageAsync<Messages::RemoteAudioDestinationManager::StartAudioDestination>(connection, decoder, this, &RemoteAudioDestinationManager::startAudioDestination);
129 if (decoder.messageName() == Messages::RemoteAudioDestinationManager::StopAudioDestination::name())
130 return IPC::handleMessageAsync<Messages::RemoteAudioDestinationManager::StopAudioDestination>(connection, decoder, this, &RemoteAudioDestinationManager::stopAudioDestination);
131#if PLATFORM(COCOA)
132 if (decoder.messageName() == Messages::RemoteAudioDestinationManager::AudioSamplesStorageChanged::name())
133 return IPC::handleMessage<Messages::RemoteAudioDestinationManager::AudioSamplesStorageChanged>(decoder, this, &RemoteAudioDestinationManager::audioSamplesStorageChanged);
134#endif
135 UNUSED_PARAM(connection);
136 UNUSED_PARAM(decoder);
137#if ENABLE(IPC_TESTING_API)
138 if (connection.ignoreInvalidMessageForTesting())
139 return;
140#endif // ENABLE(IPC_TESTING_API)
141 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
142}
143
144bool RemoteAudioDestinationManager::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
145{
146 if (decoder.messageName() == Messages::RemoteAudioDestinationManager::CreateAudioDestination::name())
147 return IPC::handleMessageSynchronous<Messages::RemoteAudioDestinationManager::CreateAudioDestination>(connection, decoder, replyEncoder, this, &RemoteAudioDestinationManager::createAudioDestination);
148 UNUSED_PARAM(connection);
149 UNUSED_PARAM(decoder);
150 UNUSED_PARAM(replyEncoder);
151#if ENABLE(IPC_TESTING_API)
152 if (connection.ignoreInvalidMessageForTesting())
153 return false;
154#endif // ENABLE(IPC_TESTING_API)
155 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
156 return false;
157}
