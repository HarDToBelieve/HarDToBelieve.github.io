44void CreateUnit::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebCore::CAAudioStreamDescription&&, size_t&&)>&& completionHandler)
45{
46 std::optional<WebCore::CAAudioStreamDescription> description;
47 decoder >> description;
48 if (!description) {
49 ASSERT_NOT_REACHED();
50 cancelReply(WTFMove(completionHandler));
51 return;
52 }
53 std::optional<size_t> frameChunkSize;
54 decoder >> frameChunkSize;
55 if (!frameChunkSize) {
56 ASSERT_NOT_REACHED();
57 cancelReply(WTFMove(completionHandler));
58 return;
59 }
60 completionHandler(WTFMove(*description), WTFMove(*frameChunkSize));
61}
62
63void CreateUnit::cancelReply(CompletionHandler<void(WebCore::CAAudioStreamDescription&&, size_t&&)>&& completionHandler)
64{
65 completionHandler(IPC::AsyncReplyError<WebCore::CAAudioStreamDescription>::create(), IPC::AsyncReplyError<size_t>::create());
66}
67
68void CreateUnit::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const WebCore::CAAudioStreamDescription& description, size_t frameChunkSize)
69{
70 encoder.get() << description;
71 encoder.get() << frameChunkSize;
72 connection.sendSyncReply(WTFMove(encoder));
73}

81void RemoteAudioMediaStreamTrackRendererInternalUnitManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
82{
83 if (decoder.messageName() == Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::CreateUnit::name())
84 return IPC::handleMessageAsync<Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::CreateUnit>(connection, decoder, this, &RemoteAudioMediaStreamTrackRendererInternalUnitManager::createUnit);
85 if (decoder.messageName() == Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::DeleteUnit::name())
86 return IPC::handleMessage<Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::DeleteUnit>(decoder, this, &RemoteAudioMediaStreamTrackRendererInternalUnitManager::deleteUnit);
87 if (decoder.messageName() == Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::StartUnit::name())
88 return IPC::handleMessage<Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::StartUnit>(decoder, this, &RemoteAudioMediaStreamTrackRendererInternalUnitManager::startUnit);
89 if (decoder.messageName() == Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::StopUnit::name())
90 return IPC::handleMessage<Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::StopUnit>(decoder, this, &RemoteAudioMediaStreamTrackRendererInternalUnitManager::stopUnit);
91 if (decoder.messageName() == Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::SetAudioOutputDevice::name())
92 return IPC::handleMessage<Messages::RemoteAudioMediaStreamTrackRendererInternalUnitManager::SetAudioOutputDevice>(decoder, this, &RemoteAudioMediaStreamTrackRendererInternalUnitManager::setAudioOutputDevice);
93 UNUSED_PARAM(connection);
94 UNUSED_PARAM(decoder);
95#if ENABLE(IPC_TESTING_API)
96 if (connection.ignoreInvalidMessageForTesting())
97 return;
98#endif // ENABLE(IPC_TESTING_API)
99 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
100}
