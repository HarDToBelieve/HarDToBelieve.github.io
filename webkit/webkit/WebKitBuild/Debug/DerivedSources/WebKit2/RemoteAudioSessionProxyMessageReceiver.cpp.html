39void TryToSetActive::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, bool suceeded)
40{
41 encoder.get() << suceeded;
42 connection.sendSyncReply(WTFMove(encoder));
43}

49namespace WebKit {

51void RemoteAudioSessionProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
52{
53 if (decoder.messageName() == Messages::RemoteAudioSessionProxy::SetCategory::name())
54 return IPC::handleMessage<Messages::RemoteAudioSessionProxy::SetCategory>(decoder, this, &RemoteAudioSessionProxy::setCategory);
55 if (decoder.messageName() == Messages::RemoteAudioSessionProxy::SetPreferredBufferSize::name())
56 return IPC::handleMessage<Messages::RemoteAudioSessionProxy::SetPreferredBufferSize>(decoder, this, &RemoteAudioSessionProxy::setPreferredBufferSize);
57 UNUSED_PARAM(connection);
58 UNUSED_PARAM(decoder);
59#if ENABLE(IPC_TESTING_API)
60 if (connection.ignoreInvalidMessageForTesting())
61 return;
62#endif // ENABLE(IPC_TESTING_API)
63 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
64}
65
66bool RemoteAudioSessionProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
67{
68 if (decoder.messageName() == Messages::RemoteAudioSessionProxy::TryToSetActive::name())
69 return IPC::handleMessageSynchronous<Messages::RemoteAudioSessionProxy::TryToSetActive>(connection, decoder, replyEncoder, this, &RemoteAudioSessionProxy::tryToSetActive);
70 UNUSED_PARAM(connection);
71 UNUSED_PARAM(decoder);
72 UNUSED_PARAM(replyEncoder);
73#if ENABLE(IPC_TESTING_API)
74 if (connection.ignoreInvalidMessageForTesting())
75 return false;
76#endif // ENABLE(IPC_TESTING_API)
77 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
78 return false;
79}
