43void RequestLicense::callReply(IPC::Decoder& decoder, CompletionHandler<void(IPC::SharedBufferCopy&&, String&&, bool&&, bool&&)>&& completionHandler)
44{
45 std::optional<IPC::SharedBufferCopy> message;
46 decoder >> message;
47 if (!message) {
48 ASSERT_NOT_REACHED();
49 cancelReply(WTFMove(completionHandler));
50 return;
51 }
52 std::optional<String> sessionId;
53 decoder >> sessionId;
54 if (!sessionId) {
55 ASSERT_NOT_REACHED();
56 cancelReply(WTFMove(completionHandler));
57 return;
58 }
59 std::optional<bool> needsIndividualization;
60 decoder >> needsIndividualization;
61 if (!needsIndividualization) {
62 ASSERT_NOT_REACHED();
63 cancelReply(WTFMove(completionHandler));
64 return;
65 }
66 std::optional<bool> succeeded;
67 decoder >> succeeded;
68 if (!succeeded) {
69 ASSERT_NOT_REACHED();
70 cancelReply(WTFMove(completionHandler));
71 return;
72 }
73 completionHandler(WTFMove(*message), WTFMove(*sessionId), WTFMove(*needsIndividualization), WTFMove(*succeeded));
74}

260void RemoteCDMInstanceSessionProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
261{
262 if (decoder.messageName() == Messages::RemoteCDMInstanceSessionProxy::RequestLicense::name())
263 return IPC::handleMessageAsync<Messages::RemoteCDMInstanceSessionProxy::RequestLicense>(connection, decoder, this, &RemoteCDMInstanceSessionProxy::requestLicense);
264 if (decoder.messageName() == Messages::RemoteCDMInstanceSessionProxy::UpdateLicense::name())
265 return IPC::handleMessageAsync<Messages::RemoteCDMInstanceSessionProxy::UpdateLicense>(connection, decoder, this, &RemoteCDMInstanceSessionProxy::updateLicense);
266 if (decoder.messageName() == Messages::RemoteCDMInstanceSessionProxy::LoadSession::name())
267 return IPC::handleMessageAsync<Messages::RemoteCDMInstanceSessionProxy::LoadSession>(connection, decoder, this, &RemoteCDMInstanceSessionProxy::loadSession);
268 if (decoder.messageName() == Messages::RemoteCDMInstanceSessionProxy::CloseSession::name())
269 return IPC::handleMessageAsync<Messages::RemoteCDMInstanceSessionProxy::CloseSession>(connection, decoder, this, &RemoteCDMInstanceSessionProxy::closeSession);
270 if (decoder.messageName() == Messages::RemoteCDMInstanceSessionProxy::RemoveSessionData::name())
271 return IPC::handleMessageAsync<Messages::RemoteCDMInstanceSessionProxy::RemoveSessionData>(connection, decoder, this, &RemoteCDMInstanceSessionProxy::removeSessionData);
272 if (decoder.messageName() == Messages::RemoteCDMInstanceSessionProxy::StoreRecordOfKeyUsage::name())
273 return IPC::handleMessage<Messages::RemoteCDMInstanceSessionProxy::StoreRecordOfKeyUsage>(decoder, this, &RemoteCDMInstanceSessionProxy::storeRecordOfKeyUsage);
274 if (decoder.messageName() == Messages::RemoteCDMInstanceSessionProxy::DisplayIDChanged::name())
275 return IPC::handleMessage<Messages::RemoteCDMInstanceSessionProxy::DisplayIDChanged>(decoder, this, &RemoteCDMInstanceSessionProxy::displayIDChanged);
276 UNUSED_PARAM(connection);
277 UNUSED_PARAM(decoder);
278#if ENABLE(IPC_TESTING_API)
279 if (connection.ignoreInvalidMessageForTesting())
280 return;
281#endif // ENABLE(IPC_TESTING_API)
282 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
283}
