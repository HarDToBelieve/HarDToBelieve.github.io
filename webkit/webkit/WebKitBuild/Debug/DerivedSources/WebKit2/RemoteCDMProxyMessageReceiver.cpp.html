44void GetSupportedConfiguration::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<WebCore::CDMKeySystemConfiguration>&&)>&& completionHandler)
45{
46 std::optional<std::optional<WebCore::CDMKeySystemConfiguration>> accumulatedConfiguration;
47 decoder >> accumulatedConfiguration;
48 if (!accumulatedConfiguration) {
49 ASSERT_NOT_REACHED();
50 cancelReply(WTFMove(completionHandler));
51 return;
52 }
53 completionHandler(WTFMove(*accumulatedConfiguration));
54}
55
56void GetSupportedConfiguration::cancelReply(CompletionHandler<void(std::optional<WebCore::CDMKeySystemConfiguration>&&)>&& completionHandler)
57{
58 completionHandler(IPC::AsyncReplyError<std::optional<WebCore::CDMKeySystemConfiguration>>::create());
59}
60
61void GetSupportedConfiguration::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const std::optional<WebCore::CDMKeySystemConfiguration>& accumulatedConfiguration)
62{
63 encoder.get() << accumulatedConfiguration;
64 connection.sendSyncReply(WTFMove(encoder));
65}
66
67void CreateInstance::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const WebKit::RemoteCDMInstanceIdentifier& identifier, const WebKit::RemoteCDMInstanceConfiguration& configuration)
68{
69 encoder.get() << identifier;
70 encoder.get() << configuration;
71 connection.sendSyncReply(WTFMove(encoder));
72}

80void RemoteCDMProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
81{
82 if (decoder.messageName() == Messages::RemoteCDMProxy::GetSupportedConfiguration::name())
83 return IPC::handleMessageAsync<Messages::RemoteCDMProxy::GetSupportedConfiguration>(connection, decoder, this, &RemoteCDMProxy::getSupportedConfiguration);
84 if (decoder.messageName() == Messages::RemoteCDMProxy::LoadAndInitialize::name())
85 return IPC::handleMessage<Messages::RemoteCDMProxy::LoadAndInitialize>(decoder, this, &RemoteCDMProxy::loadAndInitialize);
86 UNUSED_PARAM(connection);
87 UNUSED_PARAM(decoder);
88#if ENABLE(IPC_TESTING_API)
89 if (connection.ignoreInvalidMessageForTesting())
90 return;
91#endif // ENABLE(IPC_TESTING_API)
92 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
93}
94
95bool RemoteCDMProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
96{
97 if (decoder.messageName() == Messages::RemoteCDMProxy::CreateInstance::name())
98 return IPC::handleMessageSynchronous<Messages::RemoteCDMProxy::CreateInstance>(connection, decoder, replyEncoder, this, &RemoteCDMProxy::createInstance);
99 UNUSED_PARAM(connection);
100 UNUSED_PARAM(decoder);
101 UNUSED_PARAM(replyEncoder);
102#if ENABLE(IPC_TESTING_API)
103 if (connection.ignoreInvalidMessageForTesting())
104 return false;
105#endif // ENABLE(IPC_TESTING_API)
106 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
107 return false;
108}
