62bool RemoteLegacyCDMFactoryProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
63{
64 if (decoder.messageName() == Messages::RemoteLegacyCDMFactoryProxy::CreateCDM::name())
65 return IPC::handleMessageSynchronous<Messages::RemoteLegacyCDMFactoryProxy::CreateCDM>(connection, decoder, replyEncoder, this, &RemoteLegacyCDMFactoryProxy::createCDM);
66 if (decoder.messageName() == Messages::RemoteLegacyCDMFactoryProxy::SupportsKeySystem::name())
67 return IPC::handleMessageSynchronous<Messages::RemoteLegacyCDMFactoryProxy::SupportsKeySystem>(connection, decoder, replyEncoder, this, &RemoteLegacyCDMFactoryProxy::supportsKeySystem);
68 UNUSED_PARAM(connection);
69 UNUSED_PARAM(decoder);
70 UNUSED_PARAM(replyEncoder);
71#if ENABLE(IPC_TESTING_API)
72 if (connection.ignoreInvalidMessageForTesting())
73 return false;
74#endif // ENABLE(IPC_TESTING_API)
75 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
76 return false;
77}
