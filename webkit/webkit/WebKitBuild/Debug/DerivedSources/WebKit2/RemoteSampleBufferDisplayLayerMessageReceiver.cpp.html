38void RemoteSampleBufferDisplayLayer::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
39{
40 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::UpdateDisplayMode::name())
41 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::UpdateDisplayMode>(decoder, this, &RemoteSampleBufferDisplayLayer::updateDisplayMode);
42 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::UpdateAffineTransform::name())
43 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::UpdateAffineTransform>(decoder, this, &RemoteSampleBufferDisplayLayer::updateAffineTransform);
44 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::UpdateBoundsAndPosition::name())
45 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::UpdateBoundsAndPosition>(decoder, this, &RemoteSampleBufferDisplayLayer::updateBoundsAndPosition);
46 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::Flush::name())
47 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::Flush>(decoder, this, &RemoteSampleBufferDisplayLayer::flush);
48 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::FlushAndRemoveImage::name())
49 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::FlushAndRemoveImage>(decoder, this, &RemoteSampleBufferDisplayLayer::flushAndRemoveImage);
50 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::EnqueueSample::name())
51 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::EnqueueSample>(decoder, this, &RemoteSampleBufferDisplayLayer::enqueueSample);
52 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::ClearEnqueuedSamples::name())
53 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::ClearEnqueuedSamples>(decoder, this, &RemoteSampleBufferDisplayLayer::clearEnqueuedSamples);
54 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::Play::name())
55 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::Play>(decoder, this, &RemoteSampleBufferDisplayLayer::play);
56 if (decoder.messageName() == Messages::RemoteSampleBufferDisplayLayer::Pause::name())
57 return IPC::handleMessage<Messages::RemoteSampleBufferDisplayLayer::Pause>(decoder, this, &RemoteSampleBufferDisplayLayer::pause);
58 UNUSED_PARAM(connection);
59 UNUSED_PARAM(decoder);
60#if ENABLE(IPC_TESTING_API)
61 if (connection.ignoreInvalidMessageForTesting())
62 return;
63#endif // ENABLE(IPC_TESTING_API)
64 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
65}
