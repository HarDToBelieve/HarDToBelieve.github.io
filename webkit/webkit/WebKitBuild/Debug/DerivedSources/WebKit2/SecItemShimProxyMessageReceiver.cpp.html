58#if ENABLE(SEC_ITEM_SHIM)
59
60void SecItemRequest::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<WebKit::SecItemResponseData>&&)>&& completionHandler)
61{
62 std::optional<std::optional<WebKit::SecItemResponseData>> response;
63 decoder >> response;
64 if (!response) {
65 ASSERT_NOT_REACHED();
66 cancelReply(WTFMove(completionHandler));
67 return;
68 }
69 completionHandler(WTFMove(*response));
70}
71
72void SecItemRequest::cancelReply(CompletionHandler<void(std::optional<WebKit::SecItemResponseData>&&)>&& completionHandler)
73{
74 completionHandler(IPC::AsyncReplyError<std::optional<WebKit::SecItemResponseData>>::create());
75}

77void SecItemRequest::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const std::optional<WebKit::SecItemResponseData>& response)
78{
79 encoder.get() << response;
80 connection.sendSyncReply(WTFMove(encoder));
81}

91void SecItemShimProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
92{
93 auto protectedThis = makeRef(*this);
94#if ENABLE(SEC_ITEM_SHIM)
95 if (decoder.messageName() == Messages::SecItemShimProxy::SecItemRequest::name())
96 return IPC::handleMessageAsync<Messages::SecItemShimProxy::SecItemRequest>(connection, decoder, this, &SecItemShimProxy::secItemRequest);
97#endif
98 UNUSED_PARAM(connection);
99 UNUSED_PARAM(decoder);
100#if ENABLE(IPC_TESTING_API)
101 if (connection.ignoreInvalidMessageForTesting())
102 return;
103#endif // ENABLE(IPC_TESTING_API)
104 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
105}
106
107bool SecItemShimProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
108{
109 auto protectedThis = makeRef(*this);
110#if ENABLE(SEC_ITEM_SHIM)
111 if (decoder.messageName() == Messages::SecItemShimProxy::SecItemRequestSync::name())
112 return IPC::handleMessageSynchronous<Messages::SecItemShimProxy::SecItemRequestSync>(connection, decoder, replyEncoder, this, &SecItemShimProxy::secItemRequestSync);
113#endif
114 UNUSED_PARAM(connection);
115 UNUSED_PARAM(decoder);
116 UNUSED_PARAM(replyEncoder);
117#if ENABLE(IPC_TESTING_API)
118 if (connection.ignoreInvalidMessageForTesting())
119 return false;
120#endif // ENABLE(IPC_TESTING_API)
121 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
122 return false;
123}
