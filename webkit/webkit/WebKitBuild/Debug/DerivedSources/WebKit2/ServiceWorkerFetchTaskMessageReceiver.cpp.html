40void ServiceWorkerFetchTask::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
41{
42 if (decoder.messageName() == Messages::ServiceWorkerFetchTask::DidNotHandle::name())
43 return IPC::handleMessage<Messages::ServiceWorkerFetchTask::DidNotHandle>(decoder, this, &ServiceWorkerFetchTask::didNotHandle);
44 if (decoder.messageName() == Messages::ServiceWorkerFetchTask::DidFail::name())
45 return IPC::handleMessage<Messages::ServiceWorkerFetchTask::DidFail>(decoder, this, &ServiceWorkerFetchTask::didFail);
46 if (decoder.messageName() == Messages::ServiceWorkerFetchTask::DidReceiveRedirectResponse::name())
47 return IPC::handleMessage<Messages::ServiceWorkerFetchTask::DidReceiveRedirectResponse>(decoder, this, &ServiceWorkerFetchTask::didReceiveRedirectResponse);
48 if (decoder.messageName() == Messages::ServiceWorkerFetchTask::DidReceiveResponse::name())
49 return IPC::handleMessage<Messages::ServiceWorkerFetchTask::DidReceiveResponse>(decoder, this, &ServiceWorkerFetchTask::didReceiveResponse);
50 if (decoder.messageName() == Messages::ServiceWorkerFetchTask::DidReceiveData::name())
51 return IPC::handleMessage<Messages::ServiceWorkerFetchTask::DidReceiveData>(decoder, this, &ServiceWorkerFetchTask::didReceiveData);
52 if (decoder.messageName() == Messages::ServiceWorkerFetchTask::DidReceiveFormData::name())
53 return IPC::handleMessage<Messages::ServiceWorkerFetchTask::DidReceiveFormData>(decoder, this, &ServiceWorkerFetchTask::didReceiveFormData);
54 if (decoder.messageName() == Messages::ServiceWorkerFetchTask::DidFinish::name())
55 return IPC::handleMessage<Messages::ServiceWorkerFetchTask::DidFinish>(decoder, this, &ServiceWorkerFetchTask::didFinish);
56 UNUSED_PARAM(connection);
57 UNUSED_PARAM(decoder);
58#if ENABLE(IPC_TESTING_API)
59 if (connection.ignoreInvalidMessageForTesting())
60 return;
61#endif // ENABLE(IPC_TESTING_API)
62 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
63}
