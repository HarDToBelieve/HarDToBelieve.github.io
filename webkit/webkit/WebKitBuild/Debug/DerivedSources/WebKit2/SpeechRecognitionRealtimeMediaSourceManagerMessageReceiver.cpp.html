41void SpeechRecognitionRealtimeMediaSourceManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
42{
43 if (decoder.messageName() == Messages::SpeechRecognitionRealtimeMediaSourceManager::CreateSource::name())
44 return IPC::handleMessage<Messages::SpeechRecognitionRealtimeMediaSourceManager::CreateSource>(decoder, this, &SpeechRecognitionRealtimeMediaSourceManager::createSource);
45 if (decoder.messageName() == Messages::SpeechRecognitionRealtimeMediaSourceManager::DeleteSource::name())
46 return IPC::handleMessage<Messages::SpeechRecognitionRealtimeMediaSourceManager::DeleteSource>(decoder, this, &SpeechRecognitionRealtimeMediaSourceManager::deleteSource);
47 if (decoder.messageName() == Messages::SpeechRecognitionRealtimeMediaSourceManager::Start::name())
48 return IPC::handleMessage<Messages::SpeechRecognitionRealtimeMediaSourceManager::Start>(decoder, this, &SpeechRecognitionRealtimeMediaSourceManager::start);
49 if (decoder.messageName() == Messages::SpeechRecognitionRealtimeMediaSourceManager::Stop::name())
50 return IPC::handleMessage<Messages::SpeechRecognitionRealtimeMediaSourceManager::Stop>(decoder, this, &SpeechRecognitionRealtimeMediaSourceManager::stop);
51#if ENABLE(SANDBOX_EXTENSIONS)
52 if (decoder.messageName() == Messages::SpeechRecognitionRealtimeMediaSourceManager::GrantSandboxExtensions::name())
53 return IPC::handleMessage<Messages::SpeechRecognitionRealtimeMediaSourceManager::GrantSandboxExtensions>(decoder, this, &SpeechRecognitionRealtimeMediaSourceManager::grantSandboxExtensions);
54#endif
55#if ENABLE(SANDBOX_EXTENSIONS)
56 if (decoder.messageName() == Messages::SpeechRecognitionRealtimeMediaSourceManager::RevokeSandboxExtensions::name())
57 return IPC::handleMessage<Messages::SpeechRecognitionRealtimeMediaSourceManager::RevokeSandboxExtensions>(decoder, this, &SpeechRecognitionRealtimeMediaSourceManager::revokeSandboxExtensions);
58#endif
59 UNUSED_PARAM(connection);
60 UNUSED_PARAM(decoder);
61#if ENABLE(IPC_TESTING_API)
62 if (connection.ignoreInvalidMessageForTesting())
63 return;
64#endif // ENABLE(IPC_TESTING_API)
65 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
66}
