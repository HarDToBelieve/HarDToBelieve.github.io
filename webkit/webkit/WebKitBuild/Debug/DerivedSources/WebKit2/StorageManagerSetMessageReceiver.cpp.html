76void StorageManagerSet::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
77{
78 auto protectedThis = makeRef(*this);
79 if (decoder.messageName() == Messages::StorageManagerSet::DisconnectFromStorageArea::name())
80 return IPC::handleMessage<Messages::StorageManagerSet::DisconnectFromStorageArea>(connection, decoder, this, &StorageManagerSet::disconnectFromStorageArea);
81 if (decoder.messageName() == Messages::StorageManagerSet::CloneSessionStorageNamespace::name())
82 return IPC::handleMessage<Messages::StorageManagerSet::CloneSessionStorageNamespace>(connection, decoder, this, &StorageManagerSet::cloneSessionStorageNamespace);
83 if (decoder.messageName() == Messages::StorageManagerSet::SetItem::name())
84 return IPC::handleMessage<Messages::StorageManagerSet::SetItem>(connection, decoder, this, &StorageManagerSet::setItem);
85 if (decoder.messageName() == Messages::StorageManagerSet::RemoveItem::name())
86 return IPC::handleMessage<Messages::StorageManagerSet::RemoveItem>(connection, decoder, this, &StorageManagerSet::removeItem);
87 if (decoder.messageName() == Messages::StorageManagerSet::Clear::name())
88 return IPC::handleMessage<Messages::StorageManagerSet::Clear>(connection, decoder, this, &StorageManagerSet::clear);
89 UNUSED_PARAM(connection);
90 UNUSED_PARAM(decoder);
91#if ENABLE(IPC_TESTING_API)
92 if (connection.ignoreInvalidMessageForTesting())
93 return;
94#endif // ENABLE(IPC_TESTING_API)
95 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
96}
97
98bool StorageManagerSet::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
99{
100 auto protectedThis = makeRef(*this);
101 if (decoder.messageName() == Messages::StorageManagerSet::ConnectToLocalStorageArea::name())
102 return IPC::handleMessageSynchronousWantsConnection<Messages::StorageManagerSet::ConnectToLocalStorageArea>(connection, decoder, replyEncoder, this, &StorageManagerSet::connectToLocalStorageArea);
103 if (decoder.messageName() == Messages::StorageManagerSet::ConnectToTransientLocalStorageArea::name())
104 return IPC::handleMessageSynchronousWantsConnection<Messages::StorageManagerSet::ConnectToTransientLocalStorageArea>(connection, decoder, replyEncoder, this, &StorageManagerSet::connectToTransientLocalStorageArea);
105 if (decoder.messageName() == Messages::StorageManagerSet::ConnectToSessionStorageArea::name())
106 return IPC::handleMessageSynchronousWantsConnection<Messages::StorageManagerSet::ConnectToSessionStorageArea>(connection, decoder, replyEncoder, this, &StorageManagerSet::connectToSessionStorageArea);
107 if (decoder.messageName() == Messages::StorageManagerSet::GetValues::name())
108 return IPC::handleMessageSynchronousWantsConnection<Messages::StorageManagerSet::GetValues>(connection, decoder, replyEncoder, this, &StorageManagerSet::getValues);
109 UNUSED_PARAM(connection);
110 UNUSED_PARAM(decoder);
111 UNUSED_PARAM(replyEncoder);
112#if ENABLE(IPC_TESTING_API)
113 if (connection.ignoreInvalidMessageForTesting())
114 return false;
115#endif // ENABLE(IPC_TESTING_API)
116 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
117 return false;
118}
