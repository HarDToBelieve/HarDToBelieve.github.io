48void CreateMediaSourceForCaptureDeviceWithConstraints::callReply(IPC::Decoder& decoder, CompletionHandler<void(bool&&, String&&, WebCore::RealtimeMediaSourceSettings&&, WebCore::RealtimeMediaSourceCapabilities&&, Vector<WebCore::VideoPresetData>&&, WebCore::IntSize&&, double&&)>&& completionHandler)
49{
50 std::optional<bool> success;
51 decoder >> success;
52 if (!success) {
53 ASSERT_NOT_REACHED();
54 cancelReply(WTFMove(completionHandler));
55 return;
56 }
57 std::optional<String> invalidConstraints;
58 decoder >> invalidConstraints;
59 if (!invalidConstraints) {
60 ASSERT_NOT_REACHED();
61 cancelReply(WTFMove(completionHandler));
62 return;
63 }
64 std::optional<WebCore::RealtimeMediaSourceSettings> settings;
65 decoder >> settings;
66 if (!settings) {
67 ASSERT_NOT_REACHED();
68 cancelReply(WTFMove(completionHandler));
69 return;
70 }
71 std::optional<WebCore::RealtimeMediaSourceCapabilities> capabilities;
72 decoder >> capabilities;
73 if (!capabilities) {
74 ASSERT_NOT_REACHED();
75 cancelReply(WTFMove(completionHandler));
76 return;
77 }
78 std::optional<Vector<WebCore::VideoPresetData>> presets;
79 decoder >> presets;
80 if (!presets) {
81 ASSERT_NOT_REACHED();
82 cancelReply(WTFMove(completionHandler));
83 return;
84 }
85 std::optional<WebCore::IntSize> size;
86 decoder >> size;
87 if (!size) {
88 ASSERT_NOT_REACHED();
89 cancelReply(WTFMove(completionHandler));
90 return;
91 }
92 std::optional<double> frameRate;
93 decoder >> frameRate;
94 if (!frameRate) {
95 ASSERT_NOT_REACHED();
96 cancelReply(WTFMove(completionHandler));
97 return;
98 }
99 completionHandler(WTFMove(*success), WTFMove(*invalidConstraints), WTFMove(*settings), WTFMove(*capabilities), WTFMove(*presets), WTFMove(*size), WTFMove(*frameRate));
100}

125void UserMediaCaptureManagerProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
126{
127 if (decoder.messageName() == Messages::UserMediaCaptureManagerProxy::CreateMediaSourceForCaptureDeviceWithConstraints::name())

129 if (decoder.messageName() == Messages::UserMediaCaptureManagerProxy::StartProducingData::name())
130 return IPC::handleMessage<Messages::UserMediaCaptureManagerProxy::StartProducingData>(decoder, this, &UserMediaCaptureManagerProxy::startProducingData);
131 if (decoder.messageName() == Messages::UserMediaCaptureManagerProxy::StopProducingData::name())
132 return IPC::handleMessage<Messages::UserMediaCaptureManagerProxy::StopProducingData>(decoder, this, &UserMediaCaptureManagerProxy::stopProducingData);
133 if (decoder.messageName() == Messages::UserMediaCaptureManagerProxy::End::name())
134 return IPC::handleMessage<Messages::UserMediaCaptureManagerProxy::End>(decoder, this, &UserMediaCaptureManagerProxy::end);
135 if (decoder.messageName() == Messages::UserMediaCaptureManagerProxy::ApplyConstraints::name())
136 return IPC::handleMessage<Messages::UserMediaCaptureManagerProxy::ApplyConstraints>(decoder, this, &UserMediaCaptureManagerProxy::applyConstraints);
137 if (decoder.messageName() == Messages::UserMediaCaptureManagerProxy::Clone::name())
138 return IPC::handleMessage<Messages::UserMediaCaptureManagerProxy::Clone>(decoder, this, &UserMediaCaptureManagerProxy::clone);
139 if (decoder.messageName() == Messages::UserMediaCaptureManagerProxy::RequestToEnd::name())
140 return IPC::handleMessage<Messages::UserMediaCaptureManagerProxy::RequestToEnd>(decoder, this, &UserMediaCaptureManagerProxy::requestToEnd);
141 if (decoder.messageName() == Messages::UserMediaCaptureManagerProxy::SetShouldApplyRotation::name())
142 return IPC::handleMessage<Messages::UserMediaCaptureManagerProxy::SetShouldApplyRotation>(decoder, this, &UserMediaCaptureManagerProxy::setShouldApplyRotation);
143 UNUSED_PARAM(connection);
144 UNUSED_PARAM(decoder);
145#if ENABLE(IPC_TESTING_API)
146 if (connection.ignoreInvalidMessageForTesting())
147 return;
148#endif // ENABLE(IPC_TESTING_API)

150}
