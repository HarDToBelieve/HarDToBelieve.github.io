47void RequestRouteSharingPolicyAndContextUID::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebCore::RouteSharingPolicy&&, String&&)>&& completionHandler)
48{
49 std::optional<WebCore::RouteSharingPolicy> routeSharingPolicy;
50 decoder >> routeSharingPolicy;
51 if (!routeSharingPolicy) {
52 ASSERT_NOT_REACHED();
53 cancelReply(WTFMove(completionHandler));
54 return;
55 }
56 std::optional<String> routingContextUID;
57 decoder >> routingContextUID;
58 if (!routingContextUID) {
59 ASSERT_NOT_REACHED();
60 cancelReply(WTFMove(completionHandler));
61 return;
62 }
63 completionHandler(WTFMove(*routeSharingPolicy), WTFMove(*routingContextUID));
64}
65
66void RequestRouteSharingPolicyAndContextUID::cancelReply(CompletionHandler<void(WebCore::RouteSharingPolicy&&, String&&)>&& completionHandler)
67{
68 completionHandler(IPC::AsyncReplyError<WebCore::RouteSharingPolicy>::create(), IPC::AsyncReplyError<String>::create());
69}
70
71void RequestRouteSharingPolicyAndContextUID::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, WebCore::RouteSharingPolicy routeSharingPolicy, const String& routingContextUID)
72{
73 encoder.get() << routeSharingPolicy;
74 encoder.get() << routingContextUID;
75 connection.sendSyncReply(WTFMove(encoder));
76}

84void VideoFullscreenManager::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
85{
86 auto protectedThis = makeRef(*this);
87 if (decoder.messageName() == Messages::VideoFullscreenManager::RequestFullscreenMode::name())
88 return IPC::handleMessage<Messages::VideoFullscreenManager::RequestFullscreenMode>(decoder, this, &VideoFullscreenManager::requestFullscreenMode);
89 if (decoder.messageName() == Messages::VideoFullscreenManager::RequestUpdateInlineRect::name())
90 return IPC::handleMessage<Messages::VideoFullscreenManager::RequestUpdateInlineRect>(decoder, this, &VideoFullscreenManager::requestUpdateInlineRect);
91 if (decoder.messageName() == Messages::VideoFullscreenManager::RequestVideoContentLayer::name())
92 return IPC::handleMessage<Messages::VideoFullscreenManager::RequestVideoContentLayer>(decoder, this, &VideoFullscreenManager::requestVideoContentLayer);
93 if (decoder.messageName() == Messages::VideoFullscreenManager::ReturnVideoContentLayer::name())
94 return IPC::handleMessage<Messages::VideoFullscreenManager::ReturnVideoContentLayer>(decoder, this, &VideoFullscreenManager::returnVideoContentLayer);
95#if !PLATFORM(IOS_FAMILY)
96 if (decoder.messageName() == Messages::VideoFullscreenManager::DidSetupFullscreen::name())
97 return IPC::handleMessage<Messages::VideoFullscreenManager::DidSetupFullscreen>(decoder, this, &VideoFullscreenManager::didSetupFullscreen);
98#endif
99 if (decoder.messageName() == Messages::VideoFullscreenManager::WillExitFullscreen::name())
100 return IPC::handleMessage<Messages::VideoFullscreenManager::WillExitFullscreen>(decoder, this, &VideoFullscreenManager::willExitFullscreen);
101 if (decoder.messageName() == Messages::VideoFullscreenManager::DidExitFullscreen::name())
102 return IPC::handleMessage<Messages::VideoFullscreenManager::DidExitFullscreen>(decoder, this, &VideoFullscreenManager::didExitFullscreen);
103 if (decoder.messageName() == Messages::VideoFullscreenManager::DidEnterFullscreen::name())
104 return IPC::handleMessage<Messages::VideoFullscreenManager::DidEnterFullscreen>(decoder, this, &VideoFullscreenManager::didEnterFullscreen);
105 if (decoder.messageName() == Messages::VideoFullscreenManager::DidCleanupFullscreen::name())
106 return IPC::handleMessage<Messages::VideoFullscreenManager::DidCleanupFullscreen>(decoder, this, &VideoFullscreenManager::didCleanupFullscreen);
107 if (decoder.messageName() == Messages::VideoFullscreenManager::SetVideoLayerFrameFenced::name())
108 return IPC::handleMessage<Messages::VideoFullscreenManager::SetVideoLayerFrameFenced>(decoder, this, &VideoFullscreenManager::setVideoLayerFrameFenced);
109 if (decoder.messageName() == Messages::VideoFullscreenManager::SetVideoLayerGravityEnum::name())
110 return IPC::handleMessage<Messages::VideoFullscreenManager::SetVideoLayerGravityEnum>(decoder, this, &VideoFullscreenManager::setVideoLayerGravityEnum);
111 if (decoder.messageName() == Messages::VideoFullscreenManager::FullscreenModeChanged::name())
112 return IPC::handleMessage<Messages::VideoFullscreenManager::FullscreenModeChanged>(decoder, this, &VideoFullscreenManager::fullscreenModeChanged);
113 if (decoder.messageName() == Messages::VideoFullscreenManager::FullscreenMayReturnToInline::name())
114 return IPC::handleMessage<Messages::VideoFullscreenManager::FullscreenMayReturnToInline>(decoder, this, &VideoFullscreenManager::fullscreenMayReturnToInline);
115 if (decoder.messageName() == Messages::VideoFullscreenManager::RequestRouteSharingPolicyAndContextUID::name())
116 return IPC::handleMessageAsync<Messages::VideoFullscreenManager::RequestRouteSharingPolicyAndContextUID>(connection, decoder, this, &VideoFullscreenManager::requestRouteSharingPolicyAndContextUID);
117 UNUSED_PARAM(connection);
118 UNUSED_PARAM(decoder);
119#if ENABLE(IPC_TESTING_API)
120 if (connection.ignoreInvalidMessageForTesting())
121 return;
122#endif // ENABLE(IPC_TESTING_API)
123 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
124}
