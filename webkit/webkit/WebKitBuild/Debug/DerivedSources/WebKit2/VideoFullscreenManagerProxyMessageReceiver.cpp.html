43void ExitFullscreen::callReply(IPC::Decoder& decoder, CompletionHandler<void(bool&&)>&& completionHandler)
44{
45 std::optional<bool> success;
46 decoder >> success;
47 if (!success) {
48 ASSERT_NOT_REACHED();
49 cancelReply(WTFMove(completionHandler));
50 return;
51 }
52 completionHandler(WTFMove(*success));
53}
54
55void ExitFullscreen::cancelReply(CompletionHandler<void(bool&&)>&& completionHandler)
56{
57 completionHandler(IPC::AsyncReplyError<bool>::create());
58}
59
60void ExitFullscreen::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, bool success)
61{
62 encoder.get() << success;
63 connection.sendSyncReply(WTFMove(encoder));
64}

72void VideoFullscreenManagerProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
73{
74 auto protectedThis = makeRef(*this);
75 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::SetHasVideo::name())
76 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::SetHasVideo>(decoder, this, &VideoFullscreenManagerProxy::setHasVideo);
77 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::SetVideoDimensions::name())
78 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::SetVideoDimensions>(decoder, this, &VideoFullscreenManagerProxy::setVideoDimensions);
79 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::SetupFullscreenWithID::name())
80 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::SetupFullscreenWithID>(decoder, this, &VideoFullscreenManagerProxy::setupFullscreenWithID);
81#if !PLATFORM(IOS_FAMILY)
82 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::EnterFullscreen::name())
83 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::EnterFullscreen>(decoder, this, &VideoFullscreenManagerProxy::enterFullscreen);
84#endif
85 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::ExitFullscreen::name())
86 return IPC::handleMessageAsync<Messages::VideoFullscreenManagerProxy::ExitFullscreen>(connection, decoder, this, &VideoFullscreenManagerProxy::exitFullscreen);
87 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::SetInlineRect::name())
88 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::SetInlineRect>(decoder, this, &VideoFullscreenManagerProxy::setInlineRect);
89 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::SetHasVideoContentLayer::name())
90 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::SetHasVideoContentLayer>(decoder, this, &VideoFullscreenManagerProxy::setHasVideoContentLayer);
91 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::CleanupFullscreen::name())
92 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::CleanupFullscreen>(decoder, this, &VideoFullscreenManagerProxy::cleanupFullscreen);
93 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::PreparedToReturnToInline::name())
94 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::PreparedToReturnToInline>(decoder, this, &VideoFullscreenManagerProxy::preparedToReturnToInline);
95 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::PreparedToExitFullscreen::name())
96 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::PreparedToExitFullscreen>(decoder, this, &VideoFullscreenManagerProxy::preparedToExitFullscreen);
97 if (decoder.messageName() == Messages::VideoFullscreenManagerProxy::ExitFullscreenWithoutAnimationToMode::name())
98 return IPC::handleMessage<Messages::VideoFullscreenManagerProxy::ExitFullscreenWithoutAnimationToMode>(decoder, this, &VideoFullscreenManagerProxy::exitFullscreenWithoutAnimationToMode);
99 UNUSED_PARAM(connection);
100 UNUSED_PARAM(decoder);
101#if ENABLE(IPC_TESTING_API)
102 if (connection.ignoreInvalidMessageForTesting())
103 return;
104#endif // ENABLE(IPC_TESTING_API)
105 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
106}
