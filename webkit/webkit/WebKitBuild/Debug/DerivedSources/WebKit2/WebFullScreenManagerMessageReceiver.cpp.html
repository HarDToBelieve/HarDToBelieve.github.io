38void WebFullScreenManager::didReceiveWebFullScreenManagerMessage(IPC::Connection& connection, IPC::Decoder& decoder)
39{
40 auto protectedThis = makeRef(*this);
41 if (decoder.messageName() == Messages::WebFullScreenManager::RequestEnterFullScreen::name())
42 return IPC::handleMessage<Messages::WebFullScreenManager::RequestEnterFullScreen>(decoder, this, &WebFullScreenManager::requestEnterFullScreen);
43 if (decoder.messageName() == Messages::WebFullScreenManager::RequestExitFullScreen::name())
44 return IPC::handleMessage<Messages::WebFullScreenManager::RequestExitFullScreen>(decoder, this, &WebFullScreenManager::requestExitFullScreen);
45 if (decoder.messageName() == Messages::WebFullScreenManager::WillEnterFullScreen::name())
46 return IPC::handleMessage<Messages::WebFullScreenManager::WillEnterFullScreen>(decoder, this, &WebFullScreenManager::willEnterFullScreen);
47 if (decoder.messageName() == Messages::WebFullScreenManager::DidEnterFullScreen::name())
48 return IPC::handleMessage<Messages::WebFullScreenManager::DidEnterFullScreen>(decoder, this, &WebFullScreenManager::didEnterFullScreen);
49 if (decoder.messageName() == Messages::WebFullScreenManager::WillExitFullScreen::name())
50 return IPC::handleMessage<Messages::WebFullScreenManager::WillExitFullScreen>(decoder, this, &WebFullScreenManager::willExitFullScreen);
51 if (decoder.messageName() == Messages::WebFullScreenManager::DidExitFullScreen::name())
52 return IPC::handleMessage<Messages::WebFullScreenManager::DidExitFullScreen>(decoder, this, &WebFullScreenManager::didExitFullScreen);
53 if (decoder.messageName() == Messages::WebFullScreenManager::SetAnimatingFullScreen::name())
54 return IPC::handleMessage<Messages::WebFullScreenManager::SetAnimatingFullScreen>(decoder, this, &WebFullScreenManager::setAnimatingFullScreen);
55 if (decoder.messageName() == Messages::WebFullScreenManager::SaveScrollPosition::name())
56 return IPC::handleMessage<Messages::WebFullScreenManager::SaveScrollPosition>(decoder, this, &WebFullScreenManager::saveScrollPosition);
57 if (decoder.messageName() == Messages::WebFullScreenManager::RestoreScrollPosition::name())
58 return IPC::handleMessage<Messages::WebFullScreenManager::RestoreScrollPosition>(decoder, this, &WebFullScreenManager::restoreScrollPosition);
59 if (decoder.messageName() == Messages::WebFullScreenManager::SetFullscreenInsets::name())
60 return IPC::handleMessage<Messages::WebFullScreenManager::SetFullscreenInsets>(decoder, this, &WebFullScreenManager::setFullscreenInsets);
61 if (decoder.messageName() == Messages::WebFullScreenManager::SetFullscreenAutoHideDuration::name())
62 return IPC::handleMessage<Messages::WebFullScreenManager::SetFullscreenAutoHideDuration>(decoder, this, &WebFullScreenManager::setFullscreenAutoHideDuration);
63 if (decoder.messageName() == Messages::WebFullScreenManager::SetFullscreenControlsHidden::name())
64 return IPC::handleMessage<Messages::WebFullScreenManager::SetFullscreenControlsHidden>(decoder, this, &WebFullScreenManager::setFullscreenControlsHidden);
65 UNUSED_PARAM(connection);
66 UNUSED_PARAM(decoder);
67#if ENABLE(IPC_TESTING_API)
68 if (connection.ignoreInvalidMessageForTesting())
69 return;
70#endif // ENABLE(IPC_TESTING_API)
71 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
72}
