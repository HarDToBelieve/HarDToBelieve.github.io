51void WebIDBServer::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
52{
53 auto protectedThis = makeRef(*this);
54 if (decoder.messageName() == Messages::WebIDBServer::DeleteDatabase::name())
55 return IPC::handleMessage<Messages::WebIDBServer::DeleteDatabase>(decoder, this, &WebIDBServer::deleteDatabase);
56 if (decoder.messageName() == Messages::WebIDBServer::OpenDatabase::name())
57 return IPC::handleMessage<Messages::WebIDBServer::OpenDatabase>(decoder, this, &WebIDBServer::openDatabase);
58 if (decoder.messageName() == Messages::WebIDBServer::AbortTransaction::name())
59 return IPC::handleMessage<Messages::WebIDBServer::AbortTransaction>(decoder, this, &WebIDBServer::abortTransaction);
60 if (decoder.messageName() == Messages::WebIDBServer::CommitTransaction::name())
61 return IPC::handleMessage<Messages::WebIDBServer::CommitTransaction>(decoder, this, &WebIDBServer::commitTransaction);
62 if (decoder.messageName() == Messages::WebIDBServer::DidFinishHandlingVersionChangeTransaction::name())
63 return IPC::handleMessage<Messages::WebIDBServer::DidFinishHandlingVersionChangeTransaction>(decoder, this, &WebIDBServer::didFinishHandlingVersionChangeTransaction);
64 if (decoder.messageName() == Messages::WebIDBServer::CreateObjectStore::name())
65 return IPC::handleMessage<Messages::WebIDBServer::CreateObjectStore>(decoder, this, &WebIDBServer::createObjectStore);
66 if (decoder.messageName() == Messages::WebIDBServer::DeleteObjectStore::name())
67 return IPC::handleMessage<Messages::WebIDBServer::DeleteObjectStore>(decoder, this, &WebIDBServer::deleteObjectStore);
68 if (decoder.messageName() == Messages::WebIDBServer::RenameObjectStore::name())
69 return IPC::handleMessage<Messages::WebIDBServer::RenameObjectStore>(decoder, this, &WebIDBServer::renameObjectStore);
70 if (decoder.messageName() == Messages::WebIDBServer::ClearObjectStore::name())
71 return IPC::handleMessage<Messages::WebIDBServer::ClearObjectStore>(decoder, this, &WebIDBServer::clearObjectStore);
72 if (decoder.messageName() == Messages::WebIDBServer::CreateIndex::name())
73 return IPC::handleMessage<Messages::WebIDBServer::CreateIndex>(decoder, this, &WebIDBServer::createIndex);
74 if (decoder.messageName() == Messages::WebIDBServer::DeleteIndex::name())
75 return IPC::handleMessage<Messages::WebIDBServer::DeleteIndex>(decoder, this, &WebIDBServer::deleteIndex);
76 if (decoder.messageName() == Messages::WebIDBServer::RenameIndex::name())
77 return IPC::handleMessage<Messages::WebIDBServer::RenameIndex>(decoder, this, &WebIDBServer::renameIndex);
78 if (decoder.messageName() == Messages::WebIDBServer::PutOrAdd::name())
79 return IPC::handleMessage<Messages::WebIDBServer::PutOrAdd>(decoder, this, &WebIDBServer::putOrAdd);
80 if (decoder.messageName() == Messages::WebIDBServer::GetRecord::name())
81 return IPC::handleMessage<Messages::WebIDBServer::GetRecord>(decoder, this, &WebIDBServer::getRecord);
82 if (decoder.messageName() == Messages::WebIDBServer::GetAllRecords::name())
83 return IPC::handleMessage<Messages::WebIDBServer::GetAllRecords>(decoder, this, &WebIDBServer::getAllRecords);
84 if (decoder.messageName() == Messages::WebIDBServer::GetCount::name())
85 return IPC::handleMessage<Messages::WebIDBServer::GetCount>(decoder, this, &WebIDBServer::getCount);
86 if (decoder.messageName() == Messages::WebIDBServer::DeleteRecord::name())
87 return IPC::handleMessage<Messages::WebIDBServer::DeleteRecord>(decoder, this, &WebIDBServer::deleteRecord);
88 if (decoder.messageName() == Messages::WebIDBServer::OpenCursor::name())
89 return IPC::handleMessage<Messages::WebIDBServer::OpenCursor>(decoder, this, &WebIDBServer::openCursor);
90 if (decoder.messageName() == Messages::WebIDBServer::IterateCursor::name())
91 return IPC::handleMessage<Messages::WebIDBServer::IterateCursor>(decoder, this, &WebIDBServer::iterateCursor);
92 if (decoder.messageName() == Messages::WebIDBServer::EstablishTransaction::name())
93 return IPC::handleMessage<Messages::WebIDBServer::EstablishTransaction>(decoder, this, &WebIDBServer::establishTransaction);
94 if (decoder.messageName() == Messages::WebIDBServer::DatabaseConnectionPendingClose::name())
95 return IPC::handleMessage<Messages::WebIDBServer::DatabaseConnectionPendingClose>(decoder, this, &WebIDBServer::databaseConnectionPendingClose);
96 if (decoder.messageName() == Messages::WebIDBServer::DatabaseConnectionClosed::name())
97 return IPC::handleMessage<Messages::WebIDBServer::DatabaseConnectionClosed>(decoder, this, &WebIDBServer::databaseConnectionClosed);
98 if (decoder.messageName() == Messages::WebIDBServer::AbortOpenAndUpgradeNeeded::name())
99 return IPC::handleMessage<Messages::WebIDBServer::AbortOpenAndUpgradeNeeded>(decoder, this, &WebIDBServer::abortOpenAndUpgradeNeeded);
100 if (decoder.messageName() == Messages::WebIDBServer::DidFireVersionChangeEvent::name())
101 return IPC::handleMessage<Messages::WebIDBServer::DidFireVersionChangeEvent>(decoder, this, &WebIDBServer::didFireVersionChangeEvent);
102 if (decoder.messageName() == Messages::WebIDBServer::OpenDBRequestCancelled::name())
103 return IPC::handleMessage<Messages::WebIDBServer::OpenDBRequestCancelled>(decoder, this, &WebIDBServer::openDBRequestCancelled);
104 if (decoder.messageName() == Messages::WebIDBServer::GetAllDatabaseNamesAndVersions::name())
105 return IPC::handleMessage<Messages::WebIDBServer::GetAllDatabaseNamesAndVersions>(connection, decoder, this, &WebIDBServer::getAllDatabaseNamesAndVersions);
106 UNUSED_PARAM(connection);
107 UNUSED_PARAM(decoder);
108#if ENABLE(IPC_TESTING_API)
109 if (connection.ignoreInvalidMessageForTesting())
110 return;
111#endif // ENABLE(IPC_TESTING_API)
112 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
113}
