46void RegisterExtension::callReply(IPC::Decoder& decoder, CompletionHandler<void(Expected<bool, Inspector::ExtensionError>&&)>&& completionHandler)
47{
48 std::optional<Expected<bool, Inspector::ExtensionError>> result;
49 decoder >> result;
50 if (!result) {
51 ASSERT_NOT_REACHED();
52 cancelReply(WTFMove(completionHandler));
53 return;
54 }
55 completionHandler(WTFMove(*result));
56}

61}

115void EvaluateScriptForExtension::callReply(IPC::Decoder& decoder, CompletionHandler<void(IPC::DataReference&&, std::optional<WebCore::ExceptionDetails>&&, std::optional<Inspector::ExtensionError>&&)>&& completionHandler)
116{
117 std::optional<IPC::DataReference> resultData;
118 decoder >> resultData;
119 if (!resultData) {
120 ASSERT_NOT_REACHED();
121 cancelReply(WTFMove(completionHandler));
122 return;
123 }
124 std::optional<std::optional<WebCore::ExceptionDetails>> details;
125 decoder >> details;
126 if (!details) {
127 ASSERT_NOT_REACHED();
128 cancelReply(WTFMove(completionHandler));
129 return;
130 }
131 std::optional<std::optional<Inspector::ExtensionError>> error;
132 decoder >> error;
133 if (!error) {
134 ASSERT_NOT_REACHED();
135 cancelReply(WTFMove(completionHandler));
136 return;
137 }
138 completionHandler(WTFMove(*resultData), WTFMove(*details), WTFMove(*error));
139}

143 completionHandler(IPC::AsyncReplyError<IPC::DataReference>::create(), IPC::AsyncReplyError<std::optional<WebCore::ExceptionDetails>>::create(), IPC::AsyncReplyError<std::optional<Inspector::ExtensionError>>::create());
144}

183void WebInspectorUIExtensionController::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
184{
185 if (decoder.messageName() == Messages::WebInspectorUIExtensionController::RegisterExtension::name())
186 return IPC::handleMessageAsync<Messages::WebInspectorUIExtensionController::RegisterExtension>(connection, decoder, this, &WebInspectorUIExtensionController::registerExtension);
187 if (decoder.messageName() == Messages::WebInspectorUIExtensionController::UnregisterExtension::name())
188 return IPC::handleMessageAsync<Messages::WebInspectorUIExtensionController::UnregisterExtension>(connection, decoder, this, &WebInspectorUIExtensionController::unregisterExtension);
189 if (decoder.messageName() == Messages::WebInspectorUIExtensionController::CreateTabForExtension::name())
190 return IPC::handleMessageAsync<Messages::WebInspectorUIExtensionController::CreateTabForExtension>(connection, decoder, this, &WebInspectorUIExtensionController::createTabForExtension);
191 if (decoder.messageName() == Messages::WebInspectorUIExtensionController::EvaluateScriptForExtension::name())
192 return IPC::handleMessageAsync<Messages::WebInspectorUIExtensionController::EvaluateScriptForExtension>(connection, decoder, this, &WebInspectorUIExtensionController::evaluateScriptForExtension);
193 if (decoder.messageName() == Messages::WebInspectorUIExtensionController::ReloadForExtension::name())
194 return IPC::handleMessageAsync<Messages::WebInspectorUIExtensionController::ReloadForExtension>(connection, decoder, this, &WebInspectorUIExtensionController::reloadForExtension);
195 UNUSED_PARAM(connection);
196 UNUSED_PARAM(decoder);
197#if ENABLE(IPC_TESTING_API)
198 if (connection.ignoreInvalidMessageForTesting())
199 return;
200#endif // ENABLE(IPC_TESTING_API)
201 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
202}
