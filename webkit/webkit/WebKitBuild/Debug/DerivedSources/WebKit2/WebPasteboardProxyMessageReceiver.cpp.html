340void WebPasteboardProxy::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
341{
342#if PLATFORM(IOS_FAMILY)
343 if (decoder.messageName() == Messages::WebPasteboardProxy::WriteURLToPasteboard::name())
344 return IPC::handleMessage<Messages::WebPasteboardProxy::WriteURLToPasteboard>(connection, decoder, this, &WebPasteboardProxy::writeURLToPasteboard);
345#endif
346#if PLATFORM(IOS_FAMILY)
347 if (decoder.messageName() == Messages::WebPasteboardProxy::WriteWebContentToPasteboard::name())
348 return IPC::handleMessage<Messages::WebPasteboardProxy::WriteWebContentToPasteboard>(connection, decoder, this, &WebPasteboardProxy::writeWebContentToPasteboard);
349#endif
350#if PLATFORM(IOS_FAMILY)
351 if (decoder.messageName() == Messages::WebPasteboardProxy::WriteImageToPasteboard::name())
352 return IPC::handleMessage<Messages::WebPasteboardProxy::WriteImageToPasteboard>(connection, decoder, this, &WebPasteboardProxy::writeImageToPasteboard);
353#endif
354#if PLATFORM(IOS_FAMILY)
355 if (decoder.messageName() == Messages::WebPasteboardProxy::WriteStringToPasteboard::name())
356 return IPC::handleMessage<Messages::WebPasteboardProxy::WriteStringToPasteboard>(connection, decoder, this, &WebPasteboardProxy::writeStringToPasteboard);
357#endif
358#if PLATFORM(IOS_FAMILY)
359 if (decoder.messageName() == Messages::WebPasteboardProxy::UpdateSupportedTypeIdentifiers::name())
360 return IPC::handleMessage<Messages::WebPasteboardProxy::UpdateSupportedTypeIdentifiers>(decoder, this, &WebPasteboardProxy::updateSupportedTypeIdentifiers);
361#endif
362#if PLATFORM(GTK)
363 if (decoder.messageName() == Messages::WebPasteboardProxy::WriteToClipboard::name())
364 return IPC::handleMessage<Messages::WebPasteboardProxy::WriteToClipboard>(decoder, this, &WebPasteboardProxy::writeToClipboard);
365#endif
366#if PLATFORM(GTK)
367 if (decoder.messageName() == Messages::WebPasteboardProxy::ClearClipboard::name())
368 return IPC::handleMessage<Messages::WebPasteboardProxy::ClearClipboard>(decoder, this, &WebPasteboardProxy::clearClipboard);
369#endif
370#if USE(LIBWPE)
371 if (decoder.messageName() == Messages::WebPasteboardProxy::WriteWebContentToPasteboard::name())
372 return IPC::handleMessage<Messages::WebPasteboardProxy::WriteWebContentToPasteboard>(decoder, this, &WebPasteboardProxy::writeWebContentToPasteboard);
373#endif
374#if USE(LIBWPE)
375 if (decoder.messageName() == Messages::WebPasteboardProxy::WriteStringToPasteboard::name())
376 return IPC::handleMessage<Messages::WebPasteboardProxy::WriteStringToPasteboard>(decoder, this, &WebPasteboardProxy::writeStringToPasteboard);
377#endif
378 UNUSED_PARAM(connection);
379 UNUSED_PARAM(decoder);
380#if ENABLE(IPC_TESTING_API)
381 if (connection.ignoreInvalidMessageForTesting())
382 return;
383#endif // ENABLE(IPC_TESTING_API)
384 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
385}

387bool WebPasteboardProxy::didReceiveSyncMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
388{
389 if (decoder.messageName() == Messages::WebPasteboardProxy::WriteCustomData::name())
390 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::WriteCustomData>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::writeCustomData);
391 if (decoder.messageName() == Messages::WebPasteboardProxy::TypesSafeForDOMToReadAndWrite::name())
392 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::TypesSafeForDOMToReadAndWrite>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::typesSafeForDOMToReadAndWrite);
393 if (decoder.messageName() == Messages::WebPasteboardProxy::AllPasteboardItemInfo::name())
394 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::AllPasteboardItemInfo>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::allPasteboardItemInfo);
395 if (decoder.messageName() == Messages::WebPasteboardProxy::InformationForItemAtIndex::name())
396 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::InformationForItemAtIndex>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::informationForItemAtIndex);
397 if (decoder.messageName() == Messages::WebPasteboardProxy::GetPasteboardItemsCount::name())
398 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::GetPasteboardItemsCount>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::getPasteboardItemsCount);
399 if (decoder.messageName() == Messages::WebPasteboardProxy::ReadStringFromPasteboard::name())
400 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::ReadStringFromPasteboard>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::readStringFromPasteboard);
401 if (decoder.messageName() == Messages::WebPasteboardProxy::ReadURLFromPasteboard::name())
402 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::ReadURLFromPasteboard>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::readURLFromPasteboard);
403 if (decoder.messageName() == Messages::WebPasteboardProxy::ReadBufferFromPasteboard::name())
404 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::ReadBufferFromPasteboard>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::readBufferFromPasteboard);
405 if (decoder.messageName() == Messages::WebPasteboardProxy::ContainsStringSafeForDOMToReadForType::name())
406 return IPC::handleMessageSynchronousWantsConnection<Messages::WebPasteboardProxy::ContainsStringSafeForDOMToReadForType>(connection, decoder, replyEncoder, this, &WebPasteboardProxy::containsStringSafeForDOMToReadForType);
