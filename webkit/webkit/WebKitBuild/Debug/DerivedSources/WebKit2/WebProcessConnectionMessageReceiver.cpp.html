38void CreatePlugin::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, bool creationResult, bool wantsWheelEvents, uint32_t remoteLayerClientID)
39{
40 encoder.get() << creationResult;
41 encoder.get() << wantsWheelEvents;
42 encoder.get() << remoteLayerClientID;
43 connection.sendSyncReply(WTFMove(encoder));
44}
45
46void DestroyPlugin::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection)
47{
48 connection.sendSyncReply(WTFMove(encoder));
49}

57void WebProcessConnection::didReceiveWebProcessConnectionMessage(IPC::Connection& connection, IPC::Decoder& decoder)
58{
59 auto protectedThis = makeRef(*this);
60 if (decoder.messageName() == Messages::WebProcessConnection::CreatePluginAsynchronously::name())
61 return IPC::handleMessage<Messages::WebProcessConnection::CreatePluginAsynchronously>(decoder, this, &WebProcessConnection::createPluginAsynchronously);
62 UNUSED_PARAM(connection);
63 UNUSED_PARAM(decoder);
64#if ENABLE(IPC_TESTING_API)
65 if (connection.ignoreInvalidMessageForTesting())
66 return;
67#endif // ENABLE(IPC_TESTING_API)
68 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
69}
70
71bool WebProcessConnection::didReceiveSyncWebProcessConnectionMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
72{
73 auto protectedThis = makeRef(*this);
74 if (decoder.messageName() == Messages::WebProcessConnection::CreatePlugin::name())
75 return IPC::handleMessageSynchronous<Messages::WebProcessConnection::CreatePlugin>(connection, decoder, replyEncoder, this, &WebProcessConnection::createPlugin);
76 if (decoder.messageName() == Messages::WebProcessConnection::DestroyPlugin::name())
77 return IPC::handleMessageSynchronous<Messages::WebProcessConnection::DestroyPlugin>(connection, decoder, replyEncoder, this, &WebProcessConnection::destroyPlugin);
78 UNUSED_PARAM(connection);
79 UNUSED_PARAM(decoder);
80 UNUSED_PARAM(replyEncoder);
81#if ENABLE(IPC_TESTING_API)
82 if (connection.ignoreInvalidMessageForTesting())
83 return false;
84#endif // ENABLE(IPC_TESTING_API)
85 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
86 return false;
87}
