107void FetchWebsiteData::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebKit::WebsiteData&&)>&& completionHandler)
108{
109 std::optional<WebKit::WebsiteData> websiteData;
110 decoder >> websiteData;
111 if (!websiteData) {
112 ASSERT_NOT_REACHED();
113 cancelReply(WTFMove(completionHandler));
114 return;
115 }
116 completionHandler(WTFMove(*websiteData));
117}
118
119void FetchWebsiteData::cancelReply(CompletionHandler<void(WebKit::WebsiteData&&)>&& completionHandler)
120{
121 completionHandler(IPC::AsyncReplyError<WebKit::WebsiteData>::create());
122}
123
124void FetchWebsiteData::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const WebKit::WebsiteData& websiteData)
125{
126 encoder.get() << websiteData;
127 connection.sendSyncReply(WTFMove(encoder));
128}
