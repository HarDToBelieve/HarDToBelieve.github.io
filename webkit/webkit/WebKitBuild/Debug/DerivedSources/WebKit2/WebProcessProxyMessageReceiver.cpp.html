130void SendMessageToWebContextWithReply::callReply(IPC::Decoder& decoder, CompletionHandler<void(WebKit::UserMessage&&)>&& completionHandler)
131{
132 std::optional<WebKit::UserMessage> replyMessage;
133 decoder >> replyMessage;
134 if (!replyMessage) {
135 ASSERT_NOT_REACHED();
136 cancelReply(WTFMove(completionHandler));
137 return;
138 }
139 completionHandler(WTFMove(*replyMessage));
140}
141
142void SendMessageToWebContextWithReply::cancelReply(CompletionHandler<void(WebKit::UserMessage&&)>&& completionHandler)
143{
144 completionHandler(IPC::AsyncReplyError<WebKit::UserMessage>::create());
145}
146
147void SendMessageToWebContextWithReply::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const WebKit::UserMessage& replyMessage)
148{
149 encoder.get() << replyMessage;
150 connection.sendSyncReply(WTFMove(encoder));
151}

171void WebProcessProxy::didReceiveWebProcessProxyMessage(IPC::Connection& connection, IPC::Decoder& decoder)

173 auto protectedThis = makeRef(*this);

176 if (decoder.messageName() == Messages::WebProcessProxy::DidDestroyFrame::name())
177 return IPC::handleMessage<Messages::WebProcessProxy::DidDestroyFrame>(decoder, this, &WebProcessProxy::didDestroyFrame);
178 if (decoder.messageName() == Messages::WebProcessProxy::DidDestroyUserGestureToken::name())
179 return IPC::handleMessage<Messages::WebProcessProxy::DidDestroyUserGestureToken>(decoder, this, &WebProcessProxy::didDestroyUserGestureToken);
180 if (decoder.messageName() == Messages::WebProcessProxy::EnableSuddenTermination::name())
181 return IPC::handleMessage<Messages::WebProcessProxy::EnableSuddenTermination>(decoder, this, &WebProcessProxy::enableSuddenTermination);
182 if (decoder.messageName() == Messages::WebProcessProxy::DisableSuddenTermination::name())

184 if (decoder.messageName() == Messages::WebProcessProxy::SetIsHoldingLockedFiles::name())
185 return IPC::handleMessage<Messages::WebProcessProxy::SetIsHoldingLockedFiles>(decoder, this, &WebProcessProxy::setIsHoldingLockedFiles);
186 if (decoder.messageName() == Messages::WebProcessProxy::DidExceedActiveMemoryLimit::name())
187 return IPC::handleMessage<Messages::WebProcessProxy::DidExceedActiveMemoryLimit>(decoder, this, &WebProcessProxy::didExceedActiveMemoryLimit);
188 if (decoder.messageName() == Messages::WebProcessProxy::DidExceedInactiveMemoryLimit::name())
189 return IPC::handleMessage<Messages::WebProcessProxy::DidExceedInactiveMemoryLimit>(decoder, this, &WebProcessProxy::didExceedInactiveMemoryLimit);
190 if (decoder.messageName() == Messages::WebProcessProxy::DidExceedCPULimit::name())
191 return IPC::handleMessage<Messages::WebProcessProxy::DidExceedCPULimit>(decoder, this, &WebProcessProxy::didExceedCPULimit);
192 if (decoder.messageName() == Messages::WebProcessProxy::StopResponsivenessTimer::name())
193 return IPC::handleMessage<Messages::WebProcessProxy::StopResponsivenessTimer>(decoder, this, &WebProcessProxy::stopResponsivenessTimer);
194 if (decoder.messageName() == Messages::WebProcessProxy::DidReceiveBackgroundResponsivenessPing::name())
195 return IPC::handleMessage<Messages::WebProcessProxy::DidReceiveBackgroundResponsivenessPing>(decoder, this, &WebProcessProxy::didReceiveBackgroundResponsivenessPing);
196 if (decoder.messageName() == Messages::WebProcessProxy::MemoryPressureStatusChanged::name())

253bool WebProcessProxy::didReceiveSyncWebProcessProxyMessage(IPC::Connection& connection, IPC::Decoder& decoder, UniqueRef<IPC::Encoder>& replyEncoder)
254{
255 auto protectedThis = makeRef(*this);
256 if (decoder.messageName() == Messages::WebProcessProxy::ShouldTerminate::name())
257 return IPC::handleMessageSynchronous<Messages::WebProcessProxy::ShouldTerminate>(connection, decoder, replyEncoder, this, &WebProcessProxy::shouldTerminate);
258#if ENABLE(NETSCAPE_PLUGIN_API)
259 if (decoder.messageName() == Messages::WebProcessProxy::GetPlugins::name())
260 return IPC::handleMessageSynchronous<Messages::WebProcessProxy::GetPlugins>(connection, decoder, replyEncoder, this, &WebProcessProxy::getPlugins);
261#endif
262#if ENABLE(NETSCAPE_PLUGIN_API)
263 if (decoder.messageName() == Messages::WebProcessProxy::GetPluginProcessConnection::name())
264 return IPC::handleMessageSynchronous<Messages::WebProcessProxy::GetPluginProcessConnection>(connection, decoder, replyEncoder, this, &WebProcessProxy::getPluginProcessConnection);
265#endif
266 if (decoder.messageName() == Messages::WebProcessProxy::GetNetworkProcessConnection::name())
267 return IPC::handleMessageSynchronous<Messages::WebProcessProxy::GetNetworkProcessConnection>(connection, decoder, replyEncoder, this, &WebProcessProxy::getNetworkProcessConnection);
268#if ENABLE(GPU_PROCESS)
269 if (decoder.messageName() == Messages::WebProcessProxy::GetGPUProcessConnection::name())
270 return IPC::handleMessageSynchronous<Messages::WebProcessProxy::GetGPUProcessConnection>(connection, decoder, replyEncoder, this, &WebProcessProxy::getGPUProcessConnection);
271#endif
272#if ENABLE(WEB_AUTHN)
273 if (decoder.messageName() == Messages::WebProcessProxy::GetWebAuthnProcessConnection::name())
274 return IPC::handleMessageSynchronous<Messages::WebProcessProxy::GetWebAuthnProcessConnection>(connection, decoder, replyEncoder, this, &WebProcessProxy::getWebAuthnProcessConnection);
275#endif
276#if PLATFORM(MAC)
277 if (decoder.messageName() == Messages::WebProcessProxy::IsAXAuthenticated::name())
278 return IPC::handleMessageSynchronous<Messages::WebProcessProxy::IsAXAuthenticated>(connection, decoder, replyEncoder, this, &WebProcessProxy::isAXAuthenticated);
279#endif
280 UNUSED_PARAM(connection);
281 UNUSED_PARAM(decoder);
282 UNUSED_PARAM(replyEncoder);
283#if ENABLE(IPC_TESTING_API)
284 if (connection.ignoreInvalidMessageForTesting())
285 return false;
286#endif // ENABLE(IPC_TESTING_API)
287 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled synchronous message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
288 return false;
289}
