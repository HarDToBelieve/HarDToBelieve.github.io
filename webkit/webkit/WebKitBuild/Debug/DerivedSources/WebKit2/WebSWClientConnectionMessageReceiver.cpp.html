55void SetDocumentIsControlled::callReply(IPC::Decoder& decoder, CompletionHandler<void(bool&&)>&& completionHandler)
56{
57 std::optional<bool> isSuccess;
58 decoder >> isSuccess;
59 if (!isSuccess) {
60 ASSERT_NOT_REACHED();
61 cancelReply(WTFMove(completionHandler));
62 return;
63 }
64 completionHandler(WTFMove(*isSuccess));
65}
66
67void SetDocumentIsControlled::cancelReply(CompletionHandler<void(bool&&)>&& completionHandler)
68{
69 completionHandler(IPC::AsyncReplyError<bool>::create());
70}
71
72void SetDocumentIsControlled::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, bool isSuccess)
73{
74 encoder.get() << isSuccess;
75 connection.sendSyncReply(WTFMove(encoder));
76}

84void WebSWClientConnection::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
85{
86 auto protectedThis = makeRef(*this);
87 if (decoder.messageName() == Messages::WebSWClientConnection::JobRejectedInServer::name())
88 return IPC::handleMessage<Messages::WebSWClientConnection::JobRejectedInServer>(decoder, this, &WebSWClientConnection::jobRejectedInServer);
89 if (decoder.messageName() == Messages::WebSWClientConnection::RegistrationJobResolvedInServer::name())
90 return IPC::handleMessage<Messages::WebSWClientConnection::RegistrationJobResolvedInServer>(decoder, this, &WebSWClientConnection::registrationJobResolvedInServer);
91 if (decoder.messageName() == Messages::WebSWClientConnection::StartScriptFetchForServer::name())
92 return IPC::handleMessage<Messages::WebSWClientConnection::StartScriptFetchForServer>(decoder, this, &WebSWClientConnection::startScriptFetchForServer);
93 if (decoder.messageName() == Messages::WebSWClientConnection::UpdateRegistrationState::name())
94 return IPC::handleMessage<Messages::WebSWClientConnection::UpdateRegistrationState>(decoder, this, &WebSWClientConnection::updateRegistrationState);
95 if (decoder.messageName() == Messages::WebSWClientConnection::UpdateWorkerState::name())
96 return IPC::handleMessage<Messages::WebSWClientConnection::UpdateWorkerState>(decoder, this, &WebSWClientConnection::updateWorkerState);
97 if (decoder.messageName() == Messages::WebSWClientConnection::FireUpdateFoundEvent::name())
98 return IPC::handleMessage<Messages::WebSWClientConnection::FireUpdateFoundEvent>(decoder, this, &WebSWClientConnection::fireUpdateFoundEvent);
99 if (decoder.messageName() == Messages::WebSWClientConnection::SetRegistrationLastUpdateTime::name())
100 return IPC::handleMessage<Messages::WebSWClientConnection::SetRegistrationLastUpdateTime>(decoder, this, &WebSWClientConnection::setRegistrationLastUpdateTime);
101 if (decoder.messageName() == Messages::WebSWClientConnection::SetRegistrationUpdateViaCache::name())
102 return IPC::handleMessage<Messages::WebSWClientConnection::SetRegistrationUpdateViaCache>(decoder, this, &WebSWClientConnection::setRegistrationUpdateViaCache);
103 if (decoder.messageName() == Messages::WebSWClientConnection::NotifyClientsOfControllerChange::name())
104 return IPC::handleMessage<Messages::WebSWClientConnection::NotifyClientsOfControllerChange>(decoder, this, &WebSWClientConnection::notifyClientsOfControllerChange);
105 if (decoder.messageName() == Messages::WebSWClientConnection::SetSWOriginTableIsImported::name())
106 return IPC::handleMessage<Messages::WebSWClientConnection::SetSWOriginTableIsImported>(decoder, this, &WebSWClientConnection::setSWOriginTableIsImported);
107 if (decoder.messageName() == Messages::WebSWClientConnection::SetSWOriginTableSharedMemory::name())
108 return IPC::handleMessage<Messages::WebSWClientConnection::SetSWOriginTableSharedMemory>(decoder, this, &WebSWClientConnection::setSWOriginTableSharedMemory);
109 if (decoder.messageName() == Messages::WebSWClientConnection::PostMessageToServiceWorkerClient::name())
110 return IPC::handleMessage<Messages::WebSWClientConnection::PostMessageToServiceWorkerClient>(decoder, this, &WebSWClientConnection::postMessageToServiceWorkerClient);
111 if (decoder.messageName() == Messages::WebSWClientConnection::DidMatchRegistration::name())
112 return IPC::handleMessage<Messages::WebSWClientConnection::DidMatchRegistration>(decoder, this, &WebSWClientConnection::didMatchRegistration);
113 if (decoder.messageName() == Messages::WebSWClientConnection::DidGetRegistrations::name())
114 return IPC::handleMessage<Messages::WebSWClientConnection::DidGetRegistrations>(decoder, this, &WebSWClientConnection::didGetRegistrations);
115 if (decoder.messageName() == Messages::WebSWClientConnection::RegistrationReady::name())
116 return IPC::handleMessage<Messages::WebSWClientConnection::RegistrationReady>(decoder, this, &WebSWClientConnection::registrationReady);
117 if (decoder.messageName() == Messages::WebSWClientConnection::SetDocumentIsControlled::name())
118 return IPC::handleMessageAsync<Messages::WebSWClientConnection::SetDocumentIsControlled>(connection, decoder, this, &WebSWClientConnection::setDocumentIsControlled);
119 UNUSED_PARAM(connection);
120 UNUSED_PARAM(decoder);
121#if ENABLE(IPC_TESTING_API)
122 if (connection.ignoreInvalidMessageForTesting())
123 return;
124#endif // ENABLE(IPC_TESTING_API)
125 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
126}
