49void SkipWaiting::callReply(IPC::Decoder& decoder, CompletionHandler<void()>&& completionHandler)
50{
51 completionHandler();
52}
53
54void SkipWaiting::cancelReply(CompletionHandler<void()>&& completionHandler)
55{
56 completionHandler();
57}
58
59void SkipWaiting::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection)
60{
61 connection.sendSyncReply(WTFMove(encoder));
62}
63
64void Claim::callReply(IPC::Decoder& decoder, CompletionHandler<void(std::optional<WebCore::ExceptionData>&&)>&& completionHandler)
65{
66 std::optional<std::optional<WebCore::ExceptionData>> result;
67 decoder >> result;
68 if (!result) {
69 ASSERT_NOT_REACHED();
70 cancelReply(WTFMove(completionHandler));
71 return;
72 }
73 completionHandler(WTFMove(*result));
74}
75
76void Claim::cancelReply(CompletionHandler<void(std::optional<WebCore::ExceptionData>&&)>&& completionHandler)
77{
78 completionHandler(IPC::AsyncReplyError<std::optional<WebCore::ExceptionData>>::create());
79}
80
81void Claim::send(UniqueRef<IPC::Encoder>&& encoder, IPC::Connection& connection, const std::optional<WebCore::ExceptionData>& result)
82{
83 encoder.get() << result;
84 connection.sendSyncReply(WTFMove(encoder));
85}

93void WebSWServerToContextConnection::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
94{
95 if (decoder.messageName() == Messages::WebSWServerToContextConnection::ScriptContextFailedToStart::name())
96 return IPC::handleMessage<Messages::WebSWServerToContextConnection::ScriptContextFailedToStart>(decoder, this, &WebSWServerToContextConnection::scriptContextFailedToStart);
97 if (decoder.messageName() == Messages::WebSWServerToContextConnection::ScriptContextStarted::name())
98 return IPC::handleMessage<Messages::WebSWServerToContextConnection::ScriptContextStarted>(decoder, this, &WebSWServerToContextConnection::scriptContextStarted);
99 if (decoder.messageName() == Messages::WebSWServerToContextConnection::DidFinishInstall::name())
100 return IPC::handleMessage<Messages::WebSWServerToContextConnection::DidFinishInstall>(decoder, this, &WebSWServerToContextConnection::didFinishInstall);
101 if (decoder.messageName() == Messages::WebSWServerToContextConnection::DidFinishActivation::name())
102 return IPC::handleMessage<Messages::WebSWServerToContextConnection::DidFinishActivation>(decoder, this, &WebSWServerToContextConnection::didFinishActivation);
103 if (decoder.messageName() == Messages::WebSWServerToContextConnection::SetServiceWorkerHasPendingEvents::name())
104 return IPC::handleMessage<Messages::WebSWServerToContextConnection::SetServiceWorkerHasPendingEvents>(decoder, this, &WebSWServerToContextConnection::setServiceWorkerHasPendingEvents);
105 if (decoder.messageName() == Messages::WebSWServerToContextConnection::SkipWaiting::name())
106 return IPC::handleMessageAsync<Messages::WebSWServerToContextConnection::SkipWaiting>(connection, decoder, this, &WebSWServerToContextConnection::skipWaiting);
107 if (decoder.messageName() == Messages::WebSWServerToContextConnection::WorkerTerminated::name())
108 return IPC::handleMessage<Messages::WebSWServerToContextConnection::WorkerTerminated>(decoder, this, &WebSWServerToContextConnection::workerTerminated);
109 if (decoder.messageName() == Messages::WebSWServerToContextConnection::FindClientByIdentifier::name())
110 return IPC::handleMessage<Messages::WebSWServerToContextConnection::FindClientByIdentifier>(decoder, this, &WebSWServerToContextConnection::findClientByIdentifier);
111 if (decoder.messageName() == Messages::WebSWServerToContextConnection::MatchAll::name())
112 return IPC::handleMessage<Messages::WebSWServerToContextConnection::MatchAll>(decoder, this, &WebSWServerToContextConnection::matchAll);
113 if (decoder.messageName() == Messages::WebSWServerToContextConnection::Claim::name())
114 return IPC::handleMessageAsync<Messages::WebSWServerToContextConnection::Claim>(connection, decoder, this, &WebSWServerToContextConnection::claim);
115 if (decoder.messageName() == Messages::WebSWServerToContextConnection::SetScriptResource::name())
116 return IPC::handleMessage<Messages::WebSWServerToContextConnection::SetScriptResource>(decoder, this, &WebSWServerToContextConnection::setScriptResource);
117 if (decoder.messageName() == Messages::WebSWServerToContextConnection::PostMessageToServiceWorkerClient::name())
118 return IPC::handleMessage<Messages::WebSWServerToContextConnection::PostMessageToServiceWorkerClient>(decoder, this, &WebSWServerToContextConnection::postMessageToServiceWorkerClient);
119 if (decoder.messageName() == Messages::WebSWServerToContextConnection::DidFailHeartBeatCheck::name())
120 return IPC::handleMessage<Messages::WebSWServerToContextConnection::DidFailHeartBeatCheck>(decoder, this, &WebSWServerToContextConnection::didFailHeartBeatCheck);
121 UNUSED_PARAM(connection);
122 UNUSED_PARAM(decoder);
123#if ENABLE(IPC_TESTING_API)
124 if (connection.ignoreInvalidMessageForTesting())
125 return;
126#endif // ENABLE(IPC_TESTING_API)
127 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
128}
