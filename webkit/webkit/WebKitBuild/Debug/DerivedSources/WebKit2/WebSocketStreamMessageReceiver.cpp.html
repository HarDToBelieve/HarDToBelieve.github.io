37void WebSocketStream::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
38{
39 auto protectedThis = makeRef(*this);
40 if (decoder.messageName() == Messages::WebSocketStream::DidOpenSocketStream::name())
41 return IPC::handleMessage<Messages::WebSocketStream::DidOpenSocketStream>(decoder, this, &WebSocketStream::didOpenSocketStream);
42 if (decoder.messageName() == Messages::WebSocketStream::DidCloseSocketStream::name())
43 return IPC::handleMessage<Messages::WebSocketStream::DidCloseSocketStream>(decoder, this, &WebSocketStream::didCloseSocketStream);
44 if (decoder.messageName() == Messages::WebSocketStream::DidReceiveSocketStreamData::name())
45 return IPC::handleMessage<Messages::WebSocketStream::DidReceiveSocketStreamData>(decoder, this, &WebSocketStream::didReceiveSocketStreamData);
46 if (decoder.messageName() == Messages::WebSocketStream::DidFailToReceiveSocketStreamData::name())
47 return IPC::handleMessage<Messages::WebSocketStream::DidFailToReceiveSocketStreamData>(decoder, this, &WebSocketStream::didFailToReceiveSocketStreamData);
48 if (decoder.messageName() == Messages::WebSocketStream::DidUpdateBufferedAmount::name())
49 return IPC::handleMessage<Messages::WebSocketStream::DidUpdateBufferedAmount>(decoder, this, &WebSocketStream::didUpdateBufferedAmount);
50 if (decoder.messageName() == Messages::WebSocketStream::DidFailSocketStream::name())
51 return IPC::handleMessage<Messages::WebSocketStream::DidFailSocketStream>(decoder, this, &WebSocketStream::didFailSocketStream);
52 if (decoder.messageName() == Messages::WebSocketStream::DidSendData::name())
53 return IPC::handleMessage<Messages::WebSocketStream::DidSendData>(decoder, this, &WebSocketStream::didSendData);
54 if (decoder.messageName() == Messages::WebSocketStream::DidSendHandshake::name())
55 return IPC::handleMessage<Messages::WebSocketStream::DidSendHandshake>(decoder, this, &WebSocketStream::didSendHandshake);
56 UNUSED_PARAM(connection);
57 UNUSED_PARAM(decoder);
58#if ENABLE(IPC_TESTING_API)
59 if (connection.ignoreInvalidMessageForTesting())
60 return;
61#endif // ENABLE(IPC_TESTING_API)
62 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
63}
