44void WebUserContentController::didReceiveMessage(IPC::Connection& connection, IPC::Decoder& decoder)
45{
46 auto protectedThis = makeRef(*this);
47 if (decoder.messageName() == Messages::WebUserContentController::AddContentWorlds::name())
48 return IPC::handleMessage<Messages::WebUserContentController::AddContentWorlds>(decoder, this, &WebUserContentController::addContentWorlds);
49 if (decoder.messageName() == Messages::WebUserContentController::RemoveContentWorlds::name())
50 return IPC::handleMessage<Messages::WebUserContentController::RemoveContentWorlds>(decoder, this, &WebUserContentController::removeContentWorlds);
51 if (decoder.messageName() == Messages::WebUserContentController::AddUserScripts::name())
52 return IPC::handleMessage<Messages::WebUserContentController::AddUserScripts>(decoder, this, &WebUserContentController::addUserScripts);
53 if (decoder.messageName() == Messages::WebUserContentController::RemoveUserScript::name())
54 return IPC::handleMessage<Messages::WebUserContentController::RemoveUserScript>(decoder, this, &WebUserContentController::removeUserScript);
55 if (decoder.messageName() == Messages::WebUserContentController::RemoveAllUserScripts::name())
56 return IPC::handleMessage<Messages::WebUserContentController::RemoveAllUserScripts>(decoder, this, &WebUserContentController::removeAllUserScripts);
57 if (decoder.messageName() == Messages::WebUserContentController::AddUserStyleSheets::name())
58 return IPC::handleMessage<Messages::WebUserContentController::AddUserStyleSheets>(decoder, this, &WebUserContentController::addUserStyleSheets);
59 if (decoder.messageName() == Messages::WebUserContentController::RemoveUserStyleSheet::name())
60 return IPC::handleMessage<Messages::WebUserContentController::RemoveUserStyleSheet>(decoder, this, &WebUserContentController::removeUserStyleSheet);
61 if (decoder.messageName() == Messages::WebUserContentController::RemoveAllUserStyleSheets::name())
62 return IPC::handleMessage<Messages::WebUserContentController::RemoveAllUserStyleSheets>(decoder, this, &WebUserContentController::removeAllUserStyleSheets);
63 if (decoder.messageName() == Messages::WebUserContentController::AddUserScriptMessageHandlers::name())
64 return IPC::handleMessage<Messages::WebUserContentController::AddUserScriptMessageHandlers>(decoder, this, &WebUserContentController::addUserScriptMessageHandlers);
65 if (decoder.messageName() == Messages::WebUserContentController::RemoveUserScriptMessageHandler::name())
66 return IPC::handleMessage<Messages::WebUserContentController::RemoveUserScriptMessageHandler>(decoder, this, &WebUserContentController::removeUserScriptMessageHandler);
67 if (decoder.messageName() == Messages::WebUserContentController::RemoveAllUserScriptMessageHandlersForWorlds::name())
68 return IPC::handleMessage<Messages::WebUserContentController::RemoveAllUserScriptMessageHandlersForWorlds>(decoder, this, &WebUserContentController::removeAllUserScriptMessageHandlersForWorlds);
69 if (decoder.messageName() == Messages::WebUserContentController::RemoveAllUserScriptMessageHandlers::name())
70 return IPC::handleMessage<Messages::WebUserContentController::RemoveAllUserScriptMessageHandlers>(decoder, this, &WebUserContentController::removeAllUserScriptMessageHandlers);
71#if ENABLE(CONTENT_EXTENSIONS)
72 if (decoder.messageName() == Messages::WebUserContentController::AddContentRuleLists::name())
73 return IPC::handleMessage<Messages::WebUserContentController::AddContentRuleLists>(decoder, this, &WebUserContentController::addContentRuleLists);
74#endif
75#if ENABLE(CONTENT_EXTENSIONS)
76 if (decoder.messageName() == Messages::WebUserContentController::RemoveContentRuleList::name())
77 return IPC::handleMessage<Messages::WebUserContentController::RemoveContentRuleList>(decoder, this, &WebUserContentController::removeContentRuleList);
78#endif
79#if ENABLE(CONTENT_EXTENSIONS)
80 if (decoder.messageName() == Messages::WebUserContentController::RemoveAllContentRuleLists::name())
81 return IPC::handleMessage<Messages::WebUserContentController::RemoveAllContentRuleLists>(decoder, this, &WebUserContentController::removeAllContentRuleLists);
82#endif
83 UNUSED_PARAM(connection);
84 UNUSED_PARAM(decoder);
85#if ENABLE(IPC_TESTING_API)
86 if (connection.ignoreInvalidMessageForTesting())
87 return;
88#endif // ENABLE(IPC_TESTING_API)
89 ASSERT_NOT_REACHED_WITH_MESSAGE("Unhandled message %s to %" PRIu64, description(decoder.messageName()), decoder.destinationID());
90}
