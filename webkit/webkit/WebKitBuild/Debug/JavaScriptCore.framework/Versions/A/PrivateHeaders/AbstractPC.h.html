26#pragma once
27
28#include "MacroAssemblerCodeRef.h"
29
30namespace JSC {
31
32class CallFrame;
33class VM;
34struct Instruction;
35
36class AbstractPC {
37public:
38 AbstractPC()
39 : m_mode(None)
40 {
41 }
42
43 AbstractPC(VM&, CallFrame*);
44
45#if ENABLE(JIT)
46 AbstractPC(ReturnAddressPtr ptr)
47 : m_pointer(ptr.value())
48 , m_mode(JIT)
49 {
50 assertIsTaggedWith<ReturnAddressPtrTag>(m_pointer);
51 }
52
53 bool hasJITReturnAddress() const { return m_mode == JIT; }
54 ReturnAddressPtr jitReturnAddress() const
55 {
56 ASSERT(hasJITReturnAddress());
57 return ReturnAddressPtr(m_pointer);
58 }
59#endif
60
61 bool isSet() const { return m_mode != None; }
62 bool operator!() const { return !isSet(); }
63
64private:
65#if ENABLE(JIT)
66 const void* m_pointer { nullptr };
67#endif
68
69 enum Mode { None, JIT, Interpreter };
70 Mode m_mode;
71};
72
73} // namespace JSC
