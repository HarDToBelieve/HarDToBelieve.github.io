35using ReferrerToken = AbstractSlotVisitor::ReferrerToken;

37inline ReferrerToken::ReferrerToken(HeapCell* cell)
38 : m_bits(bitwise_cast<uintptr_t>(cell) | HeapCellToken)
39{
40}
41
42inline ReferrerToken::ReferrerToken(OpaqueRootTag, void* opaqueRoot)
43 : m_bits(bitwise_cast<uintptr_t>(opaqueRoot) | OpaqueRootToken)
44{
45 ASSERT(opaqueRoot);
46}
47
48inline ReferrerToken::ReferrerToken(RootMarkReason reason)
49 : m_bits((static_cast<uintptr_t>(reason) << tokenTypeShift) | RootMarkReasonToken)
50{
51}
52
53inline HeapCell* ReferrerToken::asCell() const
54{
55 return isHeapCell() ? bitwise_cast<HeapCell*>(m_bits & ~tokenTypeMask) : nullptr;
56}
57
58inline void* ReferrerToken::asOpaqueRoot() const
59{
60 return isOpaqueRoot() ? bitwise_cast<HeapCell*>(m_bits & ~tokenTypeMask) : nullptr;
61}
62
63inline RootMarkReason ReferrerToken::asRootMarkReason() const
64{
65 return isRootMarkReason() ? static_cast<RootMarkReason>(m_bits >> tokenTypeShift) : RootMarkReason::None;
66}

68inline AbstractSlotVisitor::ReferrerContext::ReferrerContext(AbstractSlotVisitor& visitor, ReferrerToken referrer)
69 : m_visitor(visitor)
70 , m_referrer(referrer)
71{
72 m_previous = m_visitor.m_context;
73 if (m_previous) {
74 // An OpaqueRoot contexts can only be on the leaf.
75 RELEASE_ASSERT(!m_previous->m_isOpaqueRootContext);
76 }
77 m_visitor.m_context = this;
78}
79
80inline AbstractSlotVisitor::ReferrerContext::ReferrerContext(AbstractSlotVisitor& visitor, AbstractSlotVisitor::OpaqueRootTag)
81 : m_visitor(visitor)
82 , m_isOpaqueRootContext(true)
83{
84 m_previous = m_visitor.m_context;
85 if (m_previous) {
86 // An OpaqueRoot contexts can only be on the leaf.
87 RELEASE_ASSERT(!m_previous->m_isOpaqueRootContext);
88 }
89 m_visitor.m_context = this;
90}
91
92inline AbstractSlotVisitor::ReferrerContext::~ReferrerContext()
93{
94 m_visitor.m_context = m_previous;
95}

119inline bool AbstractSlotVisitor::addOpaqueRoot(void* ptr)
120{
121 if (!ptr)
122 return false;
123 if (m_ignoreNewOpaqueRoots)
124 return false;
125 if (!m_opaqueRoots.add(ptr))
126 return false;
127 if (UNLIKELY(m_needsExtraOpaqueRootHandling))
128 didAddOpaqueRoot(ptr);
129 m_visitCount++;
130 return true;
131}
132
133inline bool AbstractSlotVisitor::containsOpaqueRoot(void* ptr) const
134{
135 bool found = m_opaqueRoots.contains(ptr);
136 if (UNLIKELY(found && m_needsExtraOpaqueRootHandling)) {
137 auto* nonConstThis = const_cast<AbstractSlotVisitor*>(this);
138 nonConstThis->didFindOpaqueRoot(ptr);
139 }
140 return found;
141}

143template<typename T>
144ALWAYS_INLINE void AbstractSlotVisitor::append(const Weak<T>& weak)
145{
146 appendUnbarriered(weak.get());
147}
148
149template<typename T, typename Traits>
150ALWAYS_INLINE void AbstractSlotVisitor::append(const WriteBarrierBase<T, Traits>& slot)
151{
152 appendUnbarriered(slot.get());
153}
154
155template<typename T, typename Traits>
156ALWAYS_INLINE void AbstractSlotVisitor::appendHidden(const WriteBarrierBase<T, Traits>& slot)
157{
158 appendHiddenUnbarriered(slot.get());
159}
160
161ALWAYS_INLINE void AbstractSlotVisitor::appendHiddenUnbarriered(JSValue value)
162{
163 if (value.isCell())
164 appendHiddenUnbarriered(value.asCell());
165}
166
167template<typename Iterator>
168ALWAYS_INLINE void AbstractSlotVisitor::append(Iterator begin, Iterator end)
169{
170 for (auto it = begin; it != end; ++it)
171 append(*it);
172}
173
174ALWAYS_INLINE void AbstractSlotVisitor::appendValues(const WriteBarrierBase<Unknown>* barriers, size_t count)
175{
176 for (size_t i = 0; i < count; ++i)
177 append(barriers[i]);
178}
179
180ALWAYS_INLINE void AbstractSlotVisitor::appendValuesHidden(const WriteBarrierBase<Unknown>* barriers, size_t count)
181{
182 for (size_t i = 0; i < count; ++i)
183 appendHidden(barriers[i]);
184}
185
186ALWAYS_INLINE void AbstractSlotVisitor::appendUnbarriered(JSValue value)
187{
188 if (value.isCell())
189 appendUnbarriered(value.asCell());
190}
191
192ALWAYS_INLINE void AbstractSlotVisitor::appendUnbarriered(JSValue* slot, size_t count)
193{
194 for (size_t i = count; i--;)
195 appendUnbarriered(slot[i]);
196}

198ALWAYS_INLINE ReferrerToken AbstractSlotVisitor::referrer() const
199{
200 if (!m_context)
201 return nullptr;
202 return m_context->referrer();
203}
204
205ALWAYS_INLINE void AbstractSlotVisitor::reset()
206{
207 m_visitCount = 0;
