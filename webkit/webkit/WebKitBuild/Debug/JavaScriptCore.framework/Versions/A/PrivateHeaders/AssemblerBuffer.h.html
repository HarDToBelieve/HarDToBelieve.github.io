55 struct AssemblerLabel {
56 inline AssemblerLabel() { setOffset(std::numeric_limits<uint32_t>::max()); }
57 inline AssemblerLabel(const AssemblerLabel& other) { setOffset(other.offset()); }
58 inline AssemblerLabel(AssemblerLabel&& other) { setOffset(other.offset()); }
59 inline explicit AssemblerLabel(uint32_t offset) { setOffset(offset); }
60
61 AssemblerLabel& operator=(const AssemblerLabel& other) { setOffset(other.offset()); return *this; }
62 AssemblerLabel& operator=(AssemblerLabel&& other) { setOffset(other.offset()); return *this; }
63
64 bool isSet() const { return (offset() != std::numeric_limits<uint32_t>::max()); }
65
66 inline AssemblerLabel labelAtOffset(int offset) const
67 {
68 return AssemblerLabel(this->offset() + offset);
69 }
70
71 bool operator==(const AssemblerLabel& other) const { return offset() == other.offset(); }
72
73 inline uint32_t offset() const
74 {
75#if CPU(ARM64E)
76 return static_cast<uint32_t>(untagInt(m_offset, bitwise_cast<PtrTag>(this)));
77#else
78 return m_offset;
79#endif
80 }
81
82 private:
83 inline void setOffset(uint32_t offset)
84 {
85#if CPU(ARM64E)
86 m_offset = tagInt(static_cast<uint64_t>(offset), bitwise_cast<PtrTag>(this));
87#else
88 m_offset = offset;
89#endif
90 }
91
92#if CPU(ARM64E)
93 uint64_t m_offset;
94#else
95 uint32_t m_offset;
96#endif
97 };

99 class AssemblerData {
100 WTF_MAKE_NONCOPYABLE(AssemblerData);
101 static constexpr size_t InlineCapacity = 128;
102 public:
103 AssemblerData()
104 : m_buffer(m_inlineBuffer)
105 , m_capacity(InlineCapacity)
106 {
107 }
108
109 AssemblerData(size_t initialCapacity)
110 {
111 if (initialCapacity <= InlineCapacity) {
112 m_capacity = InlineCapacity;
113 m_buffer = m_inlineBuffer;
114 } else {
115 m_capacity = initialCapacity;
116 m_buffer = static_cast<char*>(AssemblerDataMalloc::malloc(m_capacity));
117 }
118 }
119
120 AssemblerData(AssemblerData&& other)
121 {
122 if (other.isInlineBuffer()) {
123 ASSERT(other.m_capacity == InlineCapacity);
124 memcpy(m_inlineBuffer, other.m_inlineBuffer, InlineCapacity);
125 m_buffer = m_inlineBuffer;
126 } else
127 m_buffer = other.m_buffer;
128 m_capacity = other.m_capacity;
129
130 other.m_buffer = other.m_inlineBuffer;
131 other.m_capacity = InlineCapacity;
132 }
133
134 AssemblerData& operator=(AssemblerData&& other)
135 {
136 if (m_buffer && !isInlineBuffer())
137 AssemblerDataMalloc::free(m_buffer);
138
139 if (other.isInlineBuffer()) {
140 ASSERT(other.m_capacity == InlineCapacity);
141 memcpy(m_inlineBuffer, other.m_inlineBuffer, InlineCapacity);
142 m_buffer = m_inlineBuffer;
143 } else
144 m_buffer = other.m_buffer;
145 m_capacity = other.m_capacity;
146
147 other.m_buffer = other.m_inlineBuffer;
148 other.m_capacity = InlineCapacity;
149 return *this;
150 }
151
152 void takeBufferIfLarger(AssemblerData&& other)
153 {
154 if (other.isInlineBuffer())
155 return;
156
157 if (m_capacity >= other.m_capacity)
158 return;
159
160 if (m_buffer && !isInlineBuffer())
161 AssemblerDataMalloc::free(m_buffer);
162
163 m_buffer = other.m_buffer;
164 m_capacity = other.m_capacity;
165
166 other.m_buffer = other.m_inlineBuffer;
167 other.m_capacity = InlineCapacity;
168 }
169
170 ~AssemblerData()
171 {
172 clear();
173 }
174
175 void clear()
176 {
177 if (m_buffer && !isInlineBuffer()) {
178 AssemblerDataMalloc::free(m_buffer);
179 m_capacity = InlineCapacity;
180 m_buffer = m_inlineBuffer;
181 }
182 }
183
184 char* buffer() const { return m_buffer; }
185
186 unsigned capacity() const { return m_capacity; }
187
188 void grow(unsigned extraCapacity = 0)
189 {
190 m_capacity = m_capacity + m_capacity / 2 + extraCapacity;
191 if (isInlineBuffer()) {
192 m_buffer = static_cast<char*>(AssemblerDataMalloc::malloc(m_capacity));
193 memcpy(m_buffer, m_inlineBuffer, InlineCapacity);
194 } else
195 m_buffer = static_cast<char*>(AssemblerDataMalloc::realloc(m_buffer, m_capacity));
196 }
197
198 private:
199 bool isInlineBuffer() const { return m_buffer == m_inlineBuffer; }
200 char* m_buffer;
201 char m_inlineBuffer[InlineCapacity];
202 unsigned m_capacity;
203 };

205#if CPU(ARM64E)
206 class ARM64EHash {
207 public:
208 ARM64EHash(uint32_t initialHash)
209 : m_hash(initialHash)
210 {
211 }
212
213 ALWAYS_INLINE uint32_t update(uint32_t value)
214 {
215 uint64_t input = value ^ m_hash;
216 uint64_t a = static_cast<uint32_t>(tagInt(input, static_cast<PtrTag>(0)) >> 39);
217 uint64_t b = tagInt(input, static_cast<PtrTag>(0xb7e151628aed2a6a)) >> 23;
218 m_hash = a ^ b;
219 return m_hash;
220 }
221
222 private:
223 uint32_t m_hash;
224 };
225#endif

227 class AssemblerBuffer {
228 public:
229 AssemblerBuffer()
230 : m_storage()
231 , m_index(0)
232#if CPU(ARM64E)
233 , m_hash(static_cast<uint32_t>(bitwise_cast<uint64_t>(this)))
234 , m_hashes()
235#endif
236 {
237 auto& threadSpecificData = threadSpecificAssemblerData();
238 m_storage.takeBufferIfLarger(WTFMove(*threadSpecificData));
239#if CPU(ARM64E)
240 auto& threadSpecificHashes = threadSpecificAssemblerHashes();
241 m_hashes.takeBufferIfLarger(WTFMove(*threadSpecificHashes));
242 ASSERT(m_storage.capacity() == m_hashes.capacity());
243#endif
244 }
245
246 ~AssemblerBuffer()
247 {
248#if CPU(ARM64E)
249 ASSERT(m_storage.capacity() == m_hashes.capacity());
250 auto& threadSpecificHashes = threadSpecificAssemblerHashes();
251 threadSpecificHashes->takeBufferIfLarger(WTFMove(m_hashes));
252#endif
253 auto& threadSpecificData = threadSpecificAssemblerData();
254 threadSpecificData->takeBufferIfLarger(WTFMove(m_storage));
255 }

257 bool isAvailable(unsigned space)
258 {
259 return m_index + space <= m_storage.capacity();
260 }
261
262 void ensureSpace(unsigned space)
263 {
264 while (!isAvailable(space))
265 outOfLineGrow();
266 }
267
268 bool isAligned(int alignment) const
269 {
270 return !(m_index & (alignment - 1));
271 }

319 // LocalWriter is a trick to keep the storage buffer and the index
320 // in memory while issuing multiple Stores.
321 // It is created in a block scope and its attribute can stay live
322 // between writes.
323 //
324 // LocalWriter *CANNOT* be mixed with other types of access to AssemblerBuffer.
325 // AssemblerBuffer cannot be used until its LocalWriter goes out of scope.
326#if !CPU(ARM64) // If we ever need to use this on arm64e, we would need to make the checksum aware of this.
327 class LocalWriter {
328 public:
329 LocalWriter(AssemblerBuffer& buffer, unsigned requiredSpace)
330 : m_buffer(buffer)
331 {
332 buffer.ensureSpace(requiredSpace);
333 m_storageBuffer = buffer.m_storage.buffer();
334 m_index = buffer.m_index;
335#if ASSERT_ENABLED
336 m_initialIndex = m_index;
337 m_requiredSpace = requiredSpace;
338#endif
339 }
340
341 ~LocalWriter()
342 {
343 ASSERT(m_index - m_initialIndex <= m_requiredSpace);
344 ASSERT(m_buffer.m_index == m_initialIndex);
345 ASSERT(m_storageBuffer == m_buffer.m_storage.buffer());
346 m_buffer.m_index = m_index;
347 }
348
349 void putByteUnchecked(int8_t value) { putIntegralUnchecked(value); }
350 void putShortUnchecked(int16_t value) { putIntegralUnchecked(value); }
351 void putIntUnchecked(int32_t value) { putIntegralUnchecked(value); }
352 void putInt64Unchecked(int64_t value) { putIntegralUnchecked(value); }
353 private:
354 template<typename IntegralType>
355 void putIntegralUnchecked(IntegralType value)
356 {
357 ASSERT(m_index + sizeof(IntegralType) <= m_buffer.m_storage.capacity());
358 WTF::unalignedStore<IntegralType>(m_storageBuffer + m_index, value);
359 m_index += sizeof(IntegralType);
360 }
361 AssemblerBuffer& m_buffer;
362 char* m_storageBuffer;
363 unsigned m_index;
364#if ASSERT_ENABLED
365 unsigned m_initialIndex;
366 unsigned m_requiredSpace;
367#endif
368 };
369#endif // !CPU(ARM64)
370
371#if !CPU(ARM64) // If we were to define this on arm64e, we'd need a way to update the hash as we write directly into the buffer.
372 void* data() const { return m_storage.buffer(); }
373#endif
374
375 protected:
376 template<typename IntegralType>
377 void putIntegral(IntegralType value)
378 {
379 unsigned nextIndex = m_index + sizeof(IntegralType);
380 if (UNLIKELY(nextIndex > m_storage.capacity()))
381 outOfLineGrow();
382 putIntegralUnchecked<IntegralType>(value);
383 }
384
385 template<typename IntegralType>
386 void putIntegralUnchecked(IntegralType value)
387 {
388#if CPU(ARM64)
389 static_assert(sizeof(value) == 4, "");
390#if CPU(ARM64E)
391 uint32_t hash = m_hash.update(value);
392 WTF::unalignedStore<uint32_t>(m_hashes.buffer() + m_index, hash);
393#endif
394#endif
395 ASSERT(isAvailable(sizeof(IntegralType)));
396 WTF::unalignedStore<IntegralType>(m_storage.buffer() + m_index, value);
397 m_index += sizeof(IntegralType);
398 }
399
400 private:
401 void grow(int extraCapacity = 0)
402 {
403 m_storage.grow(extraCapacity);
404#if CPU(ARM64E)
405 m_hashes.grow(extraCapacity);
406#endif
407 }
408
409 NEVER_INLINE void outOfLineGrow()
410 {
411 m_storage.grow();
412#if CPU(ARM64E)
413 m_hashes.grow();
414#endif
415 }

419#endif

424#if CPU(ARM64E)

426 AssemblerData m_hashes;
427#endif
428 };
