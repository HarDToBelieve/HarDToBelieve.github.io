29#include "JSWrapperObject.h"

31namespace JSC {
32
33class BigIntObject final : public JSWrapperObject {
34public:
35 using Base = JSWrapperObject;
36
37 template<typename, SubspaceAccess mode>
38 static IsoSubspace* subspaceFor(VM& vm)
39 {
40 return vm.bigIntObjectSpace<mode>();
41 }
42
43 JS_EXPORT_PRIVATE static BigIntObject* create(VM&, JSGlobalObject*, JSValue);
44
45 DECLARE_EXPORT_INFO;
46
47 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
48 {
49 return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
50 }
51
52private:
53 JS_EXPORT_PRIVATE void finishCreation(VM&, JSValue);
54 JS_EXPORT_PRIVATE BigIntObject(VM&, Structure*);
55};
56static_assert(sizeof(BigIntObject) == sizeof(JSWrapperObject));

58} // namespace JSC
