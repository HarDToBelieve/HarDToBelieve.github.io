21#pragma once
22
23#include "JSWrapperObject.h"
24
25namespace JSC {
26
27class BooleanObject : public JSWrapperObject {
28protected:
29 JS_EXPORT_PRIVATE BooleanObject(VM&, Structure*);
30 JS_EXPORT_PRIVATE void finishCreation(VM&);
31
32public:
33 using Base = JSWrapperObject;
34
35 template<typename, SubspaceAccess mode>
36 static IsoSubspace* subspaceFor(VM& vm)
37 {
38 return vm.booleanObjectSpace<mode>();
39 }
40
41 static BooleanObject* create(VM& vm, Structure* structure)
42 {
43 BooleanObject* boolean = new (NotNull, allocateCell<BooleanObject>(vm.heap)) BooleanObject(vm, structure);
44 boolean->finishCreation(vm);
45 return boolean;
46 }
47
48 DECLARE_EXPORT_INFO;
49
50 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
51 {
52 return Structure::create(vm, globalObject, prototype, TypeInfo(BooleanObjectType, StructureFlags), info());
53 }
54};
55static_assert(sizeof(BooleanObject) == sizeof(JSWrapperObject));
56
57} // namespace JSC
