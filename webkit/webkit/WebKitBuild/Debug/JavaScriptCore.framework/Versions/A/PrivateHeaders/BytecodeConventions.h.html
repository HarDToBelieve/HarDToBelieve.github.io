28// Register numbers used in bytecode operations have different meaning according to their ranges:
29// 0x80000000-0xFFFFFFFF Negative indices from the CallFrame pointer are entries in the call frame.
30// 0x00000000-0x3FFFFFFF Forwards indices from the CallFrame pointer are local vars and temporaries with the function's callframe.
31// 0x40000000-0x7FFFFFFF Positive indices from 0x40000000 specify entries in the constant pool on the CodeBlock.
32static constexpr int FirstConstantRegisterIndex = 0x40000000;
33
34static constexpr int FirstConstantRegisterIndex8 = 16;
35static constexpr int FirstConstantRegisterIndex16 = 64;
36static constexpr int FirstConstantRegisterIndex32 = FirstConstantRegisterIndex;
