72 // The most general helper for setting arguments that fit in a GPR, if you can compute each
73 // argument without using any argument registers. You usually want one of the setupArguments*()
74 // methods below instead of this. This thing is most useful if you have *a lot* of arguments.
75 template<typename Functor>
76 void setupArgument(unsigned argumentIndex, const Functor& functor)
77 {
78 unsigned numberOfRegs = GPRInfo::numberOfArgumentRegisters; // Disguise the constant from clang's tautological compare warning.
79 if (argumentIndex < numberOfRegs) {
80 functor(GPRInfo::toArgumentRegister(argumentIndex));
81 return;
82 }
83
84 functor(GPRInfo::nonArgGPR0);
85 poke(GPRInfo::nonArgGPR0, POKE_ARGUMENT_OFFSET + argumentIndex - GPRInfo::numberOfArgumentRegisters);
86 }

90 template<unsigned NumberOfRegisters, typename RegType>
91 ALWAYS_INLINE void setupStubArgs(std::array<RegType, NumberOfRegisters> destinations, std::array<RegType, NumberOfRegisters> sources)
92 {
93 if (ASSERT_ENABLED) {
94 RegisterSet set;
95 for (RegType dest : destinations)
96 set.set(dest);
97 ASSERT_WITH_MESSAGE(set.numberOfSetRegisters() == NumberOfRegisters, "Destinations should not be aliased.");
98 }
99
100 typedef std::pair<RegType, RegType> RegPair;
101 Vector<RegPair, NumberOfRegisters> pairs;
102
103 // if constexpr avoids warnings when NumberOfRegisters is 0.
104 if constexpr (NumberOfRegisters > 0) {
105 for (unsigned i = 0; i < NumberOfRegisters; ++i) {
106 if (sources[i] != destinations[i])
107 pairs.append(std::make_pair(sources[i], destinations[i]));
108 }
109 } else {
110 // Silence some older compilers (GCC up to 9.X) about unused but set parameters.
111 UNUSED_PARAM(sources);
112 UNUSED_PARAM(destinations);
113 }
114
115#if ASSERT_ENABLED
116 auto numUniqueSources = [&] () -> unsigned {
117 RegisterSet set;
118 for (auto& pair : pairs) {
119 RegType source = pair.first;
120 set.set(source);
121 }
122 return set.numberOfSetRegisters();
123 };
124
125 auto numUniqueDests = [&] () -> unsigned {
126 RegisterSet set;
127 for (auto& pair : pairs) {
128 RegType dest = pair.second;
129 set.set(dest);
130 }
131 return set.numberOfSetRegisters();
132 };
133#endif
134
135 while (pairs.size()) {
136 RegisterSet freeDestinations;
137 for (auto& pair : pairs) {
138 RegType dest = pair.second;
139 freeDestinations.set(dest);
140 }
141 for (auto& pair : pairs) {
142 RegType source = pair.first;
143 freeDestinations.clear(source);
144 }
145
146 if (freeDestinations.numberOfSetRegisters()) {
147 bool madeMove = false;
148 for (unsigned i = 0; i < pairs.size(); i++) {
149 auto& pair = pairs[i];
150 RegType source = pair.first;
151 RegType dest = pair.second;
152 if (freeDestinations.get(dest)) {
153 move(source, dest);
154 pairs.remove(i);
155 madeMove = true;
156 break;
157 }
158 }
159 ASSERT_UNUSED(madeMove, madeMove);
160 continue;
161 }
162
163 ASSERT(numUniqueDests() == numUniqueSources());
164 ASSERT(numUniqueDests() == pairs.size());
165 // The set of source and destination registers are equivalent sets. This means we don't have
166 // any free destination registers that won't also clobber a source. We get around this by
167 // exchanging registers.
168
169 RegType source = pairs[0].first;
170 RegType dest = pairs[0].second;
171 swap(source, dest);
172 pairs.remove(0);
173
174 RegType newSource = source;
175 for (auto& pair : pairs) {
176 RegType source = pair.first;
177 if (source == dest) {
178 pair.first = newSource;
179 break;
180 }
181 }
182
183 // We may have introduced pairs that have the same source and destination. Remove those now.
184 for (unsigned i = 0; i < pairs.size(); i++) {
185 auto& pair = pairs[i];
186 if (pair.first == pair.second) {
187 pairs.remove(i);
188 i--;
189 }
190 }
191 }
192 }

325 template<unsigned TargetSize, typename RegType>
326 std::array<RegType, TargetSize> clampArrayToSize(std::array<RegType, InfoTypeForReg<RegType>::numberOfRegisters> sourceArray)
327 {
328 static_assert(TargetSize <= sourceArray.size(), "TargetSize is bigger than source.size()");
329 RELEASE_ASSERT(TargetSize <= InfoTypeForReg<RegType>::numberOfRegisters);
330
331 std::array<RegType, TargetSize> result { };
332
333 // if constexpr avoids warnings when TargetSize is 0.
334 if constexpr (TargetSize > 0) {
335 for (unsigned i = 0; i < TargetSize; i++) {
336 ASSERT(sourceArray[i] != static_cast<int32_t>(InfoTypeForReg<RegType>::InvalidIndex));
337 result[i] = sourceArray[i];
338 }
339 }
340
341 return result;

344 ALWAYS_INLINE unsigned calculatePokeOffset(unsigned currentGPRArgument, unsigned currentFPRArgument, unsigned numCrossSources, unsigned extraGPRArgs, unsigned nonArgGPRs, unsigned extraPoke)
345 {
346 // Clang claims that it cannot find the symbol for FPRReg/GPRReg::numberOfArgumentRegisters when they are passed directly to std::max... seems like a bug
347 unsigned numberOfFPArgumentRegisters = FPRInfo::numberOfArgumentRegisters;
348 unsigned numberOfGPArgumentRegisters = GPRInfo::numberOfArgumentRegisters;
349
350 UNUSED_PARAM(nonArgGPRs);
351
352 currentGPRArgument += extraGPRArgs;
353 currentFPRArgument -= numCrossSources;
354 IGNORE_WARNINGS_BEGIN("type-limits")
355 ASSERT(currentGPRArgument >= GPRInfo::numberOfArgumentRegisters || currentFPRArgument >= FPRInfo::numberOfArgumentRegisters);
356 IGNORE_WARNINGS_END
357
358 unsigned pokeOffset = POKE_ARGUMENT_OFFSET + extraPoke;
359 pokeOffset += std::max(currentGPRArgument, numberOfGPArgumentRegisters) - numberOfGPArgumentRegisters;
360 pokeOffset += std::max(currentFPRArgument, numberOfFPArgumentRegisters) - numberOfFPArgumentRegisters;
361 return pokeOffset;
362 }

371 ALWAYS_INLINE bool stackAligned(unsigned currentGPRArgument, unsigned currentFPRArgument, unsigned numCrossSources, unsigned extraGPRArgs, unsigned nonArgGPRs, unsigned extraPoke)

377 // In the auto-calling convention code below the order of operations is:
378 // 1) spill arguments to stack slots
379 // 2) shuffle incomming argument values in registers to argument registers
380 // 3) fill immediate values to argument registers
381 // To do this, we recurse forwards through our args collecting argument values in registers and spilling stack slots.
382 // when we run out of args we then run our shuffling code to relocate registers. Finally, as we unwind from our
383 // recursion we can fill immediates.
384
385#define CURRENT_ARGUMENT_TYPE typename FunctionTraits<OperationType>::template ArgumentType<numGPRArgs + numFPRArgs>
386#define RESULT_TYPE typename FunctionTraits<OperationType>::ResultType
387
388#if USE(JSVALUE64)
389
390 // Avoid MSVC optimization time explosion associated with __forceinline in recursive templates.
391 template<typename OperationType, unsigned numGPRArgs, unsigned numGPRSources, unsigned numFPRArgs, unsigned numFPRSources, unsigned numCrossSources, unsigned extraGPRArgs, unsigned nonArgGPRs, unsigned extraPoke, typename RegType, typename... Args>
392 ALWAYS_INLINE_EXCEPT_MSVC void marshallArgumentRegister(ArgCollection<numGPRArgs, numGPRSources, numFPRArgs, numFPRSources, numCrossSources, extraGPRArgs, nonArgGPRs, extraPoke> argSourceRegs, RegType arg, Args... args)
393 {
394 using InfoType = InfoTypeForReg<RegType>;
395 unsigned numArgRegisters = InfoType::numberOfArgumentRegisters;
396#if OS(WINDOWS) && CPU(X86_64)
397 unsigned currentArgCount = argSourceRegs.argCount(arg) + (std::is_same<RESULT_TYPE, SlowPathReturnType>::value ? 1 : 0);
398#else
399 unsigned currentArgCount = argSourceRegs.argCount(arg);
400#endif
401 if (currentArgCount < numArgRegisters) {
402 auto updatedArgSourceRegs = argSourceRegs.pushRegArg(arg, InfoType::toArgumentRegister(currentArgCount));
403 setupArgumentsImpl<OperationType>(updatedArgSourceRegs, args...);
404 return;
405 }
406
407 pokeForArgument(arg, numGPRArgs, numFPRArgs, numCrossSources, extraGPRArgs, nonArgGPRs, extraPoke);
408 setupArgumentsImpl<OperationType>(argSourceRegs.addStackArg(arg), args...);
409 }

439 template<typename OperationType, unsigned numGPRArgs, unsigned numGPRSources, unsigned numFPRArgs, unsigned numFPRSources, unsigned numCrossSources, unsigned extraGPRArgs, unsigned nonArgGPRs, unsigned extraPoke, typename... Args>
440 void setupArgumentsImpl(ArgCollection<numGPRArgs, numGPRSources, numFPRArgs, numFPRSources, numCrossSources, extraGPRArgs, nonArgGPRs, extraPoke> argSourceRegs, FPRReg arg, Args... args)
441 {
442 static_assert(std::is_same<CURRENT_ARGUMENT_TYPE, double>::value, "We should only be passing FPRRegs to a double");
443
444 // MIPS and ARM-hardfp pass FP arguments in FP registers.
445#if CPU(MIPS)
446 unsigned numberOfFPArgumentRegisters = FPRInfo::numberOfArgumentRegisters;
447 unsigned currentFPArgCount = argSourceRegs.argCount(arg);
448
449 // MIPS can only use FP argument registers if it isn't preceeded by any GP argument.
450 if (currentFPArgCount < numberOfFPArgumentRegisters && !numGPRArgs) {
451 auto updatedArgSourceRegs = argSourceRegs.pushRegArg(arg, FPRInfo::toArgumentRegister(currentFPArgCount));
452 setupArgumentsImpl<OperationType>(updatedArgSourceRegs.addGPRExtraArg().addGPRExtraArg(), args...);
453 return;
454 }
455#elif CPU(ARM_THUMB2) && CPU(ARM_HARDFP)
456 unsigned numberOfFPArgumentRegisters = FPRInfo::numberOfArgumentRegisters;
457 unsigned currentFPArgCount = argSourceRegs.argCount(arg);
458
459 if (currentFPArgCount < numberOfFPArgumentRegisters) {
460 auto updatedArgSourceRegs = argSourceRegs.pushRegArg(arg, FPRInfo::toArgumentRegister(currentFPArgCount));
461 setupArgumentsImpl<OperationType>(updatedArgSourceRegs, args...);
462 return;
463 }
464#endif
465
466#if CPU(MIPS) || (CPU(ARM_THUMB2) && !CPU(ARM_HARDFP))
467 // On MIPS and ARM-softfp FP arguments can be passed in GP registers.
468 unsigned numberOfGPArgumentRegisters = GPRInfo::numberOfArgumentRegisters;
469 unsigned currentGPArgCount = argSourceRegs.argCount(GPRInfo::regT0);
470 unsigned alignedGPArgCount = roundUpToMultipleOf<2>(currentGPArgCount);
471
472 if (alignedGPArgCount + 1 < numberOfGPArgumentRegisters) {
473 auto updatedArgSourceRegs = argSourceRegs.pushRegArg(arg, GPRInfo::toArgumentRegister(alignedGPArgCount));
474
475 if (alignedGPArgCount > currentGPArgCount)
476 setupArgumentsImpl<OperationType>(updatedArgSourceRegs.addGPRExtraArg().addGPRExtraArg().addGPRExtraArg(), args...);
477 else
478 setupArgumentsImpl<OperationType>(updatedArgSourceRegs.addGPRExtraArg().addGPRExtraArg(), args...);
479
480 return;
481 }
482
483 if (currentGPArgCount < numberOfGPArgumentRegisters) {
484 pokeForArgument(arg, numGPRArgs, numFPRArgs, numCrossSources, extraGPRArgs + 1, nonArgGPRs, extraPoke);
485 setupArgumentsImpl<OperationType>(argSourceRegs.addGPRExtraArg().addStackArg(arg).addPoke(), args...);
486 return;
487 }
488#endif
489
490 // Otherwise pass FP argument on stack.
491 if (stackAligned(numGPRArgs, numFPRArgs, numCrossSources, extraGPRArgs, nonArgGPRs, extraPoke)) {
492 pokeForArgument(arg, numGPRArgs, numFPRArgs, numCrossSources, extraGPRArgs, nonArgGPRs, extraPoke);
493 setupArgumentsImpl<OperationType>(argSourceRegs.addStackArg(arg).addPoke(), args...);
494 } else {
495 pokeForArgument(arg, numGPRArgs, numFPRArgs, numCrossSources, extraGPRArgs, nonArgGPRs, extraPoke + 1);
496 setupArgumentsImpl<OperationType>(argSourceRegs.addStackArg(arg).addPoke().addPoke(), args...);
497 }
498 }

516 template<typename OperationType, typename Arg1, typename Arg2, unsigned numGPRArgs, unsigned numGPRSources, unsigned numFPRArgs, unsigned numFPRSources, unsigned numCrossSources, unsigned extraGPRArgs, unsigned nonArgGPRs, unsigned extraPoke, typename... Args>
517 void pokeArgumentsAligned(ArgCollection<numGPRArgs, numGPRSources, numFPRArgs, numFPRSources, numCrossSources, extraGPRArgs, nonArgGPRs, extraPoke> argSourceRegs, Arg1 arg1, Arg2 arg2, Args... args)
518 {
519 unsigned numArgRegisters = GPRInfo::numberOfArgumentRegisters;
520 unsigned currentArgCount = argSourceRegs.argCount(GPRInfo::regT0);
521

747 void prepareForTailCallSlow(GPRReg calleeGPR = InvalidGPRReg)
748 {
749 GPRReg temp1 = calleeGPR == GPRInfo::regT0 ? GPRInfo::regT3 : GPRInfo::regT0;
750 GPRReg temp2 = calleeGPR == GPRInfo::regT1 ? GPRInfo::regT3 : GPRInfo::regT1;
751 GPRReg temp3 = calleeGPR == GPRInfo::regT2 ? GPRInfo::regT3 : GPRInfo::regT2;
752
753 GPRReg newFramePointer = temp1;
754 GPRReg newFrameSizeGPR = temp2;
755 {
756 // The old frame size is its number of arguments (or number of
757 // parameters in case of arity fixup), plus the frame header size,
758 // aligned
759 GPRReg oldFrameSizeGPR = temp2;
760 {
761 GPRReg argCountGPR = oldFrameSizeGPR;
762 load32(Address(framePointerRegister, CallFrameSlot::argumentCountIncludingThis * static_cast<int>(sizeof(Register)) + PayloadOffset), argCountGPR);
763
764 {
765 GPRReg numParametersGPR = temp1;
766 {
767 GPRReg codeBlockGPR = numParametersGPR;
768 loadPtr(Address(framePointerRegister, CallFrameSlot::codeBlock * static_cast<int>(sizeof(Register))), codeBlockGPR);
769 load32(Address(codeBlockGPR, CodeBlock::offsetOfNumParameters()), numParametersGPR);
770 }
771
772 ASSERT(numParametersGPR != argCountGPR);
773 Jump argumentCountWasNotFixedUp = branch32(BelowOrEqual, numParametersGPR, argCountGPR);
774 move(numParametersGPR, argCountGPR);
775 argumentCountWasNotFixedUp.link(this);
776 }
777
778 add32(TrustedImm32(stackAlignmentRegisters() + CallFrame::headerSizeInRegisters - 1), argCountGPR, oldFrameSizeGPR);
779 and32(TrustedImm32(-stackAlignmentRegisters()), oldFrameSizeGPR);
780 // We assume < 2^28 arguments
781 mul32(TrustedImm32(sizeof(Register)), oldFrameSizeGPR, oldFrameSizeGPR);
782 }
783
784 // The new frame pointer is at framePointer + oldFrameSize - newFrameSize
785 ASSERT(newFramePointer != oldFrameSizeGPR);
786 addPtr(framePointerRegister, oldFrameSizeGPR, newFramePointer);
787
788 // The new frame size is just the number of arguments plus the
789 // frame header size, aligned
790 ASSERT(newFrameSizeGPR != newFramePointer);
791 load32(Address(stackPointerRegister, CallFrameSlot::argumentCountIncludingThis * static_cast<int>(sizeof(Register)) + PayloadOffset - sizeof(CallerFrameAndPC)),
792 newFrameSizeGPR);
793 add32(TrustedImm32(stackAlignmentRegisters() + CallFrame::headerSizeInRegisters - 1), newFrameSizeGPR);
794 and32(TrustedImm32(-stackAlignmentRegisters()), newFrameSizeGPR);
795 // We assume < 2^28 arguments
796 mul32(TrustedImm32(sizeof(Register)), newFrameSizeGPR, newFrameSizeGPR);
797 }
798
799 GPRReg tempGPR = temp3;
800 ASSERT(tempGPR != newFramePointer && tempGPR != newFrameSizeGPR);
801
802 // We don't need the current frame beyond this point. Masquerade as our
803 // caller.
804#if CPU(ARM_THUMB2) || CPU(ARM64)
805 loadPtr(Address(framePointerRegister, CallFrame::returnPCOffset()), linkRegister);
806 subPtr(TrustedImm32(2 * sizeof(void*)), newFrameSizeGPR);
807#if CPU(ARM64E)
808 addPtr(TrustedImm32(sizeof(CallerFrameAndPC)), MacroAssembler::framePointerRegister, tempGPR);
809 untagPtr(tempGPR, linkRegister);
810 validateUntaggedPtr(linkRegister, tempGPR);
811#endif
812#elif CPU(MIPS)
813 loadPtr(Address(framePointerRegister, sizeof(void*)), returnAddressRegister);
814 subPtr(TrustedImm32(2 * sizeof(void*)), newFrameSizeGPR);
815#elif CPU(X86_64)
816 loadPtr(Address(framePointerRegister, sizeof(void*)), tempGPR);
817 push(tempGPR);
818 subPtr(TrustedImm32(sizeof(void*)), newFrameSizeGPR);
819#else
820 UNREACHABLE_FOR_PLATFORM();
821#endif
822 subPtr(newFrameSizeGPR, newFramePointer);
823 loadPtr(Address(framePointerRegister), framePointerRegister);
824
825
826 // We need to move the newFrameSizeGPR slots above the stack pointer by
827 // newFramePointer registers. We use pointer-sized chunks.
828 MacroAssembler::Label copyLoop(label());
829
830 subPtr(TrustedImm32(sizeof(void*)), newFrameSizeGPR);
831 loadPtr(BaseIndex(stackPointerRegister, newFrameSizeGPR, TimesOne), tempGPR);
832 storePtr(tempGPR, BaseIndex(newFramePointer, newFrameSizeGPR, TimesOne));
833
834 branchTest32(MacroAssembler::NonZero, newFrameSizeGPR).linkTo(copyLoop, this);
835
836 // Ready for a jump!
837 move(newFramePointer, stackPointerRegister);
838 }
