82public:

87 VM& vm() { return m_vm; }

89
90 ptrdiff_t offsetOf(const void*);
91 void cacheOffset(ptrdiff_t, void*);
92 std::optional<void*> cachedPtrForOffset(ptrdiff_t);
93 const void* ptrForOffsetFromBase(ptrdiff_t);

95 void setHandleForTDZEnvironment(CompactTDZEnvironment*, const CompactTDZEnvironmentMap::Handle&);
96 void addLeafExecutable(const UnlinkedFunctionExecutable*, ptrdiff_t);
97 RefPtr<SourceProvider> provider() const;
98
99 template<typename Functor>
100 void addFinalizer(const Functor&);
101
102private:
103 Decoder(VM&, Ref<CachedBytecode>, RefPtr<SourceProvider>);
104
105 VM& m_vm;
106 Ref<CachedBytecode> m_cachedBytecode;
107 HashMap<ptrdiff_t, void*> m_offsetToPtrMap;
108 Vector<std::function<void()>> m_finalizers;
109 HashMap<CompactTDZEnvironment*, CompactTDZEnvironmentMap::Handle> m_environmentToHandleMap;
110 RefPtr<SourceProvider> m_provider;
111};
