26#pragma once
27
28#include "CallVariant.h"
29
30namespace JSC {
31
32class CallEdge {
33public:
34 CallEdge();
35 CallEdge(CallVariant, uint32_t);
36
37 bool operator!() const { return !m_callee; }
38
39 CallVariant callee() const { return m_callee; }
40 uint32_t count() const { return m_count; }
41
42 CallEdge despecifiedClosure() const
43 {
44 return CallEdge(m_callee.despecifiedClosure(), m_count);
45 }
46
47 void dump(PrintStream&) const;
48
49private:
50 CallVariant m_callee;
51 uint32_t m_count;
52};
53
54inline CallEdge::CallEdge(CallVariant callee, uint32_t count)
55 : m_callee(callee)
56 , m_count(count)
57{
58}
59
60inline CallEdge::CallEdge()
61 : CallEdge(CallVariant(), 0)
62{
63}
64
65typedef Vector<CallEdge, 1> CallEdgeList;
66
67} // namespace JSC
