30namespace JSC {
31
32// The CellState of a cell is a kind of hint about what the state of the cell is.
33enum class CellState : uint8_t {
34 // The object is either currently being scanned, or it has finished being scanned, or this
35 // is a full collection and it's actually a white object (you'd know because its mark bit
36 // would be clear).
37 PossiblyBlack = 0,
38
39 // The object is in eden. During GC, this means that the object has not been marked yet.
40 DefinitelyWhite = 1,
41
42 // This sorta means that the object is grey - i.e. it will be scanned. Or it could be white
43 // during a full collection if its mark bit is clear. That would happen if it had been black,
44 // got barriered, and we did a full collection.
45 PossiblyGrey = 2
46};
47
48static constexpr unsigned blackThreshold = 0; // x <= blackThreshold means x is PossiblyOldOrBlack.
49static constexpr unsigned tautologicalThreshold = 100; // x <= tautologicalThreshold is always true.
50
51inline bool isWithinThreshold(CellState cellState, unsigned threshold)
52{
53 return static_cast<unsigned>(cellState) <= threshold;
54}
