124#define CREATE_MEMBER_CHECKER(member) \
125 template <typename T> \
126 struct MemberCheck##member { \
127 struct Fallback { \
128 void member(...); \
129 }; \
130 struct Derived : T, Fallback { }; \
131 template <typename U, U> struct Check; \
132 typedef char Yes[2]; \
133 typedef char No[1]; \
134 template <typename U> \
135 static No &func(Check<void (Fallback::*)(...), &U::member>*); \
136 template <typename U> \
137 static Yes &func(...); \
138 enum { has = sizeof(func<Derived>(0)) == sizeof(Yes) }; \
139 }
140
141#define HAS_MEMBER_NAMED(klass, name) (MemberCheck##name<klass>::has)

175struct ClassInfo {
176 using CheckJSCastSnippetFunctionPtr = Ref<Snippet> (*)(void);
177
178 // A string denoting the class name. Example: "Window".
179 const char* className;
180 // Pointer to the class information of the base class.
181 // nullptrif there is none.
182 const ClassInfo* parentClass;
183 const HashTable* staticPropHashTable;
184 CheckJSCastSnippetFunctionPtr checkSubClassSnippet;
185 const std::optional<JSTypeRange> inheritsJSTypeRange; // This is range of JSTypes for doing inheritance checking. Has the form: [firstJSType, lastJSType] (inclusive).
186 MethodTable methodTable;
187 const TypedArrayType typedArrayStorageType;
188 const unsigned staticClassSize;
189
190 static ptrdiff_t offsetOfParentClass()
191 {
192 return OBJECT_OFFSETOF(ClassInfo, parentClass);
193 }
194
195 bool isSubClassOf(const ClassInfo* other) const
196 {
197 for (const ClassInfo* ci = this; ci; ci = ci->parentClass) {
198 if (ci == other)
199 return true;
200 }
201 return false;
202 }
203
204 JS_EXPORT_PRIVATE void dump(PrintStream&) const;
205
206 JS_EXPORT_PRIVATE bool hasStaticSetterOrReadonlyProperties() const;
207};
