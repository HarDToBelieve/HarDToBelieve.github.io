47class CodeOrigin {
48public:
49 CodeOrigin()
50#if CPU(ADDRESS64)
51 : m_compositeValue(buildCompositeValue(nullptr, BytecodeIndex()))
52#else
53 : m_inlineCallFrame(nullptr)
54#endif
55 {
56 }
57
58 CodeOrigin(WTF::HashTableDeletedValueType)
59#if CPU(ADDRESS64)
60 : m_compositeValue(buildCompositeValue(deletedMarker(), BytecodeIndex()))
61#else
62 : m_bytecodeIndex(WTF::HashTableDeletedValue)
63 , m_inlineCallFrame(deletedMarker())
64#endif
65 {
66 }
67
68 explicit CodeOrigin(BytecodeIndex bytecodeIndex, InlineCallFrame* inlineCallFrame = nullptr)
69#if CPU(ADDRESS64)
70 : m_compositeValue(buildCompositeValue(inlineCallFrame, bytecodeIndex))
71#else
72 : m_bytecodeIndex(bytecodeIndex)
73 , m_inlineCallFrame(inlineCallFrame)
74#endif
75 {
76 ASSERT(!!bytecodeIndex);
77#if CPU(ADDRESS64)
78 ASSERT(!(bitwise_cast<uintptr_t>(inlineCallFrame) & ~s_maskCompositeValueForPointer));
79#endif
80 }
81
82#if CPU(ADDRESS64)
83 CodeOrigin& operator=(const CodeOrigin& other)
84 {
85 if (this != &other) {
86 if (UNLIKELY(isOutOfLine()))
87 delete outOfLineCodeOrigin();
88
89 if (UNLIKELY(other.isOutOfLine()))
90 m_compositeValue = buildCompositeValue(other.inlineCallFrame(), other.bytecodeIndex());
91 else
92 m_compositeValue = other.m_compositeValue;
93 }
94 return *this;
95 }
96 CodeOrigin& operator=(CodeOrigin&& other)
97 {
98 if (this != &other) {
99 if (UNLIKELY(isOutOfLine()))
100 delete outOfLineCodeOrigin();
101
102 m_compositeValue = std::exchange(other.m_compositeValue, 0);
103 }
104 return *this;
105 }
106
107 CodeOrigin(const CodeOrigin& other)
108 {
109 // We don't use the member initializer list because it would not let us optimize the common case where there is no out-of-line storage
110 // (in which case we don't have to extract the components of the composite value just to reassemble it).
111 if (UNLIKELY(other.isOutOfLine()))
112 m_compositeValue = buildCompositeValue(other.inlineCallFrame(), other.bytecodeIndex());
113 else
114 m_compositeValue = other.m_compositeValue;
115 }
116 CodeOrigin(CodeOrigin&& other)
117 : m_compositeValue(std::exchange(other.m_compositeValue, 0))
118 {
119 }
120
121 ~CodeOrigin()
122 {
123 if (UNLIKELY(isOutOfLine()))
124 delete outOfLineCodeOrigin();
125 }
126#endif
127
128 bool isSet() const
129 {
130#if CPU(ADDRESS64)
131 return !(m_compositeValue & s_maskIsBytecodeIndexInvalid);
132#else
133 return !!m_bytecodeIndex;
134#endif
135 }
136 explicit operator bool() const { return isSet(); }
137
138 bool isHashTableDeletedValue() const
139 {
140#if CPU(ADDRESS64)
141 return !isSet() && (m_compositeValue & s_maskCompositeValueForPointer);
142#else
143 return m_bytecodeIndex.isHashTableDeletedValue() && !!m_inlineCallFrame;
144#endif
145 }
146
147 // The inline depth is the depth of the inline stack, so 1 = not inlined,
148 // 2 = inlined one deep, etc.
149 unsigned inlineDepth() const;
150
151 // If the code origin corresponds to inlined code, gives you the heap object that
152 // would have owned the code if it had not been inlined. Otherwise returns 0.
153 CodeBlock* codeOriginOwner() const;
154
155 int stackOffset() const;
156
157 unsigned hash() const;
158 bool operator==(const CodeOrigin& other) const;
159 bool operator!=(const CodeOrigin& other) const { return !(*this == other); }
160
161 // This checks if the two code origins correspond to the same stack trace snippets,
162 // but ignore whether the InlineCallFrame's are identical.
163 bool isApproximatelyEqualTo(const CodeOrigin& other, InlineCallFrame* terminal = nullptr) const;
164
165 unsigned approximateHash(InlineCallFrame* terminal = nullptr) const;

167 template <typename Function>
168 void walkUpInlineStack(const Function&) const;
169
170 inline bool inlineStackContainsActiveCheckpoint() const;
171
172 // Get the inline stack. This is slow, and is intended for debugging only.
173 Vector<CodeOrigin> inlineStack() const;
174
175 JS_EXPORT_PRIVATE void dump(PrintStream&) const;
176 void dumpInContext(PrintStream&, DumpContext*) const;
177
178 BytecodeIndex bytecodeIndex() const
179 {
180#if CPU(ADDRESS64)
181 if (!isSet())
182 return BytecodeIndex();
183 if (UNLIKELY(isOutOfLine()))
184 return outOfLineCodeOrigin()->bytecodeIndex;
185 return BytecodeIndex::fromBits(m_compositeValue >> (64 - s_freeBitsAtTop));
186#else
187 return m_bytecodeIndex;
188#endif
189 }
190
191 InlineCallFrame* inlineCallFrame() const
192 {
193#if CPU(ADDRESS64)
194 if (UNLIKELY(isOutOfLine()))
195 return outOfLineCodeOrigin()->inlineCallFrame;
196 return bitwise_cast<InlineCallFrame*>(m_compositeValue & s_maskCompositeValueForPointer);
197#else
198 return m_inlineCallFrame;
199#endif
200 }
201
202private:
203#if CPU(ADDRESS64)
204 static constexpr uintptr_t s_maskIsOutOfLine = 1;
205 static constexpr uintptr_t s_maskIsBytecodeIndexInvalid = 2;
206
207 struct OutOfLineCodeOrigin {
208 WTF_MAKE_FAST_ALLOCATED;
209 public:
210 InlineCallFrame* inlineCallFrame;
211 BytecodeIndex bytecodeIndex;
212
213 OutOfLineCodeOrigin(InlineCallFrame* inlineCallFrame, BytecodeIndex bytecodeIndex)
214 : inlineCallFrame(inlineCallFrame)
215 , bytecodeIndex(bytecodeIndex)
216 {
217 }
218 };
219
220 bool isOutOfLine() const
221 {
222 return m_compositeValue & s_maskIsOutOfLine;
223 }
224 OutOfLineCodeOrigin* outOfLineCodeOrigin() const
225 {
226 ASSERT(isOutOfLine());
227 return bitwise_cast<OutOfLineCodeOrigin*>(m_compositeValue & s_maskCompositeValueForPointer);
228 }
229#endif
230
231 static InlineCallFrame* deletedMarker()
232 {
233 auto value = static_cast<uintptr_t>(1 << 3);
234#if CPU(ADDRESS64)
235 ASSERT(value & s_maskCompositeValueForPointer);
236 ASSERT(!(value & ~s_maskCompositeValueForPointer));
237#endif
238 return bitwise_cast<InlineCallFrame*>(value);
239 }
240
241#if CPU(ADDRESS64)
242 static constexpr unsigned s_freeBitsAtTop = 64 - OS_CONSTANT(EFFECTIVE_ADDRESS_WIDTH);
243 static constexpr uintptr_t s_maskCompositeValueForPointer = ((1ULL << OS_CONSTANT(EFFECTIVE_ADDRESS_WIDTH)) - 1) & ~(8ULL - 1);
244 static uintptr_t buildCompositeValue(InlineCallFrame* inlineCallFrame, BytecodeIndex bytecodeIndex)
245 {
246 if (!bytecodeIndex)
247 return bitwise_cast<uintptr_t>(inlineCallFrame) | s_maskIsBytecodeIndexInvalid;
248
249 if (UNLIKELY(bytecodeIndex.asBits() >= 1 << s_freeBitsAtTop)) {
250 auto* outOfLine = new OutOfLineCodeOrigin(inlineCallFrame, bytecodeIndex);
251 return bitwise_cast<uintptr_t>(outOfLine) | s_maskIsOutOfLine;
252 }
253
254 uintptr_t encodedBytecodeIndex = static_cast<uintptr_t>(bytecodeIndex.asBits()) << (64 - s_freeBitsAtTop);
255 ASSERT(!(encodedBytecodeIndex & bitwise_cast<uintptr_t>(inlineCallFrame)));
256 return encodedBytecodeIndex | bitwise_cast<uintptr_t>(inlineCallFrame);
257 }
258
259 // The bottom bit indicates whether to look at an out-of-line implementation (because of a bytecode index which is too big for us to store).
260 // The next bit indicates whether this is an invalid bytecode (which depending on the InlineCallFrame* can either indicate an unset CodeOrigin,
261 // or a deletion marker for a hash table).
262 // The next bit is free
263 // The next 64-s_freeBitsAtTop-3 are the InlineCallFrame* or the OutOfLineCodeOrigin*
264 // Finally the last s_freeBitsAtTop are the bytecodeIndex if it is inline
265 uintptr_t m_compositeValue;
266#else
267 BytecodeIndex m_bytecodeIndex;
268 InlineCallFrame* m_inlineCallFrame;
269#endif
270};
271
272inline unsigned CodeOrigin::hash() const
273{
274 return WTF::IntHash<unsigned>::hash(bytecodeIndex().asBits()) +
275 WTF::PtrHash<InlineCallFrame*>::hash(inlineCallFrame());
276}
277
278inline bool CodeOrigin::operator==(const CodeOrigin& other) const
279{
280#if CPU(ADDRESS64)
281 if (m_compositeValue == other.m_compositeValue)
282 return true;
283#endif
284 return bytecodeIndex() == other.bytecodeIndex()
285 && inlineCallFrame() == other.inlineCallFrame();
286}

288struct CodeOriginHash {
289 static unsigned hash(const CodeOrigin& key) { return key.hash(); }
290 static bool equal(const CodeOrigin& a, const CodeOrigin& b) { return a == b; }
291 static constexpr bool safeToCompareToEmptyOrDeleted = true;
292};
293
294struct CodeOriginApproximateHash {
295 static unsigned hash(const CodeOrigin& key) { return key.approximateHash(); }
296 static bool equal(const CodeOrigin& a, const CodeOrigin& b) { return a.isApproximatelyEqualTo(b); }
297 static constexpr bool safeToCompareToEmptyOrDeleted = true;
298};
