33ALWAYS_INLINE void* CompleteSubspace::allocateNonVirtual(VM& vm, size_t size, GCDeferralContext* deferralContext, AllocationFailureMode failureMode)
34{
35 if constexpr (validateDFGDoesGC)
36 vm.heap.verifyCanGC();
37
38 if (Allocator allocator = allocatorForNonVirtual(size, AllocatorForMode::AllocatorIfExists))
39 return allocator.allocate(vm.heap, deferralContext, failureMode);
40 return allocateSlow(vm, size, deferralContext, failureMode);
41}
