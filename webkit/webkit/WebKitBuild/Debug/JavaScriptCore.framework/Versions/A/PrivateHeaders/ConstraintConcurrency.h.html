26#pragma once
27
28#include <wtf/PrintStream.h>
29
30namespace JSC {
31
32enum class ConstraintConcurrency : uint8_t {
33 Sequential,
34 Concurrent
35};
36
37} // namespace JSC
38
39namespace WTF {
40
41inline void printInternal(PrintStream& out, JSC::ConstraintConcurrency concurrency)
42{
43 switch (concurrency) {
44 case JSC::ConstraintConcurrency::Sequential:
45 out.print("Sequential");
46 return;
47 case JSC::ConstraintConcurrency::Concurrent:
48 out.print("Concurrent");
49 return;
50 }
51 RELEASE_ASSERT_NOT_REACHED();
52}
53
54} // namespace WTF
