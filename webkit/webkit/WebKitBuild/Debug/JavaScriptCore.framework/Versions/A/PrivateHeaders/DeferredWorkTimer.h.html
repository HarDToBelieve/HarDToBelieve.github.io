28#include "JSRunLoopTimer.h"
29#include "Strong.h"

31#include <wtf/Deque.h>
32#include <wtf/HashMap.h>
33#include <wtf/Lock.h>
34#include <wtf/Vector.h>
35
36namespace JSC {
37
38class JSPromise;
39class VM;
40class JSCell;
41
42class JS_EXPORT_PRIVATE DeferredWorkTimer final : public JSRunLoopTimer {
43public:
44 using Base = JSRunLoopTimer;
45
46 struct TicketData {
47 Vector<Strong<JSCell>> dependencies;
48 Strong<JSObject> scriptExecutionOwner;
49 };
50
51 void doWork(VM&) final;
52
53 using Ticket = JSObject*;
54 void addPendingWork(VM&, Ticket, Vector<Strong<JSCell>>&& dependencies);
55 bool hasPendingWork(Ticket);
56 bool hasDependancyInPendingWork(Ticket, JSCell* dependency);
57 bool cancelPendingWork(Ticket);
58
59 // If the script execution owner your ticket is associated with gets canceled
60 // the Task will not be called and will be deallocated. So it's important
61 // to make sure your memory ownership model won't leak memory when
62 // this occurs. The easiest way is to make sure everything is either owned
63 // by a GC'd value in dependencies or by the Task lambda.
64 using Task = Function<void(Ticket, TicketData&&)>;
65 void scheduleWorkSoon(Ticket, Task&&);
66 void didResumeScriptExecutionOwner();
67
68 void stopRunningTasks() { m_runTasks = false; }
69 void runRunLoop();
70
71 static Ref<DeferredWorkTimer> create(VM& vm) { return adoptRef(*new DeferredWorkTimer(vm)); }
72private:
73 DeferredWorkTimer(VM&);
74
75 Lock m_taskLock;
76 bool m_runTasks { true };
77 bool m_shouldStopRunLoopWhenAllTicketsFinish { false };
78 bool m_currentlyRunningTask { false };
79 Deque<std::tuple<Ticket, Task>> m_tasks WTF_GUARDED_BY_LOCK(m_taskLock);
80 HashMap<Ticket, TicketData> m_pendingTickets;
81};
82
83} // namespace JSC
