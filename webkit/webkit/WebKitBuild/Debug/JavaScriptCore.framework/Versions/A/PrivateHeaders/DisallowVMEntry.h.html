26#pragma once
27
28#include <wtf/ForbidHeapAllocation.h>
29
30namespace JSC {
31
32class VM;
33
34// The only reason we implement DisallowVMEntry as specialization of a template
35// is so that we can work around having to #include VM.h, which can hurt build
36// time. This defers the cost of #include'ing VM.h to only the clients that
37// need it.
38
39template<typename VMType = VM>
40class DisallowVMEntryImpl {
41 WTF_FORBID_HEAP_ALLOCATION;
42public:
43 DisallowVMEntryImpl(VMType& vm)
44 : m_vm(&vm)
45 {
46 m_vm->disallowVMEntryCount++;
47 }
48
49 DisallowVMEntryImpl(const DisallowVMEntryImpl& other)
50 : m_vm(other.m_vm)
51 {
52 m_vm->disallowVMEntryCount++;
53 }
54
55 ~DisallowVMEntryImpl()
56 {
57 RELEASE_ASSERT(m_vm->disallowVMEntryCount);
58 m_vm->disallowVMEntryCount--;
59 m_vm = nullptr;
60 }
61
62 DisallowVMEntryImpl& operator=(const DisallowVMEntryImpl& other)
63 {
64 RELEASE_ASSERT(m_vm && m_vm == other.m_vm);
65 RELEASE_ASSERT(m_vm->disallowVMEntryCount);
66 // Conceptually, we need to decrement the disallowVMEntryCount of the
67 // old m_vm, and increment the disallowVMEntryCount of the new m_vm.
68 // But since the old and the new m_vm should always be the same, the
69 // decrementing and incrementing cancels out, and there's nothing more
70 // to do here.
71 return *this;
72 }
73
74private:
75 VMType* m_vm;
76};
77
78using DisallowVMEntry = DisallowVMEntryImpl<VM>;
79
80} // namespace JSC
