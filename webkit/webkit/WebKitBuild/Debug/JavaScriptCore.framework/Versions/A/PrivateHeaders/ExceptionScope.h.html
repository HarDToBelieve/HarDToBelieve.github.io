49class ExceptionScope {
50public:
51 VM& vm() const { return m_vm; }
52 unsigned recursionDepth() const { return m_recursionDepth; }
53 Exception* exception() const { return m_vm.exception(); }
54
55 ALWAYS_INLINE void assertNoException() { RELEASE_ASSERT_WITH_MESSAGE(!exception(), "%s", unexpectedExceptionMessage().data()); }
56 ALWAYS_INLINE void releaseAssertNoException() { RELEASE_ASSERT_WITH_MESSAGE(!exception(), "%s", unexpectedExceptionMessage().data()); }
57 ALWAYS_INLINE void assertNoExceptionExceptTermination() { RELEASE_ASSERT_WITH_MESSAGE(!exception() || m_vm.hasPendingTerminationException(), "%s", unexpectedExceptionMessage().data()); }
58 ALWAYS_INLINE void releaseAssertNoExceptionExceptTermination() { RELEASE_ASSERT_WITH_MESSAGE(!exception() || m_vm.hasPendingTerminationException(), "%s", unexpectedExceptionMessage().data()); }
59
60#if ASAN_ENABLED || ENABLE(C_LOOP)
61 const void* stackPosition() const { return m_location.stackPosition; }
62#else
63 const void* stackPosition() const { return this; }
64#endif
65
66protected:
67 ExceptionScope(VM&, ExceptionEventLocation);
68 ExceptionScope(const ExceptionScope&) = delete;
69 ExceptionScope(ExceptionScope&&) = default;
70 ~ExceptionScope();
71
72 JS_EXPORT_PRIVATE CString unexpectedExceptionMessage();
73
74 VM& m_vm;
75 ExceptionScope* m_previousScope;
76 ExceptionEventLocation m_location;
77 unsigned m_recursionDepth;
78};
79
80#else // not ENABLE(EXCEPTION_SCOPE_VERIFICATION)
81
82#define EXCEPTION_ASSERT(x) ASSERT(x)
83#define EXCEPTION_ASSERT_UNUSED(variable, assertion) ASSERT_UNUSED(variable, assertion)
84#define EXCEPTION_ASSERT_WITH_MESSAGE(assertion, message) ASSERT_WITH_MESSAGE(assertion, message)
85
86class ExceptionScope {
87public:
88 ALWAYS_INLINE VM& vm() const { return m_vm; }
89 ALWAYS_INLINE Exception* exception() const { return m_vm.exception(); }
90
91 ALWAYS_INLINE void assertNoException() { ASSERT(!exception()); }
92 ALWAYS_INLINE void releaseAssertNoException() { RELEASE_ASSERT(!exception()); }
93 ALWAYS_INLINE void assertNoExceptionExceptTermination() { ASSERT(!exception() || m_vm.hasPendingTerminationException()); }
94 ALWAYS_INLINE void releaseAssertNoExceptionExceptTermination() { RELEASE_ASSERT(!exception() || m_vm.hasPendingTerminationException()); }
95
96protected:
97 ALWAYS_INLINE ExceptionScope(VM& vm)
98 : m_vm(vm)
99 { }
100 ExceptionScope(const ExceptionScope&) = delete;
101 ExceptionScope(ExceptionScope&&) = default;
102
103 ALWAYS_INLINE CString unexpectedExceptionMessage() { return { }; }
104
105 VM& m_vm;
106};
107
108#endif // ENABLE(EXCEPTION_SCOPE_VERIFICATION)

110#define RETURN_IF_EXCEPTION(scope__, value__) do { \
111 JSC::VM& vm = (scope__).vm(); \
112 ASSERT(!!(scope__).exception() == vm.traps().needHandling(JSC::VMTraps::NeedExceptionHandling)); \
113 if (UNLIKELY(vm.traps().needHandling(JSC::VMTraps::NonDebuggerEvents))) { \
114 if (vm.hasExceptionsAfterHandlingTraps()) \
115 return value__; \
116 } \
117 } while (false)
118
119#define RELEASE_AND_RETURN(scope__, expression__) do { \
120 scope__.release(); \
121 return expression__; \
122 } while (false)
