26#pragma once
27
28#if OS(DARWIN) && CPU(ARM64)
29
30#include "JSCConfig.h"
31
32#include <wtf/Platform.h>
33
34#if USE(PTHREAD_JIT_PERMISSIONS_API)
35#include <pthread.h>
36#elif USE(APPLE_INTERNAL_SDK)
37#include <os/thread_self_restrict.h>
38#endif
39
40static ALWAYS_INLINE void threadSelfRestrictRWXToRW()
41{
42 ASSERT(g_jscConfig.useFastJITPermissions);
43
44#if USE(PTHREAD_JIT_PERMISSIONS_API)
45 pthread_jit_write_protect_np(false);
46#elif USE(APPLE_INTERNAL_SDK)
47 os_thread_self_restrict_rwx_to_rw();
48#else
49 bool tautologyToIgnoreWarning = true;
50 if (tautologyToIgnoreWarning)
51 RELEASE_ASSERT_NOT_REACHED();
52#endif
53}
54
55static ALWAYS_INLINE void threadSelfRestrictRWXToRX()
56{
57 ASSERT(g_jscConfig.useFastJITPermissions);
58
59#if USE(PTHREAD_JIT_PERMISSIONS_API)
60 pthread_jit_write_protect_np(true);
61#elif USE(APPLE_INTERNAL_SDK)
62 os_thread_self_restrict_rwx_to_rx();
63#else
64 bool tautologyToIgnoreWarning = true;
65 if (tautologyToIgnoreWarning)
66 RELEASE_ASSERT_NOT_REACHED();
67#endif
68}
69
70#else // Not OS(DARWIN) && CPU(ARM64)
71
72NO_RETURN_DUE_TO_CRASH ALWAYS_INLINE void threadSelfRestrictRWXToRW()
73{
74 CRASH();
75}
76
77NO_RETURN_DUE_TO_CRASH ALWAYS_INLINE void threadSelfRestrictRWXToRX()
78{
79 CRASH();
80}
81
82#endif // OS(DARWIN) && CPU(ARM64)
