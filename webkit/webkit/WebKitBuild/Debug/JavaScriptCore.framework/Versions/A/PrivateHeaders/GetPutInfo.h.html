37enum ResolveMode {
38 ThrowIfNotFound,
39 DoNotThrowIfNotFound

42#define FOR_EACH_RESOLVE_TYPE(v) \
43 v(GlobalProperty) \
44 v(GlobalVar) \
45 v(GlobalLexicalVar) \
46 v(ClosureVar) \
47 v(ResolvedClosureVar) \
48 v(ModuleVar) \
49 v(GlobalPropertyWithVarInjectionChecks) \
50 v(GlobalVarWithVarInjectionChecks) \
51 v(GlobalLexicalVarWithVarInjectionChecks) \
52 v(ClosureVarWithVarInjectionChecks) \
53 v(UnresolvedProperty) \
54 v(UnresolvedPropertyWithVarInjectionChecks) \
55 v(Dynamic)

57enum ResolveType : unsigned {
58 // Lexical scope guaranteed a certain type of variable access.
59 GlobalProperty,
60 GlobalVar,
61 GlobalLexicalVar,
62 ClosureVar,
63 ResolvedClosureVar,
64 ModuleVar,
65
66 // Ditto, but at least one intervening scope used non-strict eval, which
67 // can inject an intercepting var delcaration at runtime.
68 GlobalPropertyWithVarInjectionChecks,
69 GlobalVarWithVarInjectionChecks,
70 GlobalLexicalVarWithVarInjectionChecks,
71 ClosureVarWithVarInjectionChecks,
72
73 // We haven't found which scope this belongs to, and we also
74 // haven't ruled out the possibility of it being cached. Ideally,
75 // we want to transition this to GlobalVar/GlobalLexicalVar/GlobalProperty <with/without injection>
76 UnresolvedProperty,
77 UnresolvedPropertyWithVarInjectionChecks,
78
79 // Lexical scope didn't prove anything -- probably because of a 'with' scope.
80 Dynamic
81};
82
83enum class InitializationMode : unsigned {
84 Initialization, // "let x = 20;"
85 ConstInitialization, // "const x = 20;"
86 NotInitialization // "x = 20;"
87};

128ALWAYS_INLINE bool isInitialization(InitializationMode initializationMode)
129{
130 switch (initializationMode) {
131 case InitializationMode::Initialization:
132 case InitializationMode::ConstInitialization:
133 return true;
134 case InitializationMode::NotInitialization:
135 return false;
136 }
137 ASSERT_NOT_REACHED();
138 return false;
139}
140
141ALWAYS_INLINE ResolveType makeType(ResolveType type, bool needsVarInjectionChecks)
142{
143 if (!needsVarInjectionChecks)
144 return type;
145
146 switch (type) {
147 case GlobalProperty:
148 return GlobalPropertyWithVarInjectionChecks;
149 case GlobalVar:
150 return GlobalVarWithVarInjectionChecks;
151 case GlobalLexicalVar:
152 return GlobalLexicalVarWithVarInjectionChecks;
153 case ClosureVar:
154 case ResolvedClosureVar:
155 return ClosureVarWithVarInjectionChecks;
156 case UnresolvedProperty:
157 return UnresolvedPropertyWithVarInjectionChecks;
158 case ModuleVar:
159 case GlobalPropertyWithVarInjectionChecks:
160 case GlobalVarWithVarInjectionChecks:
161 case GlobalLexicalVarWithVarInjectionChecks:
162 case ClosureVarWithVarInjectionChecks:
163 case UnresolvedPropertyWithVarInjectionChecks:
164 case Dynamic:
165 return type;
166 }
167
168 RELEASE_ASSERT_NOT_REACHED();
169 return type;
170}
171
172ALWAYS_INLINE bool needsVarInjectionChecks(ResolveType type)
173{
174 switch (type) {
175 case GlobalProperty:
176 case GlobalVar:
177 case GlobalLexicalVar:
178 case ClosureVar:
179 case ResolvedClosureVar:
180 case ModuleVar:
181 case UnresolvedProperty:
182 return false;
183 case GlobalPropertyWithVarInjectionChecks:
184 case GlobalVarWithVarInjectionChecks:
185 case GlobalLexicalVarWithVarInjectionChecks:
186 case ClosureVarWithVarInjectionChecks:
187 case UnresolvedPropertyWithVarInjectionChecks:
188 case Dynamic:
189 return true;
190 default:
191 RELEASE_ASSERT_NOT_REACHED();
192 return true;
193 }
194}

217class GetPutInfo {
218 typedef unsigned Operand;
219public:
220 // Give each field 10 bits for simplicity.
221 static_assert(sizeof(Operand) * 8 > 31, "Not enough bits for GetPutInfo");
222 static constexpr unsigned isStrictShift = 30;
223 static constexpr unsigned modeShift = 20;
224 static constexpr unsigned initializationShift = 10;
225 static constexpr unsigned typeBits = (1 << initializationShift) - 1;
226 static constexpr unsigned initializationBits = ((1 << modeShift) - 1) & ~typeBits;
227 static constexpr unsigned modeBits = ((1 << 30) - 1) & ~initializationBits & ~typeBits;
228 static constexpr unsigned isStrictBit = 1 << 30;
229 static_assert((modeBits & initializationBits & typeBits & isStrictBit) == 0x0, "There should be no intersection between ResolveMode ResolveType and InitializationMode");
230
231 GetPutInfo() = default;
232
233 GetPutInfo(ResolveMode resolveMode, ResolveType resolveType, InitializationMode initializationMode, ECMAMode ecmaMode)
234 : m_operand((ecmaMode.isStrict() << isStrictShift) | (resolveMode << modeShift) | (static_cast<unsigned>(initializationMode) << initializationShift) | resolveType)
235 {
236 }
237
238 explicit GetPutInfo(unsigned operand)
239 : m_operand(operand)
240 {
241 }
242
243 ResolveType resolveType() const { return static_cast<ResolveType>(m_operand & typeBits); }
244 InitializationMode initializationMode() const { return static_cast<InitializationMode>((m_operand & initializationBits) >> initializationShift); }
245 ResolveMode resolveMode() const { return static_cast<ResolveMode>((m_operand & modeBits) >> modeShift); }
246 ECMAMode ecmaMode() const { return m_operand & isStrictBit ? ECMAMode::strict() : ECMAMode::sloppy(); }
247 unsigned operand() const { return m_operand; }
248
249 void dump(PrintStream&) const;
250
251private:
252 Operand m_operand { 0 };
253
254 friend class JSC::LLIntOffsetsExtractor;
255};
