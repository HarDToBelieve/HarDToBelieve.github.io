26#pragma once
27
28#include <wtf/DoublyLinkedList.h>
29
30namespace JSC {
31
32class HandleSet;
33class HandleNode;
34
35class HandleBlock : public DoublyLinkedListNode<HandleBlock> {
36 friend class WTF::DoublyLinkedListNode<HandleBlock>;
37public:
38 static HandleBlock* create(HandleSet*);
39 static void destroy(HandleBlock*);
40 static HandleBlock* blockFor(HandleNode*);
41
42 static constexpr size_t blockSize = 4 * KB;
43
44 HandleSet* handleSet();
45
46 HandleNode* nodes();
47 HandleNode* nodeAtIndex(unsigned);
48 unsigned nodeCapacity();
49
50private:
51 HandleBlock(HandleSet*);
52
53 char* payload();
54 char* payloadEnd();
55
56 static constexpr size_t s_blockMask = ~(blockSize - 1);
57
58 HandleBlock* m_prev;
59 HandleBlock* m_next;
60 HandleSet* m_handleSet;
61};
62
63inline HandleBlock* HandleBlock::blockFor(HandleNode* node)
64{
65 return reinterpret_cast<HandleBlock*>(reinterpret_cast<size_t>(node) & s_blockMask);
66}
67
68inline HandleSet* HandleBlock::handleSet()
69{
70 return m_handleSet;
71}
72
73} // namespace JSC
