102inline HandleSet* HandleSet::heapFor(HandleSlot handle)
103{
104 return toNode(handle)->handleSet();
105}

112inline HandleSlot HandleSet::toHandle(HandleSet::Node* node)

114 return reinterpret_cast<HandleSlot>(node);

117inline HandleSet::Node* HandleSet::toNode(HandleSlot handle)
118{
119 return reinterpret_cast<HandleSet::Node*>(handle);
120}
121
122inline HandleSlot HandleSet::allocate()
123{
124 if (m_freeList.isEmpty())
125 grow();
126
127 HandleSet::Node* node = m_freeList.pop();
128 new (NotNull, node) HandleSet::Node();
129 m_immediateList.push(node);
130 return toHandle(node);
131}
132
133inline void HandleSet::deallocate(HandleSlot handle)
134{
135 HandleSet::Node* node = toNode(handle);
136 SentinelLinkedList<HandleSet::Node>::remove(node);
137 m_freeList.push(node);
138}

182template<typename Functor> void HandleSet::forEachStrongHandle(const Functor& functor, const HashCountedSet<JSCell*>& skipSet)
183{
184 HandleSet::Node* end = m_strongList.end();
185 for (HandleSet::Node* node = m_strongList.begin(); node != end; node = node->next()) {
186 JSValue value = *node->slot();
187 if (!value || !value.isCell())
188 continue;
189 if (skipSet.contains(value.asCell()))
190 continue;
191 functor(value.asCell());
192 }
193}
