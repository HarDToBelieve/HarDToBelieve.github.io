35ALWAYS_INLINE bool areKeysEqual(JSGlobalObject* globalObject, JSValue a, JSValue b)
36{
37 // We want +0 and -0 to be compared to true here. sameValue() itself doesn't
38 // guarantee that, however, we normalize all keys before comparing and storing
39 // them in the map. The normalization will convert -0.0 and 0.0 to the integer
40 // representation for 0.
41 return sameValue(globalObject, a, b);
42}
43
44// Note that normalization is inlined in DFG's NormalizeMapKey.
45// Keep in sync with the implementation of DFG and FTL normalization.
46ALWAYS_INLINE JSValue normalizeMapKey(JSValue key)
47{
48 if (!key.isNumber()) {
49 if (key.isHeapBigInt())
50 return tryConvertToBigInt32(key.asHeapBigInt());
51 return key;
52 }
53
54 if (key.isInt32())
55 return key;
56
57 double d = key.asDouble();
58 if (std::isnan(d))
59 return jsNaN();
60
61 int i = static_cast<int>(d);
62 if (i == d) {
63 // When a key is -0, we convert it to positive zero.
64 // When a key is the double representation for an integer, we convert it to an integer.
65 return jsNumber(i);
66 }
67 // This means key is definitely not negative zero, and it's definitely not a double representation of an integer.
68 return key;
69}

71ALWAYS_INLINE uint32_t wangsInt64Hash(uint64_t key)
72{
73 key += ~(key << 32);
74 key ^= (key >> 22);
75 key += ~(key << 13);
76 key ^= (key >> 8);
77 key += (key << 3);
78 key ^= (key >> 15);
79 key += ~(key << 27);
80 key ^= (key >> 31);
81 return static_cast<unsigned>(key);
82}

84ALWAYS_INLINE uint32_t jsMapHash(JSBigInt* bigInt)
85{
86 return bigInt->hash();
87}

89template<ExceptionExpectation expection>
90ALWAYS_INLINE uint32_t jsMapHashImpl(JSGlobalObject* globalObject, VM& vm, JSValue value)
91{
92 ASSERT_WITH_MESSAGE(normalizeMapKey(value) == value, "We expect normalized values flowing into this function.");
93
94 if (value.isString()) {
95 auto scope = DECLARE_THROW_SCOPE(vm);
96 const String& wtfString = asString(value)->value(globalObject);
97 if constexpr (expection == ExceptionExpectation::CanThrow)
98 RETURN_IF_EXCEPTION(scope, UINT_MAX);
99 else
100 EXCEPTION_ASSERT_UNUSED(scope, !scope.exception());
101 return wtfString.impl()->hash();
102 }
103
104 if (value.isHeapBigInt())
105 return jsMapHash(value.asHeapBigInt());
106
107 return wangsInt64Hash(JSValue::encode(value));
108}

110ALWAYS_INLINE uint32_t jsMapHash(JSGlobalObject* globalObject, VM& vm, JSValue value)
111{
112 return jsMapHashImpl<ExceptionExpectation::CanThrow>(globalObject, vm, value);
113}

115ALWAYS_INLINE uint32_t jsMapHashForAlreadyHashedValue(JSGlobalObject* globalObject, VM& vm, JSValue value)
116{
117 return jsMapHashImpl<ExceptionExpectation::ShouldNotThrow>(globalObject, vm, value);
118}

120ALWAYS_INLINE std::optional<uint32_t> concurrentJSMapHash(JSValue key)
121{
122 key = normalizeMapKey(key);
123 if (key.isString()) {
124 JSString* string = asString(key);
125 if (string->length() > 10 * 1024)
126 return std::nullopt;
127 const StringImpl* impl = string->tryGetValueImpl();
128 if (!impl)
129 return std::nullopt;
130 return impl->concurrentHash();
131 }
132
133 if (key.isHeapBigInt())
134 return key.asHeapBigInt()->concurrentHash();
135
136 uint64_t rawValue = JSValue::encode(key);
137 return wangsInt64Hash(rawValue);
138}

140ALWAYS_INLINE uint32_t shouldShrink(uint32_t capacity, uint32_t keyCount)
141{
142 return 8 * keyCount <= capacity && capacity > 4;
143}
144
145ALWAYS_INLINE uint32_t shouldRehashAfterAdd(uint32_t capacity, uint32_t keyCount, uint32_t deleteCount)
146{
147 return 2 * (keyCount + deleteCount) >= capacity;
148}
149
150ALWAYS_INLINE uint32_t nextCapacity(uint32_t capacity, uint32_t keyCount)
151{
152 if (shouldShrink(capacity, keyCount)) {
153 ASSERT((capacity / 2) >= 4);
154 return capacity / 2;
155 }
156
157 if (3 * keyCount <= capacity && capacity > 64) {
158 // We stay at the same size if rehashing would cause us to be no more than
159 // 1/3rd full. This comes up for programs like this:
160 // Say the hash table grew to a key count of 64, causing it to grow to a capacity of 256.
161 // Then, the table added 63 items. The load is now 127. Then, 63 items are deleted.
162 // The load is still 127. Then, another item is added. The load is now 128, and we
163 // decide that we need to rehash. The key count is 65, almost exactly what it was
164 // when we grew to a capacity of 256. We don't really need to grow to a capacity
165 // of 512 in this situation. Instead, we choose to rehash at the same size. This
166 // will bring the load down to 65. We rehash into the same size when we determine
167 // that the new load ratio will be under 1/3rd. (We also pick a minumum capacity
168 // at which this rule kicks in because otherwise we will be too sensitive to rehashing
169 // at the same capacity).
170 return capacity;
171 }
172 return Checked<uint32_t>(capacity) * 2;
173}

175template <typename HashMapBucketType>
176void HashMapImpl<HashMapBucketType>::finishCreation(JSGlobalObject* globalObject, VM& vm)
177{
178 ASSERT_WITH_MESSAGE(HashMapBucket<HashMapBucketDataKey>::offsetOfKey() == HashMapBucket<HashMapBucketDataKeyValue>::offsetOfKey(), "We assume this to be true in the DFG and FTL JIT.");
179
180 auto scope = DECLARE_THROW_SCOPE(vm);
181 Base::finishCreation(vm);
182
183 makeAndSetNewBuffer(globalObject, vm);
184 RETURN_IF_EXCEPTION(scope, void());
185
186 setUpHeadAndTail(globalObject, vm);
187}

189template <typename HashMapBucketType>
190void HashMapImpl<HashMapBucketType>::finishCreation(JSGlobalObject* globalObject, VM& vm, HashMapImpl* base)
191{
192 auto scope = DECLARE_THROW_SCOPE(vm);
193 Base::finishCreation(vm);
194
195 // This size should be the same to the case when you clone the map by calling add() repeatedly.
196 uint32_t capacity = (Checked<uint32_t>(base->m_keyCount) * 2) + 1;
197 RELEASE_ASSERT(capacity <= (1U << 31));
198 capacity = std::max<uint32_t>(WTF::roundUpToPowerOfTwo(capacity), 4U);
199 m_capacity = capacity;
200 makeAndSetNewBuffer(globalObject, vm);
201 RETURN_IF_EXCEPTION(scope, void());
202
203 setUpHeadAndTail(globalObject, vm);
204
205 HashMapBucketType* bucket = base->m_head.get()->next();
206 while (bucket) {
207 if (!bucket->deleted()) {
208 addNormalizedNonExistingForCloning(globalObject, bucket->key(), HashMapBucketType::extractValue(*bucket));
209 RETURN_IF_EXCEPTION(scope, void());
210 }
211 bucket = bucket->next();
212 }
213 checkConsistency();
214}

216template <typename HashMapBucketType>
217ALWAYS_INLINE HashMapBucketType** HashMapImpl<HashMapBucketType>::findBucket(JSGlobalObject* globalObject, JSValue key)
218{
219 VM& vm = getVM(globalObject);
220 auto scope = DECLARE_THROW_SCOPE(vm);
221 key = normalizeMapKey(key);
222 uint32_t hash = jsMapHash(globalObject, vm, key);
223 RETURN_IF_EXCEPTION(scope, nullptr);
224 return findBucket(globalObject, key, hash);
225}

230 ASSERT_WITH_MESSAGE(normalizeMapKey(key) == key, "We expect normalized values flowing into this function.");
231 return findBucketAlreadyHashedAndNormalized(globalObject, key, hash);

249template <typename HashMapBucketType>
250ALWAYS_INLINE void HashMapImpl<HashMapBucketType>::add(JSGlobalObject* globalObject, JSValue key, JSValue value)
251{
252 VM& vm = getVM(globalObject);
253 auto scope = DECLARE_THROW_SCOPE(vm);
254
255 key = normalizeMapKey(key);
256 addNormalizedInternal(globalObject, key, value, [&] (HashMapBucketType* bucket) {
257 return !isDeleted(bucket) && areKeysEqual(globalObject, key, bucket->key());
258 });
259 RETURN_IF_EXCEPTION(scope, void());
260 scope.release();
261 if (shouldRehashAfterAdd())
262 rehash(globalObject);
263}

265template <typename HashMapBucketType>
266ALWAYS_INLINE HashMapBucketType* HashMapImpl<HashMapBucketType>::addNormalized(JSGlobalObject* globalObject, JSValue key, JSValue value, uint32_t hash)
267{
268 VM& vm = getVM(globalObject);
269 ASSERT_WITH_MESSAGE(normalizeMapKey(key) == key, "We expect normalized values flowing into this function.");
270 DEFER_TERMINATION_AND_ASSERT_WITH_MESSAGE(vm, jsMapHash(globalObject, getVM(globalObject), key) == hash, "We expect hash value is what we expect.");
271
272 auto* bucket = addNormalizedInternal(vm, key, value, hash, [&] (HashMapBucketType* bucket) {
273 return !isDeleted(bucket) && areKeysEqual(globalObject, key, bucket->key());
274 });
275 if (shouldRehashAfterAdd())
276 rehash(globalObject);
277 return bucket;
278}

280template <typename HashMapBucketType>
281ALWAYS_INLINE bool HashMapImpl<HashMapBucketType>::remove(JSGlobalObject* globalObject, JSValue key)
282{
283 HashMapBucketType** bucket = findBucket(globalObject, key);
284 if (!bucket)
285 return false;
286
287 VM& vm = getVM(globalObject);
288 HashMapBucketType* impl = *bucket;
289 impl->next()->setPrev(vm, impl->prev());
290 impl->prev()->setNext(vm, impl->next());
291 impl->makeDeleted(vm);
292
293 *bucket = deletedValue();
294
295 ++m_deleteCount;
296 ASSERT(m_keyCount > 0);
297 --m_keyCount;
298
299 if (shouldShrink())
300 rehash(globalObject);
301
302 return true;
303}

305template <typename HashMapBucketType>
306ALWAYS_INLINE void HashMapImpl<HashMapBucketType>::clear(JSGlobalObject* globalObject)
307{
308 VM& vm = getVM(globalObject);
309 m_keyCount = 0;
310 m_deleteCount = 0;
311 HashMapBucketType* head = m_head.get();
312 HashMapBucketType* bucket = m_head->next();
313 HashMapBucketType* tail = m_tail.get();
314 while (bucket != tail) {
315 HashMapBucketType* next = bucket->next();
316 // We restart each iterator by pointing it to the head of the list.
317 bucket->setNext(vm, head);
318 bucket->makeDeleted(vm);
319 bucket = next;
320 }
321 m_head->setNext(vm, m_tail.get());
322 m_tail->setPrev(vm, m_head.get());
323 m_capacity = 4;
324 makeAndSetNewBuffer(globalObject, vm);
325 checkConsistency();
326}

328template <typename HashMapBucketType>
329ALWAYS_INLINE void HashMapImpl<HashMapBucketType>::setUpHeadAndTail(JSGlobalObject*, VM& vm)
330{
331 m_head.set(vm, this, HashMapBucketType::create(vm));
332 m_tail.set(vm, this, HashMapBucketType::create(vm));
333
334 m_head->setNext(vm, m_tail.get());
335 m_tail->setPrev(vm, m_head.get());
336 ASSERT(m_head->deleted());
337 ASSERT(m_tail->deleted());
338}

348template <typename HashMapBucketType>
349template<typename CanUseBucket>
350ALWAYS_INLINE void HashMapImpl<HashMapBucketType>::addNormalizedInternal(JSGlobalObject* globalObject, JSValue key, JSValue value, const CanUseBucket& canUseBucket)
351{
352 VM& vm = getVM(globalObject);
353 auto scope = DECLARE_THROW_SCOPE(vm);
354
355 uint32_t hash = jsMapHash(globalObject, vm, key);
356 RETURN_IF_EXCEPTION(scope, void());
357 scope.release();
358 addNormalizedInternal(vm, key, value, hash, canUseBucket);
359}

361template <typename HashMapBucketType>
362template<typename CanUseBucket>
363ALWAYS_INLINE HashMapBucketType* HashMapImpl<HashMapBucketType>::addNormalizedInternal(VM& vm, JSValue key, JSValue value, uint32_t hash, const CanUseBucket& canUseBucket)
364{
365 ASSERT_WITH_MESSAGE(normalizeMapKey(key) == key, "We expect normalized values flowing into this function.");
366
367 const uint32_t mask = m_capacity - 1;
368 uint32_t index = hash & mask;
369 HashMapBucketType** buffer = this->buffer();
370 HashMapBucketType* bucket = buffer[index];
371 while (!isEmpty(bucket)) {
372 if (canUseBucket(bucket)) {
373 bucket->setValue(vm, value);
374 return bucket;
375 }
376 index = (index + 1) & mask;
377 bucket = buffer[index];
378 }
379
380 HashMapBucketType* newEntry = m_tail.get();
381 buffer[index] = newEntry;
382 newEntry->setKey(vm, key);
383 newEntry->setValue(vm, value);
384 ASSERT(!newEntry->deleted());
385 HashMapBucketType* newTail = HashMapBucketType::create(vm);
386 m_tail.set(vm, this, newTail);
387 newTail->setPrev(vm, newEntry);
388 ASSERT(newTail->deleted());
389 newEntry->setNext(vm, newTail);
390
391 ++m_keyCount;
392 return newEntry;
393}

395template <typename HashMapBucketType>
396ALWAYS_INLINE HashMapBucketType** HashMapImpl<HashMapBucketType>::findBucketAlreadyHashedAndNormalized(JSGlobalObject* globalObject, JSValue key, uint32_t hash)
397{
398 const uint32_t mask = m_capacity - 1;
399 uint32_t index = hash & mask;
400 HashMapBucketType** buffer = this->buffer();
401 HashMapBucketType* bucket = buffer[index];
402
403 while (!isEmpty(bucket)) {
404 if (!isDeleted(bucket) && areKeysEqual(globalObject, key, bucket->key()))
405 return buffer + index;
406 index = (index + 1) & mask;
407 bucket = buffer[index];
408 }
409 return nullptr;
410}

412template <typename HashMapBucketType>
413void HashMapImpl<HashMapBucketType>::rehash(JSGlobalObject* globalObject)
414{
415 VM& vm = getVM(globalObject);
416 auto scope = DECLARE_THROW_SCOPE(vm);
417
418 uint32_t oldCapacity = m_capacity;
419 m_capacity = nextCapacity(m_capacity, m_keyCount);
420
421 if (m_capacity != oldCapacity) {
422 makeAndSetNewBuffer(globalObject, vm);
423 RETURN_IF_EXCEPTION(scope, void());
424 } else {
425 m_buffer->reset(m_capacity);
426 assertBufferIsEmpty();
427 }
428
429 HashMapBucketType* iter = m_head->next();
430 HashMapBucketType* end = m_tail.get();
431 const uint32_t mask = m_capacity - 1;
432 RELEASE_ASSERT(!(m_capacity & (m_capacity - 1)));
433 HashMapBucketType** buffer = this->buffer();
434 while (iter != end) {
435 uint32_t index = jsMapHashForAlreadyHashedValue(globalObject, vm, iter->key()) & mask;
436 EXCEPTION_ASSERT_WITH_MESSAGE(!scope.exception(), "All keys should already be hashed before, so this should not throw because it won't resolve ropes.");
437 {
438 HashMapBucketType* bucket = buffer[index];
439 while (!isEmpty(bucket)) {
440 index = (index + 1) & mask;
441 bucket = buffer[index];
442 }
443 }
444 buffer[index] = iter;
445 iter = iter->next();
446 }
447
448 m_deleteCount = 0;
449
450 checkConsistency();
451}

453template <typename HashMapBucketType>
454ALWAYS_INLINE void HashMapImpl<HashMapBucketType>::checkConsistency() const
455{
456 if (ASSERT_ENABLED) {
457 HashMapBucketType* iter = m_head->next();
458 HashMapBucketType* end = m_tail.get();
459 uint32_t size = 0;
460 while (iter != end) {
461 ++size;
462 iter = iter->next();
463 }
464 ASSERT(size == m_keyCount);
465 }
466}

468template <typename HashMapBucketType>
469void HashMapImpl<HashMapBucketType>::makeAndSetNewBuffer(JSGlobalObject* globalObject, VM& vm)
470{
471 ASSERT(!(m_capacity & (m_capacity - 1)));
472
473 HashMapBufferType* buffer = HashMapBufferType::create(globalObject, vm, this, m_capacity);
474 if (UNLIKELY(!buffer))
475 return;
476
477 m_buffer.set(vm, this, buffer);
478 assertBufferIsEmpty();
479}

481template <typename HashMapBucketType>
482ALWAYS_INLINE void HashMapImpl<HashMapBucketType>::assertBufferIsEmpty() const
483{
484 if (ASSERT_ENABLED) {
485 for (unsigned i = 0; i < m_capacity; i++)
486 ASSERT(isEmpty(buffer()[i]));
487 }
