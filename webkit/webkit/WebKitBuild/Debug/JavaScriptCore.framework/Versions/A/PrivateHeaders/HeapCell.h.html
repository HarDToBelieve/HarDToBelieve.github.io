31namespace JSC {

33class CellContainer;
34class Heap;
35class PreciseAllocation;
36class MarkedBlock;
37class Subspace;
38class VM;
39struct CellAttributes;
40
41class HeapCell {
42public:
43 enum Kind : int8_t {
44 JSCell,
45 JSCellWithIndexingHeader,
46 Auxiliary
47 };
48
49 HeapCell() { }
50
51 // We're intentionally only zapping the bits for the structureID and leaving
52 // the rest of the cell header bits intact for crash analysis uses.
53 enum ZapReason : int8_t { Unspecified, Destruction, StopAllocating };
54 void zap(ZapReason reason)
55 {
56 uint32_t* cellWords = bitwise_cast<uint32_t*>(this);
57 cellWords[0] = 0;
58 // Leaving cellWords[1] alone for crash analysis if needed.
59 cellWords[2] = reason;
60 }
61 bool isZapped() const { return !*bitwise_cast<const uint32_t*>(this); }
62
63 bool isLive();
64
65 bool isPreciseAllocation() const;
66 CellContainer cellContainer() const;
67 MarkedBlock& markedBlock() const;
68 PreciseAllocation& preciseAllocation() const;
69
70 // If you want performance and you know that your cell is small, you can do this instead:
71 // ASSERT(!cell->isPreciseAllocation());
72 // cell->markedBlock().vm()
73 // We currently only use this hack for callees to make CallFrame::vm() fast. It's not
74 // recommended to use it for too many other things, since the large allocation cutoff is
75 // a runtime option and its default value is small (400 bytes).
76 Heap* heap() const;
77 VM& vm() const;
78
79 size_t cellSize() const;
80 CellAttributes cellAttributes() const;
81 DestructionMode destructionMode() const;
82 Kind cellKind() const;
83 Subspace* subspace() const;
84
85 // Call use() after the last point where you need `this` pointer to be kept alive. You usually don't
86 // need to use this, but it might be necessary if you're otherwise referring to an object's innards
87 // but not the object itself.
88 ALWAYS_INLINE void use() const
89 {
90 ensureStillAliveHere(this);
91 }
92};
93
94inline bool isJSCellKind(HeapCell::Kind kind)
95{
96 return kind == HeapCell::JSCell || kind == HeapCell::JSCellWithIndexingHeader;
97}
98
99inline bool mayHaveIndexingHeader(HeapCell::Kind kind)
100{
101 return kind == HeapCell::Auxiliary || kind == HeapCell::JSCellWithIndexingHeader;
102}
103
104} // namespace JSC
