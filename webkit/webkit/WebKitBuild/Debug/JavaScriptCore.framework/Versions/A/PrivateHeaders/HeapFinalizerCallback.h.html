27
28#include "JSHeapFinalizerPrivate.h"
29#include <wtf/PrintStream.h>
30
31namespace JSC {
32
33class VM;
34
35class HeapFinalizerCallback {
36public:
37 HeapFinalizerCallback(JSHeapFinalizer finalizer = nullptr, void *userData = nullptr)
38 : m_finalizer(finalizer)
39 , m_userData(userData)
40 {
41 }
42
43 bool operator==(const HeapFinalizerCallback& other) const
44 {
45 return m_finalizer == other.m_finalizer
46 && m_userData == other.m_userData;
47 }
48
49 bool operator!=(const HeapFinalizerCallback& other) const
50 {
51 return !(*this == other);
52 }
53
54 explicit operator bool() const
55 {
56 return *this != HeapFinalizerCallback();
57 }
58
59 void dump(PrintStream&) const;
60
61 void run(VM&) const;
62
63private:
64 JSHeapFinalizer m_finalizer;
65 void *m_userData;
66};

68} // namespace JSC
