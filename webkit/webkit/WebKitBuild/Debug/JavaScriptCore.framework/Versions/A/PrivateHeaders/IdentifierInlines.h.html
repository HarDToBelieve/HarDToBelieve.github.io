34inline Identifier::Identifier(VM& vm, AtomStringImpl* string)
35 : m_string(string)
36{
37#ifndef NDEBUG
38 checkCurrentAtomStringTable(vm);
39 if (string)
40 ASSERT_WITH_MESSAGE(!string->length() || string->isSymbol() || AtomStringImpl::isInAtomStringTable(string), "The atomic string comes from an other thread!");
41#else
42 UNUSED_PARAM(vm);
43#endif
44}
45
46inline Identifier::Identifier(VM& vm, const AtomString& string)
47 : m_string(string.string())
48{
49#ifndef NDEBUG
50 checkCurrentAtomStringTable(vm);
51 if (!string.isNull())
52 ASSERT_WITH_MESSAGE(!string.length() || string.impl()->isSymbol() || AtomStringImpl::isInAtomStringTable(string.impl()), "The atomic string comes from an other thread!");
53#else
54 UNUSED_PARAM(vm);
55#endif
56}
57
58inline Ref<AtomStringImpl> Identifier::add(VM& vm, StringImpl* r)
59{
60#ifndef NDEBUG
61 checkCurrentAtomStringTable(vm);
62#endif
63 return *AtomStringImpl::addWithStringTableProvider(vm, r);
64}

66inline Identifier Identifier::fromUid(VM& vm, UniquedStringImpl* uid)
67{
68 if (!uid || !uid->isSymbol())
69 return Identifier(vm, uid);
70 return static_cast<SymbolImpl&>(*uid);
71}

73inline Identifier Identifier::fromUid(const PrivateName& name)
74{

134inline JSValue identifierToJSValue(VM& vm, const Identifier& identifier)
135{
136 if (identifier.isSymbol())
137 return Symbol::create(vm, static_cast<SymbolImpl&>(*identifier.impl()));
138 return jsString(vm, identifier.impl());
139}
140
141inline JSValue identifierToSafePublicJSValue(VM& vm, const Identifier& identifier)
142{
143 if (identifier.isSymbol() && !identifier.isPrivateName())
144 return Symbol::create(vm, static_cast<SymbolImpl&>(*identifier.impl()));
145 return jsString(vm, identifier.impl());
146}
