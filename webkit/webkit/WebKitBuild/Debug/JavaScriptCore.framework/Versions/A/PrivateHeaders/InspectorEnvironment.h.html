43typedef JSC::JSValue (*InspectorFunctionCallHandler)(JSC::JSGlobalObject* globalObject, JSC::JSValue functionObject, const JSC::CallData& callData, JSC::JSValue thisValue, const JSC::ArgList& args, NakedPtr<JSC::Exception>& returnedException);

46class InspectorEnvironment {
47public:

49 virtual bool developerExtrasEnabled() const = 0;
50 virtual bool canAccessInspectedScriptState(JSC::JSGlobalObject*) const = 0;

53 virtual void frontendInitialized() = 0;
54 virtual WTF::Stopwatch& executionStopwatch() const = 0;
55 virtual JSC::Debugger& debugger() = 0;
56 virtual JSC::VM& vm() = 0;
57};
