28#include "JSExportMacros.h"
29#include <wtf/Vector.h>
30#include <wtf/text/WTFString.h>

32namespace Inspector {
33
34class FrontendChannel;
35
36class JS_EXPORT_PRIVATE FrontendRouter : public RefCounted<FrontendRouter> {
37public:
38 static Ref<FrontendRouter> create();
39
40 bool hasFrontends() const { return !m_connections.isEmpty(); }
41 bool hasLocalFrontend() const;
42 bool hasRemoteFrontend() const;
43
44 unsigned frontendCount() const { return m_connections.size(); }
45
46 void connectFrontend(FrontendChannel&);
47 void disconnectFrontend(FrontendChannel&);
48 void disconnectAllFrontends();
49
50 void sendEvent(const String& message) const;
51 void sendResponse(const String& message) const;
52
53private:
54 Vector<FrontendChannel*, 2> m_connections;
55};
56
57} // namespace Inspector
