34namespace Inspector {

36namespace Protocol {
37
38using ErrorString = String;
39
40template <typename T>
41using ErrorStringOr = Expected<T, ErrorString>;
42
43template<typename> struct BindingTraits;
44
45template<JSON::Value::Type type> struct PrimitiveBindingTraits {
46 static void assertValueHasExpectedType(JSON::Value* value)
47 {
48 ASSERT_UNUSED(value, value);
49 ASSERT_UNUSED(value, value->type() == type);
50 }
51};
52
53template<typename T> struct BindingTraits<JSON::ArrayOf<T>> {
54 static Ref<JSON::ArrayOf<T>> runtimeCast(Ref<JSON::Value>&& value)
55 {
56 auto array = value->asArray();
57 BindingTraits<JSON::ArrayOf<T>>::assertValueHasExpectedType(array.get());
58 COMPILE_ASSERT(sizeof(JSON::ArrayOf<T>) == sizeof(JSON::Array), type_cast_problem);
59 return static_reference_cast<JSON::ArrayOf<T>>(static_reference_cast<JSON::ArrayBase>(array.releaseNonNull()));
60 }
61
62 static void assertValueHasExpectedType(JSON::Value* value)
63 {
64 ASSERT_UNUSED(value, value);
65#if ASSERT_ENABLED
66 auto array = value->asArray();
67 ASSERT(array);
68 for (unsigned i = 0; i < array->length(); i++)
69 BindingTraits<T>::assertValueHasExpectedType(array->get(i).ptr());
70#endif
71 }
72};
73
74template<> struct BindingTraits<JSON::Value> {
75 static void assertValueHasExpectedType(JSON::Value*)
76 {
77 }
78};
79
80template<> struct BindingTraits<JSON::Array> : PrimitiveBindingTraits<JSON::Value::Type::Array> { };
81template<> struct BindingTraits<JSON::Object> : PrimitiveBindingTraits<JSON::Value::Type::Object> { };
82template<> struct BindingTraits<String> : PrimitiveBindingTraits<JSON::Value::Type::String> { };
83template<> struct BindingTraits<bool> : PrimitiveBindingTraits<JSON::Value::Type::Boolean> { };
84template<> struct BindingTraits<double> : PrimitiveBindingTraits<JSON::Value::Type::Double> { };
85template<> struct BindingTraits<int> : PrimitiveBindingTraits<JSON::Value::Type::Integer> { };
