30ALWAYS_INLINE void* IsoSubspace::allocateNonVirtual(VM& vm, size_t size, GCDeferralContext* deferralContext, AllocationFailureMode failureMode)
31{
32 RELEASE_ASSERT(WTF::roundUpToMultipleOf<MarkedBlock::atomSize>(size) == cellSize());
33 Allocator allocator = allocatorForNonVirtual(size, AllocatorForMode::MustAlreadyHaveAllocator);
34 void* result = allocator.allocate(vm.heap, deferralContext, failureMode);
35 return result;
36}
37
38inline void IsoSubspace::clearIsoCellSetBit(PreciseAllocation* preciseAllocation)
39{
40 unsigned lowerTierIndex = preciseAllocation->lowerTierIndex();
41 m_cellSets.forEach(
42 [&](IsoCellSet* set) {
43 set->clearLowerTierCell(lowerTierIndex);
44 });
45}
46
47inline void IsoSubspace::sweep()
48{
49 Subspace::sweepBlocks();
50 // We sweep precise-allocations eagerly, but we do not free it immediately.
51 // This part should be done by MarkedSpace::sweepPreciseAllocations.
52 m_preciseAllocations.forEach([&](PreciseAllocation* allocation) {
53 allocation->sweep();
54 });
55}

57template<typename Func>
58void IsoSubspace::forEachLowerTierFreeListedPreciseAllocation(const Func& func)
59{
60 m_lowerTierFreeList.forEach(func);
61}
