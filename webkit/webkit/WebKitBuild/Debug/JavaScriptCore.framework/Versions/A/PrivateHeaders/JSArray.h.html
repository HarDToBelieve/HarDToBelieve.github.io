69 // tryCreateUninitializedRestricted is used for fast construction of arrays whose size and
70 // contents are known at time of creation. This is a restricted API for careful use only in
71 // performance critical code paths. If you don't have a good reason to use it, you probably
72 // shouldn't use it. Instead, you should go with
73 // - JSArray::tryCreate() or JSArray::create() instead of tryCreateUninitializedRestricted(), and
74 // - putDirectIndex() instead of initializeIndex().
75 //
76 // Clients of this interface must:
77 // - null-check the result (indicating out of memory, or otherwise unable to allocate vector).
78 // - call 'initializeIndex' for all properties in sequence, for 0 <= i < initialLength.
79 // - Provide a valid GCDefferalContext* if they might garbage collect when initializing properties,
80 // otherwise the caller can provide a null GCDefferalContext*.
81 // - Provide a local stack instance of ObjectInitializationScope at the call site.

115 enum ShiftCountMode {
116 // This form of shift hints that we're doing queueing. With this assumption in hand,
117 // we convert to ArrayStorage, which has queue optimizations.
118 ShiftCountForShift,
119
120 // This form of shift hints that we're just doing care and feeding on an array that
121 // is probably typically used for ordinary accesses. With this assumption in hand,
122 // we try to preserve whatever indexing type it has already.
123 ShiftCountForSplice
124 };
125
126 bool shiftCountForShift(JSGlobalObject* globalObject, unsigned startIndex, unsigned count)
127 {
128 VM& vm = getVM(globalObject);
129 return shiftCountWithArrayStorage(vm, startIndex, count, ensureArrayStorage(vm));
130 }
131 bool shiftCountForSplice(JSGlobalObject* globalObject, unsigned& startIndex, unsigned count)
132 {
133 return shiftCountWithAnyIndexingType(globalObject, startIndex, count);
134 }
135 template<ShiftCountMode shiftCountMode>
136 bool shiftCount(JSGlobalObject* globalObject, unsigned& startIndex, unsigned count)
137 {
138 switch (shiftCountMode) {
139 case ShiftCountForShift:
140 return shiftCountForShift(globalObject, startIndex, count);
141 case ShiftCountForSplice:
142 return shiftCountForSplice(globalObject, startIndex, count);
143 default:
144 CRASH();
145 return false;
146 }
147 }
148
149 bool unshiftCountForShift(JSGlobalObject* globalObject, unsigned startIndex, unsigned count)
150 {
151 return unshiftCountWithArrayStorage(globalObject, startIndex, count, ensureArrayStorage(getVM(globalObject)));
152 }
153 bool unshiftCountForSplice(JSGlobalObject* globalObject, unsigned startIndex, unsigned count)
154 {
155 return unshiftCountWithAnyIndexingType(globalObject, startIndex, count);
156 }
157 template<ShiftCountMode shiftCountMode>
158 bool unshiftCount(JSGlobalObject* globalObject, unsigned startIndex, unsigned count)
159 {
160 switch (shiftCountMode) {
161 case ShiftCountForShift:
162 return unshiftCountForShift(globalObject, startIndex, count);
163 case ShiftCountForSplice:
164 return unshiftCountForSplice(globalObject, startIndex, count);
165 default:
166 CRASH();
167 return false;
168 }
169 }

215inline Butterfly* tryCreateArrayButterfly(VM& vm, JSObject* intendedOwner, unsigned initialLength)
216{
217 Butterfly* butterfly = Butterfly::tryCreate(
218 vm, intendedOwner, 0, 0, true, baseIndexingHeaderForArrayStorage(initialLength),
219 ArrayStorage::sizeFor(BASE_ARRAY_STORAGE_VECTOR_LEN));
220 if (!butterfly)
221 return nullptr;
222 ArrayStorage* storage = butterfly->arrayStorage();
223 storage->m_sparseMap.clear();
224 storage->m_indexBias = 0;
225 storage->m_numValuesInVector = 0;
226 return butterfly;
227}

229inline JSArray* JSArray::tryCreate(VM& vm, Structure* structure, unsigned initialLength, unsigned vectorLengthHint)
230{
231 ASSERT(vectorLengthHint >= initialLength);
232 unsigned outOfLineStorage = structure->outOfLineCapacity();
233
234 Butterfly* butterfly;
235 IndexingType indexingType = structure->indexingType();
236 if (LIKELY(!hasAnyArrayStorage(indexingType))) {
237 ASSERT(
238 hasUndecided(indexingType)
239 || hasInt32(indexingType)
240 || hasDouble(indexingType)
241 || hasContiguous(indexingType));
242
243 if (UNLIKELY(vectorLengthHint > MAX_STORAGE_VECTOR_LENGTH))
244 return nullptr;
245
246 unsigned vectorLength = Butterfly::optimalContiguousVectorLength(structure, vectorLengthHint);
247 void* temp = vm.jsValueGigacageAuxiliarySpace.allocateNonVirtual(
248 vm,
249 Butterfly::totalSize(0, outOfLineStorage, true, vectorLength * sizeof(EncodedJSValue)),
250 nullptr, AllocationFailureMode::ReturnNull);
251 if (!temp)
252 return nullptr;
253 butterfly = Butterfly::fromBase(temp, 0, outOfLineStorage);
254 butterfly->setVectorLength(vectorLength);
255 butterfly->setPublicLength(initialLength);
256 if (hasDouble(indexingType))
257 clearArray(butterfly->contiguousDouble().data(), vectorLength);
258 else
259 clearArray(butterfly->contiguous().data(), vectorLength);
260 } else {
261 ASSERT(
262 indexingType == ArrayWithSlowPutArrayStorage
263 || indexingType == ArrayWithArrayStorage);
264 butterfly = tryCreateArrayButterfly(vm, nullptr, initialLength);
265 if (!butterfly)
266 return nullptr;
267 for (unsigned i = 0; i < BASE_ARRAY_STORAGE_VECTOR_LEN; ++i)
268 butterfly->arrayStorage()->m_vector[i].clear();
269 }
270
271 return createWithButterfly(vm, nullptr, structure, butterfly);
272}
