43struct Config {
44 static Config& singleton();
45
46 JS_EXPORT_PRIVATE static void disableFreezingForTesting();
47 JS_EXPORT_PRIVATE static void enableRestrictedOptions();
48 static void permanentlyFreeze() { WTF::Config::permanentlyFreeze(); }
49
50 static void configureForTesting()
51 {
52 WTF::setPermissionsOfConfigPage();
53 disableFreezingForTesting();
54 enableRestrictedOptions();
55 }
56
57 bool isPermanentlyFrozen() { return g_wtfConfig.isPermanentlyFrozen; }
58
59 // All the fields in this struct should be chosen such that their
60 // initial value is 0 / null / falsy because Config is instantiated
61 // as a global singleton.
62
63 bool disabledFreezingForTesting;
64 bool restrictedOptionsEnabled;
65 bool jitDisabled;
66
67 bool useFastJITPermissions;
68
69 // The following HasBeenCalled flags are for auditing call_once initialization functions.
70 bool initializeHasBeenCalled;
71
72 struct {
73#if ASSERT_ENABLED
74 bool canUseJITIsSet;
75#endif
76 bool canUseJIT;
77 } vm;
78
79 ExecutableAllocator* executableAllocator;
80 FixedVMPoolExecutableAllocator* fixedVMPoolExecutableAllocator;
81 void* startExecutableMemory;
82 void* endExecutableMemory;
83 uintptr_t startOfFixedWritableMemoryPool;
84
85#if ENABLE(SEPARATED_WX_HEAP)
86 JITWriteSeparateHeapsFunction jitWriteSeparateHeaps;
87#endif
88
89 OptionsStorage options;
90
91 void (*shellTimeoutCheckCallback)(VM&);
92
93 struct {
94 uint8_t exceptionInstructions[maxOpcodeLength + 1];
95 uint8_t wasmExceptionInstructions[maxOpcodeLength + 1];
96 const void* gateMap[numberOfGates];
97 } llint;
98
99#if CPU(ARM64E) && ENABLE(PTRTAG_DEBUGGING)
100 WTF::PtrTagLookup ptrTagLookupRecord;
101#endif
102};
