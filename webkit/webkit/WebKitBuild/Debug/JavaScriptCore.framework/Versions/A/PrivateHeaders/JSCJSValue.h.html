366#if USE(JSVALUE32_64)
367 /*
368 * On 32-bit platforms USE(JSVALUE32_64) should be defined, and we use a NaN-encoded
369 * form for immediates.
370 *
371 * The encoding makes use of unused NaN space in the IEEE754 representation. Any value
372 * with the top 13 bits set represents a QNaN (with the sign bit set). QNaN values
373 * can encode a 51-bit payload. Hardware produced and C-library payloads typically
374 * have a payload of zero. We assume that non-zero payloads are available to encode
375 * pointer and integer values. Since any 64-bit bit pattern where the top 15 bits are
376 * all set represents a NaN with a non-zero payload, we can use this space in the NaN
377 * ranges to encode other values (however there are also other ranges of NaN space that
378 * could have been selected).
379 *
380 * For JSValues that do not contain a double value, the high 32 bits contain the tag
381 * values listed in the enums below, which all correspond to NaN-space. In the case of
382 * cell, integer and bool values the lower 32 bits (the 'payload') contain the pointer
383 * integer or boolean value; in the case of all other tags the payload is 0.
384 */
385 uint32_t tag() const;
386 int32_t payload() const;
387
388 // This should only be used by the LLInt C Loop interpreter and OSRExit code who needs
389 // synthesize JSValue from its "register"s holding tag and payload values.
390 explicit JSValue(int32_t tag, int32_t payload);
391
392#elif USE(JSVALUE64)
393 /*
394 * On 64-bit platforms USE(JSVALUE64) should be defined, and we use a NaN-encoded
395 * form for immediates.
396 *
397 * The encoding makes use of unused NaN space in the IEEE754 representation. Any value
398 * with the top 13 bits set represents a QNaN (with the sign bit set). QNaN values
399 * can encode a 51-bit payload. Hardware produced and C-library payloads typically
400 * have a payload of zero. We assume that non-zero payloads are available to encode
401 * pointer and integer values. Since any 64-bit bit pattern where the top 15 bits are
402 * all set represents a NaN with a non-zero payload, we can use this space in the NaN
403 * ranges to encode other values (however there are also other ranges of NaN space that
404 * could have been selected).
405 *
406 * This range of NaN space is represented by 64-bit numbers begining with the 15-bit
407 * hex patterns 0xFFFC and 0xFFFE - we rely on the fact that no valid double-precision
408 * numbers will fall in these ranges.
409 *
410 * The top 15-bits denote the type of the encoded JSValue:
411 *
412 * Pointer { 0000:PPPP:PPPP:PPPP
413 * / 0002:****:****:****
414 * Double { ...
415 * \ FFFC:****:****:****
416 * Integer { FFFE:0000:IIII:IIII
417 *
418 * The scheme we have implemented encodes double precision values by performing a
419 * 64-bit integer addition of the value 2^49 to the number. After this manipulation
420 * no encoded double-precision value will begin with the pattern 0x0000 or 0xFFFE.
421 * Values must be decoded by reversing this operation before subsequent floating point
422 * operations may be peformed.
423 *
424 * 32-bit signed integers are marked with the 16-bit tag 0xFFFE.
425 *
426 * The tag 0x0000 denotes a pointer, or another form of tagged immediate. Boolean,
427 * null and undefined values are represented by specific, invalid pointer values:
428 *
429 * False: 0x06
430 * True: 0x07
431 * Undefined: 0x0a
432 * Null: 0x02
433 *
434 * These values have the following properties:
435 * - Bit 1 (OtherTag) is set for all four values, allowing real pointers to be
436 * quickly distinguished from all immediate values, including these invalid pointers.
437 * - With bit 3 masked out (UndefinedTag), Undefined and Null share the
438 * same value, allowing null & undefined to be quickly detected.
439 *
440 * No valid JSValue will have the bit pattern 0x0, this is used to represent array
441 * holes, and as a C++ 'no value' result (e.g. JSValue() has an internal value of 0).
442 *
443 * When USE(BIGINT32), we have a special representation for BigInts that are small (32-bit at most):
444 * 0000:XXXX:XXXX:0012
445 * This representation works because of the following things:
446 * - It cannot be confused with a Double or Integer thanks to the top bits
447 * - It cannot be confused with a pointer to a Cell, thanks to bit 1 which is set to true
448 * - It cannot be confused with a pointer to wasm thanks to bit 0 which is set to false
449 * - It cannot be confused with true/false because bit 2 is set to false
450 * - It cannot be confused for null/undefined because bit 4 is set to true
451 */
452
453 // This value is 2^49, used to encode doubles such that the encoded value will begin
454 // with a 15-bit pattern within the range 0x0002..0xFFFC.
455 static constexpr size_t DoubleEncodeOffsetBit = 49;
456 static constexpr int64_t DoubleEncodeOffset = 1ll << DoubleEncodeOffsetBit;
457 // If all bits in the mask are set, this indicates an integer number,
458 // if any but not all are set this value is a double precision number.
459 static constexpr int64_t NumberTag = 0xfffe000000000000ll;
460 // The following constant is used for a trick in the implementation of strictEq, to detect if either of the arguments is a double
461 static constexpr int64_t LowestOfHighBits = 1ULL << 49;
462 static_assert(LowestOfHighBits & NumberTag);
463 static_assert(!((LowestOfHighBits>>1) & NumberTag));
464
465 // All non-numeric (bool, null, undefined) immediates have bit 2 set.
466 static constexpr int32_t OtherTag = 0x2;
467 static constexpr int32_t BoolTag = 0x4;
468 static constexpr int32_t UndefinedTag = 0x8;
469#if USE(BIGINT32)
470 static constexpr int32_t BigInt32Tag = 0x12;
471 static constexpr int64_t BigInt32Mask = NumberTag | BigInt32Tag;
472#endif
473 // Combined integer value for non-numeric immediates.
474 static constexpr int32_t ValueFalse = OtherTag | BoolTag | false;
475 static constexpr int32_t ValueTrue = OtherTag | BoolTag | true;
476 static constexpr int32_t ValueUndefined = OtherTag | UndefinedTag;
477 static constexpr int32_t ValueNull = OtherTag;
478
479 static constexpr int64_t MiscTag = OtherTag | BoolTag | UndefinedTag;
480
481 // NotCellMask is used to check for all types of immediate values (either number or 'other').
482 static constexpr int64_t NotCellMask = NumberTag | OtherTag;
483
484 // These special values are never visible to JavaScript code; Empty is used to represent
485 // Array holes, and for uninitialized JSValues. Deleted is used in hash table code.
486 // These values would map to cell types in the JSValue encoding, but not valid GC cell
487 // pointer should have either of these values (Empty is null, deleted is at an invalid
488 // alignment for a GC cell, and in the zero page).
489 static constexpr int32_t ValueEmpty = 0x0;
490 static constexpr int32_t ValueDeleted = 0x4;
491
492 static constexpr int64_t WasmTag = OtherTag | 0x1;
493 static constexpr int64_t WasmMask = NumberTag | 0x7;
494 // We tag Wasm non-JSCell pointers with a 3 at the bottom. We can test if a 64-bit JSValue pattern
495 // is a Wasm callee by masking the upper 16 bits and the lower 3 bits, and seeing if
496 // the resulting value is 3. The full test is: x & WasmMask == WasmTag
497 // This works because the lower 3 bits of the non-number immediate values are as follows:
498 // undefined: 0b010
499 // null: 0b010
500 // true: 0b111
501 // false: 0b110
502 // The test rejects all of these because none have just the value 3 in their lower 3 bits.
503 // The test rejects all numbers because they have non-zero upper 16 bits.
504 // The test also rejects normal cells because they won't have the number 3 as
505 // their lower 3 bits. Note, this bit pattern also allows the normal JSValue isCell(), etc,
506 // predicates to work on a Wasm::Callee because the various tests will fail if you
507 // bit casted a boxed Wasm::Callee* to a JSValue. isCell() would fail since it sees
508 // OtherTag. The other tests also trivially fail, since it won't be a number,
509 // and it won't be equal to null, undefined, true, or false. The isBoolean() predicate
510 // will fail because we won't have BoolTag set.
511#endif
