55{
56 // The only difference between toInt32 and toUint32 is that toUint32 reinterprets resulted int32_t value as uint32_t.

61inline uint32_t JSValue::toIndex(JSGlobalObject* globalObject, const char* errorName) const
62{
63 VM& vm = getVM(globalObject);
64 auto scope = DECLARE_THROW_SCOPE(vm);
65
66 double d = toNumber(globalObject);
67 RETURN_IF_EXCEPTION(scope, 0);
68 if (d <= -1) {
69 throwException(globalObject, scope, createRangeError(globalObject, makeString(errorName, " cannot be negative")));
70 return 0;
71 }
72 if (d > std::numeric_limits<unsigned>::max()) {
73 throwException(globalObject, scope, createRangeError(globalObject, makeString(errorName, " too large")));
74 return 0;
75 }
76
77 if (isInt32())
78 return asInt32();
79 RELEASE_AND_RETURN(scope, JSC::toInt32(d));
80}

82// https://tc39.es/ecma262/#sec-tointegerorinfinity
83inline double JSValue::toIntegerOrInfinity(JSGlobalObject* globalObject) const
84{
85 if (isInt32())
86 return asInt32();
87 double d = toNumber(globalObject);
88 if (std::isnan(d) || !d)
89 return 0.0;
90 return trunc(d);
91}

110inline std::optional<uint32_t> JSValue::tryGetAsUint32Index()
111{
112 if (isUInt32()) {
113 ASSERT(isIndex(asUInt32()));
114 return asUInt32();
115 }
116 if (isNumber()) {
117 double number = asNumber();
118 uint32_t asUint = static_cast<uint32_t>(number);
119 if (static_cast<double>(asUint) == number && isIndex(asUint))
120 return asUint;
121 }
122 return std::nullopt;
123}

125inline std::optional<int32_t> JSValue::tryGetAsInt32()
126{
127 if (isInt32())
128 return asInt32();
129 if (isNumber()) {
130 double number = asNumber();
131 int32_t asInt = static_cast<int32_t>(number);
132 if (static_cast<double>(asInt) == number)
133 return asInt;
134 }
135 return std::nullopt;
136}

589inline int64_t tryConvertToInt52(double number)
590{
591 if (number != number)
592 return JSValue::notInt52;
593#if OS(WINDOWS) && CPU(X86)
594 // The VS Compiler for 32-bit builds generates a floating point error when attempting to cast
595 // from an infinity to a 64-bit integer. We leave this routine with the floating point error
596 // left in a register, causing undefined behavior in later floating point operations.
597 //
598 // To avoid this issue, we check for infinity here, and return false in that case.
599 if (std::isinf(number))
600 return JSValue::notInt52;
601#endif
602 int64_t asInt64 = static_cast<int64_t>(number);
603 if (asInt64 != number)
604 return JSValue::notInt52;
605 if (!asInt64 && std::signbit(number))
606 return JSValue::notInt52;
607 if (asInt64 >= (static_cast<int64_t>(1) << (JSValue::numberOfInt52Bits - 1)))
608 return JSValue::notInt52;
609 if (asInt64 < -(static_cast<int64_t>(1) << (JSValue::numberOfInt52Bits - 1)))
610 return JSValue::notInt52;
611 return asInt64;
612}

760ALWAYS_INLINE Identifier JSValue::toPropertyKey(JSGlobalObject* globalObject) const
761{
762 VM& vm = getVM(globalObject);
763 auto scope = DECLARE_THROW_SCOPE(vm);
764
765 if (isString())
766 RELEASE_AND_RETURN(scope, asString(*this)->toIdentifier(globalObject));
767
768 JSValue primitive = toPrimitive(globalObject, PreferString);
769 RETURN_IF_EXCEPTION(scope, vm.propertyNames->emptyIdentifier);
770 if (primitive.isSymbol())
771 RELEASE_AND_RETURN(scope, Identifier::fromUid(asSymbol(primitive)->privateName()));
772
773 auto string = primitive.toString(globalObject);
774 RETURN_IF_EXCEPTION(scope, { });
775 RELEASE_AND_RETURN(scope, string->toIdentifier(globalObject));
776}

799inline PreferredPrimitiveType toPreferredPrimitiveType(JSGlobalObject* globalObject, JSValue value)
800{
801 VM& vm = getVM(globalObject);
802 auto scope = DECLARE_THROW_SCOPE(vm);
803
804 if (!value.isString()) {
805 throwTypeError(globalObject, scope, "Primitive hint is not a string."_s);
806 return NoPreference;
807 }
808
809 StringImpl* hintString = asString(value)->value(globalObject).impl();
810 RETURN_IF_EXCEPTION(scope, NoPreference);
811
812 if (WTF::equal(hintString, "default"))
813 return NoPreference;
814 if (WTF::equal(hintString, "number"))
815 return PreferNumber;
816 if (WTF::equal(hintString, "string"))
817 return PreferString;
818
819 throwTypeError(globalObject, scope, "Expected primitive hint to match one of 'default', 'number', 'string'."_s);
820 return NoPreference;
821}

832ALWAYS_INLINE JSValue JSValue::toNumeric(JSGlobalObject* globalObject) const
833{
834 VM& vm = globalObject->vm();
835 auto scope = DECLARE_THROW_SCOPE(vm);
836
837 if (isInt32() || isDouble() || isBigInt())
838 return *this;
839
840 JSValue primValue = this->toPrimitive(globalObject, PreferNumber);
841 RETURN_IF_EXCEPTION(scope, { });
842
843 if (primValue.isDouble() || primValue.isBigInt())

1107ALWAYS_INLINE bool JSValue::equalSlowCaseInline(JSGlobalObject* globalObject, JSValue v1, JSValue v2)
1108{
1109 VM& vm = getVM(globalObject);
1110 auto scope = DECLARE_THROW_SCOPE(vm);
1111 do {
1112 if (v1.isNumber()) {
1113 if (v2.isNumber())
1114 return v1.asNumber() == v2.asNumber();
1115 // Guaranteeing that if we have a number it is v2 makes some of the cases below simpler.
1116 std::swap(v1, v2);
1117 }
1118
1119 // This deals with Booleans, BigInt32, Objects, and is a shortcut for a few more types.
1120 // It has to come here and not before, because it is NOT true that NaN == NaN
1121 if (v1 == v2)
1122 return true;
1123
1124 if (v1.isUndefinedOrNull()) {
1125 if (v2.isUndefinedOrNull())
1126 return true;
1127 if (!v2.isCell())
1128 return false;
1129 return v2.asCell()->structure(vm)->masqueradesAsUndefined(globalObject);
1130 }
1131
1132 if (v2.isUndefinedOrNull()) {
1133 if (!v1.isCell())
1134 return false;
1135 return v1.asCell()->structure(vm)->masqueradesAsUndefined(globalObject);
1136 }
1137
1138 if (v1.isObject()) {
1139 if (v2.isObject())
1140 return false; // v1 == v2 is already dealt with previously
1141 JSValue p1 = v1.toPrimitive(globalObject);
1142 RETURN_IF_EXCEPTION(scope, false);
1143 v1 = p1;
1144 if (v1.isInt32() && v2.isInt32())
1145 return v1 == v2;
1146 continue;
1147 }
1148
1149 if (v2.isObject()) {
1150 JSValue p2 = v2.toPrimitive(globalObject);
1151 RETURN_IF_EXCEPTION(scope, false);
1152 v2 = p2;
1153 if (v1.isInt32() && v2.isInt32())
1154 return v1 == v2;
1155 continue;
1156 }
1157
1158 bool sym1 = v1.isSymbol();
1159 bool sym2 = v2.isSymbol();
1160 if (sym1 || sym2) {
1161 if (sym1 && sym2)
1162 return false; // v1 == v2 is already dealt with previously
1163 return false;
1164 }
1165
1166 bool s1 = v1.isString();
1167 bool s2 = v2.isString();
1168 if (s1) {
1169 if (s2)
1170 RELEASE_AND_RETURN(scope, asString(v1)->equal(globalObject, asString(v2)));
1171 std::swap(v1, v2);
1172 // We are guaranteed to enter the next case, so losing the invariant of only v2 being a number is fine
1173 }
1174 if (s1 || s2) {
1175 // We are guaranteed that the string is v2 (thanks to the swap above)
1176 if (v1.isBigInt()) {
1177 String v2String = asString(v2)->value(globalObject);
1178 RETURN_IF_EXCEPTION(scope, false);
1179 v2 = JSBigInt::stringToBigInt(globalObject, v2String);
1180 RETURN_IF_EXCEPTION(scope, false);
1181 if (!v2)
1182 return false;
1183 if (v1 == v2)
1184 return true; // For BigInt32
1185 // We fallthrough to the generic code for comparing BigInts (which is only missing the BigInt32/BigInt32 case, hence the check above)
1186 } else {
1187 ASSERT(v1.isNumber() || v1.isBoolean());
1188 double d1 = v1.toNumber(globalObject);
1189 RETURN_IF_EXCEPTION(scope, false);
1190 double d2 = v2.toNumber(globalObject);
1191 RETURN_IF_EXCEPTION(scope, false);
1192 return d1 == d2;
1193 }
1194 }
1195
1196 if (v1.isBoolean()) {
1197 if (v2.isNumber())
1198 return static_cast<double>(v1.asBoolean()) == v2.asNumber();
1199 v1 = JSValue(v1.toNumber(globalObject));
1200 RETURN_IF_EXCEPTION(scope, false);
1201 // We fallthrough to the BigInt/Number comparison below
1202 // We just need one more swap to repair the rule that only v2 is allowed to be a number in these comparisons
1203 std::swap(v1, v2);
1204 } else if (v2.isBoolean()) {
1205 v2 = JSValue(v2.toNumber(globalObject));
1206 RETURN_IF_EXCEPTION(scope, false);
1207 // We fallthrough to the BigInt/Number comparison below
1208 }
1209
1210#if USE(BIGINT32)
1211 if (v1.isBigInt32()) {
1212 if (v2.isInt32())
1213 return v1.bigInt32AsInt32() == v2.asInt32();
1214 if (v2.isDouble())
1215 return static_cast<double>(v1.bigInt32AsInt32()) == v2.asDouble();
1216 std::swap(v1, v2);
1217 }
1218#endif // USE(BIGINT32)
1219
1220 if (v1.isHeapBigInt()) {
1221 if (v2.isHeapBigInt())
1222 return JSBigInt::equals(v1.asHeapBigInt(), v2.asHeapBigInt());
1223#if USE(BIGINT32)
1224 if (v2.isBigInt32())
1225 return v1.asHeapBigInt()->equalsToInt32(v2.bigInt32AsInt32());
1226#endif // USE(BIGINT32)
1227 if (v2.isNumber())
1228 return v1.asHeapBigInt()->equalsToNumber(v2);
1229 }
1230
1231 return false;
1232 } while (true);
1233}

1274{

1345ALWAYS_INLINE bool sameValue(JSGlobalObject* globalObject, JSValue a, JSValue b)
1346{
