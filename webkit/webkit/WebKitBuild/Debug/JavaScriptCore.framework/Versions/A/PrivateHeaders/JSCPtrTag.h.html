82#define JSC_DECLARE_PTRTAG(name, type, target) WTF_DECLARE_PTRTAG(name)
83
84#define JSC_DECLARE_PTRTAG_TRAIT(passedTag, passedCalleeType, passedCallerType) \
85 template<> struct PtrTagTraits<JSC::passedTag> { \
86 static constexpr PtrTag tag = JSC::passedTag; \
87 static constexpr PtrTagCalleeType calleeType = passedCalleeType; \
88 static constexpr PtrTagCallerType callerType = passedCallerType; \
89 static constexpr bool isSpecialized = passedCallerType == PtrTagCallerType::JIT;\
90 \
91 template<typename PtrType> \
92 ALWAYS_INLINE static PtrType tagCodePtr(PtrType ptr) \
93 { \
94 if constexpr (!isSpecialized) \
95 return WTF::tagNativeCodePtrImpl<tag>(ptr); \
96 else \
97 return JSC::tagJSCCodePtrImpl<tag, calleeType, callerType>(ptr); \
98 } \
99 \
100 template<typename PtrType> \
101 ALWAYS_INLINE static PtrType untagCodePtr(PtrType ptr) \
102 { \
103 if constexpr (!isSpecialized) \
104 return WTF::untagNativeCodePtrImpl<tag>(ptr); \
105 else \
106 return JSC::untagJSCCodePtrImpl<tag, calleeType, callerType>(ptr); \
107 } \
108 };

124template<PtrTag tag, PtrTagCalleeType calleeType, PtrTagCallerType callerType, typename PtrType>
125ALWAYS_INLINE static PtrType tagJSCCodePtrImpl(PtrType ptr)
126{
127 static_assert(callerType == PtrTagCallerType::JIT);
128 if constexpr (calleeType == PtrTagCalleeType::Native) {
129 static_assert(tag == OperationPtrTag);
130 JITOperationList::instance().assertIsJITOperation(ptr);
131#if ENABLE(JIT_CAGE)
132 if (Options::useJITCage())
133 return bitwise_cast<PtrType>(JITOperationList::instance().map(bitwise_cast<void*>(ptr)));
134 } else {
135 if (Options::useJITCage())
136 return bitwise_cast<PtrType>(jitCagePtr(bitwise_cast<void*>(ptr), tag));
137#endif // ENABLE(JIT_CAGE)
138 }
139 return WTF::tagNativeCodePtrImpl<tag>(ptr);
140}
141
142template<PtrTag tag, PtrTagCalleeType calleeType, PtrTagCallerType callerType, typename PtrType>
143ALWAYS_INLINE static PtrType untagJSCCodePtrImpl(PtrType ptr)
144{
145 static_assert(callerType == PtrTagCallerType::JIT);
146 if constexpr (calleeType == PtrTagCalleeType::Native) {
147 static_assert(tag == OperationPtrTag);
148 JITOperationList::instance().assertIsJITOperation(ptr);
149#if ENABLE(JIT_CAGE)
150 if (Options::useJITCage()) {
151 RELEASE_ASSERT(bitwise_cast<PtrType>(JITOperationList::instance().map(bitwise_cast<void*>(ptr))) == ptr);
152 return removeCodePtrTag(ptr);
153 }
154 } else {
155 if (Options::useJITCage()) {
156 PtrType removed = removeCodePtrTag(ptr);
157 RELEASE_ASSERT((tagJSCCodePtrImpl<tag, calleeType, callerType>(removed)) == ptr);
158 return removed;
159 }
160#endif // ENABLE(JIT_CAGE)
161 }
162 return WTF::untagNativeCodePtrImpl<tag>(ptr);
163}

165template<typename PtrType>
166inline PtrType tagCodePtrWithStackPointerForJITCall(PtrType ptr, const void* stackPointer)
167{
168 ASSERT(ptr);
169 UNUSED_PARAM(stackPointer);
170#if ENABLE(JIT_CAGE)
171 if (Options::useJITCage())
172 return bitwise_cast<PtrType>(JSC_JIT_CAGE(bitwise_cast<void*>(ptr), bitwise_cast<uintptr_t>(stackPointer)));
173#endif
174#if CPU(ARM64E)
175 return ptrauth_sign_unauthenticated(ptr, ptrauth_key_process_dependent_code, stackPointer);
176#else
177 return ptr;
178#endif
179}
180
181template<typename PtrType>
182inline PtrType untagCodePtrWithStackPointerForJITCall(PtrType ptr, const void* stackPointer)
183{
184 ASSERT(ptr);
185 UNUSED_PARAM(stackPointer);
186#if ENABLE(JIT_CAGE)
187 if (Options::useJITCage()) {
188 PtrType uncaged = removeCodePtrTag(ptr);
189 RELEASE_ASSERT(tagCodePtrWithStackPointerForJITCall(uncaged, stackPointer) == ptr);
190 return uncaged;
191 }
192#endif
193#if CPU(ARM64E)
194 return __builtin_ptrauth_auth(ptr, ptrauth_key_process_dependent_code, stackPointer);
195#else
196 return ptr;
197#endif
198}
199
200template<PtrTag tag, typename PtrType>
201inline PtrType untagAddressDiversifiedCodePtr(PtrType ptr, const void* ptrAddress)
202{
203 UNUSED_PARAM(ptrAddress);
204#if CPU(ARM64E)
205 uint64_t address = bitwise_cast<uint64_t>(ptrAddress);
206 uint64_t tagBits = static_cast<uint64_t>(tag) << 48;
207 uint64_t addressDiversifiedTag = tagBits ^ address;
208 return __builtin_ptrauth_auth(ptr, ptrauth_key_process_dependent_code, addressDiversifiedTag);
209#else
210 return ptr;
211#endif
212}
