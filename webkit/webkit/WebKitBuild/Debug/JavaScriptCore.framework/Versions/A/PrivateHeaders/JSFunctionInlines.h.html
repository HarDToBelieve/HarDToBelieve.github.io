93inline bool isHostFunction(JSValue value, TaggedNativeFunction nativeFunction)
94{
95 JSFunction* function = jsCast<JSFunction*>(getJSFunction(value));
96 if (!function || !function->isHostFunction())
97 return false;
98 return function->nativeFunction() == nativeFunction;
99}

115inline bool JSFunction::canAssumeNameAndLengthAreOriginal(VM&)
116{
117 if (isHostFunction())
118 return false;
119 FunctionRareData* rareData = this->rareData();
120 if (!rareData)
121 return true;
122 if (rareData->hasModifiedNameForNonHostFunction())
123 return false;
124 if (rareData->hasModifiedLengthForNonHostFunction())
125 return false;
126 return true;
127}

129inline bool JSFunction::canUseAllocationProfile()
130{
131 if (isHostOrBuiltinFunction()) {
132 if (isHostFunction())
133 return false;
134
135 VM& vm = globalObject()->vm();
136 unsigned attributes;
137 JSValue prototype = getDirect(vm, vm.propertyNames->prototype, attributes);
138 if (!prototype || (attributes & PropertyAttribute::AccessorOrCustomAccessorOrValue))
139 return false;
140 }
141
142 // If we don't have a prototype property, we're not guaranteed it's
143 // non-configurable. For example, user code can define the prototype
144 // as a getter. JS semantics require that the getter is called every
145 // time |construct| occurs with this function as new.target.
146 return jsExecutable()->hasPrototypeProperty();
147}
148
149inline FunctionRareData* JSFunction::ensureRareDataAndAllocationProfile(JSGlobalObject* globalObject, unsigned inlineCapacity)
150{
151 ASSERT(canUseAllocationProfile());
152 FunctionRareData* rareData = this->rareData();
153 if (!rareData)
154 return allocateAndInitializeRareData(globalObject, inlineCapacity);
155 if (UNLIKELY(!rareData->isObjectAllocationProfileInitialized()))
156 return initializeRareData(globalObject, inlineCapacity);
157 return rareData;
158}

160inline JSString* JSFunction::asStringConcurrently(VM& vm) const
161{
162 if (inherits<JSBoundFunction>(vm))
163 return nullptr;
164 if (isHostFunction())
165 return static_cast<NativeExecutable*>(executable())->asStringConcurrently();
166 return jsExecutable()->asStringConcurrently();
167}
