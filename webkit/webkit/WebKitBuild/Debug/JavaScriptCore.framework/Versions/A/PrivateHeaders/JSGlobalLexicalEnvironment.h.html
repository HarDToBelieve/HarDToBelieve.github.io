26#pragma once
27
28#include "JSSegmentedVariableObject.h"
29
30namespace JSC {
31
32class JSGlobalLexicalEnvironment final : public JSSegmentedVariableObject {
33public:
34 using Base = JSSegmentedVariableObject;
35
36 static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnPropertySlot | OverridesPut;
37
38 template<typename CellType, SubspaceAccess mode>
39 static IsoSubspace* subspaceFor(VM& vm)
40 {
41 return &vm.globalLexicalEnvironmentSpace;
42 }
43
44 static JSGlobalLexicalEnvironment* create(VM& vm, Structure* structure, JSScope* parentScope)
45 {
46 JSGlobalLexicalEnvironment* result =
47 new (NotNull, allocateCell<JSGlobalLexicalEnvironment>(vm.heap)) JSGlobalLexicalEnvironment(vm, structure, parentScope);
48 result->finishCreation(vm);
49 result->symbolTable()->setScopeType(SymbolTable::ScopeType::GlobalLexicalScope);
50 return result;
51 }
52
53 static bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&);
54 static bool put(JSCell*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&);
55
56 static void destroy(JSCell*);
57 static constexpr bool needsDestruction = true;
58
59 bool isEmpty() const { return !symbolTable()->size(); }
60 bool isConstVariable(UniquedStringImpl*);
61
62 DECLARE_INFO;
63
64 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject)
65 {
66 return Structure::create(vm, globalObject, jsNull(), TypeInfo(GlobalLexicalEnvironmentType, StructureFlags), info());
67 }
68
69private:
70 JSGlobalLexicalEnvironment(VM& vm, Structure* structure, JSScope* scope)
71 : Base(vm, structure, scope)
72 {
73 }
74};
75
76} // namespace JSC
