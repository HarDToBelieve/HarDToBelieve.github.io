26#pragma once
27
28#include "JSFunction.h"
29
30namespace JSC {
31
32class JSGlobalObject;
33
34using NativeStdFunction = WTF::Function<EncodedJSValue(JSGlobalObject*, CallFrame*)>;
35
36class JSNativeStdFunction final : public JSFunction {
37public:
38 using Base = JSFunction;
39
40 static constexpr unsigned StructureFlags = Base::StructureFlags;
41 static constexpr bool needsDestruction = true;
42 static void destroy(JSCell* cell)
43 {
44 static_cast<JSNativeStdFunction*>(cell)->JSNativeStdFunction::~JSNativeStdFunction();
45 }
46
47 template<typename CellType, SubspaceAccess mode>
48 static IsoSubspace* subspaceFor(VM& vm)
49 {
50 return vm.nativeStdFunctionSpace<mode>();
51 }
52
53 DECLARE_EXPORT_INFO;
54
55 JS_EXPORT_PRIVATE static JSNativeStdFunction* create(VM&, JSGlobalObject*, unsigned length, const String& name, NativeStdFunction&&, Intrinsic = NoIntrinsic, NativeFunction nativeConstructor = callHostFunctionAsConstructor);
56
57 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
58 {
59 ASSERT(globalObject);
60 return Structure::create(vm, globalObject, prototype, TypeInfo(JSFunctionType, StructureFlags), info());
61 }
62
63 const NativeStdFunction& function() { return m_function; }
64
65private:
66 JSNativeStdFunction(VM&, NativeExecutable*, JSGlobalObject*, Structure*, NativeStdFunction&&);
67 void finishCreation(VM&, NativeExecutable*, unsigned length, const String& name);
68 DECLARE_VISIT_CHILDREN;
69
70 NativeStdFunction m_function;
71};
72
73} // namespace JSC
