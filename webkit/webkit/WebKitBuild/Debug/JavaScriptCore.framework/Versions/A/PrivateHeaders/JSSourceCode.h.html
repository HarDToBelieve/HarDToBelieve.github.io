27#pragma once
28
29#include "JSGlobalObject.h"
30#include "JSObject.h"
31#include "SourceCode.h"
32
33namespace JSC {
34
35class JSSourceCode final : public JSCell {
36public:
37 using Base = JSCell;
38
39 static constexpr unsigned StructureFlags = Base::StructureFlags | StructureIsImmortal;
40 static constexpr bool needsDestruction = true;
41
42 DECLARE_EXPORT_INFO;
43
44 template<typename CellType, SubspaceAccess mode>
45 static IsoSubspace* subspaceFor(VM& vm)
46 {
47 return vm.sourceCodeSpace<mode>();
48 }
49
50 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
51 {
52 return Structure::create(vm, globalObject, prototype, TypeInfo(JSSourceCodeType, StructureFlags), info());
53 }
54
55 static JSSourceCode* create(VM& vm, Structure* structure, SourceCode&& sourceCode)
56 {
57 auto* result = new (NotNull, allocateCell<JSSourceCode>(vm.heap)) JSSourceCode(vm, structure, WTFMove(sourceCode));
58 result->finishCreation(vm);
59 return result;
60 }
61
62 static JSSourceCode* create(VM& vm, SourceCode&& sourceCode)
63 {
64 return create(vm, vm.sourceCodeStructure.get(), WTFMove(sourceCode));
65 }
66
67 const SourceCode& sourceCode() const
68 {
69 return m_sourceCode;
70 }
71
72 static void destroy(JSCell*);
73
74private:
75 JSSourceCode(VM& vm, Structure* structure, SourceCode&& sourceCode)
76 : Base(vm, structure)
77 , m_sourceCode(WTFMove(sourceCode))
78 {
79 }
80
81 SourceCode m_sourceCode;
82};
83
84} // namespace JSC
