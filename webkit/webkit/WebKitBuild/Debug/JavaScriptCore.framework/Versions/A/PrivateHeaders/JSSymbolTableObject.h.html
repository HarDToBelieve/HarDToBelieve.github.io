39class JSSymbolTableObject : public JSScope {
40public:
41 using Base = JSScope;
42 static constexpr unsigned StructureFlags = Base::StructureFlags | OverridesGetOwnSpecialPropertyNames;
43
44 SymbolTable* symbolTable() const { return m_symbolTable.get(); }
45
46 JS_EXPORT_PRIVATE static bool deleteProperty(JSCell*, JSGlobalObject*, PropertyName, DeletePropertySlot&);
47 JS_EXPORT_PRIVATE static void getOwnSpecialPropertyNames(JSObject*, JSGlobalObject*, PropertyNameArray&, DontEnumPropertiesMode);
48
49 static ptrdiff_t offsetOfSymbolTable() { return OBJECT_OFFSETOF(JSSymbolTableObject, m_symbolTable); }
50
51 DECLARE_EXPORT_INFO;
52
53protected:
54 JSSymbolTableObject(VM& vm, Structure* structure, JSScope* scope)
55 : Base(vm, structure, scope)
56 {
57 }
58
59 JSSymbolTableObject(VM& vm, Structure* structure, JSScope* scope, SymbolTable* symbolTable)
60 : Base(vm, structure, scope)
61 {
62 ASSERT(symbolTable);
63 setSymbolTable(vm, symbolTable);
64 }
65
66 void setSymbolTable(VM& vm, SymbolTable* symbolTable)
67 {
68 ASSERT(!m_symbolTable);
69 symbolTable->notifyCreation(vm, this, "Allocated a scope");
70 m_symbolTable.set(vm, this, symbolTable);
71 }
72
73 DECLARE_VISIT_CHILDREN;
74
75private:
76 WriteBarrier<SymbolTable> m_symbolTable;
77};

79template<typename SymbolTableObjectType>
80inline bool symbolTableGet(
81 SymbolTableObjectType* object, PropertyName propertyName, PropertySlot& slot)
82{
83 SymbolTable& symbolTable = *object->symbolTable();
84 ConcurrentJSLocker locker(symbolTable.m_lock);
85 SymbolTable::Map::iterator iter = symbolTable.find(locker, propertyName.uid());
86 if (iter == symbolTable.end(locker))
87 return false;
88 SymbolTableEntry::Fast entry = iter->value;
89 ASSERT(!entry.isNull());
90
91 ScopeOffset offset = entry.scopeOffset();
92 // Defend against the inspector asking for a var after it has been optimized out.
93 if (!object->isValidScopeOffset(offset))
94 return false;
95
96 slot.setValue(object, entry.getAttributes() | PropertyAttribute::DontDelete, object->variableAt(offset).get());
97 return true;
98}
99
100template<typename SymbolTableObjectType>
101inline bool symbolTableGet(
102 SymbolTableObjectType* object, PropertyName propertyName, SymbolTableEntry& entry, PropertyDescriptor& descriptor)
103{
104 SymbolTable& symbolTable = *object->symbolTable();
105 ConcurrentJSLocker locker(symbolTable.m_lock);
106 SymbolTable::Map::iterator iter = symbolTable.find(locker, propertyName.uid());
107 if (iter == symbolTable.end(locker))
108 return false;
109 entry = iter->value;
110 ASSERT(!entry.isNull());
111
112 ScopeOffset offset = entry.scopeOffset();
113 // Defend against the inspector asking for a var after it has been optimized out.
114 if (!object->isValidScopeOffset(offset))
115 return false;
116
117 descriptor.setDescriptor(object->variableAt(offset).get(), entry.getAttributes() | PropertyAttribute::DontDelete);
118 return true;
119}

121template<typename SymbolTableObjectType>
122ALWAYS_INLINE void symbolTablePutTouchWatchpointSet(VM& vm, SymbolTableObjectType* object, PropertyName propertyName, JSValue value, WriteBarrierBase<Unknown>* reg, WatchpointSet* set)
123{
124 reg->set(vm, object, value);
125 if (set)
126 VariableWriteFireDetail::touch(vm, set, object, propertyName);
127}
128
129template<typename SymbolTableObjectType>
130ALWAYS_INLINE void symbolTablePutInvalidateWatchpointSet(VM& vm, SymbolTableObjectType* object, PropertyName propertyName, JSValue value, WriteBarrierBase<Unknown>* reg, WatchpointSet* set)
131{
132 reg->set(vm, object, value);
133 if (set)
134 set->invalidate(vm, VariableWriteFireDetail(object, propertyName)); // Don't mess around - if we had found this statically, we would have invalidated it.
135}

137enum class SymbolTablePutMode {
138 Touch,
139 Invalidate
140};
141
142template<SymbolTablePutMode symbolTablePutMode, typename SymbolTableObjectType>
143inline bool symbolTablePut(SymbolTableObjectType* object, JSGlobalObject* globalObject, PropertyName propertyName, JSValue value, bool shouldThrowReadOnlyError, bool ignoreReadOnlyErrors, bool& putResult)
144{
145 VM& vm = getVM(globalObject);
146 auto scope = DECLARE_THROW_SCOPE(vm);
147
148 WatchpointSet* set = nullptr;
149 WriteBarrierBase<Unknown>* reg;
150 {
151 SymbolTable& symbolTable = *object->symbolTable();
152 // FIXME: This is very suspicious. We shouldn't need a GC-safe lock here.
153 // https://bugs.webkit.org/show_bug.cgi?id=134601
154 GCSafeConcurrentJSLocker locker(symbolTable.m_lock, vm.heap);
155 SymbolTable::Map::iterator iter = symbolTable.find(locker, propertyName.uid());
156 if (iter == symbolTable.end(locker))
157 return false;
158 bool wasFat;
159 SymbolTableEntry::Fast fastEntry = iter->value.getFast(wasFat);
160 ASSERT(!fastEntry.isNull());
161 if (fastEntry.isReadOnly() && !ignoreReadOnlyErrors) {
162 if (shouldThrowReadOnlyError)
163 throwTypeError(globalObject, scope, ReadonlyPropertyWriteError);
164 putResult = false;
165 return true;
166 }
167
168 ScopeOffset offset = fastEntry.scopeOffset();
169
170 // Defend against the inspector asking for a var after it has been optimized out.
171 if (!object->isValidScopeOffset(offset))
172 return false;
173
174 set = iter->value.watchpointSet();
175 reg = &object->variableAt(offset);
176 }
177 // I'd prefer we not hold lock while executing barriers, since I prefer to reserve
178 // the right for barriers to be able to trigger GC. And I don't want to hold VM
179 // locks while GC'ing.
180 if (symbolTablePutMode == SymbolTablePutMode::Invalidate)
181 symbolTablePutInvalidateWatchpointSet(vm, object, propertyName, value, reg, set);
182 else
183 symbolTablePutTouchWatchpointSet(vm, object, propertyName, value, reg, set);
184 putResult = true;
185 return true;
186}
187
188template<typename SymbolTableObjectType>
189inline bool symbolTablePutTouchWatchpointSet(
190 SymbolTableObjectType* object, JSGlobalObject* globalObject, PropertyName propertyName, JSValue value,
191 bool shouldThrowReadOnlyError, bool ignoreReadOnlyErrors, bool& putResult)
192{
193 ASSERT(!Heap::heap(value) || Heap::heap(value) == Heap::heap(object));
194 return symbolTablePut<SymbolTablePutMode::Touch>(object, globalObject, propertyName, value, shouldThrowReadOnlyError, ignoreReadOnlyErrors, putResult);
195}
196
197template<typename SymbolTableObjectType>
198inline bool symbolTablePutInvalidateWatchpointSet(
199 SymbolTableObjectType* object, JSGlobalObject* globalObject, PropertyName propertyName, JSValue value,
200 bool shouldThrowReadOnlyError, bool ignoreReadOnlyErrors, bool& putResult)
201{
202 ASSERT(!Heap::heap(value) || Heap::heap(value) == Heap::heap(object));
203 return symbolTablePut<SymbolTablePutMode::Invalidate>(object, globalObject, propertyName, value, shouldThrowReadOnlyError, ignoreReadOnlyErrors, putResult);
204}
