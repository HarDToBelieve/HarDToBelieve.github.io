132 LastJSCObjectType = DerivedStringObjectType, // This is the last "JSC" Object type. After this, we have embedder's (e.g., WebCore) extended object types.
133 MaxJSType = 0b11111111,
134};
135
136static constexpr uint32_t FirstTypedArrayType = Int8ArrayType;
137static constexpr uint32_t LastTypedArrayType = DataViewType;
138static constexpr uint32_t LastTypedArrayTypeExcludingDataView = LastTypedArrayType - 1;
139
140// LastObjectType should be MaxJSType (not LastJSCObjectType) since embedders can add their extended object types after the enums listed in JSType.
141static constexpr uint32_t FirstObjectType = ObjectType;
142static constexpr uint32_t LastObjectType = MaxJSType;
143
144static constexpr uint32_t NumberOfTypedArrayTypes = LastTypedArrayType - FirstTypedArrayType + 1;
145static constexpr uint32_t NumberOfTypedArrayTypesExcludingDataView = NumberOfTypedArrayTypes - 1;
146static constexpr uint32_t NumberOfTypedArrayTypesExcludingBigIntArraysAndDataView = NumberOfTypedArrayTypes - 3;
147
148static_assert(sizeof(JSType) == sizeof(uint8_t), "sizeof(JSType) is one byte.");
149static_assert(LastJSCObjectType < 0b11100000, "Embedder can use 0b11100000 or upper.");
150
151inline constexpr bool isTypedArrayType(JSType type)
152{
153 return (static_cast<uint32_t>(type) - FirstTypedArrayType) < NumberOfTypedArrayTypesExcludingDataView;
154}
