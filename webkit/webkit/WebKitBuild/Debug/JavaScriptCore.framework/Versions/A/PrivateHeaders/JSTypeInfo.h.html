27#pragma once
28
29// This file would be called TypeInfo.h, but that conflicts with <typeinfo.h>
30// in the STL on systems without case-sensitive file systems.
31
32#include "JSType.h"
33
34namespace JSC {
35
36class LLIntOffsetsExtractor;
37
38// Inline flags.
39
40static constexpr unsigned MasqueradesAsUndefined = 1; // WebCore uses MasqueradesAsUndefined to make document.all undetectable.
41static constexpr unsigned ImplementsDefaultHasInstance = 1 << 1;
42static constexpr unsigned OverridesGetCallData = 1 << 2; // Need this flag if you implement [[Callable]] interface, which means overriding getCallData. The object may not be callable since getCallData can say it is not callable.
43static constexpr unsigned OverridesGetOwnPropertySlot = 1 << 3;
44static constexpr unsigned OverridesToThis = 1 << 4; // If this is false then this returns something other than 'this'. Non-object cells that are visible to JS have this set as do some exotic objects.
45static constexpr unsigned HasStaticPropertyTable = 1 << 5;
46static constexpr unsigned TypeInfoPerCellBit = 1 << 7; // Unlike other inline flags, this will only be set on the cell itself and will not be set on the Structure.
47
48// Out of line flags.
49
50static constexpr unsigned ImplementsHasInstance = 1 << 8;
51static constexpr unsigned OverridesGetOwnPropertyNames = 1 << 9;
52static constexpr unsigned OverridesGetOwnSpecialPropertyNames = 1 << 10;
53static constexpr unsigned ProhibitsPropertyCaching = 1 << 11;
54static constexpr unsigned GetOwnPropertySlotIsImpure = 1 << 12;
55static constexpr unsigned NewImpurePropertyFiresWatchpoints = 1 << 13;
56static constexpr unsigned IsImmutablePrototypeExoticObject = 1 << 14;
57static constexpr unsigned GetOwnPropertySlotIsImpureForPropertyAbsence = 1 << 15;
58static constexpr unsigned InterceptsGetOwnPropertySlotByIndexEvenWhenLengthIsNotZero = 1 << 16;
59static constexpr unsigned StructureIsImmortal = 1 << 17;
60static constexpr unsigned OverridesPut = 1 << 18;
61static constexpr unsigned OverridesGetPrototype = 1 << 19;
62static constexpr unsigned GetOwnPropertySlotMayBeWrongAboutDontEnum = 1 << 20;
63
64static constexpr unsigned numberOfInlineBits = 8;
65static constexpr unsigned OverridesGetPrototypeOutOfLine = OverridesGetPrototype >> numberOfInlineBits;
66
67class TypeInfo {
68public:
69 typedef uint8_t InlineTypeFlags;
70 typedef uint16_t OutOfLineTypeFlags;
71
72 TypeInfo(JSType type, unsigned flags)
73 : TypeInfo(type, flags & 0xff, flags >> numberOfInlineBits)
74 {
75 ASSERT(!(flags >> 24));
76 }
77
78 TypeInfo(JSType type, InlineTypeFlags inlineTypeFlags, OutOfLineTypeFlags outOfLineTypeFlags)
79 : m_type(type)
80 , m_flags(inlineTypeFlags)
81 , m_flags2(outOfLineTypeFlags)
82 {
83 }
84
85 JSType type() const { return static_cast<JSType>(m_type); }
86 bool isObject() const { return isObject(type()); }
87 static bool isObject(JSType type) { return type >= ObjectType; }
88 bool isFinalObject() const { return type() == FinalObjectType; }
89 bool isNumberObject() const { return type() == NumberObjectType; }
90
91 unsigned flags() const { return (static_cast<unsigned>(m_flags2) << numberOfInlineBits) | static_cast<unsigned>(m_flags); }
92 bool masqueradesAsUndefined() const { return isSetOnFlags1<MasqueradesAsUndefined>(); }
93 bool implementsHasInstance() const { return isSetOnFlags2<ImplementsHasInstance>(); }
94 bool implementsDefaultHasInstance() const { return isSetOnFlags1<ImplementsDefaultHasInstance>(); }
95 bool overridesGetCallData() const { return isSetOnFlags1<OverridesGetCallData>(); }
96 bool overridesGetOwnPropertySlot() const { return overridesGetOwnPropertySlot(inlineTypeFlags()); }
97 bool hasStaticPropertyTable() const { return isSetOnFlags1<HasStaticPropertyTable>(); }
98 static bool overridesGetOwnPropertySlot(InlineTypeFlags flags) { return flags & OverridesGetOwnPropertySlot; }
99 static bool hasStaticPropertyTable(InlineTypeFlags flags) { return flags & HasStaticPropertyTable; }
100 static bool perCellBit(InlineTypeFlags flags) { return flags & TypeInfoPerCellBit; }
101 bool overridesToThis() const { return isSetOnFlags1<OverridesToThis>(); }
102 bool structureIsImmortal() const { return isSetOnFlags2<StructureIsImmortal>(); }
103 bool overridesGetOwnPropertyNames() const { return isSetOnFlags2<OverridesGetOwnPropertyNames>(); }
104 bool overridesGetOwnSpecialPropertyNames() const { return isSetOnFlags2<OverridesGetOwnSpecialPropertyNames>(); }
105 bool overridesAnyFormOfGetOwnPropertyNames() const { return overridesGetOwnPropertyNames() || overridesGetOwnSpecialPropertyNames(); }
106 bool overridesPut() const { return isSetOnFlags2<OverridesPut>(); }
107 bool overridesGetPrototype() const { return isSetOnFlags2<OverridesGetPrototype>(); }
108 bool prohibitsPropertyCaching() const { return isSetOnFlags2<ProhibitsPropertyCaching>(); }
109 bool getOwnPropertySlotIsImpure() const { return isSetOnFlags2<GetOwnPropertySlotIsImpure>(); }
110 bool getOwnPropertySlotIsImpureForPropertyAbsence() const { return isSetOnFlags2<GetOwnPropertySlotIsImpureForPropertyAbsence>(); }
111 bool getOwnPropertySlotMayBeWrongAboutDontEnum() const { return isSetOnFlags2<GetOwnPropertySlotMayBeWrongAboutDontEnum>(); }
112 bool newImpurePropertyFiresWatchpoints() const { return isSetOnFlags2<NewImpurePropertyFiresWatchpoints>(); }
113 bool isImmutablePrototypeExoticObject() const { return isSetOnFlags2<IsImmutablePrototypeExoticObject>(); }
114 bool interceptsGetOwnPropertySlotByIndexEvenWhenLengthIsNotZero() const { return isSetOnFlags2<InterceptsGetOwnPropertySlotByIndexEvenWhenLengthIsNotZero>(); }
115
116 static bool isArgumentsType(JSType type)
117 {
118 return type == DirectArgumentsType
119 || type == ScopedArgumentsType
120 || type == ClonedArgumentsType;
121 }
122
123 static ptrdiff_t flagsOffset()
124 {
125 return OBJECT_OFFSETOF(TypeInfo, m_flags);
126 }
127
128 static ptrdiff_t typeOffset()
129 {
130 return OBJECT_OFFSETOF(TypeInfo, m_type);
131 }
132
133 // Since the Structure doesn't track TypeInfoPerCellBit, we need to make sure we copy it.
134 static InlineTypeFlags mergeInlineTypeFlags(InlineTypeFlags structureFlags, InlineTypeFlags oldCellFlags)
135 {
136 return structureFlags | (oldCellFlags & static_cast<InlineTypeFlags>(TypeInfoPerCellBit));
137 }
138
139 InlineTypeFlags inlineTypeFlags() const { return m_flags; }
140 OutOfLineTypeFlags outOfLineTypeFlags() const { return m_flags2; }
141
142private:
143 friend class LLIntOffsetsExtractor;
144
145 template<unsigned flag>
146 bool isSetOnFlags1() const
147 {
148 static_assert(flag <= (1 << 7));
149 return m_flags & flag;
150 }
151
152 template<unsigned flag>
153 bool isSetOnFlags2() const
154 {
155 static_assert(flag >= (1 << numberOfInlineBits) && flag <= (1 << 24));
156 return m_flags2 & (flag >> numberOfInlineBits);
157 }
158
159 JSType m_type;
160 uint8_t m_flags;
161 uint16_t m_flags2;
162};
163
164} // namespace JSC
