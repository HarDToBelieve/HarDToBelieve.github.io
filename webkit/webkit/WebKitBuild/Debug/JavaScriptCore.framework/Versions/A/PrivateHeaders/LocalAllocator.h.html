26#pragma once
27
28#include "AllocationFailureMode.h"
29#include "FreeList.h"
30#include "MarkedBlock.h"
31#include <wtf/Noncopyable.h>
32
33namespace JSC {
34
35class BlockDirectory;
36class GCDeferralContext;
37class Heap;
38
39class LocalAllocator : public BasicRawSentinelNode<LocalAllocator> {
40 WTF_MAKE_NONCOPYABLE(LocalAllocator);
41
42public:
43 LocalAllocator(BlockDirectory*);
44 ~LocalAllocator();
45
46 void* allocate(Heap&, GCDeferralContext*, AllocationFailureMode);
47
48 unsigned cellSize() const { return m_freeList.cellSize(); }
49
50 void stopAllocating();
51 void prepareForAllocation();
52 void resumeAllocating();
53 void stopAllocatingForGood();
54
55 static ptrdiff_t offsetOfFreeList();
56 static ptrdiff_t offsetOfCellSize();
57
58 bool isFreeListedCell(const void*) const;
59
60private:
61 friend class BlockDirectory;
62
63 void reset();
64 JS_EXPORT_PRIVATE void* allocateSlowCase(Heap&, GCDeferralContext*, AllocationFailureMode);
65 void didConsumeFreeList();
66 void* tryAllocateWithoutCollecting();
67 void* tryAllocateIn(MarkedBlock::Handle*);
68 void* allocateIn(MarkedBlock::Handle*);
69 ALWAYS_INLINE void doTestCollectionsIfNeeded(Heap&, GCDeferralContext*);
70
71 BlockDirectory* m_directory;
72 FreeList m_freeList;
73
74 MarkedBlock::Handle* m_currentBlock { nullptr };
75 MarkedBlock::Handle* m_lastActiveBlock { nullptr };
76
77 // After you do something to a block based on one of these cursors, you clear the bit in the
78 // corresponding bitvector and leave the cursor where it was.
79 unsigned m_allocationCursor { 0 }; // Points to the next block that is a candidate for allocation.
80};
81
82inline ptrdiff_t LocalAllocator::offsetOfFreeList()
83{
84 return OBJECT_OFFSETOF(LocalAllocator, m_freeList);
85}
86
87inline ptrdiff_t LocalAllocator::offsetOfCellSize()
88{
89 return OBJECT_OFFSETOF(LocalAllocator, m_freeList) + FreeList::offsetOfCellSize();
90}
91
92} // namespace JSC
