35// ASSERT_VALID_CODE_POINTER checks that ptr is a non-null pointer, and that it is a valid
36// instruction address on the platform (for example, check any alignment requirements).
37#if CPU(ARM_THUMB2) && ENABLE(JIT)
38// ARM instructions must be 16-bit aligned. Thumb2 code pointers to be loaded into
39// into the processor are decorated with the bottom bit set, while traditional ARM has
40// the lower bit clear. Since we don't know what kind of pointer, we check for both
41// decorated and undecorated null.
42#define ASSERT_NULL_OR_VALID_CODE_POINTER(ptr) \
43 ASSERT(!ptr || reinterpret_cast<intptr_t>(ptr) & ~1)
44#define ASSERT_VALID_CODE_POINTER(ptr) \
45 ASSERT(reinterpret_cast<intptr_t>(ptr) & ~1)
46#define ASSERT_VALID_CODE_OFFSET(offset) \
47 ASSERT(!(offset & 1)) // Must be multiple of 2.
48#else
49#define ASSERT_NULL_OR_VALID_CODE_POINTER(ptr) // Anything goes!
50#define ASSERT_VALID_CODE_POINTER(ptr) \
51 ASSERT(ptr)
52#define ASSERT_VALID_CODE_OFFSET(offset) // Anything goes!
53#endif

62// CFunctionPtr can only be used to hold C/C++ functions.
63class CFunctionPtr {
64public:
65 using Ptr = void(*)();
66
67 CFunctionPtr() { }
68 CFunctionPtr(std::nullptr_t) { }
69
70 template<typename ReturnType, typename... Arguments>
71 constexpr CFunctionPtr(ReturnType(&ptr)(Arguments...))
72 : m_ptr(reinterpret_cast<Ptr>(&ptr))
73 { }
74
75 template<typename ReturnType, typename... Arguments>
76 explicit CFunctionPtr(ReturnType(*ptr)(Arguments...))
77 : m_ptr(reinterpret_cast<Ptr>(ptr))
78 {
79 assertIsCFunctionPtr(m_ptr);
80 }
81
82 // MSVC doesn't seem to treat functions with different calling conventions as
83 // different types; these methods are already defined for fastcall, below.
84#if CALLING_CONVENTION_IS_STDCALL && !OS(WINDOWS)
85 template<typename ReturnType, typename... Arguments>
86 constexpr CFunctionPtr(ReturnType(CDECL &ptr)(Arguments...))
87 : m_ptr(reinterpret_cast<Ptr>(&ptr))
88 { }
89
90 template<typename ReturnType, typename... Arguments>
91 explicit CFunctionPtr(ReturnType(CDECL *ptr)(Arguments...))
92 : m_ptr(reinterpret_cast<Ptr>(ptr))
93 {
94 assertIsCFunctionPtr(m_ptr);
95 }
96
97#endif // CALLING_CONVENTION_IS_STDCALL && !OS(WINDOWS)
98
99#if COMPILER_SUPPORTS(FASTCALL_CALLING_CONVENTION)
100 template<typename ReturnType, typename... Arguments>
101 constexpr CFunctionPtr(ReturnType(FASTCALL &ptr)(Arguments...))
102 : m_ptr(reinterpret_cast<Ptr>(&ptr))
103 { }
104
105 template<typename ReturnType, typename... Arguments>
106 explicit CFunctionPtr(ReturnType(FASTCALL *ptr)(Arguments...))
107 : m_ptr(reinterpret_cast<Ptr>(ptr))
108 {
109 assertIsCFunctionPtr(m_ptr);
110 }
111#endif // COMPILER_SUPPORTS(FASTCALL_CALLING_CONVENTION)
112
113 constexpr Ptr get() const { return m_ptr; }
114 void* address() const { return reinterpret_cast<void*>(m_ptr); }
115
116 explicit operator bool() const { return !!m_ptr; }
117 bool operator!() const { return !m_ptr; }
118
119 bool operator==(const CFunctionPtr& other) const { return m_ptr == other.m_ptr; }
120 bool operator!=(const CFunctionPtr& other) const { return m_ptr != other.m_ptr; }
121
122private:
123 Ptr m_ptr { nullptr };
124};

127// FunctionPtr:
128//
129// FunctionPtr should be used to wrap pointers to C/C++ functions in JSC
130// (particularly, the stub functions).
131template<PtrTag tag = CFunctionPtrTag>
132class FunctionPtr {
133public:
134 FunctionPtr() { }
135 FunctionPtr(std::nullptr_t) { }
136
137 template<typename ReturnType, typename... Arguments>
138 FunctionPtr(ReturnType(*value)(Arguments...))
139 : m_value(tagCFunctionPtr<void*, tag>(value))
140 {
141 assertIsNullOrCFunctionPtr(value);
142 ASSERT_NULL_OR_VALID_CODE_POINTER(m_value);
143 }
144
145// MSVC doesn't seem to treat functions with different calling conventions as
146// different types; these methods already defined for fastcall, below.
147#if CALLING_CONVENTION_IS_STDCALL && !OS(WINDOWS)
148
149 template<typename ReturnType, typename... Arguments>
150 FunctionPtr(ReturnType(CDECL *value)(Arguments...))
151 : m_value(tagCFunctionPtr<void*, tag>(value))
152 {
153 assertIsNullOrCFunctionPtr(value);
154 ASSERT_NULL_OR_VALID_CODE_POINTER(m_value);
155 }
156
157#endif // CALLING_CONVENTION_IS_STDCALL && !OS(WINDOWS)
158
159#if COMPILER_SUPPORTS(FASTCALL_CALLING_CONVENTION)
160
161 template<typename ReturnType, typename... Arguments>
162 FunctionPtr(ReturnType(FASTCALL *value)(Arguments...))
163 : m_value(tagCFunctionPtr<void*, tag>(value))
164 {
165 assertIsNullOrCFunctionPtr(value);
166 ASSERT_NULL_OR_VALID_CODE_POINTER(m_value);
167 }
168
169#endif // COMPILER_SUPPORTS(FASTCALL_CALLING_CONVENTION)
170
171 template<typename PtrType, typename = std::enable_if_t<std::is_pointer<PtrType>::value && !std::is_function<typename std::remove_pointer<PtrType>::type>::value>>
172 explicit FunctionPtr(PtrType value)
173 // Using a C-ctyle cast here to avoid compiler error on RVTC:
174 // Error: #694: reinterpret_cast cannot cast away const or other type qualifiers
175 // (I guess on RVTC function pointers have a different constness to GCC/MSVC?)
176 : m_value(tagCFunctionPtr<void*, tag>(value))
177 {
178 assertIsNullOrCFunctionPtr(value);
179 ASSERT_NULL_OR_VALID_CODE_POINTER(m_value);
180 }
181
182 explicit FunctionPtr(MacroAssemblerCodePtr<tag>);
183
184 template<PtrTag otherTag>
185 FunctionPtr<otherTag> retagged() const
186 {
187 if (!m_value)
188 return FunctionPtr<otherTag>();
189 return FunctionPtr<otherTag>(*this);
190 }
191
192 void* executableAddress() const
193 {
194 return m_value;
195 }
196
197 template<PtrTag newTag>
198 void* retaggedExecutableAddress() const
199 {
200 return retagCodePtr<tag, newTag>(m_value);
201 }
202
203 explicit operator bool() const { return !!m_value; }
204 bool operator!() const { return !m_value; }
205
206 bool operator==(const FunctionPtr& other) const { return m_value == other.m_value; }
207 bool operator!=(const FunctionPtr& other) const { return m_value != other.m_value; }
208
209private:
210 template<PtrTag otherTag>
211 explicit FunctionPtr(const FunctionPtr<otherTag>& other)
212 : m_value(retagCodePtr<otherTag, tag>(other.executableAddress()))
213 {
214 ASSERT_NULL_OR_VALID_CODE_POINTER(m_value);
215 }
216
217 void* m_value { nullptr };
218
219 template<PtrTag> friend class FunctionPtr;
220};

227// ReturnAddressPtr:
228//
229// ReturnAddressPtr should be used to wrap return addresses generated by processor
230// 'call' instructions exectued in JIT code. We use return addresses to look up
231// exception and optimization information, and to repatch the call instruction
232// that is the source of the return address.
233class ReturnAddressPtr {
234public:
235 ReturnAddressPtr() { }
236
237 explicit ReturnAddressPtr(const void* returnAddress)
238 {
239#if CPU(ARM64E)
240 assertIsNotTagged(returnAddress);
241 returnAddress = retagCodePtr<NoPtrTag, ReturnAddressPtrTag>(returnAddress);
242#endif
243 m_value = returnAddress;
244 ASSERT_VALID_CODE_POINTER(m_value);
245 }
246
247 const void* value() const
248 {
249 return m_value;
250 }
251
252 const void* untaggedValue() const
253 {
254 return untagCodePtr<ReturnAddressPtrTag>(m_value);
255 }
256
257 void dump(PrintStream& out) const
258 {
259 out.print(RawPointer(m_value));
260 }
261
262private:
263 const void* m_value { nullptr };
264};

274// FIXME: Make JSC MacroAssemblerCodePtr injerit from MetaAllocatorPtr.
275// https://bugs.webkit.org/show_bug.cgi?id=185145
276template<PtrTag tag>
277class MacroAssemblerCodePtr : private MacroAssemblerCodePtrBase {
278public:
279 MacroAssemblerCodePtr() = default;
280 MacroAssemblerCodePtr(std::nullptr_t) : m_value(nullptr) { }
281
282 explicit MacroAssemblerCodePtr(const void* value)
283#if CPU(ARM_THUMB2)
284 // Decorate the pointer as a thumb code pointer.
285 : m_value(reinterpret_cast<const char*>(value) + 1)
286#else
287 : m_value(value)
288#endif
289 {
290 assertIsTaggedWith<tag>(value);
291 ASSERT(value);
292#if CPU(ARM_THUMB2)
293 ASSERT(!(reinterpret_cast<uintptr_t>(value) & 1));
294#endif
295 ASSERT_VALID_CODE_POINTER(m_value);
296 }
297
298 static MacroAssemblerCodePtr createFromExecutableAddress(const void* value)
299 {
300 ASSERT(value);
301 ASSERT_VALID_CODE_POINTER(value);
302 assertIsTaggedWith<tag>(value);
303 MacroAssemblerCodePtr result;
304 result.m_value = value;
305 return result;
306 }
307
308 explicit MacroAssemblerCodePtr(ReturnAddressPtr ra)
309 : m_value(retagCodePtr<ReturnAddressPtrTag, tag>(ra.value()))
310 {
311 ASSERT(ra.untaggedValue());
312 ASSERT_VALID_CODE_POINTER(m_value);
313 }
314
315 template<PtrTag newTag>
316 MacroAssemblerCodePtr<newTag> retagged() const
317 {
318 if (!m_value)
319 return MacroAssemblerCodePtr<newTag>();
320 return MacroAssemblerCodePtr<newTag>::createFromExecutableAddress(retaggedExecutableAddress<newTag>());
321 }
322
323 template<typename T = void*>
324 T executableAddress() const
325 {
326 return bitwise_cast<T>(m_value);
327 }
328
329 template<typename T = void*>
330 T untaggedExecutableAddress() const
331 {
332 return untagCodePtr<T, tag>(m_value);
333 }
334
335 template<PtrTag newTag, typename T = void*>
336 T retaggedExecutableAddress() const
337 {
338 return retagCodePtr<T, tag, newTag>(m_value);
339 }
340
341#if CPU(ARM_THUMB2)
342 // To use this pointer as a data address remove the decoration.
343 template<typename T = void*>
344 T dataLocation() const
345 {
346 ASSERT_VALID_CODE_POINTER(m_value);
347 return bitwise_cast<T>(m_value ? bitwise_cast<char*>(m_value) - 1 : nullptr);
348 }
349#else
350 template<typename T = void*>
351 T dataLocation() const
352 {
353 ASSERT_VALID_CODE_POINTER(m_value);
354 return untagCodePtr<T, tag>(m_value);
355 }
356#endif
357
358 bool operator!() const
359 {
360 return !m_value;
361 }
362 explicit operator bool() const { return !(!*this); }
363
364 bool operator==(const MacroAssemblerCodePtr& other) const
365 {
366 return m_value == other.m_value;
367 }
368
369 // Disallow any casting operations (except for booleans). Instead, the client
370 // should be asking executableAddress() explicitly.
371 template<typename T, typename = std::enable_if_t<!std::is_same<T, bool>::value>>
372 operator T() = delete;
373
374 void dumpWithName(const char* name, PrintStream& out) const
375 {
376 MacroAssemblerCodePtrBase::dumpWithName(executableAddress(), dataLocation(), name, out);
377 }
378
379 void dump(PrintStream& out) const { dumpWithName("CodePtr", out); }
380
381 enum EmptyValueTag { EmptyValue };
382 enum DeletedValueTag { DeletedValue };
383
384 MacroAssemblerCodePtr(EmptyValueTag)
385 : m_value(emptyValue())
386 { }
387
388 MacroAssemblerCodePtr(DeletedValueTag)
389 : m_value(deletedValue())
390 { }
391
392 bool isEmptyValue() const { return m_value == emptyValue(); }
393 bool isDeletedValue() const { return m_value == deletedValue(); }
394
395 unsigned hash() const { return PtrHash<const void*>::hash(m_value); }
396
397 static void initialize();
398
399private:
400 static const void* emptyValue() { return bitwise_cast<void*>(static_cast<intptr_t>(1)); }
401 static const void* deletedValue() { return bitwise_cast<void*>(static_cast<intptr_t>(2)); }
402
403 const void* m_value { nullptr };
404};

421class MacroAssemblerCodeRefBase {
422protected:

424 static bool tryToDisassemble(MacroAssemblerCodePtr<DisassemblyPtrTag>, size_t, const char* prefix);

428template<PtrTag tag>
429class MacroAssemblerCodeRef : private MacroAssemblerCodeRefBase {
430private:
431 // This is private because it's dangerous enough that we want uses of it
432 // to be easy to find - hence the static create method below.
433 explicit MacroAssemblerCodeRef(MacroAssemblerCodePtr<tag> codePtr)
434 : m_codePtr(codePtr)
435 {
436 ASSERT(m_codePtr);
437 }
438
439public:
440 MacroAssemblerCodeRef() = default;
441
442 MacroAssemblerCodeRef(Ref<ExecutableMemoryHandle>&& executableMemory)
443 : m_codePtr(executableMemory->start().retaggedPtr<tag>())
444 , m_executableMemory(WTFMove(executableMemory))
445 {
446 ASSERT(m_executableMemory->start());
447 ASSERT(m_codePtr);
448 }
449
450 template<PtrTag otherTag>
451 MacroAssemblerCodeRef& operator=(const MacroAssemblerCodeRef<otherTag>& otherCodeRef)
452 {
453 m_codePtr = MacroAssemblerCodePtr<tag>::createFromExecutableAddress(otherCodeRef.code().template retaggedExecutableAddress<tag>());
454 m_executableMemory = otherCodeRef.m_executableMemory;
455 return *this;
456 }
457
458 // Use this only when you know that the codePtr refers to code that is
459 // already being kept alive through some other means. Typically this means
460 // that codePtr is immortal.
461 static MacroAssemblerCodeRef createSelfManagedCodeRef(MacroAssemblerCodePtr<tag> codePtr)
462 {
463 return MacroAssemblerCodeRef(codePtr);
464 }
465
466 ExecutableMemoryHandle* executableMemory() const
467 {
468 return m_executableMemory.get();
469 }
470
471 MacroAssemblerCodePtr<tag> code() const
472 {
473 return m_codePtr;
474 }
475
476 template<PtrTag newTag>
477 MacroAssemblerCodePtr<newTag> retaggedCode() const
478 {
479 return m_codePtr.template retagged<newTag>();
480 }
481
482 template<PtrTag newTag>
483 MacroAssemblerCodeRef<newTag> retagged() const
484 {
485 return MacroAssemblerCodeRef<newTag>(*this);
486 }
487
488 size_t size() const
489 {
490 if (!m_executableMemory)
491 return 0;
492 return m_executableMemory->sizeInBytes();
493 }
494
495 bool tryToDisassemble(PrintStream& out, const char* prefix = "") const
496 {
497 return tryToDisassemble(retaggedCode<DisassemblyPtrTag>(), size(), prefix, out);
498 }
499
500 bool tryToDisassemble(const char* prefix = "") const
501 {
502 return tryToDisassemble(retaggedCode<DisassemblyPtrTag>(), size(), prefix);
503 }
504
505 CString disassembly() const
506 {
507 return MacroAssemblerCodeRefBase::disassembly(retaggedCode<DisassemblyPtrTag>(), size());
508 }
509
510 explicit operator bool() const { return !!m_codePtr; }
511
512 void dump(PrintStream& out) const
513 {
514 m_codePtr.dumpWithName("CodeRef", out);
515 }
516
517private:
518 template<PtrTag otherTag>
519 MacroAssemblerCodeRef(const MacroAssemblerCodeRef<otherTag>& otherCodeRef)
520 {
521 *this = otherCodeRef;
522 }
523
524 MacroAssemblerCodePtr<tag> m_codePtr;
525 RefPtr<ExecutableMemoryHandle> m_executableMemory;
526
527 template<PtrTag> friend class MacroAssemblerCodeRef;
528};
