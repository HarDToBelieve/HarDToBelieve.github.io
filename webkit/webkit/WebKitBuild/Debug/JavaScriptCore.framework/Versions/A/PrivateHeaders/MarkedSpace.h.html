51class MarkedSpace {
52 WTF_MAKE_NONCOPYABLE(MarkedSpace);
53public:
54 // sizeStep is really a synonym for atomSize; it's no accident that they are the same.
55 static constexpr size_t sizeStep = MarkedBlock::atomSize;
56
57 // Sizes up to this amount get a size class for each size step.
58 static constexpr size_t preciseCutoff = 80;
59
60 // The amount of available payload in a block is the block's size minus the footer.
61 static constexpr size_t blockPayload = MarkedBlock::payloadSize;
62
63 // The largest cell we're willing to allocate in a MarkedBlock the "normal way" (i.e. using size
64 // classes, rather than a large allocation) is half the size of the payload, rounded down. This
65 // ensures that we only use the size class approach if it means being able to pack two things
66 // into one block.
67 static constexpr size_t largeCutoff = (blockPayload / 2) & ~(sizeStep - 1);
68 static_assert(largeCutoff <= UINT32_MAX);
69
70 // We have an extra size class for size zero.
71 static constexpr size_t numSizeClasses = largeCutoff / sizeStep + 1;
72
73 static constexpr HeapVersion nullVersion = 0; // The version of freshly allocated blocks.
74 static constexpr HeapVersion initialVersion = 2; // The version that the heap starts out with. Set to make sure that nextVersion(nullVersion) != initialVersion.
75
76 static HeapVersion nextVersion(HeapVersion version)
77 {
78 version++;
79 if (version == nullVersion)
80 version = initialVersion;
81 return version;
82 }
83
84 static size_t sizeClassToIndex(size_t size)
85 {
86 return (size + sizeStep - 1) / sizeStep;
87 }
88
89 static size_t indexToSizeClass(size_t index)
90 {
91 size_t result = index * sizeStep;
92 ASSERT(sizeClassToIndex(result) == index);
93 return result;
94 }

228template <typename Functor> inline void MarkedSpace::forEachBlock(const Functor& functor)
229{
230 forEachDirectory(
231 [&] (BlockDirectory& directory) -> IterationStatus {
232 directory.forEachBlock(functor);
233 return IterationStatus::Continue;
234 });
235}
236
237template <typename Functor>
238void MarkedSpace::forEachDirectory(const Functor& functor)
239{
240 for (BlockDirectory* directory = m_directories.first(); directory; directory = directory->nextDirectory()) {
241 if (functor(*directory) == IterationStatus::Done)
242 return;
243 }
244}
245
246template<typename Functor>
247void MarkedSpace::forEachSubspace(const Functor& functor)
248{
249 for (auto subspace : m_subspaces) {
250 if (functor(*subspace) == IterationStatus::Done)
251 return;
252 }
253}

256ALWAYS_INLINE size_t MarkedSpace::optimalSizeFor(size_t bytes)
257{
258 ASSERT(bytes);
259 if (bytes <= preciseCutoff)
260 return WTF::roundUpToMultipleOf<sizeStep>(bytes);
261 if (bytes <= largeCutoff)
262 return s_sizeClassForSizeStep[sizeClassToIndex(bytes)];
263 return bytes;
264}
