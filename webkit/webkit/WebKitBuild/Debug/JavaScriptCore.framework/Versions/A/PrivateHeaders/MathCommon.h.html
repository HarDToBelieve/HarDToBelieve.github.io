38constexpr double maxSafeInteger()
39{
40 // 2 ^ 53 - 1
41 return 9007199254740991.0;
42}
43
44constexpr double minSafeInteger()
45{
46 // -(2 ^ 53 - 1)
47 return -9007199254740991.0;

65// This in the ToInt32 operation is defined in section 9.5 of the ECMA-262 spec.
66// Note that this operation is identical to ToUInt32 other than to interpretation
67// of the resulting bit-pattern (as such this method is also called to implement
68// ToUInt32).
69//
70// The operation can be described as round towards zero, then select the 32 least
71// bits of the resulting value in 2s-complement representation.
72enum ToInt32Mode {
73 Generic,
74 AfterSensibleConversionAttempt,
75};
76template<ToInt32Mode Mode>
77ALWAYS_INLINE int32_t toInt32Internal(double number)
78{
79 uint64_t bits = WTF::bitwise_cast<uint64_t>(number);
80 int32_t exp = (static_cast<int32_t>(bits >> 52) & 0x7ff) - 0x3ff;
81
82 // If exponent < 0 there will be no bits to the left of the decimal point
83 // after rounding; if the exponent is > 83 then no bits of precision can be
84 // left in the low 32-bit range of the result (IEEE-754 doubles have 52 bits
85 // of fractional precision).
86 // Note this case handles 0, -0, and all infinite, NaN, & denormal value.
87
88 // We need to check exp > 83 because:
89 // 1. exp may be used as a left shift value below in (exp - 52), and
90 // 2. Left shift amounts that exceed 31 results in undefined behavior. See:
91 // http://en.cppreference.com/w/cpp/language/operator_arithmetic#Bitwise_shift_operators
92 //
93 // Using an unsigned comparison here also gives us a exp < 0 check for free.
94 if (static_cast<uint32_t>(exp) > 83u)
95 return 0;
96
97 // Select the appropriate 32-bits from the floating point mantissa. If the
98 // exponent is 52 then the bits we need to select are already aligned to the
99 // lowest bits of the 64-bit integer representation of the number, no need
100 // to shift. If the exponent is greater than 52 we need to shift the value
101 // left by (exp - 52), if the value is less than 52 we need to shift right
102 // accordingly.
103 uint32_t result = (exp > 52)
104 ? static_cast<uint32_t>(bits << (exp - 52))
105 : static_cast<uint32_t>(bits >> (52 - exp));
106
107 // IEEE-754 double precision values are stored omitting an implicit 1 before
108 // the decimal point; we need to reinsert this now. We may also the shifted
109 // invalid bits into the result that are not a part of the mantissa (the sign
110 // and exponent bits from the floatingpoint representation); mask these out.
111 // Note that missingOne should be held as uint32_t since ((1 << 31) - 1) causes
112 // int32_t overflow.
113 if (Mode == ToInt32Mode::AfterSensibleConversionAttempt) {
114 if (exp == 31) {
115 // This is an optimization for when toInt32() is called in the slow path
116 // of a JIT operation. Currently, this optimization is only applicable for
117 // x86 ports. This optimization offers 5% performance improvement in
118 // kraken-crypto-pbkdf2.
119 //
120 // On x86, the fast path does a sensible double-to-int32 conversion, by
121 // first attempting to truncate the double value to int32 using the
122 // cvttsd2si_rr instruction. According to Intel's manual, cvttsd2si performs
123 // the following truncate operation:
124 //
125 // If src = NaN, +-Inf, or |(src)rz| > 0x7fffffff and (src)rz != 0x80000000,
126 // then the result becomes 0x80000000. Otherwise, the operation succeeds.
127 //
128 // Note that the ()rz notation means rounding towards zero.
129 // We'll call the slow case function only when the above cvttsd2si fails. The
130 // JIT code checks for fast path failure by checking if result == 0x80000000.
131 // Hence, the slow path will only see the following possible set of numbers:
132 //
133 // NaN, +-Inf, or |(src)rz| > 0x7fffffff.
134 //
135 // As a result, the exp of the double is always >= 31. We can take advantage
136 // of this by specifically checking for (exp == 31) and give the compiler a
137 // chance to constant fold the operations below.
138 const constexpr uint32_t missingOne = 1U << 31;
139 result &= missingOne - 1;
140 result += missingOne;
141 }
142 } else {
143 if (exp < 32) {
144 const uint32_t missingOne = 1U << exp;
145 result &= missingOne - 1;
146 result += missingOne;
147 }
148 }
149
150 // If the input value was negative (we could test either 'number' or 'bits',
151 // but testing 'bits' is likely faster) invert the result appropriately.
152 return static_cast<int64_t>(bits) < 0 ? -static_cast<int32_t>(result) : static_cast<int32_t>(result);
153}

155ALWAYS_INLINE int32_t toInt32(double number)
156{
157#if HAVE(FJCVTZS_INSTRUCTION)
158 int32_t result = 0;
159 __asm__ ("fjcvtzs %w0, %d1" : "=r" (result) : "w" (number) : "cc");
160 return result;
161#else
162 return toInt32Internal<ToInt32Mode::Generic>(number);
163#endif
164}

166// This implements ToUInt32, defined in ECMA-262 9.6.
167inline uint32_t toUInt32(double number)
168{
169 // As commented in the spec, the operation of ToInt32 and ToUint32 only differ
170 // in how the result is interpreted; see NOTEs in sections 9.5 and 9.6.
171 return toInt32(number);
172}

174ALWAYS_INLINE constexpr UCPUStrictInt32 toUCPUStrictInt32(int32_t value)
175{
176 // StrictInt32 format requires that higher bits are all zeros even if value is negative.
177 return static_cast<UCPUStrictInt32>(static_cast<uint32_t>(value));
178}

180inline std::optional<double> safeReciprocalForDivByConst(double constant)
181{
182 // No "weird" numbers (NaN, Denormal, etc).
183 if (!constant || !std::isnormal(constant))
184 return std::nullopt;
185
186 int exponent;
187 if (std::frexp(constant, &exponent) != 0.5)
188 return std::nullopt;
189
190 // Note that frexp() returns the value divided by two
191 // so we to offset this exponent by one.
192 exponent -= 1;
193
194 // A double exponent is between -1022 and 1023.
195 // Nothing we can do to invert 1023.
196 if (exponent == 1023)
197 return std::nullopt;
198
199 double reciprocal = std::ldexp(1, -exponent);
200 ASSERT(std::isnormal(reciprocal));
201 ASSERT(1. / constant == reciprocal);
202 ASSERT(constant == 1. / reciprocal);
203 ASSERT(1. == constant * reciprocal);
204
205 return reciprocal;
206}

208ALWAYS_INLINE bool canBeStrictInt32(double value)
209{
210 // Note: while this behavior is undefined for NaN and inf, the subsequent statement will catch these cases.
211 const int32_t asInt32 = static_cast<int32_t>(value);
212 return !(asInt32 != value || (!asInt32 && std::signbit(value))); // true for -0.0
213}
214
215ALWAYS_INLINE bool canBeInt32(double value)
216{
217 // Note: Strictly speaking this is an undefined behavior.
218 return static_cast<int32_t>(value) == value;
219}
