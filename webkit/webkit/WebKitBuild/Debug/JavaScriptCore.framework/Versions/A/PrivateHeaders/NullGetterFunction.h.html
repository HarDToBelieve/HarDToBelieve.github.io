28#include "InternalFunction.h"

30namespace JSC {
31
32class NullGetterFunction final : public InternalFunction {
33public:
34 typedef InternalFunction Base;
35
36 static NullGetterFunction* create(VM& vm, Structure* structure)
37 {
38 // Since NullGetterFunction is per JSGlobalObject, we use put-without-transition in InternalFunction::finishCreation.
39 NullGetterFunction* function = new (NotNull, allocateCell< NullGetterFunction>(vm.heap)) NullGetterFunction(vm, structure);
40 function->finishCreation(vm, 0, String(), PropertyAdditionMode::WithoutStructureTransition);
41 return function;
42 }
43
44 DECLARE_EXPORT_INFO;
45
46 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
47 {
48 return Structure::create(vm, globalObject, prototype, TypeInfo(InternalFunctionType, StructureFlags), info());
49 }
50
51private:
52 NullGetterFunction(VM&, Structure*);
53};
54STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(NullGetterFunction, InternalFunction);

56} // namespace JSC
