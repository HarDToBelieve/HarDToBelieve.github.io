25namespace JSC {

27class ObjectPrototype final : public JSNonFinalObject {
28public:
29 using Base = JSNonFinalObject;
30 static constexpr unsigned StructureFlags = Base::StructureFlags | IsImmutablePrototypeExoticObject;
31
32 template<typename CellType, SubspaceAccess>
33 static IsoSubspace* subspaceFor(VM& vm)
34 {
35 STATIC_ASSERT_ISO_SUBSPACE_SHARABLE(ObjectPrototype, Base);
36 return &vm.plainObjectSpace;
37 }
38
39 static ObjectPrototype* create(VM&, JSGlobalObject*, Structure*);
40
41 DECLARE_EXPORT_INFO;
42
43 static Structure* createStructure(VM& vm, JSGlobalObject* globalObject, JSValue prototype)
44 {
45 return Structure::create(vm, globalObject, prototype, TypeInfo(ObjectType, StructureFlags), info());
46 }
47
48private:
49 ObjectPrototype(VM&, Structure*);
50 void finishCreation(VM&, JSGlobalObject*);
51};
