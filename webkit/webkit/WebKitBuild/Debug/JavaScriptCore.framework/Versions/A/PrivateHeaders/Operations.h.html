37template<Concurrency concurrency>
38ALWAYS_INLINE TriState jsTypeofIsObjectWithConcurrency(JSGlobalObject* globalObject, JSValue value)
39{
40 VM& vm = globalObject->vm();
41 if (!value.isObject())
42 return triState(value.isNull());
43 JSObject* object = asObject(value);
44 if (object->structure(vm)->masqueradesAsUndefined(globalObject))
45 return TriState::False;
46 return invert(object->isCallableWithConcurrency<concurrency>(vm));
47}

49template<Concurrency concurrency>
50ALWAYS_INLINE TriState jsTypeofIsFunctionWithConcurrency(JSGlobalObject* globalObject, JSValue value)
51{
52 VM& vm = globalObject->vm();
53 if (!value.isObject())
54 return TriState::False;
55 JSObject* object = asObject(value);
56 if (object->structure(vm)->masqueradesAsUndefined(globalObject))
57 return TriState::False;
58 return object->isCallableWithConcurrency<concurrency>(vm);
59}

80ALWAYS_INLINE JSString* jsString(JSGlobalObject* globalObject, const String& u1, JSString* s2)
81{
82 VM& vm = getVM(globalObject);
83 auto scope = DECLARE_THROW_SCOPE(vm);
84
85 unsigned length1 = u1.length();
86 if (!length1)
87 return s2;
88 unsigned length2 = s2->length();
89 if (!length2)
90 return jsString(vm, u1);
91 static_assert(JSString::MaxLength == std::numeric_limits<int32_t>::max(), "");
92 if (sumOverflows<int32_t>(length1, length2)) {
93 throwOutOfMemoryError(globalObject, scope);
94 return nullptr;
95 }
96
97 // (1) Cost of making JSString : sizeof(JSString) (for new string) + sizeof(StringImpl header) + length1 + length2
98 // (2) Cost of making JSRopeString: sizeof(JSString) (for u1) + sizeof(JSRopeString)
99 // We do not account u1 cost in (2) since u1 may be shared StringImpl, and it may not introduce additional cost.
100 // We conservatively consider the cost of u1. Currently, we are not considering about is8Bit() case because 16-bit
101 // strings are relatively rare. But we can do that if we need to consider it.
102 if (s2->isRope() || (StringImpl::headerSize<LChar>() + length1 + length2) >= sizeof(JSRopeString))
103 return JSRopeString::create(vm, jsString(vm, u1), s2);
104
105 ASSERT(!s2->isRope());
106 const String& u2 = s2->value(globalObject);
107 scope.assertNoException();
108 String newString = tryMakeString(u1, u2);
109 if (!newString) {
110 throwOutOfMemoryError(globalObject, scope);
111 return nullptr;
112 }
113 return JSString::create(vm, newString.releaseImpl().releaseNonNull());
114}

225ALWAYS_INLINE JSString* jsString(JSGlobalObject* globalObject, const String& u1, const String& u2, const String& u3)
226{
227 VM& vm = getVM(globalObject);
228 auto scope = DECLARE_THROW_SCOPE(vm);
229
230 unsigned length1 = u1.length();
231 unsigned length2 = u2.length();
232 unsigned length3 = u3.length();
233 ASSERT(length1 <= JSString::MaxLength);
234 ASSERT(length2 <= JSString::MaxLength);
235 ASSERT(length3 <= JSString::MaxLength);
236
237 if (!length1)
238 RELEASE_AND_RETURN(scope, jsString(globalObject, u2, u3));
239
240 if (!length2)
241 RELEASE_AND_RETURN(scope, jsString(globalObject, u1, u3));
242
243 if (!length3)
244 RELEASE_AND_RETURN(scope, jsString(globalObject, u1, u2));
245
246 static_assert(JSString::MaxLength == std::numeric_limits<int32_t>::max(), "");
247 if (sumOverflows<int32_t>(length1, length2, length3)) {
248 throwOutOfMemoryError(globalObject, scope);
249 return nullptr;
250 }
251
252 // (1) Cost of making JSString : sizeof(JSString) (for new string) + sizeof(StringImpl header) + length1 + length2 + length3
253 // (2) Cost of making JSRopeString: sizeof(JSString) (for u1) + sizeof(JSString) (for u2) + sizeof(JSString) (for u3) + sizeof(JSRopeString)
254 if ((StringImpl::headerSize<LChar>() + length1 + length2 + length3) >= (sizeof(JSRopeString) + sizeof(JSString) * 2))
255 return JSRopeString::create(vm, jsString(vm, u1), jsString(vm, u2), jsString(vm, u3));
256
257 String newString = tryMakeString(u1, u2, u3);
258 if (!newString) {
259 throwOutOfMemoryError(globalObject, scope);
260 return nullptr;
261 }
262 return JSString::create(vm, newString.releaseImpl().releaseNonNull());
263}

265ALWAYS_INLINE JSValue jsStringFromRegisterArray(JSGlobalObject* globalObject, Register* strings, unsigned count)
266{
267 VM& vm = getVM(globalObject);
268 auto scope = DECLARE_THROW_SCOPE(vm);
269 JSRopeString::RopeBuilder<RecordOverflow> ropeBuilder(vm);
270
271 for (unsigned i = 0; i < count; ++i) {
272 JSValue v = strings[-static_cast<int>(i)].jsValue();
273 JSString* string = v.toString(globalObject);
274 RETURN_IF_EXCEPTION(scope, { });
275 if (!ropeBuilder.append(string))
276 return throwOutOfMemoryError(globalObject, scope);
277 }
278
279 return ropeBuilder.release();
280}

282ALWAYS_INLINE JSBigInt::ComparisonResult compareBigInt(JSValue left, JSValue right)
283{
284 ASSERT(left.isBigInt());
285 ASSERT(right.isBigInt());
286#if USE(BIGINT32)
287 if (left.isBigInt32()) {
288 if (right.isBigInt32())
289 return JSBigInt::compare(left.bigInt32AsInt32(), right.bigInt32AsInt32());
290 ASSERT(right.isHeapBigInt());
291 return JSBigInt::compare(left.bigInt32AsInt32(), right.asHeapBigInt());
292 }
293 if (right.isBigInt32()) {
294 ASSERT(left.isHeapBigInt());
295 return JSBigInt::compare(left.asHeapBigInt(), right.bigInt32AsInt32());
296 }
297#endif
298 return JSBigInt::compare(left.asHeapBigInt(), right.asHeapBigInt());
299}

301ALWAYS_INLINE JSBigInt::ComparisonResult compareBigIntToOtherPrimitive(JSGlobalObject* globalObject, JSBigInt* v1, JSValue primValue)
302{
303 VM& vm = globalObject->vm();
304 auto scope = DECLARE_THROW_SCOPE(vm);
305
306 ASSERT(primValue.isPrimitive());
307 ASSERT(!primValue.isBigInt());
308
309 if (primValue.isString()) {
310 String string = asString(primValue)->value(globalObject);
311 RETURN_IF_EXCEPTION(scope, JSBigInt::ComparisonResult::Undefined);
312 JSValue bigIntValue = JSBigInt::stringToBigInt(globalObject, string);
313 RETURN_IF_EXCEPTION(scope, JSBigInt::ComparisonResult::Undefined);
314 if (!bigIntValue)
315 return JSBigInt::ComparisonResult::Undefined;
316
317 if (bigIntValue.isHeapBigInt())
318 return JSBigInt::compare(v1, bigIntValue.asHeapBigInt());
319
320 ASSERT(bigIntValue.isBigInt32());
321#if USE(BIGINT32)
322 // FIXME: use something less hacky, e.g. some kind of JSBigInt::compareToInt32
323 return JSBigInt::compareToDouble(v1, static_cast<double>(bigIntValue.bigInt32AsInt32()));
324#endif
325 }
326
327 double numberValue = primValue.toNumber(globalObject);
328 RETURN_IF_EXCEPTION(scope, JSBigInt::ComparisonResult::Undefined);
329 return JSBigInt::compareToDouble(v1, numberValue);
330}

332ALWAYS_INLINE JSBigInt::ComparisonResult compareBigInt32ToOtherPrimitive(JSGlobalObject* globalObject, int32_t v1, JSValue primValue)
333{
334 VM& vm = globalObject->vm();
335 auto scope = DECLARE_THROW_SCOPE(vm);
336
337 ASSERT(primValue.isPrimitive());
338 ASSERT(!primValue.isBigInt());
339
340 auto compare = [&] (auto v1, auto v2) {
341 static_assert(std::is_same_v<decltype(v1), decltype(v2)>);
342 if (v1 == v2)
343 return JSBigInt::ComparisonResult::Equal;
344 if (v1 < v2)
345 return JSBigInt::ComparisonResult::LessThan;
346 return JSBigInt::ComparisonResult::GreaterThan;
347 };
348
349 if (primValue.isString()) {
350 String string = asString(primValue)->value(globalObject);
351 RETURN_IF_EXCEPTION(scope, JSBigInt::ComparisonResult::Undefined);
352 JSValue bigIntValue = JSBigInt::stringToBigInt(globalObject, string);
353 RETURN_IF_EXCEPTION(scope, JSBigInt::ComparisonResult::Undefined);
354 if (!bigIntValue)
355 return JSBigInt::ComparisonResult::Undefined;
356
357 if (bigIntValue.isHeapBigInt())
358 return JSBigInt::compare(v1, bigIntValue.asHeapBigInt());
359
360 ASSERT(bigIntValue.isBigInt32());
361#if USE(BIGINT32)
362 return compare(v1, bigIntValue.bigInt32AsInt32());
363#endif
364 }
365
366 // Note that 0n <=> -0.0 is handling -0.0 as +0.
367 double numberValue = primValue.toNumber(globalObject);
368 RETURN_IF_EXCEPTION(scope, JSBigInt::ComparisonResult::Undefined);
369 if (std::isnan(numberValue))
370 return JSBigInt::ComparisonResult::Undefined;
371 return compare(static_cast<double>(v1), numberValue);
372}

374ALWAYS_INLINE bool bigIntCompareResult(JSBigInt::ComparisonResult comparisonResult, JSBigInt::ComparisonMode comparisonMode)
375{
376 if (comparisonMode == JSBigInt::ComparisonMode::LessThan)
377 return comparisonResult == JSBigInt::ComparisonResult::LessThan;
378
379 ASSERT(comparisonMode == JSBigInt::ComparisonMode::LessThanOrEqual);
380 return comparisonResult == JSBigInt::ComparisonResult::LessThan || comparisonResult == JSBigInt::ComparisonResult::Equal;
381}

383ALWAYS_INLINE bool bigIntCompare(JSGlobalObject* globalObject, JSValue v1, JSValue v2, JSBigInt::ComparisonMode comparisonMode)
384{
385 ASSERT(v1.isBigInt() || v2.isBigInt());
386 ASSERT(v1.isPrimitive() && v2.isPrimitive());
387
388 VM& vm = globalObject->vm();
389 auto scope = DECLARE_THROW_SCOPE(vm);
390
391 if (v1.isBigInt() && v2.isBigInt())
392 return bigIntCompareResult(compareBigInt(v1, v2), comparisonMode);
393
394#if USE(BIGINT32)
395 if (v1.isBigInt32()) {
396 ASSERT(!v2.isBigInt());
397 auto comparisonResult = compareBigInt32ToOtherPrimitive(globalObject, v1.bigInt32AsInt32(), v2);
398 RETURN_IF_EXCEPTION(scope, false);
399 return bigIntCompareResult(comparisonResult, comparisonMode);
400 }
401 if (v2.isBigInt32()) {
402 ASSERT(!v1.isBigInt());
403 auto comparisonResult = compareBigInt32ToOtherPrimitive(globalObject, v2.bigInt32AsInt32(), v1);
404 RETURN_IF_EXCEPTION(scope, false);
405 return bigIntCompareResult(invertBigIntCompareResult(comparisonResult), comparisonMode);
406 }
407 ASSERT(!v1.isBigInt32() && !v2.isBigInt32());
408#endif
409
410 if (v1.isHeapBigInt()) {
411 ASSERT(!v2.isBigInt());
412 auto comparisonResult = compareBigIntToOtherPrimitive(globalObject, v1.asHeapBigInt(), v2);
413 RETURN_IF_EXCEPTION(scope, false);
414 return bigIntCompareResult(comparisonResult, comparisonMode);
415 }
416
417 // Here we check inverted because BigInt is the v2
418 ASSERT(!v1.isBigInt());
419 ASSERT(v2.isHeapBigInt());
420 auto comparisonResult = compareBigIntToOtherPrimitive(globalObject, v2.asHeapBigInt(), v1);
421 RETURN_IF_EXCEPTION(scope, false);
422 return bigIntCompareResult(invertBigIntCompareResult(comparisonResult), comparisonMode);
423}

425ALWAYS_INLINE bool toPrimitiveNumeric(JSGlobalObject* globalObject, JSValue v, JSValue& p, double& n)
426{
427 VM& vm = globalObject->vm();
428 auto scope = DECLARE_THROW_SCOPE(vm);
429
430 p = v.toPrimitive(globalObject, PreferNumber);
431 RETURN_IF_EXCEPTION(scope, false);
432 if (p.isBigInt())
433 return true;
434
435 n = p.toNumber(globalObject);
436 RETURN_IF_EXCEPTION(scope, false);
437 return !p.isString();
438}

440// See ES5 11.8.1/11.8.2/11.8.5 for definition of leftFirst, this value ensures correct
441// evaluation ordering for argument conversions for '<' and '>'. For '<' pass the value
442// true, for leftFirst, for '>' pass the value false (and reverse operand order).
443template<bool leftFirst>
444ALWAYS_INLINE bool jsLess(JSGlobalObject* globalObject, JSValue v1, JSValue v2)
445{
446 VM& vm = globalObject->vm();
447 auto scope = DECLARE_THROW_SCOPE(vm);
448
449 if (v1.isInt32() && v2.isInt32())
450 return v1.asInt32() < v2.asInt32();
451
452 if (v1.isNumber() && v2.isNumber())
453 return v1.asNumber() < v2.asNumber();
454
455 if (isJSString(v1) && isJSString(v2)) {
456 String s1 = asString(v1)->value(globalObject);
457 RETURN_IF_EXCEPTION(scope, false);
458 String s2 = asString(v2)->value(globalObject);
459 RETURN_IF_EXCEPTION(scope, false);
460 return codePointCompareLessThan(s1, s2);
461 }
462
463 double n1;
464 double n2;
465 JSValue p1;
466 JSValue p2;
467 bool wasNotString1;
468 bool wasNotString2;
469 if (leftFirst) {
470 wasNotString1 = toPrimitiveNumeric(globalObject, v1, p1, n1);
471 RETURN_IF_EXCEPTION(scope, false);
472 wasNotString2 = toPrimitiveNumeric(globalObject, v2, p2, n2);
473 } else {
474 wasNotString2 = toPrimitiveNumeric(globalObject, v2, p2, n2);
475 RETURN_IF_EXCEPTION(scope, false);
476 wasNotString1 = toPrimitiveNumeric(globalObject, v1, p1, n1);
477 }
478 RETURN_IF_EXCEPTION(scope, false);
479
480 if (wasNotString1 || wasNotString2) {
481 if (p1.isBigInt() || p2.isBigInt())
482 RELEASE_AND_RETURN(scope, bigIntCompare(globalObject, p1, p2, JSBigInt::ComparisonMode::LessThan));
483
484 return n1 < n2;
485 }
486
487 return codePointCompareLessThan(asString(p1)->value(globalObject), asString(p2)->value(globalObject));
488}
489
490// See ES5 11.8.3/11.8.4/11.8.5 for definition of leftFirst, this value ensures correct
491// evaluation ordering for argument conversions for '<=' and '=>'. For '<=' pass the
492// value true, for leftFirst, for '=>' pass the value false (and reverse operand order).
493template<bool leftFirst>
494ALWAYS_INLINE bool jsLessEq(JSGlobalObject* globalObject, JSValue v1, JSValue v2)
495{
496 VM& vm = globalObject->vm();
497 auto scope = DECLARE_THROW_SCOPE(vm);
498
499 if (v1.isInt32() && v2.isInt32())
500 return v1.asInt32() <= v2.asInt32();
501
502 if (v1.isNumber() && v2.isNumber())
503 return v1.asNumber() <= v2.asNumber();
504
505 if (isJSString(v1) && isJSString(v2)) {
506 String s1 = asString(v1)->value(globalObject);
507 RETURN_IF_EXCEPTION(scope, false);
508 String s2 = asString(v2)->value(globalObject);
509 RETURN_IF_EXCEPTION(scope, false);
510 return !codePointCompareLessThan(s2, s1);
511 }
512
513 double n1;
514 double n2;
515 JSValue p1;
516 JSValue p2;
517 bool wasNotString1;
518 bool wasNotString2;
519 if (leftFirst) {
520 wasNotString1 = toPrimitiveNumeric(globalObject, v1, p1, n1);
521 RETURN_IF_EXCEPTION(scope, false);
522 wasNotString2 = toPrimitiveNumeric(globalObject, v2, p2, n2);
523 } else {
524 wasNotString2 = toPrimitiveNumeric(globalObject, v2, p2, n2);
525 RETURN_IF_EXCEPTION(scope, false);
526 wasNotString1 = toPrimitiveNumeric(globalObject, v1, p1, n1);
527 }
528 RETURN_IF_EXCEPTION(scope, false);
529
530 if (wasNotString1 || wasNotString2) {
531 if (p1.isBigInt() || p2.isBigInt())
532 RELEASE_AND_RETURN(scope, bigIntCompare(globalObject, p1, p2, JSBigInt::ComparisonMode::LessThanOrEqual));
533
534 return n1 <= n2;
535 }
536 return !codePointCompareLessThan(asString(p2)->value(globalObject), asString(p1)->value(globalObject));
537}

539// Fast-path choices here are based on frequency data from SunSpider:
540// <times> Add case: <t1> <t2>
541// ---------------------------
542// 5626160 Add case: 3 3 (of these, 3637690 are for immediate values)
543// 247412 Add case: 5 5
544// 20900 Add case: 5 6
545// 13962 Add case: 5 3
546// 4000 Add case: 3 5

549ALWAYS_INLINE JSValue jsAddNonNumber(JSGlobalObject* globalObject, JSValue v1, JSValue v2)
550{
551 VM& vm = globalObject->vm();
552 auto scope = DECLARE_THROW_SCOPE(vm);
553 ASSERT(!v1.isNumber() || !v2.isNumber());
554
555 if (LIKELY(v1.isString() && !v2.isObject())) {
556 if (v2.isString())
557 RELEASE_AND_RETURN(scope, jsString(globalObject, asString(v1), asString(v2)));
558 String s2 = v2.toWTFString(globalObject);
559 RETURN_IF_EXCEPTION(scope, { });
560 RELEASE_AND_RETURN(scope, jsString(globalObject, asString(v1), s2));
561 }
562
563 // All other cases are pretty uncommon
564 RELEASE_AND_RETURN(scope, jsAddSlowCase(globalObject, v1, v2));
565}

575template<typename DoubleOperation, typename BigIntOp>
576ALWAYS_INLINE JSValue arithmeticBinaryOp(JSGlobalObject* globalObject, JSValue v1, JSValue v2, DoubleOperation&& doubleOp, BigIntOp&& bigIntOp, const char* errorMessage)
577{
578 VM& vm = globalObject->vm();
579 auto scope = DECLARE_THROW_SCOPE(vm);
580
581 JSValue leftNumeric = v1.toNumeric(globalObject);
582 RETURN_IF_EXCEPTION(scope, { });
583 JSValue rightNumeric = v2.toNumeric(globalObject);
584 RETURN_IF_EXCEPTION(scope, { });
585
586 if (leftNumeric.isNumber() && rightNumeric.isNumber())
587 return jsNumber(doubleOp(leftNumeric.asNumber(), rightNumeric.asNumber()));
588
589#if USE(BIGINT32)
590 if (leftNumeric.isBigInt32()) {
591 if (rightNumeric.isBigInt32())
592 RELEASE_AND_RETURN(scope, bigIntOp(globalObject, leftNumeric.bigInt32AsInt32(), rightNumeric.bigInt32AsInt32()));
593 if (rightNumeric.isHeapBigInt())
594 RELEASE_AND_RETURN(scope, bigIntOp(globalObject, leftNumeric.bigInt32AsInt32(), rightNumeric.asHeapBigInt()));
595 } else if (leftNumeric.isHeapBigInt()) {

734template <bool isLeft>
735ALWAYS_INLINE JSValue shift(JSGlobalObject* globalObject, JSValue v1, JSValue v2)
736{
737 VM& vm = globalObject->vm();
738 auto scope = DECLARE_THROW_SCOPE(vm);
739
740 auto leftNumeric = v1.toBigIntOrInt32(globalObject);
741 RETURN_IF_EXCEPTION(scope, { });
742 auto rightNumeric = v2.toBigIntOrInt32(globalObject);
743 RETURN_IF_EXCEPTION(scope, { });
744
745 if (leftNumeric.isInt32() && rightNumeric.isInt32()) {
746 int32_t leftInt32 = leftNumeric.asInt32();
747 int32_t rightInt32 = rightNumeric.asInt32() & 31;
748 int32_t result = isLeft ? (leftInt32 << rightInt32) : (leftInt32 >> rightInt32);
749 return jsNumber(result);
750 }
751
752#if USE(BIGINT32)
753 if (leftNumeric.isBigInt32()) {
754 if (rightNumeric.isBigInt32()) {
755 if (isLeft)
756 RELEASE_AND_RETURN(scope, JSBigInt::leftShift(globalObject, leftNumeric.bigInt32AsInt32(), rightNumeric.bigInt32AsInt32()));
757 RELEASE_AND_RETURN(scope, JSBigInt::signedRightShift(globalObject, leftNumeric.bigInt32AsInt32(), rightNumeric.bigInt32AsInt32()));
758 }
759 if (rightNumeric.isHeapBigInt()) {
760 if (isLeft)
761 RELEASE_AND_RETURN(scope, JSBigInt::leftShift(globalObject, leftNumeric.bigInt32AsInt32(), rightNumeric.asHeapBigInt()));
762 RELEASE_AND_RETURN(scope, JSBigInt::signedRightShift(globalObject, leftNumeric.bigInt32AsInt32(), rightNumeric.asHeapBigInt()));
763 }
764 } else if (leftNumeric.isHeapBigInt()) {
765 if (rightNumeric.isBigInt32()) {
766 if (isLeft)
767 RELEASE_AND_RETURN(scope, JSBigInt::leftShift(globalObject, leftNumeric.asHeapBigInt(), rightNumeric.bigInt32AsInt32()));
768 RELEASE_AND_RETURN(scope, JSBigInt::signedRightShift(globalObject, leftNumeric.asHeapBigInt(), rightNumeric.bigInt32AsInt32()));
769 } else if (rightNumeric.isHeapBigInt()) {
770 if (isLeft)
771 RELEASE_AND_RETURN(scope, JSBigInt::leftShift(globalObject, leftNumeric.asHeapBigInt(), rightNumeric.asHeapBigInt()));
772 RELEASE_AND_RETURN(scope, JSBigInt::signedRightShift(globalObject, leftNumeric.asHeapBigInt(), rightNumeric.asHeapBigInt()));
773 }
774 }
775#else
776 if (leftNumeric.isHeapBigInt() && rightNumeric.isHeapBigInt()) {
777 if (isLeft)
778 RELEASE_AND_RETURN(scope, JSBigInt::leftShift(globalObject, leftNumeric.asHeapBigInt(), rightNumeric.asHeapBigInt()));
779 RELEASE_AND_RETURN(scope, JSBigInt::signedRightShift(globalObject, leftNumeric.asHeapBigInt(), rightNumeric.asHeapBigInt()));
780 }
781#endif
782
783 auto errorMessage = isLeft ? "Invalid mix of BigInt and other type in left shift operation." : "Invalid mix of BigInt and other type in signed right shift operation.";
784 return throwTypeError(globalObject, scope, errorMessage);
785}

799ALWAYS_INLINE JSValue jsURShift(JSGlobalObject* globalObject, JSValue left, JSValue right)
800{
801 VM& vm = globalObject->vm();
802 auto scope = DECLARE_THROW_SCOPE(vm);
803
804 std::optional<uint32_t> leftUint32 = left.toUInt32AfterToNumeric(globalObject);
805 RETURN_IF_EXCEPTION(scope, { });
806 std::optional<uint32_t> rightUint32 = right.toUInt32AfterToNumeric(globalObject);
807 RETURN_IF_EXCEPTION(scope, { });
808
809 if (UNLIKELY(!leftUint32 || !rightUint32)) {
810 throwTypeError(globalObject, scope, "BigInt does not support >>> operator"_s);
811 return { };
812 }
813
814 return jsNumber(static_cast<int32_t>(leftUint32.value() >> (rightUint32.value() & 31)));
815}
