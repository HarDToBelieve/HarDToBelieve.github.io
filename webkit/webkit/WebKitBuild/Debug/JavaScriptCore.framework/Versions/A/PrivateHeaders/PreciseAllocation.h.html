31namespace JSC {
32
33class IsoSubspace;
34class SlotVisitor;

36// WebKit has a good malloc that already knows what to do for large allocations. The GC shouldn't
37// have to think about such things. That's where PreciseAllocation comes in. We will allocate large
38// objects directly using malloc, and put the PreciseAllocation header just before them. We can detect
39// when a HeapCell* is a PreciseAllocation because it will have the MarkedBlock::atomSize / 2 bit set.
40
41class PreciseAllocation : public PackedRawSentinelNode<PreciseAllocation> {
42public:
43 friend class LLIntOffsetsExtractor;
44 friend class IsoSubspace;
45
46 static PreciseAllocation* tryCreate(Heap&, size_t, Subspace*, unsigned indexInSpace);
47
48 static PreciseAllocation* createForLowerTier(Heap&, size_t, Subspace*, uint8_t lowerTierIndex);
49 PreciseAllocation* reuseForLowerTier();
50
51 PreciseAllocation* tryReallocate(size_t, Subspace*);
52
53 ~PreciseAllocation();
54
55 static PreciseAllocation* fromCell(const void* cell)
56 {
57 return bitwise_cast<PreciseAllocation*>(bitwise_cast<char*>(cell) - headerSize());
58 }
59
60 HeapCell* cell() const
61 {
62 return bitwise_cast<HeapCell*>(bitwise_cast<char*>(this) + headerSize());
63 }
64
65 static bool isPreciseAllocation(HeapCell* cell)
66 {
67 return bitwise_cast<uintptr_t>(cell) & halfAlignment;
68 }
69
70 Subspace* subspace() const { return m_subspace; }
71
72 void lastChanceToFinalize();
73
74 Heap* heap() const { return m_weakSet.heap(); }
75 VM& vm() const { return m_weakSet.vm(); }
76 WeakSet& weakSet() { return m_weakSet; }
77
78 static ptrdiff_t offsetOfWeakSet() { return OBJECT_OFFSETOF(PreciseAllocation, m_weakSet); }
79
80 unsigned indexInSpace() { return m_indexInSpace; }
81 void setIndexInSpace(unsigned indexInSpace) { m_indexInSpace = indexInSpace; }
82
83 void clearNewlyAllocated() { m_isNewlyAllocated = false; }
84 void flip();
85
86 bool isNewlyAllocated() const { return m_isNewlyAllocated; }
87 ALWAYS_INLINE bool isMarked() { return m_isMarked.load(std::memory_order_relaxed); }
88 ALWAYS_INLINE bool isMarked(HeapCell*) { return isMarked(); }
89 ALWAYS_INLINE bool isMarked(HeapCell*, Dependency) { return isMarked(); }
90 ALWAYS_INLINE bool isMarked(HeapVersion, HeapCell*) { return isMarked(); }
91 bool isLive() { return isMarked() || isNewlyAllocated(); }
92
93 bool hasValidCell() const { return m_hasValidCell; }
94
95 bool isEmpty();
96
97 size_t cellSize() const { return m_cellSize; }
98
99 uint8_t lowerTierIndex() const { return m_lowerTierIndex; }
100
101 bool aboveLowerBound(const void* rawPtr)
102 {
103 char* ptr = bitwise_cast<char*>(rawPtr);
104 char* begin = bitwise_cast<char*>(cell());
105 return ptr >= begin;
106 }
107
108 bool belowUpperBound(const void* rawPtr)
109 {
110 char* ptr = bitwise_cast<char*>(rawPtr);
111 char* begin = bitwise_cast<char*>(cell());
112 char* end = begin + cellSize();
113 // We cannot #include IndexingHeader.h because reasons. The fact that IndexingHeader is 8
114 // bytes is wired deep into our engine, so this isn't so bad.
115 size_t sizeOfIndexingHeader = 8;
116 return ptr <= end + sizeOfIndexingHeader;
117 }
118
119 bool contains(const void* rawPtr)
120 {
121 return aboveLowerBound(rawPtr) && belowUpperBound(rawPtr);
122 }
123
124 const CellAttributes& attributes() const { return m_attributes; }
125
126 Dependency aboutToMark(HeapVersion) { return Dependency(); }
127
128 ALWAYS_INLINE bool testAndSetMarked()
129 {
130 // This method is usually called when the object is already marked. This avoids us
131 // having to CAS in that case. It's profitable to reduce the total amount of CAS
132 // traffic.
133 if (isMarked())
134 return true;
135 return m_isMarked.compareExchangeStrong(false, true);
136 }
137 ALWAYS_INLINE bool testAndSetMarked(HeapCell*, Dependency) { return testAndSetMarked(); }
138 void clearMarked() { m_isMarked.store(false); }
139
140 void noteMarked() { }
141
142#if ASSERT_ENABLED
143 void assertValidCell(VM&, HeapCell*) const;
144#else
145 void assertValidCell(VM&, HeapCell*) const { }
146#endif
147
148 void sweep();
149
150 void destroy();
151
152 void dump(PrintStream&) const;
153
154 bool isLowerTier() const { return m_lowerTierIndex != UINT8_MAX; }
155
156 static constexpr unsigned alignment = MarkedBlock::atomSize;
157 static constexpr unsigned halfAlignment = alignment / 2;
158 static constexpr unsigned headerSize() { return ((sizeof(PreciseAllocation) + halfAlignment - 1) & ~(halfAlignment - 1)) | halfAlignment; }
159
160private:
161 PreciseAllocation(Heap&, size_t, Subspace*, unsigned indexInSpace, bool adjustedAlignment);
162
163 void* basePointer() const;
164
165 unsigned m_indexInSpace { 0 };
166 size_t m_cellSize;
167 bool m_isNewlyAllocated : 1;
168 bool m_hasValidCell : 1;
169 bool m_adjustedAlignment : 1;
170 Atomic<bool> m_isMarked;
171 CellAttributes m_attributes;
172 uint8_t m_lowerTierIndex { UINT8_MAX };
173 Subspace* m_subspace;
174 WeakSet m_weakSet;
175};
176
177inline void* PreciseAllocation::basePointer() const
178{
179 if (m_adjustedAlignment)
180 return bitwise_cast<char*>(this) - halfAlignment;
181 return bitwise_cast<void*>(this);
182}
