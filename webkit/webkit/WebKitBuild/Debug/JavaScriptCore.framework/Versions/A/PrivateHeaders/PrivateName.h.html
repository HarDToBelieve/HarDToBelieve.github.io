26#pragma once
27
28#include <wtf/text/SymbolImpl.h>
29#include <wtf/text/WTFString.h>
30
31namespace JSC {
32
33class PrivateName {
34public:
35 PrivateName()
36 : m_uid(SymbolImpl::createNullSymbol())
37 {
38 }
39
40 explicit PrivateName(SymbolImpl& uid)
41 : m_uid(uid)
42 {
43 }
44
45 enum DescriptionTag { Description };
46 explicit PrivateName(DescriptionTag, const String& description)
47 : m_uid(SymbolImpl::create(*description.impl()))
48 {
49 }
50
51 enum PrivateSymbolTag { PrivateSymbol };
52 explicit PrivateName(PrivateSymbolTag, const String& description)
53 : m_uid(PrivateSymbolImpl::create(*description.impl()))
54 {
55 }
56
57 PrivateName(const PrivateName& privateName)
58 : m_uid(privateName.m_uid.copyRef())
59 {
60 }
61
62 PrivateName(PrivateName&&) = default;
63
64 SymbolImpl& uid() const { return m_uid; }
65
66 bool operator==(const PrivateName& other) const { return &uid() == &other.uid(); }
67 bool operator!=(const PrivateName& other) const { return &uid() != &other.uid(); }
68
69private:
70 Ref<SymbolImpl> m_uid;
71};
72
73} // namespace JSC
