26#pragma once
27
28#include "JSCJSValue.h"
29#include <wtf/text/CString.h>
30
31namespace JSC {
32
33enum OpcodeID : unsigned;
34
35namespace Profiler {
36
37class Bytecode {
38public:
39 Bytecode()
40 : m_bytecodeIndex(std::numeric_limits<unsigned>::max())
41 {
42 }
43
44 Bytecode(unsigned bytecodeIndex, OpcodeID opcodeID, const CString& description)
45 : m_bytecodeIndex(bytecodeIndex)
46 , m_opcodeID(opcodeID)
47 , m_description(description)
48 {
49 }
50
51 unsigned bytecodeIndex() const { return m_bytecodeIndex; }
52 OpcodeID opcodeID() const { return m_opcodeID; }
53 const CString& description() const { return m_description; }
54
55 JSValue toJS(JSGlobalObject*) const;
56private:
57 unsigned m_bytecodeIndex;
58 OpcodeID m_opcodeID;
59 CString m_description;
60};
61
62inline unsigned getBytecodeIndexForBytecode(Bytecode* bytecode) { return bytecode->bytecodeIndex(); }
63
64} } // namespace JSC::Profiler
