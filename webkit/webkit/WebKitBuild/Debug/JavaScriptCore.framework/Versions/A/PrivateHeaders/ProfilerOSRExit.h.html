28#include "ExitKind.h"

32namespace JSC { namespace Profiler {
33
34class OSRExit {
35public:
36 OSRExit(unsigned id, const OriginStack&, ExitKind, bool isWatchpoint);
37 ~OSRExit();
38
39 unsigned id() const { return m_id; }
40 const OriginStack& origin() const { return m_origin; }
41 ExitKind exitKind() const { return m_exitKind; }
42 bool isWatchpoint() const { return m_isWatchpoint; }
43
44 uint64_t* counterAddress() { return &m_counter; }
45 uint64_t count() const { return m_counter; }
46 void incCount() { m_counter++; }
47
48 JSValue toJS(JSGlobalObject*) const;
49
50private:
51 OriginStack m_origin;
52 unsigned m_id;
53 ExitKind m_exitKind;
54 bool m_isWatchpoint;
55 uint64_t m_counter;
56};

58} } // namespace JSC::Profiler
