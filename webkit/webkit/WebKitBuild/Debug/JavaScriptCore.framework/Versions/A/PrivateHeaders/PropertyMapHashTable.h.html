59inline bool isPowerOf2(unsigned v)
60{
61 return hasOneBitSet(v);
62}
63
64inline unsigned nextPowerOf2(unsigned v)
65{
66 // Taken from http://www.cs.utk.edu/~vose/c-stuff/bithacks.html
67 // Devised by Sean Anderson, Sepember 14, 2001
68
69 v--;
70 v |= v >> 1;
71 v |= v >> 2;
72 v |= v >> 4;
73 v |= v >> 8;
74 v |= v >> 16;
75 v++;
76
77 return v;
78}

82 // This is the implementation for 'iterator' and 'const_iterator',
83 // used for iterating over the table in insertion order.
84 template<typename T>
85 class ordered_iterator {
86 public:
87 ordered_iterator<T>& operator++()
88 {
89 m_valuePtr = skipDeletedEntries(m_valuePtr + 1, m_endValuePtr);
90 return *this;
91 }
92
93 bool operator==(const ordered_iterator<T>& other)
94 {
95 return m_valuePtr == other.m_valuePtr;
96 }
97
98 bool operator!=(const ordered_iterator<T>& other)
99 {
100 return m_valuePtr != other.m_valuePtr;
101 }
102
103 T& operator*()
104 {
105 return *m_valuePtr;
106 }
107
108 T* operator->()
109 {
110 return m_valuePtr;
111 }
112
113 ordered_iterator(T* valuePtr, T* endValuePtr)
114 : m_valuePtr(valuePtr)
115 , m_endValuePtr(endValuePtr)
116 {
117 }
118
119 private:
120 T* m_valuePtr;
121 T* m_endValuePtr;
122 };

224 // The capacity of the table of values is half of the size of the index.

226
227 // We keep an extra deleted slot after the array to make iteration work,
228 // and to use for deleted values. Index values into the array are 1-based,
229 // so this is tableCapacity() + 1.
230 // For example, if m_tableSize is 16, then tableCapacity() is 8 - but the
231 // values array is actually 9 long (the 9th used for the deleted value/
232 // iteration guard). The 8 valid entries are numbered 1..8, so the
233 // deleted index is 9 (0 being reserved for empty).
234 unsigned deletedEntryIndex() const;

293inline PropertyTable::find_iterator PropertyTable::find(const KeyType& key)
294{
295 ASSERT(key);
296 ASSERT(key->isAtom() || key->isSymbol());
297 unsigned hash = IdentifierRepHash::hash(key);
298
299#if DUMP_PROPERTYMAP_STATS
300 ++propertyMapHashTableStats->numFinds;
301#endif
302
303 while (true) {
304 unsigned entryIndex = m_index[hash & m_indexMask];
305 if (entryIndex == EmptyEntryIndex)
306 return std::make_pair((ValueType*)nullptr, hash & m_indexMask);
307 if (key == table()[entryIndex - 1].key)
308 return std::make_pair(&table()[entryIndex - 1], hash & m_indexMask);
309
310#if DUMP_PROPERTYMAP_STATS
311 ++propertyMapHashTableStats->numCollisions;
312#endif
313
314#if DUMP_PROPERTYMAP_COLLISIONS
315 dataLog("PropertyTable collision for ", key, " (", hash, ")\n");
316 dataLog("Collided with ", table()[entryIndex - 1].key, "(", IdentifierRepHash::hash(table()[entryIndex - 1].key), ")\n");
317#endif
318
319 hash++;
320 }
321}

323inline PropertyTable::ValueType* PropertyTable::get(const KeyType& key)
324{
325 ASSERT(key);
326 ASSERT(key->isAtom() || key->isSymbol());
327 ASSERT(key != PROPERTY_MAP_DELETED_ENTRY_KEY);
328
329 if (!m_keyCount)
330 return nullptr;
331
332 unsigned hash = IdentifierRepHash::hash(key);
333
334#if DUMP_PROPERTYMAP_STATS
335 ++propertyMapHashTableStats->numLookups;
336#endif
337
338 while (true) {
339 unsigned entryIndex = m_index[hash & m_indexMask];
340 if (entryIndex == EmptyEntryIndex)
341 return nullptr;
342 if (key == table()[entryIndex - 1].key) {
343 ASSERT(!m_deletedOffsets || !m_deletedOffsets->contains(table()[entryIndex - 1].offset));
344 return &table()[entryIndex - 1];
345 }
346
347#if DUMP_PROPERTYMAP_STATS
348 ++propertyMapHashTableStats->numLookupProbing;
349#endif
350
351 hash++;
352 }
353}

355inline std::pair<PropertyTable::find_iterator, bool> WARN_UNUSED_RETURN PropertyTable::add(VM& vm, const ValueType& entry)
356{
357 ASSERT(!m_deletedOffsets || !m_deletedOffsets->contains(entry.offset));
358
359 // Look for a value with a matching key already in the array.
360 find_iterator iter = find(entry.key);
361 if (iter.first)
362 return std::make_pair(iter, false);
363
364#if DUMP_PROPERTYMAP_STATS
365 ++propertyMapHashTableStats->numAdds;
366#endif
367
368 // Ref the key
369 entry.key->ref();
370
371 // ensure capacity is available.
372 if (!canInsert()) {
373 rehash(vm, m_keyCount + 1);
374 iter = find(entry.key);
375 ASSERT(!iter.first);
376 }
377
378 // Allocate a slot in the hashtable, and set the index to reference this.
379 unsigned entryIndex = usedCount() + 1;
380 m_index[iter.second] = entryIndex;
381 iter.first = &table()[entryIndex - 1];
382 *iter.first = entry;
383
384 ++m_keyCount;
385
386 return std::make_pair(iter, true);
387}

389inline void PropertyTable::remove(VM& vm, const find_iterator& iter)
390{
391 // Removing a key that doesn't exist does nothing!
392 if (!iter.first)
393 return;
394
395#if DUMP_PROPERTYMAP_STATS
396 ++propertyMapHashTableStats->numRemoves;
397#endif
398
399 // Replace this one element with the deleted sentinel. Also clear out
400 // the entry so we can iterate all the entries as needed.
401 m_index[iter.second] = deletedEntryIndex();
402 iter.first->key->deref();
403 iter.first->key = PROPERTY_MAP_DELETED_ENTRY_KEY;
404
405 ASSERT(m_keyCount >= 1);
406 --m_keyCount;
407 ++m_deletedCount;
408
409 if (m_deletedCount * 4 >= m_indexSize)
410 rehash(vm, m_keyCount);
411}
412
413inline void PropertyTable::remove(VM& vm, const KeyType& key)
414{
415 remove(vm, find(key));
416}

467inline PropertyTable* PropertyTable::copy(VM& vm, unsigned newCapacity)
468{
469 ASSERT(newCapacity >= m_keyCount);
470
471 // Fast case; if the new table will be the same m_indexSize as this one, we can memcpy it,
472 // save rehashing all keys.
473 if (sizeForCapacity(newCapacity) == m_indexSize)
474 return PropertyTable::clone(vm, *this);
475 return PropertyTable::clone(vm, newCapacity, *this);
476}

488inline void PropertyTable::reinsert(const ValueType& entry)
489{
490#if DUMP_PROPERTYMAP_STATS
491 ++propertyMapHashTableStats->numReinserts;
492#endif
493
494 // Used to insert a value known not to be in the table, and where
495 // we know capacity to be available.
496 ASSERT(canInsert());
497 find_iterator iter = find(entry.key);
498 ASSERT(!iter.first);
499
500 unsigned entryIndex = usedCount() + 1;
501 m_index[iter.second] = entryIndex;
502 table()[entryIndex - 1] = entry;
503
504 ++m_keyCount;
505}
506
507inline void PropertyTable::rehash(VM& vm, unsigned newCapacity)
508{
509#if DUMP_PROPERTYMAP_STATS
510 ++propertyMapHashTableStats->numRehashes;
511#endif
512
513 size_t oldDataSize = dataSize();
514 unsigned* oldEntryIndices = m_index;
515 iterator iter = this->begin();
516 iterator end = this->end();
517
518 m_indexSize = sizeForCapacity(newCapacity);
519 m_indexMask = m_indexSize - 1;
520 m_keyCount = 0;
521 m_deletedCount = 0;
522
523 m_index = static_cast<unsigned*>(PropertyTableMalloc::zeroedMalloc(dataSize()));
524
525 for (; iter != end; ++iter) {
526 ASSERT(canInsert());
527 reinsert(*iter);
528 }
529
530 PropertyTableMalloc::free(oldEntryIndices);
531
532 if (oldDataSize < dataSize())
533 vm.heap.reportExtraMemoryAllocated(dataSize() - oldDataSize);
534}

536inline unsigned PropertyTable::tableCapacity() const { return m_indexSize >> 1; }

538inline unsigned PropertyTable::deletedEntryIndex() const { return tableCapacity() + 1; }

540template<typename T>
541inline T* PropertyTable::skipDeletedEntries(T* valuePtr, T* endValuePtr)
542{
543 while (valuePtr < endValuePtr && valuePtr->key == PROPERTY_MAP_DELETED_ENTRY_KEY)
544 ++valuePtr;
545 return valuePtr;
546}

548inline PropertyTable::ValueType* PropertyTable::table()

560inline unsigned PropertyTable::usedCount() const
561{
562 // Total number of used entries in the values array - by either valid entries, or deleted ones.
563 return m_keyCount + m_deletedCount;
564}
565
566inline size_t PropertyTable::dataSize()
567{
568 // The size in bytes of data needed for by the table.
569 // Ensure that this function can be called concurrently.
570 unsigned indexSize = m_indexSize;
571 return indexSize * sizeof(unsigned) + ((indexSize >> 1) + 1) * sizeof(ValueType);
572}
573
574inline unsigned PropertyTable::sizeForCapacity(unsigned capacity)
575{
576 if (capacity < MinimumTableSize / 2)
577 return MinimumTableSize;
578 return nextPowerOf2(capacity + 1) * 2;
579}
580
581inline bool PropertyTable::canInsert()
582{
583 return usedCount() < tableCapacity();
584}
