26#pragma once
27
28#include "JSCJSValue.h"
29#include "PropertySlot.h"
30#include <wtf/Assertions.h>
31
32namespace JSC {
33
34class JSObject;
35class JSFunction;
36
37class PutPropertySlot {
38public:
39 enum Type : uint8_t { Uncachable, ExistingProperty, NewProperty, SetterProperty, CustomValue, CustomAccessor };
40 enum Context : uint8_t { UnknownContext, PutById, PutByIdEval, ReflectSet };
41 using PutValueFunc = JSC::PutValueFunc;
42 using PutValueFuncWithPtr = JSC::PutValueFuncWithPtr;
43
44 PutPropertySlot(JSValue thisValue, bool isStrictMode = false, Context context = UnknownContext, bool isInitialization = false)
45 : m_base(nullptr)
46 , m_thisValue(thisValue)
47 , m_offset(invalidOffset)
48 , m_isStrictMode(isStrictMode)
49 , m_isInitialization(isInitialization)
50 , m_isTaintedByOpaqueObject(false)
51 , m_type(Uncachable)
52 , m_context(context)
53 , m_cacheability(CachingAllowed)
54 {
55 }
56
57 void setExistingProperty(JSObject* base, PropertyOffset offset)
58 {
59 m_type = ExistingProperty;
60 m_base = base;
61 m_offset = offset;
62 }
63
64 void setNewProperty(JSObject* base, PropertyOffset offset)
65 {
66 m_type = NewProperty;
67 m_base = base;
68 m_offset = offset;
69 }
70
71 void setCustomValue(JSObject* base, PutValueFunc function)
72 {
73 m_type = CustomValue;
74 m_base = base;
75 m_putFunction = function;
76 }
77
78 void setCustomAccessor(JSObject* base, PutValueFunc function)
79 {
80 m_type = CustomAccessor;
81 m_base = base;
82 m_putFunction = function;
83 }
84
85 void setCacheableSetter(JSObject* base, PropertyOffset offset)
86 {
87 m_type = SetterProperty;
88 m_base = base;
89 m_offset = offset;
90 }
91
92 void setThisValue(JSValue thisValue)
93 {
94 m_thisValue = thisValue;
95 }
96
97 void setStrictMode(bool value)
98 {
99 m_isStrictMode = value;
100 }
101
102 FunctionPtr<CustomAccessorPtrTag> customSetter() const
103 {
104 ASSERT(isCacheableCustom());
105 return m_putFunction;
106 }
107
108 Type type() const { return m_type; }
109 Context context() const { return m_context; }
110 JSObject* base() const { return m_base; }
111 JSValue thisValue() const { return m_thisValue; }
112
113 bool isStrictMode() const { return m_isStrictMode; }
114 bool isCacheablePut() const { return isCacheable() && (m_type == NewProperty || m_type == ExistingProperty); }
115 bool isCacheableSetter() const { return isCacheable() && m_type == SetterProperty; }
116 bool isCacheableCustom() const { return isCacheable() && (m_type == CustomValue || m_type == CustomAccessor) && !!m_putFunction; }
117 bool isCustomAccessor() const { return isCacheable() && m_type == CustomAccessor; }
118 bool isInitialization() const { return m_isInitialization; }
119 bool isTaintedByOpaqueObject() const { return m_isTaintedByOpaqueObject; }
120 void setIsTaintedByOpaqueObject() { m_isTaintedByOpaqueObject = true; }
121
122 PropertyOffset cachedOffset() const
123 {
124 return m_offset;
125 }
126
127 void disableCaching()
128 {
129 m_cacheability = CachingDisallowed;
130 }
131
132private:
133 bool isCacheable() const { return m_cacheability == CachingAllowed; }
134
135 JSObject* m_base;
136 JSValue m_thisValue;
137 PropertyOffset m_offset;
138 bool m_isStrictMode : 1;
139 bool m_isInitialization : 1;
140 bool m_isTaintedByOpaqueObject : 1;
141 Type m_type;
142 Context m_context;
143 CacheabilityType m_cacheability;
144 FunctionPtr<CustomAccessorPtrTag> m_putFunction;
145};
146
147} // namespace JSC
