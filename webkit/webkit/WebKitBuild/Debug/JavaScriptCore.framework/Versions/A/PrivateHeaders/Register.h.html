35namespace JSC {
36
37 class CallFrame;
38 class CodeBlock;
39 class JSLexicalEnvironment;
40 class JSObject;
41 class JSScope;
42
43 class Register {
44 WTF_MAKE_FAST_ALLOCATED;
45 public:
46 Register();
47
48 Register(const JSValue&);
49 JSValue jsValue() const;
50 JSValue asanUnsafeJSValue() const;
51 EncodedJSValue encodedJSValue() const;
52
53 ALWAYS_INLINE Register& operator=(CallFrame*);
54 ALWAYS_INLINE Register& operator=(CodeBlock*);
55 ALWAYS_INLINE Register& operator=(JSScope*);
56 ALWAYS_INLINE Register& operator=(JSObject*);
57 ALWAYS_INLINE Register& operator=(EncodedJSValue);
58
59 int32_t i() const;
60 ALWAYS_INLINE CallFrame* callFrame() const;
61 ALWAYS_INLINE CodeBlock* codeBlock() const;
62 ALWAYS_INLINE CodeBlock* asanUnsafeCodeBlock() const;
63 ALWAYS_INLINE JSObject* object() const;
64 ALWAYS_INLINE JSScope* scope() const;
65 int32_t unboxedInt32() const;
66 uint32_t unboxedUInt32() const;
67 int32_t asanUnsafeUnboxedInt32() const;
68 int64_t unboxedInt52() const;
69 int64_t asanUnsafeUnboxedInt52() const;
70 int64_t unboxedStrictInt52() const;
71 int64_t asanUnsafeUnboxedStrictInt52() const;
72 int64_t unboxedInt64() const;
73 int64_t asanUnsafeUnboxedInt64() const;
74 bool unboxedBoolean() const;
75 double unboxedDouble() const;
76 double asanUnsafeUnboxedDouble() const;
77 JSCell* unboxedCell() const;
78 JSCell* asanUnsafeUnboxedCell() const;
79 int32_t payload() const;
80 int32_t tag() const;
81 int32_t unsafePayload() const;
82 int32_t unsafeTag() const;
83 int32_t& payload();
84 int32_t& tag();
85
86 void* pointer() const;
87 void* asanUnsafePointer() const;
88
89 static Register withInt(int32_t i)
90 {
91 Register r = jsNumber(i);
92 return r;
93 }
94
95 private:
96 union {
97 EncodedJSValue value;
98 CallFrame* callFrame;
99 CodeBlock* codeBlock;
100 EncodedValueDescriptor encodedValue;
101 double number;
102 int64_t integer;
103 } u;
104 };
105
106 ALWAYS_INLINE Register::Register()
107 {
108#ifndef NDEBUG
109 *this = JSValue();
110#endif
111 }
112
113 ALWAYS_INLINE Register::Register(const JSValue& v)
114 {
115 u.value = JSValue::encode(v);
116 }
117
118 // FIXME (rdar://problem/19379214): ASan only needs to be suppressed for Register::jsValue() when called from prepareOSREntry(), but there is currently no way to express this short of adding a separate copy of the function.
119 SUPPRESS_ASAN ALWAYS_INLINE JSValue Register::asanUnsafeJSValue() const
120 {
121 return JSValue::decode(u.value);
122 }
123
124 ALWAYS_INLINE JSValue Register::jsValue() const
125 {
126 return JSValue::decode(u.value);
127 }
128
129 ALWAYS_INLINE EncodedJSValue Register::encodedJSValue() const
130 {
131 return u.value;
132 }
133
134 // Interpreter functions
135
136 ALWAYS_INLINE int32_t Register::i() const
137 {
138 return jsValue().asInt32();
139 }
140
141 ALWAYS_INLINE int32_t Register::unboxedInt32() const
142 {
143 return payload();
144 }
145
146 ALWAYS_INLINE uint32_t Register::unboxedUInt32() const
147 {
148 return static_cast<uint32_t>(unboxedInt32());
149 }
150
151 SUPPRESS_ASAN ALWAYS_INLINE int32_t Register::asanUnsafeUnboxedInt32() const
152 {
153 return unsafePayload();
154 }
155
156 ALWAYS_INLINE int64_t Register::unboxedInt52() const
157 {
158 return u.integer >> JSValue::int52ShiftAmount;
159 }
160
161 SUPPRESS_ASAN ALWAYS_INLINE int64_t Register::asanUnsafeUnboxedInt52() const
162 {
163 return u.integer >> JSValue::int52ShiftAmount;
164 }
165
166 ALWAYS_INLINE int64_t Register::unboxedStrictInt52() const
167 {
168 return u.integer;
169 }
170
171 SUPPRESS_ASAN ALWAYS_INLINE int64_t Register::asanUnsafeUnboxedStrictInt52() const
172 {
173 return u.integer;
174 }
175
176 ALWAYS_INLINE int64_t Register::unboxedInt64() const
177 {
178 return u.integer;
179 }
180
181 SUPPRESS_ASAN ALWAYS_INLINE int64_t Register::asanUnsafeUnboxedInt64() const
182 {
183 return u.integer;
184 }
185
186 ALWAYS_INLINE bool Register::unboxedBoolean() const
187 {
188 return !!payload();
189 }
190
191 ALWAYS_INLINE double Register::unboxedDouble() const
192 {
193 return u.number;
194 }
195
196 SUPPRESS_ASAN ALWAYS_INLINE double Register::asanUnsafeUnboxedDouble() const
197 {
198 return u.number;
199 }
200
201 ALWAYS_INLINE JSCell* Register::unboxedCell() const
202 {
203#if USE(JSVALUE64)
204 return u.encodedValue.ptr;
205#else
206 return bitwise_cast<JSCell*>(payload());
207#endif
208 }
209
210 SUPPRESS_ASAN ALWAYS_INLINE JSCell* Register::asanUnsafeUnboxedCell() const
211 {
212#if USE(JSVALUE64)
213 return u.encodedValue.ptr;
214#else
215 return bitwise_cast<JSCell*>(payload());
216#endif
217 }
218
219 ALWAYS_INLINE void* Register::pointer() const
220 {
221#if USE(JSVALUE64)
222 return u.encodedValue.ptr;
223#else
224 return bitwise_cast<void*>(payload());
225#endif
226 }
227
228 SUPPRESS_ASAN ALWAYS_INLINE void* Register::asanUnsafePointer() const
229 {
230#if USE(JSVALUE64)
231 return u.encodedValue.ptr;
232#else
233 return bitwise_cast<void*>(unsafePayload());
234#endif
235 }
236
237 ALWAYS_INLINE int32_t Register::payload() const
238 {
239 return u.encodedValue.asBits.payload;
240 }
241
242 ALWAYS_INLINE int32_t Register::tag() const
243 {
244 return u.encodedValue.asBits.tag;
245 }
246
247 SUPPRESS_ASAN ALWAYS_INLINE int32_t Register::unsafePayload() const
248 {
249 return u.encodedValue.asBits.payload;
250 }
251
252 SUPPRESS_ASAN ALWAYS_INLINE int32_t Register::unsafeTag() const
253 {
254 return u.encodedValue.asBits.tag;
255 }
256
257 ALWAYS_INLINE int32_t& Register::payload()
258 {
259 return u.encodedValue.asBits.payload;
260 }
261
262 ALWAYS_INLINE int32_t& Register::tag()
263 {
264 return u.encodedValue.asBits.tag;
265 }
