28#include "CodeBlock.h"
29#include "JSScope.h"
30#include "Register.h"

32namespace JSC {
33
34ALWAYS_INLINE CallFrame* Register::callFrame() const
35{
36 return u.callFrame;
37}
38
39ALWAYS_INLINE CodeBlock* Register::codeBlock() const
40{
41 return u.codeBlock;
42}
43
44SUPPRESS_ASAN ALWAYS_INLINE CodeBlock* Register::asanUnsafeCodeBlock() const
45{
46 return u.codeBlock;
47}
48
49ALWAYS_INLINE JSObject* Register::object() const
50{
51 return asObject(jsValue());
52}
53
54ALWAYS_INLINE Register& Register::operator=(CallFrame* callFrame)
55{
56 u.callFrame = callFrame;
57 return *this;
58}
59
60ALWAYS_INLINE Register& Register::operator=(CodeBlock* codeBlock)
61{
62 u.codeBlock = codeBlock;
63 return *this;
64}
65
66ALWAYS_INLINE Register& Register::operator=(JSObject* object)
67{
68 u.value = JSValue::encode(JSValue(object));
69 return *this;
70}
71
72ALWAYS_INLINE Register& Register::operator=(JSScope* scope)
73{
74 *this = JSValue(scope);
75 return *this;
76}
77
78ALWAYS_INLINE Register& Register::operator=(EncodedJSValue encodedJSValue)
79{
80 u.value = encodedJSValue;
81 return *this;
82}
83
84ALWAYS_INLINE JSScope* Register::scope() const
85{
86 return jsCast<JSScope*>(unboxedCell());
87}
