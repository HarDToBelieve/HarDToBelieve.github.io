28#include "JSCJSValue.h"
29#include "JSCell.h"
30#include "Options.h"

32namespace JSC {
33
34inline bool scribbleFreeCells()
35{
36 return ASSERT_ENABLED || Options::scribbleFreeCells();
37}
38
39#define SCRIBBLE_WORD static_cast<intptr_t>(0xbadbeef0)
40
41inline bool isScribbledValue(JSValue value)
42{
43 return JSValue::encode(value) == JSValue::encode(bitwise_cast<JSCell*>(SCRIBBLE_WORD));
44}
45
46inline void scribble(void* base, size_t size)
47{
48 for (size_t i = size / sizeof(EncodedJSValue); i--;) {
49 // Use a 16-byte aligned value to ensure that it passes the cell check.
50 static_cast<EncodedJSValue*>(base)[i] = JSValue::encode(bitwise_cast<JSCell*>(SCRIBBLE_WORD));
51 }
52}
