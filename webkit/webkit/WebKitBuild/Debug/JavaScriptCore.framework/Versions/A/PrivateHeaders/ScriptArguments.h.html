39namespace JSC {
40class CallFrame;
41class JSGlobalObject;
42}
43
44namespace Inspector {
45
46class JS_EXPORT_PRIVATE ScriptArguments : public RefCounted<ScriptArguments> {
47public:
48 static Ref<ScriptArguments> create(JSC::JSGlobalObject*, Vector<JSC::Strong<JSC::Unknown>>&& arguments);
49 ~ScriptArguments();
50
51 JSC::JSValue argumentAt(size_t) const;
52 size_t argumentCount() const { return m_arguments.size(); }
53
54 JSC::JSGlobalObject* globalObject() const;
55
56 bool getFirstArgumentAsString(String& result) const;
57 bool isEqual(const ScriptArguments&) const;
58
59 static String truncateStringForConsoleMessage(const String& message)
60 {
61 constexpr size_t maxMessageLength = 10000;
62 if (message.length() <= maxMessageLength)
63 return message;
64 return makeString(message.substring(0, maxMessageLength), "..."_s);
65 }
66
67private:
68 ScriptArguments(JSC::JSGlobalObject*, Vector<JSC::Strong<JSC::Unknown>>&& arguments);
69
70 JSC::Strong<JSC::JSGlobalObject> m_globalObject;
71 Vector<JSC::Strong<JSC::Unknown>> m_arguments;
72};

74} // namespace Inspector
