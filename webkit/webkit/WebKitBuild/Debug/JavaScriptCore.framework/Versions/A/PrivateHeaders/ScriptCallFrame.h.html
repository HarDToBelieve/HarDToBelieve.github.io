41class JS_EXPORT_PRIVATE ScriptCallFrame {
42public:
43 ScriptCallFrame(const String& functionName, const String& scriptName, JSC::SourceID sourceID, unsigned lineNumber, unsigned column);
44 ~ScriptCallFrame();
45
46 const String& functionName() const { return m_functionName; }
47 const String& sourceURL() const { return m_scriptName; }
48 unsigned lineNumber() const { return m_lineNumber; }
49 unsigned columnNumber() const { return m_column; }
50 JSC::SourceID sourceID() const { return m_sourceID; }
51
52 bool isEqual(const ScriptCallFrame&) const;
53 bool isNative() const;
54
55 bool operator==(const ScriptCallFrame& other) const { return isEqual(other); }
56
57 Ref<Protocol::Console::CallFrame> buildInspectorObject() const;
58
59private:
60 String m_functionName;
61 String m_scriptName;
62 JSC::SourceID m_sourceID;
63 unsigned m_lineNumber;
64 unsigned m_column;
65};
