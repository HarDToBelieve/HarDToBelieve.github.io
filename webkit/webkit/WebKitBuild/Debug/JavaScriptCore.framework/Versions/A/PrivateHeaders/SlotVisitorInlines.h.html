35ALWAYS_INLINE void SlotVisitor::appendUnbarriered(JSValue* slot, size_t count)
36{
37 for (size_t i = count; i--;)
38 appendUnbarriered(slot[i]);
39}
40
41ALWAYS_INLINE void SlotVisitor::appendUnbarriered(JSCell* cell)
42{
43 // This needs to be written in a very specific way to ensure that it gets inlined
44 // properly. In particular, it appears that using templates here breaks ALWAYS_INLINE.
45
46 if (!cell)
47 return;
48
49 Dependency dependency;
50 if (UNLIKELY(cell->isPreciseAllocation())) {
51 if (LIKELY(cell->preciseAllocation().isMarked())) {
52 if (LIKELY(!m_heapAnalyzer))
53 return;
54 }
55 } else {
56 MarkedBlock& block = cell->markedBlock();
57 dependency = block.aboutToMark(m_markingVersion);
58 if (LIKELY(block.isMarked(cell, dependency))) {
59 if (LIKELY(!m_heapAnalyzer))
60 return;
61 }
62 }
63
64 appendSlow(cell, dependency);
65}
66
67ALWAYS_INLINE void SlotVisitor::appendUnbarriered(JSValue value)
68{
69 if (value.isCell())
70 appendUnbarriered(value.asCell());
71}
72
73ALWAYS_INLINE void SlotVisitor::appendHiddenUnbarriered(JSValue value)
74{
75 if (value.isCell())
76 appendHiddenUnbarriered(value.asCell());
77}
78
79ALWAYS_INLINE void SlotVisitor::appendHiddenUnbarriered(JSCell* cell)
80{
81 // This needs to be written in a very specific way to ensure that it gets inlined
82 // properly. In particular, it appears that using templates here breaks ALWAYS_INLINE.
83
84 if (!cell)
85 return;
86
87 Dependency dependency;
88 if (UNLIKELY(cell->isPreciseAllocation())) {
89 if (LIKELY(cell->preciseAllocation().isMarked()))
90 return;
91 } else {
92 MarkedBlock& block = cell->markedBlock();
93 dependency = block.aboutToMark(m_markingVersion);
94 if (LIKELY(block.isMarked(cell, dependency)))
95 return;
96 }
97
98 appendHiddenSlow(cell, dependency);
99}

101template<typename T>
102ALWAYS_INLINE void SlotVisitor::append(const Weak<T>& weak)
103{
104 appendUnbarriered(weak.get());
105}
106
107template<typename T, typename Traits>
108ALWAYS_INLINE void SlotVisitor::append(const WriteBarrierBase<T, Traits>& slot)
109{
110 appendUnbarriered(slot.get());
111}
112
113template<typename T, typename Traits>
114ALWAYS_INLINE void SlotVisitor::appendHidden(const WriteBarrierBase<T, Traits>& slot)
115{
116 appendHiddenUnbarriered(slot.get());
117}
118
119template<typename Iterator>
120ALWAYS_INLINE void SlotVisitor::append(Iterator begin, Iterator end)
121{
122 for (auto it = begin; it != end; ++it)
123 append(*it);
124}

126ALWAYS_INLINE void SlotVisitor::appendValues(const WriteBarrierBase<Unknown>* barriers, size_t count)
127{
128 for (size_t i = 0; i < count; ++i)
129 append(barriers[i]);
130}
131
132ALWAYS_INLINE void SlotVisitor::appendValuesHidden(const WriteBarrierBase<Unknown>* barriers, size_t count)
133{
134 for (size_t i = 0; i < count; ++i)
135 appendHidden(barriers[i]);
136}

153inline void SlotVisitor::reportExtraMemoryVisited(size_t size)
154{
155 if (m_isFirstVisit) {
156 m_nonCellVisitCount += size;
157 // FIXME: Change this to use SaturatedArithmetic when available.
158 // https://bugs.webkit.org/show_bug.cgi?id=170411
159 m_extraMemorySize += size;
160 }
161}

171template<typename Func>
172IterationStatus SlotVisitor::forEachMarkStack(const Func& func)
173{
174 if (func(m_collectorStack) == IterationStatus::Done)
175 return IterationStatus::Done;
176 if (func(m_mutatorStack) == IterationStatus::Done)
177 return IterationStatus::Done;
178 return IterationStatus::Continue;
179}
