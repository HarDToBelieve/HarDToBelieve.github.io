26#pragma once
27
28#include "CPU.h"
29#include <wtf/StdLibExtras.h>
30
31namespace JSC {
32
33#if USE(JSVALUE64)
34// According to C++ rules, a type used for the return signature of function with C linkage (i.e.
35// 'extern "C"') needs to be POD; hence putting any constructors into it could cause either compiler
36// warnings, or worse, a change in the ABI used to return these types.
37struct SlowPathReturnType {
38 CPURegister a;
39 CPURegister b;
40};
41static_assert(sizeof(SlowPathReturnType) >= sizeof(void*) * 2, "SlowPathReturnType should fit in two machine registers");
42
43inline SlowPathReturnType encodeResult(const void* a, const void* b)
44{
45 SlowPathReturnType result;
46 result.a = reinterpret_cast<CPURegister>(a);
47 result.b = reinterpret_cast<CPURegister>(b);
48 return result;
49}
50
51inline void decodeResult(SlowPathReturnType result, const void*& a, const void*& b)
52{
53 a = reinterpret_cast<void*>(result.a);
54 b = reinterpret_cast<void*>(result.b);
55}
56
57#else // USE(JSVALUE32_64)
58typedef int64_t SlowPathReturnType;
59
60typedef union {
61 struct {
62 const void* a;
63 const void* b;
64 } pair;
65 int64_t i;
66} SlowPathReturnTypeEncoding;
67
68inline SlowPathReturnType encodeResult(const void* a, const void* b)
69{
70 SlowPathReturnTypeEncoding u;
71 u.pair.a = a;
72 u.pair.b = b;
73 return u.i;
74}
75
76inline void decodeResult(SlowPathReturnType result, const void*& a, const void*& b)
77{
78 SlowPathReturnTypeEncoding u;
79 u.i = result;
80 a = u.pair.a;
81 b = u.pair.b;
82}
83#endif // USE(JSVALUE32_64)
84
85} // namespace JSC
