26#pragma once
27
28#include "ScriptFetcher.h"
29#include <wtf/URL.h>
30
31namespace JSC {
32
33class SourceOrigin {
34public:
35 explicit SourceOrigin(const URL& url)
36 : m_url(url)
37 {
38 }
39
40 explicit SourceOrigin(const URL& url, Ref<ScriptFetcher>&& fetcher)
41 : m_url(url)
42 , m_fetcher(WTFMove(fetcher))
43 {
44 }
45
46 SourceOrigin() = default;
47
48 const URL& url() const { return m_url; }
49 const String& string() const { return m_url.string(); }
50 bool isNull() const { return url().isNull(); }
51
52 ScriptFetcher* fetcher() const { return m_fetcher.get(); }
53
54private:
55 URL m_url;
56 RefPtr<ScriptFetcher> m_fetcher;
57};
58
59} // namespace JSC
