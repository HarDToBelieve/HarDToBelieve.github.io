395 {
396 if (offset == invalidOffset)
397 m_maxOffset = shortInvalidOffset;
398 else if (offset < useRareDataFlag && offset < shortInvalidOffset)

434 static unsigned outOfLineCapacity(PropertyOffset maxOffset)
435 {
436 unsigned outOfLineSize = Structure::outOfLineSize(maxOffset);
437
438 // This algorithm completely determines the out-of-line property storage growth algorithm.
439 // The JSObject code will only trigger a resize if the value returned by this algorithm
440 // changed between the new and old structure. So, it's important to keep this simple because
441 // it's on a fast path.
442
443 if (!outOfLineSize)
444 return 0;
445
446 if (outOfLineSize <= initialOutOfLineCapacity)
447 return initialOutOfLineCapacity;
448
449 ASSERT(outOfLineSize > initialOutOfLineCapacity);
450 COMPILE_ASSERT(outOfLineGrowthFactor == 2, outOfLineGrowthFactor_is_two);
451 return WTF::roundUpToPowerOfTwo(outOfLineSize);
452 }

619 bool dfgShouldWatchIfPossible() const
620 {
621 // FIXME: We would like to not watch things that are unprofitable to watch, like
622 // dictionaries. Unfortunately, we can't do such things: a dictionary could get flattened,
623 // in which case it will start to appear watchable and so the DFG will think that it is
624 // watching it. We should come up with a comprehensive story for not watching things that
625 // aren't profitable to watch.
626 // https://bugs.webkit.org/show_bug.cgi?id=133625
627
628 // - We don't watch Structures that either decided not to be watched, or whose predecessors
629 // decided not to be watched. This happens when a transition is fired while being watched.
630 if (transitionWatchpointIsLikelyToBeFired())
631 return false;
632
633 // - Don't watch Structures that had been dictionaries.
634 if (hasBeenDictionary())
635 return false;
636
637 return true;
638 }

706public:
707#define DEFINE_BITFIELD(type, lowerName, upperName, width, offset) \
708 static constexpr uint32_t s_##lowerName##Shift = offset;\
709 static constexpr uint32_t s_##lowerName##Mask = ((1 << (width - 1)) | ((1 << (width - 1)) - 1));\
710 static constexpr uint32_t s_bitWidthOf##upperName = width;\
711 type lowerName() const { return static_cast<type>((m_bitField >> offset) & s_##lowerName##Mask); }\
712 void set##upperName(type newValue) \
713 {\
714 m_bitField &= ~(s_##lowerName##Mask << offset);\
715 m_bitField |= (static_cast<uint32_t>(newValue) & s_##lowerName##Mask) << offset;\
716 }
717
718 DEFINE_BITFIELD(DictionaryKind, dictionaryKind, DictionaryKind, 2, 0);
