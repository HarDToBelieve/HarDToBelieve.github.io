36class Structure;

38#if USE(JSVALUE64)
39using StructureID = uint32_t;
40
41inline StructureID nukedStructureIDBit()
42{
43 return 0x80000000u;
44}
45
46inline StructureID nuke(StructureID id)
47{
48 return id | nukedStructureIDBit();
49}
50
51inline bool isNuked(StructureID id)
52{
53 return !!(id & nukedStructureIDBit());
54}
55
56inline StructureID decontaminate(StructureID id)
57{
58 return id & ~nukedStructureIDBit();
59}
60#else // not USE(JSVALUE64)
61using StructureID = Structure*;
62
63inline StructureID nukedStructureIDBit()
64{
65 return bitwise_cast<StructureID>(static_cast<uintptr_t>(1));
66}
67
68inline StructureID nuke(StructureID id)
69{
70 return bitwise_cast<StructureID>(bitwise_cast<uintptr_t>(id) | bitwise_cast<uintptr_t>(nukedStructureIDBit()));
71}
72
73inline bool isNuked(StructureID id)
74{
75 return !!(bitwise_cast<uintptr_t>(id) & bitwise_cast<uintptr_t>(nukedStructureIDBit()));
76}
77
78inline StructureID decontaminate(StructureID id)
79{
80 return bitwise_cast<StructureID>(bitwise_cast<uintptr_t>(id) & ~bitwise_cast<uintptr_t>(nukedStructureIDBit()));
81}
82#endif // not USE(JSVALUE64)

84#if USE(JSVALUE64)

86using EncodedStructureBits = uintptr_t;
87
88class StructureIDTable {
89 friend class LLIntOffsetsExtractor;
90public:
91 StructureIDTable();
92
93 void** base() { return reinterpret_cast<void**>(&m_table); }
94
95 ALWAYS_INLINE void validate(StructureID);
96
97 // FIXME: rdar://69036888: remove this when no longer needed.
98 // This is only used for a special case mitigation. It is not for general use.
99 Structure* tryGet(StructureID);
100
101 Structure* get(StructureID);
102 void deallocateID(Structure*, StructureID);
103 StructureID allocateID(Structure*);
104
105 void flushOldTables();
106
107 size_t size() const { return m_size; }
108
109private:
110 void resize(size_t newCapacity);
111 void makeFreeListFromRange(uint32_t first, uint32_t last);
112
113 union StructureOrOffset {
114 WTF_MAKE_FAST_ALLOCATED;
115 public:
116 EncodedStructureBits encodedStructureBits;
117 uintptr_t offset;
118 };
119
120 StructureOrOffset* table() const { return m_table.get(); }
121 static Structure* decode(EncodedStructureBits, StructureID);
122 static EncodedStructureBits encode(Structure*, StructureID);
123
124 static constexpr size_t s_initialSize = 512;
125
126 Vector<UniqueArray<StructureOrOffset>> m_oldTables;
127
128 uint32_t m_firstFreeOffset { 0 };
129 uint32_t m_lastFreeOffset { 0 };
130 UniqueArray<StructureOrOffset> m_table;
131
132 size_t m_size { 0 };
133 size_t m_capacity;
134
135 WeakRandom m_weakRandom;
136
137 static constexpr StructureID s_unusedID = 0;
138
139public:
140 // 1. StructureID is encoded as:
141 //
142 // ----------------------------------------------------------------
143 // | 1 Nuke Bit | 24 StructureIDTable index bits | 7 entropy bits |
144 // ----------------------------------------------------------------
145 //
146 // The entropy bits are chosen at random and assigned when a StructureID
147 // is allocated.
148 //
149 // 2. For each StructureID, the StructureIDTable stores encodedStructureBits
150 // which are encoded from the structure pointer as such:
151 //
152 // -----------------------------------------------------------------
153 // | 9 low index bits | 7 entropy bits | 48 structure pointer bits |
154 // -----------------------------------------------------------------
155 //
156 // The entropy bits here are the same 7 bits used in the encoding of the
157 // StructureID for this structure entry in the StructureIDTable.
158
159 static constexpr uint32_t s_numberOfNukeBits = 1;
160 static constexpr uint32_t s_numberOfEntropyBits = 7;
161 static constexpr uint32_t s_entropyBitsShiftForStructurePointer = (sizeof(EncodedStructureBits) * 8) - 16;
162
163 static constexpr uint32_t s_maximumNumberOfStructures = 1 << (32 - s_numberOfEntropyBits - s_numberOfNukeBits);
164};
165
166ALWAYS_INLINE Structure* StructureIDTable::decode(EncodedStructureBits bits, StructureID structureID)
167{
168 return bitwise_cast<Structure*>(bits ^ (static_cast<uintptr_t>(structureID) << s_entropyBitsShiftForStructurePointer));
169}
170
171ALWAYS_INLINE EncodedStructureBits StructureIDTable::encode(Structure* structure, StructureID structureID)
172{
173 return bitwise_cast<EncodedStructureBits>(structure) ^ (static_cast<EncodedStructureBits>(structureID) << s_entropyBitsShiftForStructurePointer);
174}
175
176inline Structure* StructureIDTable::get(StructureID structureID)
177{
178 ASSERT_WITH_SECURITY_IMPLICATION(structureID);
179 ASSERT_WITH_SECURITY_IMPLICATION(!isNuked(structureID));
180 uint32_t structureIndex = structureID >> s_numberOfEntropyBits;
181 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(structureIndex < m_capacity);
182 return decode(table()[structureIndex].encodedStructureBits, structureID);
183}
184
185// FIXME: rdar://69036888: remove this function when no longer needed.
186inline Structure* StructureIDTable::tryGet(StructureID structureID)
187{
188 uint32_t structureIndex = structureID >> s_numberOfEntropyBits;
189 if (structureIndex >= m_capacity)
190 return nullptr;
191 return decode(table()[structureIndex].encodedStructureBits, structureID);
192}
193
194ALWAYS_INLINE void StructureIDTable::validate(StructureID structureID)
195{
196 uint32_t structureIndex = structureID >> s_numberOfEntropyBits;
197 Structure* structure = decode(table()[structureIndex].encodedStructureBits, structureID);
198 RELEASE_ASSERT(structureIndex < m_capacity);
199 *bitwise_cast<volatile uint64_t*>(structure);
200}
