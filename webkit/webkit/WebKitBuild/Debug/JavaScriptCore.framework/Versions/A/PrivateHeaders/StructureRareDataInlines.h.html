60inline JSValue StructureRareData::cachedSpecialProperty(CachedSpecialPropertyKey key) const
61{
62 auto* cache = m_specialPropertyCache.get();
63 if (!cache)
64 return JSValue();
65 JSValue value = cache->m_cache[static_cast<unsigned>(key)].m_value.get();
66 if (value == JSCell::seenMultipleCalleeObjects())
67 return JSValue();
68#if ASSERT_ENABLED
69 if (value && value.isCell())
70 validateCell(value.asCell());
71#endif
72 return value;
73}

85inline JSImmutableButterfly* StructureRareData::cachedPropertyNames(CachedPropertyNamesKind kind) const
86{
87 ASSERT(!isCompilationThread());
88 auto* butterfly = m_cachedPropertyNames[static_cast<unsigned>(kind)].unvalidatedGet();
89 if (butterfly == cachedPropertyNamesSentinel())
90 return nullptr;
91 return butterfly;
92}
93
94inline JSImmutableButterfly* StructureRareData::cachedPropertyNamesIgnoringSentinel(CachedPropertyNamesKind kind) const
95{
96 ASSERT(!isCompilationThread());
97 return m_cachedPropertyNames[static_cast<unsigned>(kind)].unvalidatedGet();
98}

100inline JSImmutableButterfly* StructureRareData::cachedPropertyNamesConcurrently(CachedPropertyNamesKind kind) const
101{
102 auto* butterfly = m_cachedPropertyNames[static_cast<unsigned>(kind)].unvalidatedGet();
103 if (butterfly == cachedPropertyNamesSentinel())
104 return nullptr;
105 return butterfly;
106}
107
108inline void StructureRareData::setCachedPropertyNames(VM& vm, CachedPropertyNamesKind kind, JSImmutableButterfly* butterfly)
109{
110 if (butterfly == cachedPropertyNamesSentinel()) {
111 m_cachedPropertyNames[static_cast<unsigned>(kind)].setWithoutWriteBarrier(butterfly);
112 return;
113 }
114
115 WTF::storeStoreFence();
116 m_cachedPropertyNames[static_cast<unsigned>(kind)].set(vm, this, butterfly);
117}

119inline bool StructureRareData::canCacheSpecialProperty(CachedSpecialPropertyKey key)
120{
121 ASSERT(!isCompilationThread() && !Thread::mayBeGCThread());
122 auto* cache = m_specialPropertyCache.get();
123 if (!cache)
124 return true;
125 return cache->m_cache[static_cast<unsigned>(key)].m_value.get() != JSCell::seenMultipleCalleeObjects();
126}
127
128inline SpecialPropertyCache& StructureRareData::ensureSpecialPropertyCache()
129{
130 ASSERT(!isCompilationThread() && !Thread::mayBeGCThread());
131 if (auto* cache = m_specialPropertyCache.get())
132 return *cache;
133 return ensureSpecialPropertyCacheSlow();
134}
135
136inline void StructureRareData::cacheSpecialProperty(JSGlobalObject* globalObject, VM& vm, Structure* ownStructure, JSValue value, CachedSpecialPropertyKey key, const PropertySlot& slot)
137{
138 if (!canCacheSpecialProperty(key))
139 return;
140 return cacheSpecialPropertySlow(globalObject, vm, ownStructure, value, key, slot);
141}
