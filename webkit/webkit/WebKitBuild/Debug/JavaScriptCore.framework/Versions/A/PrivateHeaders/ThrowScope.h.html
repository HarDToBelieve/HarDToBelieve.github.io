30namespace JSC {

32class CallFrame;
33class JSObject;

35#if ENABLE(EXCEPTION_SCOPE_VERIFICATION)
36
37// If a function can throw a JS exception, it should declare a ThrowScope at the
38// top of the function (as early as possible) using the DECLARE_THROW_SCOPE macro.
39// Declaring a ThrowScope in a function means that the function may throw an
40// exception that its caller will have to handle.
41
42class ThrowScope : public ExceptionScope {
43public:
44 JS_EXPORT_PRIVATE ThrowScope(VM&, ExceptionEventLocation);
45 JS_EXPORT_PRIVATE ~ThrowScope();
46
47 ThrowScope(const ThrowScope&) = delete;
48 ThrowScope(ThrowScope&&) = default;
49
50 JS_EXPORT_PRIVATE Exception* throwException(JSGlobalObject*, Exception*);
51 JS_EXPORT_PRIVATE Exception* throwException(JSGlobalObject*, JSValue);
52
53 void release() { m_isReleased = true; }
54
55 void clearException() { m_vm.clearException(); }
56
57 JS_EXPORT_PRIVATE void printIfNeedCheck(const char* functionName, const char* file, unsigned line);
58
59private:
60 void simulateThrow();
61
62 bool m_isReleased { false };
63};
64
65#define DECLARE_THROW_SCOPE(vm__) \
66 JSC::ThrowScope((vm__), JSC::ExceptionEventLocation(EXCEPTION_SCOPE_POSITION_FOR_ASAN(vm__), __FUNCTION__, __FILE__, __LINE__))
67
68#define throwScopePrintIfNeedCheck(scope__) \
69 scope__.printIfNeedCheck(__FUNCTION__, __FILE__, __LINE__)
70
71#else // not ENABLE(EXCEPTION_SCOPE_VERIFICATION)
72
73class ThrowScope : public ExceptionScope {
74public:
75 ALWAYS_INLINE ThrowScope(VM& vm)
76 : ExceptionScope(vm)
77 { }
78 ThrowScope(const ThrowScope&) = delete;
79 ThrowScope(ThrowScope&&) = default;
80
81 ALWAYS_INLINE Exception* throwException(JSGlobalObject* globalObject, Exception* exception) { return m_vm.throwException(globalObject, exception); }
82 ALWAYS_INLINE Exception* throwException(JSGlobalObject* globalObject, JSValue value) { return m_vm.throwException(globalObject, value); }
83
84 ALWAYS_INLINE void release() { }
85
86 ALWAYS_INLINE void clearException() { m_vm.clearException(); }
87};
88
89#define DECLARE_THROW_SCOPE(vm__) \
90 JSC::ThrowScope((vm__))
91
92#endif // ENABLE(EXCEPTION_SCOPE_VERIFICATION)
93
94ALWAYS_INLINE Exception* throwException(JSGlobalObject* globalObject, ThrowScope& scope, Exception* exception)
95{
96 return scope.throwException(globalObject, exception);
97}
98
99ALWAYS_INLINE Exception* throwException(JSGlobalObject* globalObject, ThrowScope& scope, JSValue value)
100{
101 return scope.throwException(globalObject, value);
102}
103
104ALWAYS_INLINE EncodedJSValue throwVMException(JSGlobalObject* globalObject, ThrowScope& scope, Exception* exception)
105{
106 throwException(globalObject, scope, exception);
107 return encodedJSValue();
108}
109
110ALWAYS_INLINE EncodedJSValue throwVMException(JSGlobalObject* globalObject, ThrowScope& scope, JSValue value)
111{
112 throwException(globalObject, scope, value);
113 return encodedJSValue();
114}
