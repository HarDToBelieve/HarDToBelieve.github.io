36template<
37 typename TypeArg, typename ViewTypeArg, typename JSViewTypeArg,
38 TypedArrayType typeValueArg>
39struct IntegralTypedArrayAdaptor {
40 typedef TypeArg Type;
41 typedef ViewTypeArg ViewType;
42 typedef JSViewTypeArg JSViewType;
43 static constexpr TypedArrayType typeValue = typeValueArg;
44 static constexpr TypeArg minValue = std::numeric_limits<TypeArg>::lowest();
45 static constexpr TypeArg maxValue = std::numeric_limits<TypeArg>::max();
46 static constexpr bool canConvertToJSQuickly = true;
47 static constexpr TypedArrayContentType contentType = JSC::contentType(typeValue);
48
49 static JSValue toJSValue(JSGlobalObject*, Type value)
50 {
51 static_assert(!std::is_floating_point<Type>::value, "");
52 return jsNumber(value);
53 }
54
55 static Type toNativeFromInt32(int32_t value)
56 {
57 return static_cast<Type>(value);
58 }
59
60 static Type toNativeFromUint32(uint32_t value)
61 {
62 return static_cast<Type>(value);
63 }
64
65 static Type toNativeFromDouble(double value)
66 {
67 int32_t result = static_cast<int32_t>(value);
68 if (static_cast<double>(result) != value)
69 result = toInt32(value);
70 return static_cast<Type>(result);
71 }
72
73 template<typename OtherAdaptor>
74 static typename OtherAdaptor::Type convertTo(Type value)
75 {
76 if (typeValue == TypeUint32)
77 return OtherAdaptor::toNativeFromUint32(value);
78 return OtherAdaptor::toNativeFromInt32(value);
79 }
80
81 static std::optional<Type> toNativeFromInt32WithoutCoercion(int32_t value)
82 {
83 if ((value >= 0 && static_cast<uint32_t>(value) > static_cast<uint32_t>(maxValue)) || value < static_cast<int32_t>(minValue))
84 return std::nullopt;
85 return static_cast<Type>(value);
86 }
87
88 static std::optional<Type> toNativeFromUint32WithoutCoercion(uint32_t value)
89 {
90 if (value > static_cast<uint32_t>(maxValue))
91 return std::nullopt;
92
93 return static_cast<Type>(value);
94 }
95
96 static std::optional<Type> toNativeFromDoubleWithoutCoercion(double value)
97 {
98 Type integer = static_cast<Type>(value);
99 if (static_cast<double>(integer) != value)
100 return std::nullopt;
101
102 if (value < 0)
103 return toNativeFromInt32WithoutCoercion(static_cast<int32_t>(value));
104
105 return toNativeFromUint32WithoutCoercion(static_cast<uint32_t>(value));
106 }
107};

109template<
110 typename TypeArg, typename ViewTypeArg, typename JSViewTypeArg,
111 TypedArrayType typeValueArg>
112struct FloatTypedArrayAdaptor {
113 typedef TypeArg Type;
114 typedef ViewTypeArg ViewType;
115 typedef JSViewTypeArg JSViewType;
116 static constexpr TypedArrayType typeValue = typeValueArg;
117 static constexpr TypeArg minValue = std::numeric_limits<TypeArg>::lowest();
118 static constexpr TypeArg maxValue = std::numeric_limits<TypeArg>::max();
119 static constexpr bool canConvertToJSQuickly = true;
120 static constexpr TypedArrayContentType contentType = JSC::contentType(typeValue);
121
122 static JSValue toJSValue(JSGlobalObject*, Type value)
123 {
124 return jsDoubleNumber(purifyNaN(value));
125 }
126
127 static Type toNativeFromInt32(int32_t value)
128 {
129 return static_cast<Type>(value);
130 }
131
132 static Type toNativeFromUint32(uint32_t value)
133 {
134 return static_cast<Type>(value);
135 }
136
137 static Type toNativeFromDouble(double value)
138 {
139 return static_cast<Type>(value);
140 }
141
142 template<typename OtherAdaptor>
143 static typename OtherAdaptor::Type convertTo(Type value)
144 {
145 return OtherAdaptor::toNativeFromDouble(value);
146 }
147
148 static std::optional<Type> toNativeFromInt32WithoutCoercion(int32_t value)
149 {
150 return static_cast<Type>(value);
151 }
152
153 static std::optional<Type> toNativeFromDoubleWithoutCoercion(double value)
154 {
155 if (std::isnan(value) || std::isinf(value))
156 return static_cast<Type>(value);
157
158 Type valueResult = static_cast<Type>(value);
159
160 if (static_cast<double>(valueResult) != value)
161 return std::nullopt;
162
163 if (value < minValue || value > maxValue)
164 return std::nullopt;
165
166 return valueResult;
167 }
168};

172 TypedArrayType typeValueArg>

174 typedef TypeArg Type;
175 typedef ViewTypeArg ViewType;

182
183 static JSValue toJSValue(JSGlobalObject* globalObject, Type value)
184 {
185 ASSERT(globalObject);
186 return JSBigInt::makeHeapBigIntOrBigInt32(globalObject, value);
187 }
188
189 static Type toNativeFromInt32(int32_t value)
190 {
191 return static_cast<Type>(value);
192 }
193
194 static Type toNativeFromUint32(uint32_t value)
195 {
196 return static_cast<Type>(value);
197 }
198
199 static Type toNativeFromDouble(double value)
200 {
201 return static_cast<Type>(value);
202 }
203

207 return static_cast<typename OtherAdaptor::Type>(value);
208 }
209};

260struct Uint8ClampedAdaptor {
261 typedef uint8_t Type;
262 typedef Uint8ClampedArray ViewType;
263 typedef JSUint8ClampedArray JSViewType;
264 static constexpr TypedArrayType typeValue = TypeUint8Clamped;
265 static constexpr uint8_t minValue = std::numeric_limits<uint8_t>::lowest();
266 static constexpr uint8_t maxValue = std::numeric_limits<uint8_t>::max();
267 static constexpr bool canConvertToJSQuickly = true;
268 static constexpr TypedArrayContentType contentType = JSC::contentType(typeValue);
269
270 static JSValue toJSValue(JSGlobalObject*, uint8_t value)
271 {
272 return jsNumber(value);
273 }
274
275 static Type toNativeFromInt32(int32_t value)
276 {
277 return clamp(value);
278 }
279
280 static Type toNativeFromUint32(uint32_t value)
281 {
282 return std::min(static_cast<uint32_t>(255), value);
283 }
284
285 static Type toNativeFromDouble(double value)
286 {
287 if (std::isnan(value) || value < 0)
288 return 0;
289 if (value > 255)
290 return 255;
291 return static_cast<uint8_t>(lrint(value));
292 }
293
294 template<typename OtherAdaptor>
295 static typename OtherAdaptor::Type convertTo(uint8_t value)
296 {
297 return OtherAdaptor::toNativeFromInt32(value);
298 }
299
300 static std::optional<Type> toNativeFromInt32WithoutCoercion(int32_t value)
301 {
302 if (value > maxValue || value < minValue)
303 return std::nullopt;
304
305 return static_cast<Type>(value);
306 }
307
308 static std::optional<Type> toNativeFromDoubleWithoutCoercion(double value)
309 {
310 uint8_t integer = static_cast<uint8_t>(value);
311 if (static_cast<double>(integer) != value)
312 return std::nullopt;
313
314 return integer;
315 }
316
317private:
318 static uint8_t clamp(int32_t value)
319 {
320 if (value < 0)
321 return 0;
322 if (value > 255)
323 return 255;
324 return static_cast<uint8_t>(value);
325 }
326};
