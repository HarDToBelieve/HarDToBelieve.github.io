251 bool codeBlockEdgeMayBeWeak() const
252 {
253 // Currently, bytecode cache assumes that the tree of UnlinkedFunctionExecutable and UnlinkedCodeBlock will not be destroyed while the parent is live.
254 // Bytecode cache uses this asumption to avoid duplicate materialization by bookkeeping the heap cells in the offste-to-pointer map.
255 return VM::useUnlinkedCodeBlockJettisoning() && !m_isGeneratedFromCache;
256 }
