251struct ScratchBuffer {
252 ScratchBuffer()
253 {
254 u.m_activeLength = 0;
255 }
256
257 static ScratchBuffer* create(size_t size)
258 {
259 ScratchBuffer* result = new (VMMalloc::malloc(ScratchBuffer::allocationSize(size))) ScratchBuffer;
260 return result;
261 }
262
263 static ScratchBuffer* fromData(void* buffer)
264 {
265 return bitwise_cast<ScratchBuffer*>(static_cast<char*>(buffer) - OBJECT_OFFSETOF(ScratchBuffer, m_buffer));
266 }
267
268 static size_t allocationSize(Checked<size_t> bufferSize) { return sizeof(ScratchBuffer) + bufferSize; }
269 void setActiveLength(size_t activeLength) { u.m_activeLength = activeLength; }
270 size_t activeLength() const { return u.m_activeLength; };
271 size_t* addressOfActiveLength() { return &u.m_activeLength; };
272 void* dataBuffer() { return m_buffer; }
273
274 union {
275 size_t m_activeLength;
276 double pad; // Make sure m_buffer is double aligned.
277 } u;
278#if CPU(MIPS) && (defined WTF_MIPS_ARCH_REV && WTF_MIPS_ARCH_REV == 2)
279 alignas(8) void* m_buffer[0];
280#else
281 void* m_buffer[0];
282#endif
283};
284#if COMPILER(MSVC)
285#pragma warning(pop)
286#endif

463 // We make cross-cutting assumptions about typed arrays being in the primitive Gigacage and butterflies
464 // being in the JSValue gigacage. For some types, it's super obvious where they should go, and so we
465 // can hardcode that fact. But sometimes it's not clear, so we abstract it by having a Gigacage::Kind
466 // constant somewhere.
467 // FIXME: Maybe it would be better if everyone abstracted this?
468 // https://bugs.webkit.org/show_bug.cgi?id=175248
469 ALWAYS_INLINE CompleteSubspace& gigacageAuxiliarySpace(Gigacage::Kind kind)
470 {
471 switch (kind) {
472 case Gigacage::Primitive:
473 return primitiveGigacageAuxiliarySpace;
474 case Gigacage::JSValue:
475 return jsValueGigacageAuxiliarySpace;
476 case Gigacage::NumberOfKinds:
477 break;
478 }
479 RELEASE_ASSERT_NOT_REACHED();
480 return primitiveGigacageAuxiliarySpace;
481 }
