32ALWAYS_INLINE VM& VMTraps::vm() const
33{
34 return *bitwise_cast<VM*>(bitwise_cast<uintptr_t>(this) - OBJECT_OFFSETOF(VM, m_traps));
35}
36
37inline void VMTraps::deferTermination(DeferAction deferAction)
38{
39 auto originalCount = m_deferTerminationCount++;
40 ASSERT(m_deferTerminationCount < UINT_MAX);
41 if (UNLIKELY(originalCount == 0 && vm().exception()))
42 deferTerminationSlow(deferAction);
43}
44
45inline void VMTraps::undoDeferTermination(DeferAction deferAction)
46{
47 ASSERT(m_deferTerminationCount > 0);
48 ASSERT(!m_suspendedTerminationException || vm().terminationInProgress());
49 if (UNLIKELY(--m_deferTerminationCount == 0 && vm().terminationInProgress()))
50 undoDeferTerminationSlow(deferAction);
51}
