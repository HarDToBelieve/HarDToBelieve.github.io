35namespace JSC { namespace Wasm {
36
37class LLIntTierUpCounter : public BaselineExecutionCounter {
38 WTF_MAKE_NONCOPYABLE(LLIntTierUpCounter);
39
40public:
41 enum class CompilationStatus : uint8_t {
42 NotCompiled,
43 Compiling,
44 Compiled,
45 };
46
47 LLIntTierUpCounter()
48 {
49 optimizeAfterWarmUp();
50 }
51
52 void optimizeAfterWarmUp()
53 {
54 if (Options::wasmLLIntTiersUpToBBQ())
55 setNewThreshold(Options::thresholdForBBQOptimizeAfterWarmUp(), nullptr);
56 else
57 setNewThreshold(Options::thresholdForOMGOptimizeAfterWarmUp(), nullptr);
58 }
59
60 bool checkIfOptimizationThresholdReached()
61 {
62 return checkIfThresholdCrossedAndSet(nullptr);
63 }
64
65 void optimizeSoon()
66 {
67 if (Options::wasmLLIntTiersUpToBBQ())
68 setNewThreshold(Options::thresholdForBBQOptimizeSoon(), nullptr);
69 else
70 setNewThreshold(Options::thresholdForOMGOptimizeSoon(), nullptr);
71 }
72
73 struct OSREntryData {
74 uint32_t loopIndex;
75 Vector<VirtualRegister> values;
76 };
77
78 void addOSREntryDataForLoop(InstructionStream::Offset, OSREntryData&&);
79
80 const OSREntryData& osrEntryDataForLoop(InstructionStream::Offset) const;
81
82 Lock m_lock;
83 CompilationStatus m_compilationStatus { CompilationStatus::NotCompiled };
84 CompilationStatus m_loopCompilationStatus { CompilationStatus::NotCompiled };
85 HashMap<InstructionStream::Offset, OSREntryData> m_osrEntryData;
86};
87
88} } // namespace JSC::Wasm
