43class StreamingParserClient {

45 virtual ~StreamingParserClient() = default;

51class StreamingParser {
52 WTF_MAKE_FAST_ALLOCATED;
53public:
54 // The layout of the Wasm module is the following.
55 //
56 // Module: [ Header ][ Section ]*
57 // Section: [ ID ][ SizeOfPayload ][ Payload ]
58 // Code Section: [ ID ][ SizeOfPayload ][ Payload of Code Section ]
59 // [ NumberOfFunctions ]([ SizeOfFunction ][ PayloadOfFunction ])*
60 //
61 // Basically we can parse Wasm sections by repeatedly (1) reading the size of the payload, and (2) reading the payload based on the size read in (1).
62 // So this streaming parser handles Section as the unit for incremental parsing. The exception is the Code section. The Code section is large since it
63 // includes all the functions in the wasm module. Since we would like to compile each function and parse the Code section concurrently, the streaming
64 // parser specially handles the Code section. In the Code section, the streaming parser uses Function as the unit for incremental parsing.
65 enum class State : uint8_t {
66 ModuleHeader,
67 SectionID,
68 SectionSize,
69 SectionPayload,
70 CodeSectionSize,
71 FunctionSize,
72 FunctionPayload,
73 Finished,
74 FatalError,
75 };
76
77 enum class IsEndOfStream { Yes, No };
78
79 StreamingParser(ModuleInformation&, StreamingParserClient&);
80
81 State addBytes(const uint8_t* bytes, size_t length) { return addBytes(bytes, length, IsEndOfStream::No); }
82 State finalize();
83
84 String errorMessage() const { return crossThreadCopy(m_errorMessage); }
85
86 void reportError() { moveToStateIfNotFailed(failOnState(State::FatalError)); }
87
88private:
89 static constexpr unsigned moduleHeaderSize = 8;
90 static constexpr unsigned sectionIDSize = 1;
91
92 JS_EXPORT_PRIVATE State addBytes(const uint8_t* bytes, size_t length, IsEndOfStream);
93
94 State parseModuleHeader(Vector<uint8_t>&&);
95 State parseSectionID(Vector<uint8_t>&&);
96 State parseSectionSize(uint32_t);
97 State parseSectionPayload(Vector<uint8_t>&&);
98
99 State parseCodeSectionSize(uint32_t);
100 State parseFunctionSize(uint32_t);
101 State parseFunctionPayload(Vector<uint8_t>&&);
102
103 std::optional<Vector<uint8_t>> consume(const uint8_t* bytes, size_t, size_t&, size_t);
104 Expected<uint32_t, State> consumeVarUInt32(const uint8_t* bytes, size_t, size_t&, IsEndOfStream);
105
106 void moveToStateIfNotFailed(State);
107 template <typename ...Args> NEVER_INLINE State WARN_UNUSED_RETURN fail(Args...);
108
109 State failOnState(State);
110
111 Ref<ModuleInformation> m_info;
112 StreamingParserClient& m_client;
113 Vector<uint8_t> m_remaining;
114 String m_errorMessage;
115
116 CheckedSize m_totalSize { 0 };
117 size_t m_offset { 0 };
118 size_t m_nextOffset { 0 };
119 size_t m_codeOffset { 0 };
120
121 SHA1 m_hasher;
122
123 uint32_t m_sectionLength { 0 };
124
125 uint32_t m_functionCount { 0 };
126 uint32_t m_functionIndex { 0 };
127
128 uint32_t m_functionSize { 0 };
129
130 Lock m_stateLock;
131 State m_state { State::ModuleHeader };
132 Section m_section { Section::Begin };
133 Section m_previousKnownSection { Section::Begin };
134};
