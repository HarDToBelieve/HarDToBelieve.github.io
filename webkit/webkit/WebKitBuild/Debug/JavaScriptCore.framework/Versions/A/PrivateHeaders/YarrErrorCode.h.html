36enum class ErrorCode : uint8_t {
37 NoError = 0,
38 PatternTooLarge,
39 QuantifierOutOfOrder,
40 QuantifierWithoutAtom,
41 QuantifierTooLarge,
42 QuantifierIncomplete,
43 MissingParentheses,
44 BracketUnmatched,
45 ParenthesesUnmatched,
46 ParenthesesTypeInvalid,
47 InvalidGroupName,
48 DuplicateGroupName,
49 CharacterClassUnmatched,
50 CharacterClassRangeOutOfOrder,
51 CharacterClassRangeInvalid,
52 EscapeUnterminated,
53 InvalidUnicodeEscape,
54 InvalidUnicodeCodePointEscape,
55 InvalidBackreference,
56 InvalidNamedBackReference,
57 InvalidIdentityEscape,
58 InvalidOctalEscape,
59 InvalidControlLetterEscape,
60 InvalidUnicodePropertyExpression,
61 TooManyDisjunctions,
62 OffsetTooLarge,
63 InvalidRegularExpressionFlags,
64};
65
66JS_EXPORT_PRIVATE const char* errorMessage(ErrorCode);
67inline bool hasError(ErrorCode errorCode)
68{
69 return errorCode != ErrorCode::NoError;
70}
71
72inline bool hasHardError(ErrorCode errorCode)
73{
74 // TooManyDisjunctions means that we ran out stack compiling.
75 // All other errors are due to problems in the expression.
76 return hasError(errorCode) && errorCode != ErrorCode::TooManyDisjunctions;
77}
78JS_EXPORT_PRIVATE JSObject* errorToThrow(JSGlobalObject*, ErrorCode);
