48 /*
49 * CharacterClassParserDelegate:
50 *
51 * The class CharacterClassParserDelegate is used in the parsing of character
52 * classes. This class handles detection of character ranges. This class
53 * implements enough of the delegate interface such that it can be passed to
54 * parseEscape() as an EscapeDelegate. This allows parseEscape() to be reused
55 * to perform the parsing of escape characters in character sets.
56 */
57 class CharacterClassParserDelegate {
58 public:
59 CharacterClassParserDelegate(Delegate& delegate, ErrorCode& err, bool isUnicode)
60 : m_delegate(delegate)
61 , m_errorCode(err)
62 , m_isUnicode(isUnicode)
63 , m_state(Empty)
64 , m_character(0)
65 {
66 }
67
68 /*
69 * begin():
70 *
71 * Called at beginning of construction.
72 */
73 void begin(bool invert)
74 {
75 m_delegate.atomCharacterClassBegin(invert);
76 }

78 /*
79 * atomPatternCharacter():
80 *
81 * This method is called either from parseCharacterClass() (for an unescaped
82 * character in a character class), or from parseEscape(). In the former case
83 * the value true will be passed for the argument 'hyphenIsRange', and in this
84 * mode we will allow a hypen to be treated as indicating a range (i.e. /[a-z]/
85 * is different to /[a\-z]/).
86 */
87 void atomPatternCharacter(UChar32 ch, bool hyphenIsRange = false)
88 {
89 switch (m_state) {
90 case AfterCharacterClass:
91 // Following a built-in character class we need look out for a hyphen.
92 // We're looking for invalid ranges, such as /[\d-x]/ or /[\d-\d]/.
93 // If we see a hyphen following a character class then unlike usual
94 // we'll report it to the delegate immediately, and put ourself into
95 // a poisoned state. In a unicode pattern, any following calls to add
96 // another character or character class will result in syntax error.
97 // A hypen following a character class is itself valid, but only at
98 // the end of a regex.
99 if (hyphenIsRange && ch == '-') {
100 m_delegate.atomCharacterClassAtom('-');
101 m_state = AfterCharacterClassHyphen;
102 return;
103 }
104 // Otherwise just fall through - cached character so treat this as Empty.
105 FALLTHROUGH;
106
107 case Empty:
108 m_character = ch;
109 m_state = CachedCharacter;
110 return;
111
112 case CachedCharacter:
113 if (hyphenIsRange && ch == '-')
114 m_state = CachedCharacterHyphen;
115 else {
116 m_delegate.atomCharacterClassAtom(m_character);
117 m_character = ch;
118 }
119 return;
120
121 case CachedCharacterHyphen:
122 if (ch < m_character) {
123 m_errorCode = ErrorCode::CharacterClassRangeOutOfOrder;
124 return;
125 }
126 m_delegate.atomCharacterClassRange(m_character, ch);
127 m_state = Empty;
128 return;
129
130 // If we hit this case, we have an invalid range like /[\d-a]/.
131 // See coment in atomBuiltInCharacterClass() below.
132 case AfterCharacterClassHyphen:
133 if (m_isUnicode) {
134 m_errorCode = ErrorCode::CharacterClassRangeInvalid;
135 return;
136 }
137 m_delegate.atomCharacterClassAtom(ch);
138 m_state = Empty;
139 return;
140 }
141 }
142
143 /*
144 * atomBuiltInCharacterClass():
145 *
146 * Adds a built-in character class, called by parseEscape().
147 */
148 void atomBuiltInCharacterClass(BuiltInCharacterClassID classID, bool invert)
149 {
150 switch (m_state) {
151 case CachedCharacter:
152 // Flush the currently cached character, then fall through.
153 m_delegate.atomCharacterClassAtom(m_character);
154 FALLTHROUGH;
155 case Empty:
156 case AfterCharacterClass:
157 m_delegate.atomCharacterClassBuiltIn(classID, invert);
158 m_state = AfterCharacterClass;
159 return;
160
161 // If we hit either of these cases, we have an invalid range that
162 // looks something like /[a-\d]/ or /[\d-\d]/.
163 // Since ES2015, this should be syntax error in a unicode pattern,
164 // yet gracefully handled in a regular regex to avoid breaking the web.
165 // Effectively we handle the hyphen as if it was (implicitly) escaped,
166 // e.g. /[\d-a-z]/ is treated as /[\d\-a\-z]/.
167 // See usages of CharacterRangeOrUnion abstract op in
168 // https://tc39.es/ecma262/#sec-regular-expression-patterns-semantics
169 case CachedCharacterHyphen:
170 m_delegate.atomCharacterClassAtom(m_character);
171 m_delegate.atomCharacterClassAtom('-');
172 FALLTHROUGH;
173 case AfterCharacterClassHyphen:
174 if (m_isUnicode) {
175 m_errorCode = ErrorCode::CharacterClassRangeInvalid;
176 return;
177 }
178 m_delegate.atomCharacterClassBuiltIn(classID, invert);
179 m_state = Empty;
180 return;
181 }
182 }
183
184 /*
185 * end():
186 *
187 * Called at end of construction.
188 */
189 void end()
190 {
191 if (m_state == CachedCharacter)
192 m_delegate.atomCharacterClassAtom(m_character);
193 else if (m_state == CachedCharacterHyphen) {
194 m_delegate.atomCharacterClassAtom(m_character);
195 m_delegate.atomCharacterClassAtom('-');
196 }
197 m_delegate.atomCharacterClassEnd();
198 }

200 // parseEscape() should never call these delegate methods when
201 // invoked with inCharacterClass set.
202 NO_RETURN_DUE_TO_ASSERT void assertionWordBoundary(bool) { RELEASE_ASSERT_NOT_REACHED(); }
203 NO_RETURN_DUE_TO_ASSERT void atomBackReference(unsigned) { RELEASE_ASSERT_NOT_REACHED(); }
204 NO_RETURN_DUE_TO_ASSERT void atomNamedBackReference(const String&) { RELEASE_ASSERT_NOT_REACHED(); }
205 NO_RETURN_DUE_TO_ASSERT void atomNamedForwardReference(const String&) { RELEASE_ASSERT_NOT_REACHED(); }

214 CachedCharacterHyphen,

216 AfterCharacterClassHyphen,

231 // The handling of IdentityEscapes is different depending on the unicode flag.
232 // For Unicode patterns, IdentityEscapes only include SyntaxCharacters or '/'.
233 // For non-unicode patterns, most any character can be escaped.
234 bool isIdentityEscapeAnError(int ch)
235 {
236 if (m_isUnicode && (!strchr("^$\\.*+?()[]{}|/", ch) || !ch)) {
237 m_errorCode = ErrorCode::InvalidIdentityEscape;
238 return true;
239 }
240
241 return false;
242 }

244 /*
245 * parseEscape():
246 *
247 * Helper for parseTokens() AND parseCharacterClass().
248 * Unlike the other parser methods, this function does not report tokens
249 * directly to the member delegate (m_delegate), instead tokens are
250 * emitted to the delegate provided as an argument. In the case of atom
251 * escapes, parseTokens() will call parseEscape() passing m_delegate as
252 * an argument, and as such the escape will be reported to the delegate.
253 *
254 * However this method may also be used by parseCharacterClass(), in which
255 * case a CharacterClassParserDelegate will be passed as the delegate that
256 * tokens should be added to. A boolean flag is also provided to indicate
257 * whether that an escape in a CharacterClass is being parsed (some parsing
258 * rules change in this context).
259 *
260 * The boolean value returned by this method indicates whether the token
261 * parsed was an atom (outside of a characted class \b and \B will be
262 * interpreted as assertions).
263 */
264 template<bool inCharacterClass, class EscapeDelegate>
265 bool parseEscape(EscapeDelegate& delegate)
266 {
267 ASSERT(!hasError(m_errorCode));
268 ASSERT(peek() == '\\');
269 consume();
270
271 if (atEndOfPattern()) {
272 m_errorCode = ErrorCode::EscapeUnterminated;
273 return false;
274 }
275
276 switch (peek()) {
277 // Assertions
278 case 'b':
279 consume();
280 if (inCharacterClass)
281 delegate.atomPatternCharacter('\b');
282 else {
283 delegate.assertionWordBoundary(false);
284 return false;
285 }
286 break;
287 case 'B':
288 consume();
289 if (inCharacterClass) {
290 if (isIdentityEscapeAnError('B'))
291 break;
292
293 delegate.atomPatternCharacter('B');
294 } else {
295 delegate.assertionWordBoundary(true);
296 return false;
297 }
298 break;
299
300 // CharacterClassEscape
301 case 'd':
302 consume();
303 delegate.atomBuiltInCharacterClass(BuiltInCharacterClassID::DigitClassID, false);
304 break;
305 case 's':
306 consume();
307 delegate.atomBuiltInCharacterClass(BuiltInCharacterClassID::SpaceClassID, false);
308 break;
309 case 'w':
310 consume();
311 delegate.atomBuiltInCharacterClass(BuiltInCharacterClassID::WordClassID, false);
312 break;
313 case 'D':
314 consume();
315 delegate.atomBuiltInCharacterClass(BuiltInCharacterClassID::DigitClassID, true);
316 break;
317 case 'S':
318 consume();
319 delegate.atomBuiltInCharacterClass(BuiltInCharacterClassID::SpaceClassID, true);
320 break;
321 case 'W':
322 consume();
323 delegate.atomBuiltInCharacterClass(BuiltInCharacterClassID::WordClassID, true);
324 break;
325
326 case '0': {
327 consume();
328
329 if (!peekIsDigit()) {
330 delegate.atomPatternCharacter(0);
331 break;
332 }
333
334 if (m_isUnicode) {
335 m_errorCode = ErrorCode::InvalidOctalEscape;
336 break;
337 }
338
339 delegate.atomPatternCharacter(consumeOctal(2));
340 break;
341 }
342
343 // DecimalEscape
344 case '1':
345 case '2':
346 case '3':
347 case '4':
348 case '5':
349 case '6':
350 case '7':
351 case '8':
352 case '9': {
353 // For non-Unicode patterns, invalid backreferences are parsed as octal or decimal escapes.
354 // First, try to parse this as backreference.
355 if (!inCharacterClass) {
356 ParseState state = saveState();
357
358 unsigned backReference = consumeNumber();
359 if (backReference <= m_backReferenceLimit) {
360 m_maxSeenBackReference = std::max(m_maxSeenBackReference, backReference);
361 delegate.atomBackReference(backReference);
362 break;
363 }
364
365 restoreState(state);
366 if (m_isUnicode) {
367 m_errorCode = ErrorCode::InvalidBackreference;
368 break;
369 }
370 }
371
372 if (m_isUnicode) {
373 m_errorCode = ErrorCode::InvalidOctalEscape;
374 break;
375 }
376
377 delegate.atomPatternCharacter(peek() < '8' ? consumeOctal(3) : consume());
378 break;
379 }
380
381 // ControlEscape
382 case 'f':
383 consume();
384 delegate.atomPatternCharacter('\f');
385 break;
386 case 'n':
387 consume();
388 delegate.atomPatternCharacter('\n');
389 break;
390 case 'r':
391 consume();
392 delegate.atomPatternCharacter('\r');
393 break;
394 case 't':
395 consume();
396 delegate.atomPatternCharacter('\t');
397 break;
398 case 'v':
399 consume();
400 delegate.atomPatternCharacter('\v');
401 break;
402
403 // ControlLetter
404 case 'c': {
405 ParseState state = saveState();
406 consume();
407 if (!atEndOfPattern()) {
408 int control = consume();
409
410 if (WTF::isASCIIAlpha(control)) {
411 delegate.atomPatternCharacter(control & 0x1f);
412 break;
413 }
414
415 if (m_isUnicode) {
416 m_errorCode = ErrorCode::InvalidControlLetterEscape;
417 break;
418 }
419
420 // https://tc39.es/ecma262/#prod-annexB-ClassControlLetter
421 if (inCharacterClass && (WTF::isASCIIDigit(control) || control == '_')) {
422 delegate.atomPatternCharacter(control & 0x1f);
423 break;
424 }
425 }
426
427 if (m_isUnicode) {
428 m_errorCode = ErrorCode::InvalidIdentityEscape;
429 break;
430 }
431
432 restoreState(state);
433 delegate.atomPatternCharacter('\\');
434 break;
435 }
436
437 // HexEscape
438 case 'x': {
439 consume();
440 int x = tryConsumeHex(2);
441 if (x == -1) {
442 if (isIdentityEscapeAnError('x'))
443 break;
444
445 delegate.atomPatternCharacter('x');
446 } else
447 delegate.atomPatternCharacter(x);
448 break;
449 }
450
451 // Named backreference
452 case 'k': {
453 consume();
454 ParseState state = saveState();
455 if (!inCharacterClass && tryConsume('<')) {
456 auto groupName = tryConsumeGroupName();
457 if (hasError(m_errorCode))
458 break;
459
460 if (groupName) {
461 if (m_captureGroupNames.contains(groupName.value())) {
462 delegate.atomNamedBackReference(groupName.value());
463 break;
464 }
465
466 if (m_isNamedForwardReferenceAllowed) {
467 m_forwardReferenceNames.add(groupName.value());
468 delegate.atomNamedForwardReference(groupName.value());
469 break;
470 }
471 }
472 }
473
474 restoreState(state);
475 if (!isIdentityEscapeAnError('k')) {
476 delegate.atomPatternCharacter('k');
477 m_kIdentityEscapeSeen = true;
478 }
479 break;
480 }
481
482 // Unicode property escapes
483 case 'p':
484 case 'P': {
485 int escapeChar = consume();
486
487 if (!m_isUnicode) {
488 if (isIdentityEscapeAnError(escapeChar))
489 break;
490 delegate.atomPatternCharacter(escapeChar);
491 break;
492 }
493
494 if (!atEndOfPattern() && peek() == '{') {
495 consume();
496 auto optClassID = tryConsumeUnicodePropertyExpression();
497 if (!optClassID) {
498 // tryConsumeUnicodePropertyExpression() will set m_errorCode for a malformed property expression
499 break;
500 }
501 delegate.atomBuiltInCharacterClass(optClassID.value(), escapeChar == 'P');
502 } else
503 m_errorCode = ErrorCode::InvalidUnicodePropertyExpression;
504 break;
505 }
506
507 // UnicodeEscape
508 case 'u': {
509 int codePoint = tryConsumeUnicodeEscape<UnicodeParseContext::PatternCodePoint>();
510 if (hasError(m_errorCode))
511 break;
512
513 delegate.atomPatternCharacter(codePoint == -1 ? 'u' : codePoint);
514 break;
515 }
516
517 // IdentityEscape
518 default:
519 int ch = peek();
520
521 if (ch == '-' && m_isUnicode && inCharacterClass) {
522 // \- is allowed for ClassEscape with unicode flag.
523 delegate.atomPatternCharacter(consume());
524 break;
525 }
526
527 if (isIdentityEscapeAnError(ch))
528 break;
529
530 delegate.atomPatternCharacter(consume());
531 }
532
533 return true;
534 }
535
536 template<UnicodeParseContext context>
537 UChar32 consumePossibleSurrogatePair()
538 {
539 bool unicodePatternOrGroupName = m_isUnicode || context == UnicodeParseContext::GroupName;
540
541 UChar32 ch = consume();
542 if (U16_IS_LEAD(ch) && unicodePatternOrGroupName && !atEndOfPattern()) {
543 ParseState state = saveState();
544
545 UChar32 surrogate2 = consume();
546 if (U16_IS_TRAIL(surrogate2))
547 ch = U16_GET_SUPPLEMENTARY(ch, surrogate2);
548 else
549 restoreState(state);
550 }
551
552 return ch;
553 }

569 /*
570 * parseCharacterClass():
571 *
572 * Helper for parseTokens(); calls directly and indirectly (via parseCharacterClassEscape)
573 * to an instance of CharacterClassParserDelegate, to describe the character class to the
574 * delegate.
575 */
576 void parseCharacterClass()
577 {
578 ASSERT(!hasError(m_errorCode));
579 ASSERT(peek() == '[');
580 consume();
581
582 CharacterClassParserDelegate characterClassConstructor(m_delegate, m_errorCode, m_isUnicode);
583
584 characterClassConstructor.begin(tryConsume('^'));
585
586 while (!atEndOfPattern()) {
587 switch (peek()) {
588 case ']':
589 consume();
590 characterClassConstructor.end();
591 return;
592
593 case '\\':
594 parseCharacterClassEscape(characterClassConstructor);
595 break;
596
597 default:
598 characterClassConstructor.atomPatternCharacter(consumePossibleSurrogatePair<UnicodeParseContext::PatternCodePoint>(), true);
599 }
600
601 if (hasError(m_errorCode))
602 return;
603 }
604
605 m_errorCode = ErrorCode::CharacterClassUnmatched;
606 }

608 /*
609 * parseParenthesesBegin():
610 *
611 * Helper for parseTokens(); checks for parentheses types other than regular capturing subpatterns.
612 */
613 void parseParenthesesBegin()
614 {
615 ASSERT(!hasError(m_errorCode));
616 ASSERT(peek() == '(');
617 consume();
618
619 auto type = ParenthesesType::Subpattern;
620
621 if (tryConsume('?')) {
622 if (atEndOfPattern()) {
623 m_errorCode = ErrorCode::ParenthesesTypeInvalid;
624 return;
625 }
626
627 switch (consume()) {
628 case ':':
629 m_delegate.atomParenthesesSubpatternBegin(false);
630 break;
631
632 case '=':
633 m_delegate.atomParentheticalAssertionBegin();
634 type = ParenthesesType::Assertion;
635 break;
636
637 case '!':
638 m_delegate.atomParentheticalAssertionBegin(true);
639 type = ParenthesesType::Assertion;
640 break;
641
642 case '<': {
643 auto groupName = tryConsumeGroupName();
644 if (hasError(m_errorCode))
645 break;
646
647 if (groupName) {
648 if (m_kIdentityEscapeSeen) {
649 m_errorCode = ErrorCode::InvalidNamedBackReference;
650 break;
651 }
652
653 auto setAddResult = m_captureGroupNames.add(groupName.value());
654 if (setAddResult.isNewEntry)
655 m_delegate.atomParenthesesSubpatternBegin(true, groupName);
656 else
657 m_errorCode = ErrorCode::DuplicateGroupName;
658 } else
659 m_errorCode = ErrorCode::InvalidGroupName;
660
661 break;
662 }
663
664 default:
665 m_errorCode = ErrorCode::ParenthesesTypeInvalid;
666 }
667 } else
668 m_delegate.atomParenthesesSubpatternBegin();
669
670 if (type == ParenthesesType::Subpattern)
671 ++m_numSubpatterns;
672
673 m_parenthesesStack.append(type);
674 }
675
676 /*
677 * parseParenthesesEnd():
678 *
679 * Helper for parseTokens(); checks for parse errors (due to unmatched parentheses).
680 *
681 * The boolean value returned by this method indicates whether the token parsed
682 * was either an Atom or, for web compatibility reasons, QuantifiableAssertion
683 * in non-Unicode pattern.
684 */
685 bool parseParenthesesEnd()
686 {
687 ASSERT(!hasError(m_errorCode));
688 ASSERT(peek() == ')');
689 consume();
690
691 if (m_parenthesesStack.isEmpty()) {
692 m_errorCode = ErrorCode::ParenthesesUnmatched;
693 return false;
694 }
695
696 m_delegate.atomParenthesesEnd();
697 auto type = m_parenthesesStack.takeLast();
698 return type == ParenthesesType::Subpattern || !m_isUnicode;
699 }

701 /*
702 * parseQuantifier():
703 *
704 * Helper for parseTokens(); checks for parse errors and non-greedy quantifiers.
705 */
706 void parseQuantifier(bool lastTokenWasAnAtom, unsigned min, unsigned max)
707 {
708 ASSERT(!hasError(m_errorCode));
709 ASSERT(min <= max);
710
711 if (min == UINT_MAX) {
712 m_errorCode = ErrorCode::QuantifierTooLarge;
713 return;
714 }
715
716 if (lastTokenWasAnAtom)
717 m_delegate.quantifyAtom(min, max, !tryConsume('?'));
718 else
719 m_errorCode = ErrorCode::QuantifierWithoutAtom;
720 }
721
722 /*
723 * parseTokens():
724 *
725 * This method loops over the input pattern reporting tokens to the delegate.
726 * The method returns when a parse error is detected, or the end of the pattern
727 * is reached. One piece of state is tracked around the loop, which is whether
728 * the last token passed to the delegate was an atom (this is necessary to detect
729 * a parse error when a quantifier provided without an atom to quantify).
730 */
731 void parseTokens()
732 {
733 bool lastTokenWasAnAtom = false;
734
735 while (!atEndOfPattern()) {
736 switch (peek()) {
737 case '|':
738 consume();
739 m_delegate.disjunction();
740 lastTokenWasAnAtom = false;
741 break;
742
743 case '(':
744 parseParenthesesBegin();
745 lastTokenWasAnAtom = false;
746 break;
747
748 case ')':
749 lastTokenWasAnAtom = parseParenthesesEnd();
750 break;
751
752 case '^':
753 consume();
754 m_delegate.assertionBOL();
755 lastTokenWasAnAtom = false;
756 break;
757
758 case '$':
759 consume();
760 m_delegate.assertionEOL();
761 lastTokenWasAnAtom = false;
762 break;
763
764 case '.':
765 consume();
766 m_delegate.atomBuiltInCharacterClass(BuiltInCharacterClassID::DotClassID, false);
767 lastTokenWasAnAtom = true;
768 break;
769
770 case '[':
771 parseCharacterClass();
772 lastTokenWasAnAtom = true;
773 break;
774
775 case ']':
776 case '}':
777 if (m_isUnicode) {
778 m_errorCode = ErrorCode::BracketUnmatched;
779 break;
780 }
781
782 m_delegate.atomPatternCharacter(consume());
783 lastTokenWasAnAtom = true;
784 break;
785
786 case '\\':
787 lastTokenWasAnAtom = parseAtomEscape();
788 break;
789
790 case '*':
791 consume();
792 parseQuantifier(lastTokenWasAnAtom, 0, quantifyInfinite);
793 lastTokenWasAnAtom = false;
794 break;
795
796 case '+':
797 consume();
798 parseQuantifier(lastTokenWasAnAtom, 1, quantifyInfinite);
799 lastTokenWasAnAtom = false;
800 break;
801
802 case '?':
803 consume();
804 parseQuantifier(lastTokenWasAnAtom, 0, 1);
805 lastTokenWasAnAtom = false;
806 break;
807
808 case '{': {
809 ParseState state = saveState();
810
811 consume();
812 if (peekIsDigit()) {
813 unsigned min = consumeNumber();
814 unsigned max = min;
815
816 if (tryConsume(','))
817 max = peekIsDigit() ? consumeNumber() : quantifyInfinite;
818
819 if (tryConsume('}')) {
820 if (min <= max)
821 parseQuantifier(lastTokenWasAnAtom, min, max);
822 else
823 m_errorCode = ErrorCode::QuantifierOutOfOrder;
824 lastTokenWasAnAtom = false;
825 break;
826 }
827 }
828
829 if (m_isUnicode) {
830 m_errorCode = ErrorCode::QuantifierIncomplete;
831 break;
832 }
833
834 restoreState(state);
835 // if we did not find a complete quantifer, fall through to the default case.
836 FALLTHROUGH;
837 }
838
839 default:
840 m_delegate.atomPatternCharacter(consumePossibleSurrogatePair<UnicodeParseContext::PatternCodePoint>());
841 lastTokenWasAnAtom = true;
842 }
843
844 if (hasError(m_errorCode))
845 return;
846 }
847
848 if (!m_parenthesesStack.isEmpty())
849 m_errorCode = ErrorCode::MissingParentheses;
850 }

852 /*
853 * parse():
854 *
855 * This method calls parseTokens() to parse over the input and returns error code for a result.
856 */
857 ErrorCode parse()
858 {

860 return ErrorCode::PatternTooLarge;
861

863
864 if (!hasError(m_errorCode)) {
865 ASSERT(atEndOfPattern());

867 ASSERT(atEndOfPattern());
868 }
869
870 return m_errorCode;
871 }

873 void handleIllegalReferences()
874 {
875 bool shouldReparse = false;
876
877 if (m_maxSeenBackReference > m_numSubpatterns) {
878 // Contains illegal numeric backreference. See https://tc39.es/ecma262/#prod-annexB-AtomEscape
879 if (m_isUnicode) {
880 m_errorCode = ErrorCode::InvalidBackreference;
881 return;
882 }
883
884 m_backReferenceLimit = m_numSubpatterns;
885 shouldReparse = true;
886 }
887
888 if (m_kIdentityEscapeSeen && !m_captureGroupNames.isEmpty()) {
889 m_errorCode = ErrorCode::InvalidNamedBackReference;
890 return;
891 }
892
893 if (containsIllegalNamedForwardReference()) {
894 // \k<a> is parsed as named reference in Unicode patterns because of strict IdentityEscape grammar.
895 // See https://tc39.es/ecma262/#sec-patterns-static-semantics-early-errors
896 if (m_isUnicode || !m_captureGroupNames.isEmpty()) {
897 m_errorCode = ErrorCode::InvalidNamedBackReference;
898 return;
899 }
900
901 m_isNamedForwardReferenceAllowed = false;
902 shouldReparse = true;
903 }
904
905 if (shouldReparse) {
906 resetForReparsing();
907 parseTokens();
908 }
909 }
910
911 bool containsIllegalNamedForwardReference()
912 {
913 if (m_forwardReferenceNames.isEmpty())
914 return false;
915
916 if (m_captureGroupNames.isEmpty())
917 return true;
918
919 for (auto& entry : m_forwardReferenceNames) {
920 if (!m_captureGroupNames.contains(entry))
921 return true;
922 }
923
924 return false;
925 }

984 template<UnicodeParseContext context>
985 int tryConsumeUnicodeEscape()
986 {
987 ASSERT(!hasError(m_errorCode));
988
989 bool unicodePatternOrGroupName = m_isUnicode || context == UnicodeParseContext::GroupName;
990
991 if (!tryConsume('u') || atEndOfPattern()) {
992 if (unicodePatternOrGroupName)
993 m_errorCode = ErrorCode::InvalidUnicodeEscape;
994 return -1;
995 }
996
997 if (unicodePatternOrGroupName && tryConsume('{')) {
998 int codePoint = 0;
999 do {
1000 if (atEndOfPattern() || !isASCIIHexDigit(peek())) {
1001 m_errorCode = ErrorCode::InvalidUnicodeCodePointEscape;
1002 return -1;
1003 }
1004
1005 codePoint = (codePoint << 4) | toASCIIHexValue(consume());
1006
1007 if (codePoint > UCHAR_MAX_VALUE) {
1008 m_errorCode = ErrorCode::InvalidUnicodeCodePointEscape;
1009 return -1;
1010 }
1011 } while (!atEndOfPattern() && peek() != '}');
1012
1013 if (!tryConsume('}')) {
1014 m_errorCode = ErrorCode::InvalidUnicodeCodePointEscape;
1015 return -1;
1016 }
1017
1018 return codePoint;
1019 }
1020
1021 int codeUnit = tryConsumeHex(4);
1022 if (codeUnit == -1) {
1023 if (unicodePatternOrGroupName)
1024 m_errorCode = ErrorCode::InvalidUnicodeEscape;
1025 return -1;
1026 }
1027
1028 // If we have the first of a surrogate pair, look for the second.
1029 if (U16_IS_LEAD(codeUnit) && unicodePatternOrGroupName && patternRemaining() >= 6 && peek() == '\\') {
1030 ParseState state = saveState();
1031 consume();
1032
1033 if (tryConsume('u')) {
1034 int surrogate2 = tryConsumeHex(4);
1035 if (U16_IS_TRAIL(surrogate2))
1036 return U16_GET_SUPPLEMENTARY(codeUnit, surrogate2);
1037 }
1038
1039 restoreState(state);
1040 }
1041
1042 return codeUnit;
1043 }

1152 if (atEndOfPattern() || !isUnicodePropertyValueExpressionChar(peek())) {
1153 m_errorCode = ErrorCode::InvalidUnicodePropertyExpression;
1154 return std::nullopt;
1155 }
1156

1170 return std::nullopt;
1171 }

1174 auto result = unicodeMatchPropertyValue(unicodePropertyName, expressionBuilder.toString());
1175 if (!result)
1176 m_errorCode = ErrorCode::InvalidUnicodePropertyExpression;
1177 return result;
1178 }
1179
1180 auto result = unicodeMatchProperty(expressionBuilder.toString());
1181 if (!result)
1182 m_errorCode = ErrorCode::InvalidUnicodePropertyExpression;
1183 return result;
1184 }
1185
1186 consume();
1187 if (ch == '=') {
1188 if (!foundEquals) {
1189 foundEquals = true;
1190 unicodePropertyName = expressionBuilder.toString();
1191 expressionBuilder.clear();
1192 } else
1193 errors++;
1194 } else if (!isUnicodePropertyValueExpressionChar(ch))
1195 errors++;

1198 }

1225/*
1226 * Yarr::parse():
1227 *
1228 * The parse method is passed a pattern to be parsed and a delegate upon which
1229 * callbacks will be made to record the parsed tokens forming the regex.
1230 * Yarr::parse() returns null on success, or a const C string providing an error
1231 * message where a parse error occurs.
1232 *
1233 * The Delegate must implement the following interface:
1234 *
1235 * void assertionBOL();
1236 * void assertionEOL();
1237 * void assertionWordBoundary(bool invert);
1238 *
1239 * void atomPatternCharacter(UChar32 ch);
1240 * void atomBuiltInCharacterClass(BuiltInCharacterClassID classID, bool invert);
1241 * void atomCharacterClassBegin(bool invert)
1242 * void atomCharacterClassAtom(UChar32 ch)
1243 * void atomCharacterClassRange(UChar32 begin, UChar32 end)
1244 * void atomCharacterClassBuiltIn(BuiltInCharacterClassID classID, bool invert)
1245 * void atomCharacterClassEnd()
1246 * void atomParenthesesSubpatternBegin(bool capture = true, std::optional<String> groupName);
1247 * void atomParentheticalAssertionBegin(bool invert = false);
1248 * void atomParenthesesEnd();
1249 * void atomBackReference(unsigned subpatternId);
1250 * void atomNamedBackReference(const String& subpatternName);
1251 * void atomNamedForwardReference(const String& subpatternName);
1252 *
1253 * void quantifyAtom(unsigned min, unsigned max, bool greedy);
1254 *
1255 * void disjunction();
1256 *
1257 * void resetForReparsing();
1258 *
1259 * The regular expression is described by a sequence of assertion*() and atom*()
1260 * callbacks to the delegate, describing the terms in the regular expression.
1261 * Following an atom a quantifyAtom() call may occur to indicate that the previous
1262 * atom should be quantified. In the case of atoms described across multiple
1263 * calls (parentheses and character classes) the call to quantifyAtom() will come
1264 * after the call to the atom*End() method, never after atom*Begin().
1265 *
1266 * Character classes may either be described by a single call to
1267 * atomBuiltInCharacterClass(), or by a sequence of atomCharacterClass*() calls.
1268 * In the latter case, ...Begin() will be called, followed by a sequence of
1269 * calls to ...Atom(), ...Range(), and ...BuiltIn(), followed by a call to ...End().
1270 *
1271 * Sequences of atoms and assertions are broken into alternatives via calls to
1272 * disjunction(). Assertions, atoms, and disjunctions emitted between calls to
1273 * atomParenthesesBegin() and atomParenthesesEnd() form the body of a subpattern.
1274 * atomParenthesesBegin() is passed a subpatternId. In the case of a regular
1275 * capturing subpattern, this will be the subpatternId associated with these
1276 * parentheses, and will also by definition be the lowest subpatternId of these
1277 * parentheses and of any nested paretheses. The atomParenthesesEnd() method
1278 * is passed the subpatternId of the last capturing subexpression nested within
1279 * these paretheses. In the case of a capturing subpattern with no nested
1280 * capturing subpatterns, the same subpatternId will be passed to the begin and
1281 * end functions. In the case of non-capturing subpatterns the subpatternId
1282 * passed to the begin method is also the first possible subpatternId that might
1283 * be nested within these paretheses. If a set of non-capturing parentheses does
1284 * not contain any capturing subpatterns, then the subpatternId passed to begin
1285 * will be greater than the subpatternId passed to end.
1286 */
1287
1288template<class Delegate>
1289ErrorCode parse(Delegate& delegate, const String& pattern, bool isUnicode, unsigned backReferenceLimit = quantifyInfinite, bool isNamedForwardReferenceAllowed = true)
1290{
1291 if (pattern.is8Bit())
1292 return Parser<Delegate, LChar>(delegate, pattern, isUnicode, backReferenceLimit, isNamedForwardReferenceAllowed).parse();
1293 return Parser<Delegate, UChar>(delegate, pattern, isUnicode, backReferenceLimit, isNamedForwardReferenceAllowed).parse();
1294}
