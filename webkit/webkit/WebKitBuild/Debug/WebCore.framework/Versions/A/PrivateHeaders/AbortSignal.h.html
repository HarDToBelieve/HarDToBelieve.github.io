38class AbortAlgorithm;

41class AbortSignal final : public RefCounted<AbortSignal>, public EventTargetWithInlineData, private ContextDestructionObserver {
42 WTF_MAKE_ISO_ALLOCATED_EXPORT(AbortSignal, WEBCORE_EXPORT);
43public:
44 static Ref<AbortSignal> create(ScriptExecutionContext&);
45
46 static Ref<AbortSignal> abort(ScriptExecutionContext&);
47
48 static bool whenSignalAborted(AbortSignal&, Ref<AbortAlgorithm>&&);
49
50 void signalAbort();
51 void signalFollow(AbortSignal&);
52
53 bool aborted() const { return m_aborted; }
54
55 using RefCounted::ref;
56 using RefCounted::deref;
57
58 using Algorithm = Function<void()>;
59 void addAlgorithm(Algorithm&& algorithm) { m_algorithms.append(WTFMove(algorithm)); }
60
61 bool isFollowingSignal() const { return !!m_followingSignal; }
62
63private:
64 enum class Aborted : bool { No, Yes };
65 explicit AbortSignal(ScriptExecutionContext&, Aborted = Aborted::No);
66
67 // EventTarget.
68 EventTargetInterface eventTargetInterface() const final { return AbortSignalEventTargetInterfaceType; }
69 ScriptExecutionContext* scriptExecutionContext() const final { return ContextDestructionObserver::scriptExecutionContext(); }
70 void refEventTarget() final { ref(); }
71 void derefEventTarget() final { deref(); }
72
73 bool m_aborted { false };
74 Vector<Algorithm> m_algorithms;
75 WeakPtr<AbortSignal> m_followingSignal;
76};
