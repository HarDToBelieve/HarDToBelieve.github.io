26#pragma once
27
28#include "DictationContext.h"
29#include "FloatRect.h"
30#include <wtf/Vector.h>
31#include <wtf/text/WTFString.h>

33namespace WebCore {
34
35enum ReasonForDismissingAlternativeText {
36 ReasonForDismissingAlternativeTextCancelled = 0,
37 ReasonForDismissingAlternativeTextIgnored,
38 ReasonForDismissingAlternativeTextAccepted
39};
40
41enum AlternativeTextType {
42 AlternativeTextTypeCorrection = 0,
43 AlternativeTextTypeReversion,
44 AlternativeTextTypeSpellingSuggestions,
45 AlternativeTextTypeDictationAlternatives
46};
47
48enum class AutocorrectionResponse {
49 Edited,
50 Reverted,
51 Accepted
52};
53
54class AlternativeTextClient {
55public:
56 virtual ~AlternativeTextClient() = default;
57#if USE(AUTOCORRECTION_PANEL)
58 virtual void showCorrectionAlternative(AlternativeTextType, const FloatRect& boundingBoxOfReplacedString, const String& replacedString, const String& replacmentString, const Vector<String>& alternativeReplacementStrings) = 0;
59 virtual void dismissAlternative(ReasonForDismissingAlternativeText) = 0;
60 virtual String dismissAlternativeSoon(ReasonForDismissingAlternativeText) = 0;
61 virtual void recordAutocorrectionResponse(AutocorrectionResponse, const String& replacedString, const String& replacementString) = 0;
62#endif
63#if USE(DICTATION_ALTERNATIVES)
64 virtual void showDictationAlternativeUI(const FloatRect& boundingBoxOfDictatedText, DictationContext) = 0;
65 virtual void removeDictationAlternatives(DictationContext) = 0;
66 virtual Vector<String> dictationAlternatives(DictationContext) = 0;
67#endif
68};

70} // namespace WebCore
