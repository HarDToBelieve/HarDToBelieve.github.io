34namespace WebCore {
35
36struct ApplePayDetailsUpdateBase : public ApplePayDetailsUpdateData {
37 ApplePayLineItem newTotal;
38 Vector<ApplePayLineItem> newLineItems;
39
40 template<class Encoder> void encode(Encoder&) const;
41 template<class Decoder> static std::optional<ApplePayDetailsUpdateBase> decode(Decoder&);
42
43 template<class Decoder> WARN_UNUSED_RETURN bool decodeBase(Decoder&);
44};
45
46template<class Encoder>
47void ApplePayDetailsUpdateBase::encode(Encoder& encoder) const
48{
49 ApplePayDetailsUpdateData::encode(encoder);
50 encoder << newTotal;
51 encoder << newLineItems;
52}
53
54template<class Decoder>
55std::optional<ApplePayDetailsUpdateBase> ApplePayDetailsUpdateBase::decode(Decoder& decoder)
56{
57 ApplePayDetailsUpdateBase result;
58 if (!result.decodeBase(decoder))
59 return std::nullopt;
60 return result;
61}
62
63template<class Decoder>
64bool ApplePayDetailsUpdateBase::decodeBase(Decoder& decoder)
65{
66 if (!decodeData(decoder))
67 return false;
68
69#define DECODE(name, type) \
70 std::optional<type> name; \
71 decoder >> name; \
72 if (!name) \
73 return false; \
74 this->name = WTFMove(*name); \
75
76 DECODE(newTotal, ApplePayLineItem)
77 DECODE(newLineItems, Vector<ApplePayLineItem>)
78
79#undef DECODE
80
81 return true;
82}
