26#pragma once
27
28#if ENABLE(APPLE_PAY)
29
30#include "ApplePayPaymentTiming.h"
31#include "ApplePayRecurringPaymentDateUnit.h"
32#include <limits>
33#include <optional>
34#include <wtf/text/WTFString.h>

36namespace WebCore {
37
38struct ApplePayLineItem final {
39 enum class Type : bool {
40 Pending,
41 Final,
42 };
43
44 Type type { Type::Final };
45 String label;
46 String amount;
47
48#if ENABLE(APPLE_PAY_RECURRING_LINE_ITEM) || ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM)
49 ApplePayPaymentTiming paymentTiming { ApplePayPaymentTiming::Immediate };
50#endif
51
52#if ENABLE(APPLE_PAY_RECURRING_LINE_ITEM)
53 double recurringPaymentStartDate { std::numeric_limits<double>::quiet_NaN() };
54 ApplePayRecurringPaymentDateUnit recurringPaymentIntervalUnit { ApplePayRecurringPaymentDateUnit::Month };
55 unsigned recurringPaymentIntervalCount = 1;
56 double recurringPaymentEndDate { std::numeric_limits<double>::quiet_NaN() };
57#endif
58
59#if ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM)
60 double deferredPaymentDate { std::numeric_limits<double>::quiet_NaN() };
61#endif
62
63 template<class Encoder> void encode(Encoder&) const;
64 template<class Decoder> static std::optional<ApplePayLineItem> decode(Decoder&);
65};
66
67template<class Encoder>
68void ApplePayLineItem::encode(Encoder& encoder) const
69{
70 encoder << type;
71 encoder << label;
72 encoder << amount;
73#if ENABLE(APPLE_PAY_RECURRING_LINE_ITEM) || ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM)
74 encoder << paymentTiming;
75#endif
76#if ENABLE(APPLE_PAY_RECURRING_LINE_ITEM)
77 encoder << recurringPaymentStartDate;
78 encoder << recurringPaymentIntervalUnit;
79 encoder << recurringPaymentIntervalCount;
80 encoder << recurringPaymentEndDate;
81#endif
82#if ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM)
83 encoder << deferredPaymentDate;
84#endif
85}
86
87template<class Decoder>
88std::optional<ApplePayLineItem> ApplePayLineItem::decode(Decoder& decoder)
89{
90#define DECODE(name, type) \
91 std::optional<type> name; \
92 decoder >> name; \
93 if (!name) \
94 return std::nullopt; \
95
96 DECODE(type, Type)
97 DECODE(label, String)
98 DECODE(amount, String)
99#if ENABLE(APPLE_PAY_RECURRING_LINE_ITEM) || ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM)
100 DECODE(paymentTiming, ApplePayPaymentTiming)
101#endif
102#if ENABLE(APPLE_PAY_RECURRING_LINE_ITEM)
103 DECODE(recurringPaymentStartDate, double)
104 DECODE(recurringPaymentIntervalUnit, ApplePayRecurringPaymentDateUnit)
105 DECODE(recurringPaymentIntervalCount, unsigned)
106 DECODE(recurringPaymentEndDate, double)
107#endif
108#if ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM)
109 DECODE(deferredPaymentDate, double)
110#endif
111
112#undef DECODE
113
114 return { {
115 WTFMove(*type),
116 WTFMove(*label),
117 WTFMove(*amount),
118#if ENABLE(APPLE_PAY_RECURRING_LINE_ITEM) || ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM)
119 WTFMove(*paymentTiming),
120#endif
121#if ENABLE(APPLE_PAY_RECURRING_LINE_ITEM)
122 WTFMove(*recurringPaymentStartDate),
123 WTFMove(*recurringPaymentIntervalUnit),
124 WTFMove(*recurringPaymentIntervalCount),
125 WTFMove(*recurringPaymentEndDate),
126#endif
127#if ENABLE(APPLE_PAY_DEFERRED_LINE_ITEM)
128 WTFMove(*deferredPaymentDate),
129#endif
130 } };
131}
