166#if ENABLE(SCROLLING_THREAD)
167class LayerTreeHitTestLocker {
168public:
169 LayerTreeHitTestLocker(ScrollingCoordinator* scrollingCoordinator)
170 {
171 if (is<AsyncScrollingCoordinator>(scrollingCoordinator)) {
172 m_scrollingTree = downcast<AsyncScrollingCoordinator>(*scrollingCoordinator).scrollingTree();
173 if (m_scrollingTree)
174 m_scrollingTree->lockLayersForHitTesting();
175 }
176 }
177
178 ~LayerTreeHitTestLocker()
179 {
180 if (m_scrollingTree)
181 m_scrollingTree->unlockLayersForHitTesting();
182 }
183
184private:
185 RefPtr<ScrollingTree> m_scrollingTree;
186};
