29#ifndef AudioBus_h
30#define AudioBus_h
31
32#include "AudioChannel.h"
33#include <memory>
34#include <wtf/Noncopyable.h>
35#include <wtf/ThreadSafeRefCounted.h>
36#include <wtf/Vector.h>
37
38namespace WebCore {
39
40// An AudioBus represents a collection of one or more AudioChannels.
41// The data layout is "planar" as opposed to "interleaved".
42// An AudioBus with one channel is mono, an AudioBus with two channels is stereo, etc.
43
44enum class ChannelInterpretation {
45 Speakers,
46 Discrete,
47};
48
49class AudioBus final : public ThreadSafeRefCounted<AudioBus> {
50 WTF_MAKE_NONCOPYABLE(AudioBus);
51public:
52 enum {
53 ChannelLeft = 0,
54 ChannelRight = 1,
55 ChannelCenter = 2, // center and mono are the same
56 ChannelMono = 2,
57 ChannelLFE = 3,
58 ChannelSurroundLeft = 4,
59 ChannelSurroundRight = 5,
60 };
61
62 enum {
63 LayoutCanonical = 0
64 // Can define non-standard layouts here
65 };
66
67 // allocate indicates whether or not to initially have the AudioChannels created with managed storage.
68 // Normal usage is to pass true here, in which case the AudioChannels will memory-manage their own storage.
69 // If allocate is false then setChannelMemory() has to be called later on for each channel before the AudioBus is useable...
70 WEBCORE_EXPORT static RefPtr<AudioBus> create(unsigned numberOfChannels, size_t length, bool allocate = true);
71
72 // Tells the given channel to use an externally allocated buffer.
73 WEBCORE_EXPORT void setChannelMemory(unsigned channelIndex, float* storage, size_t length);
74
75 // Channels
76 unsigned numberOfChannels() const { return m_channels.size(); }
77
78 AudioChannel* channel(unsigned channel) { return m_channels[channel].get(); }
79 const AudioChannel* channel(unsigned channel) const { return m_channels[channel].get(); }
80 AudioChannel* channelByType(unsigned type);
81 const AudioChannel* channelByType(unsigned type) const;
82
83 // Number of sample-frames
84 size_t length() const { return m_length; }
85
86 // Sample-rate : 0.0 if unknown or "don't care"
87 float sampleRate() const { return m_sampleRate; }
88 void setSampleRate(float sampleRate) { m_sampleRate = sampleRate; }
89
90 // Zeroes all channels.
91 void zero();
92
93 // Clears the silent flag on all channels.
94 void clearSilentFlag();
95
96 // Returns true if the silent bit is set on all channels.
97 bool isSilent() const;
98
99 // Returns true if the channel count and frame-size match.
100 bool topologyMatches(const AudioBus &sourceBus) const;
101
102 // Creates a new buffer from a range in the source buffer.
103 // 0 may be returned if the range does not fit in the sourceBuffer
104 static RefPtr<AudioBus> createBufferFromRange(const AudioBus* sourceBuffer, unsigned startFrame, unsigned endFrame);
105
106
107 // Creates a new AudioBus by sample-rate converting sourceBus to the newSampleRate.
108 // setSampleRate() must have been previously called on sourceBus.
109 // Note: sample-rate conversion is already handled in the file-reading code for the mac port, so we don't need this.
110 static RefPtr<AudioBus> createBySampleRateConverting(const AudioBus* sourceBus, bool mixToMono, double newSampleRate);
111
112 // Creates a new AudioBus by mixing all the channels down to mono.
113 // If sourceBus is already mono, then the returned AudioBus will simply be a copy.
114 static RefPtr<AudioBus> createByMixingToMono(const AudioBus* sourceBus);
115
116 // Scales all samples by the same amount.
117 void scale(float scale);
118
119 void reset() { m_isFirstTime = true; } // for de-zippering
120
121 // Copies the samples from the source bus to this one.
122 void copyFromRange(const AudioBus& sourceBus, unsigned startFrame, unsigned endFrame);
123
124 // Copies the samples from the source bus to this one.
125 // This is just a simple per-channel copy if the number of channels match, otherwise an up-mix or down-mix is done.
126 WEBCORE_EXPORT void copyFrom(const AudioBus& sourceBus, ChannelInterpretation = ChannelInterpretation::Speakers);
127
128 // Sums the samples from the source bus to this one.
129 // This is just a simple per-channel summing if the number of channels match, otherwise an up-mix or down-mix is done.
130 void sumFrom(const AudioBus& sourceBus, ChannelInterpretation = ChannelInterpretation::Speakers);
131
132 // Copy each channel from sourceBus into our corresponding channel.
133 // We scale by targetGain (and our own internal gain m_busGain).
134 void copyWithGainFrom(const AudioBus& sourceBus, float targetGain);
135
136 // Copies the sourceBus by scaling with sample-accurate gain values.
137 void copyWithSampleAccurateGainValuesFrom(const AudioBus &sourceBus, float* gainValues, unsigned numberOfGainValues);
138
139 // Returns maximum absolute value across all channels (useful for normalization).
140 float maxAbsValue() const;
141
142 // Makes maximum absolute value == 1.0 (if possible).
143 void normalize();
144
145 static RefPtr<AudioBus> loadPlatformResource(const char* name, float sampleRate);
146
147private:
148 AudioBus() { }
149
150 AudioBus(unsigned numberOfChannels, size_t length, bool allocate);
151
152 void speakersSumFromByUpMixing(const AudioBus&);
153 void speakersSumFromByDownMixing(const AudioBus&);
154 void discreteSumFrom(const AudioBus&);
155
156 size_t m_length;
157 Vector<std::unique_ptr<AudioChannel>> m_channels;
158 int m_layout;
159 float m_busGain { 1 };
160 std::unique_ptr<AudioFloatArray> m_dezipperGainValues;
161 bool m_isFirstTime { 0 };
162 float m_sampleRate { 0 }; // 0.0 if unknown or N/A
163};
164
165} // WebCore
166
167#endif // AudioBus_h
