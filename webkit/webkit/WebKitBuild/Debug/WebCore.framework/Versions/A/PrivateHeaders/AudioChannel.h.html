29#ifndef AudioChannel_h
30#define AudioChannel_h
31
32#include "AudioArray.h"
33#include <memory>
34#include <wtf/Noncopyable.h>
35
36namespace WebCore {
37
38// An AudioChannel represents a buffer of non-interleaved floating-point audio samples.
39// The PCM samples are normally assumed to be in a nominal range -1.0 -> +1.0
40class AudioChannel final {
41 WTF_MAKE_FAST_ALLOCATED;
42 WTF_MAKE_NONCOPYABLE(AudioChannel);
43public:
44 // Memory can be externally referenced, or can be internally allocated with an AudioFloatArray.
45
46 // Reference an external buffer.
47 AudioChannel(float* storage, size_t length)
48 : m_rawPointer(storage)
49 , m_length(length)
50 , m_silent(false)
51 {
52 }
53
54 // Manage storage for us.
55 explicit AudioChannel(size_t length)
56 : m_memBuffer(makeUnique<AudioFloatArray>(length))
57 , m_length(length)
58 {
59 }
60
61 // A "blank" audio channel -- must call set() before it's useful...
62 AudioChannel() = default;
63
64 // Redefine the memory for this channel.
65 // storage represents external memory not managed by this object.
66 void set(float* storage, size_t length)
67 {
68 m_memBuffer = nullptr; // cleanup managed storage
69 m_rawPointer = storage;
70 m_length = length;
71 m_silent = false;
72 }
73
74 // How many sample-frames do we contain?
75 size_t length() const { return m_length; }
76
77 // Direct access to PCM sample data. Non-const accessor clears silent flag.
78 float* mutableData()
79 {
80 clearSilentFlag();
81 return m_rawPointer ? m_rawPointer : m_memBuffer->data();
82 }
83
84 const float* data() const { return m_rawPointer ? m_rawPointer : m_memBuffer->data(); }
85
86 // Zeroes out all sample values in buffer.
87 void zero()
88 {
89 if (m_silent)
90 return;
91
92 m_silent = true;
93
94 if (m_memBuffer.get())
95 m_memBuffer->zero();
96 else
97 memset(m_rawPointer, 0, sizeof(float) * m_length);
98 }
99
100 // Clears the silent flag.
101 void clearSilentFlag() { m_silent = false; }
102
103 bool isSilent() const { return m_silent; }
104
105 // Scales all samples by the same amount.
106 void scale(float scale);
107
108 // A simple memcpy() from the source channel
109 void copyFrom(const AudioChannel* sourceChannel);
110
111 // Copies the given range from the source channel.
112 void copyFromRange(const AudioChannel* sourceChannel, unsigned startFrame, unsigned endFrame);
113
114 // Sums (with unity gain) from the source channel.
115 void sumFrom(const AudioChannel* sourceChannel);
116
117 // Returns maximum absolute value (useful for normalization).
118 float maxAbsValue() const;
119
120private:
121 float* m_rawPointer { nullptr };
122 std::unique_ptr<AudioFloatArray> m_memBuffer;
123 size_t m_length { 0 };
124 bool m_silent { true };
125};
126
127} // WebCore
128
129#endif // AudioChannel_h
