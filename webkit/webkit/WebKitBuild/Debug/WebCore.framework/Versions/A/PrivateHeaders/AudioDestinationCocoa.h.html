27
28#if ENABLE(WEB_AUDIO)
29
30#include "AudioDestination.h"
31#include "AudioOutputUnitAdaptor.h"
32#include <AudioUnit/AudioUnit.h>
33#include <wtf/Lock.h>
34#include <wtf/RefPtr.h>
35#include <wtf/UniqueRef.h>
36
37namespace WebCore {
38
39class AudioBus;
40class MultiChannelResampler;
41class PushPullFIFO;
42
43using CreateAudioDestinationCocoaOverride = Ref<AudioDestination>(*)(AudioIOCallback&, float sampleRate);
44
45// An AudioDestination using CoreAudio's default output AudioUnit
46class AudioDestinationCocoa : public AudioDestination, public AudioUnitRenderer {
47public:
48 WEBCORE_EXPORT AudioDestinationCocoa(AudioIOCallback&, unsigned numberOfOutputChannels, float sampleRate, bool configureAudioOutputUnit = true);
49 WEBCORE_EXPORT virtual ~AudioDestinationCocoa();
50
51 WEBCORE_EXPORT static CreateAudioDestinationCocoaOverride createOverride;
52
53protected:
54 WEBCORE_EXPORT bool hasEnoughFrames(UInt32 numberOfFrames) const;
55 WEBCORE_EXPORT OSStatus render(double sampleTime, uint64_t hostTime, UInt32 numberOfFrames, AudioBufferList* ioData) final;
56
57 WEBCORE_EXPORT void setIsPlaying(bool);
58 bool isPlaying() final { return m_isPlaying; }
59 float sampleRate() const final { return m_contextSampleRate; }
60 WEBCORE_EXPORT unsigned framesPerBuffer() const final;
61
62 WEBCORE_EXPORT unsigned numberOfOutputChannels() const;
63
64 WEBCORE_EXPORT void getAudioStreamBasicDescription(AudioStreamBasicDescription&);
65
66private:
67 friend Ref<AudioDestination> AudioDestination::create(AudioIOCallback&, const String&, unsigned, unsigned, float);
68
69 WEBCORE_EXPORT void start(Function<void(Function<void()>&&)>&& dispatchToRenderThread, CompletionHandler<void(bool)>&&) final;
70 WEBCORE_EXPORT void stop(CompletionHandler<void(bool)>&&) final;
71
72 virtual void startRendering(CompletionHandler<void(bool)>&&);
73 virtual void stopRendering(CompletionHandler<void(bool)>&&);
74
75 void renderOnRenderingTheadIfPlaying(size_t framesToRender);
76 void renderOnRenderingThead(size_t framesToRender);
77
78 AudioOutputUnitAdaptor m_audioOutputUnitAdaptor;
79
80 // To pass the data from FIFO to the audio device callback.
81 Ref<AudioBus> m_outputBus;
82
83 // To push the rendered result from WebAudio graph into the FIFO.
84 Ref<AudioBus> m_renderBus;
85
86 // Resolves the buffer size mismatch between the WebAudio engine and
87 // the callback function from the actual audio device.
88 Lock m_fifoLock;
89 UniqueRef<PushPullFIFO> m_fifo WTF_GUARDED_BY_LOCK(m_fifoLock);
90
91 std::unique_ptr<MultiChannelResampler> m_resampler;
92 AudioIOPosition m_outputTimestamp;
93
94 Lock m_dispatchToRenderThreadLock;
95 Function<void(Function<void()>&&)> m_dispatchToRenderThread WTF_GUARDED_BY_LOCK(m_dispatchToRenderThreadLock);
96
97 float m_contextSampleRate;
98
99 std::atomic<bool> m_isPlaying;
100};
101
102} // namespace WebCore
103
104#endif // ENABLE(WEB_AUDIO)
