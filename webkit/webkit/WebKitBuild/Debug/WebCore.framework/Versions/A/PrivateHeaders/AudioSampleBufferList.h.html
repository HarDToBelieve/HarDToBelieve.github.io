35typedef struct AudioStreamBasicDescription AudioStreamBasicDescription;
36typedef struct OpaqueAudioConverter* AudioConverterRef;
37
38namespace WebCore {
39
40class AudioSampleBufferList : public RefCounted<AudioSampleBufferList> {
41public:
42 static Ref<AudioSampleBufferList> create(const CAAudioStreamDescription&, size_t);
43
44 ~AudioSampleBufferList();
45
46 static inline size_t audioBufferListSizeForStream(const CAAudioStreamDescription&);
47
48 static void applyGain(AudioBufferList&, float, AudioStreamDescription::PCMFormat);
49 void applyGain(float);
50
51 OSStatus copyFrom(const AudioSampleBufferList&, size_t count = SIZE_MAX);
52 OSStatus copyFrom(const AudioBufferList&, size_t frameCount, AudioConverterRef);
53 OSStatus copyFrom(AudioSampleBufferList&, size_t frameCount, AudioConverterRef);
54 OSStatus copyFrom(CARingBuffer&, size_t frameCount, uint64_t startFrame, CARingBuffer::FetchMode);
55
56 OSStatus mixFrom(const AudioSampleBufferList&, size_t count = SIZE_MAX);
57
58 OSStatus mixFrom(const AudioBufferList&, size_t count = SIZE_MAX);
59 OSStatus copyTo(AudioBufferList&, size_t count = SIZE_MAX);
60
61 const AudioStreamBasicDescription& streamDescription() const { return m_internalFormat->streamDescription(); }
62 const WebAudioBufferList& bufferList() const { return m_bufferList; }
63 WebAudioBufferList& bufferList() { return m_bufferList; }
64
65 uint32_t sampleCapacity() const { return m_sampleCapacity; }
66 uint32_t sampleCount() const { return m_sampleCount; }
67 void setSampleCount(size_t);
68
69 uint64_t timestamp() const { return m_timestamp; }
70 double hostTime() const { return m_hostTime; }
71 void setTimes(uint64_t time, double hostTime) { m_timestamp = time; m_hostTime = hostTime; }
72
73 void reset();
74
75 WEBCORE_EXPORT static void zeroABL(AudioBufferList&, size_t);
76 void zero();
77
78protected:
79 AudioSampleBufferList(const CAAudioStreamDescription&, size_t);
80
81 UniqueRef<CAAudioStreamDescription> m_internalFormat;
82
83 uint64_t m_timestamp { 0 };
84 double m_hostTime { -1 };
85 size_t m_sampleCount { 0 };
86 size_t m_sampleCapacity { 0 };
87 size_t m_maxBufferSizePerChannel { 0 };
88 size_t m_bufferListBaseSize { 0 };
89 UniqueRef<WebAudioBufferList> m_bufferList;
90};
91
92inline size_t AudioSampleBufferList::audioBufferListSizeForStream(const CAAudioStreamDescription& description)
93{
94 return offsetof(AudioBufferList, mBuffers) + (sizeof(AudioBuffer) * std::max<uint32_t>(1, description.numberOfChannelStreams()));
95}
