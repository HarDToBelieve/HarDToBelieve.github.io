26#pragma once
27
28#include "AudioSampleBufferList.h"
29#include <CoreAudio/CoreAudioTypes.h>
30#include <wtf/LoggerHelper.h>
31#include <wtf/MediaTime.h>
32#include <wtf/RefPtr.h>
33#include <wtf/ThreadSafeRefCounted.h>
34#include <wtf/text/WTFString.h>
35
36typedef const struct opaqueCMFormatDescription *CMFormatDescriptionRef;
37typedef struct opaqueCMSampleBuffer *CMSampleBufferRef;
38
39namespace WebCore {
40
41class CAAudioStreamDescription;
42class CARingBuffer;
43
44class AudioSampleDataSource : public ThreadSafeRefCounted<AudioSampleDataSource, WTF::DestructionThread::MainRunLoop>
45#if !RELEASE_LOG_DISABLED
46 , private LoggerHelper
47#endif
48 {
49public:
50 static Ref<AudioSampleDataSource> create(size_t, WTF::LoggerHelper&, size_t waitToStartForPushCount = 2);
51
52 ~AudioSampleDataSource();
53
54 OSStatus setInputFormat(const CAAudioStreamDescription&);
55 OSStatus setOutputFormat(const CAAudioStreamDescription&);
56
57 void pushSamples(const MediaTime&, const PlatformAudioData&, size_t);
58 void pushSamples(const AudioStreamBasicDescription&, CMSampleBufferRef);
59
60 enum PullMode { Copy, Mix };
61 bool pullSamples(AudioSampleBufferList&, size_t, uint64_t, double, PullMode);
62 bool pullSamples(AudioBufferList&, size_t, uint64_t, double, PullMode);
63
64 bool pullAvailableSamplesAsChunks(AudioBufferList&, size_t frameCount, uint64_t timeStamp, Function<void()>&&);
65
66 void setVolume(float volume) { m_volume = volume; }
67 float volume() const { return m_volume; }
68
69 void setMuted(bool muted) { m_muted = muted; }
70 bool muted() const { return m_muted; }
71
72 const CAAudioStreamDescription* inputDescription() const { return m_inputDescription ? &m_inputDescription.value() : nullptr; }
73
74#if !RELEASE_LOG_DISABLED
75 const Logger& logger() const final { return m_logger; }
76 const void* logIdentifier() const final { return m_logIdentifier; }
77 void setLogger(Ref<const Logger>&&, const void*);
78#endif
79
80 static constexpr float EquivalentToMaxVolume = 0.95;
81
82private:
83 AudioSampleDataSource(size_t, LoggerHelper&, size_t waitToStartForPushCount);
84
85 OSStatus setupConverter();
86 bool pullSamplesInternal(AudioBufferList&, size_t, uint64_t, double, PullMode);
87
88 void pushSamplesInternal(const AudioBufferList&, const MediaTime&, size_t frameCount);
89
90 std::optional<CAAudioStreamDescription> m_inputDescription;
91 std::optional<CAAudioStreamDescription> m_outputDescription;
92
93 MediaTime hostTime() const;
94
95#if !RELEASE_LOG_DISABLED
96 const char* logClassName() const final { return "AudioSampleDataSource"; }
97 WTFLogChannel& logChannel() const final;
98#endif
99
100 uint64_t m_lastPushedSampleCount { 0 };
101 size_t m_waitToStartForPushCount { 2 };
102 MediaTime m_expectedNextPushedSampleTime { MediaTime::invalidTime() };
103 bool m_isFirstPull { true };
104
105 MediaTime m_inputSampleOffset;
106 int64_t m_outputSampleOffset { 0 };
107
108 AudioConverterRef m_converter;
109 RefPtr<AudioSampleBufferList> m_scratchBuffer;
110
111 UniqueRef<CARingBuffer> m_ringBuffer;
112 size_t m_maximumSampleCount { 0 };
113
114 float m_volume { 1.0 };
115 bool m_muted { false };
116 bool m_shouldComputeOutputSampleOffset { true };
117 uint64_t m_endFrameWhenNotEnoughData { 0 };
118
119 bool m_isInNeedOfMoreData { false };
120
121#if !RELEASE_LOG_DISABLED
122 Ref<const Logger> m_logger;
123 const void* m_logIdentifier;
124#endif
125};
126
127} // namespace WebCore
