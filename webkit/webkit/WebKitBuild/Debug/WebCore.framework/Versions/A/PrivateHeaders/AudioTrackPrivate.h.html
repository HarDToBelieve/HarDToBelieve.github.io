33namespace WebCore {
34
35class AudioTrackPrivate;
36
37class AudioTrackPrivateClient : public TrackPrivateBaseClient {
38public:
39 virtual void enabledChanged(bool) = 0;
40};
41
42class AudioTrackPrivate : public TrackPrivateBase {
43public:
44 static Ref<AudioTrackPrivate> create()
45 {
46 return adoptRef(*new AudioTrackPrivate);
47 }
48
49 void setClient(AudioTrackPrivateClient* client) { m_client = client; }
50 AudioTrackPrivateClient* client() const override { return m_client; }
51
52 virtual void setEnabled(bool enabled)
53 {
54 if (m_enabled == enabled)
55 return;
56 m_enabled = enabled;
57 if (m_client)
58 m_client->enabledChanged(enabled);
59 if (m_enabledChangedCallback)
60 m_enabledChangedCallback(*this, m_enabled);
61 }
62
63 bool enabled() const { return m_enabled; }
64
65 enum Kind { Alternative, Description, Main, MainDesc, Translation, Commentary, None };
66 virtual Kind kind() const { return None; }
67
68 virtual bool isBackedByMediaStreamTrack() const { return false; }
69
70 using EnabledChangedCallback = Function<void(AudioTrackPrivate&, bool enabled)>;
71 void setEnabledChangedCallback(EnabledChangedCallback&& callback) { m_enabledChangedCallback = WTFMove(callback); }
72
73#if !RELEASE_LOG_DISABLED
74 const char* logClassName() const override { return "AudioTrackPrivate"; }
75#endif
76
77protected:
78 AudioTrackPrivate() = default;
79
80private:
81 AudioTrackPrivateClient* m_client { nullptr };
82 bool m_enabled { false };
83 EnabledChangedCallback m_enabledChangedCallback;
84};
85
86} // namespace WebCore
