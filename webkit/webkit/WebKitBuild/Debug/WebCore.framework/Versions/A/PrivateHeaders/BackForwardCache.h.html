26#pragma once
27
28#include "HistoryItem.h"
29#include <wtf/Forward.h>
30#include <wtf/ListHashSet.h>
31#include <wtf/Noncopyable.h>
32
33namespace WebCore {
34
35class CachedPage;
36class Frame;
37class Page;
38
39enum class PruningReason { None, ProcessSuspended, MemoryPressure, ReachedMaxSize };
40
41class BackForwardCache {
42 WTF_MAKE_NONCOPYABLE(BackForwardCache);
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 // Function to obtain the global back/forward cache.
46 WEBCORE_EXPORT static BackForwardCache& singleton();
47
48 bool canCache(Page&) const;
49
50 // Used when memory is low to prune some cached pages.
51 WEBCORE_EXPORT void pruneToSizeNow(unsigned maxSize, PruningReason);
52 WEBCORE_EXPORT void setMaxSize(unsigned); // number of pages to cache.
53 unsigned maxSize() const { return m_maxSize; }
54
55 WEBCORE_EXPORT std::unique_ptr<CachedPage> suspendPage(Page&);
56 WEBCORE_EXPORT bool addIfCacheable(HistoryItem&, Page*); // Prunes if maxSize() is exceeded.
57 WEBCORE_EXPORT void remove(HistoryItem&);
58 CachedPage* get(HistoryItem&, Page*);
59 std::unique_ptr<CachedPage> take(HistoryItem&, Page*);
60
61 void removeAllItemsForPage(Page&);
62
63 unsigned pageCount() const { return m_items.size(); }
64 WEBCORE_EXPORT unsigned frameCount() const;
65
66 void markPagesForDeviceOrPageScaleChanged(Page&);
67 void markPagesForContentsSizeChanged(Page&);
68#if ENABLE(VIDEO)
69 void markPagesForCaptionPreferencesChanged();
70#endif
71
72private:
73 BackForwardCache();
74 ~BackForwardCache() = delete; // Make sure nobody accidentally calls delete -- WebCore does not delete singletons.
75
76 static bool canCachePageContainingThisFrame(Frame&);
77
78 enum class ForceSuspension : bool { No, Yes };
79 std::unique_ptr<CachedPage> trySuspendPage(Page&, ForceSuspension);
80 void prune(PruningReason);
81 void dump() const;
82
83 ListHashSet<RefPtr<HistoryItem>> m_items;
84 unsigned m_maxSize {0};
85
86#if ASSERT_ENABLED
87 bool m_isInRemoveAllItemsForPage { false };
88#endif
89
90 friend class WTF::NeverDestroyed<BackForwardCache>;
91};
92
93} // namespace WebCore
