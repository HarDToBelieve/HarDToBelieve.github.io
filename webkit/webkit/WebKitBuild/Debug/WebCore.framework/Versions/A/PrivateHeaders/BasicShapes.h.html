77class BasicShapeCenterCoordinate {
78public:
79 enum Direction {
80 TopLeft,
81 BottomRight
82 };
83
84 BasicShapeCenterCoordinate()
85 {
86 updateComputedLength();
87 }
88
89 BasicShapeCenterCoordinate(Direction direction, Length length)
90 : m_direction(direction)
91 , m_length(length)
92 {
93 updateComputedLength();
94 }
95
96 Direction direction() const { return m_direction; }
97 const Length& length() const { return m_length; }
98 const Length& computedLength() const { return m_computedLength; }
99
100 BasicShapeCenterCoordinate blend(const BasicShapeCenterCoordinate& from, const BlendingContext& context) const
101 {
102 return BasicShapeCenterCoordinate(TopLeft, WebCore::blend(from.m_computedLength, m_computedLength, context));
103 }
104
105 bool operator==(const BasicShapeCenterCoordinate& other) const
106 {
107 return m_direction == other.m_direction
108 && m_length == other.m_length
109 && m_computedLength == other.m_computedLength;
110 }
111
112private:
113 void updateComputedLength();
114
115 Direction m_direction { TopLeft };
116 Length m_length { LengthType::Undefined };
117 Length m_computedLength;
118};

120class BasicShapeRadius {
121public:
122 enum Type {
123 Value,
124 ClosestSide,
125 FarthestSide
126 };
127
128 BasicShapeRadius() = default;
129
130 explicit BasicShapeRadius(Length v)
131 : m_value(v)
132 , m_type(Value)
133 { }
134 explicit BasicShapeRadius(Type t)
135 : m_value(LengthType::Undefined)
136 , m_type(t)
137 { }
138
139 const Length& value() const { return m_value; }
140 Type type() const { return m_type; }
141
142 bool canBlend(const BasicShapeRadius& other) const
143 {
144 // FIXME determine how to interpolate between keywords. See bug 125108.
145 return m_type == Value && other.type() == Value;
146 }
147
148 BasicShapeRadius blend(const BasicShapeRadius& from, const BlendingContext& context) const
149 {
150 if (m_type != Value || from.type() != Value)
151 return BasicShapeRadius(from);
152
153 return BasicShapeRadius(WebCore::blend(from.value(), value(), context));
154 }
155
156 bool operator==(const BasicShapeRadius& other) const
157 {
158 return m_value == other.m_value && m_type == other.m_type;
159 }
160
161private:
162 Length m_value { LengthType::Undefined };
163 Type m_type { ClosestSide };
164};
