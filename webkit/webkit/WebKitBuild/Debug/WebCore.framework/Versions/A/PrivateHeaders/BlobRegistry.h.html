32#pragma once
33
34#include <wtf/Forward.h>
35
36namespace WebCore {
37
38class BlobDataFileReference;
39class BlobPart;
40class BlobRegistry;
41class BlobRegistryImpl;
42
43WEBCORE_EXPORT BlobRegistry& blobRegistry();
44
45// BlobRegistry is not thread-safe. It should only be called from main thread.
46class WEBCORE_EXPORT BlobRegistry {
47public:
48
49 // Registers a blob URL referring to the specified file.
50 virtual void registerFileBlobURL(const URL&, Ref<BlobDataFileReference>&&, const String& path, const String& contentType) = 0;
51
52 // Registers a blob URL referring to the specified blob data.
53 virtual void registerBlobURL(const URL&, Vector<BlobPart>&&, const String& contentType) = 0;
54
55 // Registers a new blob URL referring to the blob data identified by the specified srcURL.
56 virtual void registerBlobURL(const URL&, const URL& srcURL) = 0;
57
58 // Registers a new blob URL referring to the blob data identified by the specified srcURL or, if none found, referring to the file found at the given path.
59 virtual void registerBlobURLOptionallyFileBacked(const URL&, const URL& srcURL, RefPtr<BlobDataFileReference>&&, const String& contentType) = 0;
60
61 // Negative start and end values select from the end.
62 virtual void registerBlobURLForSlice(const URL&, const URL& srcURL, long long start, long long end, const String& contentType) = 0;
63
64 virtual void unregisterBlobURL(const URL&) = 0;
65
66 virtual unsigned long long blobSize(const URL&) = 0;
67
68 virtual void writeBlobsToTemporaryFiles(const Vector<String>& blobURLs, CompletionHandler<void(Vector<String>&& filePaths)>&&) = 0;
69
70 virtual BlobRegistryImpl* blobRegistryImpl() { return nullptr; }
71
72protected:
73 virtual ~BlobRegistry();
74};
75
76} // namespace WebCore
