49class Field {
50 WTF_MAKE_FAST_ALLOCATED;
51public:
52 virtual JSValue valueFromInstance(JSGlobalObject*, const Instance*) const = 0;
53 virtual bool setValueToInstance(JSGlobalObject*, const Instance*, JSValue) const = 0;
54
55 virtual ~Field() = default;
56};
57
58class Class {
59 WTF_MAKE_NONCOPYABLE(Class); WTF_MAKE_FAST_ALLOCATED;
60public:
61 Class() = default;
62 virtual Method* methodNamed(PropertyName, Instance*) const = 0;
63 virtual Field* fieldNamed(PropertyName, Instance*) const = 0;
64 virtual JSValue fallbackObject(JSGlobalObject*, Instance*, PropertyName) { return jsUndefined(); }
65
66 virtual ~Class() = default;
67};
68
69class Instance : public RefCounted<Instance> {
70public:
71 WEBCORE_EXPORT Instance(RefPtr<RootObject>&&);
72
73 // These functions are called before and after the main entry points into
74 // the native implementations. They can be used to establish and cleanup
75 // any needed state.
76 void begin();
77 void end();
78
79 virtual Class* getClass() const = 0;
80 WEBCORE_EXPORT JSObject* createRuntimeObject(JSGlobalObject*);
81 void willInvalidateRuntimeObject();
82
83 // Returns false if the value was not set successfully.
84 virtual bool setValueOfUndefinedField(JSGlobalObject*, PropertyName, JSValue) { return false; }
85
86 virtual JSValue getMethod(JSGlobalObject*, PropertyName) = 0;
87 virtual JSValue invokeMethod(JSGlobalObject*, CallFrame*, RuntimeMethod* method) = 0;
88
89 virtual bool supportsInvokeDefaultMethod() const { return false; }
90 virtual JSValue invokeDefaultMethod(JSGlobalObject*, CallFrame*) { return jsUndefined(); }
91
92 virtual bool supportsConstruct() const { return false; }
93 virtual JSValue invokeConstruct(JSGlobalObject*, CallFrame*, const ArgList&) { return JSValue(); }
94
95 virtual void getPropertyNames(JSGlobalObject*, PropertyNameArray&) { }
96
97 virtual JSValue defaultValue(JSGlobalObject*, PreferredPrimitiveType) const = 0;
98
99 virtual JSValue valueOf(JSGlobalObject* exec) const = 0;
100
101 RootObject* rootObject() const;
102
103 WEBCORE_EXPORT virtual ~Instance();
104
105 virtual bool getOwnPropertySlot(JSObject*, JSGlobalObject*, PropertyName, PropertySlot&) { return false; }
106 virtual bool put(JSObject*, JSGlobalObject*, PropertyName, JSValue, PutPropertySlot&) { return false; }
107
108protected:
109 virtual void virtualBegin() { }
110 virtual void virtualEnd() { }
111 WEBCORE_EXPORT virtual RuntimeObject* newRuntimeObject(JSGlobalObject*);
112
113 RefPtr<RootObject> m_rootObject;
114
115private:
116 Weak<RuntimeObject> m_runtimeObject;
117};

126 virtual JSValue valueAt(JSGlobalObject*, unsigned index) const = 0;
127 virtual unsigned int getLength() const = 0;
128
129protected:
130 RefPtr<RootObject> m_rootObject;
131};
