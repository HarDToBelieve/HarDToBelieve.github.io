60// JavaScript's handle to a CDMInstance, must be used from the
61// main-thread only!
62class CDMInstance : public RefCounted<CDMInstance> {
63public:
64 virtual ~CDMInstance() = default;
65
66 virtual void setClient(WeakPtr<CDMInstanceClient>&&) { }

74 Mock,
75 ClearKey,

84 enum SuccessValue : bool {
85 Failed,
86 Succeeded,
87 };

90 enum class AllowDistinctiveIdentifiers : bool {
91 No,
92 Yes,
93 };
94
95 enum class AllowPersistentState : bool {
96 No,
97 Yes,
98 };
99
100 virtual void initializeWithConfiguration(const CDMKeySystemConfiguration&, AllowDistinctiveIdentifiers, AllowPersistentState, SuccessCallback&&) = 0;
101 virtual void setServerCertificate(Ref<SharedBuffer>&&, SuccessCallback&&) = 0;
102 virtual void setStorageDirectory(const String&) = 0;
103 virtual const String& keySystem() const = 0;
104 virtual RefPtr<CDMInstanceSession> createSession() = 0;

106 enum class HDCPStatus : uint8_t {
107 Unknown,
108 Valid,
109 OutputRestricted,
110 OutputDownscaled,
111 };

113};
