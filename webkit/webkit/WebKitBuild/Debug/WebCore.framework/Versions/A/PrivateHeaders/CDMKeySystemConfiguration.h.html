29#pragma once
30
31#if ENABLE(ENCRYPTED_MEDIA)
32
33#include "CDMMediaCapability.h"
34#include "CDMRequirement.h"
35#include "CDMSessionType.h"
36#include <wtf/Vector.h>
37#include <wtf/text/WTFString.h>
38
39namespace WebCore {
40
41struct CDMKeySystemConfiguration {
42 using KeysRequirement = CDMRequirement;
43
44 String label;
45 Vector<String> initDataTypes;
46 Vector<CDMMediaCapability> audioCapabilities;
47 Vector<CDMMediaCapability> videoCapabilities;
48 CDMRequirement distinctiveIdentifier { CDMRequirement::Optional };
49 CDMRequirement persistentState { CDMRequirement::Optional };
50 Vector<CDMSessionType> sessionTypes;
51
52 template<class Encoder>
53 void encode(Encoder& encoder) const
54 {
55 encoder << label;
56 encoder << initDataTypes;
57 encoder << audioCapabilities;
58 encoder << videoCapabilities;
59 encoder << distinctiveIdentifier;
60 encoder << persistentState;
61 encoder << sessionTypes;
62 }
63
64 template <class Decoder>
65 static std::optional<CDMKeySystemConfiguration> decode(Decoder& decoder)
66 {
67 std::optional<String> label;
68 decoder >> label;
69 if (!label)
70 return std::nullopt;
71
72 std::optional<Vector<String>> initDataTypes;
73 decoder >> initDataTypes;
74 if (!initDataTypes)
75 return std::nullopt;
76
77 std::optional<Vector<CDMMediaCapability>> audioCapabilities;
78 decoder >> audioCapabilities;
79 if (!audioCapabilities)
80 return std::nullopt;
81
82 std::optional<Vector<CDMMediaCapability>> videoCapabilities;
83 decoder >> videoCapabilities;
84 if (!videoCapabilities)
85 return std::nullopt;
86
87 std::optional<CDMRequirement> distinctiveIdentifier;
88 decoder >> distinctiveIdentifier;
89 if (!distinctiveIdentifier)
90 return std::nullopt;
91
92 std::optional<CDMRequirement> persistentState;
93 decoder >> persistentState;
94 if (!persistentState)
95 return std::nullopt;
96
97 std::optional<Vector<CDMSessionType>> sessionTypes;
98 decoder >> sessionTypes;
99 if (!sessionTypes)
100 return std::nullopt;
101
102 return {{
103 WTFMove(*label),
104 WTFMove(*initDataTypes),
105 WTFMove(*audioCapabilities),
106 WTFMove(*videoCapabilities),
107 *distinctiveIdentifier,
108 *persistentState,
109 WTFMove(*sessionTypes),
110 }};
111 }
112};
113
114} // namespace WebCore
115
116#endif // ENABLE(ENCRYPTED_MEDIA)
