26#pragma once
27
28#include "CSSValue.h"
29#include "CSSVariableData.h"
30#include "CSSVariableReferenceValue.h"
31#include "Length.h"
32#include "StyleImage.h"
33#include <wtf/Variant.h>
34
35namespace WebCore {
36
37class CSSParserToken;
38
39class CSSCustomPropertyValue final : public CSSValue {
40public:
41 using VariantValue = Variant<Monostate, Ref<CSSVariableReferenceValue>, CSSValueID, Ref<CSSVariableData>, Length, Ref<StyleImage>>;
42
43 static Ref<CSSCustomPropertyValue> createEmpty(const AtomString& name);
44
45 static Ref<CSSCustomPropertyValue> createUnresolved(const AtomString& name, Ref<CSSVariableReferenceValue>&& value)
46 {
47 return adoptRef(*new CSSCustomPropertyValue(name, { WTFMove(value) }));
48 }
49
50 static Ref<CSSCustomPropertyValue> createUnresolved(const AtomString& name, CSSValueID value)
51 {
52 return adoptRef(*new CSSCustomPropertyValue(name, { value }));
53 }
54
55 static Ref<CSSCustomPropertyValue> createWithID(const AtomString& name, CSSValueID id)
56 {
57 ASSERT(id == CSSValueInherit || id == CSSValueInitial || id == CSSValueUnset || id == CSSValueRevert || id == CSSValueInvalid);
58 return adoptRef(*new CSSCustomPropertyValue(name, { id }));
59 }
60
61 static Ref<CSSCustomPropertyValue> createSyntaxAll(const AtomString& name, Ref<CSSVariableData>&& value)
62 {
63 return adoptRef(*new CSSCustomPropertyValue(name, { WTFMove(value) }));
64 }
65
66 static Ref<CSSCustomPropertyValue> createSyntaxLength(const AtomString& name, Length value)
67 {
68 ASSERT(!value.isUndefined());
69 ASSERT(!value.isCalculated());
70 return adoptRef(*new CSSCustomPropertyValue(name, { WTFMove(value) }));
71 }
72
73 static Ref<CSSCustomPropertyValue> createSyntaxImage(const AtomString& name, Ref<StyleImage>&& value)
74 {
75 return adoptRef(*new CSSCustomPropertyValue(name, { WTFMove(value) }));
76 }
77
78 static Ref<CSSCustomPropertyValue> create(const CSSCustomPropertyValue& other)
79 {
80 return adoptRef(*new CSSCustomPropertyValue(other));
81 }
82
83 String customCSSText() const;
84
85 const AtomString& name() const { return m_name; }
86 bool isResolved() const { return !WTF::holds_alternative<Ref<CSSVariableReferenceValue>>(m_value); }
87 bool isUnset() const { return WTF::holds_alternative<CSSValueID>(m_value) && WTF::get<CSSValueID>(m_value) == CSSValueUnset; }
88 bool isInvalid() const { return WTF::holds_alternative<CSSValueID>(m_value) && WTF::get<CSSValueID>(m_value) == CSSValueInvalid; }
89
90 const VariantValue& value() const { return m_value; }
91
92 Vector<CSSParserToken> tokens() const;
93 bool equals(const CSSCustomPropertyValue&) const;
94
95private:
96 CSSCustomPropertyValue(const AtomString& name, VariantValue&& value)
97 : CSSValue(CustomPropertyClass)
98 , m_name(name)
99 , m_value(WTFMove(value))
100 {
101 }
102
103 CSSCustomPropertyValue(const CSSCustomPropertyValue& other)
104 : CSSValue(CustomPropertyClass)
105 , m_name(other.m_name)
106 , m_value(other.m_value)
107 , m_stringValue(other.m_stringValue)
108 {
109 }
110
111 const AtomString m_name;
112 const VariantValue m_value;
113 mutable String m_stringValue;
114};
115
116} // namespace WebCore
117
118SPECIALIZE_TYPE_TRAITS_CSS_VALUE(CSSCustomPropertyValue, isCustomPropertyValue())
