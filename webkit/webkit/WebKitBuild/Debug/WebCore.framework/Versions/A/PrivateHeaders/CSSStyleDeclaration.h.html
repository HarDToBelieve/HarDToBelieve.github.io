21#pragma once
22
23#include "CSSPropertyNames.h"
24#include "ExceptionOr.h"
25#include "ScriptWrappable.h"
26
27namespace WebCore {
28
29class CSSProperty;
30class CSSRule;
31class CSSStyleSheet;
32class CSSValue;
33class DeprecatedCSSOMValue;
34class MutableStyleProperties;
35class StyleProperties;
36class StyledElement;
37
38class CSSStyleDeclaration : public ScriptWrappable {
39 WTF_MAKE_NONCOPYABLE(CSSStyleDeclaration);
40 WTF_MAKE_ISO_ALLOCATED(CSSStyleDeclaration);
41public:
42 virtual ~CSSStyleDeclaration() = default;
43
44 virtual void ref() = 0;
45 virtual void deref() = 0;
46
47 virtual StyledElement* parentElement() const { return nullptr; }
48 virtual CSSRule* parentRule() const = 0;
49 virtual String cssText() const = 0;
50 virtual ExceptionOr<void> setCssText(const String&) = 0;
51 virtual unsigned length() const = 0;
52 virtual String item(unsigned index) const = 0;
53 virtual RefPtr<DeprecatedCSSOMValue> getPropertyCSSValue(const String& propertyName) = 0;
54 virtual String getPropertyValue(const String& propertyName) = 0;
55 virtual String getPropertyPriority(const String& propertyName) = 0;
56 virtual String getPropertyShorthand(const String& propertyName) = 0;
57 virtual bool isPropertyImplicit(const String& propertyName) = 0;
58 virtual ExceptionOr<void> setProperty(const String& propertyName, const String& value, const String& priority) = 0;
59 virtual ExceptionOr<String> removeProperty(const String& propertyName) = 0;
60
61 String cssFloat();
62 ExceptionOr<void> setCssFloat(const String&);
63
64 // CSSPropertyID versions of the CSSOM functions to support bindings and editing.
65 // Use the non-virtual methods in the concrete subclasses when possible.
66 // The CSSValue returned by this function should not be exposed to the web as it may be used by multiple documents at the same time.
67 virtual RefPtr<CSSValue> getPropertyCSSValueInternal(CSSPropertyID) = 0;
68 virtual String getPropertyValueInternal(CSSPropertyID) = 0;
69 virtual ExceptionOr<void> setPropertyInternal(CSSPropertyID, const String& value, bool important) = 0;
70
71 virtual Ref<MutableStyleProperties> copyProperties() const = 0;
72
73 virtual CSSStyleSheet* parentStyleSheet() const { return nullptr; }
74
75 virtual const Settings* settings() const;
76
77 // FIXME: It would be more efficient, by virtue of avoiding the text transformation and hash lookup currently
78 // required in the implementation, if we could could smuggle the CSSPropertyID through the bindings, perhaps
79 // by encoding it into the HashTableValue and then passing it together with the PropertyName.
80
81 // Shared implementation for all properties that match https://drafts.csswg.org/cssom/#dom-cssstyledeclaration-camel_cased_attribute.
82 String propertyValueForCamelCasedIDLAttribute(const AtomString&);
83 ExceptionOr<void> setPropertyValueForCamelCasedIDLAttribute(const AtomString&, const String&);
84
85 // Shared implementation for all properties that match https://drafts.csswg.org/cssom/#dom-cssstyledeclaration-webkit_cased_attribute.
86 String propertyValueForWebKitCasedIDLAttribute(const AtomString&);
87 ExceptionOr<void> setPropertyValueForWebKitCasedIDLAttribute(const AtomString&, const String&);
88
89 // Shared implementation for all properties that match https://drafts.csswg.org/cssom/#dom-cssstyledeclaration-dashed_attribute.
90 String propertyValueForDashedIDLAttribute(const AtomString&);
91 ExceptionOr<void> setPropertyValueForDashedIDLAttribute(const AtomString&, const String&);
92
93 // Shared implementation for all properties that match non-standard Epub-cased.
94 String propertyValueForEpubCasedIDLAttribute(const AtomString&);
95 ExceptionOr<void> setPropertyValueForEpubCasedIDLAttribute(const AtomString&, const String&);
96
97 // FIXME: This needs to pass in a Settings& to work correctly.
98 static CSSPropertyID getCSSPropertyIDFromJavaScriptPropertyName(const AtomString&);
99
100protected:
101 CSSStyleDeclaration() = default;
102};
103
104} // namespace WebCore
