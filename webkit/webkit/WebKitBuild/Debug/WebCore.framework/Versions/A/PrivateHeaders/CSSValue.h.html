57 static constexpr unsigned refCountFlagIsStatic = 0x1;
58 static constexpr unsigned refCountIncrement = 0x2; // This allows us to ref / deref without disturbing the static CSSValue flag.
59 void ref() const
60 {
61 m_refCount += refCountIncrement;
62 }
63 bool hasOneRef() const { return m_refCount == refCountIncrement; }
64 unsigned refCount() const { return m_refCount / refCountIncrement; }
65 bool hasAtLeastOneRef() const { return m_refCount; }
66
67 void deref()
68 {
69 // Customized deref() to ensure operator delete is called on
70 // the appropriate subclass type.
71 unsigned tempRefCount = m_refCount - refCountIncrement;
72 if (!tempRefCount) {
73IGNORE_ERRONEOUS_GCC_NULL_CHECK_WARNINGS_BEGIN
74 destroy();
75IGNORE_ERRONEOUS_GCC_NULL_CHECK_WARNINGS_END
76 return;
77 }
78 m_refCount = tempRefCount;
79 }

273template<typename CSSValueType>
274inline bool compareCSSValueVector(const Vector<Ref<CSSValueType>>& firstVector, const Vector<Ref<CSSValueType>>& secondVector)
275{
276 size_t size = firstVector.size();
277 if (size != secondVector.size())
278 return false;
279
280 for (size_t i = 0; i < size; ++i) {
281 auto& firstPtr = firstVector[i];
282 auto& secondPtr = secondVector[i];
283 if (firstPtr.ptr() == secondPtr.ptr() || firstPtr->equals(secondPtr))
284 continue;
285 return false;
286 }
287 return true;
288}

290template<typename CSSValueType>
291inline bool compareCSSValuePtr(const RefPtr<CSSValueType>& first, const RefPtr<CSSValueType>& second)
292{
293 return first ? second && first->equals(*second) : !second;
294}
