39namespace Style {

43class CSSVariableReferenceValue : public CSSValue {
44public:
45 static Ref<CSSVariableReferenceValue> create(const CSSParserTokenRange&);
46
47 bool equals(const CSSVariableReferenceValue&) const;
48 String customCSSText() const;
49
50 RefPtr<CSSVariableData> resolveVariableReferences(Style::BuilderState&) const;
51
52 // The maximum number of tokens that may be produced by a var()
53 // reference or var() fallback value.
54 // https://drafts.csswg.org/css-variables/#long-variables
55 static constexpr size_t maxSubstitutionTokens = 65536;
56
57private:
58 explicit CSSVariableReferenceValue(Ref<CSSVariableData>&&);
59
60 Ref<CSSVariableData> m_data;
61 mutable String m_stringValue;
62};

64} // namespace WebCore
