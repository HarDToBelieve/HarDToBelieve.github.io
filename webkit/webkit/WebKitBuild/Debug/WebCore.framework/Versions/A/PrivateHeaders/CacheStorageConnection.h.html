28
29#include "DOMCacheEngine.h"
30#include "RetrieveRecordsOptions.h"
31#include <wtf/HashMap.h>
32#include <wtf/ThreadSafeRefCounted.h>
33
34namespace WebCore {
35
36struct ClientOrigin;
37class FetchResponse;
38
39class CacheStorageConnection : public ThreadSafeRefCounted<CacheStorageConnection> {
40public:
41 virtual ~CacheStorageConnection() = default;
42
43 virtual void open(const ClientOrigin&, const String& cacheName, DOMCacheEngine::CacheIdentifierCallback&&) = 0;
44 virtual void remove(uint64_t cacheIdentifier, DOMCacheEngine::CacheIdentifierCallback&&) = 0;
45 virtual void retrieveCaches(const ClientOrigin&, uint64_t updateCounter, DOMCacheEngine::CacheInfosCallback&&) = 0;
46
47 virtual void retrieveRecords(uint64_t cacheIdentifier, const RetrieveRecordsOptions&, DOMCacheEngine::RecordsCallback&&) = 0;
48 virtual void batchDeleteOperation(uint64_t cacheIdentifier, const ResourceRequest&, CacheQueryOptions&&, DOMCacheEngine::RecordIdentifiersCallback&&) = 0;
49 virtual void batchPutOperation(uint64_t cacheIdentifier, Vector<DOMCacheEngine::Record>&&, DOMCacheEngine::RecordIdentifiersCallback&&) = 0;
50
51 virtual void reference(uint64_t /* cacheIdentifier */) = 0;
52 virtual void dereference(uint64_t /* cacheIdentifier */) = 0;
53
54 uint64_t computeRecordBodySize(const FetchResponse&, const DOMCacheEngine::ResponseBody&);
55
56 // Used only for testing purposes.
57 virtual void clearMemoryRepresentation(const ClientOrigin&, DOMCacheEngine::CompletionCallback&& callback) { callback(DOMCacheEngine::Error::NotImplemented); }
58 virtual void engineRepresentation(CompletionHandler<void(const String&)>&& callback) { callback(String { }); }
59 virtual void updateQuotaBasedOnSpaceUsage(const ClientOrigin&) { }
60
61private:
62 uint64_t computeRealBodySize(const DOMCacheEngine::ResponseBody&);
63
64protected:
65 HashMap<uint64_t, uint64_t> m_opaqueResponseToSizeWithPaddingMap;
66};

68} // namespace WebCore
