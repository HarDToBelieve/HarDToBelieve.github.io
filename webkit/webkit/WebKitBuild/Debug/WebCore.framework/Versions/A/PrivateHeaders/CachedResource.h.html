57// A resource that is held in the cache. Classes who want to use this object should derive
58// from CachedResourceClient, to get the function calls in case the requested data has arrived.
59// This class also does the actual communication with the loader to obtain the resource from the network.
60DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(CachedResource);
61class CachedResource {
62 WTF_MAKE_NONCOPYABLE(CachedResource);
63 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(CachedResource);
64 friend class MemoryCache;

66public:
67 enum class Type : uint8_t {
68 MainResource,
69 ImageResource,
70 CSSStyleSheet,
71 Script,
72 FontResource,
73 SVGFontResource,
74 MediaResource,
75#if ENABLE(MODEL_ELEMENT)
76 ModelResource,
77#endif
78 RawResource,
79 Icon,
80 Beacon,
81 Ping,
82#if ENABLE(XSLT)
83 XSLStyleSheet,
84#endif
85 LinkPrefetch,
86 TextTrackResource,
87#if ENABLE(APPLICATION_MANIFEST)
88 ApplicationManifest,
89#endif
90 SVGDocumentResource,
91 LastType = SVGDocumentResource,
92 };
93 static constexpr unsigned bitWidthOfType = 5;
94 static_assert(static_cast<unsigned>(Type::LastType) <= ((1U << bitWidthOfType) - 1));
95
96 enum Status : uint8_t {
97 Unknown, // let cache decide what to do with it
98 Pending, // only partially loaded
99 Cached, // regular case
100 LoadError,
101 DecodeError
102 };
103 static constexpr unsigned bitWidthOfStatus = 3;
104 static_assert(static_cast<unsigned>(DecodeError) <= ((1ULL << bitWidthOfStatus) - 1));
