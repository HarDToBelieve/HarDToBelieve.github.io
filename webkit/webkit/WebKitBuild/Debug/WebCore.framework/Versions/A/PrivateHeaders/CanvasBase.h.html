57class CanvasBase {
58public:
59 virtual ~CanvasBase();
60
61 virtual void refCanvasBase() = 0;
62 virtual void derefCanvasBase() = 0;
63
64 virtual bool isHTMLCanvasElement() const { return false; }
65 virtual bool isOffscreenCanvas() const { return false; }
66 virtual bool isCustomPaintCanvas() const { return false; }
67
68 virtual unsigned width() const { return m_size.width(); }
69 virtual unsigned height() const { return m_size.height(); }
70 const IntSize& size() const { return m_size; }
71
72 ImageBuffer* buffer() const;
73
74 virtual AffineTransform baseTransform() const;
75
76 void makeRenderingResultsAvailable();
77
78 size_t memoryCost() const;
79 size_t externalMemoryCost() const;
80
81 void setOriginClean() { m_originClean = true; }
82 void setOriginTainted() { m_originClean = false; }
83 bool originClean() const { return m_originClean; }
84
85 virtual SecurityOrigin* securityOrigin() const { return nullptr; }
86 ScriptExecutionContext* scriptExecutionContext() const { return canvasBaseScriptExecutionContext(); }
87
88 virtual CanvasRenderingContext* renderingContext() const = 0;
89
90 void addObserver(CanvasObserver&);
91 void removeObserver(CanvasObserver&);
92 void notifyObserversCanvasChanged(const std::optional<FloatRect>&);
93 void notifyObserversCanvasResized();
94 void notifyObserversCanvasDestroyed(); // Must be called in destruction before clearing m_context.
95
96 HashSet<Element*> cssCanvasClients() const;
97
98 virtual GraphicsContext* drawingContext() const;
99 virtual GraphicsContext* existingDrawingContext() const;
100
101 virtual void didDraw(const std::optional<FloatRect>&) = 0;
102
103 virtual Image* copiedImage() const = 0;
104 virtual void clearCopiedImage() const = 0;
105
106 bool hasActiveInspectorCanvasCallTracer() const;
107
108protected:
109 explicit CanvasBase(IntSize);
110
111 virtual ScriptExecutionContext* canvasBaseScriptExecutionContext() const = 0;
112
113 virtual void setSize(const IntSize& size) { m_size = size; }
114
115 RefPtr<ImageBuffer> setImageBuffer(RefPtr<ImageBuffer>&&) const;
116 virtual bool hasCreatedImageBuffer() const { return false; }
117 static size_t activePixelMemory();
118
119 void resetGraphicsContextState() const;
120
121private:
122 virtual void createImageBuffer() const { }
123
124 mutable IntSize m_size;
125 mutable Lock m_imageBufferAssignmentLock;
126 mutable RefPtr<ImageBuffer> m_imageBuffer;
127 mutable size_t m_imageBufferCost { 0 };
128 mutable std::unique_ptr<GraphicsContextStateSaver> m_contextStateSaver;
129
130 bool m_originClean { true };
131#if ASSERT_ENABLED
132 bool m_didNotifyObserversCanvasDestroyed { false };
133#endif
134 HashSet<CanvasObserver*> m_observers;
135};
