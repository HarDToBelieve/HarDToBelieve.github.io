33template<typename, size_t> struct ColorComponents;
34
35template<size_t ColumnCount, size_t RowCount>
36class ColorMatrix {
37public:
38 template<typename ...Ts>
39 explicit constexpr ColorMatrix(Ts ...input)
40 : m_matrix {{ input ... }}
41 {
42 static_assert(sizeof...(Ts) == RowCount * ColumnCount);
43 }
44
45 constexpr ColorComponents<float, 4> transformedColorComponents(const ColorComponents<float, 4>&) const;
46
47 constexpr float at(size_t row, size_t column) const
48 {
49 return m_matrix[(row * ColumnCount) + column];
50 }
51
52private:
53 std::array<float, RowCount * ColumnCount> m_matrix;
54};
55
56template<size_t ColumnCount, size_t RowCount>
57constexpr bool operator==(const ColorMatrix<ColumnCount, RowCount>& a, const ColorMatrix<ColumnCount, RowCount>& b)
58{
59 for (size_t row = 0; row < RowCount; ++row) {
60 for (size_t column = 0; column < ColumnCount; ++column) {
61 if (a.at(row, column) != b.at(row, column))
62 return false;
63 }
64 }
65 return true;
66}
67
68template<size_t ColumnCount, size_t RowCount>
69constexpr bool operator!=(const ColorMatrix<ColumnCount, RowCount>& a, const ColorMatrix<ColumnCount, RowCount>& b)
70{
71 return !(a == b);
72}

76constexpr ColorMatrix<3, 3> grayscaleColorMatrix(float amount)
77{
78 // Values from https://www.w3.org/TR/filter-effects-1/#grayscaleEquivalent
79 float oneMinusAmount = std::clamp(1.0f - amount, 0.0f, 1.0f);
80 return ColorMatrix<3, 3> {
81 0.2126f + 0.7874f * oneMinusAmount, 0.7152f - 0.7152f * oneMinusAmount, 0.0722f - 0.0722f * oneMinusAmount,
82 0.2126f - 0.2126f * oneMinusAmount, 0.7152f + 0.2848f * oneMinusAmount, 0.0722f - 0.0722f * oneMinusAmount,
83 0.2126f - 0.2126f * oneMinusAmount, 0.7152f - 0.7152f * oneMinusAmount, 0.0722f + 0.9278f * oneMinusAmount
84 };
85}

87constexpr ColorMatrix<3, 3> sepiaColorMatrix(float amount)
88{

98constexpr ColorMatrix<3, 3> saturationColorMatrix(float amount)
99{
100 // Values from https://www.w3.org/TR/filter-effects-1/#feColorMatrixElement
101 return ColorMatrix<3, 3> {
102 0.213f + 0.787f * amount, 0.715f - 0.715f * amount, 0.072f - 0.072f * amount,
103 0.213f - 0.213f * amount, 0.715f + 0.285f * amount, 0.072f - 0.072f * amount,
104 0.213f - 0.213f * amount, 0.715f - 0.715f * amount, 0.072f + 0.928f * amount
105 };
106}

108// NOTE: hueRotateColorMatrix is not constexpr due to use of cos/sin which are not constexpr yet.
109inline ColorMatrix<3, 3> hueRotateColorMatrix(float angleInDegrees)
110{
111 float cosHue = cos(deg2rad(angleInDegrees));
112 float sinHue = sin(deg2rad(angleInDegrees));
113
114 // Values from https://www.w3.org/TR/filter-effects-1/#feColorMatrixElement
115 return ColorMatrix<3, 3> {
116 0.213f + cosHue * 0.787f - sinHue * 0.213f, 0.715f - cosHue * 0.715f - sinHue * 0.715f, 0.072f - cosHue * 0.072f + sinHue * 0.928f,
117 0.213f - cosHue * 0.213f + sinHue * 0.143f, 0.715f + cosHue * 0.285f + sinHue * 0.140f, 0.072f - cosHue * 0.072f - sinHue * 0.283f,
118 0.213f - cosHue * 0.213f - sinHue * 0.787f, 0.715f - cosHue * 0.715f + sinHue * 0.715f, 0.072f + cosHue * 0.928f + sinHue * 0.072f
119 };
120}

122template<size_t ColumnCount, size_t RowCount>
123constexpr ColorComponents<float, 4> ColorMatrix<ColumnCount, RowCount>::transformedColorComponents(const ColorComponents<float, 4>& inputVector) const
124{
125 static_assert(ColorComponents<float, 4>::Size >= RowCount);
126
127 ColorComponents<float, 4> result;
128 for (size_t row = 0; row < RowCount; ++row) {
129 if constexpr (ColumnCount <= ColorComponents<float, 4>::Size) {
130 for (size_t column = 0; column < ColumnCount; ++column)
131 result[row] += at(row, column) * inputVector[column];
132 } else if constexpr (ColumnCount > ColorComponents<float, 4>::Size) {
133 for (size_t column = 0; column < ColorComponents<float, 4>::Size; ++column)
134 result[row] += at(row, column) * inputVector[column];
135 for (size_t additionalColumn = ColorComponents<float, 4>::Size; additionalColumn < ColumnCount; ++additionalColumn)
136 result[row] += at(row, additionalColumn);
137 }
138 }
139 if constexpr (ColorComponents<float, 4>::Size > RowCount) {
140 for (size_t additionalRow = RowCount; additionalRow < ColorComponents<float, 4>::Size; ++additionalRow)
141 result[additionalRow] = inputVector[additionalRow];
142 }
143
144 return result;
145}

147template<typename T, typename M> inline constexpr auto applyMatricesToColorComponents(const ColorComponents<T, 4>& components, M matrix) -> ColorComponents<T, 4>
148{
149 return matrix.transformedColorComponents(components);
150}
151
152template<typename T, typename M, typename... Matrices> inline constexpr auto applyMatricesToColorComponents(const ColorComponents<T, 4>& components, M matrix, Matrices... matrices) -> ColorComponents<T, 4>
153{
154 return applyMatricesToColorComponents(matrix.transformedColorComponents(components), matrices...);
155}
