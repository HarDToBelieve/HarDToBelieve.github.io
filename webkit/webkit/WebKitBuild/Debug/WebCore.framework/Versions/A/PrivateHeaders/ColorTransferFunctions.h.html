32// Transfer functions for colors that can be gamma encoded.
33
34enum class TransferFunctionMode : uint8_t {
35 Clamped,
36 Unclamped
37};
38
39template<typename T, TransferFunctionMode mode>
40struct A98RGBTransferFunction {
41 static T toGammaEncoded(T);
42 static T toLinear(T);
43};
44
45template<typename T, TransferFunctionMode mode>
46struct ProPhotoRGBTransferFunction {
47 static constexpr T gamma = 1.8;
48
49 static T toGammaEncoded(T);
50 static T toLinear(T);
51};
52
53template<typename T, TransferFunctionMode mode>
54struct Rec2020TransferFunction {
55 static constexpr T alpha = 1.09929682680944;
56 static constexpr T beta = 0.018053968510807;
57 static constexpr T gamma = 0.45;
58
59 static T toGammaEncoded(T);
60 static T toLinear(T);
61};
62
63template<typename T, TransferFunctionMode mode>
64struct SRGBTransferFunction {
65 static T toGammaEncoded(T);
66 static T toLinear(T);
67};
68
69// MARK: A98RGBTransferFunction.
70
71template<typename T, TransferFunctionMode mode> T A98RGBTransferFunction<T, mode>::toGammaEncoded(T c)
72{
73 auto sign = std::signbit(c) ? -1.0f : 1.0f;
74 auto result = std::pow(std::abs(c), 256.0f / 563.0f) * sign;
75 if constexpr (mode == TransferFunctionMode::Clamped)
76 return clampTo<T>(result, 0, 1);
77 return result;
78}
79
80template<typename T, TransferFunctionMode mode> T A98RGBTransferFunction<T, mode>::toLinear(T c)
81{
82 auto sign = std::signbit(c) ? -1.0f : 1.0f;
83 auto result = std::pow(std::abs(c), 563.0f / 256.0f) * sign;
84 if constexpr (mode == TransferFunctionMode::Clamped)
85 return clampTo<T>(result, 0, 1);
86 return result;
87}
88
89// MARK: ProPhotoRGBTransferFunction.
90
91template<typename T, TransferFunctionMode mode> T ProPhotoRGBTransferFunction<T, mode>::toGammaEncoded(T c)
92{
93 if constexpr (mode == TransferFunctionMode::Clamped) {
94 if (c < 1.0 / 512.0)
95 return 16.0 * c;
96
97 return clampTo<T>(std::pow(c, 1.0 / gamma), 0, 1);
98 } else {
99 if (std::abs(c) < 1.0 / 512.0)
100 return 16.0 * c;
101
102 float sign = std::signbit(c) ? -1.0 : 1.0;
103 return std::pow(c, 1.0 / gamma) * sign;
104 }
105}
106
107template<typename T, TransferFunctionMode mode> T ProPhotoRGBTransferFunction<T, mode>::toLinear(T c)
108{
109 if constexpr (mode == TransferFunctionMode::Clamped) {
110 if (c <= 16.0 / 512.0)
111 return c / 16.0;
112
113 return clampTo<T>(std::pow(c, gamma), 0, 1);
114 } else {
115 if (std::abs(c) <= 16.0 / 512.0)
116 return c / 16.0;
117
118 float sign = std::signbit(c) ? -1.0 : 1.0;
119 return std::pow(c, gamma) * sign;
120 }
121}
122
123// MARK: Rec2020TransferFunction.
124
125template<typename T, TransferFunctionMode mode> T Rec2020TransferFunction<T, mode>::toGammaEncoded(T c)
126{
127 if constexpr (mode == TransferFunctionMode::Clamped) {
128 if (c <= beta)
129 return 4.5f * c;
130
131 return clampTo<T>(alpha * std::pow(c, gamma) - (alpha - 1.0f), 0, 1);
132 } else {
133 if (std::abs(c) <= beta)
134 return 4.5f * c;
135
136 float sign = std::signbit(c) ? -1.0 : 1.0;
137 return (alpha * std::pow(c, gamma) - (alpha - 1.0)) * sign;
138 }
139}
140
141template<typename T, TransferFunctionMode mode> T Rec2020TransferFunction<T, mode>::toLinear(T c)
142{
143 if constexpr (mode == TransferFunctionMode::Clamped) {
144 if (c < beta * 4.5f)
145 return c / 4.5f;
146 return clampTo<T>(std::pow((c + alpha - 1.0) / alpha, 1.0 / gamma), 0, 1);
147 } else {
148 if (std::abs(c) < beta * 4.5f)
149 return c / 4.5f;
150
151 float sign = std::signbit(c) ? -1.0 : 1.0;
152 return std::pow((c + alpha - 1.0) / alpha, 1.0 / gamma) * sign;
153 }
154}
155
156// MARK: SRGBTransferFunction.
157
158template<typename T, TransferFunctionMode mode> T SRGBTransferFunction<T, mode>::toGammaEncoded(T c)
159{
160 if constexpr (mode == TransferFunctionMode::Clamped) {
161 if (c < 0.0031308f)
162 return std::max<T>(12.92f * c, 0);
163 return clampTo<T>(1.055f * std::pow(c, 1.0f / 2.4f) - 0.055f, 0, 1);
164 } else {
165 auto sign = std::signbit(c) ? -1.0f : 1.0f;
166 c = std::abs(c);
167
168 if (c < 0.0031308f)
169 return 12.92f * c * sign;
170 return (1.055f * std::pow(c, 1.0f / 2.4f) - 0.055f) * sign;
171 }
172}
173
174template<typename T, TransferFunctionMode mode> T SRGBTransferFunction<T, mode>::toLinear(T c)
175{
176 if constexpr (mode == TransferFunctionMode::Clamped) {
177 if (c <= 0.04045f)
178 return std::max<float>(c / 12.92f, 0);
179 return clampTo<float>(std::pow((c + 0.055f) / 1.055f, 2.4f), 0, 1);
180 } else {
181 auto sign = std::signbit(c) ? -1.0f : 1.0f;
182 c = std::abs(c);
183
184 if (c <= 0.04045f)
185 return c / 12.92f * sign;
186 return std::pow((c + 0.055f) / 1.055f, 2.4f) * sign;
187 }
188}
