26#pragma once
27
28#include "Color.h"
29
30namespace WebCore {
31
32struct CompositionHighlight {
33 CompositionHighlight() = default;
34 CompositionHighlight(unsigned startOffset, unsigned endOffset, const Color& c)
35 : startOffset(startOffset)
36 , endOffset(endOffset)
37 , color(c)
38 {
39 }
40
41#if PLATFORM(IOS_FAMILY)
42 static constexpr auto defaultCompositionFillColor = SRGBA<uint8_t> { 175, 192, 227, 60 };
43#else
44 static constexpr auto defaultCompositionFillColor = SRGBA<uint8_t> { 225, 221, 85 };
45#endif
46
47 unsigned startOffset { 0 };
48 unsigned endOffset { 0 };
49 Color color { defaultCompositionFillColor };
50
51 template<class Encoder> void encode(Encoder&) const;
52 template<class Decoder> static std::optional<CompositionHighlight> decode(Decoder&);
53};
54
55template<class Encoder>
56void CompositionHighlight::encode(Encoder& encoder) const
57{
58 encoder << startOffset;
59 encoder << endOffset;
60 encoder << color;
61}
62
63template<class Decoder>
64std::optional<CompositionHighlight> CompositionHighlight::decode(Decoder& decoder)
65{
66 std::optional<unsigned> startOffset;
67 decoder >> startOffset;
68 if (!startOffset)
69 return std::nullopt;
70
71 std::optional<unsigned> endOffset;
72 decoder >> endOffset;
73 if (!endOffset)
74 return std::nullopt;
75
76 std::optional<Color> color;
77 decoder >> color;
78 if (!color)
79 return std::nullopt;
80
81 return {{ *startOffset, *endOffset, *color }};
82}
83
84} // namespace WebCore
