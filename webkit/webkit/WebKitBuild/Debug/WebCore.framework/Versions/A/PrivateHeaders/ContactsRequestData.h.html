26#pragma once
27
28#include "ContactProperty.h"
29#include <wtf/Forward.h>
30#include <wtf/Vector.h>
31#include <wtf/text/WTFString.h>
32
33namespace WebCore {
34
35struct ContactsRequestData {
36 Vector<ContactProperty> properties;
37 bool multiple { false };
38 String url;
39
40 template<class Encoder> void encode(Encoder&) const;
41 template<class Decoder> static std::optional<ContactsRequestData> decode(Decoder&);
42};
43
44template<class Encoder>
45void ContactsRequestData::encode(Encoder& encoder) const
46{
47 encoder << properties;
48 encoder << multiple;
49 encoder << url;
50}
51
52template<class Decoder>
53std::optional<ContactsRequestData> ContactsRequestData::decode(Decoder& decoder)
54{
55 std::optional<Vector<ContactProperty>> properties;
56 decoder >> properties;
57 if (!properties)
58 return std::nullopt;
59
60 std::optional<bool> multiple;
61 decoder >> multiple;
62 if (!multiple)
63 return std::nullopt;
64
65 std::optional<String> url;
66 decoder >> url;
67 if (!url)
68 return std::nullopt;
69
70 return {{ *properties, *multiple, *url }};
71}
