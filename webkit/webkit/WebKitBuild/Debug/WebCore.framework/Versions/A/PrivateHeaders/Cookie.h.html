41namespace WebCore {
42
43struct Cookie {
44 Cookie() = default;
45
46 template<class Encoder> void encode(Encoder&) const;
47 template<class Decoder> static std::optional<Cookie> decode(Decoder&);
48
49 WEBCORE_EXPORT bool operator==(const Cookie&) const;
50 WEBCORE_EXPORT unsigned hash() const;
51
52#ifdef __OBJC__
53 WEBCORE_EXPORT Cookie(NSHTTPCookie *);
54 WEBCORE_EXPORT operator NSHTTPCookie *() const;
55#elif USE(SOUP)
56 explicit Cookie(SoupCookie*);
57 SoupCookie* toSoupCookie() const;
58#endif
59
60 bool isNull() const
61 {
62 return name.isNull()
63 && value.isNull()
64 && domain.isNull()
65 && path.isNull()
66 && !created
67 && !expires
68 && !httpOnly
69 && !secure
70 && !session
71 && comment.isNull()
72 && commentURL.isNull();
73 }
74
75 bool isKeyEqual(const Cookie& otherCookie) const
76 {
77 return name == otherCookie.name
78 && domain == otherCookie.domain
79 && path == otherCookie.path;
80 }
81
82 String name;
83 String value;
84 String domain;
85 String path;
86 // Creation and expiration dates are expressed as milliseconds since the UNIX epoch.
87 double created { 0 };
88 std::optional<double> expires;
89 bool httpOnly { false };
90 bool secure { false };
91 bool session { false };
92 String comment;
93 URL commentURL;
94 Vector<uint16_t> ports;
95
96 enum class SameSitePolicy { None, Lax, Strict };
97 SameSitePolicy sameSite { SameSitePolicy::None };
98};
99
100struct CookieHash {
101 static unsigned hash(const Cookie& key)
102 {
103 return key.hash();
104 }
105
106 static bool equal(const Cookie& a, const Cookie& b)
107 {
108 return a == b;
109 }
110 static const bool safeToCompareToEmptyOrDeleted = false;
111};
112
113template<class Encoder>
114void Cookie::encode(Encoder& encoder) const
115{
116 encoder << name;
117 encoder << value;
118 encoder << domain;
119 encoder << path;
120 encoder << created;
121 encoder << expires;
122 encoder << httpOnly;
123 encoder << secure;
124 encoder << session;
125 encoder << comment;
126 encoder << commentURL;
127 encoder << ports;
128 encoder << sameSite;
129}
130
131template<class Decoder>
132std::optional<Cookie> Cookie::decode(Decoder& decoder)
133{
134 Cookie cookie;
135 if (!decoder.decode(cookie.name))
136 return std::nullopt;
137 if (!decoder.decode(cookie.value))
138 return std::nullopt;
139 if (!decoder.decode(cookie.domain))
140 return std::nullopt;
141 if (!decoder.decode(cookie.path))
142 return std::nullopt;
143 if (!decoder.decode(cookie.created))
144 return std::nullopt;
145 if (!decoder.decode(cookie.expires))
146 return std::nullopt;
147 if (!decoder.decode(cookie.httpOnly))
148 return std::nullopt;
149 if (!decoder.decode(cookie.secure))
150 return std::nullopt;
151 if (!decoder.decode(cookie.session))
152 return std::nullopt;
153 if (!decoder.decode(cookie.comment))
154 return std::nullopt;
155 if (!decoder.decode(cookie.commentURL))
156 return std::nullopt;
157 if (!decoder.decode(cookie.ports))
158 return std::nullopt;
159 if (!decoder.decode(cookie.sameSite))
160 return std::nullopt;
161 return cookie;
162}
