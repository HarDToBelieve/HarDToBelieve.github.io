26#pragma once
27
28#include "CookieJar.h"
29#include "FrameIdentifier.h"
30#include "PageIdentifier.h"
31#include "SameSiteInfo.h"
32#include <wtf/URL.h>
33
34namespace WebCore {
35
36struct CookieRequestHeaderFieldProxy {
37 URL firstParty;
38 SameSiteInfo sameSiteInfo;
39 URL url;
40 std::optional<FrameIdentifier> frameID;
41 std::optional<PageIdentifier> pageID;
42 IncludeSecureCookies includeSecureCookies { IncludeSecureCookies::No };
43
44 template<class Encoder> void encode(Encoder&) const;
45 template<class Decoder> static std::optional<CookieRequestHeaderFieldProxy> decode(Decoder&);
46};
47
48template<class Encoder>
49void CookieRequestHeaderFieldProxy::encode(Encoder& encoder) const
50{
51 encoder << firstParty;
52 encoder << sameSiteInfo;
53 encoder << url;
54 encoder << frameID;
55 encoder << pageID;
56 encoder << includeSecureCookies;
57}
58
59template<class Decoder>
60std::optional<CookieRequestHeaderFieldProxy> CookieRequestHeaderFieldProxy::decode(Decoder& decoder)
61{
62 URL firstParty;
63 if (!decoder.decode(firstParty))
64 return std::nullopt;
65
66 SameSiteInfo sameSiteInfo;
67 if (!decoder.decode(sameSiteInfo))
68 return std::nullopt;
69
70 URL url;
71 if (!decoder.decode(url))
72 return std::nullopt;
73
74 std::optional<std::optional<FrameIdentifier>> frameID;
75 decoder >> frameID;
76 if (!frameID)
77 return std::nullopt;
78
79 std::optional<std::optional<PageIdentifier>> pageID;
80 decoder >> pageID;
81 if (!pageID)
82 return std::nullopt;
83
84 IncludeSecureCookies includeSecureCookies;
85 if (!decoder.decode(includeSecureCookies))
86 return std::nullopt;
87
88 return CookieRequestHeaderFieldProxy { WTFMove(firstParty), WTFMove(sameSiteInfo), WTFMove(url), *frameID, *pageID, includeSecureCookies };
89}

91} // namespace WebCore
