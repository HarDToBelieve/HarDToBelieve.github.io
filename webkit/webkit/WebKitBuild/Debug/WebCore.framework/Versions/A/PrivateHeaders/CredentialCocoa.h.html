28#include "CredentialBase.h"
29#include <Security/SecBase.h>
30#include <wtf/RetainPtr.h>

32#if USE(CFURLCONNECTION)
33#include <pal/spi/cf/CFNetworkSPI.h>

36OBJC_CLASS NSURLCredential;
37
38namespace WebCore {
39
40class Credential : public CredentialBase {
41public:
42 Credential()
43 : CredentialBase()
44 {
45 }
46
47 Credential(const String& user, const String& password, CredentialPersistence persistence)
48 : CredentialBase(user, password, persistence)
49 {
50 }
51
52 Credential(const Credential&, CredentialPersistence);
53
54#if USE(CFURLCONNECTION)
55 explicit Credential(CFURLCredentialRef);
56#endif
57 WEBCORE_EXPORT explicit Credential(NSURLCredential *);
58
59 WEBCORE_EXPORT bool isEmpty() const;
60
61 bool encodingRequiresPlatformData() const { return m_nsCredential && encodingRequiresPlatformData(m_nsCredential.get()); }
62
63#if USE(CFURLCONNECTION)
64 CFURLCredentialRef cfCredential() const;
65#endif
66 WEBCORE_EXPORT NSURLCredential *nsCredential() const;
67
68 static bool platformCompare(const Credential&, const Credential&);
69
70private:
71 WEBCORE_EXPORT static bool encodingRequiresPlatformData(NSURLCredential *);
72
73 mutable RetainPtr<NSURLCredential> m_nsCredential;
74};
75
76} // namespace WebCore
