26#pragma once
27
28#if ENABLE(WEB_CRYPTO)
29
30#include "CryptoKey.h"
31#include "ExceptionOr.h"
32#include <wtf/Function.h>
33#include <wtf/Vector.h>
34
35namespace WebCore {
36
37class CryptoAlgorithmParameters;
38struct JsonWebKey;
39
40class CryptoKeyHMAC final : public CryptoKey {
41public:
42 static Ref<CryptoKeyHMAC> create(const Vector<uint8_t>& key, CryptoAlgorithmIdentifier hash, bool extractable, CryptoKeyUsageBitmap usage)
43 {
44 return adoptRef(*new CryptoKeyHMAC(key, hash, extractable, usage));
45 }
46 virtual ~CryptoKeyHMAC();
47
48 static RefPtr<CryptoKeyHMAC> generate(size_t lengthBits, CryptoAlgorithmIdentifier hash, bool extractable, CryptoKeyUsageBitmap);
49 static RefPtr<CryptoKeyHMAC> importRaw(size_t lengthBits, CryptoAlgorithmIdentifier hash, Vector<uint8_t>&& keyData, bool extractable, CryptoKeyUsageBitmap);
50 using CheckAlgCallback = Function<bool(CryptoAlgorithmIdentifier, const String&)>;
51 static RefPtr<CryptoKeyHMAC> importJwk(size_t lengthBits, CryptoAlgorithmIdentifier hash, JsonWebKey&&, bool extractable, CryptoKeyUsageBitmap, CheckAlgCallback&&);
52
53 CryptoKeyClass keyClass() const final { return CryptoKeyClass::HMAC; }
54
55 const Vector<uint8_t>& key() const { return m_key; }
56 JsonWebKey exportJwk() const;
57
58 CryptoAlgorithmIdentifier hashAlgorithmIdentifier() const { return m_hash; }
59
60 static ExceptionOr<size_t> getKeyLength(const CryptoAlgorithmParameters&);
61
62private:
63 CryptoKeyHMAC(const Vector<uint8_t>& key, CryptoAlgorithmIdentifier hash, bool extractable, CryptoKeyUsageBitmap);
64 CryptoKeyHMAC(Vector<uint8_t>&& key, CryptoAlgorithmIdentifier hash, bool extractable, CryptoKeyUsageBitmap);
65
66 KeyAlgorithm algorithm() const final;
67
68 CryptoAlgorithmIdentifier m_hash;
69 Vector<uint8_t> m_key;
70};
71
72} // namespace WebCore
73
74SPECIALIZE_TYPE_TRAITS_CRYPTO_KEY(CryptoKeyHMAC, CryptoKeyClass::HMAC)
75
76#endif // ENABLE(WEB_CRYPTO)
