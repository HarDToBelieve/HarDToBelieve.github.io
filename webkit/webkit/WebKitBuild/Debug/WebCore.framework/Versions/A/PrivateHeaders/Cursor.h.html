140 WEBCORE_EXPORT Cursor(Image*, const IntPoint& hotSpot, float imageScaleFactor);
141#endif
142
143 explicit Cursor(Type);
144
145 Type type() const;
146 Image* image() const { return m_image.get(); }
147 const IntPoint& hotSpot() const { return m_hotSpot; }
148
149#if ENABLE(MOUSE_CURSOR_SCALE)
150 // Image scale in image pixels per logical (UI) pixel.
151 float imageScaleFactor() const { return m_imageScaleFactor; }
152#endif
153
154 WEBCORE_EXPORT PlatformCursor platformCursor() const;
155
156 WEBCORE_EXPORT void setAsPlatformCursor() const;
157
158private:
159 void ensurePlatformCursor() const;
160
161 // The type of -1 indicates an invalid Cursor that should never actually get used.
162 Type m_type { static_cast<Type>(-1) };
163 RefPtr<Image> m_image;
164 IntPoint m_hotSpot;
165
166#if ENABLE(MOUSE_CURSOR_SCALE)
167 float m_imageScaleFactor { 1 };
168#endif
169
170#if !HAVE(NSCURSOR)
171 mutable PlatformCursor m_platformCursor { nullptr };
172#else
173 mutable RetainPtr<NSCursor> m_platformCursor;
174#endif
