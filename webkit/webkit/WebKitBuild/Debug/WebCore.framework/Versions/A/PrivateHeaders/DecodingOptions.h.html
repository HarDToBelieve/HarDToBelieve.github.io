26#pragma once
27
28#include "IntSize.h"
29#include <optional>
30#include <wtf/EnumTraits.h>
31#include <wtf/Variant.h>
32
33namespace WebCore {
34
35enum class DecodingMode : uint8_t {
36 Auto,
37 Synchronous,
38 Asynchronous
39};
40
41class DecodingOptions {
42public:
43 explicit DecodingOptions(DecodingMode decodingMode = DecodingMode::Auto)
44 : m_decodingModeOrSize(decodingMode)
45 {
46 }
47
48 DecodingOptions(const std::optional<IntSize>& sizeForDrawing)
49 : m_decodingModeOrSize(sizeForDrawing)
50 {
51 }
52
53 bool operator==(const DecodingOptions& other) const
54 {
55 return m_decodingModeOrSize == other.m_decodingModeOrSize;
56 }
57
58 bool isAuto() const
59 {
60 return hasDecodingMode() && WTF::get<DecodingMode>(m_decodingModeOrSize) == DecodingMode::Auto;
61 }
62
63 bool isSynchronous() const
64 {
65 return hasDecodingMode() && WTF::get<DecodingMode>(m_decodingModeOrSize) == DecodingMode::Synchronous;
66 }
67
68 bool isAsynchronous() const
69 {
70 return hasDecodingMode() && WTF::get<DecodingMode>(m_decodingModeOrSize) == DecodingMode::Asynchronous;
71 }
72
73 bool isAsynchronousCompatibleWith(const DecodingOptions& decodingOptions) const
74 {
75 if (isAuto() || decodingOptions.isAuto())
76 return false;
77
78 // Comparing DecodingOptions with isAsynchronous() should not happen.
79 ASSERT(!isAsynchronous());
80 if (isAsynchronous() || decodingOptions.isSynchronous())
81 return false;
82
83 // If the image was synchronously decoded, then it should fit any size.
84 // If we want an image regardless of its size, then the current decoded
85 // image should be fine.
86 if (isSynchronous() || decodingOptions.isAsynchronous())
87 return true;
88
89 ASSERT(decodingOptions.hasSize());
90 if (decodingOptions.hasFullSize())
91 return hasFullSize();
92
93 ASSERT(decodingOptions.hasSizeForDrawing());
94 if (hasFullSize())
95 return true;
96
97 ASSERT(hasSizeForDrawing());
98 return maxDimension(*sizeForDrawing()) >= maxDimension(*decodingOptions.sizeForDrawing());
99 }
100
101 bool hasFullSize() const
102 {
103 return hasSize() && !sizeForDrawing();
104 }
105
106 bool hasSizeForDrawing() const
107 {
108 return hasSize() && sizeForDrawing();
109 }
110
111 std::optional<IntSize> sizeForDrawing() const
112 {
113 ASSERT(hasSize());
114 return WTF::get<std::optional<IntSize>>(m_decodingModeOrSize);
115 }
116
117 static int maxDimension(const IntSize& size)
118 {
119 return std::max(size.width(), size.height());
120 }
121
122private:
123 template<typename T>
124 bool has() const
125 {
126 return WTF::holds_alternative<T>(m_decodingModeOrSize);
127 }
128
129 bool hasDecodingMode() const
130 {
131 return has<DecodingMode>();
132 }
133
134 bool hasSize() const
135 {
136 return has<std::optional<IntSize>>();
137 }
138
139 // Four states of the decoding:
140 // - Synchronous: DecodingMode::Synchronous
141 // - Asynchronous + anySize: DecodingMode::Asynchronous
142 // - Asynchronous + intrinsicSize: an empty std::optional<IntSize>>
143 // - Asynchronous + sizeForDrawing: a none empty std::optional<IntSize>>
144 using DecodingModeOrSize = Variant<DecodingMode, std::optional<IntSize>>;
145 DecodingModeOrSize m_decodingModeOrSize;
146};
147
148} // namespace WebCore

157 WebCore::DecodingMode::Asynchronous
158 >;
159};
