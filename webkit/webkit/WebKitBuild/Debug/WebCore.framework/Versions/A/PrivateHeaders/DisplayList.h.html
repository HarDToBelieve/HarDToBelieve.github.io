48enum AsTextFlag {
49 None = 0,
50 IncludesPlatformOperations = 1 << 0,
51};
52
53typedef unsigned AsTextFlags;

55class DisplayList {

57 friend class Recorder;
58 friend class Replayer;

68 void dump(WTF::TextStream&) const;

72 WEBCORE_EXPORT size_t sizeInBytes() const;
73
74 String asText(AsTextFlags) const;
75
76 const ImageBufferHashMap& imageBuffers() const { return m_imageBuffers; }
77 const NativeImageHashMap& nativeImages() const { return m_nativeImages; }
78 const FontRenderingResourceMap& fonts() const { return m_fonts; }
79
80 WEBCORE_EXPORT void setItemBufferReadingClient(ItemBufferReadingClient*);

101 WEBCORE_EXPORT Iterator begin() const;

104private:
105 ItemBuffer* itemBufferIfExists() const { return m_items.get(); }
106 WEBCORE_EXPORT ItemBuffer& itemBuffer();
107
108 void addDrawingItemExtent(std::optional<FloatRect>&& extent)
109 {
110 ASSERT(m_tracksDrawingItemExtents);
111 m_drawingItemExtents.append(WTFMove(extent));
112 }
113
114 void cacheImageBuffer(WebCore::ImageBuffer& imageBuffer)
115 {
116 m_imageBuffers.ensure(imageBuffer.renderingResourceIdentifier(), [&]() {
117 return makeRef(imageBuffer);
118 });
119 }
120
121 void cacheNativeImage(NativeImage& image)
122 {
123 m_nativeImages.ensure(image.renderingResourceIdentifier(), [&]() {
124 return makeRef(image);
125 });
126 }
127
128 void cacheFont(Font& font)
129 {
130 m_fonts.ensure(font.renderingResourceIdentifier(), [&]() {
131 return makeRef(font);
132 });
133 }
134
135 static bool shouldDumpForFlags(AsTextFlags, ItemHandle);
136
137 ImageBufferHashMap m_imageBuffers;
138 NativeImageHashMap m_nativeImages;
139 FontRenderingResourceMap m_fonts;
140 std::unique_ptr<ItemBuffer> m_items;
141 Vector<std::optional<FloatRect>> m_drawingItemExtents;
142 bool m_tracksDrawingItemExtents { true };
143};
144
145template<typename T, class... Args>
146void DisplayList::append(Args&&... args)
147{
148 itemBuffer().append<T>(std::forward<Args>(args)...);
149}
