36namespace WebCore {
37
38class DisplayAnimationClient;
39class DisplayRefreshMonitorClient;
40class DisplayRefreshMonitorFactory;
41
42class DisplayRefreshMonitor : public ThreadSafeRefCounted<DisplayRefreshMonitor> {
43 friend class DisplayRefreshMonitorManager;
44public:
45 static RefPtr<DisplayRefreshMonitor> create(DisplayRefreshMonitorFactory*, PlatformDisplayID);
46 WEBCORE_EXPORT virtual ~DisplayRefreshMonitor();
47
48 WEBCORE_EXPORT virtual void stop();
49
50 // Return true if callback request was scheduled, false if it couldn't be
51 // (e.g., hardware refresh is not available)
52 WEBCORE_EXPORT virtual bool requestRefreshCallback();
53
54 void windowScreenDidChange(PlatformDisplayID);
55
56 bool hasClients() const { return m_clients.size(); }
57 void addClient(DisplayRefreshMonitorClient&);
58 bool removeClient(DisplayRefreshMonitorClient&);
59
60 void clientPreferredFramesPerSecondChanged(DisplayRefreshMonitorClient&);
61 std::optional<FramesPerSecond> maxClientPreferredFramesPerSecond() const { return m_maxClientPreferredFramesPerSecond; }
62
63 virtual std::optional<FramesPerSecond> displayNominalFramesPerSecond() { return std::nullopt; }
64
65 PlatformDisplayID displayID() const { return m_displayID; }
66
67 static RefPtr<DisplayRefreshMonitor> createDefaultDisplayRefreshMonitor(PlatformDisplayID);
68 WEBCORE_EXPORT virtual void displayLinkFired(const DisplayUpdate&);
69
70protected:
71 WEBCORE_EXPORT explicit DisplayRefreshMonitor(PlatformDisplayID);
72
73 WEBCORE_EXPORT virtual void dispatchDisplayDidRefresh(const DisplayUpdate&);
74
75 Lock& lock() { return m_lock; }
76 void setMaxUnscheduledFireCount(unsigned count) { m_maxUnscheduledFireCount = count; }
77
78 // Returns true if the start was successful.
79 WEBCORE_EXPORT virtual bool startNotificationMechanism() = 0;
80 WEBCORE_EXPORT virtual void stopNotificationMechanism() = 0;
81
82 bool isScheduled() const { return m_scheduled; }
83 void setIsScheduled(bool scheduled) { m_scheduled = scheduled; }
84
85 bool isPreviousFrameDone() const { return m_previousFrameDone; }
86 void setIsPreviousFrameDone(bool done) { m_previousFrameDone = done; }
87
88 WEBCORE_EXPORT void displayDidRefresh(const DisplayUpdate&);
89
90private:
91 bool firedAndReachedMaxUnscheduledFireCount();
92
93 virtual void adjustPreferredFramesPerSecond(FramesPerSecond) { }
94
95 std::optional<FramesPerSecond> maximumClientPreferredFramesPerSecond() const;
96 void computeMaxPreferredFramesPerSecond();
97
98 HashSet<DisplayRefreshMonitorClient*> m_clients;
99 HashSet<DisplayRefreshMonitorClient*>* m_clientsToBeNotified { nullptr };
100
101 PlatformDisplayID m_displayID { 0 };
102 std::optional<FramesPerSecond> m_maxClientPreferredFramesPerSecond;
103
104 Lock m_lock;
105 bool m_scheduled { false };
106 bool m_previousFrameDone { true };
107
108 unsigned m_unscheduledFireCount { 0 };
109 unsigned m_maxUnscheduledFireCount { 0 };
110};
