26#include "ContainerNode.h"

29namespace WebCore {
30
31class DocumentFragment : public ContainerNode {
32 WTF_MAKE_ISO_ALLOCATED(DocumentFragment);
33public:
34 static Ref<DocumentFragment> create(Document&);
35
36 void parseHTML(const String&, Element* contextElement, ParserContentPolicy = AllowScriptingContent);
37 bool parseXML(const String&, Element* contextElement, ParserContentPolicy = AllowScriptingContent);
38
39 bool canContainRangeEndPoint() const final { return true; }
40 virtual bool isTemplateContent() const { return false; }
41
42 void setIsDocumentFragmentForInnerOuterHTML() { setNodeFlag(NodeFlag::IsDocumentFragmentForInnerOuterHTML); }
43
44 // From the NonElementParentNode interface - https://dom.spec.whatwg.org/#interface-nonelementparentnode
45 WEBCORE_EXPORT Element* getElementById(const AtomString&) const;
46
47protected:
48 DocumentFragment(Document&, ConstructionType = CreateContainer);
49 String nodeName() const final;
50
51private:
52 NodeType nodeType() const final;
53 Ref<Node> cloneNodeInternal(Document&, CloningOperation) override;
54 bool childTypeAllowed(NodeType) const override;
55};
56
57} // namespace WebCore
