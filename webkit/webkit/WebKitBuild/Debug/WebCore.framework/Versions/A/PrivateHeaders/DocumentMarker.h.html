36// A range of a node within a document that is "marked", such as the range of a misspelled word.
37// It optionally includes a description that could be displayed in the user interface.
38class DocumentMarker {
39public:
40 enum MarkerType {
41 Spelling = 1 << 0,
42 Grammar = 1 << 1,
43 TextMatch = 1 << 2,
44 // Text has been modified by spell correction, reversion of spell correction or other type of substitution.
45 // On some platforms, this prevents the text from being autocorrected again. On post Snow Leopard Mac OS X,
46 // if a Replacement marker contains non-empty description, a reversion UI will be shown.
47 Replacement = 1 << 3,
48 // Renderer needs to add underline indicating that the text has been modified by spell
49 // correction. Text with Replacement marker doesn't necessarily has CorrectionIndicator
50 // marker. For instance, after some text has been corrected, it will have both Replacement
51 // and CorrectionIndicator. However, if user further modifies such text, we would remove
52 // CorrectionIndicator marker, but retain Replacement marker.
53 CorrectionIndicator = 1 << 4,
54 // Correction suggestion has been offered, but got rejected by user.
55 RejectedCorrection = 1 << 5,
56 // Text has been modified by autocorrection. The description of this marker is the original text before autocorrection.
57 Autocorrected = 1 << 6,
58 // On some platforms, this prevents the text from being spellchecked again.
59 SpellCheckingExemption = 1 << 7,
60 // This marker indicates user has deleted an autocorrection starting at the end of the
61 // range that bears this marker. In some platforms, if the user later inserts the same original
62 // word again at this position, it will not be autocorrected again. The description of this
63 // marker is the original word before autocorrection was applied.
64 DeletedAutocorrection = 1 << 8,
65 // This marker indicates that the range of text spanned by the marker is entered by voice dictation,
66 // and it has alternative text.
67 DictationAlternatives = 1 << 9,
68#if ENABLE(TELEPHONE_NUMBER_DETECTION)
69 TelephoneNumber = 1 << 10,
70#endif
71#if PLATFORM(IOS_FAMILY)
72 // FIXME: iOS should share the same dictation mark system with the other platforms.
73 DictationPhraseWithAlternatives = 1 << 11,
74 DictationResult = 1 << 12,
75#endif
76 // This marker indicates that the user has selected a text candidate.
77 AcceptedCandidate = 1 << 13,
78 // This marker indicates that the user has initiated a drag with this content.
79 DraggedContent = 1 << 14,
80#if ENABLE(PLATFORM_DRIVEN_TEXT_CHECKING)
81 // This marker maintains state for the platform text checker.
82 PlatformTextChecking = 1 << 15,
83#endif
84 };
85
86 static constexpr OptionSet<MarkerType> allMarkers();
87
88 struct DictationData {
89 DictationContext context;
90 String originalText;
91 };
92#if ENABLE(PLATFORM_DRIVEN_TEXT_CHECKING)
93 struct PlatformTextCheckingData {
94 String key;
95 String value;
96 };
97#endif
98
99 using Data = Variant<
100 String
101 , DictationData // DictationAlternatives
102#if PLATFORM(IOS_FAMILY)
103 , Vector<String> // DictationPhraseWithAlternatives
104 , RetainPtr<id> // DictationResult
105#endif
106 , RefPtr<Node> // DraggedContent
107#if ENABLE(PLATFORM_DRIVEN_TEXT_CHECKING)
108 , PlatformTextCheckingData // PlatformTextChecking
109#endif
110 >;
111
112 DocumentMarker(MarkerType, OffsetRange, Data&& = { });
113
114 MarkerType type() const { return m_type; }
115 unsigned startOffset() const { return m_range.start; }
116 unsigned endOffset() const { return m_range.end; }
117
118 const String& description() const;
119
120 const Data& data() const { return m_data; }
121 void clearData() { m_data = String { }; }
122
123 // Offset modifications are done by DocumentMarkerController.
124 // Other classes should not call following setters.
125 void setStartOffset(unsigned offset) { m_range.start = offset; }
126 void setEndOffset(unsigned offset) { m_range.end = offset; }
127 void shiftOffsets(int delta);
128
129private:
130 MarkerType m_type;
131 OffsetRange m_range;
132 Data m_data;
133};
134
135constexpr auto DocumentMarker::allMarkers() -> OptionSet<MarkerType>
136{
137 return {
138 AcceptedCandidate,
139 Autocorrected,
140 CorrectionIndicator,
141 DeletedAutocorrection,
142 DictationAlternatives,
143 DraggedContent,
144 Grammar,
145 RejectedCorrection,
146 Replacement,
147 SpellCheckingExemption,
148 Spelling,
149 TextMatch,
150#if ENABLE(TELEPHONE_NUMBER_DETECTION)
151 TelephoneNumber,
152#endif
153#if PLATFORM(IOS_FAMILY)
154 DictationPhraseWithAlternatives,
155 DictationResult,
156#endif
157#if ENABLE(PLATFORM_DRIVEN_TEXT_CHECKING)
158 PlatformTextChecking
159#endif
160 };
161}
162
163inline DocumentMarker::DocumentMarker(MarkerType type, OffsetRange range, Data&& data)
164 : m_type(type)
165 , m_range(range)
166 , m_data(WTFMove(data))
167{
168}
169
170inline void DocumentMarker::shiftOffsets(int delta)
171{
172 m_range.start += delta;
173 m_range.end += delta;
174}
175
176inline const String& DocumentMarker::description() const
177{
178 return WTF::holds_alternative<String>(m_data) ? WTF::get<String>(m_data) : emptyString();
179}
