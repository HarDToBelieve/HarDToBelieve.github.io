120inline bool DocumentMarkerController::hasMarkers() const
121{
122 ASSERT(m_markers.isEmpty() == !m_possiblyExistingMarkerTypes.containsAny(DocumentMarker::allMarkers()));
123 return !m_markers.isEmpty();
124}
