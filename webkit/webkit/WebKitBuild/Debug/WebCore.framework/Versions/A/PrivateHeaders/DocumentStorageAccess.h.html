34namespace WebCore {
35
36class DeferredPromise;
37class Document;
38class UserGestureIndicator;
39
40enum class StorageAccessWasGranted : bool {
41 No,
42 Yes
43};
44
45enum class StorageAccessPromptWasShown : bool {
46 No,
47 Yes
48};
49
50enum class StorageAccessScope : bool {
51 PerFrame,
52 PerPage
53};
54
55enum class StorageAccessQuickResult : bool {
56 Grant,
57 Reject
58};
59
60struct RequestStorageAccessResult {
61 StorageAccessWasGranted wasGranted;
62 StorageAccessPromptWasShown promptWasShown;
63 StorageAccessScope scope;
64 RegistrableDomain topFrameDomain;
65 RegistrableDomain subFrameDomain;
66
67 template<class Encoder> void encode(Encoder&) const;
68 template<class Decoder> static std::optional<RequestStorageAccessResult> decode(Decoder&);
69};
70
71const unsigned maxNumberOfTimesExplicitlyDeniedStorageAccess = 2;
72
73class DocumentStorageAccess final : public Supplement<Document>, public CanMakeWeakPtr<DocumentStorageAccess> {
74 WTF_MAKE_FAST_ALLOCATED;
75public:
76 explicit DocumentStorageAccess(Document&);
77 ~DocumentStorageAccess();
78
79 static void hasStorageAccess(Document&, Ref<DeferredPromise>&&);
80 static bool hasStorageAccessForDocumentQuirk(Document&);
81
82 static void requestStorageAccess(Document&, Ref<DeferredPromise>&&);
83 static void requestStorageAccessForDocumentQuirk(Document&, CompletionHandler<void(StorageAccessWasGranted)>&&);
84 static void requestStorageAccessForNonDocumentQuirk(Document& hostingDocument, RegistrableDomain&& requestingDomain, CompletionHandler<void(StorageAccessWasGranted)>&&);
85
86private:
87 std::optional<bool> hasStorageAccessQuickCheck();
88 void hasStorageAccess(Ref<DeferredPromise>&&);
89 bool hasStorageAccessQuirk();
90
91 std::optional<StorageAccessQuickResult> requestStorageAccessQuickCheck();
92 void requestStorageAccess(Ref<DeferredPromise>&&);
93 void requestStorageAccessForDocumentQuirk(CompletionHandler<void(StorageAccessWasGranted)>&&);
94 void requestStorageAccessForNonDocumentQuirk(RegistrableDomain&& requestingDomain, CompletionHandler<void(StorageAccessWasGranted)>&&);
95 void requestStorageAccessQuirk(RegistrableDomain&& requestingDomain, CompletionHandler<void(StorageAccessWasGranted)>&&);
96
97 static DocumentStorageAccess* from(Document&);
98 static const char* supplementName();
99 bool hasFrameSpecificStorageAccess() const;
100 void setWasExplicitlyDeniedFrameSpecificStorageAccess() { ++m_numberOfTimesExplicitlyDeniedFrameSpecificStorageAccess; };
101 bool isAllowedToRequestStorageAccess() { return m_numberOfTimesExplicitlyDeniedFrameSpecificStorageAccess < maxNumberOfTimesExplicitlyDeniedStorageAccess; };
102 void enableTemporaryTimeUserGesture();
103 void consumeTemporaryTimeUserGesture();
104
105 std::unique_ptr<UserGestureIndicator> m_temporaryUserGesture;
106
107 Document& m_document;
108
109 uint8_t m_numberOfTimesExplicitlyDeniedFrameSpecificStorageAccess = 0;
110
111 StorageAccessScope m_storageAccessScope = StorageAccessScope::PerPage;
112};
113
114template<class Encoder>
115void RequestStorageAccessResult::encode(Encoder& encoder) const
116{
117 encoder << wasGranted << promptWasShown << scope << topFrameDomain << subFrameDomain;
118}
119
120template<class Decoder>
121std::optional<RequestStorageAccessResult> RequestStorageAccessResult::decode(Decoder& decoder)
122{
123 std::optional<StorageAccessWasGranted> wasGranted;
124 decoder >> wasGranted;
125 if (!wasGranted)
126 return std::nullopt;
127
128 std::optional<StorageAccessPromptWasShown> promptWasShown;
129 decoder >> promptWasShown;
130 if (!promptWasShown)
131 return std::nullopt;
132
133 std::optional<StorageAccessScope> scope;
134 decoder >> scope;
135 if (!scope)
136 return std::nullopt;
137
138 std::optional<RegistrableDomain> topFrameDomain;
139 decoder >> topFrameDomain;
140 if (!topFrameDomain)
141 return std::nullopt;
142
143 std::optional<RegistrableDomain> subFrameDomain;
144 decoder >> subFrameDomain;
145 if (!subFrameDomain)
146 return std::nullopt;
147
148 return { { WTFMove(*wasGranted), WTFMove(*promptWasShown), WTFMove(*scope), WTFMove(*topFrameDomain), WTFMove(*subFrameDomain) } };
149}
