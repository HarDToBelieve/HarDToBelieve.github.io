33namespace WebCore {
34
35// See WebDragDestinationAction and WKDragDestinationAction.
36enum class DragDestinationAction : uint8_t {
37 DHTML = 1,
38 Edit = 2,
39 Load = 4
40};
41
42constexpr OptionSet<DragDestinationAction> anyDragDestinationAction()
43{
44 return OptionSet<DragDestinationAction> { DragDestinationAction::DHTML, DragDestinationAction::Edit, DragDestinationAction::Load };
45}
46
47// See WebDragSourceAction.
48enum class DragSourceAction : uint8_t {
49 DHTML = 1,
50 Image = 2,
51 Link = 4,
52 Selection = 8,
53#if ENABLE(ATTACHMENT_ELEMENT)
54 Attachment = 16,
55#endif
56#if ENABLE(INPUT_TYPE_COLOR)
57 Color = 32,
58#endif
59};
60
61constexpr OptionSet<DragSourceAction> anyDragSourceAction()
62{
63 return OptionSet<DragSourceAction> {
64 DragSourceAction::DHTML,
65 DragSourceAction::Image,
66 DragSourceAction::Link,
67 DragSourceAction::Selection
68#if ENABLE(ATTACHMENT_ELEMENT)
69 , DragSourceAction::Attachment
70#endif
71#if ENABLE(INPUT_TYPE_COLOR)
72 , DragSourceAction::Color
73#endif
74 };
75}
76
77// See NSDragOperation, _UIDragOperation and UIDropOperation.
78enum class DragOperation : uint8_t {
79 Copy = 1,
80 Link = 2,
81 Generic = 4,
82 Private = 8,
83 Move = 16,
84 Delete = 32,
85};
86
87constexpr OptionSet<DragOperation> anyDragOperation()
88{
89 return { DragOperation::Copy, DragOperation::Link, DragOperation::Generic, DragOperation::Private, DragOperation::Move, DragOperation::Delete };
90}

93enum class HasNonDefaultPasteboardData : bool { No, Yes };
94enum class DragHandlingMethod : uint8_t { None, EditPlainText, EditRichText, UploadFile, PageLoad, SetColor, NonDefault };
