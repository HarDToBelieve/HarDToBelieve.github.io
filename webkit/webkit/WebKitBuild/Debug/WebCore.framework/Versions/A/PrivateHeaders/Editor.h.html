112enum class TemporarySelectionOption : uint8_t {
113 RevealSelection = 1 << 0,
114 DoNotSetFocus = 1 << 1,
115
116 // Don't propagate selection changes to the client layer.
117 IgnoreSelectionChanges = 1 << 2,
118
119 // Force the render tree to update selection state. Only respected on iOS.
120 EnableAppearanceUpdates = 1 << 3,
121
122 SmoothScroll = 1 << 4,
123
124 OverrideSmoothScrollFeatureEnablment = 1 << 5,
125};
