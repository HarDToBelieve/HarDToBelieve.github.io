26#pragma once
27
28#include "ElementIterator.h"
29
30namespace WebCore {
31
32template<typename> class ElementChildRange;
33
34// Range for iterating through child elements.
35template <typename ElementType> ElementChildRange<ElementType> childrenOfType(ContainerNode&);
36template <typename ElementType> ElementChildRange<const ElementType> childrenOfType(const ContainerNode&);
37
38template <typename ElementType>
39class ElementChildIterator : public ElementIterator<ElementType> {
40public:
41 ElementChildIterator() = default;
42 ElementChildIterator(const ContainerNode& parent, ElementType* current);
43 ElementChildIterator& operator--();
44 ElementChildIterator& operator++();
45};
46
47template <typename ElementType>
48class ElementChildRange {
49public:
50 ElementChildRange(const ContainerNode& parent);
51
52 ElementChildIterator<ElementType> begin() const;
53 static constexpr std::nullptr_t end() { return nullptr; }
54 ElementChildIterator<ElementType> beginAt(ElementType&) const;
55
56 ElementType* first() const;
57 ElementType* last() const;
58
59private:
60 const ContainerNode& m_parent;
61};
62
63// ElementChildIterator
64
65template <typename ElementType>
66inline ElementChildIterator<ElementType>::ElementChildIterator(const ContainerNode& parent, ElementType* current)
67 : ElementIterator<ElementType>(&parent, current)
68{
69}
70
71template <typename ElementType>
72inline ElementChildIterator<ElementType>& ElementChildIterator<ElementType>::operator--()
73{
74 ElementIterator<ElementType>::traversePreviousSibling();
75 return *this;
76}
77
78template <typename ElementType>
79inline ElementChildIterator<ElementType>& ElementChildIterator<ElementType>::operator++()
80{
81 ElementIterator<ElementType>::traverseNextSibling();
82 return *this;
83}
84
85// ElementChildRange
86
87template <typename ElementType>
88inline ElementChildRange<ElementType>::ElementChildRange(const ContainerNode& parent)
89 : m_parent(parent)
90{
91}
92
93template <typename ElementType>
94inline ElementChildIterator<ElementType> ElementChildRange<ElementType>::begin() const
95{
96 return ElementChildIterator<ElementType>(m_parent, Traversal<ElementType>::firstChild(m_parent));
97}
98
99template <typename ElementType>
100inline ElementType* ElementChildRange<ElementType>::first() const
101{
102 return Traversal<ElementType>::firstChild(m_parent);
103}
104
105template <typename ElementType>
106inline ElementType* ElementChildRange<ElementType>::last() const
107{
108 return Traversal<ElementType>::lastChild(m_parent);
109}
110
111template <typename ElementType>
112inline ElementChildIterator<ElementType> ElementChildRange<ElementType>::beginAt(ElementType& child) const
113{
114 ASSERT(child.parentNode() == &m_parent);
115 return ElementChildIterator<ElementType>(m_parent, &child);
116}
117
118// Standalone functions
119
120template <typename ElementType>
121inline ElementChildRange<ElementType> childrenOfType(ContainerNode& parent)
122{
123 return ElementChildRange<ElementType>(parent);
124}
125
126template <typename ElementType>
127inline ElementChildRange<const ElementType> childrenOfType(const ContainerNode& parent)
128{
129 return ElementChildRange<const ElementType>(parent);
130}
