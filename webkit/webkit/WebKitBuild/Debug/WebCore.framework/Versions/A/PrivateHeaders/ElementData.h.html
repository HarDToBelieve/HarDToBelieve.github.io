40class AttributeConstIterator {
41public:
42 AttributeConstIterator(const Attribute* array, unsigned offset)
43 : m_array(array)
44 , m_offset(offset)
45 {
46 }
47
48 const Attribute& operator*() const { return m_array[m_offset]; }
49 const Attribute* operator->() const { return &m_array[m_offset]; }
50 AttributeConstIterator& operator++() { ++m_offset; return *this; }
51
52 bool operator==(const AttributeConstIterator& other) const { return m_offset == other.m_offset; }
53 bool operator!=(const AttributeConstIterator& other) const { return !(*this == other); }
54
55private:
56 const Attribute* m_array;
57 unsigned m_offset;
58};

72
73private:
74 const Attribute* m_array;

76};

267ALWAYS_INLINE const Attribute* ElementData::findAttributeByName(const AtomString& name, bool shouldIgnoreAttributeCase) const
268{
269 unsigned index = findAttributeIndexByName(name, shouldIgnoreAttributeCase);
270 if (index != attributeNotFound)
271 return &attributeAt(index);
272 return nullptr;
273}
274
275ALWAYS_INLINE unsigned ElementData::findAttributeIndexByName(const QualifiedName& name) const
276{
277 const Attribute* attributes = attributeBase();
278 for (unsigned i = 0, count = length(); i < count; ++i) {
279 if (attributes[i].name().matches(name))
280 return i;
281 }
282 return attributeNotFound;
283}
284
285// We use a boolean parameter instead of calling shouldIgnoreAttributeCase so that the caller
286// can tune the behavior (hasAttribute is case sensitive whereas getAttribute is not).
287ALWAYS_INLINE unsigned ElementData::findAttributeIndexByName(const AtomString& name, bool shouldIgnoreAttributeCase) const
288{
289 unsigned attributeCount = length();
290 if (!attributeCount)
291 return attributeNotFound;
292
293 const Attribute* attributes = attributeBase();
294 const AtomString& caseAdjustedName = shouldIgnoreAttributeCase ? name.convertToASCIILowercase() : name;
295
296 unsigned attributeIndex = 0;
297 do {
298 const Attribute& attribute = attributes[attributeIndex];
299 if (!attribute.name().hasPrefix()) {
300 if (attribute.localName() == caseAdjustedName)
301 return attributeIndex;
302 } else {
303 if (attribute.name().toString() == caseAdjustedName)
304 return attributeIndex;
305 }
306
307 ++attributeIndex;
308 } while (attributeIndex < attributeCount);
309
310 return attributeNotFound;
311}
312
313ALWAYS_INLINE const Attribute* ElementData::findAttributeByName(const QualifiedName& name) const
314{
315 const Attribute* attributes = attributeBase();
316 for (unsigned i = 0, count = length(); i < count; ++i) {
317 if (attributes[i].name().matches(name))
318 return &attributes[i];
319 }
320 return 0;
321}
