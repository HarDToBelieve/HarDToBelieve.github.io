36template <typename ElementType>
37class ElementIterator : public std::iterator<std::forward_iterator_tag, ElementType> {
38public:
39 ElementIterator() = default;
40
41 ElementType& operator*() const;
42 ElementType* operator->() const;
43
44 constexpr operator bool() const;
45 constexpr bool operator!() const;
46 constexpr bool operator!=(std::nullptr_t) const;
47 constexpr bool operator!=(const ElementIterator&) const;
48
49 ElementIterator& traverseNext();
50 ElementIterator& traversePrevious();
51 ElementIterator& traverseNextSibling();
52 ElementIterator& traversePreviousSibling();
53 ElementIterator& traverseNextSkippingChildren();
54 ElementIterator& traverseAncestor();
55
56 void dropAssertions();
57
58protected:
59 ElementIterator(const ContainerNode* root, ElementType* current);
60
61private:
62 const ContainerNode* m_root { nullptr };
63 ElementType* m_current { nullptr };
64
65#if ASSERT_ENABLED
66 ElementIteratorAssertions m_assertions;
67#endif
68};

72template <typename ElementType>
73inline ElementIterator<ElementType>::ElementIterator(const ContainerNode* root, ElementType* current)
74 : m_root(root)
75 , m_current(current)
76#if ASSERT_ENABLED
77 , m_assertions(current)
78#endif
79{
80}

82template <typename ElementType>
83inline ElementIterator<ElementType>& ElementIterator<ElementType>::traverseNext()
84{
85 ASSERT(m_current);
86 ASSERT(!m_assertions.domTreeHasMutated());
87 m_current = Traversal<ElementType>::next(*m_current, m_root);
88#if ASSERT_ENABLED
89 // Drop the assertion when the iterator reaches the end.
90 if (!m_current)
91 m_assertions.dropEventDispatchAssertion();
92#endif
93 return *this;
94}

160template <typename ElementType>
161inline ElementType* findElementAncestorOfType(const Node& current)
162{
163 for (Element* ancestor = current.parentElement(); ancestor; ancestor = ancestor->parentElement()) {
164 if (is<ElementType>(*ancestor))
165 return downcast<ElementType>(ancestor);
166 }
167 return nullptr;
168}
169
170template <>
171inline Element* findElementAncestorOfType<Element>(const Node& current)
172{
173 return current.parentElement();
174}
175
176template <typename ElementType>
177inline ElementIterator<ElementType>& ElementIterator<ElementType>::traverseAncestor()
178{
179 ASSERT(m_current);
180 ASSERT(m_current != m_root);
181 ASSERT(!m_assertions.domTreeHasMutated());
182 m_current = findElementAncestorOfType<ElementType>(*m_current);
183#if ASSERT_ENABLED
184 // Drop the assertion when the iterator reaches the end.
185 if (!m_current)
186 m_assertions.dropEventDispatchAssertion();
187#endif
188 return *this;
189}

191template <typename ElementType>

196 return *m_current;
197}

199template <typename ElementType>
200inline ElementType* ElementIterator<ElementType>::operator->() const
201{
202 ASSERT(m_current);

204 return m_current;

207template<typename ElementType> constexpr ElementIterator<ElementType>::operator bool() const
208{
209 return m_current;
210}
211
212template<typename ElementType> constexpr bool ElementIterator<ElementType>::operator!() const
213{
214 return !m_current;
215}
216
217template<typename ElementType> constexpr bool ElementIterator<ElementType>::operator!=(std::nullptr_t) const
218{
219 return m_current;
220}
221
222template<typename ElementType> constexpr bool ElementIterator<ElementType>::operator!=(const ElementIterator& other) const
223{
224 ASSERT(m_root == other.m_root || !m_current || !other.m_current);
225 return m_current != other.m_current;
226}
