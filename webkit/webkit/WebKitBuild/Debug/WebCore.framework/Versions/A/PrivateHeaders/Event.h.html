41class EventPath;
42class EventTarget;
43class ScriptExecutionContext;
44
45class Event : public ScriptWrappable, public RefCounted<Event> {
46 WTF_MAKE_ISO_ALLOCATED(Event);
47public:
48 enum class IsTrusted : uint8_t { No, Yes };
49 enum class CanBubble : uint8_t { No, Yes };
50 enum class IsCancelable : uint8_t { No, Yes };
51 enum class IsComposed : uint8_t { No, Yes };
52
53 enum PhaseType : uint8_t {
54 NONE = 0,
55 CAPTURING_PHASE = 1,
56 AT_TARGET = 2,
57 BUBBLING_PHASE = 3
58 };
59
60 WEBCORE_EXPORT static Ref<Event> create(const AtomString& type, CanBubble, IsCancelable, IsComposed = IsComposed::No);
61 static Ref<Event> createForBindings();
62 static Ref<Event> create(const AtomString& type, const EventInit&, IsTrusted = IsTrusted::No);
63
64 virtual ~Event();
65
66 WEBCORE_EXPORT void initEvent(const AtomString& type, bool canBubble, bool cancelable);
67
68 bool isInitialized() const { return m_isInitialized; }
69
70 const AtomString& type() const { return m_type; }
71 void setType(const AtomString& type) { m_type = type; }
72
73 EventTarget* target() const { return m_target.get(); }
74 void setTarget(RefPtr<EventTarget>&&);
75
76 EventTarget* currentTarget() const { return m_currentTarget.get(); }
77 void setCurrentTarget(EventTarget*, std::optional<bool> isInShadowTree = std::nullopt);
78 bool currentTargetIsInShadowTree() const { return m_currentTargetIsInShadowTree; }
79
80 unsigned short eventPhase() const { return m_eventPhase; }
81 void setEventPhase(PhaseType phase) { m_eventPhase = phase; }
82
83 bool bubbles() const { return m_canBubble; }
84 bool cancelable() const { return m_cancelable; }
85 bool composed() const { return m_composed; }
86
87 DOMHighResTimeStamp timeStampForBindings(ScriptExecutionContext&) const;
88 MonotonicTime timeStamp() const { return m_createTime; }
89
90 void setEventPath(const EventPath& path) { m_eventPath = &path; }
91 Vector<EventTarget*> composedPath() const;
92
93 void stopPropagation() { m_propagationStopped = true; }
94 void stopImmediatePropagation() { m_immediatePropagationStopped = true; }
95
96 bool isTrusted() const { return m_isTrusted; }
97 void setUntrusted() { m_isTrusted = false; }
98
99 bool legacyReturnValue() const { return !m_wasCanceled; }
100 void setLegacyReturnValue(bool);
101
102 virtual EventInterface eventInterface() const { return EventInterfaceType; }
103
104 virtual bool isBeforeTextInsertedEvent() const { return false; }
105 virtual bool isBeforeUnloadEvent() const { return false; }
106 virtual bool isClipboardEvent() const { return false; }
107 virtual bool isCompositionEvent() const { return false; }
108 virtual bool isErrorEvent() const { return false; }
109 virtual bool isFocusEvent() const { return false; }
110 virtual bool isInputEvent() const { return false; }
111 virtual bool isKeyboardEvent() const { return false; }
112 virtual bool isMouseEvent() const { return false; }
113 virtual bool isPointerEvent() const { return false; }
114 virtual bool isTextEvent() const { return false; }
115 virtual bool isTouchEvent() const { return false; }
116 virtual bool isUIEvent() const { return false; }
117 virtual bool isVersionChangeEvent() const { return false; }
118 virtual bool isWheelEvent() const { return false; }
119
120 bool propagationStopped() const { return m_propagationStopped || m_immediatePropagationStopped; }
121 bool immediatePropagationStopped() const { return m_immediatePropagationStopped; }
122
123 void resetBeforeDispatch();
124 void resetAfterDispatch();
125
126 bool defaultPrevented() const { return m_wasCanceled; }
127 void preventDefault();
128
129 bool defaultHandled() const { return m_defaultHandled; }
130 void setDefaultHandled() { m_defaultHandled = true; }
131
132 bool isDefaultEventHandlerIgnored() const { return m_isDefaultEventHandlerIgnored; }
133 void setIsDefaultEventHandlerIgnored() { m_isDefaultEventHandlerIgnored = true; }
134
135 void setInPassiveListener(bool value) { m_isExecutingPassiveEventListener = value; }
136
137 bool cancelBubble() const { return propagationStopped(); }
138 void setCancelBubble(bool);
139
140 Event* underlyingEvent() const { return m_underlyingEvent.get(); }
141 void setUnderlyingEvent(Event*);
142
143 // Returns true if the dispatch flag is set.
144 // https://dom.spec.whatwg.org/#dispatch-flag
145 bool isBeingDispatched() const { return eventPhase(); }
146
147 virtual EventTarget* relatedTarget() const { return nullptr; }
148 virtual void setRelatedTarget(EventTarget*) { }
149
150 virtual String debugDescription() const;
151
152protected:
153 explicit Event(IsTrusted = IsTrusted::No);
154 Event(const AtomString& type, CanBubble, IsCancelable, IsComposed = IsComposed::No);
155 Event(const AtomString& type, CanBubble, IsCancelable, IsComposed, MonotonicTime timestamp, IsTrusted isTrusted = IsTrusted::Yes);
156 Event(const AtomString& type, const EventInit&, IsTrusted);
157
158 virtual void receivedTarget() { }
159
160private:
161 explicit Event(MonotonicTime createTime, const AtomString& type, IsTrusted, CanBubble, IsCancelable, IsComposed);
162
163 void setCanceledFlagIfPossible();
164
165 unsigned m_isInitialized : 1;
166 unsigned m_canBubble : 1;
167 unsigned m_cancelable : 1;
168 unsigned m_composed : 1;
169
170 unsigned m_propagationStopped : 1;
171 unsigned m_immediatePropagationStopped : 1;
172 unsigned m_wasCanceled : 1;
173 unsigned m_defaultHandled : 1;
174 unsigned m_isDefaultEventHandlerIgnored : 1;
175 unsigned m_isTrusted : 1;
176 unsigned m_isExecutingPassiveEventListener : 1;
177 unsigned m_currentTargetIsInShadowTree : 1;
178
179 unsigned m_eventPhase : 2;
180
181 AtomString m_type;
182
183 RefPtr<EventTarget> m_currentTarget;
184 const EventPath* m_eventPath { nullptr };
185 RefPtr<EventTarget> m_target;
186 MonotonicTime m_createTime;
187
188 RefPtr<Event> m_underlyingEvent;
189};
190
191inline void Event::preventDefault()
192{
193 setCanceledFlagIfPossible();
194}
195
196inline void Event::setLegacyReturnValue(bool returnValue)
197{
198 if (!returnValue)
199 setCanceledFlagIfPossible();
200}
201
202// https://dom.spec.whatwg.org/#set-the-canceled-flag
203inline void Event::setCanceledFlagIfPossible()
204{
205 if (m_cancelable && !m_isExecutingPassiveEventListener)
206 m_wasCanceled = true;
207 // FIXME: Specification suggests we log something to the console when preventDefault is called but
208 // doesn't do anything because the event is not cancelable or is executing passive event listeners.
209}
210
211inline void Event::setCancelBubble(bool cancel)
212{
213 if (cancel)
214 m_propagationStopped = true;
215}

217WTF::TextStream& operator<<(WTF::TextStream&, const Event&);
