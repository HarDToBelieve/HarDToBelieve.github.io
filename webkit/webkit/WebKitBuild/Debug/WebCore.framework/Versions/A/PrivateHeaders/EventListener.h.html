22
23#include <wtf/RefCounted.h>
24#include <wtf/WeakPtr.h>

26namespace JSC {
27class AbstractSlotVisitor;
28class JSObject;
29class SlotVisitor;
30}
31
32namespace WebCore {
33
34class ScriptExecutionContext;
35class Event;
36class EventTarget;
37
38class EventListener : public RefCounted<EventListener>, public CanMakeWeakPtr<EventListener> {
39public:
40 enum Type {
41 JSEventListenerType,
42 ImageEventListenerType,
43 ObjCEventListenerType,
44 CPPEventListenerType,
45 ConditionEventListenerType,
46 GObjectEventListenerType,
47 NativeEventListenerType,
48 SVGTRefTargetEventListenerType,
49 };
50
51 virtual ~EventListener() = default;
52 virtual bool operator==(const EventListener&) const = 0;
53 virtual void handleEvent(ScriptExecutionContext&, Event&) = 0;
54 virtual bool wasCreatedFromMarkup() const { return false; }
55
56 virtual void visitJSFunction(JSC::AbstractSlotVisitor&) { }
57 virtual void visitJSFunction(JSC::SlotVisitor&) { }
58
59 virtual bool isAttribute() const { return false; }
60 Type type() const { return m_type; }
61
62#if ASSERT_ENABLED
63 virtual void checkValidityForEventTarget(EventTarget&) { }
64#endif
65
66 virtual JSC::JSObject* jsFunction() const { return nullptr; }
67 virtual JSC::JSObject* wrapper() const { return nullptr; }
68
69protected:
70 explicit EventListener(Type type)
71 : m_type(type)
72 {
73 }
74
75private:
76 Type m_type;
77};

79} // namespace WebCore
