103class EventLoopTaskGroup : public CanMakeWeakPtr<EventLoopTaskGroup> {
104 WTF_MAKE_NONCOPYABLE(EventLoopTaskGroup);
105 WTF_MAKE_FAST_ALLOCATED;
106
107public:
108 EventLoopTaskGroup(EventLoop& eventLoop)
109 : m_eventLoop(makeWeakPtr(eventLoop))
110 {
111 eventLoop.registerGroup(*this);
112 }
113
114 ~EventLoopTaskGroup()
115 {
116 if (auto* eventLoop = m_eventLoop.get())
117 eventLoop->unregisterGroup(*this);
118 }
119
120 bool hasSameEventLoopAs(EventLoopTaskGroup& otherGroup)
121 {
122 ASSERT(m_eventLoop);
123 return m_eventLoop == otherGroup.m_eventLoop;
124 }
125
126 bool matchesTask(EventLoopTask& task) const
127 {
128 auto* group = task.group();
129 return group == this;
130 }
131
132 // Marks the group as ready to stop but it won't actually be stopped
133 // until all groups in this event loop are ready to stop.
134 void markAsReadyToStop()
135 {
136 if (isReadyToStop() || isStoppedPermanently())
137 return;
138
139 m_state = State::ReadyToStop;
140 if (auto* eventLoop = m_eventLoop.get())
141 eventLoop->stopAssociatedGroupsIfNecessary();
142 }
143
144 // This gets called by the event loop when all groups in the EventLoop as ready to stop.
145 void stopAndDiscardAllTasks()
146 {
147 ASSERT(isReadyToStop());
148 m_state = State::Stopped;
149 if (auto* eventLoop = m_eventLoop.get())
150 eventLoop->stopGroup(*this);
151 }
152
153 void suspend()
154 {
155 ASSERT(!isStoppedPermanently());
156 ASSERT(!isReadyToStop());
157 m_state = State::Suspended;
158 // We don't remove suspended tasks to preserve the ordering.
159 // EventLoop::run checks whether each task's group is suspended or not.
160 }
161
162 void resume()
163 {
164 ASSERT(!isStoppedPermanently());
165 ASSERT(!isReadyToStop());
166 m_state = State::Running;
167 if (auto* eventLoop = m_eventLoop.get())
168 eventLoop->resumeGroup(*this);
169 }
170
171 bool isStoppedPermanently() { return m_state == State::Stopped; }
172 bool isSuspended() { return m_state == State::Suspended; }
173 bool isReadyToStop() const { return m_state == State::ReadyToStop; }
174
175 void queueTask(std::unique_ptr<EventLoopTask>&&);
176 WEBCORE_EXPORT void queueTask(TaskSource, EventLoop::TaskFunction&&);
177
178 // https://html.spec.whatwg.org/multipage/webappapis.html#queue-a-microtask
179 WEBCORE_EXPORT void queueMicrotask(EventLoop::TaskFunction&&);
180 MicrotaskQueue& microtaskQueue() { return m_eventLoop->microtaskQueue(); }
181
182 // https://html.spec.whatwg.org/multipage/webappapis.html#perform-a-microtask-checkpoint
183 void performMicrotaskCheckpoint();
184
185 void runAtEndOfMicrotaskCheckpoint(EventLoop::TaskFunction&&);
186
187private:
188 enum class State : uint8_t { Running, Suspended, ReadyToStop, Stopped };
189
190 WeakPtr<EventLoop> m_eventLoop;
191 State m_state { State::Running };
192};

196 , m_group(makeWeakPtr(group))
197{ }
