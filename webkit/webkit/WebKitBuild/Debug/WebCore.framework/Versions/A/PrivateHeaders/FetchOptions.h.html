38struct FetchOptions {
39 enum class Destination : uint8_t { EmptyString, Audio, Audioworklet, Document, Embed, Font, Image, Manifest, Model, Object, Paintworklet, Report, Script, Serviceworker, Sharedworker, Style, Track, Video, Worker, Xslt };
40 enum class Mode : uint8_t { Navigate, SameOrigin, NoCors, Cors };
41 enum class Credentials : uint8_t { Omit, SameOrigin, Include };
42 enum class Cache : uint8_t { Default, NoStore, Reload, NoCache, ForceCache, OnlyIfCached };
43 enum class Redirect : uint8_t { Follow, Error, Manual };
44
45 FetchOptions() = default;
46 FetchOptions(Destination, Mode, Credentials, Cache, Redirect, ReferrerPolicy, String&&, bool);
47 FetchOptions isolatedCopy() const { return { destination, mode, credentials, cache, redirect, referrerPolicy, integrity.isolatedCopy(), keepAlive }; }
48
49 template<class Encoder> void encodePersistent(Encoder&) const;
50 template<class Decoder> static WARN_UNUSED_RETURN bool decodePersistent(Decoder&, FetchOptions&);
51 template<class Encoder> void encode(Encoder&) const;
52 template<class Decoder> static std::optional<FetchOptions> decode(Decoder&);
53
54 Destination destination { Destination::EmptyString };
55 Mode mode { Mode::NoCors };
56 Credentials credentials { Credentials::Omit };
57 Cache cache { Cache::Default };
58 Redirect redirect { Redirect::Follow };
59 ReferrerPolicy referrerPolicy { ReferrerPolicy::EmptyString };
60 bool keepAlive { false };
61 String integrity;
62 Markable<DocumentIdentifier, DocumentIdentifier::MarkableTraits> clientIdentifier;
63};
64
65inline FetchOptions::FetchOptions(Destination destination, Mode mode, Credentials credentials, Cache cache, Redirect redirect, ReferrerPolicy referrerPolicy, String&& integrity, bool keepAlive)
66 : destination(destination)
67 , mode(mode)

69 , cache(cache)
70 , redirect(redirect)
71 , referrerPolicy(referrerPolicy)
72 , keepAlive(keepAlive)
73 , integrity(WTFMove(integrity))
74{
75}

77inline bool isPotentialNavigationOrSubresourceRequest(FetchOptions::Destination destination)
78{
79 return destination == FetchOptions::Destination::Object
80 || destination == FetchOptions::Destination::Embed;
81}
82
83inline bool isNonSubresourceRequest(FetchOptions::Destination destination)
84{
85 return destination == FetchOptions::Destination::Document
86 || destination == FetchOptions::Destination::Report
87 || destination == FetchOptions::Destination::Serviceworker
88 || destination == FetchOptions::Destination::Sharedworker
89 || destination == FetchOptions::Destination::Worker;
90}
91
92inline bool isScriptLikeDestination(FetchOptions::Destination destination)
93{
94 return destination == FetchOptions::Destination::Audioworklet
95 || destination == FetchOptions::Destination::Paintworklet
96 || destination == FetchOptions::Destination::Script
97 || destination == FetchOptions::Destination::Serviceworker
98 || destination == FetchOptions::Destination::Worker;
99}

175template<class Encoder>
176inline void FetchOptions::encodePersistent(Encoder& encoder) const
177{
178 // Changes to encoding here should bump NetworkCache Storage format version.
179 encoder << destination;
180 encoder << mode;
181 encoder << credentials;
182 encoder << cache;
183 encoder << redirect;
184 encoder << referrerPolicy;
185 encoder << integrity;
186 encoder << keepAlive;
187}
188
189template<class Decoder>
190inline bool FetchOptions::decodePersistent(Decoder& decoder, FetchOptions& options)
191{
192 std::optional<FetchOptions::Destination> destination;
193 decoder >> destination;
194 if (!destination)
195 return false;
196
197 std::optional<FetchOptions::Mode> mode;
198 decoder >> mode;
199 if (!mode)
200 return false;
201
202 std::optional<FetchOptions::Credentials> credentials;
203 decoder >> credentials;
204 if (!credentials)
205 return false;
206
207 std::optional<FetchOptions::Cache> cache;
208 decoder >> cache;
209 if (!cache)
210 return false;
211
212 std::optional<FetchOptions::Redirect> redirect;
213 decoder >> redirect;
214 if (!redirect)
215 return false;
216
217 std::optional<ReferrerPolicy> referrerPolicy;
218 decoder >> referrerPolicy;
219 if (!referrerPolicy)
220 return false;
221
222 std::optional<String> integrity;
223 decoder >> integrity;
224 if (!integrity)
225 return false;
226
227 std::optional<bool> keepAlive;
228 decoder >> keepAlive;
229 if (!keepAlive)
230 return false;
231
232 options.destination = *destination;
233 options.mode = *mode;
234 options.credentials = *credentials;
235 options.cache = *cache;
236 options.redirect = *redirect;
237 options.referrerPolicy = *referrerPolicy;
238 options.integrity = WTFMove(*integrity);
239 options.keepAlive = *keepAlive;
240
241 return true;
242}
243
244template<class Encoder>
245inline void FetchOptions::encode(Encoder& encoder) const
246{
247 encodePersistent(encoder);
248 encoder << clientIdentifier.asOptional();
249}
250
251template<class Decoder>
252inline std::optional<FetchOptions> FetchOptions::decode(Decoder& decoder)
253{
254 FetchOptions options;
255 if (!decodePersistent(decoder, options))
256 return std::nullopt;
257
258 std::optional<std::optional<DocumentIdentifier>> clientIdentifier;
259 decoder >> clientIdentifier;
260 if (!clientIdentifier)
261 return std::nullopt;
262 options.clientIdentifier = WTFMove(clientIdentifier.value());
263
264 return options;
265}
