32
33#include "BlobResourceHandle.h"
34#include "ExceptionCode.h"
35#include <wtf/URL.h>
36#include "TextEncoding.h"
37#include "ThreadableLoaderClient.h"

40#include <wtf/text/WTFString.h>
41
42namespace JSC {
43class ArrayBuffer;
44}
45
46namespace WebCore {
47
48class Blob;
49class FileReaderLoaderClient;
50class ScriptExecutionContext;
51class TextResourceDecoder;
52class ThreadableLoader;
53
54class FileReaderLoader : public ThreadableLoaderClient {
55public:
56 enum ReadType {
57 ReadAsArrayBuffer,
58 ReadAsBinaryString,
59 ReadAsBlob,
60 ReadAsText,
61 ReadAsDataURL
62 };
63
64 // If client is given, do the loading asynchronously. Otherwise, load synchronously.
65 WEBCORE_EXPORT FileReaderLoader(ReadType, FileReaderLoaderClient*);
66 ~FileReaderLoader();
67
68 WEBCORE_EXPORT void start(ScriptExecutionContext*, Blob&);
69 WEBCORE_EXPORT void cancel();
70
71 // ThreadableLoaderClient
72 void didReceiveResponse(unsigned long, const ResourceResponse&) override;
73 void didReceiveData(const uint8_t*, int) override;
74 void didFinishLoading(unsigned long) override;
75 void didFail(const ResourceError&) override;
76
77 String stringResult();
78 WEBCORE_EXPORT RefPtr<JSC::ArrayBuffer> arrayBufferResult() const;
79 unsigned bytesLoaded() const { return m_bytesLoaded; }
80 unsigned totalBytes() const { return m_totalBytes; }
81 std::optional<ExceptionCode> errorCode() const { return m_errorCode; }
82
83 void setEncoding(const String&);
84 void setDataType(const String& dataType) { m_dataType = dataType; }
85
86 const URL& url() { return m_urlForReading; }
87
88 bool isCompleted() const;
89
90private:
91 void terminate();
92 void cleanup();
93 void failed(ExceptionCode);
94 void convertToText();
95 void convertToDataURL();
96
97 static ExceptionCode httpStatusCodeToErrorCode(int);
98 static ExceptionCode toErrorCode(BlobResourceHandle::Error);
99
100 ReadType m_readType;
101 WeakPtr<FileReaderLoaderClient> m_client;
102 TextEncoding m_encoding;
103 String m_dataType;
104
105 URL m_urlForReading;
106 RefPtr<ThreadableLoader> m_loader;
107
108 RefPtr<JSC::ArrayBuffer> m_rawData;
109 bool m_isRawDataConverted;
110
111 String m_stringResult;
112 RefPtr<Blob> m_blobResult;
113
114 // The decoder used to decode the text data.
115 RefPtr<TextResourceDecoder> m_decoder;
116
117 bool m_variableLength;
118 unsigned m_bytesLoaded;
119 unsigned m_totalBytes;
120
121 std::optional<ExceptionCode> m_errorCode;
122};
123
124} // namespace WebCore
