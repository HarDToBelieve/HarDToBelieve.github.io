23#ifndef FloatPoint3D_h
24#define FloatPoint3D_h
25
26#include "FloatPoint.h"
27
28namespace WebCore {
29
30class FloatPoint3D {
31public:
32 FloatPoint3D() = default;
33
34 FloatPoint3D(float x, float y, float z)
35 : m_x(x)
36 , m_y(y)
37 , m_z(z)
38 {
39 }
40
41 FloatPoint3D(const FloatPoint& p)
42 : m_x(p.x())
43 , m_y(p.y())
44 {
45 }
46
47 float x() const { return m_x; }
48 void setX(float x) { m_x = x; }
49
50 float y() const { return m_y; }
51 void setY(float y) { m_y = y; }
52
53 FloatPoint xy() const { return { m_x, m_y }; }
54 void setXY(FloatPoint p)
55 {
56 m_x = p.x();
57 m_y = p.y();
58 }
59
60 float z() const { return m_z; }
61 void setZ(float z) { m_z = z; }
62 void set(float x, float y, float z)
63 {
64 m_x = x;
65 m_y = y;
66 m_z = z;
67 }
68
69 void move(float dx, float dy, float dz = 0)
70 {
71 m_x += dx;
72 m_y += dy;
73 m_z += dz;
74 }
75
76 void scale(float sx, float sy, float sz)
77 {
78 m_x *= sx;
79 m_y *= sy;
80 m_z *= sz;
81 }
82
83 bool isZero() const
84 {
85 return !m_x && !m_y && !m_z;
86 }
87
88 void normalize();
89
90 float dot(const FloatPoint3D& a) const
91 {
92 return m_x * a.x() + m_y * a.y() + m_z * a.z();
93 }
94
95 // Sets this FloatPoint3D to the cross product of the passed two.
96 // It is safe for "this" to be the same as either or both of the
97 // arguments.
98 void cross(const FloatPoint3D& a, const FloatPoint3D& b)
99 {
100 float x = a.y() * b.z() - a.z() * b.y();
101 float y = a.z() * b.x() - a.x() * b.z();
102 float z = a.x() * b.y() - a.y() * b.x();
103 m_x = x;
104 m_y = y;
105 m_z = z;
106 }
107
108 // Convenience function returning "this cross point" as a
109 // stack-allocated result.
110 FloatPoint3D cross(const FloatPoint3D& point) const
111 {
112 FloatPoint3D result;
113 result.cross(*this, point);
114 return result;
115 }
116
117 float lengthSquared() const { return this->dot(*this); }
118 float length() const { return std::hypot(m_x, m_y, m_z); }
119
120 float distanceTo(const FloatPoint3D& a) const;
121
122private:
123 float m_x { 0 };
124 float m_y { 0 };
125 float m_z { 0 };
126};
127
128inline FloatPoint3D& operator +=(FloatPoint3D& a, const FloatPoint3D& b)
129{
130 a.move(b.x(), b.y(), b.z());
131 return a;
132}
133
134inline FloatPoint3D& operator +=(FloatPoint3D& a, const FloatPoint& b)
135{
136 a.move(b.x(), b.y());
137 return a;
138}
139
140inline FloatPoint3D& operator -=(FloatPoint3D& a, const FloatPoint3D& b)
141{
142 a.move(-b.x(), -b.y(), -b.z());
143 return a;
144}
145
146inline FloatPoint3D& operator -=(FloatPoint3D& a, const FloatPoint& b)
147{
148 a.move(-b.x(), -b.y());
149 return a;
150}
151
152inline FloatPoint3D operator+(const FloatPoint3D& a, const FloatPoint3D& b)
153{
154 return FloatPoint3D(a.x() + b.x(), a.y() + b.y(), a.z() + b.z());
155}
156
157inline FloatPoint3D operator+(const FloatPoint3D& a, const FloatPoint& b)
158{
159 return FloatPoint3D(a.x() + b.x(), a.y() + b.y(), a.z());
160}
161
162inline FloatPoint3D operator-(const FloatPoint3D& a, const FloatPoint3D& b)
163{
164 return FloatPoint3D(a.x() - b.x(), a.y() - b.y(), a.z() - b.z());
165}
166
167inline FloatPoint3D operator-(const FloatPoint3D& a, const FloatPoint& b)
168{
169 return FloatPoint3D(a.x() - b.x(), a.y() - b.y(), a.z());
170}
171
172inline bool operator==(const FloatPoint3D& a, const FloatPoint3D& b)
173{
174 return a.x() == b.x() && a.y() == b.y() && a.z() == b.z();
175}
176
177inline bool operator!=(const FloatPoint3D& a, const FloatPoint3D& b)
178{
179 return a.x() != b.x() || a.y() != b.y() || a.z() != b.z();
180}
181
182inline float operator*(const FloatPoint3D& a, const FloatPoint3D& b)
183{
184 // dot product
185 return a.dot(b);
186}
187
188inline FloatPoint3D operator*(float k, const FloatPoint3D& v)
189{
190 return FloatPoint3D(k * v.x(), k * v.y(), k * v.z());
191}
192
193inline FloatPoint3D operator*(const FloatPoint3D& v, float k)
194{
195 return FloatPoint3D(k * v.x(), k * v.y(), k * v.z());
196}
197
198inline float FloatPoint3D::distanceTo(const FloatPoint3D& a) const
199{
200 return (*this - a).length();
201}
