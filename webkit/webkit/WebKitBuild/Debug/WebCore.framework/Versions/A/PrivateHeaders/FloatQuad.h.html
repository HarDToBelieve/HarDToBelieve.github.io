43// A FloatQuad is a collection of 4 points, often representing the result of
44// mapping a rectangle through transforms. When initialized from a rect, the
45// points are in clockwise order from top left.
46class FloatQuad {
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 FloatQuad()
50 {
51 }
52
53 FloatQuad(const FloatPoint& p1, const FloatPoint& p2, const FloatPoint& p3, const FloatPoint& p4)
54 : m_p1(p1)
55 , m_p2(p2)
56 , m_p3(p3)
57 , m_p4(p4)
58 {
59 }
60
61 FloatQuad(const FloatRect& inRect)
62 : m_p1(inRect.location())
63 , m_p2(inRect.maxX(), inRect.y())
64 , m_p3(inRect.maxX(), inRect.maxY())
65 , m_p4(inRect.x(), inRect.maxY())
66 {
67 }
68
69 const FloatPoint& p1() const { return m_p1; }
70 const FloatPoint& p2() const { return m_p2; }
71 const FloatPoint& p3() const { return m_p3; }
72 const FloatPoint& p4() const { return m_p4; }
73
74 void setP1(const FloatPoint& p) { m_p1 = p; }
75 void setP2(const FloatPoint& p) { m_p2 = p; }
76 void setP3(const FloatPoint& p) { m_p3 = p; }
77 void setP4(const FloatPoint& p) { m_p4 = p; }
78
79 WEBCORE_EXPORT bool isEmpty() const;
80
81 // This method will not identify "slanted" empty quads.
82 bool boundingBoxIsEmpty() const { return boundingBox().isEmpty(); }
83
84 // Tests whether this quad can be losslessly represented by a FloatRect,
85 // that is, if two edges are parallel to the x-axis and the other two
86 // are parallel to the y-axis. If this method returns true, the
87 // corresponding FloatRect can be retrieved with boundingBox().
88 WEBCORE_EXPORT bool isRectilinear() const;
89
90 // Tests whether the given point is inside, or on an edge or corner of this quad.
91 WEBCORE_EXPORT bool containsPoint(const FloatPoint&) const;
92
93 // Tests whether the four corners of other are inside, or coincident with the sides of this quad.
94 // Note that this only works for convex quads, but that includes all quads that originate
95 // from transformed rects.
96 WEBCORE_EXPORT bool containsQuad(const FloatQuad&) const;
97
98 // Tests whether any part of the rectangle intersects with this quad.
99 // This only works for convex quads.
100 bool intersectsRect(const FloatRect&) const;
101
102 // Test whether any part of the circle/ellipse intersects with this quad.
103 // Note that these two functions only work for convex quads.
104 bool intersectsCircle(const FloatPoint& center, float radius) const;
105 bool intersectsEllipse(const FloatPoint& center, const FloatSize& radii) const;
106
107 // The center of the quad. If the quad is the result of a affine-transformed rectangle this is the same as the original center transformed.
108 FloatPoint center() const
109 {
110 return FloatPoint((m_p1.x() + m_p2.x() + m_p3.x() + m_p4.x()) / 4.0,
111 (m_p1.y() + m_p2.y() + m_p3.y() + m_p4.y()) / 4.0);
112 }
113
114 WEBCORE_EXPORT FloatRect boundingBox() const;
115 IntRect enclosingBoundingBox() const
116 {
117 return enclosingIntRect(boundingBox());
118 }
119
120 void move(const FloatSize& offset)
121 {
122 m_p1 += offset;
123 m_p2 += offset;
124 m_p3 += offset;
125 m_p4 += offset;
126 }
127
128 void move(float dx, float dy)
129 {
130 m_p1.move(dx, dy);
131 m_p2.move(dx, dy);
132 m_p3.move(dx, dy);
133 m_p4.move(dx, dy);
134 }
135
136 void scale(float s)
137 {
138 scale(s, s);
139 }
140
141 void scale(float dx, float dy)
142 {
143 m_p1.scale(dx, dy);
144 m_p2.scale(dx, dy);
145 m_p3.scale(dx, dy);
146 m_p4.scale(dx, dy);
147 }
148
149 // Tests whether points are in clock-wise, or counter clock-wise order.
150 // Note that output is undefined when all points are colinear.
151 bool isCounterclockwise() const;
152
153 template<class Encoder> void encode(Encoder&) const;
154 template<class Decoder> static std::optional<FloatQuad> decode(Decoder&);
155
156private:
157 FloatPoint m_p1;
158 FloatPoint m_p2;
159 FloatPoint m_p3;
160 FloatPoint m_p4;
161};
162
163template<class Encoder> void FloatQuad::encode(Encoder& encoder) const
164{
165 encoder << m_p1;
166 encoder << m_p2;
167 encoder << m_p3;
168 encoder << m_p4;
169}
170
171template<class Decoder> std::optional<FloatQuad> FloatQuad::decode(Decoder& decoder)
172{
173 std::optional<FloatPoint> p1;
174 decoder >> p1;
175 if (!p1)
176 return std::nullopt;
177
178 std::optional<FloatPoint> p2;
179 decoder >> p2;
180 if (!p2)
181 return std::nullopt;
182
183 std::optional<FloatPoint> p3;
184 decoder >> p3;
185 if (!p3)
186 return std::nullopt;
187
188 std::optional<FloatPoint> p4;
189 decoder >> p4;
190 if (!p4)
191 return std::nullopt;
192
193 return {{ *p1, *p2, *p3, *p4 }};
194}
195
196inline FloatQuad& operator+=(FloatQuad& a, const FloatSize& b)
197{
198 a.move(b);
199 return a;
200}
201
202inline FloatQuad& operator-=(FloatQuad& a, const FloatSize& b)
203{
204 a.move(-b.width(), -b.height());
205 return a;
206}
207
208inline bool operator==(const FloatQuad& a, const FloatQuad& b)
209{
210 return a.p1() == b.p1() && a.p2() == b.p2() && a.p3() == b.p3() && a.p4() == b.p4();
211}
212
213inline bool operator!=(const FloatQuad& a, const FloatQuad& b)
214{
215 return !(a == b);
216}
