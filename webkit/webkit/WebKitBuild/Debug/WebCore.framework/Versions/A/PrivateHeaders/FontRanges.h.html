26#ifndef FontRanges_h
27#define FontRanges_h
28
29#include "Font.h"
30#include <wtf/Vector.h>
31
32namespace WebCore {
33
34class FontAccessor;
35
36enum class ExternalResourceDownloadPolicy {
37 Forbid,
38 Allow
39};
40
41class FontRanges {
42public:
43 struct Range {
44 Range(UChar32 from, UChar32 to, Ref<FontAccessor>&& fontAccessor)
45 : m_from(from)
46 , m_to(to)
47 , m_fontAccessor(WTFMove(fontAccessor))
48 {
49 }
50
51 Range(const Range& range)
52 : m_from(range.m_from)
53 , m_to(range.m_to)
54 , m_fontAccessor(range.m_fontAccessor.copyRef())
55 {
56 }
57
58 Range(Range&&) = default;
59 Range& operator=(const Range&) = delete;
60 Range& operator=(Range&&) = default;
61
62 UChar32 from() const { return m_from; }
63 UChar32 to() const { return m_to; }
64 const Font* font(ExternalResourceDownloadPolicy) const;
65 const FontAccessor& fontAccessor() const { return m_fontAccessor; }
66
67 private:
68 UChar32 m_from;
69 UChar32 m_to;
70 Ref<FontAccessor> m_fontAccessor;
71 };
72
73 FontRanges();
74 explicit FontRanges(RefPtr<Font>&&);
75 ~FontRanges();
76
77 FontRanges(const FontRanges&) = default;
78 FontRanges& operator=(FontRanges&&) = default;
79
80 bool isNull() const { return m_ranges.isEmpty(); }
81
82 void appendRange(Range&& range) { m_ranges.append(WTFMove(range)); }
83 unsigned size() const { return m_ranges.size(); }
84 const Range& rangeAt(unsigned i) const { return m_ranges[i]; }
85
86 void shrinkToFit() { m_ranges.shrinkToFit(); }
87
88 WEBCORE_EXPORT GlyphData glyphDataForCharacter(UChar32, ExternalResourceDownloadPolicy) const;
89 WEBCORE_EXPORT const Font* fontForCharacter(UChar32) const;
90 WEBCORE_EXPORT const Font& fontForFirstRange() const;
91 bool isLoading() const;
92
93private:
94 Vector<Range, 1> m_ranges;
95};
96
97}

99#endif
