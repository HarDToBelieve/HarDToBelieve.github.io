34namespace WebCore {
35
36class Document;
37class Frame;
38class SecurityOrigin;
39
40class FrameLoadRequest {
41public:
42 WEBCORE_EXPORT FrameLoadRequest(Document&, SecurityOrigin&, ResourceRequest&&, const String& frameName, InitiatedByMainFrame, const AtomString& downloadAttribute = { }, const SystemPreviewInfo& = { });
43 WEBCORE_EXPORT FrameLoadRequest(Frame&, const ResourceRequest&, const SubstituteData& = SubstituteData());
44
45 WEBCORE_EXPORT ~FrameLoadRequest();
46
47 WEBCORE_EXPORT FrameLoadRequest(FrameLoadRequest&&);
48 WEBCORE_EXPORT FrameLoadRequest& operator=(FrameLoadRequest&&);
49
50 bool isEmpty() const { return m_resourceRequest.isEmpty(); }
51
52 Document& requester();
53 const SecurityOrigin& requesterSecurityOrigin() const;
54
55 ResourceRequest& resourceRequest() { return m_resourceRequest; }
56 const ResourceRequest& resourceRequest() const { return m_resourceRequest; }
57
58 const String& frameName() const { return m_frameName; }
59 void setFrameName(const String& frameName) { m_frameName = frameName; }
60
61 void setShouldCheckNewWindowPolicy(bool checkPolicy) { m_shouldCheckNewWindowPolicy = checkPolicy; }
62 bool shouldCheckNewWindowPolicy() const { return m_shouldCheckNewWindowPolicy; }
63
64 void setShouldTreatAsContinuingLoad(bool value) { m_shouldTreatAsContinuingLoad = value; }
65 bool shouldTreatAsContinuingLoad() const { return m_shouldTreatAsContinuingLoad; }
66
67 const SubstituteData& substituteData() const { return m_substituteData; }
68 void setSubstituteData(const SubstituteData& data) { m_substituteData = data; }
69 bool hasSubstituteData() { return m_substituteData.isValid(); }
70
71 LockHistory lockHistory() const { return m_lockHistory; }
72 void setLockHistory(LockHistory value) { m_lockHistory = value; }
73
74 LockBackForwardList lockBackForwardList() const { return m_lockBackForwardList; }
75 void setLockBackForwardList(LockBackForwardList value) { m_lockBackForwardList = value; }
76
77 const String& clientRedirectSourceForHistory() const { return m_clientRedirectSourceForHistory; }
78 void setClientRedirectSourceForHistory(const String& clientRedirectSourceForHistory) { m_clientRedirectSourceForHistory = clientRedirectSourceForHistory; }
79
80 ReferrerPolicy referrerPolicy() const { return m_referrerPolicy; }
81 void setReferrerPolicy(const ReferrerPolicy& referrerPolicy) { m_referrerPolicy = referrerPolicy; }
82
83 AllowNavigationToInvalidURL allowNavigationToInvalidURL() const { return m_allowNavigationToInvalidURL; }
84 void disableNavigationToInvalidURL() { m_allowNavigationToInvalidURL = AllowNavigationToInvalidURL::No; }
85
86 NewFrameOpenerPolicy newFrameOpenerPolicy() const { return m_newFrameOpenerPolicy; }
87 void setNewFrameOpenerPolicy(NewFrameOpenerPolicy newFrameOpenerPolicy) { m_newFrameOpenerPolicy = newFrameOpenerPolicy; }
88
89 // The shouldReplaceDocumentIfJavaScriptURL parameter will go away when the FIXME to eliminate the
90 // corresponding parameter from ScriptController::executeIfJavaScriptURL() is addressed.
91 ShouldReplaceDocumentIfJavaScriptURL shouldReplaceDocumentIfJavaScriptURL() const { return m_shouldReplaceDocumentIfJavaScriptURL; }
92 void disableShouldReplaceDocumentIfJavaScriptURL() { m_shouldReplaceDocumentIfJavaScriptURL = DoNotReplaceDocumentIfJavaScriptURL; }
93
94 void setShouldOpenExternalURLsPolicy(ShouldOpenExternalURLsPolicy policy) { m_shouldOpenExternalURLsPolicy = policy; }
95 ShouldOpenExternalURLsPolicy shouldOpenExternalURLsPolicy() const { return m_shouldOpenExternalURLsPolicy; }
96
97 const AtomString& downloadAttribute() const { return m_downloadAttribute; }
98
99 InitiatedByMainFrame initiatedByMainFrame() const { return m_initiatedByMainFrame; }
100
101 bool isSystemPreview() const { return m_systemPreviewInfo.isPreview; }
102 const SystemPreviewInfo& systemPreviewInfo() const { return m_systemPreviewInfo; }
103
104 void setIsRequestFromClientOrUserInput() { m_isRequestFromClientOrUserInput = true; }
105 bool isRequestFromClientOrUserInput() const { return m_isRequestFromClientOrUserInput; }
106
107private:
108 Ref<Document> m_requester;
109 Ref<SecurityOrigin> m_requesterSecurityOrigin;
110 ResourceRequest m_resourceRequest;
111 String m_frameName;
112 SubstituteData m_substituteData;
113 String m_clientRedirectSourceForHistory;
114
115 bool m_shouldCheckNewWindowPolicy { false };
116 bool m_shouldTreatAsContinuingLoad { false };
117 LockHistory m_lockHistory { LockHistory::No };
118 LockBackForwardList m_lockBackForwardList { LockBackForwardList::No };
119 ReferrerPolicy m_referrerPolicy { ReferrerPolicy::EmptyString };
120 AllowNavigationToInvalidURL m_allowNavigationToInvalidURL { AllowNavigationToInvalidURL::Yes };
121 NewFrameOpenerPolicy m_newFrameOpenerPolicy { NewFrameOpenerPolicy::Allow };
122 ShouldReplaceDocumentIfJavaScriptURL m_shouldReplaceDocumentIfJavaScriptURL { ReplaceDocumentIfJavaScriptURL };
123 ShouldOpenExternalURLsPolicy m_shouldOpenExternalURLsPolicy { ShouldOpenExternalURLsPolicy::ShouldNotAllow };
124 AtomString m_downloadAttribute;
125 InitiatedByMainFrame m_initiatedByMainFrame { InitiatedByMainFrame::Unknown };
126 SystemPreviewInfo m_systemPreviewInfo;
127 bool m_isRequestFromClientOrUserInput { false };
128};
