20#pragma once
21
22#include "Document.h"
23#include "Frame.h"
24#include "NetworkingContext.h"
25#include "ReferrerPolicy.h"
26
27namespace WebCore {
28
29class FrameNetworkingContext : public NetworkingContext {
30public:
31 void invalidate()
32 {
33 m_frame = nullptr;
34 }
35
36 bool shouldClearReferrerOnHTTPSToHTTPRedirect() const override
37 {
38 // FIXME: PingLoader finishes without a frame, but it should use its document's referrer policy.
39 if (!m_frame)
40 return true;
41
42 return m_frame->document()->referrerPolicy() == ReferrerPolicy::NoReferrerWhenDowngrade;
43 }
44
45protected:
46 explicit FrameNetworkingContext(Frame* frame)
47 : m_frame(makeWeakPtr(frame))
48 {
49 }
50
51 Frame* frame() const { return m_frame.get(); }
52
53private:
54 bool isValid() const override { return !!m_frame; }
55
56 WeakPtr<Frame> m_frame;
57};
58
59}
