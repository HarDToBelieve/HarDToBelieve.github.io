27
28#if ENABLE(FULLSCREEN_API)
29
30#include "Document.h"
31#include "LayoutRect.h"
32#include <wtf/Deque.h>
33#include <wtf/WeakPtr.h>
34
35namespace WebCore {
36
37class RenderFullScreen;
38class RenderTreeBuilder;
39class RenderStyle;
40
41class FullscreenManager final : public CanMakeWeakPtr<FullscreenManager> {
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 FullscreenManager(Document&);
45 ~FullscreenManager();
46
47 Document& document() { return m_document; }
48 const Document& document() const { return m_document; }
49 Document& topDocument() const { return m_document.topDocument(); }
50 Page* page() const { return m_document.page(); }
51 Frame* frame() const { return m_document.frame(); }
52 Element* documentElement() const { return m_document.documentElement(); }
53 bool hasLivingRenderTree() const { return m_document.hasLivingRenderTree(); }
54 Document::BackForwardCacheState backForwardCacheState() const { return m_document.backForwardCacheState(); }
55 void scheduleFullStyleRebuild() { m_document.scheduleFullStyleRebuild(); }
56
57 // W3C Fullscreen API
58 Element* fullscreenElement() const { return !m_fullscreenElementStack.isEmpty() ? m_fullscreenElementStack.last().get() : nullptr; }
59 WEBCORE_EXPORT bool isFullscreenEnabled() const;
60 WEBCORE_EXPORT void exitFullscreen();
61
62 // Mozilla versions.
63 bool isFullscreen() const { return m_fullscreenElement.get(); }
64 bool isFullscreenKeyboardInputAllowed() const { return m_fullscreenElement.get() && m_areKeysEnabledInFullscreen; }
65 Element* currentFullscreenElement() const { return m_fullscreenElement.get(); }
66 WEBCORE_EXPORT void cancelFullscreen();
67
68 enum FullscreenCheckType {
69 EnforceIFrameAllowFullscreenRequirement,
70 ExemptIFrameAllowFullscreenRequirement,
71 };
72 WEBCORE_EXPORT void requestFullscreenForElement(Element*, FullscreenCheckType);
73
74 WEBCORE_EXPORT bool willEnterFullscreen(Element&);
75 WEBCORE_EXPORT bool didEnterFullscreen();
76 WEBCORE_EXPORT bool willExitFullscreen();
77 WEBCORE_EXPORT bool didExitFullscreen();
78
79 void setFullscreenRenderer(RenderTreeBuilder&, RenderFullScreen&);
80 RenderFullScreen* fullscreenRenderer() const;
81
82 void dispatchFullscreenChangeEvents();
83 void fullscreenElementRemoved();
84
85 void adjustFullscreenElementOnNodeRemoval(Node&, Document::NodeRemoval = Document::NodeRemoval::Node);
86
87 WEBCORE_EXPORT bool isAnimatingFullscreen() const;
88 WEBCORE_EXPORT void setAnimatingFullscreen(bool);
89
90 WEBCORE_EXPORT bool areFullscreenControlsHidden() const;
91 WEBCORE_EXPORT void setFullscreenControlsHidden(bool);
92
93 void clear();
94 void emptyEventQueue();
95
96protected:
97 friend class Document;
98
99 void dispatchFullscreenChangeOrErrorEvent(Deque<RefPtr<Node>>&, const AtomString& eventName, bool shouldNotifyMediaElement);
100 void clearFullscreenElementStack();
101 void popFullscreenElementStack();
102 void pushFullscreenElementStack(Element&);
103 void addDocumentToFullscreenChangeEventQueue(Document&);
104
105private:
106#if !RELEASE_LOG_DISABLED
107 const WTF::Logger& logger() const { return m_document.logger(); }
108 const void* logIdentifier() const { return m_logIdentifier; }
109 const char* logClassName() const { return "FullscreenManager"; }
110 WTFLogChannel& logChannel() const;
111#endif
112
113 Document& m_document;
114
115 RefPtr<Element> fullscreenOrPendingElement() const { return m_fullscreenElement ? m_fullscreenElement : m_pendingFullscreenElement; }
116
117 bool m_pendingExitFullscreen { false };
118 RefPtr<Element> m_pendingFullscreenElement;
119 RefPtr<Element> m_fullscreenElement;
120 Vector<RefPtr<Element>> m_fullscreenElementStack;
121 WeakPtr<RenderFullScreen> m_fullscreenRenderer { nullptr };
122 Deque<RefPtr<Node>> m_fullscreenChangeEventTargetQueue;
123 Deque<RefPtr<Node>> m_fullscreenErrorEventTargetQueue;
124 LayoutRect m_savedPlaceholderFrameRect;
125 std::unique_ptr<RenderStyle> m_savedPlaceholderRenderStyle;
126
127 bool m_areKeysEnabledInFullscreen { false };
128 bool m_isAnimatingFullscreen { false };
129 bool m_areFullscreenControlsHidden { false };
130
131#if !RELEASE_LOG_DISABLED
132 const void* m_logIdentifier;
133#endif
134};
135
136}

138#endif
