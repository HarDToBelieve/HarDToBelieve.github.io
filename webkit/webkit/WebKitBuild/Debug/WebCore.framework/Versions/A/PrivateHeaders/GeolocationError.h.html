26#pragma once
27
28#include <wtf/RefCounted.h>
29#include <wtf/text/WTFString.h>
30
31namespace WebCore {
32
33class GeolocationError : public RefCounted<GeolocationError> {
34public:
35 enum ErrorCode {
36 PermissionDenied,
37 PositionUnavailable
38 };
39
40 static Ref<GeolocationError> create(ErrorCode code, const String& message) { return adoptRef(*new GeolocationError(code, message)); }
41
42 ErrorCode code() const { return m_code; }
43 const String& message() const { return m_message; }
44
45private:
46 GeolocationError(ErrorCode code, const String& message)
47 : m_code(code)
48 , m_message(message)
49 {
50 }
51
52 ErrorCode m_code;
53 String m_message;
54};
55
56} // namespace WebCore
