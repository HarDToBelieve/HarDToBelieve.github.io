69 const Font& fontAt(unsigned index) const
70 {
71 ASSERT(m_fonts[index]);
72 return *m_fonts[index];
73 }
74 Glyph glyphAt(unsigned index) const { return m_glyphs[index]; }
75 GlyphBufferAdvance advanceAt(unsigned index) const { return m_advances[index]; }
76 GlyphBufferOrigin originAt(unsigned index) const { return m_origins[index]; }
77 GlyphBufferStringOffset uncheckedStringOffsetAt(unsigned index) const { return m_offsetsInString[index]; }
78 std::optional<GlyphBufferStringOffset> checkedStringOffsetAt(unsigned index, unsigned stringLength) const
79 {
80 auto result = uncheckedStringOffsetAt(index);
81 if (static_cast<std::make_unsigned_t<GlyphBufferStringOffset>>(result) >= stringLength)
82 return std::nullopt;
83 return result;
84 }

86 void setInitialAdvance(GlyphBufferAdvance initialAdvance) { m_initialAdvance = initialAdvance; }
87 const GlyphBufferAdvance& initialAdvance() const { return m_initialAdvance; }
88 void expandInitialAdvance(float width) { setWidth(m_initialAdvance, WebCore::width(m_initialAdvance) + width); }
89 void expandInitialAdvance(GlyphBufferAdvance additionalAdvance)
90 {
91 setWidth(m_initialAdvance, width(m_initialAdvance) + width(additionalAdvance));
92 setHeight(m_initialAdvance, height(m_initialAdvance) + height(additionalAdvance));
93 }

95 static constexpr GlyphBufferStringOffset noOffset = std::numeric_limits<GlyphBufferStringOffset>::max();
96 void add(Glyph glyph, const Font& font, float width, GlyphBufferStringOffset offsetInString = noOffset)
97 {
98 GlyphBufferAdvance advance = makeGlyphBufferAdvance(width, 0);
99 add(glyph, font, advance, offsetInString);
100 }

102 void add(Glyph glyph, const Font& font, GlyphBufferAdvance advance, GlyphBufferStringOffset offsetInString)
103 {
104 m_fonts.append(&font);
105 m_glyphs.append(glyph);
106 m_advances.append(advance);
107 m_origins.append(makeGlyphBufferOrigin());
108 m_offsetsInString.append(offsetInString);
109 }

111 void remove(unsigned location, unsigned length)
112 {
113 m_fonts.remove(location, length);
114 m_glyphs.remove(location, length);
115 m_advances.remove(location, length);
116 m_origins.remove(location, length);
117 m_offsetsInString.remove(location, length);
118 }

120 void makeHole(unsigned location, unsigned length, const Font* font)
121 {
122 ASSERT(location <= size());
123
124 m_fonts.insertVector(location, Vector<const Font*>(length, font));
125 m_glyphs.insertVector(location, Vector<GlyphBufferGlyph>(length, std::numeric_limits<GlyphBufferGlyph>::max()));
126 m_advances.insertVector(location, Vector<GlyphBufferAdvance>(length, makeGlyphBufferAdvance()));
127 m_origins.insertVector(location, Vector<GlyphBufferOrigin>(length, makeGlyphBufferOrigin()));
128 m_offsetsInString.insertVector(location, Vector<GlyphBufferStringOffset>(length, 0));
129 }
130
131 void reverse(unsigned from, unsigned length)
132 {
133 for (unsigned i = from, end = from + length - 1; i < end; ++i, --end)
134 swap(i, end);
135 }
136
137 void expandLastAdvance(float width)
138 {
139 ASSERT(!isEmpty());
140 GlyphBufferAdvance& lastAdvance = m_advances.last();
141 setWidth(lastAdvance, WebCore::width(lastAdvance) + width);
142 }
143
144 void expandAdvance(unsigned index, float width)
145 {
146 ASSERT(index < size());
147 auto& lastAdvance = m_advances[index];
148 setWidth(lastAdvance, WebCore::width(lastAdvance) + width);
149 }
150
151 void expandLastAdvance(GlyphBufferAdvance expansion)
152 {
153 ASSERT(!isEmpty());
154 GlyphBufferAdvance& lastAdvance = m_advances.last();
155 setWidth(lastAdvance, width(lastAdvance) + width(expansion));
156 setHeight(lastAdvance, height(lastAdvance) + height(expansion));
157 }

159 void shrink(unsigned truncationPoint)
160 {
161 m_fonts.shrink(truncationPoint);
162 m_glyphs.shrink(truncationPoint);
163 m_advances.shrink(truncationPoint);
164 m_origins.shrink(truncationPoint);
165 m_offsetsInString.shrink(truncationPoint);
166 }

168 // FontCascade::layoutText() returns a GlyphBuffer which includes layout information that is split
169 // into "advances" and "origins". See the ASCII-art diagram in ComplexTextController.h
170 // In order to get paint advances, we need to run this "flatten" operation.
171 // This merges the layout advances and origins together,
172 // leaves the paint advances in the "m_advances" field,
173 // and zeros-out the origins in the "m_origins" field.
174 void flatten()
175 {
176 for (unsigned i = 0; i < size(); ++i) {
177 m_advances[i] = makeGlyphBufferAdvance(
178 -x(m_origins[i]) + width(m_advances[i]) + (i + 1 < size() ? x(m_origins[i + 1]) : 0),
179 -y(m_origins[i]) + height(m_advances[i]) + (i + 1 < size() ? y(m_origins[i + 1]) : 0)
180 );
181 m_origins[i] = makeGlyphBufferOrigin();
182 }
183 }
184
185 bool isFlattened() const
186 {
187 for (unsigned i = 0; i < size(); ++i) {
188 if (m_origins[i] != makeGlyphBufferOrigin())
189 return false;
190 }
191 return true;
192 }
193
194private:
195 void swap(unsigned index1, unsigned index2)
196 {
197 auto font = m_fonts[index1];
198 m_fonts[index1] = m_fonts[index2];
199 m_fonts[index2] = font;
200
201 auto glyph = m_glyphs[index1];
202 m_glyphs[index1] = m_glyphs[index2];
203 m_glyphs[index2] = glyph;
204
205 auto advance = m_advances[index1];
206 m_advances[index1] = m_advances[index2];
207 m_advances[index2] = advance;
208
209 auto origin = m_origins[index1];
210 m_origins[index1] = m_origins[index2];
211 m_origins[index2] = origin;
212
213 auto offset = m_offsetsInString[index1];
214 m_offsetsInString[index1] = m_offsetsInString[index2];
215 m_offsetsInString[index2] = offset;
216 }
217
218 Vector<const Font*, 1024> m_fonts;
219 Vector<GlyphBufferGlyph, 1024> m_glyphs;
220 Vector<GlyphBufferAdvance, 1024> m_advances;
221 Vector<GlyphBufferOrigin, 1024> m_origins;
222 Vector<GlyphBufferStringOffset, 1024> m_offsetsInString;
223 GlyphBufferAdvance m_initialAdvance { makeGlyphBufferAdvance() };
224};
