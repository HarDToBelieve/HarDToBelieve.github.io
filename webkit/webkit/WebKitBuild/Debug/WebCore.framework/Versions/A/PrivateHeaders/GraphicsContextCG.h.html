35class CGContextStateSaver {
36public:
37 CGContextStateSaver(CGContextRef context, bool saveAndRestore = true)
38 : m_context(context)
39 , m_saveAndRestore(saveAndRestore)
40 {
41 if (m_saveAndRestore)
42 CGContextSaveGState(m_context);
43 }
44
45 ~CGContextStateSaver()
46 {
47 if (m_saveAndRestore)
48 CGContextRestoreGState(m_context);
49 }
50
51 void save()
52 {
53 ASSERT(!m_saveAndRestore);
54 CGContextSaveGState(m_context);
55 m_saveAndRestore = true;
56 }
57
58 void restore()
59 {
60 ASSERT(m_saveAndRestore);
61 CGContextRestoreGState(m_context);
62 m_saveAndRestore = false;
63 }
64
65 bool didSave() const
66 {
67 return m_saveAndRestore;
68 }
69
70private:
71 CGContextRef m_context;
72 bool m_saveAndRestore;
73};
