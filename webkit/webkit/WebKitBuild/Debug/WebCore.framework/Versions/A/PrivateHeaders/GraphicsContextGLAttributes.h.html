45struct GraphicsContextGLAttributes {
46 // WebGLContextAttributes
47 bool alpha { true };
48 bool depth { true };
49 bool stencil { false };
50 bool antialias { true };
51 bool premultipliedAlpha { true };
52 bool preserveDrawingBuffer { false };
53 bool failIfMajorPerformanceCaveat { false };

56
57 // Additional attributes.
58 bool shareResources { true };
59 bool noExtensions { false };
60 float devicePixelRatio { 1 };
61 PowerPreference initialPowerPreference { PowerPreference::Default };
62 using WebGLVersion = GraphicsContextGLWebGLVersion;

64#if PLATFORM(COCOA)
65 bool useMetal { false };
66#endif
67#if ENABLE(WEBXR)
68 bool xrCompatible { false };
69#endif
70};

72}
