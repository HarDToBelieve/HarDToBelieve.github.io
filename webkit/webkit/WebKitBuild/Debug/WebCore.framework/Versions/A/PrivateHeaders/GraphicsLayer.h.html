71// Base class for animation values (also used for transitions). Here to
72// represent values for properties being animated via the GraphicsLayer,
73// without pulling in style-related data from outside of the platform directory.
74// FIXME: Should be moved to its own header file.
75class AnimationValue {
76 WTF_MAKE_FAST_ALLOCATED;
77public:
78 virtual ~AnimationValue() = default;
79
80 double keyTime() const { return m_keyTime; }
81 const TimingFunction* timingFunction() const { return m_timingFunction.get(); }
82 virtual std::unique_ptr<AnimationValue> clone() const = 0;
83
84protected:
85 AnimationValue(double keyTime, TimingFunction* timingFunction = nullptr)
86 : m_keyTime(keyTime)
87 , m_timingFunction(timingFunction)
88 {
89 }
90
91 AnimationValue(const AnimationValue& other)
92 : m_keyTime(other.m_keyTime)
93 , m_timingFunction(other.m_timingFunction ? RefPtr<TimingFunction> { other.m_timingFunction->clone() } : nullptr)
94 {
95 }
96
97 AnimationValue(AnimationValue&&) = default;
98
99private:
100 void operator=(const AnimationValue&) = delete;
101
102 double m_keyTime;
103 RefPtr<TimingFunction> m_timingFunction;
104};

127// Used to store one transform value in a keyframe list.
128// FIXME: Should be moved to its own header file.
129class TransformAnimationValue : public AnimationValue {
130public:
131 TransformAnimationValue(double keyTime, const TransformOperations& value, TimingFunction* timingFunction = nullptr)
132 : AnimationValue(keyTime, timingFunction)
133 , m_value(value)
134 {
135 }
136
137 TransformAnimationValue(double keyTime, TransformOperation* value, TimingFunction* timingFunction = nullptr)
138 : AnimationValue(keyTime, timingFunction)
139 {
140 if (value)
141 m_value.operations().append(value);
142 }
143
144 std::unique_ptr<AnimationValue> clone() const override
145 {
146 return makeUnique<TransformAnimationValue>(*this);
147 }
148
149 TransformAnimationValue(const TransformAnimationValue& other)
150 : AnimationValue(other)
151 {
152 m_value.operations().reserveInitialCapacity(other.m_value.operations().size());
153 for (auto& operation : other.m_value.operations())
154 m_value.operations().uncheckedAppend(operation->clone());
155 }
156
157 TransformAnimationValue(TransformAnimationValue&&) = default;
158
159 const TransformOperations& value() const { return m_value; }
160
161private:
162 TransformOperations m_value;
163};

196// Used to store a series of values in a keyframe list.
197// Values will all be of the same type, which can be inferred from the property.
198// FIXME: Should be moved to its own header file.
199class KeyframeValueList {
200public:
201 explicit KeyframeValueList(AnimatedPropertyID property)
202 : m_property(property)
203 {
204 }
205
206 KeyframeValueList(const KeyframeValueList& other)
207 : m_property(other.property())
208 {
209 m_values.reserveInitialCapacity(other.m_values.size());
210 for (auto& value : other.m_values)
211 m_values.uncheckedAppend(value->clone());
212 }
213
214 KeyframeValueList(KeyframeValueList&&) = default;
215
216 KeyframeValueList& operator=(const KeyframeValueList& other)
217 {
218 KeyframeValueList copy(other);
219 swap(copy);
220 return *this;
221 }
222
223 KeyframeValueList& operator=(KeyframeValueList&&) = default;
224
225 void swap(KeyframeValueList& other)
226 {
227 std::swap(m_property, other.m_property);
228 m_values.swap(other.m_values);
229 }
230
231 AnimatedPropertyID property() const { return m_property; }
232
233 size_t size() const { return m_values.size(); }
234 const AnimationValue& at(size_t i) const { return *m_values.at(i); }
235
236 // Insert, sorted by keyTime.
237 WEBCORE_EXPORT void insert(std::unique_ptr<const AnimationValue>);
238
239protected:
240 Vector<std::unique_ptr<const AnimationValue>> m_values;
241 AnimatedPropertyID m_property;
242};
