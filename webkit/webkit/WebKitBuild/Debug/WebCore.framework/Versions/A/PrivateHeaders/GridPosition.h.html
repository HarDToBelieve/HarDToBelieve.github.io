32#pragma once
33
34#include <wtf/text/WTFString.h>
35
36namespace WebCore {
37
38enum GridPositionType {
39 AutoPosition,
40 ExplicitPosition, // [ <integer> || <string> ]
41 SpanPosition, // span && [ <integer> || <string> ]
42 NamedGridAreaPosition // <ident>
43};
44
45enum GridPositionSide {
46 ColumnStartSide,
47 ColumnEndSide,
48 RowStartSide,
49 RowEndSide
50};
51
52class GridPosition {
53public:
54 GridPosition()
55 : m_type(AutoPosition)
56 , m_integerPosition(0)
57 {
58 }
59
60 bool isPositive() const { return integerPosition() > 0; }
61
62 GridPositionType type() const { return m_type; }
63 bool isAuto() const { return m_type == AutoPosition; }
64 bool isSpan() const { return m_type == SpanPosition; }
65 bool isNamedGridArea() const { return m_type == NamedGridAreaPosition; }
66
67 WEBCORE_EXPORT void setExplicitPosition(int position, const String& namedGridLine);
68 void setAutoPosition();
69 WEBCORE_EXPORT void setSpanPosition(int position, const String& namedGridLine);
70 void setNamedGridArea(const String&);
71
72 WEBCORE_EXPORT int integerPosition() const;
73 String namedGridLine() const;
74 WEBCORE_EXPORT int spanPosition() const;
75
76 bool operator==(const GridPosition& other) const;
77
78 bool shouldBeResolvedAgainstOppositePosition() const { return isAuto() || isSpan(); }
79
80 // Note that grid line 1 is internally represented by the index 0, that's why the max value for
81 // a position is kGridMaxTracks instead of kGridMaxTracks + 1.
82 static int max();
83 static int min();
84
85 WEBCORE_EXPORT static void setMaxPositionForTesting(unsigned);
86
87private:
88 static std::optional<int> gMaxPositionForTesting;
89
90 void setIntegerPosition(int integerPosition) { m_integerPosition = clampTo(integerPosition, min(), max()); }
91
92 GridPositionType m_type;
93 int m_integerPosition;
94 String m_namedGridLine;
95};
96
97WTF::TextStream& operator<<(WTF::TextStream&, const GridPosition&);
98
99} // namespace WebCore
