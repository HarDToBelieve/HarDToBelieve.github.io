132inline SharedStringHash HTMLAnchorElement::visitedLinkHash() const
133{
134 ASSERT(isLink());
135 if (!m_storedVisitedLinkHash)
136 m_storedVisitedLinkHash = computeVisitedLinkHash(document().baseURL(), attributeWithoutSynchronization(HTMLNames::hrefAttr));
137 return *m_storedVisitedLinkHash;
138}
