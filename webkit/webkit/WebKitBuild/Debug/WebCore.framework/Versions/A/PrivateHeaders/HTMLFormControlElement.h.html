42// HTMLFormControlElement is the default implementation of FormAssociatedElement,
43// and form-associated element implementations should use HTMLFormControlElement
44// unless there is a special reason.
45class HTMLFormControlElement : public LabelableElement, public FormAssociatedElement {
46 WTF_MAKE_ISO_ALLOCATED(HTMLFormControlElement);
47 friend class DelayedUpdateValidityScope;
48public:
49 virtual ~HTMLFormControlElement();
50
51 HTMLFormElement* form() const final { return FormAssociatedElement::form(); }
52
53 WEBCORE_EXPORT String formEnctype() const;
54 WEBCORE_EXPORT void setFormEnctype(const String&);
55 WEBCORE_EXPORT String formMethod() const;
56 WEBCORE_EXPORT void setFormMethod(const String&);
57 bool formNoValidate() const;
58 WEBCORE_EXPORT String formAction() const;
59 WEBCORE_EXPORT void setFormAction(const AtomString&);
60
61 void setAncestorDisabled(bool isDisabled);
62
63 virtual void reset() { }
64
65 bool formControlValueMatchesRenderer() const { return m_valueMatchesRenderer; }
66 void setFormControlValueMatchesRenderer(bool b) { m_valueMatchesRenderer = b; }
67
68 bool wasChangedSinceLastFormControlChangeEvent() const { return m_wasChangedSinceLastFormControlChangeEvent; }
69 void setChangedSinceLastFormControlChangeEvent(bool);

71 virtual void dispatchFormControlChangeEvent();
72 void dispatchChangeEvent();

77 bool isEnumeratable() const override { return false; }

79 bool isRequired() const;

81 const AtomString& type() const { return formControlType(); }
82
83 virtual const AtomString& formControlType() const = 0;
84
85 virtual bool canTriggerImplicitSubmission() const { return false; }
86
87 // Override in derived classes to get the encoded name=value pair for submitting.
88 // Return true for a successful control (see HTML4-17.13.2).
89 bool appendFormData(DOMFormData&, bool) override { return false; }
90
91 virtual bool isSuccessfulSubmitButton() const { return false; }
92 virtual bool isActivatedSubmit() const { return false; }
93 virtual void setActivatedSubmit(bool) { }
94
95#if ENABLE(AUTOCORRECT)
96 WEBCORE_EXPORT bool shouldAutocorrect() const final;

99#if ENABLE(AUTOCAPITALIZE)
100 WEBCORE_EXPORT AutocapitalizeType autocapitalizeType() const final;

103 // "willValidate" means "is a candidate for constraint validation".
104 WEBCORE_EXPORT bool willValidate() const final;
105 void updateVisibleValidationMessage();

108 bool reportValidity();

110 bool isShowingValidationMessage() const;

116 bool isReadOnly() const { return m_isReadOnly; }

119 bool hasAutofocused() { return m_hasAutofocused; }

193
194 unsigned m_delayedUpdateValidityCount { 0 };
195
196 bool m_isFocusingWithValidationMessage { false };
197
198 unsigned m_disabled : 1;
199 unsigned m_isReadOnly : 1;
200 unsigned m_isRequired : 1;
201 unsigned m_valueMatchesRenderer : 1;
202 unsigned m_disabledByAncestorFieldset : 1;
203
204 enum DataListAncestorState { Unknown, InsideDataList, NotInsideDataList };
205 mutable unsigned m_dataListAncestorState : 2;
206
207 // The initial value of m_willValidate depends on the derived class. We can't
208 // initialize it with a virtual function in the constructor. m_willValidate
209 // is not deterministic as long as m_willValidateInitialized is false.
210 mutable bool m_willValidateInitialized: 1;
211 mutable bool m_willValidate : 1;
212
213 // Cache of validity()->valid().
214 // But "candidate for constraint validation" doesn't affect m_isValid.
215 unsigned m_isValid : 1;
216
217 unsigned m_wasChangedSinceLastFormControlChangeEvent : 1;
218
219 unsigned m_hasAutofocused : 1;
220};

222class DelayedUpdateValidityScope {
223public:
224 DelayedUpdateValidityScope(HTMLFormControlElement& element)
225 : m_element(element)
226 {
227 m_element.startDelayingUpdateValidity();
228 }
229
230 ~DelayedUpdateValidityScope()
231 {
232 m_element.endDelayingUpdateValidity();
233 }
234
235private:
236 HTMLFormControlElement& m_element;
237};
