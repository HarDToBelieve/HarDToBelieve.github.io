33class HTMLFrameOwnerElement : public HTMLElement {
34 WTF_MAKE_ISO_ALLOCATED(HTMLFrameOwnerElement);
35public:
36 virtual ~HTMLFrameOwnerElement();
37
38 Frame* contentFrame() const { return m_contentFrame.get(); }
39 WEBCORE_EXPORT WindowProxy* contentWindow() const;
40 WEBCORE_EXPORT Document* contentDocument() const;
41
42 void setContentFrame(Frame&);
43 void clearContentFrame();
44
45 void disconnectContentFrame();
46
47 // Most subclasses use RenderWidget (either RenderEmbeddedObject or RenderIFrame)
48 // except for HTMLObjectElement and HTMLEmbedElement which may return any
49 // RenderElement when using fallback content.
50 RenderWidget* renderWidget() const;
51
52 ExceptionOr<Document&> getSVGDocument() const;
53
54 virtual ScrollbarMode scrollingMode() const { return ScrollbarAuto; }
55
56 SandboxFlags sandboxFlags() const { return m_sandboxFlags; }
57
58 void scheduleInvalidateStyleAndLayerComposition();
59
60 virtual bool canLoadScriptURL(const URL&) const = 0;
61
62 virtual ReferrerPolicy referrerPolicy() const { return ReferrerPolicy::EmptyString; }
63
64 virtual bool shouldLoadFrameLazily() { return false; }
65 virtual bool isLazyLoadObserverActive() const { return false; }
66
67protected:
68 HTMLFrameOwnerElement(const QualifiedName& tagName, Document&);
69 void setSandboxFlags(SandboxFlags);
70 bool isProhibitedSelfReference(const URL&) const;
71
72private:
73 bool isKeyboardFocusable(KeyboardEvent*) const override;
74 bool isFrameOwnerElement() const final { return true; }
75
76 WeakPtr<Frame> m_contentFrame;
77 SandboxFlags m_sandboxFlags { SandboxNone };
78};
79
80class SubframeLoadingDisabler {
81public:
82 explicit SubframeLoadingDisabler(ContainerNode* root)
83 : m_root(root)
84 {
85 if (m_root)
86 disabledSubtreeRoots().add(m_root);
87 }
88
89 ~SubframeLoadingDisabler()
90 {
91 if (m_root)
92 disabledSubtreeRoots().remove(m_root);
93 }
94
95 static bool canLoadFrame(HTMLFrameOwnerElement&);
96
97private:
98 static HashCountedSet<ContainerNode*>& disabledSubtreeRoots()
99 {
100 static NeverDestroyed<HashCountedSet<ContainerNode*>> nodes;
101 return nodes;
102 }
103
104 ContainerNode* m_root;
105};
