35// FIXME: Not every header fits into a map. Notably, multiple Set-Cookie header fields are needed to set multiple cookies.
36
37class HTTPHeaderMap {
38public:
39 struct CommonHeader {
40 HTTPHeaderName key;
41 String value;
42
43 CommonHeader isolatedCopy() const { return { key , value.isolatedCopy() }; }
44 template <class Encoder> void encode(Encoder&) const;
45 template <class Decoder> static std::optional<CommonHeader> decode(Decoder&);
46
47 bool operator==(const CommonHeader& other) const { return key == other.key && value == other.value; }
48 };
49
50 struct UncommonHeader {
51 String key;
52 String value;
53
54 UncommonHeader isolatedCopy() const { return { key.isolatedCopy() , value.isolatedCopy() }; }
55 template <class Encoder> void encode(Encoder&) const;
56 template <class Decoder> static std::optional<UncommonHeader> decode(Decoder&);
57
58 bool operator==(const UncommonHeader& other) const { return key == other.key && value == other.value; }
59 };
60
61 typedef Vector<CommonHeader, 0, CrashOnOverflow, 6> CommonHeadersVector;
62 typedef Vector<UncommonHeader, 0, CrashOnOverflow, 0> UncommonHeadersVector;
63
64 class HTTPHeaderMapConstIterator {
65 public:
66 HTTPHeaderMapConstIterator(const HTTPHeaderMap& table, CommonHeadersVector::const_iterator commonHeadersIt, UncommonHeadersVector::const_iterator uncommonHeadersIt)
67 : m_table(table)
68 , m_commonHeadersIt(commonHeadersIt)
69 , m_uncommonHeadersIt(uncommonHeadersIt)
70 {
71 if (!updateKeyValue(m_commonHeadersIt))
72 updateKeyValue(m_uncommonHeadersIt);
73 }
74
75 struct KeyValue {
76 String key;
77 std::optional<HTTPHeaderName> keyAsHTTPHeaderName;
78 String value;
79 };
80
81 const KeyValue* get() const
82 {
83 ASSERT(*this != m_table.end());
84 return &m_keyValue;
85 }
86 const KeyValue& operator*() const { return *get(); }
87 const KeyValue* operator->() const { return get(); }
88
89 HTTPHeaderMapConstIterator& operator++()
90 {
91 if (m_commonHeadersIt != m_table.m_commonHeaders.end()) {
92 if (updateKeyValue(++m_commonHeadersIt))
93 return *this;
94 } else
95 ++m_uncommonHeadersIt;
96
97 updateKeyValue(m_uncommonHeadersIt);
98 return *this;
99 }
100
101 bool operator!=(const HTTPHeaderMapConstIterator& other) const { return !(*this == other); }
102 bool operator==(const HTTPHeaderMapConstIterator& other) const
103 {
104 return m_commonHeadersIt == other.m_commonHeadersIt && m_uncommonHeadersIt == other.m_uncommonHeadersIt;
105 }
106
107 private:
108 bool updateKeyValue(CommonHeadersVector::const_iterator it)
109 {
110 if (it == m_table.commonHeaders().end())
111 return false;
112 m_keyValue.key = httpHeaderNameString(it->key).toStringWithoutCopying();
113 m_keyValue.keyAsHTTPHeaderName = it->key;
114 m_keyValue.value = it->value;
115 return true;
116 }
117 bool updateKeyValue(UncommonHeadersVector::const_iterator it)
118 {
119 if (it == m_table.uncommonHeaders().end())
120 return false;
121 m_keyValue.key = it->key;
122 m_keyValue.keyAsHTTPHeaderName = std::nullopt;
123 m_keyValue.value = it->value;
124 return true;
125 }
126
127 const HTTPHeaderMap& m_table;
128 CommonHeadersVector::const_iterator m_commonHeadersIt;
129 UncommonHeadersVector::const_iterator m_uncommonHeadersIt;
130 KeyValue m_keyValue;
131 };
132 typedef HTTPHeaderMapConstIterator const_iterator;

134 WEBCORE_EXPORT HTTPHeaderMap();
135
136 // Gets a copy of the data suitable for passing to another thread.
137 WEBCORE_EXPORT HTTPHeaderMap isolatedCopy() const;
138
139 bool isEmpty() const { return m_commonHeaders.isEmpty() && m_uncommonHeaders.isEmpty(); }
140 int size() const { return m_commonHeaders.size() + m_uncommonHeaders.size(); }
141
142 void clear()
143 {
144 m_commonHeaders.clear();
145 m_uncommonHeaders.clear();
146 }
147
148 void shrinkToFit()
149 {
150 m_commonHeaders.shrinkToFit();
151 m_uncommonHeaders.shrinkToFit();
152 }
153
154 WEBCORE_EXPORT String get(const String& name) const;
155 WEBCORE_EXPORT void set(const String& name, const String& value);
156 WEBCORE_EXPORT void add(const String& name, const String& value);
157 WEBCORE_EXPORT void append(const String& name, const String& value);
158 WEBCORE_EXPORT bool contains(const String&) const;
159 WEBCORE_EXPORT bool remove(const String&);
160
161#if USE(CF)
162 void set(CFStringRef name, const String& value);
163#ifdef __OBJC__
164 void set(NSString *name, const String& value) { set((__bridge CFStringRef)name, value); }
165#endif
166#endif
167
168 WEBCORE_EXPORT String get(HTTPHeaderName) const;
169 void set(HTTPHeaderName, const String& value);
170 void add(HTTPHeaderName, const String& value);
171 bool addIfNotPresent(HTTPHeaderName, const String&);
172 WEBCORE_EXPORT bool contains(HTTPHeaderName) const;
173 WEBCORE_EXPORT bool remove(HTTPHeaderName);
174
175 // Instead of passing a string literal to any of these functions, just use a HTTPHeaderName instead.
176 template<size_t length> String get(const char (&)[length]) const = delete;
177 template<size_t length> void set(const char (&)[length], const String&) = delete;
178 template<size_t length> bool contains(const char (&)[length]) = delete;
179 template<size_t length> bool remove(const char (&)[length]) = delete;
180
181 const CommonHeadersVector& commonHeaders() const { return m_commonHeaders; }
182 const UncommonHeadersVector& uncommonHeaders() const { return m_uncommonHeaders; }
183 CommonHeadersVector& commonHeaders() { return m_commonHeaders; }
184 UncommonHeadersVector& uncommonHeaders() { return m_uncommonHeaders; }
185
186 const_iterator begin() const { return const_iterator(*this, m_commonHeaders.begin(), m_uncommonHeaders.begin()); }
187 const_iterator end() const { return const_iterator(*this, m_commonHeaders.end(), m_uncommonHeaders.end()); }
188
189 friend bool operator==(const HTTPHeaderMap& a, const HTTPHeaderMap& b)
190 {
191 if (a.m_commonHeaders.size() != b.m_commonHeaders.size() || a.m_uncommonHeaders.size() != b.m_uncommonHeaders.size())
192 return false;
193 for (auto& commonHeader : a.m_commonHeaders) {
194 if (b.get(commonHeader.key) != commonHeader.value)
195 return false;
196 }
197 for (auto& uncommonHeader : a.m_uncommonHeaders) {
198 if (b.getUncommonHeader(uncommonHeader.key) != uncommonHeader.value)
199 return false;
200 }
201 return true;
202 }
203
204 friend bool operator!=(const HTTPHeaderMap& a, const HTTPHeaderMap& b)
205 {
206 return !(a == b);
207 }
208
209 template <class Encoder> void encode(Encoder&) const;
210 template <class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, HTTPHeaderMap&);
211
212private:
213 void setUncommonHeader(const String& name, const String& value);
214 WEBCORE_EXPORT String getUncommonHeader(const String& name) const;
215
216 CommonHeadersVector m_commonHeaders;
217 UncommonHeadersVector m_uncommonHeaders;
218};
219
220template <class Encoder>
221void HTTPHeaderMap::CommonHeader::encode(Encoder& encoder) const
222{
223 encoder << key;
224 encoder << value;
225}
226
227template <class Decoder>
228auto HTTPHeaderMap::CommonHeader::decode(Decoder& decoder) -> std::optional<CommonHeader>
229{
230 HTTPHeaderName name;
231 if (!decoder.decode(name))
232 return std::nullopt;
233 String value;
234 if (!decoder.decode(value))
235 return std::nullopt;
236
237 return CommonHeader { name, WTFMove(value) };
238}
239
240template <class Encoder>
241void HTTPHeaderMap::UncommonHeader::encode(Encoder& encoder) const
242{
243 encoder << key;
244 encoder << value;
245}
246
247template <class Decoder>
248auto HTTPHeaderMap::UncommonHeader::decode(Decoder& decoder) -> std::optional<UncommonHeader>
249{
250 String name;
251 if (!decoder.decode(name))
252 return std::nullopt;
253 String value;
254 if (!decoder.decode(value))
255 return std::nullopt;
256
257 return UncommonHeader { WTFMove(name), WTFMove(value) };
258}
259
260template <class Encoder>
261void HTTPHeaderMap::encode(Encoder& encoder) const
262{
263 encoder << m_commonHeaders;
264 encoder << m_uncommonHeaders;
265}
266
267template <class Decoder>
268bool HTTPHeaderMap::decode(Decoder& decoder, HTTPHeaderMap& headerMap)
269{
270 if (!decoder.decode(headerMap.m_commonHeaders))
271 return false;
272
273 if (!decoder.decode(headerMap.m_uncommonHeaders))
274 return false;
275
276 return true;
277}
