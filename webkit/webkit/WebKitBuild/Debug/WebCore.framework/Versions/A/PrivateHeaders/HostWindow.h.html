26#pragma once
27
28#include "GraphicsContextGL.h"
29#include "Widget.h"
30
31namespace WebCore {
32
33class Cursor;
34class DestinationColorSpace;
35class ImageBuffer;
36
37enum class PixelFormat : uint8_t;
38enum class RenderingMode : bool;
39enum class RenderingPurpose : uint8_t;
40
41using FramesPerSecond = unsigned;
42
43class HostWindow {
44 WTF_MAKE_NONCOPYABLE(HostWindow); WTF_MAKE_FAST_ALLOCATED;
45public:
46 HostWindow() = default;
47 virtual ~HostWindow() = default;
48
49 // Requests the host invalidate the root view, not the contents.
50 virtual void invalidateRootView(const IntRect& updateRect) = 0;
51
52 // Requests the host invalidate the contents and the root view.
53 virtual void invalidateContentsAndRootView(const IntRect& updateRect) = 0;
54
55 // Requests the host scroll backingstore by the specified delta, rect to scroll, and clip rect.
56 virtual void scroll(const IntSize& scrollDelta, const IntRect& rectToScroll, const IntRect& clipRect) = 0;
57
58 // Requests the host invalidate the contents, not the root view. This is the slow path for scrolling.
59 virtual void invalidateContentsForSlowScroll(const IntRect& updateRect) = 0;
60
61 // Methods for doing coordinate conversions to and from screen coordinates.
62 virtual IntPoint screenToRootView(const IntPoint&) const = 0;
63 virtual IntRect rootViewToScreen(const IntRect&) const = 0;
64 virtual IntPoint accessibilityScreenToRootView(const IntPoint&) const = 0;
65 virtual IntRect rootViewToAccessibilityScreen(const IntRect&) const = 0;
66
67 virtual RefPtr<ImageBuffer> createImageBuffer(const FloatSize&, RenderingMode, RenderingPurpose, float resolutionScale, const DestinationColorSpace&, PixelFormat) const = 0;
68
69#if ENABLE(WEBGL)
70 virtual RefPtr<GraphicsContextGL> createGraphicsContextGL(const GraphicsContextGLAttributes&) const = 0;
71#endif
72
73 // Method for retrieving the native client of the page.
74 virtual PlatformPageClient platformPageClient() const = 0;
75
76 // Request that the cursor change.
77 virtual void setCursor(const Cursor&) = 0;
78
79 virtual void setCursorHiddenUntilMouseMoves(bool) = 0;
80
81 virtual PlatformDisplayID displayID() const = 0;
82 virtual void windowScreenDidChange(PlatformDisplayID, std::optional<FramesPerSecond> nominalFramesPerSecond) = 0;
83
84 virtual FloatSize screenSize() const = 0;
85 virtual FloatSize availableScreenSize() const = 0;
86 virtual FloatSize overrideScreenSize() const = 0;
87};
88
89} // namespace WebCore
