26#pragma once
27
28#include "DOMException.h"
29#include "ExceptionCode.h"
30#include <wtf/text/WTFString.h>
31
32namespace WebCore {
33
34class IDBError {
35public:
36 WEBCORE_EXPORT explicit IDBError(std::optional<ExceptionCode> = std::nullopt, const String& message = { });
37
38 static IDBError userDeleteError()
39 {
40 return IDBError { UnknownError, "Database deleted by request of the user"_s };
41 }
42
43 static IDBError serverConnectionLostError()
44 {
45 return IDBError { UnknownError, "Connection to Indexed Database server lost. Refresh the page to try again"_s };
46 }
47
48 RefPtr<DOMException> toDOMException() const;
49
50 std::optional<ExceptionCode> code() const { return m_code; }
51 String name() const;
52 String message() const;
53
54 bool isNull() const { return !m_code; }
55
56 WEBCORE_EXPORT IDBError isolatedCopy() const;
57
58 template<class Encoder> void encode(Encoder&) const;
59 template<class Decoder> static WARN_UNUSED_RETURN bool decode(Decoder&, IDBError&);
60
61private:
62 std::optional<ExceptionCode> m_code;
63 String m_message;
64};
65
66template<class Encoder>
67void IDBError::encode(Encoder& encoder) const
68{
69 if (m_code) {
70 encoder << true;
71 encoder << m_code.value();
72 } else
73 encoder << false;
74 encoder << m_message;
75}
76
77template<class Decoder>
78bool IDBError::decode(Decoder& decoder, IDBError& error)
79{
80 bool hasCode = false;
81 if (!decoder.decode(hasCode))
82 return false;
83
84 if (hasCode) {
85 ExceptionCode ec;
86 if (!decoder.decode(ec))
87 return false;
88 error.m_code = ec;
89 } else
90 error.m_code = std::nullopt;
91
92 if (!decoder.decode(error.m_message))
93 return false;
94
95 return true;
96}

98} // namespace WebCore
