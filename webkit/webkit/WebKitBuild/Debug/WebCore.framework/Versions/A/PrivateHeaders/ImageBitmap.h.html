39namespace WebCore {
40
41class Blob;
42class CanvasBase;
43class HTMLCanvasElement;
44class HTMLImageElement;
45class HTMLVideoElement;
46class ImageBitmapImageObserver;
47class ImageData;
48class IntRect;
49class IntSize;
50#if ENABLE(OFFSCREEN_CANVAS)
51class OffscreenCanvas;

53class PendingImageBitmap;

56struct ImageBitmapOptions;
57
58template<typename IDLType> class DOMPromiseDeferred;
59
60class ImageBitmap final : public ScriptWrappable, public RefCounted<ImageBitmap> {
61 WTF_MAKE_ISO_ALLOCATED(ImageBitmap);
62public:
63 using Source = Variant<
64 RefPtr<HTMLImageElement>,
65#if ENABLE(VIDEO)
66 RefPtr<HTMLVideoElement>,
67#endif
68 RefPtr<HTMLCanvasElement>,
69 RefPtr<ImageBitmap>,
70#if ENABLE(OFFSCREEN_CANVAS)
71 RefPtr<OffscreenCanvas>,
72#endif
73#if ENABLE(CSS_TYPED_OM)
74 RefPtr<TypedOMCSSImageValue>,
75#endif
76 RefPtr<Blob>,
77 RefPtr<ImageData>
78 >;
79
80 using Promise = DOMPromiseDeferred<IDLInterface<ImageBitmap>>;
81
82 static void createPromise(ScriptExecutionContext&, Source&&, ImageBitmapOptions&&, Promise&&);
83 static void createPromise(ScriptExecutionContext&, Source&&, ImageBitmapOptions&&, int sx, int sy, int sw, int sh, Promise&&);
84
85 static RefPtr<ImageBuffer> createImageBuffer(ScriptExecutionContext&, const FloatSize&, RenderingMode, float resolutionScale = 1);
86
87 static Ref<ImageBitmap> create(ScriptExecutionContext&, const IntSize&);
88 static Ref<ImageBitmap> create(std::optional<ImageBitmapBacking>&&);
89
90 ~ImageBitmap();
91
92 ImageBuffer* buffer() const { return m_backingStore ? m_backingStore->buffer() : nullptr; }
93 // This function has the implicit side-effect of detaching the backing store.
94 // It returns nullptr if the ImageBitmap's already detached.
95 RefPtr<ImageBuffer> takeImageBuffer();
96 OptionSet<SerializationState> serializationState() const { return m_backingStore ? m_backingStore->serializationState() : SerializationState(); }
97
98 unsigned width() const { return m_backingStore ? m_backingStore->width() : 0; }
99 unsigned height() const { return m_backingStore ? m_backingStore->height() : 0; }
100
101 bool originClean() const { return m_backingStore && m_backingStore->originClean(); }
102 bool premultiplyAlpha() const { return m_backingStore && m_backingStore->premultiplyAlpha(); }
103 bool forciblyPremultiplyAlpha() const { return m_backingStore && m_backingStore->forciblyPremultiplyAlpha(); }
104
105 std::optional<ImageBitmapBacking> takeImageBitmapBacking();
106 bool isDetached() const { return !m_backingStore; }
107 void close() { takeImageBitmapBacking(); }
108
109 static Vector<std::optional<ImageBitmapBacking>> detachBitmaps(Vector<RefPtr<ImageBitmap>>&&);
110
111private:
112 friend class ImageBitmapImageObserver;
113 friend class PendingImageBitmap;
114
115 ImageBitmap(std::optional<ImageBitmapBacking>&&);
116
117 static void resolveWithBlankImageBuffer(ScriptExecutionContext&, bool originClean, Promise&&);
118
119 static void createPromise(ScriptExecutionContext&, RefPtr<HTMLImageElement>&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
120#if ENABLE(VIDEO)
121 static void createPromise(ScriptExecutionContext&, RefPtr<HTMLVideoElement>&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
122#endif
123 static void createPromise(ScriptExecutionContext&, RefPtr<ImageBitmap>&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
124 static void createPromise(ScriptExecutionContext&, RefPtr<HTMLCanvasElement>&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
125#if ENABLE(OFFSCREEN_CANVAS)
126 static void createPromise(ScriptExecutionContext&, RefPtr<OffscreenCanvas>&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
127#endif
128 static void createPromise(ScriptExecutionContext&, CanvasBase&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
129 static void createPromise(ScriptExecutionContext&, RefPtr<Blob>&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
130 static void createPromise(ScriptExecutionContext&, RefPtr<ImageData>&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
131 static void createPromise(ScriptExecutionContext&, RefPtr<TypedOMCSSImageValue>&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
132 static void createFromBuffer(ScriptExecutionContext&, Ref<ArrayBuffer>&&, String mimeType, long long expectedContentLength, const URL&, ImageBitmapOptions&&, std::optional<IntRect>, Promise&&);
133
134 std::optional<ImageBitmapBacking> m_backingStore;
135};
