48 enum class MediaType {
49 Image,
50 Video,

53 struct FrameMetadata {
54 ImageOrientation orientation;
55 std::optional<IntSize> densityCorrectedSize;
56 };
57
58 struct FrameInfo {
59 bool hasAlpha;
60 Seconds duration;
61
62 template<class Encoder>
63 void encode(Encoder& encoder) const
64 {
65 encoder << hasAlpha;
66 encoder << duration;
67 }
68
69 template<class Decoder>
70 static std::optional<FrameInfo> decode(Decoder& decoder)
71 {
72 std::optional<bool> hasAlpha;
73 decoder >> hasAlpha;
74 if (!hasAlpha)
75 return std::nullopt;
76
77 std::optional<Seconds> duration;
78 decoder >> duration;
79 if (!duration)
80 return std::nullopt;
81
82 return {{
83 *hasAlpha,
84 *duration
85 }};
86 }
87 };
