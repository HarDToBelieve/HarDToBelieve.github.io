26#pragma once
27
28#include "Color.h"
29#include "DecodingOptions.h"
30#include "ImageOrientation.h"
31#include "ImageTypes.h"
32#include "IntSize.h"
33#include "NativeImage.h"
34#include <wtf/Seconds.h>
35
36namespace WebCore {
37
38class ImageFrame {
39 friend class ImageSource;
40public:
41 enum class Caching { Metadata, MetadataAndImage };
42
43 ImageFrame();
44 ImageFrame(const ImageFrame& other) { operator=(other); }
45
46 ~ImageFrame();
47
48 static const ImageFrame& defaultFrame();
49
50 ImageFrame& operator=(const ImageFrame& other);
51
52 unsigned clearImage();
53 unsigned clear();
54
55 void setDecodingStatus(DecodingStatus);
56 DecodingStatus decodingStatus() const;
57
58 bool isInvalid() const { return m_decodingStatus == DecodingStatus::Invalid; }
59 bool isPartial() const { return m_decodingStatus == DecodingStatus::Partial; }
60 bool isComplete() const { return m_decodingStatus == DecodingStatus::Complete; }
61
62 IntSize size() const;
63 unsigned frameBytes() const { return hasNativeImage() ? (size().area() * sizeof(uint32_t)).value() : 0; }
64 SubsamplingLevel subsamplingLevel() const { return m_subsamplingLevel; }
65
66 RefPtr<NativeImage> nativeImage() const { return m_nativeImage; }
67
68 void setOrientation(ImageOrientation orientation) { m_orientation = orientation; };
69 ImageOrientation orientation() const { return m_orientation; }
70
71 void setDensityCorrectedSize(const IntSize& size) { m_densityCorrectedSize = size; }
72 std::optional<IntSize> densityCorrectedSize() const { return m_densityCorrectedSize; }
73
74 void setDuration(const Seconds& duration) { m_duration = duration; }
75 Seconds duration() const { return m_duration; }
76
77 void setHasAlpha(bool hasAlpha) { m_hasAlpha = hasAlpha; }
78 bool hasAlpha() const { return !hasMetadata() || m_hasAlpha; }
79
80 bool hasNativeImage(const std::optional<SubsamplingLevel>& = { }) const;
81 bool hasFullSizeNativeImage(const std::optional<SubsamplingLevel>& = { }) const;
82 bool hasDecodedNativeImageCompatibleWithOptions(const std::optional<SubsamplingLevel>&, const DecodingOptions&) const;
83 bool hasMetadata() const { return !size().isEmpty(); }
84
85 Color singlePixelSolidColor() const;
86
87private:
88 DecodingStatus m_decodingStatus { DecodingStatus::Invalid };
89 IntSize m_size;
90
91 RefPtr<NativeImage> m_nativeImage;
92 SubsamplingLevel m_subsamplingLevel { SubsamplingLevel::Default };
93 DecodingOptions m_decodingOptions;
94
95 ImageOrientation m_orientation { ImageOrientation::None };
96 std::optional<IntSize> m_densityCorrectedSize;
97 Seconds m_duration;
98 bool m_hasAlpha { true };
99};
100
101}
