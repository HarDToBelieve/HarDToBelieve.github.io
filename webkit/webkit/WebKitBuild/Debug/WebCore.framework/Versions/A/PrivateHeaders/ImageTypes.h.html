34// There are four subsampling levels: 0 = 1x, 1 = 0.5x, 2 = 0.25x, 3 = 0.125x.
35enum class SubsamplingLevel {
36 First = 0,
37 Default = First,
38 Level0 = First,
39 Level1,
40 Level2,
41 Level3,
42 Last = Level3,
43 Max
44};
45
46inline SubsamplingLevel& operator++(SubsamplingLevel& subsamplingLevel)
47{
48 subsamplingLevel = static_cast<SubsamplingLevel>(static_cast<int>(subsamplingLevel) + 1);
49 ASSERT(subsamplingLevel <= SubsamplingLevel::Max);
50 return subsamplingLevel;
51}
