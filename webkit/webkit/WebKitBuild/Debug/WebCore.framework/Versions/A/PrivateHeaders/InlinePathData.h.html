30#include "FloatPoint.h"

33namespace WebCore {
34
35struct LineData {
36 FloatPoint start;
37 FloatPoint end;
38
39 template<class Encoder> void encode(Encoder&) const;
40 template<class Decoder> static std::optional<LineData> decode(Decoder&);
41};
42
43struct ArcData {
44 enum class Type : uint8_t {
45 ArcOnly,
46 LineAndArc,
47 ClosedLineAndArc
48 };
49
50 FloatPoint start;
51 FloatPoint center;
52 float radius { 0 };
53 float startAngle { 0 };
54 float endAngle { 0 };
55 bool clockwise { false };
56 Type type { Type::ArcOnly };
57
58 template<class Encoder> void encode(Encoder&) const;
59 template<class Decoder> static std::optional<ArcData> decode(Decoder&);
60};
61
62struct MoveData {
63 FloatPoint location;
64
65 template<class Encoder> void encode(Encoder&) const;
66 template<class Decoder> static std::optional<MoveData> decode(Decoder&);
67};
68
69struct QuadCurveData {
70 FloatPoint startPoint;
71 FloatPoint controlPoint;
72 FloatPoint endPoint;
73
74 template<class Encoder> void encode(Encoder&) const;
75 template<class Decoder> static std::optional<QuadCurveData> decode(Decoder&);
76};
77
78struct BezierCurveData {
79 FloatPoint startPoint;
80 FloatPoint controlPoint1;
81 FloatPoint controlPoint2;
82 FloatPoint endPoint;
83
84 template<class Encoder> void encode(Encoder&) const;
85 template<class Decoder> static std::optional<BezierCurveData> decode(Decoder&);
86};
87
88template<class Encoder> void MoveData::encode(Encoder& encoder) const
89{
90 encoder << location;
91}
92
93template<class Decoder> std::optional<MoveData> MoveData::decode(Decoder& decoder)
94{
95 MoveData data;
96 if (!decoder.decode(data.location))
97 return std::nullopt;
98 return data;
99}
100
101template<class Encoder> void LineData::encode(Encoder& encoder) const
102{
103 encoder << start;
104 encoder << end;
105}
106
107template<class Decoder> std::optional<LineData> LineData::decode(Decoder& decoder)
108{
109 LineData data;
110 if (!decoder.decode(data.start))
111 return std::nullopt;
112
113 if (!decoder.decode(data.end))
114 return std::nullopt;
115
116 return data;
117}
118
119template<class Encoder> void ArcData::encode(Encoder& encoder) const
120{
121 encoder << start;
122 encoder << center;
123 encoder << radius;
124 encoder << startAngle;
125 encoder << endAngle;
126 encoder << clockwise;
127 encoder << type;
128}
129
130template<class Decoder> std::optional<ArcData> ArcData::decode(Decoder& decoder)
131{
132 ArcData data;
133 if (!decoder.decode(data.start))
134 return std::nullopt;
135
136 if (!decoder.decode(data.center))
137 return std::nullopt;
138
139 if (!decoder.decode(data.radius))
140 return std::nullopt;
141
142 if (!decoder.decode(data.startAngle))
143 return std::nullopt;
144
145 if (!decoder.decode(data.endAngle))
146 return std::nullopt;
147
148 if (!decoder.decode(data.clockwise))
149 return std::nullopt;
150
151 if (!decoder.decode(data.type))
152 return std::nullopt;
153
154 return data;
155}
156
157template<class Encoder> void QuadCurveData::encode(Encoder& encoder) const
158{
159 encoder << startPoint;
160 encoder << controlPoint;
161 encoder << endPoint;
162}
163
164template<class Decoder> std::optional<QuadCurveData> QuadCurveData::decode(Decoder& decoder)
165{
166 QuadCurveData data;
167 if (!decoder.decode(data.startPoint))
168 return std::nullopt;
169
170 if (!decoder.decode(data.controlPoint))
171 return std::nullopt;
172
173 if (!decoder.decode(data.endPoint))
174 return std::nullopt;
175
176 return data;
177}
178
179template<class Encoder> void BezierCurveData::encode(Encoder& encoder) const
180{
181 encoder << startPoint;
182 encoder << controlPoint1;
183 encoder << controlPoint2;
184 encoder << endPoint;
185}
186
187template<class Decoder> std::optional<BezierCurveData> BezierCurveData::decode(Decoder& decoder)
188{
189 BezierCurveData data;
190 if (!decoder.decode(data.startPoint))
191 return std::nullopt;
192
193 if (!decoder.decode(data.controlPoint1))
194 return std::nullopt;
195
196 if (!decoder.decode(data.controlPoint2))
197 return std::nullopt;
198
199 if (!decoder.decode(data.endPoint))
200 return std::nullopt;
201
202 return data;
203}
