33namespace WebCore {
34
35class IntRectExtent {
36public:
37 IntRectExtent() = default;
38
39 IntRectExtent(int top, int right, int bottom, int left)
40 : m_top(top)
41 , m_right(right)
42 , m_bottom(bottom)
43 , m_left(left)
44 {
45 }
46
47 int top() const { return m_top; }
48 void setTop(int top) { m_top = top; }
49
50 int right() const { return m_right; }
51 void setRight(int right) { m_right = right; }
52
53 int bottom() const { return m_bottom; }
54 void setBottom(int bottom) { m_bottom = bottom; }
55
56 int left() const { return m_left; }
57 void setLeft(int left) { m_left = left; }
58
59 bool isZero() const { return !left() && !right() && !top() && !bottom(); }
60
61private:
62 int m_top { 0 };
63 int m_right { 0 };
64 int m_bottom { 0 };
65 int m_left { 0 };
66};
67
68inline bool operator==(const IntRectExtent& a, const IntRectExtent& b)
69{
70 return a.top() == b.top() && a.right() == b.right() && a.bottom() == b.bottom() && a.left() == b.left();
71}
72
73inline bool operator!=(const IntRectExtent& a, const IntRectExtent& b)
74{
75 return !(a == b);
76}
77
78template<typename RectType>
79inline RectType& operator+=(RectType& rect, const IntRectExtent& extent)
80{
81 rect.move(-extent.left(), -extent.top());
82 rect.expand(extent.left() + extent.right(), extent.top() + extent.bottom());
83 return rect;
84}
85
86inline IntRectExtent& operator+=(IntRectExtent& a, const IntRectExtent& b)
87{
88 a.setTop(a.top() + b.top());
89 a.setRight(a.right() + b.right());
90 a.setBottom(a.bottom() + b.bottom());
91 a.setLeft(a.left() + b.left());
92 return a;
93}
94
95using IntOutsets = IntRectExtent;
