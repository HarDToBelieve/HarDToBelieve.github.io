32namespace WTF {
33
34template<> struct IntHash<WebCore::IntRect> {
35 static unsigned hash(const WebCore::IntRect& key)
36 {
37 return pairIntHash(DefaultHash<WebCore::IntPoint>::hash(key.location()), DefaultHash<WebCore::IntSize>::hash(key.size()));
38 }
39 static bool equal(const WebCore::IntRect& a, const WebCore::IntRect& b)
40 {
41 return DefaultHash<WebCore::IntPoint>::equal(a.location(), b.location()) && DefaultHash<WebCore::IntSize>::equal(a.size(), b.size());
42 }
43 static const bool safeToCompareToEmptyOrDeleted = true;
44};
45template<> struct DefaultHash<WebCore::IntRect> : IntHash<WebCore::IntRect> { };
46
47template<> struct HashTraits<WebCore::IntRect> : GenericHashTraits<WebCore::IntRect> {
48 static const bool emptyValueIsZero = true;
49 static void constructDeletedValue(WebCore::IntRect& slot) { new (NotNull, &slot) WebCore::IntRect(-1, -1, -1, -1); }
50 static bool isDeletedValue(const WebCore::IntRect& value) { return value.x() == -1 && value.y() == -1 && value.width() == -1 && value.height() == -1; }
51};
