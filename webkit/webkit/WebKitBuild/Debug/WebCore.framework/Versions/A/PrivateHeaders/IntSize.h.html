64namespace WebCore {
65
66class FloatSize;
67
68class IntSize {
69public:
70 IntSize() : m_width(0), m_height(0) { }
71 IntSize(int width, int height) : m_width(width), m_height(height) { }
72 WEBCORE_EXPORT explicit IntSize(const FloatSize&); // don't do this implicitly since it's lossy
73
74 int width() const { return m_width; }
75 int height() const { return m_height; }
76
77 void setWidth(int width) { m_width = width; }
78 void setHeight(int height) { m_height = height; }
79
80 bool isEmpty() const { return m_width <= 0 || m_height <= 0; }
81 bool isZero() const { return !m_width && !m_height; }
82
83 float aspectRatio() const { return static_cast<float>(m_width) / static_cast<float>(m_height); }
84
85 void expand(int width, int height)
86 {
87 m_width += width;
88 m_height += height;
89 }
90
91 void contract(int width, int height)
92 {
93 m_width -= width;
94 m_height -= height;
95 }
96
97 void scale(float widthScale, float heightScale)
98 {
99 m_width = static_cast<int>(static_cast<float>(m_width) * widthScale);
100 m_height = static_cast<int>(static_cast<float>(m_height) * heightScale);
101 }
102
103 void scale(float scale)
104 {
105 this->scale(scale, scale);
106 }
107
108 IntSize expandedTo(const IntSize& other) const
109 {
110 return IntSize(std::max(m_width, other.m_width), std::max(m_height, other.m_height));
111 }
112
113 IntSize shrunkTo(const IntSize& other) const
114 {
115 return IntSize(std::min(m_width, other.m_width), std::min(m_height, other.m_height));
116 }
117
118 void clampNegativeToZero()
119 {
120 *this = expandedTo(IntSize());
121 }
122
123 void clampToMinimumSize(const IntSize& minimumSize)
124 {
125 if (m_width < minimumSize.width())
126 m_width = minimumSize.width();
127 if (m_height < minimumSize.height())
128 m_height = minimumSize.height();
129 }
130
131 WEBCORE_EXPORT IntSize constrainedBetween(const IntSize& min, const IntSize& max) const;
132
133 template <typename T = WTF::CrashOnOverflow>
134 Checked<unsigned, T> area() const
135 {
136 return Checked<unsigned, T>(abs(m_width)) * abs(m_height);
137 }
138
139 uint64_t unclampedArea() const
140 {
141 return static_cast<uint64_t>(abs(m_width)) * abs(m_height);
142 }
143
144 int diagonalLengthSquared() const
145 {
146 return m_width * m_width + m_height * m_height;
147 }
148
149 IntSize transposedSize() const
150 {
151 return IntSize(m_height, m_width);
152 }
153
154#if USE(CG)
155 WEBCORE_EXPORT explicit IntSize(const CGSize&); // don't do this implicitly since it's lossy
156 WEBCORE_EXPORT operator CGSize() const;
157#endif
158
159#if PLATFORM(MAC) && !defined(NSGEOMETRY_TYPES_SAME_AS_CGGEOMETRY_TYPES)
160 WEBCORE_EXPORT explicit IntSize(const NSSize &); // don't do this implicitly since it's lossy
161 WEBCORE_EXPORT operator NSSize() const;
162#endif
163
164#if PLATFORM(WIN)
165 IntSize(const SIZE&);
166 operator SIZE() const;
167 IntSize(const D2D1_SIZE_U&);
168 explicit IntSize(const D2D1_SIZE_F&); // don't do this implicitly since it's lossy;
169 operator D2D1_SIZE_U() const;
170 operator D2D1_SIZE_F() const;
171#endif
172
173 String toJSONString() const;
174 Ref<JSON::Object> toJSONObject() const;
175
176private:
177 int m_width, m_height;
178};
179
180inline IntSize& operator+=(IntSize& a, const IntSize& b)
181{
182 a.setWidth(a.width() + b.width());
183 a.setHeight(a.height() + b.height());
184 return a;
185}
186
187inline IntSize& operator-=(IntSize& a, const IntSize& b)
188{
189 a.setWidth(a.width() - b.width());
190 a.setHeight(a.height() - b.height());
191 return a;
192}
193
194inline IntSize operator+(const IntSize& a, const IntSize& b)
195{
196 return IntSize(a.width() + b.width(), a.height() + b.height());
197}
198
199inline IntSize operator-(const IntSize& a, const IntSize& b)
200{
201 return IntSize(a.width() - b.width(), a.height() - b.height());
202}
203
204inline IntSize operator-(const IntSize& size)
205{
206 return IntSize(-size.width(), -size.height());
207}
208
209inline bool operator==(const IntSize& a, const IntSize& b)
210{
211 return a.width() == b.width() && a.height() == b.height();
212}
213
214inline bool operator!=(const IntSize& a, const IntSize& b)
215{
216 return a.width() != b.width() || a.height() != b.height();
217}
