61template<typename T> inline T* BindingSecurity::checkSecurityForNode(JSC::JSGlobalObject& lexicalGlobalObject, T& node)
62{
63 return shouldAllowAccessToNode(lexicalGlobalObject, &node) ? &node : nullptr;
64}
65
66template<typename T> inline T* BindingSecurity::checkSecurityForNode(JSC::JSGlobalObject& lexicalGlobalObject, T* node)
67{
68 return shouldAllowAccessToNode(lexicalGlobalObject, node) ? node : nullptr;
69}
70
71template<typename T> inline ExceptionOr<T*> BindingSecurity::checkSecurityForNode(JSC::JSGlobalObject& lexicalGlobalObject, ExceptionOr<T*>&& value)
72{
73 if (value.hasException())
74 return value.releaseException();
75 return checkSecurityForNode(lexicalGlobalObject, value.releaseReturnValue());
76}
77
78template<typename T> inline ExceptionOr<T*> BindingSecurity::checkSecurityForNode(JSC::JSGlobalObject& lexicalGlobalObject, ExceptionOr<T&>&& value)
79{
80 if (value.hasException())
81 return value.releaseException();
82 return checkSecurityForNode(lexicalGlobalObject, value.releaseReturnValue());
83}
