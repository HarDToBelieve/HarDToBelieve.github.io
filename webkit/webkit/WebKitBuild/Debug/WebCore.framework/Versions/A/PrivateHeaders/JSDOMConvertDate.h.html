28#include "IDLTypes.h"
29#include "JSDOMConvertBase.h"
30#include <JavaScriptCore/JSGlobalObject.h>

32namespace WebCore {
33
34JSC::JSValue jsDate(JSC::JSGlobalObject&, double value);
35double valueToDate(JSC::JSGlobalObject&, JSC::JSValue); // NaN if the value can't be converted to a date.
36
37template<> struct Converter<IDLDate> : DefaultConverter<IDLDate> {
38 static double convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
39 {
40 return valueToDate(lexicalGlobalObject, value);
41 }
42};
43
44template<> struct JSConverter<IDLDate> {
45 static constexpr bool needsState = true;
46 static constexpr bool needsGlobalObject = false;
47
48 // FIXME: This should be taking a JSDOMGlobalObject and passing it to jsDate.
49 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, double value)
50 {
51 return jsDate(lexicalGlobalObject, value);
52 }
53};

55} // namespace WebCore
