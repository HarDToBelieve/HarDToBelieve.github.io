43inline String propertyNameToString(JSC::PropertyName propertyName)
44{
45 ASSERT(!propertyName.isSymbol());
46 return propertyName.uid() ? propertyName.uid() : propertyName.publicName();
47}
48
49inline AtomString propertyNameToAtomString(JSC::PropertyName propertyName)
50{
51 return AtomString(propertyName.uid() ? propertyName.uid() : propertyName.publicName());
52}

57template<> struct Converter<IDLDOMString> : DefaultConverter<IDLDOMString> {
58 static String convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
59 {
60 return value.toWTFString(&lexicalGlobalObject);
61 }
62};

64template<> struct JSConverter<IDLDOMString> {
65 static constexpr bool needsState = true;
66 static constexpr bool needsGlobalObject = false;
67
68 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, const String& value)
69 {
70 return JSC::jsStringWithCache(JSC::getVM(&lexicalGlobalObject), value);
71 }
72
73 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, const UncachedString& value)
74 {
75 return JSC::jsString(JSC::getVM(&lexicalGlobalObject), value.string);
76 }
77
78 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, const OwnedString& value)
79 {
80 return JSC::jsOwnedString(JSC::getVM(&lexicalGlobalObject), value.string);
81 }
82
83 static JSC::JSValue convert(JSC::JSGlobalObject& lexicalGlobalObject, const URL& value)
84 {
85 return JSC::jsOwnedString(JSC::getVM(&lexicalGlobalObject), value.string());
86 }

155template<typename T> struct Converter<IDLLegacyNullToEmptyStringAdaptor<T>> : DefaultConverter<IDLLegacyNullToEmptyStringAdaptor<T>> {
156 static String convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
157 {
158 if (value.isNull())
159 return emptyString();
160 return Converter<T>::convert(lexicalGlobalObject, value);
161 }
162};

174template<typename T> struct Converter<IDLAtomStringAdaptor<T>> : DefaultConverter<IDLAtomStringAdaptor<T>> {
175 static AtomString convert(JSC::JSGlobalObject& lexicalGlobalObject, JSC::JSValue value)
176 {
177 static_assert(std::is_same<T, IDLDOMString>::value, "This adaptor is only supported for IDLDOMString at the moment.");
178
179 return value.toString(&lexicalGlobalObject)->toAtomString(&lexicalGlobalObject);
180 }
181};
