35namespace WebCore {
36
37class WEBCORE_EXPORT DOMGuardedObject : public RefCounted<DOMGuardedObject>, public ActiveDOMCallback {
38public:
39 ~DOMGuardedObject();
40
41 bool isSuspended() const { return !m_guarded || !canInvokeCallback(); } // The wrapper world has gone away or active DOM objects have been suspended.
42
43 template<typename Visitor> void visitAggregate(Visitor& visitor) { visitor.append(m_guarded); }
44
45 JSC::JSValue guardedObject() const { return m_guarded.get(); }
46 JSDOMGlobalObject* globalObject() const { return m_globalObject.get(); }
47
48 void clear();
49
50protected:
51 DOMGuardedObject(JSDOMGlobalObject&, JSC::JSCell&);
52
53 void contextDestroyed() override;
54 bool isEmpty() const { return !m_guarded; }
55
56 JSC::Weak<JSC::JSCell> m_guarded;
57 JSC::Weak<JSDOMGlobalObject> m_globalObject;
58
59private:
60 void removeFromGlobalObject();
61};
62
63template <typename T> class DOMGuarded : public DOMGuardedObject {
64protected:
65 DOMGuarded(JSDOMGlobalObject& globalObject, T& guarded) : DOMGuardedObject(globalObject, guarded) { }
66 T* guarded() const { return JSC::jsDynamicCast<T*>(globalObject()->vm(), guardedObject()); }
67};
