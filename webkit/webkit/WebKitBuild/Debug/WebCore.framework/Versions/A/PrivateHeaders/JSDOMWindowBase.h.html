107WEBCORE_EXPORT JSC::JSValue toJS(JSC::JSGlobalObject*, DOMWindow&);
108// The following return a JSWindowProxy or jsNull()
109// JSDOMGlobalObject* is ignored, accessing a window in any context will use that DOMWindow's prototype chain.
110inline JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject*, DOMWindow& window) { return toJS(lexicalGlobalObject, window); }
111inline JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, DOMWindow* window) { return window ? toJS(lexicalGlobalObject, globalObject, *window) : JSC::jsNull(); }
112inline JSC::JSValue toJS(JSC::JSGlobalObject* lexicalGlobalObject, DOMWindow* window) { return window ? toJS(lexicalGlobalObject, *window) : JSC::jsNull(); }

119// on the stack, or the author function or script that originally scheduled the currently-running callback."
120// (<https://html.spec.whatwg.org/multipage/webappapis.html#concept-incumbent-everything>, 27 April 2017)
121// FIXME: Make this work for an "author function or script that originally scheduled the currently-running callback."
122// See <https://bugs.webkit.org/show_bug.cgi?id=163412>.

128// FIXME: This should probably be removed in favor of one of the other DOMWindow accessors. It is intended
129// to provide the document specfied as the 'responsible document' in the algorithm for document.open()
