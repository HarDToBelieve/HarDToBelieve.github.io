76inline JSDOMGlobalObject* deprecatedGlobalObjectForPrototype(JSC::JSGlobalObject* lexicalGlobalObject)
77{
78 // FIXME: Callers to this function should be using the global object
79 // from which the object is being created, instead of assuming the lexical one.
80 // e.g. subframe.document.body should use the subframe's global object, not the lexical one.
81 return JSC::jsCast<JSDOMGlobalObject*>(lexicalGlobalObject);
82}

84template<typename WrapperClass> inline JSC::Structure* getDOMStructure(JSC::VM& vm, JSDOMGlobalObject& globalObject)
85{
86 if (JSC::Structure* structure = getCachedDOMStructure(globalObject, WrapperClass::info()))
87 return structure;
88 return cacheDOMStructure(globalObject, WrapperClass::createStructure(vm, &globalObject, WrapperClass::createPrototype(vm, globalObject)), WrapperClass::info());
89}

162template<typename DOMClass> inline JSC::JSObject* getCachedWrapper(DOMWrapperWorld& world, DOMClass& domObject)
163{
164 if (auto* wrapper = getInlineCachedWrapper(world, &domObject))
165 return wrapper;
166 return world.wrappers().get(wrapperKey(&domObject));
167}
168
169template<typename DOMClass, typename WrapperClass> inline void cacheWrapper(DOMWrapperWorld& world, DOMClass* domObject, WrapperClass* wrapper)
170{
171 JSC::WeakHandleOwner* owner = wrapperOwner(world, domObject);
172 if (setInlineCachedWrapper(world, domObject, wrapper, owner))
173 return;
174 weakAdd(world.wrappers(), wrapperKey(domObject), JSC::Weak<JSC::JSObject>(wrapper, owner, &world));
175}
176
177template<typename DOMClass, typename WrapperClass> inline void uncacheWrapper(DOMWrapperWorld& world, DOMClass* domObject, WrapperClass* wrapper)
178{
179 if (clearInlineCachedWrapper(world, domObject, wrapper))
180 return;
181 weakRemove(world.wrappers(), wrapperKey(domObject), wrapper);
182}

184template<typename DOMClass, typename T> inline auto createWrapper(JSDOMGlobalObject* globalObject, Ref<T>&& domObject) -> typename std::enable_if<std::is_same<DOMClass, T>::value, typename JSDOMWrapperConverterTraits<DOMClass>::WrapperClass*>::type
185{
186 using WrapperClass = typename JSDOMWrapperConverterTraits<DOMClass>::WrapperClass;
187
188 ASSERT(!getCachedWrapper(globalObject->world(), domObject));
189 auto* domObjectPtr = domObject.ptr();
190 auto* wrapper = WrapperClass::create(getDOMStructure<WrapperClass>(globalObject->vm(), *globalObject), globalObject, WTFMove(domObject));
191 cacheWrapper(globalObject->world(), domObjectPtr, wrapper);
192 return wrapper;
193}
194
195template<typename DOMClass, typename T> inline auto createWrapper(JSDOMGlobalObject* globalObject, Ref<T>&& domObject) -> typename std::enable_if<!std::is_same<DOMClass, T>::value, typename JSDOMWrapperConverterTraits<DOMClass>::WrapperClass*>::type
196{
197 return createWrapper<DOMClass>(globalObject, static_reference_cast<DOMClass>(WTFMove(domObject)));
198}

200template<typename DOMClass> inline JSC::JSValue wrap(JSC::JSGlobalObject* lexicalGlobalObject, JSDOMGlobalObject* globalObject, DOMClass& domObject)
201{
202 if (auto* wrapper = getCachedWrapper(globalObject->world(), domObject))
203 return wrapper;
204 return toJSNewlyCreated(lexicalGlobalObject, globalObject, Ref<DOMClass>(domObject));
205}

207template<typename DOMClass> inline void setSubclassStructureIfNeeded(JSC::JSGlobalObject* lexicalGlobalObject, JSC::CallFrame* callFrame, JSC::JSObject* jsObject)
208{
209 JSC::JSObject* newTarget = callFrame->newTarget().getObject();
210 JSC::JSObject* constructor = callFrame->jsCallee();
211 if (!newTarget || newTarget == constructor)
212 return;
213
214 using WrapperClass = typename JSDOMWrapperConverterTraits<DOMClass>::WrapperClass;
215
216 JSC::VM& vm = lexicalGlobalObject->vm();
217 auto scope = DECLARE_THROW_SCOPE(vm);
218
219 auto* functionGlobalObject = JSC::getFunctionRealm(lexicalGlobalObject, newTarget);
220 RETURN_IF_EXCEPTION(scope, void());
221 auto* newTargetGlobalObject = JSC::jsCast<JSDOMGlobalObject*>(functionGlobalObject);
222 auto* baseStructure = getDOMStructure<WrapperClass>(vm, *newTargetGlobalObject);
223 auto* subclassStructure = JSC::InternalFunction::createSubclassStructure(lexicalGlobalObject, newTarget, baseStructure);
224 RETURN_IF_EXCEPTION(scope, void());
225 jsObject->setStructure(vm, subclassStructure);
226}
