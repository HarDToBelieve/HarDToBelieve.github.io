45class LayoutRect {
46public:
47 LayoutRect() { }
48 LayoutRect(const LayoutPoint& location, const LayoutSize& size)
49 : m_location(location), m_size(size) { }
50 template<typename T1, typename T2, typename U1, typename U2>
51 LayoutRect(T1 x, T2 y, U1 width, U2 height)
52 : m_location(LayoutPoint(x, y)), m_size(LayoutSize(width, height)) { }
53 LayoutRect(const LayoutPoint& topLeft, const LayoutPoint& bottomRight)
54 : m_location(topLeft), m_size(LayoutSize(bottomRight.x() - topLeft.x(), bottomRight.y() - topLeft.y())) { }
55 LayoutRect(const FloatPoint& location, const FloatSize& size)
56 : m_location(location), m_size(size) { }
57 LayoutRect(const IntRect& rect) : m_location(rect.location()), m_size(rect.size()) { }
58
59 WEBCORE_EXPORT explicit LayoutRect(const FloatRect&); // don't do this implicitly since it's lossy
60
61 template<class Encoder>
62 void encode(Encoder& encoder) const
63 {
64 encoder << m_location << m_size;
65 }
66
67 template<class Decoder>
68 static std::optional<LayoutRect> decode(Decoder& decoder)
69 {
70 std::optional<LayoutPoint> layoutPoint;
71 decoder >> layoutPoint;
72 if (!layoutPoint)
73 return std::nullopt;
74
75 std::optional<LayoutSize> layoutSize;
76 decoder >> layoutSize;
77 if (!layoutSize)
78 return std::nullopt;
79
80 return {{ *layoutPoint, *layoutSize }};
81 }
82
83 LayoutPoint location() const { return m_location; }
84 LayoutSize size() const { return m_size; }
85
86 void setLocation(const LayoutPoint& location) { m_location = location; }
87 void setSize(const LayoutSize& size) { m_size = size; }
88
89 LayoutUnit x() const { return m_location.x(); }
90 LayoutUnit y() const { return m_location.y(); }
91 LayoutUnit maxX() const { return x() + width(); }
92 LayoutUnit maxY() const { return y() + height(); }
93 LayoutUnit width() const { return m_size.width(); }
94 LayoutUnit height() const { return m_size.height(); }
95
96 template<typename T> void setX(T x) { m_location.setX(x); }
97 template<typename T> void setY(T y) { m_location.setY(y); }
98 template<typename T> void setWidth(T width) { m_size.setWidth(width); }
99 template<typename T> void setHeight(T height) { m_size.setHeight(height); }
100
101 bool isEmpty() const { return m_size.isEmpty(); }
102
103 // NOTE: The result is rounded to integer values, and thus may be not the exact
104 // center point.
105 LayoutPoint center() const { return LayoutPoint(x() + width() / 2, y() + height() / 2); }
106
107 void move(const LayoutSize& size) { m_location += size; }
108 void moveBy(const LayoutPoint& offset) { m_location.move(offset.x(), offset.y()); }
109 template<typename T, typename U> void move(T dx, U dy) { m_location.move(dx, dy); }
110
111 void expand(const LayoutSize& size) { m_size += size; }
112 void expand(const LayoutBoxExtent& box)
113 {
114 m_location.move(-box.left(), -box.top());
115 m_size.expand(box.left() + box.right(), box.top() + box.bottom());
116 }
117 template<typename T, typename U> void expand(T dw, U dh) { m_size.expand(dw, dh); }
118 void contract(const LayoutSize& size) { m_size -= size; }
119 void contract(const LayoutBoxExtent& box)
120 {
121 m_location.move(box.left(), box.top());
122 m_size.shrink(box.left() + box.right(), box.top() + box.bottom());
123 }
124 template<typename T, typename U> void contract(T dw, U dh) { m_size.expand(-dw, -dh); }
125
126 void shiftXEdgeTo(LayoutUnit edge)
127 {
128 LayoutUnit delta = edge - x();
129 setX(edge);
130 setWidth(std::max<LayoutUnit>(0, width() - delta));
131 }
132
133 void shiftMaxXEdgeTo(LayoutUnit edge)
134 {
135 LayoutUnit delta = edge - maxX();
136 setWidth(std::max<LayoutUnit>(0, width() + delta));
137 }
138
139 void shiftYEdgeTo(LayoutUnit edge)
140 {
141 LayoutUnit delta = edge - y();
142 setY(edge);
143 setHeight(std::max<LayoutUnit>(0, height() - delta));
144 }
145
146 void shiftMaxYEdgeTo(LayoutUnit edge)
147 {
148 LayoutUnit delta = edge - maxY();
149 setHeight(std::max<LayoutUnit>(0, height() + delta));
150 }
151
152 void shiftXEdgeBy(LayoutUnit delta)
153 {
154 move(delta, 0);
155 setWidth(std::max<LayoutUnit>(0, width() - delta));
156 }
157
158 void shiftYEdgeBy(LayoutUnit delta)
159 {
160 move(0, delta);
161 setHeight(std::max<LayoutUnit>(0, height() - delta));
162 }
163
164 template<typename T> void shiftXEdgeTo(T edge) { shiftXEdgeTo(LayoutUnit(edge)); }
165 template<typename T> void shiftMaxXEdgeTo(T edge) { shiftMaxXEdgeTo(LayoutUnit(edge)); }
166 template<typename T> void shiftYEdgeTo(T edge) { shiftYEdgeTo(LayoutUnit(edge)); }
167 template<typename T> void shiftMaxYEdgeTo(T edge) { shiftMaxYEdgeTo(LayoutUnit(edge)); }

169 LayoutPoint minXMinYCorner() const { return m_location; } // typically topLeft
170 LayoutPoint maxXMinYCorner() const { return LayoutPoint(m_location.x() + m_size.width(), m_location.y()); } // typically topRight
171 LayoutPoint minXMaxYCorner() const { return LayoutPoint(m_location.x(), m_location.y() + m_size.height()); } // typically bottomLeft
172 LayoutPoint maxXMaxYCorner() const { return LayoutPoint(m_location.x() + m_size.width(), m_location.y() + m_size.height()); } // typically bottomRight
173 bool isMaxXMaxYRepresentable() const
174 {
175 FloatRect rect = *this;
176 float maxX = rect.maxX();
177 float maxY = rect.maxY();
178 return maxX > LayoutUnit::nearlyMin() && maxX < LayoutUnit::nearlyMax() && maxY > LayoutUnit::nearlyMin() && maxY < LayoutUnit::nearlyMax();
179 }

184 // This checks to see if the rect contains x,y in the traditional sense.
185 // Equivalent to checking if the rect contains a 1x1 rect below and to the right of (px,py).
186 bool contains(LayoutUnit px, LayoutUnit py) const
187 { return px >= x() && px < maxX() && py >= y() && py < maxY(); }
188 bool contains(const LayoutPoint& point) const { return contains(point.x(), point.y()); }

197 void inflateX(LayoutUnit dx)
198 {
199 m_location.setX(m_location.x() - dx);
200 m_size.setWidth(m_size.width() + dx + dx);
201 }
202 void inflateY(LayoutUnit dy)
203 {
204 m_location.setY(m_location.y() - dy);
205 m_size.setHeight(m_size.height() + dy + dy);
206 }
207 void inflate(LayoutSize size) { inflateX(size.width()); inflateY(size.height()); }
208 template<typename T> void inflateX(T dx) { inflateX(LayoutUnit(dx)); }
209 template<typename T> void inflateY(T dy) { inflateY(LayoutUnit(dy)); }
210 template<typename T> void inflate(T d) { inflateX(d); inflateY(d); }

218 static LayoutRect infiniteRect()
219 {
220 // Return a rect that is slightly smaller than the true max rect to allow pixelSnapping to round up to the nearest IntRect without overflowing.
221 return LayoutRect(LayoutUnit::nearlyMin() / 2, LayoutUnit::nearlyMin() / 2, LayoutUnit::nearlyMax(), LayoutUnit::nearlyMax());
222 }
223
224 operator FloatRect() const { return FloatRect(m_location, m_size); }

227 void setLocationAndSizeFromEdges(LayoutUnit left, LayoutUnit top, LayoutUnit right, LayoutUnit bottom);

229 LayoutPoint m_location;
230 LayoutSize m_size;

233inline LayoutRect intersection(const LayoutRect& a, const LayoutRect& b)
234{
235 LayoutRect c = a;
236 c.intersect(b);
237 return c;
238}
239
240inline LayoutRect unionRect(const LayoutRect& a, const LayoutRect& b)
241{
242 LayoutRect c = a;
243 c.unite(b);
244 return c;
245}

249inline bool operator==(const LayoutRect& a, const LayoutRect& b)
250{
251 return a.location() == b.location() && a.size() == b.size();
252}
253
254inline bool operator!=(const LayoutRect& a, const LayoutRect& b)
255{
256 return a.location() != b.location() || a.size() != b.size();
257}
258
259inline bool LayoutRect::isInfinite() const
260{
261 return *this == LayoutRect::infiniteRect();
262}
263
264inline void LayoutRect::setLocationAndSizeFromEdges(LayoutUnit left, LayoutUnit top, LayoutUnit right, LayoutUnit bottom)
265{
266 m_location = { left, top };
267 m_size.setWidth(right - left);
268 m_size.setHeight(bottom - top);
269}

271// Integral snapping functions.
272inline IntRect snappedIntRect(const LayoutRect& rect)
273{
274 return IntRect(roundedIntPoint(rect.location()), snappedIntSize(rect.size(), rect.location()));
275}
276
277inline IntRect snappedIntRect(LayoutUnit left, LayoutUnit top, LayoutUnit width, LayoutUnit height)
278{
279 return IntRect(IntPoint(left.round(), top.round()), snappedIntSize(LayoutSize(width, height), LayoutPoint(left, top)));
280}
281
282inline IntRect snappedIntRect(LayoutPoint location, LayoutSize size)
283{
284 return IntRect(roundedIntPoint(location), snappedIntSize(size, location));
285}

290// Device pixel snapping functions.
291inline FloatRect snapRectToDevicePixels(const LayoutRect& rect, float pixelSnappingFactor)
292{
293 return FloatRect(FloatPoint(roundToDevicePixel(rect.x(), pixelSnappingFactor), roundToDevicePixel(rect.y(), pixelSnappingFactor)), snapSizeToDevicePixel(rect.size(), rect.location(), pixelSnappingFactor));
294}
295
296inline FloatRect snapRectToDevicePixels(LayoutUnit x, LayoutUnit y, LayoutUnit width, LayoutUnit height, float pixelSnappingFactor)
297{
298 return snapRectToDevicePixels(LayoutRect(x, y, width, height), pixelSnappingFactor);
299}

301// FIXME: This needs to take vertical centering into account too.
302inline FloatRect snapRectToDevicePixelsWithWritingDirection(const LayoutRect& rect, float deviceScaleFactor, bool ltr)
303{
304 if (!ltr) {
305 FloatPoint snappedTopRight = roundPointToDevicePixels(rect.maxXMinYCorner(), deviceScaleFactor, ltr);
306 FloatSize snappedSize = snapSizeToDevicePixel(rect.size(), rect.maxXMinYCorner(), deviceScaleFactor);
307 return FloatRect(snappedTopRight.x() - snappedSize.width(), snappedTopRight.y(), snappedSize.width(), snappedSize.height());
308 }
309 return snapRectToDevicePixels(rect, deviceScaleFactor);
310}
