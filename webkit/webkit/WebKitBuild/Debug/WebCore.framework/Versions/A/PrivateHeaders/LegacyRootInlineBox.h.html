136 using CleanLineFloatList = Vector<WeakPtr<RenderBox>>;
137 void appendFloat(RenderBox& floatingBox)
138 {
139 ASSERT(!isDirty());
140 if (m_floats)
141 m_floats->append(makeWeakPtr(floatingBox));
142 else
143 m_floats = makeUnique<CleanLineFloatList>(1, makeWeakPtr(floatingBox));
144 }
145
146 void removeFloat(RenderBox& floatingBox)
147 {
148 ASSERT(m_floats);
149 ASSERT(m_floats->contains(&floatingBox));
150 m_floats->remove(m_floats->find(&floatingBox));
151 }
152
153 CleanLineFloatList* floatsPtr() { ASSERT(!isDirty()); return m_floats.get(); }
