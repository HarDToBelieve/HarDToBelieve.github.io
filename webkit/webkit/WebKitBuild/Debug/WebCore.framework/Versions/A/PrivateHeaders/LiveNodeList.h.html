40class LiveNodeList : public NodeList {
41 WTF_MAKE_ISO_NONALLOCATABLE(LiveNodeList);
42public:
43 virtual ~LiveNodeList();
44
45 virtual bool elementMatches(Element&) const = 0;
46 virtual bool isRootedAtDocument() const = 0;
47
48 NodeListInvalidationType invalidationType() const { return m_invalidationType; }
49 ContainerNode& ownerNode() const { return m_ownerNode; }
50 void invalidateCacheForAttribute(const QualifiedName& attributeName) const;
51 virtual void invalidateCacheForDocument(Document&) const = 0;
52 void invalidateCache() const { invalidateCacheForDocument(document()); }
53
54 bool isRegisteredForInvalidationAtDocument() const { return m_isRegisteredForInvalidationAtDocument; }
55 void setRegisteredForInvalidationAtDocument(bool isRegistered) { m_isRegisteredForInvalidationAtDocument = isRegistered; }
56
57protected:
58 LiveNodeList(ContainerNode& ownerNode, NodeListInvalidationType);
59
60 Document& document() const { return m_ownerNode->document(); }
61 ContainerNode& rootNode() const;
62
63private:
64 bool isLiveNodeList() const final { return true; }
65
66 Ref<ContainerNode> m_ownerNode;
67
68 const NodeListInvalidationType m_invalidationType;
69 bool m_isRegisteredForInvalidationAtDocument { false };
70};

72template <class NodeListType>
73class CachedLiveNodeList : public LiveNodeList {
74 WTF_MAKE_ISO_NONALLOCATABLE(CachedLiveNodeList);
75public:
76 virtual ~CachedLiveNodeList();
77
78 unsigned length() const final { return m_indexCache.nodeCount(nodeList()); }
79 Element* item(unsigned offset) const final { return m_indexCache.nodeAt(nodeList(), offset); }
80
81 // For CollectionIndexCache
82 using Traversal = CollectionTraversal<CollectionTraversalType::Descendants>;
83 using Iterator = Traversal::Iterator;
84 auto collectionBegin() const { return Traversal::begin(nodeList(), rootNode()); }
85 auto collectionLast() const { return Traversal::last(nodeList(), rootNode()); }
86 void collectionTraverseForward(Iterator& current, unsigned count, unsigned& traversedCount) const { Traversal::traverseForward(nodeList(), current, count, traversedCount); }
87 void collectionTraverseBackward(Iterator& current, unsigned count) const { Traversal::traverseBackward(nodeList(), current, count); }
88 bool collectionCanTraverseBackward() const { return true; }
89 void willValidateIndexCache() const { document().registerNodeListForInvalidation(const_cast<CachedLiveNodeList&>(*this)); }
90
91 void invalidateCacheForDocument(Document&) const final;
92 size_t memoryCost() const final
93 {
94 // memoryCost() may be invoked concurrently from a GC thread, and we need to be careful
95 // about what data we access here and how. Accessing m_indexCache is safe because
96 // because it doesn't involve any pointer chasing.
97 return m_indexCache.memoryCost();
98 }
99
100protected:
101 CachedLiveNodeList(ContainerNode& rootNode, NodeListInvalidationType);
102
103private:
104 NodeListType& nodeList() { return static_cast<NodeListType&>(*this); }
105 const NodeListType& nodeList() const { return static_cast<const NodeListType&>(*this); }
106
107 mutable CollectionIndexCache<NodeListType, Iterator> m_indexCache;
108};

110ALWAYS_INLINE bool shouldInvalidateTypeOnAttributeChange(NodeListInvalidationType type, const QualifiedName& attrName)
111{
112 switch (type) {
113 case InvalidateOnClassAttrChange:
114 return attrName == HTMLNames::classAttr;
115 case InvalidateOnNameAttrChange:
116 return attrName == HTMLNames::nameAttr;
117 case InvalidateOnIdNameAttrChange:
118 return attrName == HTMLNames::idAttr || attrName == HTMLNames::nameAttr;
119 case InvalidateOnForTypeAttrChange:
120 return attrName == HTMLNames::forAttr || attrName == HTMLNames::typeAttr;
121 case InvalidateForFormControls:
122 return attrName == HTMLNames::nameAttr || attrName == HTMLNames::idAttr || attrName == HTMLNames::forAttr
123 || attrName == HTMLNames::formAttr || attrName == HTMLNames::typeAttr;
124 case InvalidateOnHRefAttrChange:
125 return attrName == HTMLNames::hrefAttr;
126 case DoNotInvalidateOnAttributeChanges:
127 return false;
128 case InvalidateOnAnyAttrChange:
129 return true;
130 }
131 return false;
132}
133
134ALWAYS_INLINE void LiveNodeList::invalidateCacheForAttribute(const QualifiedName& attributeName) const
135{
136 if (shouldInvalidateTypeOnAttributeChange(m_invalidationType, attributeName))
137 invalidateCache();
138}
139
140inline ContainerNode& LiveNodeList::rootNode() const
141{
142 if (isRootedAtDocument() && m_ownerNode->isConnected())
143 return document();
144
145 return m_ownerNode;
146}

148template <class NodeListType>
149CachedLiveNodeList<NodeListType>::CachedLiveNodeList(ContainerNode& ownerNode, NodeListInvalidationType invalidationType)
150 : LiveNodeList(ownerNode, invalidationType)
151{
152}
153
154template <class NodeListType>
155CachedLiveNodeList<NodeListType>::~CachedLiveNodeList()
156{
157 if (m_indexCache.hasValidCache())
158 document().unregisterNodeListForInvalidation(*this);
159}
160
161template <class NodeListType>
162void CachedLiveNodeList<NodeListType>::invalidateCacheForDocument(Document& document) const
163{
164 if (m_indexCache.hasValidCache()) {
165 document.unregisterNodeListForInvalidation(const_cast<NodeListType&>(nodeList()));
166 m_indexCache.invalidate();
167 }
168}
