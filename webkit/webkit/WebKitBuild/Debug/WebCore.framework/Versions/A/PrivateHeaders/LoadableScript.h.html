28#include "ScriptElementCachedScriptFetcher.h"
29#include <JavaScriptCore/ConsoleTypes.h>
30#include <wtf/HashCountedSet.h>

33namespace WebCore {
34
35class LoadableScriptClient;
36class ScriptElement;
37
38class LoadableScript : public ScriptElementCachedScriptFetcher {
39public:
40 enum class ErrorType {
41 CachedScript,
42 CrossOriginLoad,
43 MIMEType,
44 Nosniff,
45 FailedIntegrityCheck,
46 };
47
48 struct ConsoleMessage {
49 MessageSource source;
50 MessageLevel level;
51 String message;
52 };
53
54 struct Error {
55 ErrorType type;
56 std::optional<ConsoleMessage> consoleMessage;
57 };
58
59 virtual ~LoadableScript() = default;
60
61 virtual bool isLoaded() const = 0;
62 virtual std::optional<Error> error() const = 0;
63 virtual bool wasCanceled() const = 0;
64
65 virtual void execute(ScriptElement&) = 0;
66
67 void addClient(LoadableScriptClient&);
68 void removeClient(LoadableScriptClient&);
69
70protected:
71 LoadableScript(const String& nonce, ReferrerPolicy policy, const String& crossOriginMode, const String& charset, const AtomString& initiatorName, bool isInUserAgentShadowTree)
72 : ScriptElementCachedScriptFetcher(nonce, policy, crossOriginMode, charset, initiatorName, isInUserAgentShadowTree)
73 {
74 }
75
76 void notifyClientFinished();
77
78private:
79 HashCountedSet<LoadableScriptClient*> m_clients;
80};
