40struct MIMETypeRegistryThreadGlobalData {
41 WTF_MAKE_NONCOPYABLE(MIMETypeRegistryThreadGlobalData);
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 MIMETypeRegistryThreadGlobalData(HashSet<String, ASCIICaseInsensitiveHash>&& supportedImageMIMETypesForEncoding)
45 : m_supportedImageMIMETypesForEncoding(supportedImageMIMETypesForEncoding)
46 { }
47
48 const HashSet<String, ASCIICaseInsensitiveHash>& supportedImageMIMETypesForEncoding() const { return m_supportedImageMIMETypesForEncoding; }
49
50private:
51 HashSet<String, ASCIICaseInsensitiveHash> m_supportedImageMIMETypesForEncoding;
52};

54class MIMETypeRegistry {
55public:
56 WEBCORE_EXPORT static String mimeTypeForExtension(const String& extension);
57 WEBCORE_EXPORT static Vector<String> extensionsForMIMEType(const String& type);
58 WEBCORE_EXPORT static String preferredExtensionForMIMEType(const String& type);
59 WEBCORE_EXPORT static String mediaMIMETypeForExtension(const String& extension);
60
61 WEBCORE_EXPORT static String mimeTypeForPath(const String& path);
62
63 static std::unique_ptr<MIMETypeRegistryThreadGlobalData> createMIMETypeRegistryThreadGlobalData();
64
65 // Check to see if a MIME type is suitable for being loaded inline as an
66 // image (e.g., <img> tags).
67 WEBCORE_EXPORT static bool isSupportedImageMIMEType(const String& mimeType);
68
69 // Check to see if a MIME type is suitable for being loaded as an image, including SVG and Video (where supported).
70 WEBCORE_EXPORT static bool isSupportedImageVideoOrSVGMIMEType(const String& mimeType);
71
72 // Check to see if a MIME type is suitable for being encoded.
73 WEBCORE_EXPORT static bool isSupportedImageMIMETypeForEncoding(const String& mimeType);
74
75 // Check to see if a MIME type is suitable for being loaded as a JavaScript or JSON resource.
76 WEBCORE_EXPORT static bool isSupportedJavaScriptMIMEType(const String& mimeType);
77 WEBCORE_EXPORT static bool isSupportedJSONMIMEType(const String& mimeType);
78
79 // Check to see if a MIME type is suitable for being loaded as a style sheet.
80 static bool isSupportedStyleSheetMIMEType(const String& mimeType);
81
82 // Check to see if a MIME type is suitable for being loaded as a font.
83 static bool isSupportedFontMIMEType(const String& mimeType);
84
85 // Check to see if a MIME type is a text media playlist type, such as an m3u8.
86 static bool isTextMediaPlaylistMIMEType(const String& mimeType);
87
88 // Check to see if a non-image MIME type is suitable for being loaded as a
89 // document in a frame. Does not include supported JavaScript and JSON MIME types.
90 WEBCORE_EXPORT static bool isSupportedNonImageMIMEType(const String& mimeType);
91
92 // Check to see if a MIME type is suitable for being loaded using <video> and <audio>.
93 WEBCORE_EXPORT static bool isSupportedMediaMIMEType(const String& mimeType);
94
95 // Check to see if a MIME type is suitable for being loaded using <track>>.
96 WEBCORE_EXPORT static bool isSupportedTextTrackMIMEType(const String& mimeType);
97
98 // Check to see if a MIME type is a valid Java applet mime type.
99 WEBCORE_EXPORT static bool isJavaAppletMIMEType(const String& mimeType);
100
101 // Check to see if a MIME type is a plugin implemented by the browser.
102 static bool isApplicationPluginMIMEType(const String& mimeType);
103
104 // Check to see if a MIME type is one of the common PDF/PS types.
105 static bool isPDFMIMEType(const String& mimeType);
106 static bool isPostScriptMIMEType(const String& mimeType);
107 WEBCORE_EXPORT static bool isPDFOrPostScriptMIMEType(const String& mimeType);
108
109 WEBCORE_EXPORT static bool isSystemPreviewMIMEType(const String& mimeType);
110
111 // Check to see if a MIME type is suitable for being shown inside a page.
112 // Returns true if any of isSupportedImageMIMEType(), isSupportedNonImageMIMEType(),
113 // isSupportedMediaMIMEType(), isSupportedJavaScriptMIMEType(), isSupportedJSONMIMEType(),
114 // returns true or if the given MIME type begins with "text/" and
115 // isUnsupportedTextMIMEType() returns false.
116 WEBCORE_EXPORT static bool canShowMIMEType(const String& mimeType);
117
118 // Check to see if a MIME type is one where an XML document should be created
119 // rather than an HTML document.
120 WEBCORE_EXPORT static bool isXMLMIMEType(const String& mimeType);
121
122 // Check to see if a MIME type is for an XML external entity resource.
123 WEBCORE_EXPORT static bool isXMLEntityMIMEType(StringView mimeType);
124
125 // Used in page load algorithm to decide whether to display as a text
126 // document in a frame. Not a good idea to use elsewhere, because that code
127 // makes this test is after many other tests are done on the MIME type.
128 WEBCORE_EXPORT static bool isTextMIMEType(const String& mimeType);
129
130 WEBCORE_EXPORT static FixedVector<const char*> supportedImageMIMETypes();
131 static HashSet<String, ASCIICaseInsensitiveHash>& additionalSupportedImageMIMETypes();
132 WEBCORE_EXPORT static HashSet<String, ASCIICaseInsensitiveHash>& supportedNonImageMIMETypes();
133 WEBCORE_EXPORT static const HashSet<String, ASCIICaseInsensitiveHash>& supportedMediaMIMETypes();
134 WEBCORE_EXPORT static FixedVector<const char*> pdfMIMETypes();
135 WEBCORE_EXPORT static FixedVector<const char*> unsupportedTextMIMETypes();
136 WEBCORE_EXPORT static FixedVector<const char*> systemPreviewMIMETypes();
137
138 WEBCORE_EXPORT static String appendFileExtensionIfNecessary(const String& filename, const String& mimeType);
139
140 WEBCORE_EXPORT static String preferredImageMIMETypeForEncoding(const Vector<String>& mimeTypes, const Vector<String>& extensions);
141 WEBCORE_EXPORT static bool containsImageMIMETypeForEncoding(const Vector<String>& mimeTypes, const Vector<String>& extensions);
142 WEBCORE_EXPORT static Vector<String> allowedMIMETypes(const Vector<String>& mimeTypes, const Vector<String>& extensions);
143 WEBCORE_EXPORT static Vector<String> allowedFileExtensions(const Vector<String>& mimeTypes, const Vector<String>& extensions);
144
145private:
146 // Check to see if the MIME type is not suitable for being loaded as a text
147 // document in a frame. Only valid for MIME types begining with "text/".
148 static bool isUnsupportedTextMIMEType(const String& mimeType);
149};
