28#include <wtf/Vector.h>
29#include <wtf/text/WTFString.h>

31namespace WebCore {
32
33class RenderBoxModelObject;
34class RenderText;
35class RenderedDocumentMarker;
36struct TextBoxSelectableRange;
37
38struct MarkedText {
39 // Sorted by paint order
40 enum Type {
41 Unmarked,
42 GrammarError,
43 Correction,
44 SpellingError,
45 TextMatch,
46 DictationAlternatives,
47 Highlight,
48#if ENABLE(APP_HIGHLIGHTS)
49 AppHighlight,
50#endif
51#if PLATFORM(IOS_FAMILY)
52 // FIXME: See <rdar://problem/8933352>. Also, remove the PLATFORM(IOS_FAMILY)-guard.
53 DictationPhraseWithAlternatives,
54#endif
55 Selection,
56 DraggedContent,
57 };
58
59 enum class PaintPhase {
60 Background,
61 Foreground,
62 Decoration
63 };
64
65 enum class OverlapStrategy {
66 None,
67 Frontmost
68 };
69
70 unsigned startOffset;
71 unsigned endOffset;
72 Type type;
73 const RenderedDocumentMarker* marker { nullptr };
74 String highlightName { };
75
76 bool isEmpty() const { return endOffset <= startOffset; }
77 bool operator!=(const MarkedText& other) const { return !(*this == other); }
78 bool operator==(const MarkedText& other) const
79 {
80 return startOffset == other.startOffset && endOffset == other.endOffset && type == other.type && marker == other.marker && highlightName == other.highlightName;
81 }
82
83 WEBCORE_EXPORT static Vector<MarkedText> subdivide(const Vector<MarkedText>&, OverlapStrategy = OverlapStrategy::None);
84
85 static Vector<MarkedText> collectForDocumentMarkers(RenderText&, const TextBoxSelectableRange&, PaintPhase);
86 static Vector<MarkedText> collectForHighlights(RenderText&, RenderBoxModelObject& parentRenderer, const TextBoxSelectableRange&, PaintPhase);
87 static Vector<MarkedText> collectForDraggedContent(RenderText&, const TextBoxSelectableRange&);
88};
