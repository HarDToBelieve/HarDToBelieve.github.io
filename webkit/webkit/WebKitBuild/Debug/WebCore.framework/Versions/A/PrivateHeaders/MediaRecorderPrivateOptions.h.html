25#pragma once
26
27#include <wtf/Forward.h>
28
29#if ENABLE(MEDIA_STREAM)
30
31namespace WebCore {
32
33struct MediaRecorderPrivateOptions {
34 String mimeType;
35 std::optional<unsigned> audioBitsPerSecond;
36 std::optional<unsigned> videoBitsPerSecond;
37 std::optional<unsigned> bitsPerSecond;
38
39 template<class Encoder> void encode(Encoder&) const;
40 template<class Decoder> static std::optional<MediaRecorderPrivateOptions> decode(Decoder&);
41};
42
43template<class Encoder>
44inline void MediaRecorderPrivateOptions::encode(Encoder& encoder) const
45{
46 encoder << mimeType;
47 encoder << audioBitsPerSecond;
48 encoder << videoBitsPerSecond;
49 encoder << bitsPerSecond;
50}
51
52template<class Decoder>
53inline std::optional<MediaRecorderPrivateOptions> MediaRecorderPrivateOptions::decode(Decoder& decoder)
54{
55 String mimeType;
56 if (!decoder.decode(mimeType))
57 return std::nullopt;
58
59 std::optional<std::optional<unsigned>> audioBitsPerSecond;
60 decoder >> audioBitsPerSecond;
61 if (!audioBitsPerSecond)
62 return std::nullopt;
63
64 std::optional<std::optional<unsigned>> videoBitsPerSecond;
65 decoder >> videoBitsPerSecond;
66 if (!videoBitsPerSecond)
67 return std::nullopt;
68
69 std::optional<std::optional<unsigned>> bitsPerSecond;
70 decoder >> bitsPerSecond;
71 if (!bitsPerSecond)
72 return std::nullopt;
73
74 return MediaRecorderPrivateOptions { WTFMove(mimeType), *audioBitsPerSecond, *videoBitsPerSecond, *bitsPerSecond };
75}
76
77} // namespace WebCore
78
79#endif // ENABLE(MEDIA_STREAM)
