26#pragma once
27
28#if ENABLE(GAMEPAD) && HAVE(MULTIGAMEPADPROVIDER_SUPPORT)
29
30#include "GamepadProvider.h"
31#include "GamepadProviderClient.h"
32#include "PlatformGamepad.h"
33#include <wtf/Forward.h>
34#include <wtf/HashSet.h>
35
36namespace WebCore {
37
38class MultiGamepadProvider : public GamepadProvider, public GamepadProviderClient {
39public:
40 virtual ~MultiGamepadProvider() = default;
41
42 WEBCORE_EXPORT static MultiGamepadProvider& singleton();
43
44 void setUsesOnlyHIDGamepadProvider(bool hidProviderOnly) { m_usesOnlyHIDProvider = hidProviderOnly; }
45
46 // GamepadProvider
47 void startMonitoringGamepads(GamepadProviderClient&) final;
48 void stopMonitoringGamepads(GamepadProviderClient&) final;
49 const Vector<PlatformGamepad*>& platformGamepads() final { return m_gamepadVector; }
50 bool isMockGamepadProvider() const { return false; }
51
52 // GamepadProviderClient
53 void platformGamepadConnected(PlatformGamepad&, EventMakesGamepadsVisible) final;
54 void platformGamepadDisconnected(PlatformGamepad&) final;
55 void platformGamepadInputActivity(EventMakesGamepadsVisible) final;
56
57protected:
58 WEBCORE_EXPORT void dispatchPlatformGamepadInputActivity();
59
60private:
61 unsigned indexForNewlyConnectedDevice();
62
63 bool m_shouldMakeGamepadsVisible { false };
64 size_t m_initialGamepadsCount { 0 };
65 Vector<PlatformGamepad*> m_gamepadVector;
66
67 // We create our own Gamepad type - to wrap both HID and GameController gamepads -
68 // because MultiGamepadProvider needs to manage the indexes of its own gamepads
69 // no matter what the HID or GameController index is.
70 class PlatformGamepadWrapper : public PlatformGamepad {
71 public:
72 PlatformGamepadWrapper(unsigned index, PlatformGamepad* wrapped)
73 : PlatformGamepad(index)
74 , m_platformGamepad(wrapped)
75 {
76 m_id = wrapped->id();
77 m_mapping = wrapped->mapping();
78 m_connectTime = wrapped->connectTime();
79 }
80
81 MonotonicTime lastUpdateTime() const final { return m_platformGamepad->lastUpdateTime(); }
82 const Vector<SharedGamepadValue>& axisValues() const final { return m_platformGamepad->axisValues(); }
83 const Vector<SharedGamepadValue>& buttonValues() const final { return m_platformGamepad->buttonValues(); }
84
85 const char* source() const final { return m_platformGamepad->source(); }
86
87 private:
88 PlatformGamepad* m_platformGamepad;
89 };
90
91 HashMap<PlatformGamepad*, std::unique_ptr<PlatformGamepadWrapper>> m_gamepadMap;
92 bool m_hidImportComplete { false };
93 bool m_usesOnlyHIDProvider { false };
94};
95
96} // namespace WebCore
97
98#endif // ENABLE(GAMEPAD) && HAVE(MULTIGAMEPADPROVIDER_SUPPORT)
