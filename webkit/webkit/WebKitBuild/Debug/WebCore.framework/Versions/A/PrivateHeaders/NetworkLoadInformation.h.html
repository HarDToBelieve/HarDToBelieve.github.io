28#include "NetworkLoadMetrics.h"

33namespace WebCore {
34
35struct NetworkTransactionInformation {
36 enum class Type : bool { Redirection, Preflight };
37 Type type;
38 ResourceRequest request;
39 ResourceResponse response;
40 NetworkLoadMetrics metrics;
41
42 template<class Encoder> void encode(Encoder&) const;
43 template<class Decoder> static std::optional<NetworkTransactionInformation> decode(Decoder&);
44};
45
46struct NetworkLoadInformation {
47 WTF_MAKE_STRUCT_FAST_ALLOCATED;
48 ResourceRequest request;
49 ResourceResponse response;
50 NetworkLoadMetrics metrics;
51 Vector<NetworkTransactionInformation> transactions;
52};
53
54template<class Encoder> inline void NetworkTransactionInformation::encode(Encoder& encoder) const
55{
56 encoder << type;
57 encoder << request;
58 encoder << response;
59 encoder << metrics;
60}
61
62template<class Decoder> inline std::optional<NetworkTransactionInformation> NetworkTransactionInformation::decode(Decoder& decoder)
63{
64 NetworkTransactionInformation information;
65
66 if (!decoder.decode(information.type))
67 return std::nullopt;
68 if (!decoder.decode(information.request))
69 return std::nullopt;
70 if (!decoder.decode(information.response))
71 return std::nullopt;
72 if (!decoder.decode(information.metrics))
73 return std::nullopt;
74
75 return information;
76}
