25#pragma once
26
27#include "BorderValue.h"
28
29namespace WebCore {
30
31class OutlineValue : public BorderValue {
32friend class RenderStyle;
33public:
34 bool operator==(const OutlineValue& o) const
35 {
36 return m_width == o.m_width && m_style == o.m_style && m_color == o.m_color && m_offset == o.m_offset && m_isAuto == o.m_isAuto;
37 }
38
39 bool operator!=(const OutlineValue& o) const
40 {
41 return !(*this == o);
42 }
43
44 float offset() const { return m_offset; }
45 OutlineIsAuto isAuto() const { return static_cast<OutlineIsAuto>(m_isAuto); }
46
47private:
48 float m_offset { 0 };
49};
50
51} // namespace WebCore
