26#pragma once
27
28#include "Event.h"
29
30namespace WebCore {
31
32class OverflowEvent final : public Event {
33 WTF_MAKE_ISO_ALLOCATED(OverflowEvent);
34public:
35 enum orientType {
36 HORIZONTAL = 0,
37 VERTICAL = 1,
38 BOTH = 2
39 };
40
41 static Ref<OverflowEvent> create(bool horizontalOverflowChanged, bool horizontalOverflow, bool verticalOverflowChanged, bool verticalOverflow)
42 {
43 return adoptRef(*new OverflowEvent(horizontalOverflowChanged, horizontalOverflow, verticalOverflowChanged, verticalOverflow));
44 }
45
46 static Ref<OverflowEvent> createForBindings()
47 {
48 return adoptRef(*new OverflowEvent);
49 }
50
51 struct Init : EventInit {
52 unsigned short orient { 0 };
53 bool horizontalOverflow { false };
54 bool verticalOverflow { false };
55 };
56
57 static Ref<OverflowEvent> create(const AtomString& type, const Init& initializer, IsTrusted isTrusted = IsTrusted::No)
58 {
59 return adoptRef(*new OverflowEvent(type, initializer, isTrusted));
60 }
61
62 WEBCORE_EXPORT void initOverflowEvent(unsigned short orient, bool horizontalOverflow, bool verticalOverflow);
63
64 unsigned short orient() const { return m_orient; }
65 bool horizontalOverflow() const { return m_horizontalOverflow; }
66 bool verticalOverflow() const { return m_verticalOverflow; }
67
68 EventInterface eventInterface() const override;
69
70private:
71 OverflowEvent();
72 OverflowEvent(bool horizontalOverflowChanged, bool horizontalOverflow, bool verticalOverflowChanged, bool verticalOverflow);
73 OverflowEvent(const AtomString&, const Init&, IsTrusted);
74
75 unsigned short m_orient;
76 bool m_horizontalOverflow;
77 bool m_verticalOverflow;
78};
79
80} // namespace WebCore
