51#endif // PLATFORM(IOS_FAMILY)

54 virtual RefPtr<SharedBuffer> bufferForType(const String& pasteboardType, const String& pasteboardName, const PasteboardContext*) = 0;

57 virtual Vector<String> allStringsForType(const String& pasteboardType, const String& pasteboardName, const PasteboardContext*) = 0;
58 virtual int64_t changeCount(const String& pasteboardName, const PasteboardContext*) = 0;
59 virtual Color color(const String& pasteboardName, const PasteboardContext*) = 0;
60 virtual URL url(const String& pasteboardName, const PasteboardContext*) = 0;

66 virtual int64_t setURL(const PasteboardURL&, const String& pasteboardName, const PasteboardContext*) = 0;
67 virtual int64_t setColor(const Color&, const String& pasteboardName, const PasteboardContext*) = 0;

69
70 virtual bool containsURLStringSuitableForLoading(const String& pasteboardName, const PasteboardContext*) = 0;
71 virtual String urlStringSuitableForLoading(const String& pasteboardName, String& title, const PasteboardContext*) = 0;
72#endif
73 virtual String readStringFromPasteboard(size_t index, const String& pasteboardType, const String& pasteboardName, const PasteboardContext*) = 0;

75 virtual URL readURLFromPasteboard(size_t index, const String& pasteboardName, String& title, const PasteboardContext*) = 0;
76 virtual std::optional<PasteboardItemInfo> informationForItemAtIndex(size_t index, const String& pasteboardName, int64_t changeCount, const PasteboardContext*) = 0;
77 virtual std::optional<Vector<PasteboardItemInfo>> allPasteboardItemInfo(const String& pasteboardName, int64_t changeCount, const PasteboardContext*) = 0;
78 virtual int getPasteboardItemsCount(const String& pasteboardName, const PasteboardContext*) = 0;
79
80 virtual Vector<String> typesSafeForDOMToReadAndWrite(const String& pasteboardName, const String& origin, const PasteboardContext*) = 0;

82 virtual bool containsStringSafeForDOMToReadForType(const String&, const String& pasteboardName, const PasteboardContext*) = 0;
83
84#if PLATFORM(GTK)
