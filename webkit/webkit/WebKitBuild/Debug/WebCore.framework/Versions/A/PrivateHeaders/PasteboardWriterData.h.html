28#include <wtf/URL.h>
29#include <wtf/text/WTFString.h>

31namespace WebCore {
32
33class SharedBuffer;
34
35class PasteboardWriterData final {
36public:
37 WEBCORE_EXPORT PasteboardWriterData();
38 WEBCORE_EXPORT ~PasteboardWriterData();
39
40 WEBCORE_EXPORT bool isEmpty() const;
41
42 struct PlainText {
43 bool canSmartCopyOrDelete;
44 String text;
45 };
46
47 struct WebContent {
48 WebContent();
49 WEBCORE_EXPORT ~WebContent();
50
51#if PLATFORM(COCOA)
52 String contentOrigin;
53 bool canSmartCopyOrDelete;
54 RefPtr<SharedBuffer> dataInWebArchiveFormat;
55 RefPtr<SharedBuffer> dataInRTFDFormat;
56 RefPtr<SharedBuffer> dataInRTFFormat;
57 RefPtr<SharedBuffer> dataInAttributedStringFormat;
58 String dataInHTMLFormat;
59 String dataInStringFormat;
60 Vector<String> clientTypes;
61 Vector<RefPtr<SharedBuffer>> clientData;
62#endif
63 };
64
65 const std::optional<PlainText>& plainText() const { return m_plainText; }
66 void setPlainText(PlainText);
67
68 struct URLData {
69 URL url;
70 String title;
71#if PLATFORM(MAC)
72 String userVisibleForm;
73#elif PLATFORM(GTK)
74 String markup;
75#endif
76 };
77
78 const std::optional<URLData>& urlData() const { return m_url; }
79 void setURLData(URLData);
80
81 const std::optional<WebContent>& webContent() const { return m_webContent; }
82 void setWebContent(WebContent);
83
84private:
85 std::optional<PlainText> m_plainText;
86 std::optional<URLData> m_url;
87 std::optional<WebContent> m_webContent;
88};
