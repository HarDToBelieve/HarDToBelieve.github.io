38class Document;
39class PaymentMerchantSession;
40struct ApplePayCouponCodeUpdate;
41struct ApplePayPaymentMethodUpdate;
42struct ApplePaySetupConfiguration;

45struct PaymentAuthorizationResult;

47class PaymentCoordinatorClient {
48public:
49 bool supportsVersion(unsigned version);
50
51 virtual std::optional<String> validatedPaymentNetwork(const String&) = 0;
52 virtual bool canMakePayments() = 0;
53 virtual void canMakePaymentsWithActiveCard(const String& merchantIdentifier, const String& domainName, CompletionHandler<void(bool)>&&) = 0;
54 virtual void openPaymentSetup(const String& merchantIdentifier, const String& domainName, CompletionHandler<void(bool)>&&) = 0;
55
56 virtual bool showPaymentUI(const URL& originatingURL, const Vector<URL>& linkIconURLs, const ApplePaySessionPaymentRequest&) = 0;
57 virtual void completeMerchantValidation(const PaymentMerchantSession&) = 0;
58 virtual void completeShippingMethodSelection(std::optional<ApplePayShippingMethodUpdate>&&) = 0;
59 virtual void completeShippingContactSelection(std::optional<ApplePayShippingContactUpdate>&&) = 0;
60 virtual void completePaymentMethodSelection(std::optional<ApplePayPaymentMethodUpdate>&&) = 0;
61#if ENABLE(APPLE_PAY_COUPON_CODE)
62 virtual void completeCouponCodeChange(std::optional<ApplePayCouponCodeUpdate>&&) = 0;
63#endif
64 virtual void completePaymentSession(std::optional<PaymentAuthorizationResult>&&) = 0;
65 virtual void abortPaymentSession() = 0;
66 virtual void cancelPaymentSession() = 0;
67 virtual void paymentCoordinatorDestroyed() = 0;
68 virtual bool supportsUnrestrictedApplePay() const = 0;
69
70 virtual String userAgentScriptsBlockedErrorMessage() const { return { }; }
71
72 virtual bool isMockPaymentCoordinator() const { return false; }
73 virtual bool isWebPaymentCoordinator() const { return false; }
74
75 virtual void getSetupFeatures(const ApplePaySetupConfiguration&, const URL&, CompletionHandler<void(Vector<Ref<ApplePaySetupFeature>>&&)>&& completionHandler) { completionHandler({ }); }
76 virtual void beginApplePaySetup(const ApplePaySetupConfiguration&, const URL&, Vector<RefPtr<ApplePaySetupFeature>>&&, CompletionHandler<void(bool)>&& completionHandler) { completionHandler(false); }
77 virtual void endApplePaySetup() { }
78
79protected:
80 virtual ~PaymentCoordinatorClient() = default;
81};
