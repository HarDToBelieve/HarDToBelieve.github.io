26#pragma once
27
28#include "AlphaPremultiplication.h"
29#include "DestinationColorSpace.h"
30#include "PixelFormat.h"
31#include <wtf/Forward.h>
32
33namespace WebCore {
34
35struct PixelBufferFormat {
36 AlphaPremultiplication alphaFormat;
37 PixelFormat pixelFormat;
38 DestinationColorSpace colorSpace;
39
40 template<class Encoder> void encode(Encoder&) const;
41 template<class Decoder> static std::optional<PixelBufferFormat> decode(Decoder&);
42};
43
44WEBCORE_EXPORT TextStream& operator<<(TextStream&, const PixelBufferFormat&);
45
46template<class Encoder> void PixelBufferFormat::encode(Encoder& encoder) const
47{
48 encoder << alphaFormat << pixelFormat << colorSpace;
49}
50
51template<class Decoder> std::optional<PixelBufferFormat> PixelBufferFormat::decode(Decoder& decoder)
52{
53 std::optional<AlphaPremultiplication> alphaFormat;
54 decoder >> alphaFormat;
55 if (!alphaFormat)
56 return std::nullopt;
57
58 std::optional<PixelFormat> pixelFormat;
59 decoder >> pixelFormat;
60 if (!pixelFormat)
61 return std::nullopt;
62
63 std::optional<DestinationColorSpace> colorSpace;
64 decoder >> colorSpace;
65 if (!colorSpace)
66 return std::nullopt;
67
68 return { { WTFMove(*alphaFormat), WTFMove(*pixelFormat), WTFMove(*colorSpace) } };
69}
