38namespace WebCore {
39
40class FloatRect;
41class TimingFunction;

43class PlatformCAAnimation : public RefCounted<PlatformCAAnimation> {
44public:
45 enum AnimationType { Basic, Group, Keyframe, Spring };
46 enum FillModeType { NoFillMode, Forwards, Backwards, Both };
47 enum ValueFunctionType { NoValueFunction, RotateX, RotateY, RotateZ, ScaleX, ScaleY, ScaleZ, Scale, TranslateX, TranslateY, TranslateZ, Translate };
48
49 virtual ~PlatformCAAnimation() = default;
50
51 virtual bool isPlatformCAAnimationCocoa() const { return false; }
52 virtual bool isPlatformCAAnimationWin() const { return false; }
53 virtual bool isPlatformCAAnimationRemote() const { return false; }
54
55 virtual Ref<PlatformCAAnimation> copy() const = 0;
56
57 AnimationType animationType() const { return m_type; }
58 virtual String keyPath() const = 0;
59
60 virtual CFTimeInterval beginTime() const = 0;
61 virtual void setBeginTime(CFTimeInterval) = 0;
62
63 virtual CFTimeInterval duration() const = 0;
64 virtual void setDuration(CFTimeInterval) = 0;
65
66 virtual float speed() const = 0;
67 virtual void setSpeed(float) = 0;
68
69 virtual CFTimeInterval timeOffset() const = 0;
70 virtual void setTimeOffset(CFTimeInterval) = 0;
71
72 virtual float repeatCount() const = 0;
73 virtual void setRepeatCount(float) = 0;
74
75 virtual bool autoreverses() const = 0;
76 virtual void setAutoreverses(bool) = 0;
77
78 virtual FillModeType fillMode() const = 0;
79 virtual void setFillMode(FillModeType) = 0;
80
81 virtual void setTimingFunction(const TimingFunction*, bool reverse = false) = 0;
82 virtual void copyTimingFunctionFrom(const PlatformCAAnimation&) = 0;
83
84 virtual bool isRemovedOnCompletion() const = 0;
85 virtual void setRemovedOnCompletion(bool) = 0;
86
87 virtual bool isAdditive() const = 0;
88 virtual void setAdditive(bool) = 0;
89
90 virtual ValueFunctionType valueFunction() const = 0;
91 virtual void setValueFunction(ValueFunctionType) = 0;
92
93 // Basic-animation properties.
94 virtual void setFromValue(float) = 0;
95 virtual void setFromValue(const WebCore::TransformationMatrix&) = 0;
96 virtual void setFromValue(const FloatPoint3D&) = 0;
97 virtual void setFromValue(const WebCore::Color&) = 0;
98 virtual void setFromValue(const FilterOperation*, int internalFilterPropertyIndex) = 0;
99 virtual void copyFromValueFrom(const PlatformCAAnimation&) = 0;
100
101 virtual void setToValue(float) = 0;
102 virtual void setToValue(const WebCore::TransformationMatrix&) = 0;
103 virtual void setToValue(const FloatPoint3D&) = 0;
104 virtual void setToValue(const WebCore::Color&) = 0;
105 virtual void setToValue(const FilterOperation*, int internalFilterPropertyIndex) = 0;
106 virtual void copyToValueFrom(const PlatformCAAnimation&) = 0;
107
108 // Keyframe-animation properties.
109 virtual void setValues(const Vector<float>&) = 0;
110 virtual void setValues(const Vector<WebCore::TransformationMatrix>&) = 0;
111 virtual void setValues(const Vector<FloatPoint3D>&) = 0;
112 virtual void setValues(const Vector<WebCore::Color>&) = 0;
113 virtual void setValues(const Vector<RefPtr<FilterOperation>>&, int internalFilterPropertyIndex) = 0;
114 virtual void copyValuesFrom(const PlatformCAAnimation&) = 0;
115
116 virtual void setKeyTimes(const Vector<float>&) = 0;
117 virtual void copyKeyTimesFrom(const PlatformCAAnimation&) = 0;
118
119 virtual void setTimingFunctions(const Vector<const TimingFunction*>&, bool reverse = false) = 0;
120 virtual void copyTimingFunctionsFrom(const PlatformCAAnimation&) = 0;
121
122 // Animation group properties.
123 virtual void setAnimations(const Vector<RefPtr<PlatformCAAnimation>>&) = 0;
124 virtual void copyAnimationsFrom(const PlatformCAAnimation&) = 0;
125
126 void setActualStartTimeIfNeeded(CFTimeInterval t)
127 {
128 if (beginTime() <= 0)
129 setBeginTime(t);
130 }
131
132 bool isBasicAnimation() const;
133
134protected:
135 PlatformCAAnimation(AnimationType type = Basic)
136 : m_type(type)
137 {
138 }
139
140 void setType(AnimationType type) { m_type = type; }
141
142private:
143 AnimationType m_type;
144};
