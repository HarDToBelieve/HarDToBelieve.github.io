26#pragma once
27
28#include <wtf/OptionSet.h>
29#include <wtf/WallTime.h>
30
31namespace WebCore {
32
33enum class EventHandling : uint8_t {
34 DispatchedToDOM = 1 << 0,
35 DefaultPrevented = 1 << 1,
36 DefaultHandled = 1 << 2,
37};
38
39class PlatformEvent {
40public:
41 enum Type : uint8_t {
42 NoType = 0,
43
44 // PlatformKeyboardEvent
45 KeyDown,
46 KeyUp,
47 RawKeyDown,
48 Char,
49
50 // PlatformMouseEvent
51 MouseMoved,
52 MousePressed,
53 MouseReleased,
54 MouseForceChanged,
55 MouseForceDown,
56 MouseForceUp,
57 MouseScroll,
58
59 // PlatformWheelEvent
60 Wheel,
61
62#if ENABLE(TOUCH_EVENTS)
63 // PlatformTouchEvent
64 TouchStart,
65 TouchMove,
66 TouchEnd,
67 TouchCancel,
68 TouchForceChange,
69#endif
70
71#if ENABLE(MAC_GESTURE_EVENTS)
72 // PlatformGestureEvent
73 GestureStart,
74 GestureChange,
75 GestureEnd,
76#endif
77 };
78
79 enum class Modifier : uint8_t {
80 AltKey = 1 << 0,
81 ControlKey = 1 << 1,
82 MetaKey = 1 << 2,
83 ShiftKey = 1 << 3,
84 CapsLockKey = 1 << 4,
85
86 // Never used in native platforms but added for initEvent
87 AltGraphKey = 1 << 5,
88 };
89
90 Type type() const { return static_cast<Type>(m_type); }
91
92 bool shiftKey() const { return m_modifiers.contains(Modifier::ShiftKey); }
93 bool controlKey() const { return m_modifiers.contains(Modifier::ControlKey); }
94 bool altKey() const { return m_modifiers.contains(Modifier::AltKey); }
95 bool metaKey() const { return m_modifiers.contains(Modifier::MetaKey); }
96
97 OptionSet<Modifier> modifiers() const { return m_modifiers; }
98
99 WallTime timestamp() const { return m_timestamp; }
100
101protected:
102 PlatformEvent()
103 : m_type(NoType)
104 {
105 }
106
107 explicit PlatformEvent(Type type)
108 : m_type(type)
109 {
110 }
111
112 PlatformEvent(Type type, OptionSet<Modifier> modifiers, WallTime timestamp)
113 : m_timestamp(timestamp)
114 , m_type(type)
115 , m_modifiers(modifiers)
116 {
117 }
118
119 PlatformEvent(Type type, bool shiftKey, bool ctrlKey, bool altKey, bool metaKey, WallTime timestamp)
120 : m_timestamp(timestamp)
121 , m_type(type)
122 {
123 if (shiftKey)
124 m_modifiers.add(Modifier::ShiftKey);
125 if (ctrlKey)
126 m_modifiers.add(Modifier::ControlKey);
127 if (altKey)
128 m_modifiers.add(Modifier::AltKey);
129 if (metaKey)
130 m_modifiers.add(Modifier::MetaKey);
131 }
132
133 // Explicit protected destructor so that people don't accidentally
134 // delete a PlatformEvent.
135 ~PlatformEvent() = default;
136
137 WallTime m_timestamp;
138 unsigned m_type;
139 OptionSet<Modifier> m_modifiers;
140};
141
142} // namespace WebCore
