286template<class Encoder> inline void PlatformMediaSession::RemoteCommandArgument::encode(Encoder& encoder) const
287{
288 encoder << time << fastSeek;
289}
290
291template<class Decoder> inline std::optional<PlatformMediaSession::RemoteCommandArgument> PlatformMediaSession::RemoteCommandArgument::decode(Decoder& decoder)
292{
293#define DECODE(name, type) \
294 std::optional<std::optional<type>> name; \
295 decoder >> name; \
296 if (!name) \
297 return std::nullopt; \
298
299 DECODE(time, double);
300 DECODE(fastSeek, bool);
301
302#undef DECODE
303
304 return { { *time, *fastSeek } };
305}
