35const double ForceAtClick = 1;
36const double ForceAtForceClick = 2;
37
38 // These button numbers match the ones used in the DOM API, 0 through 2, except for NoButton which isn't specified.
39 // We use -2 for NoButton because -1 is a valid value in the DOM API for Pointer Events for pointermove events that
40 // indicate that the pressed mouse button hasn't changed since the last event.
41 enum MouseButton : int8_t { LeftButton = 0, MiddleButton, RightButton, NoButton = -2 };
42 enum SyntheticClickType : int8_t { NoTap, OneFingerTap, TwoFingerTap };
43
44 class PlatformMouseEvent : public PlatformEvent {
45 public:
46 PlatformMouseEvent()
47 : PlatformEvent(PlatformEvent::MouseMoved)
48 {
49 }
50
51 PlatformMouseEvent(const IntPoint& position, const IntPoint& globalPosition, MouseButton button, PlatformEvent::Type type,
52 int clickCount, bool shiftKey, bool ctrlKey, bool altKey, bool metaKey, WallTime timestamp, double force, SyntheticClickType syntheticClickType, PointerID pointerId = mousePointerID)
53 : PlatformEvent(type, shiftKey, ctrlKey, altKey, metaKey, timestamp)
54 , m_button(button)
55 , m_syntheticClickType(syntheticClickType)
56 , m_position(position)
57 , m_globalPosition(globalPosition)
58 , m_force(force)
59 , m_pointerId(pointerId)
60 , m_clickCount(clickCount)
61 {
62 }
63
64 // This position is relative to the enclosing NSWindow in WebKit1, and is WKWebView-relative in WebKit2.
65 // Use ScrollView::windowToContents() to convert it to into the contents of a given view.
66 const IntPoint& position() const { return m_position; }
67 const IntPoint& globalPosition() const { return m_globalPosition; }
68 const IntPoint& movementDelta() const { return m_movementDelta; }
69
70 MouseButton button() const { return m_button; }
71 unsigned short buttons() const { return m_buttons; }
72 int clickCount() const { return m_clickCount; }
73 unsigned modifierFlags() const { return m_modifierFlags; }
74 double force() const { return m_force; }
75 SyntheticClickType syntheticClickType() const { return m_syntheticClickType; }
76 PointerID pointerId() const { return m_pointerId; }
77 const String& pointerType() const { return m_pointerType; }
78
79#if PLATFORM(MAC)
80 int eventNumber() const { return m_eventNumber; }
81 int menuTypeForEvent() const { return m_menuTypeForEvent; }
82#endif
83
84#if PLATFORM(WIN)
85 PlatformMouseEvent(HWND, UINT, WPARAM, LPARAM, bool didActivateWebView = false);
86 void setClickCount(int count) { m_clickCount = count; }
87 bool didActivateWebView() const { return m_didActivateWebView; }
88#endif
89
90 protected:
91 MouseButton m_button { NoButton };
92 SyntheticClickType m_syntheticClickType { NoTap };
93
94 IntPoint m_position;
95 IntPoint m_globalPosition;
96 IntPoint m_movementDelta;
97 double m_force { 0 };
98 PointerID m_pointerId { mousePointerID };
99 String m_pointerType { "mouse"_s };
100 int m_clickCount { 0 };
101 unsigned m_modifierFlags { 0 };
102 unsigned short m_buttons { 0 };
103#if PLATFORM(MAC)
104 int m_eventNumber { 0 };
105 int m_menuTypeForEvent { 0 };
106#elif PLATFORM(WIN)
107 bool m_didActivateWebView { false };
108#endif
109 };
110
111#if COMPILER(MSVC)
112 // These functions are necessary to work around the fact that MSVC will not find a most-specific
113 // operator== to use after implicitly converting MouseButton to a short.
114 inline bool operator==(short a, MouseButton b)
115 {
116 return a == static_cast<short>(b);
117 }
118
119 inline bool operator!=(short a, MouseButton b)
120 {
121 return a != static_cast<short>(b);
122 }
123#endif
