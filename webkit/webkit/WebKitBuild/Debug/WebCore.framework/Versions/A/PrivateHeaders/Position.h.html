273// FIXME: Positions at the same document location with different anchoring will return false; that's unlike <= and >=.
274// FIXME: This ignores differences in m_isLegacyEditingPosition in a subtle way.
275// FIXME: For legacy editing positions, <div><img></div> [div, 0] != [img, 0] even though most of editing code will treat them as identical.
276inline bool operator==(const Position& a, const Position& b)
277{
278 return a.anchorNode() == b.anchorNode() && a.deprecatedEditingOffset() == b.deprecatedEditingOffset() && a.anchorType() == b.anchorType();
279}

319// firstPositionInNode and lastPositionInNode return parent-anchored positions, lastPositionInNode construction is O(n) due to countChildNodes()
320inline Position firstPositionInNode(Node* anchorNode)
321{
322 if (anchorNode->isTextNode())
323 return Position(anchorNode, 0, Position::PositionIsOffsetInAnchor);
324 return Position(anchorNode, Position::PositionIsBeforeChildren);
325}
326
327inline Position lastPositionInNode(Node* anchorNode)
328{
329 if (anchorNode->isTextNode())
330 return Position(anchorNode, anchorNode->length(), Position::PositionIsOffsetInAnchor);
331 return Position(anchorNode, Position::PositionIsAfterChildren);
332}

334inline bool offsetIsBeforeLastNodeOffset(unsigned offset, Node* anchorNode)
335{
336 if (is<CharacterData>(*anchorNode))
337 return offset < downcast<CharacterData>(*anchorNode).length();
338
339 unsigned currentOffset = 0;
340 for (Node* node = anchorNode->firstChild(); node && currentOffset < offset; node = node->nextSibling())
341 currentOffset++;
342 return offset < currentOffset;
343}
