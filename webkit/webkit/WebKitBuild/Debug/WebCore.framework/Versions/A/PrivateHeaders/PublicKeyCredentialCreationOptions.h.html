38namespace WebCore {
39
40enum class AuthenticatorAttachment;
41
42struct PublicKeyCredentialCreationOptions {
43#if ENABLE(WEB_AUTHN)
44 struct Entity {
45 String name;
46 String icon;
47 };
48
49 struct RpEntity : public Entity {
50 mutable String id;
51 };
52
53 struct UserEntity : public Entity {
54 BufferSource id; // id becomes idVector once it is passed to UIProcess.
55 Vector<uint8_t> idVector;
56 String displayName;
57 };
58
59 struct Parameters {
60 PublicKeyCredentialType type;
61 int64_t alg;
62
63 template<class Encoder> void encode(Encoder&) const;
64 template<class Decoder> static std::optional<Parameters> decode(Decoder&);
65 };
66
67 struct AuthenticatorSelectionCriteria {
68 std::optional<AuthenticatorAttachment> authenticatorAttachment;
69 bool requireResidentKey { false };
70 UserVerificationRequirement userVerification { UserVerificationRequirement::Preferred };
71
72 template<class Encoder> void encode(Encoder&) const;
73 template<class Decoder> static std::optional<AuthenticatorSelectionCriteria> decode(Decoder&);
74 };
75
76 RpEntity rp;
77 UserEntity user;
78
79 BufferSource challenge;
80 Vector<Parameters> pubKeyCredParams;
81
82 std::optional<unsigned> timeout;
83 Vector<PublicKeyCredentialDescriptor> excludeCredentials;
84 std::optional<AuthenticatorSelectionCriteria> authenticatorSelection;
85 AttestationConveyancePreference attestation;
86 mutable std::optional<AuthenticationExtensionsClientInputs> extensions;
87
88 template<class Encoder> void encode(Encoder&) const;
89 template<class Decoder> static std::optional<PublicKeyCredentialCreationOptions> decode(Decoder&);
90#endif // ENABLE(WEB_AUTHN)
91};
92
93#if ENABLE(WEB_AUTHN)
94template<class Encoder>
95void PublicKeyCredentialCreationOptions::Parameters::encode(Encoder& encoder) const
96{
97 encoder << type << alg;
98}
99
100template<class Decoder>
101std::optional<PublicKeyCredentialCreationOptions::Parameters> PublicKeyCredentialCreationOptions::Parameters::decode(Decoder& decoder)
102{
103 PublicKeyCredentialCreationOptions::Parameters result;
104 if (!decoder.decode(result.type))
105 return std::nullopt;
106 if (!decoder.decode(result.alg))
107 return std::nullopt;
108 return result;
109}
110
111template<class Encoder>
112void PublicKeyCredentialCreationOptions::AuthenticatorSelectionCriteria::encode(Encoder& encoder) const
113{
114 encoder << authenticatorAttachment << requireResidentKey << userVerification;
115}
116
117template<class Decoder>
118std::optional<PublicKeyCredentialCreationOptions::AuthenticatorSelectionCriteria> PublicKeyCredentialCreationOptions::AuthenticatorSelectionCriteria::decode(Decoder& decoder)
119{
120 PublicKeyCredentialCreationOptions::AuthenticatorSelectionCriteria result;
121
122 std::optional<std::optional<AuthenticatorAttachment>> authenticatorAttachment;
123 decoder >> authenticatorAttachment;
124 if (!authenticatorAttachment)
125 return std::nullopt;
126 result.authenticatorAttachment = WTFMove(*authenticatorAttachment);
127
128 std::optional<bool> requireResidentKey;
129 decoder >> requireResidentKey;
130 if (!requireResidentKey)
131 return std::nullopt;
132 result.requireResidentKey = *requireResidentKey;
133
134 if (!decoder.decode(result.userVerification))
135 return std::nullopt;
136 return result;
137}
138
139// Not every member is encoded.
140template<class Encoder>
141void PublicKeyCredentialCreationOptions::encode(Encoder& encoder) const
142{
143 encoder << rp.id << rp.name << rp.icon;
144 encoder << static_cast<uint64_t>(user.id.length());
145 encoder.encodeFixedLengthData(user.id.data(), user.id.length(), 1);
146 encoder << user.displayName << user.name << user.icon << pubKeyCredParams << timeout << excludeCredentials << authenticatorSelection << attestation << extensions;
147}
148
149template<class Decoder>
150std::optional<PublicKeyCredentialCreationOptions> PublicKeyCredentialCreationOptions::decode(Decoder& decoder)
151{
152 PublicKeyCredentialCreationOptions result;
153 if (!decoder.decode(result.rp.id))
154 return std::nullopt;
155 if (!decoder.decode(result.rp.name))
156 return std::nullopt;
157 if (!decoder.decode(result.rp.icon))
158 return std::nullopt;
159 if (!decoder.decode(result.user.idVector))
160 return std::nullopt;
161 if (!decoder.decode(result.user.displayName))
162 return std::nullopt;
163 if (!decoder.decode(result.user.name))
164 return std::nullopt;
165 if (!decoder.decode(result.user.icon))
166 return std::nullopt;
167 if (!decoder.decode(result.pubKeyCredParams))
168 return std::nullopt;
169
170 std::optional<std::optional<unsigned>> timeout;
171 decoder >> timeout;
172 if (!timeout)
173 return std::nullopt;
174 result.timeout = WTFMove(*timeout);
175
176 if (!decoder.decode(result.excludeCredentials))
177 return std::nullopt;
178
179 std::optional<std::optional<AuthenticatorSelectionCriteria>> authenticatorSelection;
180 decoder >> authenticatorSelection;
181 if (!authenticatorSelection)
182 return std::nullopt;
183 result.authenticatorSelection = WTFMove(*authenticatorSelection);
184
185 std::optional<AttestationConveyancePreference> attestation;
186 decoder >> attestation;
187 if (!attestation)
188 return std::nullopt;
189 result.attestation = WTFMove(*attestation);
190
191 std::optional<std::optional<AuthenticationExtensionsClientInputs>> extensions;
192 decoder >> extensions;
193 if (!extensions)
194 return std::nullopt;
195 result.extensions = WTFMove(*extensions);
196
197 return result;
198}
199#endif // ENABLE(WEB_AUTHN)
