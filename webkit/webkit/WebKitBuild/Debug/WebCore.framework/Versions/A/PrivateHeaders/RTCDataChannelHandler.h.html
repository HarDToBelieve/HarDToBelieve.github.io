36struct RTCDataChannelInit {
37 std::optional<bool> ordered;
38 std::optional<unsigned short> maxPacketLifeTime;
39 std::optional<unsigned short> maxRetransmits;
40 String protocol;
41 std::optional<bool> negotiated;
42 std::optional<unsigned short> id;
43 RTCPriorityType priority { RTCPriorityType::Low };
44
45 RTCDataChannelInit isolatedCopy() const;
46
47 template<class Encoder> void encode(Encoder&) const;
48 template<class Decoder> static std::optional<RTCDataChannelInit> decode(Decoder&);
49};
50
51inline RTCDataChannelInit RTCDataChannelInit::isolatedCopy() const
52{
53 auto copy = *this;
54 copy.protocol = protocol.isolatedCopy();
55 return copy;
56}
57
58template<class Encoder> void RTCDataChannelInit::encode(Encoder& encoder) const
59{
60 encoder << ordered << maxPacketLifeTime << maxRetransmits << protocol << negotiated << id << priority;
61}
62
63template<class Decoder> std::optional<RTCDataChannelInit> RTCDataChannelInit::decode(Decoder& decoder)
64{
65 std::optional<std::optional<bool>> ordered;
66 decoder >> ordered;
67 if (!ordered)
68 return { };
69
70 std::optional<std::optional<unsigned short>> maxPacketLifeTime;
71 decoder >> maxPacketLifeTime;
72 if (!maxPacketLifeTime)
73 return { };
74
75 std::optional<std::optional<unsigned short>> maxRetransmits;
76 decoder >> maxRetransmits;
77 if (!maxRetransmits)
78 return { };
79
80 String protocol;
81 if (!decoder.decode(protocol))
82 return { };
83
84 std::optional<std::optional<bool>> negotiated;
85 decoder >> negotiated;
86 if (!negotiated)
87 return { };
88
89 std::optional<std::optional<unsigned short>> id;
90 decoder >> id;
91 if (!id)
92 return { };
93
94 std::optional<RTCPriorityType> priority;
95 decoder >> priority;
96 if (!priority)
97 return { };
98
99 return RTCDataChannelInit { *ordered, *maxPacketLifeTime, *maxRetransmits, WTFMove(protocol), *negotiated, *id, *priority };
100}

113};
