34namespace WebCore {
35
36class CaptureDevice;
37class CaptureDeviceManager;
38class RealtimeMediaSource;
39
40struct CaptureSourceOrError;
41struct MediaConstraints;
42
43class WEBCORE_EXPORT SingleSourceFactory {
44public:
45 virtual ~SingleSourceFactory() = default;
46
47 virtual void setActiveSource(RealtimeMediaSource&);
48 virtual void unsetActiveSource(RealtimeMediaSource&);
49
50 virtual RealtimeMediaSource* activeSource() { return m_activeSource; }
51
52private:
53 RealtimeMediaSource* m_activeSource { nullptr };
54};
55
56class AudioCaptureFactory
57#if PLATFORM(IOS_FAMILY)
58 : public SingleSourceFactory
59#endif
60{
61public:
62 virtual ~AudioCaptureFactory() = default;
63 virtual CaptureSourceOrError createAudioCaptureSource(const CaptureDevice&, String&&, const MediaConstraints*) = 0;
64 virtual CaptureDeviceManager& audioCaptureDeviceManager() = 0;
65 virtual const Vector<CaptureDevice>& speakerDevices() const = 0;
66 virtual void computeSpeakerDevices(CompletionHandler<void()>&& callback) const { callback(); }
67
68 class ExtensiveObserver : public CanMakeWeakPtr<ExtensiveObserver> { };
69 virtual void addExtensiveObserver(ExtensiveObserver&) { };
70 virtual void removeExtensiveObserver(ExtensiveObserver&) { };
71
72protected:
73 AudioCaptureFactory() = default;
74};
