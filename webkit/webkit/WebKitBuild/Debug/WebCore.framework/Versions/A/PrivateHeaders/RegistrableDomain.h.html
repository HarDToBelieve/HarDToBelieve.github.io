34namespace WebCore {

36class RegistrableDomain {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 RegistrableDomain() = default;
40
41 explicit RegistrableDomain(const URL& url)
42 : RegistrableDomain(registrableDomainFromHost(url.host().toString()))
43 {
44 }
45
46 explicit RegistrableDomain(const SecurityOriginData& origin)
47 : RegistrableDomain(registrableDomainFromHost(origin.host))
48 {
49 }
50
51 bool isEmpty() const { return m_registrableDomain.isEmpty() || m_registrableDomain == "nullOrigin"_s; }
52 String& string() { return m_registrableDomain; }
53 const String& string() const { return m_registrableDomain; }
54
55 bool operator!=(const RegistrableDomain& other) const { return m_registrableDomain != other.m_registrableDomain; }
56 bool operator==(const RegistrableDomain& other) const { return m_registrableDomain == other.m_registrableDomain; }
57 bool operator==(const char* other) const { return m_registrableDomain == other; }
58
59 bool matches(const URL& url) const
60 {
61 return matches(url.host());
62 }
63
64 bool matches(const SecurityOriginData& origin) const
65 {
66 return matches(origin.host);
67 }
68
69 RegistrableDomain isolatedCopy() const { return RegistrableDomain { m_registrableDomain.isolatedCopy() }; }
70
71 RegistrableDomain(WTF::HashTableDeletedValueType)
72 : m_registrableDomain(WTF::HashTableDeletedValue) { }
73 bool isHashTableDeletedValue() const { return m_registrableDomain.isHashTableDeletedValue(); }
74 unsigned hash() const { return m_registrableDomain.hash(); }
75
76 struct RegistrableDomainHash {
77 static unsigned hash(const RegistrableDomain& registrableDomain) { return registrableDomain.m_registrableDomain.hash(); }
78 static bool equal(const RegistrableDomain& a, const RegistrableDomain& b) { return a == b; }
79 static const bool safeToCompareToEmptyOrDeleted = false;
80 };
81
82 static RegistrableDomain uncheckedCreateFromRegistrableDomainString(const String& domain)
83 {
84 return RegistrableDomain { String { domain } };
85 }
86
87 static RegistrableDomain uncheckedCreateFromHost(const String& host)
88 {
89#if ENABLE(PUBLIC_SUFFIX_LIST)
90 auto registrableDomain = topPrivatelyControlledDomain(host);
91 if (registrableDomain.isEmpty())
92 return uncheckedCreateFromRegistrableDomainString(host);
93 return RegistrableDomain { WTFMove(registrableDomain) };
94#else
95 return uncheckedCreateFromRegistrableDomainString(host);
96#endif
97 }
98
99 template<class Encoder> void encode(Encoder&) const;
100 template<class Decoder> static std::optional<RegistrableDomain> decode(Decoder&);
101
102protected:
103
104private:
105 explicit RegistrableDomain(String&& domain)
106 : m_registrableDomain { domain.isEmpty() ? "nullOrigin"_s : WTFMove(domain) }
107 {
108 }
109
110 bool matches(StringView host) const
111 {
112 if (host.isEmpty() && m_registrableDomain == "nullOrigin"_s)
113 return true;
114 if (!host.endsWith(m_registrableDomain))
115 return false;
116 if (host.length() == m_registrableDomain.length())
117 return true;
118 return host[host.length() - m_registrableDomain.length() - 1] == '.';
119 }
120
121 static inline String registrableDomainFromHost(const String& host)
122 {
123#if ENABLE(PUBLIC_SUFFIX_LIST)
124 auto domain = topPrivatelyControlledDomain(host);
125#else
126 auto domain = host;
127#endif
128 if (host.isEmpty())
129 domain = "nullOrigin"_s;
130 else if (domain.isEmpty())
131 domain = host;
132 return domain;
133 }
134
135 String m_registrableDomain;
136};
137
138template<class Encoder>
139void RegistrableDomain::encode(Encoder& encoder) const
140{
141 encoder << m_registrableDomain;
142}
143
144template<class Decoder>
145std::optional<RegistrableDomain> RegistrableDomain::decode(Decoder& decoder)
146{
147 std::optional<String> domain;
148 decoder >> domain;
149 if (!domain)
150 return std::nullopt;
151
152 RegistrableDomain registrableDomain;
153 registrableDomain.m_registrableDomain = WTFMove(*domain);
154 return registrableDomain;
155}
156
157inline bool areRegistrableDomainsEqual(const URL& a, const URL& b)
158{
159 return RegistrableDomain(a).matches(b);
160}
