56 void setSupportedCommands(const RemoteCommandsSet&);
57 const RemoteCommandsSet& supportedCommands() const;

59 virtual void updateSupportedCommands();

61
62 void setSupportsSeeking(bool);
63 bool supportsSeeking() const;

65 RemoteCommandListenerClient& client() const { return m_client; }

67private:
