32#include <wtf/TypeCasts.h>

34namespace JSC {
35class CallFrame;
36class JSGlobalObject;
37class JSObject;
38class JSValue;
39}
40
41namespace WebCore {
42
43class DOMWindow;
44class Document;
45class Location;
46
47class RemoteDOMWindow final : public AbstractDOMWindow {
48 WTF_MAKE_ISO_ALLOCATED_EXPORT(RemoteDOMWindow, WEBCORE_EXPORT);
49public:
50 static Ref<RemoteDOMWindow> create(Ref<RemoteFrame>&& frame, GlobalWindowIdentifier&& identifier)
51 {
52 return adoptRef(*new RemoteDOMWindow(WTFMove(frame), WTFMove(identifier)));
53 }
54
55 ~RemoteDOMWindow() final;
56
57 RemoteFrame* frame() const final { return m_frame.get(); }
58 ScriptExecutionContext* scriptExecutionContext() const final { return nullptr; }
59
60 // DOM API exposed cross-origin.
61 WindowProxy* self() const;
62 Location* location() const;
63 void close(Document&);
64 bool closed() const;
65 void focus(DOMWindow& incumbentWindow);
66 void blur();
67 unsigned length() const;
68 WindowProxy* top() const;
69 WindowProxy* opener() const;
70 WindowProxy* parent() const;
71 void postMessage(JSC::JSGlobalObject&, DOMWindow& incumbentWindow, JSC::JSValue message, const String& targetOrigin, Vector<JSC::Strong<JSC::JSObject>>&&);
72
73private:
74 WEBCORE_EXPORT RemoteDOMWindow(Ref<RemoteFrame>&&, GlobalWindowIdentifier&&);
75
76 bool isRemoteDOMWindow() const final { return true; }
77 bool isLocalDOMWindow() const final { return false; }
78
79 RefPtr<RemoteFrame> m_frame;
80};
81
82} // namespace WebCore
83
84SPECIALIZE_TYPE_TRAITS_BEGIN(WebCore::RemoteDOMWindow)
85 static bool isType(const WebCore::AbstractDOMWindow& window) { return window.isRemoteDOMWindow(); }
86SPECIALIZE_TYPE_TRAITS_END()
