50// A base class for RemoteGraphicsContextGL proxy side implementation
51// This implements the parts that are using WebCore internal functionality:
52// - Drawing buffer tracking management.
53// - Compositing support.
54class WEBCORE_EXPORT RemoteGraphicsContextGLProxyBase : public GraphicsContextGL, public ExtensionsGL {
55public:
56 RemoteGraphicsContextGLProxyBase(const GraphicsContextGLAttributes&);
57 ~RemoteGraphicsContextGLProxyBase() override;
58
59 // Other WebCore::GraphicsContextGL overrides.
60 using GraphicsContextGL::isEnabled;
61 PlatformLayer* platformLayer() const final;
62 ExtensionsGL& getExtensions() final;
63 void setContextVisibility(bool) final;
64 bool isGLES2Compliant() const final;
65 void markContextChanged() final;
66 bool layerComposited() const final;
67 void setBuffersToAutoClear(GCGLbitfield) final;
68 GCGLbitfield getBuffersToAutoClear() const final;
69 void markLayerComposited() final;
70 void enablePreserveDrawingBuffer() final;
71
72#if ENABLE(VIDEO) && USE(AVFOUNDATION)
73 GraphicsContextGLCV* asCV() final;
74#endif
75 // Other ExtensionGL overrides.
76 using ExtensionsGL::isEnabled;
77 bool supports(const String&) final;
78 void ensureEnabled(const String&) final;
79 bool isEnabled(const String&) final;
80
81#if !USE(ANGLE)
82 void readnPixelsEXT(GCGLint, GCGLint, GCGLsizei, GCGLsizei, GCGLenum, GCGLenum, GCGLsizei, GCGLvoid*) final;
83 void getnUniformfvEXT(GCGLuint, GCGLint, GCGLsizei, GCGLfloat*) final;
84 void getnUniformivEXT(GCGLuint, GCGLint, GCGLsizei, GCGLint*) final;
85#endif
86
87protected:
88 void initialize(const String& availableExtensions, const String& requestableExtensions);
89 virtual void waitUntilInitialized() = 0;
90 virtual void ensureExtensionEnabled(const String&) = 0;
91 virtual void notifyMarkContextChanged() = 0;
92#if PLATFORM(COCOA)
93 GraphicsContextGLIOSurfaceSwapChain& platformSwapChain();
94#endif
95
96private:
97 void platformInitialize();
98#if PLATFORM(COCOA)
99 RetainPtr<WebGLLayer> m_webGLLayer;
100#endif
101
102 // Guarded by waitUntilInitialized().
103 HashSet<String> m_availableExtensions;
104 HashSet<String> m_requestableExtensions;
105
106 HashSet<String> m_enabledExtensions;
107
108 // FIXME: Drawing buffer clear tracking should be moved to the client.
109 bool m_layerComposited = false;
110 GCGLbitfield m_buffersToAutoClear = { 0 };
111};

113}
