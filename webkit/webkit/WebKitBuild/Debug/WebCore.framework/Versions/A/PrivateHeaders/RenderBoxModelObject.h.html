71class BackgroundImageGeometry {
72public:
73 BackgroundImageGeometry(const LayoutRect& destinationRect, const LayoutSize& tile, const LayoutSize& phase, const LayoutSize& space, bool fixedAttachment)
74 : m_destRect(destinationRect)
75 , m_destOrigin(m_destRect.location())
76 , m_tileSize(tile)
77 , m_phase(phase)
78 , m_space(space)
79 , m_hasNonLocalGeometry(fixedAttachment)
80 {
81 }
82
83 LayoutRect destRect() const { return m_destRect; }
84 LayoutSize phase() const { return m_phase; }
85 LayoutSize tileSize() const { return m_tileSize; }
86 LayoutSize spaceSize() const { return m_space; }
87 bool hasNonLocalGeometry() const { return m_hasNonLocalGeometry; }
88
89 LayoutSize relativePhase() const
90 {
91 LayoutSize phase = m_phase;
92 phase += m_destRect.location() - m_destOrigin;
93 return phase;
94 }
95
96 void clip(const LayoutRect& clipRect) { m_destRect.intersect(clipRect); }
97
98private:
99 LayoutRect m_destRect;
100 LayoutPoint m_destOrigin;
101 LayoutSize m_tileSize;
102 LayoutSize m_phase;
103 LayoutSize m_space;
104 bool m_hasNonLocalGeometry; // Has background-attachment: fixed. Implies that we can't always cheaply compute destRect.
105};
