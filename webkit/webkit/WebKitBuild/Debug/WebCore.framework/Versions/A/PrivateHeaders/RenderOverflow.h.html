23#include "LayoutRect.h"
24#include <wtf/RefCounted.h>

26namespace WebCore
27{
28// RenderOverflow is a class for tracking content that spills out of a box. This class is used by RenderBox and
29// LegacyInlineFlowBox.
30//
31// There are two types of overflow: layout overflow (which is expected to be reachable via scrolling mechanisms) and
32// visual overflow (which is not expected to be reachable via scrolling mechanisms).
33//
34// Layout overflow examples include other boxes that spill out of our box, For example, in the inline case a tall image
35// could spill out of a line box.
36
37// Examples of visual overflow are shadows, text stroke, outline (and eventually border-image).
38
39// This object is allocated only when some of these fields have non-default values in the owning box.
40class RenderOverflow : public WTF::RefCounted<RenderOverflow> {
41 WTF_MAKE_NONCOPYABLE(RenderOverflow); WTF_MAKE_FAST_ALLOCATED;
42public:
43 RenderOverflow(const LayoutRect& layoutRect, const LayoutRect& visualRect)
44 : m_layoutOverflow(layoutRect)
45 , m_visualOverflow(visualRect)
46 {
47 }
48
49 const LayoutRect layoutOverflowRect() const { return m_layoutOverflow; }
50 const LayoutRect visualOverflowRect() const { return m_visualOverflow; }
51
52 void move(LayoutUnit dx, LayoutUnit dy);
53
54 void addLayoutOverflow(const LayoutRect&);
55 void addVisualOverflow(const LayoutRect&);
56
57 void setLayoutOverflow(const LayoutRect&);
58 void setVisualOverflow(const LayoutRect&);
59
60 LayoutUnit layoutClientAfterEdge() const { return m_layoutClientAfterEdge; }
61 void setLayoutClientAfterEdge(LayoutUnit clientAfterEdge) { m_layoutClientAfterEdge = clientAfterEdge; }
62
63private:
64 LayoutRect m_layoutOverflow;
65 LayoutRect m_visualOverflow;
66
67 LayoutUnit m_layoutClientAfterEdge;
68};
69
70inline void RenderOverflow::move(LayoutUnit dx, LayoutUnit dy)
71{
72 m_layoutOverflow.move(dx, dy);
73 m_visualOverflow.move(dx, dy);
74}
75
76inline void RenderOverflow::addLayoutOverflow(const LayoutRect& rect)
77{
78 LayoutUnit maxX = std::max(rect.maxX(), m_layoutOverflow.maxX());
79 LayoutUnit maxY = std::max(rect.maxY(), m_layoutOverflow.maxY());
80 m_layoutOverflow.setX(std::min(rect.x(), m_layoutOverflow.x()));
81 m_layoutOverflow.setY(std::min(rect.y(), m_layoutOverflow.y()));
82 m_layoutOverflow.setWidth(maxX - m_layoutOverflow.x());
83 m_layoutOverflow.setHeight(maxY - m_layoutOverflow.y());
84}
85
86inline void RenderOverflow::addVisualOverflow(const LayoutRect& rect)
87{
88 LayoutUnit maxX = std::max(rect.maxX(), m_visualOverflow.maxX());
89 LayoutUnit maxY = std::max(rect.maxY(), m_visualOverflow.maxY());
90 m_visualOverflow.setX(std::min(rect.x(), m_visualOverflow.x()));
91 m_visualOverflow.setY(std::min(rect.y(), m_visualOverflow.y()));
92 m_visualOverflow.setWidth(maxX - m_visualOverflow.x());
93 m_visualOverflow.setHeight(maxY - m_visualOverflow.y());
94}
95
96inline void RenderOverflow::setLayoutOverflow(const LayoutRect& rect)
97{
98 m_layoutOverflow = rect;
99}
100
101inline void RenderOverflow::setVisualOverflow(const LayoutRect& rect)
102{
103 m_visualOverflow = rect;
104}
