27#pragma once
28
29#include "DocumentMarker.h"
30#include <wtf/Vector.h>
31
32namespace WebCore {
33
34class RenderedDocumentMarker : public DocumentMarker {
35public:
36 explicit RenderedDocumentMarker(DocumentMarker&& marker)
37 : DocumentMarker(WTFMove(marker))
38 {
39 }
40
41 bool contains(const FloatPoint& point) const
42 {
43 ASSERT(m_isValid);
44 for (const auto& rect : m_rects) {
45 if (rect.contains(point))
46 return true;
47 }
48 return false;
49 }
50
51 void setUnclippedAbsoluteRects(Vector<FloatRect>&& rects)
52 {
53 m_isValid = true;
54 m_rects = WTFMove(rects);
55 }
56
57 const Vector<FloatRect, 1>& unclippedAbsoluteRects() const
58 {
59 ASSERT(m_isValid);
60 return m_rects;
61 }
62
63 void invalidate()
64 {
65 m_isValid = false;
66 m_rects.clear();
67 }
68
69 bool isValid() const { return m_isValid; }
70
71private:
72 Vector<FloatRect, 1> m_rects;
73 bool m_isValid { false };
74};
75
76} // namespace WebCore
