37enum class ResourceType : uint16_t {
38 Document = 0x0001,
39 Image = 0x0002,
40 StyleSheet = 0x0004,
41 Script = 0x0008,
42 Font = 0x0010,
43 SVGDocument = 0x0020,
44 Media = 0x0040,
45 Popup = 0x0080,
46 Ping = 0x0100,
47 Fetch = 0x0200,
48 WebSocket = 0x0400,
49 Other = 0x0800,
50};
51const uint16_t ResourceTypeMask = 0x0FFF;
52
53enum class LoadType : uint16_t {
54 FirstParty = 0x1000,
55 ThirdParty = 0x2000,
56};
57const uint16_t LoadTypeMask = 0x3000;
58
59enum class LoadContext : uint16_t {
60 TopFrame = 0x4000,
61 ChildFrame = 0x8000,
62};
63const uint16_t LoadContextMask = 0xC000;
64
65static_assert(!(ResourceTypeMask & LoadTypeMask), "ResourceTypeMask and LoadTypeMask should be mutually exclusive because they are stored in the same uint16_t");
66static_assert(!(ResourceTypeMask & LoadContextMask), "ResourceTypeMask and LoadContextMask should be mutually exclusive because they are stored in the same uint16_t");
67static_assert(!(LoadContextMask & LoadTypeMask), "LoadContextMask and LoadTypeMask should be mutually exclusive because they are stored in the same uint16_t");
68
69typedef uint16_t ResourceFlags;
70
71// The first 32 bits of a uint64_t action are used for the action location.
72// The next 16 bits are used for the flags (ResourceType and LoadType).
73// The next bit is used to mark actions that are from a rule with an if-domain.
74// Actions from rules with unless-domain conditions are distinguished from
75// rules with if-domain conditions by not having this bit set.
76// Actions from rules with no conditions are put in the DFA without conditions.
77// The values -1 and -2 are used for removed and empty values in HashTables.
78const uint64_t ActionFlagMask = 0x0000FFFF00000000;
79const uint64_t IfConditionFlag = 0x0001000000000000;
80
81OptionSet<ResourceType> toResourceType(CachedResource::Type, ResourceRequestBase::Requester);
82std::optional<OptionSet<ResourceType>> readResourceType(StringView);
83std::optional<OptionSet<LoadType>> readLoadType(StringView);
84std::optional<OptionSet<LoadContext>> readLoadContext(StringView);
85
86struct ResourceLoadInfo {
87 URL resourceURL;
88 URL mainDocumentURL;
89 OptionSet<ResourceType> type;
90 bool mainFrameContext { false };
91
92 bool isThirdParty() const;
93 ResourceFlags getResourceFlags() const;
94};
