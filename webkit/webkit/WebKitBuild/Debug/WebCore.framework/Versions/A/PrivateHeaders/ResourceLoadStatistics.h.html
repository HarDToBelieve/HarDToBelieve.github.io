38namespace WebCore {
39
40class KeyedDecoder;
41class KeyedEncoder;
42
43struct ResourceLoadStatistics {
44 WTF_MAKE_STRUCT_FAST_ALLOCATED;
45 explicit ResourceLoadStatistics(const RegistrableDomain& domain)
46 : registrableDomain { domain }
47 {
48 }
49
50 ResourceLoadStatistics() = default;
51
52 ResourceLoadStatistics(const ResourceLoadStatistics&) = delete;
53 ResourceLoadStatistics& operator=(const ResourceLoadStatistics&) = delete;
54 ResourceLoadStatistics(ResourceLoadStatistics&&) = default;
55 ResourceLoadStatistics& operator=(ResourceLoadStatistics&&) = default;
56
57 static constexpr Seconds NoExistingTimestamp { -1 };
58
59 WEBCORE_EXPORT static WallTime reduceTimeResolution(WallTime);
60
61 WEBCORE_EXPORT void encode(KeyedEncoder&) const;
62 WEBCORE_EXPORT bool decode(KeyedDecoder&, unsigned modelVersion);
63
64 WEBCORE_EXPORT String toString() const;
65
66 WEBCORE_EXPORT void merge(const ResourceLoadStatistics&);
67
68 RegistrableDomain registrableDomain;
69
70 WallTime lastSeen;
71
72 // User interaction
73 bool hadUserInteraction { false };
74 // Timestamp. Default value is negative, 0 means it was reset.
75 WallTime mostRecentUserInteractionTime { WallTime::fromRawSeconds(-1) };
76 bool grandfathered { false };
77
78 // Storage access
79 HashSet<RegistrableDomain> storageAccessUnderTopFrameDomains;
80
81 // Top frame stats
82 HashSet<RegistrableDomain> topFrameUniqueRedirectsTo;
83 HashSet<RegistrableDomain> topFrameUniqueRedirectsToSinceSameSiteStrictEnforcement;
84 HashSet<RegistrableDomain> topFrameUniqueRedirectsFrom;
85 HashSet<RegistrableDomain> topFrameLinkDecorationsFrom;
86 bool gotLinkDecorationFromPrevalentResource { false };
87 HashSet<RegistrableDomain> topFrameLoadedThirdPartyScripts;
88
89 // Subframe stats
90 HashSet<RegistrableDomain> subframeUnderTopFrameDomains;
91
92 // Subresource stats
93 HashSet<RegistrableDomain> subresourceUnderTopFrameDomains;
94 HashSet<RegistrableDomain> subresourceUniqueRedirectsTo;
95 HashSet<RegistrableDomain> subresourceUniqueRedirectsFrom;
96
97 // Prevalent resource stats
98 bool isPrevalentResource { false };
99 bool isVeryPrevalentResource { false };
100 unsigned dataRecordsRemoved { 0 };
101 unsigned timesAccessedAsFirstPartyDueToUserInteraction { 0 };
102 unsigned timesAccessedAsFirstPartyDueToStorageAccessAPI { 0 };
103
104 enum class IsEphemeral : bool { No, Yes };
105
106 enum class NavigatorAPI : uint64_t {
107 AppVersion = 1 << 0,
108 UserAgent = 1 << 1,
109 Plugins = 1 << 2,
110 MimeTypes = 1 << 3,
111 CookieEnabled = 1 << 4,
112 JavaEnabled = 1 << 5,
113 };
114 enum class ScreenAPI : uint64_t {
115 Height = 1 << 0,
116 Width = 1 << 1,
117 ColorDepth = 1 << 2,
118 PixelDepth = 1 << 3,
119 AvailLeft = 1 << 4,
120 AvailTop = 1 << 5,
121 AvailHeight = 1 << 6,
122 AvailWidth = 1 << 7,
123 };
124#if ENABLE(WEB_API_STATISTICS)
125 // This set represents the registrable domain of the top frame where web API
126 // were used in the top frame or one of its subframes.
127 HashSet<RegistrableDomain> topFrameRegistrableDomainsWhichAccessedWebAPIs;
128 HashSet<String> fontsFailedToLoad;
129 HashSet<String> fontsSuccessfullyLoaded;
130 CanvasActivityRecord canvasActivityRecord;
131 OptionSet<NavigatorAPI> navigatorFunctionsAccessed;
132 OptionSet<ScreenAPI> screenFunctionsAccessed;
133#endif
134};
135
