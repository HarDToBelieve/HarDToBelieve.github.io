95 // This enables automatic WAL truncation via a commit hook that uses SQLITE_CHECKPOINT_TRUNCATE.
96 // However, it shouldn't be used if you use a custom busy handler or timeout. This is because
97 // SQLITE_CHECKPOINT_TRUNCATE will invoke the busy handler if it can't acquire the necessary
98 // locks, which can lead to unintended delays.
99 void enableAutomaticWALTruncation();
100
101 // Gets/sets the maximum size in bytes
102 // Depending on per-database attributes, the size will only be settable in units that are the page size of the database, which is established at creation
103 // These chunks will never be anything other than 512, 1024, 2048, 4096, 8192, 16384, or 32768 bytes in size.
104 // setMaximumSize() will round the size down to the next smallest chunk if the passed size doesn't align.
105 int64_t maximumSize();
106 WEBCORE_EXPORT void setMaximumSize(int64_t);
107
108 // Gets the number of unused bytes in the database file.
109 int64_t freeSpaceSize();
110 int64_t totalSize();
111
112 // The SQLite SYNCHRONOUS pragma can be either FULL, NORMAL, or OFF
113 // FULL - Any writing calls to the DB block until the data is actually on the disk surface
114 // NORMAL - SQLite pauses at some critical moments when writing, but much less than FULL
115 // OFF - Calls return immediately after the data has been passed to disk
116 enum SynchronousPragma { SyncOff = 0, SyncNormal = 1, SyncFull = 2 };
117 void setSynchronous(SynchronousPragma);

119 WEBCORE_EXPORT int lastError();

128 }

135 // The SQLite AUTO_VACUUM pragma can be either NONE, FULL, or INCREMENTAL.
136 // NONE - SQLite does not do any vacuuming
137 // FULL - SQLite moves all empty pages to the end of the DB file and truncates
138 // the file to remove those pages after every transaction. This option
139 // requires SQLite to store additional information about each page in
140 // the database file.
141 // INCREMENTAL - SQLite stores extra information for each page in the database
142 // file, but removes the empty pages only when PRAGMA INCREMANTAL_VACUUM
143 // is called.
144 enum AutoVacuumPragma { AutoVacuumNone = 0, AutoVacuumFull = 1, AutoVacuumIncremental = 2 };
145 WEBCORE_EXPORT bool turnOnIncrementalAutoVacuum();

194 int m_openError { SQLITE_ERROR };

198inline void SQLiteDatabase::incrementStatementCount()
199{
200#if ASSERT_ENABLED
201 ++m_statementCount;
202#endif
203}
204
205inline void SQLiteDatabase::decrementStatementCount()
206{
207#if ASSERT_ENABLED
208 ASSERT(m_statementCount);
209 --m_statementCount;
210#endif
211}
