26#ifndef SQLiteTransaction_h
27#define SQLiteTransaction_h
28
29#include <wtf/FastMalloc.h>
30#include <wtf/Noncopyable.h>
31
32namespace WebCore {
33
34class SQLiteDatabase;
35
36class SQLiteTransaction {
37 WTF_MAKE_NONCOPYABLE(SQLiteTransaction); WTF_MAKE_FAST_ALLOCATED;
38public:
39 WEBCORE_EXPORT SQLiteTransaction(SQLiteDatabase& db, bool readOnly = false);
40 WEBCORE_EXPORT ~SQLiteTransaction();
41
42 WEBCORE_EXPORT void begin();
43 WEBCORE_EXPORT void commit();
44 WEBCORE_EXPORT void rollback();
45 void stop();
46
47 bool inProgress() const { return m_inProgress; }
48 bool wasRolledBackBySqlite() const;
49
50 SQLiteDatabase& database() const { return m_db; }
51
52private:
53 SQLiteDatabase& m_db;
54 bool m_inProgress;
55 bool m_readOnly;
56};
57
58} // namespace WebCore
59
60#endif // SQLiteTransation_H
