22#pragma once
23
24#include "ExceptionOr.h"
25#include "SVGParsingError.h"
26#include "SVGPropertyTraits.h"

28namespace WebCore {
29
30class CSSPrimitiveValue;
31class SVGLengthContext;
32
33enum class SVGLengthType : uint8_t {
34 Unknown = 0,
35 Number,
36 Percentage,
37 Ems,
38 Exs,
39 Pixels,
40 Centimeters,
41 Millimeters,
42 Inches,
43 Points,
44 Picas
45};
46
47enum class SVGLengthMode : uint8_t {
48 Width,
49 Height,
50 Other
51};
52
53enum class SVGLengthNegativeValuesMode : uint8_t {
54 Allow,
55 Forbid
56};
57
58class SVGLengthValue {
59 WTF_MAKE_FAST_ALLOCATED;
60public:
61 SVGLengthValue(SVGLengthMode = SVGLengthMode::Other, const String& valueAsString = { });
62 SVGLengthValue(float valueInSpecifiedUnits, SVGLengthType, SVGLengthMode = SVGLengthMode::Other);
63 SVGLengthValue(const SVGLengthContext&, float, SVGLengthType = SVGLengthType::Number, SVGLengthMode = SVGLengthMode::Other);
64
65 static std::optional<SVGLengthValue> construct(SVGLengthMode, StringView);
66 static SVGLengthValue construct(SVGLengthMode, StringView, SVGParsingError&, SVGLengthNegativeValuesMode = SVGLengthNegativeValuesMode::Allow);
67 static SVGLengthValue blend(const SVGLengthValue& from, const SVGLengthValue& to, float progress);
68
69 static SVGLengthValue fromCSSPrimitiveValue(const CSSPrimitiveValue&);
70 static Ref<CSSPrimitiveValue> toCSSPrimitiveValue(const SVGLengthValue&);
71
72 SVGLengthType lengthType() const { return m_lengthType; }
73 SVGLengthMode lengthMode() const { return m_lengthMode; }
74
75 bool isZero() const { return !m_valueInSpecifiedUnits; }
76 bool isRelative() const { return m_lengthType == SVGLengthType::Percentage || m_lengthType == SVGLengthType::Ems || m_lengthType == SVGLengthType::Exs; }
77
78 float value(const SVGLengthContext&) const;
79 float valueAsPercentage() const { return m_lengthType == SVGLengthType::Percentage ? m_valueInSpecifiedUnits / 100 : m_valueInSpecifiedUnits; }
80 float valueInSpecifiedUnits() const { return m_valueInSpecifiedUnits; }
81
82 String valueAsString() const;
83 ExceptionOr<float> valueForBindings(const SVGLengthContext&) const;
84
85 void setValueInSpecifiedUnits(float value) { m_valueInSpecifiedUnits = value; }
86 ExceptionOr<void> setValue(const SVGLengthContext&, float);
87 ExceptionOr<void> setValue(const SVGLengthContext&, float, SVGLengthType, SVGLengthMode);
88
89 ExceptionOr<void> setValueAsString(StringView);
90 ExceptionOr<void> setValueAsString(StringView, SVGLengthMode);
91
92 ExceptionOr<void> convertToSpecifiedUnits(const SVGLengthContext&, SVGLengthType);
93
94private:
95 float m_valueInSpecifiedUnits { 0 };
96 SVGLengthType m_lengthType { SVGLengthType::Number };
97 SVGLengthMode m_lengthMode { SVGLengthMode::Other };
98};
99
100inline bool operator==(const SVGLengthValue& a, const SVGLengthValue& b)
101{
102 return a.valueInSpecifiedUnits() == b.valueInSpecifiedUnits() && a.lengthType() == b.lengthType() && a.lengthMode() == b.lengthMode();
103}
104
105inline bool operator!=(const SVGLengthValue& a, const SVGLengthValue& b)
106{
107 return a.valueInSpecifiedUnits() != b.valueInSpecifiedUnits() || a.lengthType() != b.lengthType() || a.lengthMode() != b.lengthMode();
108}

110WTF::TextStream& operator<<(WTF::TextStream&, const SVGLengthValue&);

112} // namespace WebCore
