44class SWContextManager {
45public:
46 WEBCORE_EXPORT static SWContextManager& singleton();
47
48 class Connection {
49 WTF_MAKE_FAST_ALLOCATED;
50 public:
51 virtual ~Connection() { }
52
53 virtual void establishConnection(CompletionHandler<void()>&&) = 0;
54 virtual void postMessageToServiceWorkerClient(const ServiceWorkerClientIdentifier& destinationIdentifier, const MessageWithMessagePorts&, ServiceWorkerIdentifier source, const String& sourceOrigin) = 0;
55 virtual void serviceWorkerStarted(std::optional<ServiceWorkerJobDataIdentifier>, ServiceWorkerIdentifier, bool doesHandleFetch) = 0;
56 virtual void serviceWorkerFailedToStart(std::optional<ServiceWorkerJobDataIdentifier>, ServiceWorkerIdentifier, const String& message) = 0;
57 virtual void didFinishInstall(std::optional<ServiceWorkerJobDataIdentifier>, ServiceWorkerIdentifier, bool wasSuccessful) = 0;
58 virtual void didFinishActivation(ServiceWorkerIdentifier) = 0;
59 virtual void setServiceWorkerHasPendingEvents(ServiceWorkerIdentifier, bool) = 0;
60 virtual void workerTerminated(ServiceWorkerIdentifier) = 0;
61 virtual void skipWaiting(ServiceWorkerIdentifier, CompletionHandler<void()>&&) = 0;
62 virtual void setScriptResource(ServiceWorkerIdentifier, const URL&, const ServiceWorkerContextData::ImportedScript&) = 0;
63
64 using FindClientByIdentifierCallback = CompletionHandler<void(ExceptionOr<std::optional<ServiceWorkerClientData>>&&)>;
65 virtual void findClientByIdentifier(ServiceWorkerIdentifier, ServiceWorkerClientIdentifier, FindClientByIdentifierCallback&&) = 0;
66 virtual void matchAll(ServiceWorkerIdentifier, const ServiceWorkerClientQueryOptions&, ServiceWorkerClientsMatchAllCallback&&) = 0;
67 virtual void claim(ServiceWorkerIdentifier, CompletionHandler<void(ExceptionOr<void>&&)>&&) = 0;
68
69 virtual void didFailHeartBeatCheck(ServiceWorkerIdentifier) = 0;
70
71 virtual bool isThrottleable() const = 0;
72
73 bool isClosed() const { return m_isClosed; }
74 bool shouldUseShortTimeout() const { return m_shouldUseShortTimeout; }
75
76 protected:
77 void setAsClosed() { m_isClosed = true; }
78 void setShouldUseShortTimeout(bool value) { m_shouldUseShortTimeout = value; }
79
80 private:
81 bool m_isClosed { false };
82 bool m_shouldUseShortTimeout { false };
83 };
84
85 WEBCORE_EXPORT void setConnection(std::unique_ptr<Connection>&&);
86 WEBCORE_EXPORT Connection* connection() const;
87
88 WEBCORE_EXPORT void registerServiceWorkerThreadForInstall(Ref<ServiceWorkerThreadProxy>&&);
89 WEBCORE_EXPORT ServiceWorkerThreadProxy* serviceWorkerThreadProxy(ServiceWorkerIdentifier) const;
90 WEBCORE_EXPORT void postMessageToServiceWorker(ServiceWorkerIdentifier destination, MessageWithMessagePorts&&, ServiceWorkerOrClientData&& sourceData);
91 WEBCORE_EXPORT void fireInstallEvent(ServiceWorkerIdentifier);
92 WEBCORE_EXPORT void fireActivateEvent(ServiceWorkerIdentifier);
93 WEBCORE_EXPORT void terminateWorker(ServiceWorkerIdentifier, Seconds timeout, Function<void()>&&);

96 void forEachServiceWorkerThread(const WTF::Function<void(ServiceWorkerThreadProxy&)>&);
97
98 WEBCORE_EXPORT bool postTaskToServiceWorker(ServiceWorkerIdentifier, WTF::Function<void(ServiceWorkerGlobalScope&)>&&);
99
100 using ServiceWorkerCreationCallback = void(uint64_t);
101 void setServiceWorkerCreationCallback(ServiceWorkerCreationCallback* callback) { m_serviceWorkerCreationCallback = callback; }
102
103 ServiceWorkerThreadProxy* workerByID(ServiceWorkerIdentifier identifier) { return m_workerMap.get(identifier); }
104
105 WEBCORE_EXPORT void stopAllServiceWorkers();
106
107 static constexpr Seconds workerTerminationTimeout { 10_s };
108 static constexpr Seconds syncWorkerTerminationTimeout { 100_ms }; // Only used by layout tests.
109
110 WEBCORE_EXPORT void whenTerminationRequestsAreDone(CompletionHandler<void()>&&);

112private:
113 SWContextManager() = default;
114
115 void startedServiceWorker(std::optional<ServiceWorkerJobDataIdentifier>, ServiceWorkerIdentifier, const String& exceptionMessage, bool doesHandleFetch);

117
118 void stopWorker(ServiceWorkerThreadProxy&, Seconds, Function<void()>&&);

120 HashMap<ServiceWorkerIdentifier, Ref<ServiceWorkerThreadProxy>> m_workerMap;
121 std::unique_ptr<Connection> m_connection;
122 ServiceWorkerCreationCallback* m_serviceWorkerCreationCallback { nullptr };

124 class ServiceWorkerTerminationRequest {
125 WTF_MAKE_FAST_ALLOCATED;
126 public:
127 ServiceWorkerTerminationRequest(SWContextManager&, ServiceWorkerIdentifier, Seconds timeout);
128
129 private:
130 Timer m_timeoutTimer;
131 };
132 HashMap<ServiceWorkerIdentifier, std::unique_ptr<ServiceWorkerTerminationRequest>> m_pendingServiceWorkerTerminationRequests;
133 Vector<CompletionHandler<void()>> m_whenTerminationRequestsAreDoneHandlers;
134};
