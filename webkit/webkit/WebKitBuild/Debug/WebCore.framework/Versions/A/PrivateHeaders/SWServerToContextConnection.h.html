30#include "ExceptionData.h"

38namespace WebCore {
39
40class SWServer;
41struct ServiceWorkerClientData;
42struct ServiceWorkerClientIdentifier;
43struct ServiceWorkerContextData;
44struct ServiceWorkerJobDataIdentifier;
45
46class SWServerToContextConnection {
47 WTF_MAKE_FAST_ALLOCATED;
48public:
49 WEBCORE_EXPORT virtual ~SWServerToContextConnection();
50
51 SWServerToContextConnectionIdentifier identifier() const { return m_identifier; }
52
53 // Messages to the SW host process
54 virtual void installServiceWorkerContext(const ServiceWorkerContextData&, const String& userAgent) = 0;
55 virtual void updateAppBoundValue(ServiceWorkerIdentifier, LastNavigationWasAppBound) = 0;
56 virtual void fireInstallEvent(ServiceWorkerIdentifier) = 0;
57 virtual void fireActivateEvent(ServiceWorkerIdentifier) = 0;
58 virtual void terminateWorker(ServiceWorkerIdentifier) = 0;
59 virtual void didSaveScriptsToDisk(ServiceWorkerIdentifier, const ScriptBuffer&, const HashMap<URL, ScriptBuffer>& importedScripts) = 0;
60 virtual void findClientByIdentifierCompleted(uint64_t requestIdentifier, const std::optional<ServiceWorkerClientData>&, bool hasSecurityError) = 0;
61 virtual void matchAllCompleted(uint64_t requestIdentifier, const Vector<ServiceWorkerClientData>&) = 0;
62
63 // Messages back from the SW host process
64 WEBCORE_EXPORT void scriptContextFailedToStart(const std::optional<ServiceWorkerJobDataIdentifier>&, ServiceWorkerIdentifier, const String& message);
65 WEBCORE_EXPORT void scriptContextStarted(const std::optional<ServiceWorkerJobDataIdentifier>&, ServiceWorkerIdentifier, bool doesHandleFetch);
66 WEBCORE_EXPORT void didFinishInstall(const std::optional<ServiceWorkerJobDataIdentifier>&, ServiceWorkerIdentifier, bool wasSuccessful);
67 WEBCORE_EXPORT void didFinishActivation(ServiceWorkerIdentifier);
68 WEBCORE_EXPORT void setServiceWorkerHasPendingEvents(ServiceWorkerIdentifier, bool hasPendingEvents);
69 WEBCORE_EXPORT void skipWaiting(ServiceWorkerIdentifier, CompletionHandler<void()>&&);
70 WEBCORE_EXPORT void workerTerminated(ServiceWorkerIdentifier);
71 WEBCORE_EXPORT void findClientByIdentifier(uint64_t clientIdRequestIdentifier, ServiceWorkerIdentifier, ServiceWorkerClientIdentifier);
72 WEBCORE_EXPORT void matchAll(uint64_t requestIdentifier, ServiceWorkerIdentifier, const ServiceWorkerClientQueryOptions&);
73 WEBCORE_EXPORT void claim(ServiceWorkerIdentifier, CompletionHandler<void(std::optional<ExceptionData>&&)>&&);
74 WEBCORE_EXPORT void setScriptResource(ServiceWorkerIdentifier, URL&& scriptURL, ServiceWorkerContextData::ImportedScript&&);
75 WEBCORE_EXPORT void didFailHeartBeatCheck(ServiceWorkerIdentifier);
76
77 const RegistrableDomain& registrableDomain() const { return m_registrableDomain; }
78
79 virtual void connectionIsNoLongerNeeded() = 0;
80 virtual void terminateDueToUnresponsiveness() = 0;
81
82protected:
83 WEBCORE_EXPORT explicit SWServerToContextConnection(RegistrableDomain&&);
84
85private:
86 SWServerToContextConnectionIdentifier m_identifier;
87 RegistrableDomain m_registrableDomain;
88};
89
90} // namespace WebCore

92#endif // ENABLE(SERVICE_WORKER)
