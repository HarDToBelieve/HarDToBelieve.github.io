31#pragma once
32
33#include "SharedBuffer.h"
34
35namespace WebCore {
36
37class ScriptBuffer {
38public:
39 ScriptBuffer() = default;
40
41 ScriptBuffer(RefPtr<SharedBuffer>&& buffer)
42 : m_buffer(WTFMove(buffer))
43 {
44 }
45
46 static ScriptBuffer empty();
47
48 WEBCORE_EXPORT explicit ScriptBuffer(const String&);
49
50 String toString() const;
51 SharedBuffer* buffer() const { return m_buffer.get(); }
52
53 ScriptBuffer isolatedCopy() const { return ScriptBuffer(m_buffer ? RefPtr<SharedBuffer>(m_buffer->copy()) : nullptr); }
54 explicit operator bool() const { return !!m_buffer; }
55 bool isEmpty() const { return !m_buffer || !m_buffer->size(); }
56
57 WEBCORE_EXPORT bool containsSingleFileMappedSegment() const;
58 void append(const String&);
59
60private:
61 RefPtr<SharedBuffer> m_buffer; // Contains the UTF-8 encoded script.
62};
63
64bool operator==(const ScriptBuffer&, const ScriptBuffer&);
65bool operator!=(const ScriptBuffer&, const ScriptBuffer&);
66
67} // namespace WebCore
