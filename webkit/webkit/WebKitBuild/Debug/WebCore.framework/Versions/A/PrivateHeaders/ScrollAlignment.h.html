44#pragma once
45
46namespace WTF {
47class TextStream;
48}
49
50namespace WebCore {
51
52struct ScrollAlignment {
53
54 enum class Behavior {
55 NoScroll,
56 AlignCenter,
57 AlignTop,
58 AlignBottom,
59 AlignLeft,
60 AlignRight,
61 AlignToClosestEdge
62 };
63
64 static Behavior getVisibleBehavior(const ScrollAlignment& s) { return s.m_rectVisible; }
65 static Behavior getPartialBehavior(const ScrollAlignment& s) { return s.m_rectPartial; }
66 static Behavior getHiddenBehavior(const ScrollAlignment& s) { return s.m_rectHidden; }
67
68 static const ScrollAlignment alignCenterIfNotVisible;
69 static const ScrollAlignment alignToEdgeIfNotVisible;
70 WEBCORE_EXPORT static const ScrollAlignment alignCenterIfNeeded;
71 WEBCORE_EXPORT static const ScrollAlignment alignToEdgeIfNeeded;
72 WEBCORE_EXPORT static const ScrollAlignment alignCenterAlways;
73 static const ScrollAlignment alignTopAlways;
74 static const ScrollAlignment alignRightAlways;
75 static const ScrollAlignment alignLeftAlways;
76 static const ScrollAlignment alignBottomAlways;
77
78 Behavior m_rectVisible;
79 Behavior m_rectHidden;
80 Behavior m_rectPartial;
81};
82
83WEBCORE_EXPORT WTF::TextStream& operator<<(WTF::TextStream&, ScrollAlignment::Behavior);
84WEBCORE_EXPORT WTF::TextStream& operator<<(WTF::TextStream&, const ScrollAlignment&);
85
86}; // namespace WebCore
