47class ScrollControllerTimer : public RunLoop::TimerBase {
48public:
49 ScrollControllerTimer(RunLoop& runLoop, Function<void()>&& callback)
50 : RunLoop::TimerBase(runLoop)
51 , m_callback(WTFMove(callback))
52 {
53 }
54
55 void fired() final
56 {
57 m_callback();
58 }
59
60private:
61 Function<void()> m_callback;
62};

73 virtual void stopAnimationCallback(ScrollController&) = 0;

75#if ENABLE(RUBBER_BANDING)

79
80 virtual bool isPinnedForScrollDelta(const FloatSize&) const = 0;
81

84 virtual bool allowsHorizontalScrolling() const = 0;

86 virtual bool shouldRubberBandInDirection(ScrollDirection) const = 0;
87
88 // FIXME: use ScrollClamping to collapse these to one.

92 virtual void willStartRubberBandSnapAnimation() { }
93 virtual void didStopRubberbandSnapAnimation() { }
94
95 virtual void rubberBandingStateChanged(bool) { }
96

101

105 virtual FloatPoint scrollOffset() const = 0;
106 virtual void immediateScrollOnAxis(ScrollEventAxis, float delta) = 0;
107 virtual void willStartScrollSnapAnimation() { }
108 virtual void didStopScrollSnapAnimation() { }
109 virtual float pageScaleFactor() const = 0;
110 virtual LayoutSize scrollExtent() const = 0;
111 virtual FloatSize viewportSize() const = 0;
112};
