58using FramesPerSecond = unsigned;

61class ScrollingCoordinator : public ThreadSafeRefCounted<ScrollingCoordinator> {
62 WTF_MAKE_FAST_ALLOCATED;
63public:
64 static Ref<ScrollingCoordinator> create(Page*);
65 virtual ~ScrollingCoordinator();
66
67 WEBCORE_EXPORT virtual void pageDestroyed();
68
69 virtual bool isAsyncScrollingCoordinator() const { return false; }
70 virtual bool isRemoteScrollingCoordinator() const { return false; }
71
72 // Return whether this scrolling coordinator handles scrolling for the given frame view.
73 WEBCORE_EXPORT virtual bool coordinatesScrollingForFrameView(const FrameView&) const;
74
75 // Return whether this scrolling coordinator handles scrolling for the given overflow scroll layer.
76 WEBCORE_EXPORT virtual bool coordinatesScrollingForOverflowLayer(const RenderLayer&) const;
77
78 // Returns the ScrollingNodeID of the innermost scrolling node that scrolls the renderer.
79 WEBCORE_EXPORT virtual ScrollingNodeID scrollableContainerNodeID(const RenderObject&) const;
80
81 // Should be called whenever the given frame view has been laid out.
82 virtual void frameViewLayoutUpdated(FrameView&) { }
83
84 using LayoutViewportOriginOrOverrideRect = WTF::Variant<std::optional<FloatPoint>, std::optional<FloatRect>>;
85 virtual void reconcileScrollingState(FrameView&, const FloatPoint&, const LayoutViewportOriginOrOverrideRect&, ScrollType, ViewportRectStability, ScrollingLayerPositionAction) { }
86
87 // Should be called whenever the set of fixed objects changes.
88 void frameViewFixedObjectsDidChange(FrameView&);
89
90 // Should be called whenever the FrameView's visual viewport changed.
91 virtual void frameViewVisualViewportChanged(FrameView&) { }
92
93 // Called whenever the non-fast scrollable region changes for reasons other than layout.
94 virtual void frameViewEventTrackingRegionsChanged(FrameView&) { }
95
96 // Should be called whenever the root layer for the given frame view changes.
97 virtual void frameViewRootLayerDidChange(FrameView&);
98
99 // Traverses the scrolling tree, setting layer positions to represent the current scrolled state.
100 virtual void applyScrollingTreeLayerPositions() { }
101
102 virtual void willStartRenderingUpdate() { }
103 virtual void didCompleteRenderingUpdate() { }
104
105#if ENABLE(KINETIC_SCROLLING)
106 // Dispatched by the scrolling tree during handleWheelEvent. This is required as long as scrollbars are painted on the main thread.
107 virtual void handleWheelEventPhase(ScrollingNodeID, PlatformWheelEventPhase) { }
108#endif
109
110 // Force all scroll layer position updates to happen on the main thread.
111 WEBCORE_EXPORT void setForceSynchronousScrollLayerPositionUpdates(bool);
112
113 // These virtual functions are currently unique to the threaded scrolling architecture.
114 virtual void commitTreeStateIfNeeded() { }
115 virtual bool requestScrollPositionUpdate(ScrollableArea&, const IntPoint&, ScrollType = ScrollType::Programmatic, ScrollClamping = ScrollClamping::Clamped) { return false; }
116 virtual bool handleWheelEventForScrolling(const PlatformWheelEvent&, ScrollingNodeID, std::optional<WheelScrollGestureState>) { return false; }
117 virtual void wheelEventWasProcessedByMainThread(const PlatformWheelEvent&, std::optional<WheelScrollGestureState>) { }
118
119 // Create an unparented node.
120 virtual ScrollingNodeID createNode(ScrollingNodeType, ScrollingNodeID newNodeID) { return newNodeID; }
121 // Parent a node in the scrolling tree. This may return a new nodeID if the node type changed. parentID = 0 sets the root node.
122 virtual ScrollingNodeID insertNode(ScrollingNodeType, ScrollingNodeID newNodeID, ScrollingNodeID /*parentID*/, size_t /*childIndex*/ = notFound) { return newNodeID; }
123 // Node will be unparented, but not destroyed. It's the client's responsibility to either re-parent or destroy this node.
124 virtual void unparentNode(ScrollingNodeID) { }
125 // Node will be destroyed, and its children left unparented.
126 virtual void unparentChildrenAndDestroyNode(ScrollingNodeID) { }
127 // Node will be unparented, and it and its children destroyed.
128 virtual void detachAndDestroySubtree(ScrollingNodeID) { }
129 // Destroy the tree, including both parented and unparented nodes.
130 virtual void clearAllNodes() { }
131
132 virtual ScrollingNodeID parentOfNode(ScrollingNodeID) const { return 0; }
133 virtual Vector<ScrollingNodeID> childrenOfNode(ScrollingNodeID) const { return { }; }
134
135 virtual void scrollBySimulatingWheelEventForTesting(ScrollingNodeID, FloatSize) { }
136
137 struct NodeLayers {
138 GraphicsLayer* layer { nullptr };
139 GraphicsLayer* scrollContainerLayer { nullptr };
140 GraphicsLayer* scrolledContentsLayer { nullptr };
141 GraphicsLayer* counterScrollingLayer { nullptr };
142 GraphicsLayer* insetClipLayer { nullptr };
143 GraphicsLayer* rootContentsLayer { nullptr };
144 GraphicsLayer* horizontalScrollbarLayer { nullptr };
145 GraphicsLayer* verticalScrollbarLayer { nullptr };
146 };
147 virtual void setNodeLayers(ScrollingNodeID, const NodeLayers&) { }
148
149 virtual void setScrollingNodeScrollableAreaGeometry(ScrollingNodeID, ScrollableArea&) { }
150 virtual void setFrameScrollingNodeState(ScrollingNodeID, const FrameView&) { }
151 virtual void setViewportConstraintedNodeConstraints(ScrollingNodeID, const ViewportConstraints&) { }
152 virtual void setPositionedNodeConstraints(ScrollingNodeID, const AbsolutePositionConstraints&) { }
153 virtual void setRelatedOverflowScrollingNodes(ScrollingNodeID, Vector<ScrollingNodeID>&&) { }
154 virtual void setSynchronousScrollingReasons(ScrollingNodeID, OptionSet<SynchronousScrollingReason>) { }
155 virtual OptionSet<SynchronousScrollingReason> synchronousScrollingReasons(ScrollingNodeID) const { return { }; };
156 bool hasSynchronousScrollingReasons(ScrollingNodeID nodeID) const { return !!synchronousScrollingReasons(nodeID); }
157
158 virtual void reconcileViewportConstrainedLayerPositions(ScrollingNodeID, const LayoutRect&, ScrollingLayerPositionAction) { }
159 virtual String scrollingStateTreeAsText(ScrollingStateTreeAsTextBehavior = ScrollingStateTreeAsTextBehaviorNormal) const;
160 virtual String scrollingTreeAsText(ScrollingStateTreeAsTextBehavior = ScrollingStateTreeAsTextBehaviorNormal) const;
161 virtual bool isRubberBandInProgress(ScrollingNodeID) const { return false; }
162 virtual bool isUserScrollInProgress(ScrollingNodeID) const { return false; }
163 virtual bool isScrollSnapInProgress(ScrollingNodeID) const { return false; }
164 virtual void updateScrollSnapPropertiesWithFrameView(const FrameView&) { }
165 virtual void setScrollPinningBehavior(ScrollPinningBehavior) { }
166 virtual bool hasSubscrollers() const { return false; }
167
168 // Generated a unique id for scrolling nodes.
169 ScrollingNodeID uniqueScrollingNodeID();
170
171 bool shouldUpdateScrollLayerPositionSynchronously(const FrameView&) const;
172
173 virtual void willDestroyScrollableArea(ScrollableArea&) { }
174 virtual void scrollableAreaScrollbarLayerDidChange(ScrollableArea&, ScrollbarOrientation) { }
175
176 virtual void windowScreenDidChange(PlatformDisplayID, std::optional<FramesPerSecond> /* nominalFramesPerSecond */) { }
177
178 static String synchronousScrollingReasonsAsText(OptionSet<SynchronousScrollingReason>);
179 String synchronousScrollingReasonsAsText() const;
180
181 EventTrackingRegions absoluteEventTrackingRegions() const;
182 virtual void updateIsMonitoringWheelEventsForFrameView(const FrameView&) { }
183
184 virtual void startMonitoringWheelEvents(bool /* clearLatchingState */) { }
185 virtual void stopMonitoringWheelEvents() { }
186
187protected:
188 explicit ScrollingCoordinator(Page*);
189
190 GraphicsLayer* scrollContainerLayerForFrameView(FrameView&);
191 GraphicsLayer* scrolledContentsLayerForFrameView(FrameView&);
192 GraphicsLayer* counterScrollingLayerForFrameView(FrameView&);
193 GraphicsLayer* insetClipLayerForFrameView(FrameView&);
194 GraphicsLayer* rootContentsLayerForFrameView(FrameView&);
195 GraphicsLayer* contentShadowLayerForFrameView(FrameView&);
196 GraphicsLayer* headerLayerForFrameView(FrameView&);
197 GraphicsLayer* footerLayerForFrameView(FrameView&);
198
199 virtual void willCommitTree() { }
200
201 Page* m_page; // FIXME: ideally this would be a reference but it gets nulled on async teardown.
202
203private:
204 virtual bool hasVisibleSlowRepaintViewportConstrainedObjects(const FrameView&) const;
205
206 void updateSynchronousScrollingReasons(FrameView&);
207 void updateSynchronousScrollingReasonsForAllFrames();
208
209 EventTrackingRegions absoluteEventTrackingRegionsForFrame(const Frame&) const;
210
211 bool m_forceSynchronousScrollLayerPositionUpdates { false };
212};
