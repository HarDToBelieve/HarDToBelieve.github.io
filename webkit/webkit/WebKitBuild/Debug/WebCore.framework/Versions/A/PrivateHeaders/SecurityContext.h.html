30#include <memory>
31#include <wtf/Forward.h>
32#include <wtf/OptionSet.h>
33#include <wtf/RefPtr.h>

35namespace WebCore {
36
37class SecurityOrigin;
38class SecurityOriginPolicy;
39class ContentSecurityPolicy;
40
41enum SandboxFlag {
42 // See http://www.whatwg.org/specs/web-apps/current-work/#attr-iframe-sandbox for a list of the sandbox flags.
43 SandboxNone = 0,
44 SandboxNavigation = 1,
45 SandboxPlugins = 1 << 1,
46 SandboxOrigin = 1 << 2,
47 SandboxForms = 1 << 3,
48 SandboxScripts = 1 << 4,
49 SandboxTopNavigation = 1 << 5,
50 SandboxPopups = 1 << 6, // See https://www.w3.org/Bugs/Public/show_bug.cgi?id=12393
51 SandboxAutomaticFeatures = 1 << 7,
52 SandboxPointerLock = 1 << 8,
53 SandboxPropagatesToAuxiliaryBrowsingContexts = 1 << 9,
54 SandboxTopNavigationByUserActivation = 1 << 10,
55 SandboxDocumentDomain = 1 << 11,
56 SandboxModals = 1 << 12,
57 SandboxStorageAccessByUserActivation = 1 << 13,
58 SandboxAll = -1 // Mask with all bits set to 1.
59};
60
61typedef int SandboxFlags;
62
63class SecurityContext {
64public:
65 SandboxFlags sandboxFlags() const { return m_sandboxFlags; }
66 ContentSecurityPolicy* contentSecurityPolicy() { return m_contentSecurityPolicy.get(); }
67
68 bool isSecureTransitionTo(const URL&) const;
69
70 void enforceSandboxFlags(SandboxFlags mask);
71 bool isSandboxed(SandboxFlags mask) const { return m_sandboxFlags & mask; }
72
73 SecurityOriginPolicy* securityOriginPolicy() const { return m_securityOriginPolicy.get(); }
74
75 // Explicitly override the security origin for this security context.
76 // Note: It is dangerous to change the security origin of a script context
77 // that already contains content.
78 void setSecurityOriginPolicy(RefPtr<SecurityOriginPolicy>&&);
79
80 // Explicitly override the content security policy for this security context.
81 // Note: It is dangerous to change the content security policy of a script
82 // context that already contains content.
83 void setContentSecurityPolicy(std::unique_ptr<ContentSecurityPolicy>&&);
84
85 WEBCORE_EXPORT SecurityOrigin* securityOrigin() const;
86
87 static SandboxFlags parseSandboxPolicy(const String& policy, String& invalidTokensErrorMessage);
88 static bool isSupportedSandboxPolicy(StringView);
89
90 enum MixedContentType {
91 Inactive = 1 << 0,
92 Active = 1 << 1,
93 };
94
95 bool usedLegacyTLS() const { return m_usedLegacyTLS; }
96 void setUsedLegacyTLS(bool used) { m_usedLegacyTLS = used; }
97 const OptionSet<MixedContentType>& foundMixedContent() const { return m_mixedContentTypes; }
98 void setFoundMixedContent(MixedContentType type) { m_mixedContentTypes.add(type); }
99 bool geolocationAccessed() const { return m_geolocationAccessed; }
100 void setGeolocationAccessed() { m_geolocationAccessed = true; }
101 bool secureCookiesAccessed() const { return m_secureCookiesAccessed; }
102 void setSecureCookiesAccessed() { m_secureCookiesAccessed = true; }
103
104 bool isStrictMixedContentMode() const { return m_isStrictMixedContentMode; }
105 void setStrictMixedContentMode(bool strictMixedContentMode) { m_isStrictMixedContentMode = strictMixedContentMode; }
106
107 // This method implements the "Is the environment settings object settings a secure context?" algorithm from
108 // the Secure Context spec: https://w3c.github.io/webappsec-secure-contexts/#settings-object (Editor's Draft, 17 November 2016)
109 virtual bool isSecureContext() const = 0;
110
111 bool haveInitializedSecurityOrigin() const { return m_haveInitializedSecurityOrigin; }
112
113protected:
114 SecurityContext();
115 virtual ~SecurityContext();
116
117 // It's only appropriate to call this during security context initialization; it's needed for
118 // flags that can't be disabled with allow-* attributes, such as SandboxNavigation.
119 void disableSandboxFlags(SandboxFlags mask) { m_sandboxFlags &= ~mask; }
120
121 void didFailToInitializeSecurityOrigin() { m_haveInitializedSecurityOrigin = false; }
122
123private:
124 RefPtr<SecurityOriginPolicy> m_securityOriginPolicy;
125 std::unique_ptr<ContentSecurityPolicy> m_contentSecurityPolicy;
126 SandboxFlags m_sandboxFlags { SandboxNone };
127 OptionSet<MixedContentType> m_mixedContentTypes;
128 bool m_haveInitializedSecurityOrigin { false };
129 bool m_geolocationAccessed { false };
130 bool m_secureCookiesAccessed { false };
131 bool m_isStrictMixedContentMode { false };
132 bool m_usedLegacyTLS { false };
133};
134
135} // namespace WebCore
