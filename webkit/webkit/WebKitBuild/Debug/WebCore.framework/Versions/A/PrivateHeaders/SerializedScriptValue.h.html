157template<class Encoder>
158void SerializedScriptValue::encode(Encoder& encoder) const
159{
160 encoder << m_data;
161
162 auto hasArray = m_arrayBufferContentsArray && m_arrayBufferContentsArray->size();
163 encoder << hasArray;
164
165 if (hasArray) {
166 encoder << static_cast<uint64_t>(m_arrayBufferContentsArray->size());
167 for (const auto &arrayBufferContents : *m_arrayBufferContentsArray) {
168 encoder << arrayBufferContents.sizeInBytes();
169 encoder.encodeFixedLengthData(static_cast<const uint8_t*>(arrayBufferContents.data()), arrayBufferContents.sizeInBytes(), 1);
170 }
171 }
172
173#if ENABLE(WEB_RTC)
174 encoder << static_cast<uint64_t>(m_detachedRTCDataChannels.size());
175 for (const auto &channel : m_detachedRTCDataChannels)
176 encoder << *channel;
177#endif
178}

180template<class Decoder>
181RefPtr<SerializedScriptValue> SerializedScriptValue::decode(Decoder& decoder)
182{
183 Vector<uint8_t> data;
184 if (!decoder.decode(data))
185 return nullptr;
186
187 bool hasArray;
188 if (!decoder.decode(hasArray))
189 return nullptr;
190
191 std::unique_ptr<ArrayBufferContentsArray> arrayBufferContentsArray;
192 if (hasArray) {
193 uint64_t arrayLength;
194 if (!decoder.decode(arrayLength))
195 return nullptr;
196 ASSERT(arrayLength);
197
198 arrayBufferContentsArray = makeUnique<ArrayBufferContentsArray>();
199 while (arrayLength--) {
200 unsigned bufferSize;
201 if (!decoder.decode(bufferSize))
202 return nullptr;
203 if (!decoder.template bufferIsLargeEnoughToContain<uint8_t>(bufferSize))
204 return nullptr;
205
206 auto buffer = Gigacage::tryMalloc(Gigacage::Primitive, bufferSize);
207 if (!buffer)
208 return nullptr;
209 if (!decoder.decodeFixedLengthData(static_cast<uint8_t*>(buffer), bufferSize, 1)) {
210 Gigacage::free(Gigacage::Primitive, buffer);
211 return nullptr;
212 }
213 arrayBufferContentsArray->append({ buffer, bufferSize, ArrayBuffer::primitiveGigacageDestructor() });
214 }
215 }
216
217#if ENABLE(WEB_RTC)
218 uint64_t detachedRTCDataChannelsSize;
219 if (!decoder.decode(detachedRTCDataChannelsSize))
220 return nullptr;
221
222 Vector<std::unique_ptr<DetachedRTCDataChannel>> detachedRTCDataChannels;
223 while (detachedRTCDataChannelsSize--) {
224 auto detachedRTCDataChannel = DetachedRTCDataChannel::decode(decoder);
225 if (!detachedRTCDataChannel)
226 return nullptr;
227 detachedRTCDataChannels.append(WTFMove(detachedRTCDataChannel));
228 }
229#endif
230
231 return adoptRef(*new SerializedScriptValue(WTFMove(data), WTFMove(arrayBufferContentsArray)
232#if ENABLE(WEB_RTC)
233 , WTFMove(detachedRTCDataChannels)
234#endif
235 ));
236}
