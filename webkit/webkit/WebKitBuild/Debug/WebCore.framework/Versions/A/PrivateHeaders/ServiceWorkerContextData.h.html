40namespace WebCore {
41
42enum class LastNavigationWasAppBound : bool;
43
44struct ServiceWorkerContextData {
45 struct ImportedScript {
46 ScriptBuffer script;
47 URL responseURL;
48 String mimeType;
49
50 template<class Encoder> void encode(Encoder& encoder) const
51 {
52 encoder << script << responseURL << mimeType;
53 }
54
55 template<class Decoder> static std::optional<ImportedScript> decode(Decoder& decoder)
56 {
57 std::optional<ScriptBuffer> script;
58 decoder >> script;
59 if (!script)
60 return std::nullopt;
61
62 std::optional<URL> responseURL;
63 decoder >> responseURL;
64 if (!responseURL)
65 return std::nullopt;
66
67 std::optional<String> mimeType;
68 decoder >> mimeType;
69 if (!mimeType)
70 return std::nullopt;
71
72 return {{
73 WTFMove(*script),
74 WTFMove(*responseURL),
75 WTFMove(*mimeType)
76 }};
77 }
78
79 ImportedScript isolatedCopy() const { return { script.isolatedCopy(), responseURL.isolatedCopy(), mimeType.isolatedCopy() }; }
80 };
81
82 std::optional<ServiceWorkerJobDataIdentifier> jobDataIdentifier;
83 ServiceWorkerRegistrationData registration;
84 ServiceWorkerIdentifier serviceWorkerIdentifier;
85 ScriptBuffer script;
86 CertificateInfo certificateInfo;
87 ContentSecurityPolicyResponseHeaders contentSecurityPolicy;
88 String referrerPolicy;
89 URL scriptURL;
90 WorkerType workerType;
91 bool loadedFromDisk;
92 std::optional<LastNavigationWasAppBound> lastNavigationWasAppBound;
93 HashMap<URL, ImportedScript> scriptResourceMap;
94
95 template<class Encoder> void encode(Encoder&) const;
96 template<class Decoder> static std::optional<ServiceWorkerContextData> decode(Decoder&);
97
98 ServiceWorkerContextData isolatedCopy() const;
99};
100
101template<class Encoder>
102void ServiceWorkerContextData::encode(Encoder& encoder) const
103{
104 encoder << jobDataIdentifier << registration << serviceWorkerIdentifier << script << contentSecurityPolicy << referrerPolicy
105 << scriptURL << workerType << loadedFromDisk << lastNavigationWasAppBound << scriptResourceMap << certificateInfo;
106}
107
108template<class Decoder>
109std::optional<ServiceWorkerContextData> ServiceWorkerContextData::decode(Decoder& decoder)
110{
111 std::optional<std::optional<ServiceWorkerJobDataIdentifier>> jobDataIdentifier;
112 decoder >> jobDataIdentifier;
113 if (!jobDataIdentifier)
114 return std::nullopt;
115
116 std::optional<ServiceWorkerRegistrationData> registration;
117 decoder >> registration;
118 if (!registration)
119 return std::nullopt;
120
121 auto serviceWorkerIdentifier = ServiceWorkerIdentifier::decode(decoder);
122 if (!serviceWorkerIdentifier)
123 return std::nullopt;
124
125 std::optional<ScriptBuffer> script;
126 decoder >> script;
127 if (!script)
128 return std::nullopt;
129
130 ContentSecurityPolicyResponseHeaders contentSecurityPolicy;
131 if (!decoder.decode(contentSecurityPolicy))
132 return std::nullopt;
133
134 String referrerPolicy;
135 if (!decoder.decode(referrerPolicy))
136 return std::nullopt;
137
138 URL scriptURL;
139 if (!decoder.decode(scriptURL))
140 return std::nullopt;
141
142 WorkerType workerType;
143 if (!decoder.decode(workerType))
144 return std::nullopt;
145
146 bool loadedFromDisk;
147 if (!decoder.decode(loadedFromDisk))
148 return std::nullopt;
149
150 std::optional<LastNavigationWasAppBound> lastNavigationWasAppBound;
151 if (!decoder.decode(lastNavigationWasAppBound))
152 return std::nullopt;
153
154 HashMap<URL, ImportedScript> scriptResourceMap;
155 if (!decoder.decode(scriptResourceMap))
156 return std::nullopt;
157
158 std::optional<CertificateInfo> certificateInfo;
159 decoder >> certificateInfo;
160 if (!certificateInfo)
161 return std::nullopt;
162
163 return {{
164 WTFMove(*jobDataIdentifier),
165 WTFMove(*registration),
166 WTFMove(*serviceWorkerIdentifier),
167 WTFMove(*script),
168 WTFMove(*certificateInfo),
169 WTFMove(contentSecurityPolicy),
170 WTFMove(referrerPolicy),
171 WTFMove(scriptURL),
172 workerType,
173 loadedFromDisk,
174 WTFMove(lastNavigationWasAppBound),
175 WTFMove(scriptResourceMap)
176 }};
177}
178
179} // namespace WebCore
