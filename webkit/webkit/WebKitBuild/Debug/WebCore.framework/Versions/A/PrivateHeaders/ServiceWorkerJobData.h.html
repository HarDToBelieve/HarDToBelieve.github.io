39namespace WebCore {
40
41struct ServiceWorkerJobData {
42 using Identifier = ServiceWorkerJobDataIdentifier;
43 ServiceWorkerJobData(SWServerConnectionIdentifier, const DocumentOrWorkerIdentifier& sourceContext);
44 ServiceWorkerJobData(Identifier, const DocumentOrWorkerIdentifier& sourceContext);
45
46 SWServerConnectionIdentifier connectionIdentifier() const { return m_identifier.connectionIdentifier; }
47
48 bool isEquivalent(const ServiceWorkerJobData&) const;
49
50 URL scriptURL;
51 URL clientCreationURL;
52 SecurityOriginData topOrigin;
53 URL scopeURL;
54 ServiceWorkerOrClientIdentifier sourceContext;
55 WorkerType workerType;
56 ServiceWorkerJobType type;
57
58 ServiceWorkerRegistrationOptions registrationOptions;
59
60 Identifier identifier() const { return m_identifier; }
61 WEBCORE_EXPORT ServiceWorkerRegistrationKey registrationKey() const;
62 ServiceWorkerJobData isolatedCopy() const;
63
64 template<class Encoder> void encode(Encoder&) const;
65 template<class Decoder> static std::optional<ServiceWorkerJobData> decode(Decoder&);
66
67private:
68 ServiceWorkerJobData() = default;
69
70 Identifier m_identifier;
71};
72
73template<class Encoder>
74void ServiceWorkerJobData::encode(Encoder& encoder) const
75{
76 encoder << identifier() << scriptURL << clientCreationURL << topOrigin << scopeURL << sourceContext << workerType;
77 encoder << type;
78 switch (type) {
79 case ServiceWorkerJobType::Register:
80 encoder << registrationOptions;
81 break;
82 case ServiceWorkerJobType::Unregister:
83 case ServiceWorkerJobType::Update:
84 break;
85 }
86}
87
88template<class Decoder>
89std::optional<ServiceWorkerJobData> ServiceWorkerJobData::decode(Decoder& decoder)
90{
91 std::optional<ServiceWorkerJobDataIdentifier> identifier;
92 decoder >> identifier;
93 if (!identifier)
94 return std::nullopt;
95
96 ServiceWorkerJobData jobData;
97 jobData.m_identifier = *identifier;
98
99 if (!decoder.decode(jobData.scriptURL))
100 return std::nullopt;
101 if (!decoder.decode(jobData.clientCreationURL))
102 return std::nullopt;
103
104 std::optional<SecurityOriginData> topOrigin;
105 decoder >> topOrigin;
106 if (!topOrigin)
107 return std::nullopt;
108 jobData.topOrigin = WTFMove(*topOrigin);
109
110 if (!decoder.decode(jobData.scopeURL))
111 return std::nullopt;
112 if (!decoder.decode(jobData.sourceContext))
113 return std::nullopt;
114 if (!decoder.decode(jobData.workerType))
115 return std::nullopt;
116 if (!decoder.decode(jobData.type))
117 return std::nullopt;
118
119 switch (jobData.type) {
120 case ServiceWorkerJobType::Register: {
121 std::optional<ServiceWorkerRegistrationOptions> registrationOptions;
122 decoder >> registrationOptions;
123 if (!registrationOptions)
124 return std::nullopt;
125 jobData.registrationOptions = WTFMove(*registrationOptions);
126 break;
127 }
128 case ServiceWorkerJobType::Unregister:
129 case ServiceWorkerJobType::Update:
130 break;
131 }
132
133 return jobData;
134}
