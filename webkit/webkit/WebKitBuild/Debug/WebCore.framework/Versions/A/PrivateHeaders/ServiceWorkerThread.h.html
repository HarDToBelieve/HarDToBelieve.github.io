38namespace WebCore {
39
40class CacheStorageProvider;
41class ContentSecurityPolicyResponseHeaders;
42class ExtendableEvent;
43class MessagePortChannel;
44class SerializedScriptValue;
45class WorkerObjectProxy;
46struct MessageWithMessagePorts;
47struct ServiceWorkerClientIdentifier;
48
49class ServiceWorkerThread : public WorkerThread, public CanMakeWeakPtr<ServiceWorkerThread, WeakPtrFactoryInitialization::Eager> {
50public:
51 template<typename... Args> static Ref<ServiceWorkerThread> create(Args&&... args)
52 {
53 return adoptRef(*new ServiceWorkerThread(std::forward<Args>(args)...));
54 }
55 virtual ~ServiceWorkerThread();
56
57 WorkerObjectProxy& workerObjectProxy() const { return m_workerObjectProxy; }
58
59 void start(Function<void(const String&, bool)>&&);
60
61 void willPostTaskToFireInstallEvent();
62 void willPostTaskToFireActivateEvent();
63 void willPostTaskToFireMessageEvent();
64
65 void queueTaskToFireFetchEvent(Ref<ServiceWorkerFetch::Client>&&, std::optional<ServiceWorkerClientIdentifier>&&, ResourceRequest&&, String&& referrer, FetchOptions&&);
66 void queueTaskToPostMessage(MessageWithMessagePorts&&, ServiceWorkerOrClientData&& sourceData);
67 void queueTaskToFireInstallEvent();
68 void queueTaskToFireActivateEvent();
69
70 ServiceWorkerIdentifier identifier() const { return m_serviceWorkerIdentifier; }
71 std::optional<ServiceWorkerJobDataIdentifier> jobDataIdentifier() const { return m_jobDataIdentifier; }
72 bool doesHandleFetch() const { return m_doesHandleFetch; }
73
74 void startFetchEventMonitoring();
75 void stopFetchEventMonitoring() { m_isHandlingFetchEvent = false; }
76
77protected:
78 Ref<WorkerGlobalScope> createWorkerGlobalScope(const WorkerParameters&, Ref<SecurityOrigin>&&, Ref<SecurityOrigin>&& topOrigin) final;
79 void runEventLoop() override;
80
81private:
82 WEBCORE_EXPORT ServiceWorkerThread(ServiceWorkerContextData&&, String&& userAgent, const Settings::Values&, WorkerLoaderProxy&, WorkerDebuggerProxy&, IDBClient::IDBConnectionProxy*, SocketProvider*);
83
84 bool isServiceWorkerThread() const final { return true; }
85 void finishedEvaluatingScript() final;
86
87 void finishedFiringInstallEvent(bool hasRejectedAnyPromise);
88 void finishedFiringActivateEvent();
89 void finishedFiringMessageEvent();
90 void finishedStarting();
91
92 void startHeartBeatTimer();
93 void heartBeatTimerFired();
94 void installEventTimerFired();
95
96 ServiceWorkerIdentifier m_serviceWorkerIdentifier;
97 std::optional<ServiceWorkerJobDataIdentifier> m_jobDataIdentifier;
98 std::optional<ServiceWorkerContextData> m_data; // Becomes std::nullopt after the ServiceWorkerGlobalScope has been created.
99 WorkerObjectProxy& m_workerObjectProxy;
100 bool m_doesHandleFetch { false };
101
102 bool m_isHandlingFetchEvent { false };
103 uint64_t m_messageEventCount { 0 };
104 enum class State { Idle, Starting, Installing, Activating };
105 State m_state { State::Idle };
106 bool m_ongoingHeartBeatCheck { false };
107
108 static constexpr Seconds heartBeatTimeout { 60_s };
109 static constexpr Seconds heartBeatTimeoutForTest { 1_s };
110 Seconds m_heartBeatTimeout { heartBeatTimeout };
111 Timer m_heartBeatTimer;
112};
