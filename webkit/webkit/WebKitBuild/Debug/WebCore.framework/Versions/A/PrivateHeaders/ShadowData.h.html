26
27#include "Color.h"
28#include "FloatRect.h"
29#include "LayoutRect.h"
30
31namespace WebCore {
32
33enum class ShadowStyle : uint8_t { Normal, Inset };
34
35// This class holds information about shadows for the text-shadow and box-shadow properties.
36
37class ShadowData {
38 WTF_MAKE_FAST_ALLOCATED;
39public:
40 ShadowData() = default;
41
42 ShadowData(const LayoutPoint& location, int radius, LayoutUnit spread, ShadowStyle style, bool isWebkitBoxShadow, const Color& color)
43 : m_location(location)
44 , m_spread(spread)
45 , m_radius(radius)
46 , m_color(color)
47 , m_style(style)
48 , m_isWebkitBoxShadow(isWebkitBoxShadow)
49 {
50 }
51
52 ShadowData(const ShadowData&);
53 static std::optional<ShadowData> clone(const ShadowData*);
54
55 ShadowData& operator=(ShadowData&&) = default;
56
57 bool operator==(const ShadowData& o) const;
58 bool operator!=(const ShadowData& o) const
59 {
60 return !(*this == o);
61 }
62
63 LayoutUnit x() const { return m_location.x(); }
64 LayoutUnit y() const { return m_location.y(); }
65 LayoutPoint location() const { return m_location; }
66 int radius() const { return m_radius; }
67 LayoutUnit paintingExtent() const
68 {
69 // Blurring uses a Gaussian function whose std. deviation is m_radius/2, and which in theory
70 // extends to infinity. In 8-bit contexts, however, rounding causes the effect to become
71 // undetectable at around 1.4x the radius.
72 const float radiusExtentMultiplier = 1.4;
73 return LayoutUnit(ceilf(m_radius * radiusExtentMultiplier));
74 }
75 LayoutUnit spread() const { return m_spread; }
76 ShadowStyle style() const { return m_style; }
77
78 void setColor(const Color& color) { m_color = color; }
79 const Color& color() const { return m_color; }
80
81 bool isWebkitBoxShadow() const { return m_isWebkitBoxShadow; }
82
83 const ShadowData* next() const { return m_next.get(); }
84 void setNext(std::unique_ptr<ShadowData>&& shadow) { m_next = WTFMove(shadow); }
85
86 void adjustRectForShadow(LayoutRect&, int additionalOutlineSize = 0) const;
87 void adjustRectForShadow(FloatRect&, int additionalOutlineSize = 0) const;
88
89private:
90 LayoutPoint m_location;
91 LayoutUnit m_spread;
92 int m_radius { 0 }; // This is the "blur radius", or twice the standard deviation of the Gaussian blur.
93 Color m_color;
94 ShadowStyle m_style { ShadowStyle::Normal };
95 bool m_isWebkitBoxShadow { false };
96 std::unique_ptr<ShadowData> m_next;
97};
98
99WTF::TextStream& operator<<(WTF::TextStream&, const ShadowData&);
100
101} // namespace WebCore
