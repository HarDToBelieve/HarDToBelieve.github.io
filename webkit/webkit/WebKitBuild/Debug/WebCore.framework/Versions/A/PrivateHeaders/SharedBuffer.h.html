67class WEBCORE_EXPORT SharedBuffer : public RefCounted<SharedBuffer> {

69 static Ref<SharedBuffer> create() { return adoptRef(*new SharedBuffer); }

71 static Ref<SharedBuffer> create(const char* data, size_t size) { return adoptRef(*new SharedBuffer(data, size)); }

74 enum class MayUseFileMapping : bool { No, Yes };

79#if USE(FOUNDATION)
80 RetainPtr<NSData> createNSData() const;

97 static Ref<SharedBuffer> create(GstMappedOwnedBuffer&);

99 // Calling data() causes all the data segments to be copied into one segment if they are not already.
100 // Iterate the segments using begin() and end() instead.
101 // FIXME: Audit the call sites of this function and replace them with iteration if possible.
102 const uint8_t* data() const;
103 const char* dataAsCharPtr() const { return reinterpret_cast<const char*>(data()); }
104 Vector<uint8_t> copyData() { return { data(), size() }; }
105
106 // Combines all the segments into a Vector and returns that vector after clearing the SharedBuffer.
107 Vector<uint8_t> takeData();
108
109 // Creates an ArrayBuffer and copies this SharedBuffer's contents to that
110 // ArrayBuffer without merging segmented buffers into a flat buffer.
111 RefPtr<ArrayBuffer> tryCreateArrayBuffer() const;
112
113 size_t size() const { return m_size; }
114
115 bool isEmpty() const { return !size(); }

127 // Data wrapped by a DataSegment should be immutable because it can be referenced by other objects.
128 // To modify or combine the data, allocate a new DataSegment.
129 class DataSegment : public ThreadSafeRefCounted<DataSegment> {
130 public:
131 WEBCORE_EXPORT const uint8_t* data() const;
132 WEBCORE_EXPORT size_t size() const;
133
134 static Ref<DataSegment> create(Vector<uint8_t>&& data)
135 {
136 data.shrinkToFit();
137 return adoptRef(*new DataSegment(WTFMove(data)));
138 }
139
140#if USE(CF)
141 static Ref<DataSegment> create(RetainPtr<CFDataRef>&& data) { return adoptRef(*new DataSegment(WTFMove(data))); }
142#endif
143#if USE(GLIB)
144 static Ref<DataSegment> create(GRefPtr<GBytes>&& data) { return adoptRef(*new DataSegment(WTFMove(data))); }
145#endif
146#if USE(GSTREAMER)
147 static Ref<DataSegment> create(RefPtr<GstMappedOwnedBuffer>&& data) { return adoptRef(*new DataSegment(WTFMove(data))); }
148#endif
149 static Ref<DataSegment> create(FileSystem::MappedFileData&& data) { return adoptRef(*new DataSegment(WTFMove(data))); }
150
151#if USE(FOUNDATION)
152 RetainPtr<NSData> createNSData() const;
153#endif
154
155 bool containsMappedFileData() const;
156
157 private:
158 DataSegment(Vector<uint8_t>&& data)
159 : m_immutableData(WTFMove(data)) { }
160#if USE(CF)
161 DataSegment(RetainPtr<CFDataRef>&& data)
162 : m_immutableData(WTFMove(data)) { }
163#endif
164#if USE(GLIB)
165 DataSegment(GRefPtr<GBytes>&& data)
166 : m_immutableData(WTFMove(data)) { }
167#endif
168#if USE(GSTREAMER)
169 DataSegment(RefPtr<GstMappedOwnedBuffer>&& data)
170 : m_immutableData(WTFMove(data)) { }
171#endif
172 DataSegment(FileSystem::MappedFileData&& data)
173 : m_immutableData(WTFMove(data)) { }
174
175 Variant<Vector<uint8_t>,
176#if USE(CF)
177 RetainPtr<CFDataRef>,
178#endif
179#if USE(GLIB)
180 GRefPtr<GBytes>,
181#endif
182#if USE(GSTREAMER)
183 RefPtr<GstMappedOwnedBuffer>,
184#endif
185 FileSystem::MappedFileData> m_immutableData;
186 friend class SharedBuffer;
187 };
