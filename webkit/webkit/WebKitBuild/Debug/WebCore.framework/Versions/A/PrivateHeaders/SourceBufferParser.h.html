38namespace WTF {
39class Logger;
40}
41
42namespace WebCore {
43
44class ContentType;
45class MediaSample;
46
47class WEBCORE_EXPORT SourceBufferParser : public ThreadSafeRefCounted<SourceBufferParser> {
48public:
49 static MediaPlayerEnums::SupportsType isContentTypeSupported(const ContentType&);
50
51 static RefPtr<SourceBufferParser> create(const ContentType&, bool webMParserEnabled);
52 virtual ~SourceBufferParser() = default;
53
54 enum class Type : uint8_t {
55 AVFObjC,
56 WebM,
57 };
58 virtual Type type() const = 0;
59 enum class AppendFlags : uint8_t {
60 None,
61 Discontinuity,
62 };
63
64 class Segment {
65 public:
66#if HAVE(MT_PLUGIN_FORMAT_READER)
67 Segment(RetainPtr<MTPluginByteSourceRef>&&);
68#endif
69 Segment(Vector<uint8_t>&&);
70 Segment(Segment&&) = default;
71 Vector<uint8_t> takeVector();
72
73 size_t size() const;
74 size_t read(size_t position, size_t, uint8_t* destination) const;
75
76 private:
77 Variant<
78#if HAVE(MT_PLUGIN_FORMAT_READER)
79 RetainPtr<MTPluginByteSourceRef>,
80#endif
81 Vector<uint8_t>
82 > m_segment;
83 };
84
85 using CallOnClientThreadCallback = WTF::Function<void(WTF::Function<void()>&&)>;
86 void setCallOnClientThreadCallback(CallOnClientThreadCallback&&);
87
88 // appendData will be called on the SourceBufferPrivateAVFObjC data parser queue.
89 // Other methods will be called on the main thread, but only once appendData has returned.
90 virtual void appendData(Segment&&, CompletionHandler<void()>&& = [] { }, AppendFlags = AppendFlags::None) = 0;
91 virtual void flushPendingMediaData() = 0;
92 virtual void setShouldProvideMediaDataForTrackID(bool, uint64_t) = 0;
93 virtual bool shouldProvideMediadataForTrackID(uint64_t) = 0;
94 virtual void resetParserState() = 0;
95 virtual void invalidate() = 0;
96 virtual void setMinimumAudioSampleDuration(float);
97#if !RELEASE_LOG_DISABLED
98 virtual void setLogger(const WTF::Logger&, const void* logIdentifier) = 0;
99#endif
100
101 // Will be called on the main thread.
102 using InitializationSegment = SourceBufferPrivateClient::InitializationSegment;
103 using DidParseInitializationDataCallback = WTF::Function<void(InitializationSegment&&)>;
104 void setDidParseInitializationDataCallback(DidParseInitializationDataCallback&& callback)
105 {
106 m_didParseInitializationDataCallback = WTFMove(callback);
107 }
108
109 // Will be called on the main thread.
110 using DidEncounterErrorDuringParsingCallback = WTF::Function<void(uint64_t errorCode)>;
111 void setDidEncounterErrorDuringParsingCallback(DidEncounterErrorDuringParsingCallback&& callback)
112 {
113 m_didEncounterErrorDuringParsingCallback = WTFMove(callback);
114 }
115
116 // Will be called on the main thread.
117 using DidProvideMediaDataCallback = WTF::Function<void(Ref<MediaSample>&&, uint64_t trackID, const String& mediaType)>;
118 void setDidProvideMediaDataCallback(DidProvideMediaDataCallback&& callback)
119 {
120 m_didProvideMediaDataCallback = WTFMove(callback);
121 }
122
123 // Will be called synchronously on the parser thead.
124 using WillProvideContentKeyRequestInitializationDataForTrackIDCallback = WTF::Function<void(uint64_t trackID)>;
125 void setWillProvideContentKeyRequestInitializationDataForTrackIDCallback(WillProvideContentKeyRequestInitializationDataForTrackIDCallback&& callback)
126 {
127 m_willProvideContentKeyRequestInitializationDataForTrackIDCallback = WTFMove(callback);
128 }
129
130 // Will be called synchronously on the parser thead.
131 using DidProvideContentKeyRequestInitializationDataForTrackIDCallback = WTF::Function<void(Ref<Uint8Array>&&, uint64_t trackID)>;
132 void setDidProvideContentKeyRequestInitializationDataForTrackIDCallback(DidProvideContentKeyRequestInitializationDataForTrackIDCallback&& callback)
133 {
134 m_didProvideContentKeyRequestInitializationDataForTrackIDCallback = WTFMove(callback);
135 }
136
137protected:
138 SourceBufferParser();
139
140 CallOnClientThreadCallback m_callOnClientThreadCallback;
141 DidParseInitializationDataCallback m_didParseInitializationDataCallback;
142 DidEncounterErrorDuringParsingCallback m_didEncounterErrorDuringParsingCallback;
143 DidProvideMediaDataCallback m_didProvideMediaDataCallback;
144 WillProvideContentKeyRequestInitializationDataForTrackIDCallback m_willProvideContentKeyRequestInitializationDataForTrackIDCallback;
145 DidProvideContentKeyRequestInitializationDataForTrackIDCallback m_didProvideContentKeyRequestInitializationDataForTrackIDCallback;
146};
147
148}
