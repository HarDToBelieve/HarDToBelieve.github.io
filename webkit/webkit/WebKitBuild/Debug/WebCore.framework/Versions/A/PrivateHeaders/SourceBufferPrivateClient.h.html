30#include "MediaDescription.h"

32#include <wtf/Vector.h>

34namespace WebCore {
35
36class AudioTrackPrivate;
37class InbandTextTrackPrivate;
38class MediaSample;
39class MediaDescription;
40class PlatformTimeRanges;
41class VideoTrackPrivate;
42
43class SourceBufferPrivateClient {
44public:
45 virtual ~SourceBufferPrivateClient() = default;
46
47 struct InitializationSegment {
48 MediaTime duration;
49
50 struct AudioTrackInformation {
51 RefPtr<MediaDescription> description;
52 RefPtr<AudioTrackPrivate> track;
53 };
54 Vector<AudioTrackInformation> audioTracks;
55
56 struct VideoTrackInformation {
57 RefPtr<MediaDescription> description;
58 RefPtr<VideoTrackPrivate> track;
59 };
60 Vector<VideoTrackInformation> videoTracks;
61
62 struct TextTrackInformation {
63 RefPtr<MediaDescription> description;
64 RefPtr<InbandTextTrackPrivate> track;
65 };
66 Vector<TextTrackInformation> textTracks;
67 };
68 virtual void sourceBufferPrivateDidReceiveInitializationSegment(InitializationSegment&&, CompletionHandler<void()>&&) = 0;
69 virtual void sourceBufferPrivateStreamEndedWithDecodeError() = 0;
70 virtual void sourceBufferPrivateAppendError(bool decodeError) = 0;
71 enum class AppendResult : uint8_t {
72 AppendSucceeded,
73 ReadStreamFailed,
74 ParsingFailed
75 };
76 virtual void sourceBufferPrivateAppendComplete(AppendResult) = 0;
77 virtual void sourceBufferPrivateDurationChanged(const MediaTime&) = 0;
78 virtual void sourceBufferPrivateHighestPresentationTimestampChanged(const MediaTime&) = 0;
79 virtual void sourceBufferPrivateDidParseSample(double frameDuration) = 0;
80 virtual void sourceBufferPrivateDidDropSample() = 0;
81 virtual void sourceBufferPrivateBufferedDirtyChanged(bool) = 0;
82 virtual void sourceBufferPrivateDidReceiveRenderingError(int64_t errorCode) = 0;
83 virtual void sourceBufferPrivateReportExtraMemoryCost(uint64_t) = 0;
84};
85
86} // namespace WebCore
87
88namespace WTF {

96 >;
97};
