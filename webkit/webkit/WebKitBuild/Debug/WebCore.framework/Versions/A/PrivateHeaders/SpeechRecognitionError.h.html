26#pragma once
27
28#include <wtf/EnumTraits.h>
29
30namespace WebCore {
31
32enum class SpeechRecognitionErrorType : uint8_t {
33 NoSpeech,
34 Aborted,
35 AudioCapture,
36 Network,
37 NotAllowed,
38 ServiceNotAllowed,
39 BadGrammar,
40 LanguageNotSupported
41};
42
43struct SpeechRecognitionError {
44 SpeechRecognitionErrorType type;
45 String message;
46
47 template<class Encoder> void encode(Encoder&) const;
48 template<class Decoder> static std::optional<SpeechRecognitionError> decode(Decoder&);
49};
50
51template<class Encoder>
52void SpeechRecognitionError::encode(Encoder& encoder) const
53{
54 encoder << type << message;
55}
56
57template<class Decoder>
58std::optional<SpeechRecognitionError> SpeechRecognitionError::decode(Decoder& decoder)
59{
60 std::optional<SpeechRecognitionErrorType> type;
61 decoder >> type;
62 if (!type)
63 return std::nullopt;
64
65 std::optional<String> message;
66 decoder >> message;
67 if (!message)
68 return std::nullopt;
69
70 return SpeechRecognitionError { WTFMove(*type), WTFMove(*message) };
71}
72
73} // namespace WebCore
