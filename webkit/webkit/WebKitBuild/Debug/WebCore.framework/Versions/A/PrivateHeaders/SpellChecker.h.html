40class SpellCheckRequest final : public TextCheckingRequest {
41public:
42 static RefPtr<SpellCheckRequest> create(OptionSet<TextCheckingType>, TextCheckingProcessType, const SimpleRange& checkingRange, const SimpleRange& automaticReplacementRange, const SimpleRange& paragraphRange);
43 virtual ~SpellCheckRequest();
44
45 const SimpleRange& checkingRange() const { return m_checkingRange; }
46 const SimpleRange& paragraphRange() const { return m_paragraphRange; }
47 const SimpleRange& automaticReplacementRange() const { return m_automaticReplacementRange; }
48 Element* rootEditableElement() const { return m_rootEditableElement.get(); }
49
50 void setCheckerAndIdentifier(SpellChecker*, TextCheckingRequestIdentifier);
51 void requesterDestroyed();
52 bool isStarted() const { return m_checker; }
53
54 const TextCheckingRequestData& data() const final;
55
56private:
57 void didSucceed(const Vector<TextCheckingResult>&) final;
58 void didCancel() final;
59
60 SpellCheckRequest(const SimpleRange& checkingRange, const SimpleRange& automaticReplacementRange, const SimpleRange& paragraphRange, const String&, OptionSet<TextCheckingType>, TextCheckingProcessType);
61
62 SpellChecker* m_checker { nullptr };
63 SimpleRange m_checkingRange;
64 SimpleRange m_automaticReplacementRange;
65 SimpleRange m_paragraphRange;
66 RefPtr<Element> m_rootEditableElement;
67 TextCheckingRequestData m_requestData;
68};

70class SpellChecker {
71 WTF_MAKE_FAST_ALLOCATED;
72public:
73 friend class SpellCheckRequest;
74
75 explicit SpellChecker(Document&);
76 ~SpellChecker();
77
78 bool isAsynchronousEnabled() const;
79 bool isCheckable(const SimpleRange&) const;
80
81 void requestCheckingFor(Ref<SpellCheckRequest>&&);
82
83 TextCheckingRequestIdentifier lastRequestIdentifier() const { return m_lastRequestIdentifier; }
84 TextCheckingRequestIdentifier lastProcessedIdentifier() const { return m_lastProcessedIdentifier; }
85
86private:
87 bool canCheckAsynchronously(const SimpleRange&) const;
88 TextCheckerClient* client() const;
89 void timerFiredToProcessQueuedRequest();
90 void invokeRequest(Ref<SpellCheckRequest>&&);
91 void enqueueRequest(Ref<SpellCheckRequest>&&);
92 void didCheckSucceed(TextCheckingRequestIdentifier, const Vector<TextCheckingResult>&);
93 void didCheckCancel(TextCheckingRequestIdentifier);
94 void didCheck(TextCheckingRequestIdentifier, const Vector<TextCheckingResult>&);
95
96 Document& m_document;
97 TextCheckingRequestIdentifier m_lastRequestIdentifier;
98 TextCheckingRequestIdentifier m_lastProcessedIdentifier;
99
100 Timer m_timerToProcessQueuedRequest;
101
102 RefPtr<SpellCheckRequest> m_processingRequest;
103 Deque<Ref<SpellCheckRequest>> m_requestQueue;
104};
