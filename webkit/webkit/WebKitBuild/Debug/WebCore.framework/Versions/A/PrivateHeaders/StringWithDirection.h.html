34#include "WritingMode.h"
35#include <wtf/text/WTFString.h>

37namespace WebCore {
38
39// In some circumstances we want to store a String along with the TextDirection
40// of the String as learned from the context of the String. For example,
41// consider storing the title derived from <title dir='rtl'>some title</title>
42// in the history.
43//
44// Note that is explicitly *not* the direction of the string as learned
45// from the characters of the string; it's extra metadata we have external
46// to the string.
47
48struct StringWithDirection {
49 String string;
50 TextDirection direction { TextDirection::LTR };
51};
52
53inline bool operator==(const StringWithDirection& a, const StringWithDirection& b)
54{
55 return a.string == b.string && a.direction == b.direction;
56}
57
58inline bool operator!=(const StringWithDirection& a, const StringWithDirection& b)
59{
60 return !(a == b);
61}
62
63inline StringWithDirection truncateFromEnd(const StringWithDirection& string, unsigned maxLength)
64{
65 return { string.string.left(maxLength), string.direction };
66}
