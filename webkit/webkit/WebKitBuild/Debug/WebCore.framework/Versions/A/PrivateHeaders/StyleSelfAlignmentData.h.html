30namespace WTF {

34namespace WebCore {
35
36class StyleSelfAlignmentData {
37public:
38 // Style data for Self-Aligment and Default-Alignment properties: align-{self, items}, justify-{self, items}.
39 // [ <self-position> && <overflow-position>? ] | [ legacy && [ left | right | center ] ]
40 StyleSelfAlignmentData(ItemPosition position, OverflowAlignment overflow, ItemPositionType positionType = ItemPositionType::NonLegacy)
41 : m_position(static_cast<unsigned>(position))
42 , m_positionType(static_cast<unsigned>(positionType))
43 , m_overflow(static_cast<unsigned>(overflow))
44 {
45 }
46
47 void setPosition(ItemPosition position) { m_position = static_cast<unsigned>(position); }
48 void setPositionType(ItemPositionType positionType) { m_positionType = static_cast<unsigned>(positionType); }
49 void setOverflow(OverflowAlignment overflow) { m_overflow = static_cast<unsigned>(overflow); }
50
51 ItemPosition position() const { return static_cast<ItemPosition>(m_position); }
52 ItemPositionType positionType() const { return static_cast<ItemPositionType>(m_positionType); }
53 OverflowAlignment overflow() const { return static_cast<OverflowAlignment>(m_overflow); }
54
55 bool operator==(const StyleSelfAlignmentData& o) const
56 {
57 return m_position == o.m_position && m_positionType == o.m_positionType && m_overflow == o.m_overflow;
58 }
59
60 bool operator!=(const StyleSelfAlignmentData& o) const
61 {
62 return !(*this == o);
63 }
64
65private:
66 unsigned m_position : 4; // ItemPosition
67 unsigned m_positionType: 1; // Whether or not alignment uses the 'legacy' keyword.
68 unsigned m_overflow : 2; // OverflowAlignment
69};
70
71WTF::TextStream& operator<<(WTF::TextStream&, const StyleSelfAlignmentData&);

73} // namespace WebCore
