49enum class TextIndicatorLifetime : uint8_t {
50 // The TextIndicator should indicate the text until dismissed.
51 Permanent,

54 Temporary

64 None,

74// Make sure to keep these in sync with the ones in Internals.idl.
75enum class TextIndicatorOption : uint16_t {
76 // Use the styled text color instead of forcing black text (the default)
77 RespectTextColor = 1 << 0,
78
79 // Paint backgrounds, even if they're not part of the Range
80 PaintBackgrounds = 1 << 1,
81
82 // Don't restrict painting to the given Range
83 PaintAllContent = 1 << 2,
84
85 // Take two snapshots:
86 // - one including the selection highlight and ignoring other painting-related options
87 // - one respecting the other painting-related options
88 IncludeSnapshotWithSelectionHighlight = 1 << 3,
89
90 // Tightly fit the content instead of expanding to cover the bounds of the selection highlight
91 TightlyFitContent = 1 << 4,
92
93 // If there are any non-inline or replaced elements in the Range, indicate the bounding rect
94 // of the range instead of the individual subrects, and don't restrict painting to the given Range
95 UseBoundingRectAndPaintAllContentForComplexRanges = 1 << 5,
96
97 // By default, TextIndicator removes any margin if the given Range matches the
98 // selection Range. If this option is set, maintain the margin in any case.
99 IncludeMarginIfRangeMatchesSelection = 1 << 6,
100
101 // By default, TextIndicator clips the indicated rects to the visible content rect.
102 // If this option is set, expand the clip rect outward so that slightly offscreen content will be included.
103 ExpandClipBeyondVisibleRect = 1 << 7,
104
105 // By default, TextIndicator clips the indicated rects to the visible content rect.
106 // If this option is set, do not clip to the visible rect.
107 DoNotClipToVisibleRect = 1 << 8,
108
109 // Include an additional snapshot of everything in view, with the exception of nodes within the currently selected range.
110 IncludeSnapshotOfAllVisibleContentWithoutSelection = 1 << 9,
111
112 // By default, TextIndicator uses text rects to size the snapshot. Enabling this flag causes it to use the bounds of the
113 // selection rects that would enclose the given Range instead.
114 // Currently, this is only supported on iOS.
115 UseSelectionRectForSizing = 1 << 10,
116
117 // Compute a background color to use when rendering a platter around the content image, falling back to a default if the
118 // content's background is too complex to be captured by a single color.
119 ComputeEstimatedBackgroundColor = 1 << 11,
120};

122struct TextIndicatorData {
123 FloatRect selectionRectInRootViewCoordinates;
124 FloatRect textBoundingRectInRootViewCoordinates;
125 FloatRect contentImageWithoutSelectionRectInRootViewCoordinates;
126 Vector<FloatRect> textRectsInBoundingRectCoordinates;
127 float contentImageScaleFactor { 1 };
128 RefPtr<Image> contentImageWithHighlight;
129 RefPtr<Image> contentImageWithoutSelection;
130 RefPtr<Image> contentImage;
131 Color estimatedBackgroundColor;
132 TextIndicatorPresentationTransition presentationTransition { TextIndicatorPresentationTransition::None };
133 OptionSet<TextIndicatorOption> options;
134};
135
136class TextIndicator : public RefCounted<TextIndicator> {
137public:
138 // FIXME: These are fairly Mac-specific, and they don't really belong here.
139 // But they're needed at TextIndicator creation time, so they can't go in TextIndicatorWindow.
140 // Maybe they can live in some Theme code somewhere?
141 constexpr static float defaultHorizontalMargin { 2 };
142 constexpr static float defaultVerticalMargin { 1 };
143
144 WEBCORE_EXPORT static Ref<TextIndicator> create(const TextIndicatorData&);
145 WEBCORE_EXPORT static RefPtr<TextIndicator> createWithSelectionInFrame(Frame&, OptionSet<TextIndicatorOption>, TextIndicatorPresentationTransition, FloatSize margin = FloatSize(defaultHorizontalMargin, defaultVerticalMargin));
146 WEBCORE_EXPORT static RefPtr<TextIndicator> createWithRange(const SimpleRange&, OptionSet<TextIndicatorOption>, TextIndicatorPresentationTransition, FloatSize margin = FloatSize(defaultHorizontalMargin, defaultVerticalMargin));
147
148 WEBCORE_EXPORT ~TextIndicator();
149
150 FloatRect selectionRectInRootViewCoordinates() const { return m_data.selectionRectInRootViewCoordinates; }
151 FloatRect textBoundingRectInRootViewCoordinates() const { return m_data.textBoundingRectInRootViewCoordinates; }
152 const Vector<FloatRect>& textRectsInBoundingRectCoordinates() const { return m_data.textRectsInBoundingRectCoordinates; }
153 float contentImageScaleFactor() const { return m_data.contentImageScaleFactor; }
154 Image* contentImageWithHighlight() const { return m_data.contentImageWithHighlight.get(); }
155 Image* contentImage() const { return m_data.contentImage.get(); }
156
157 TextIndicatorPresentationTransition presentationTransition() const { return m_data.presentationTransition; }
158 void setPresentationTransition(TextIndicatorPresentationTransition transition) { m_data.presentationTransition = transition; }
159
160 TextIndicatorData data() const { return m_data; }
161
162private:
163 TextIndicator(const TextIndicatorData&);
164
165 TextIndicatorData m_data;
166};
