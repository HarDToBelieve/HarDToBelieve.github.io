201template<class Encoder>
202void TextManipulationController::ManipulationTokenInfo::encode(Encoder& encoder) const
203{
204 encoder << tagName;
205 encoder << roleAttribute;
206 encoder << documentURL;
207 encoder << isVisible;
208}
209
210template<class Decoder>
211std::optional<TextManipulationController::ManipulationTokenInfo> TextManipulationController::ManipulationTokenInfo::decode(Decoder& decoder)
212{
213 ManipulationTokenInfo result;
214 if (!decoder.decode(result.tagName))
215 return std::nullopt;
216
217 if (!decoder.decode(result.roleAttribute))
218 return std::nullopt;
219
220 if (!decoder.decode(result.documentURL))
221 return std::nullopt;
222
223 if (!decoder.decode(result.isVisible))
224 return std::nullopt;
225
226 return result;
227}

235template<class Decoder>
236std::optional<TextManipulationController::ManipulationToken> TextManipulationController::ManipulationToken::decode(Decoder& decoder)
237{
238 ManipulationToken result;

240 return std::nullopt;

242 return std::nullopt;
243 if (!decoder.decode(result.info))
244 return std::nullopt;
245 if (!decoder.decode(result.isExcluded))
246 return std::nullopt;
247 return result;
248}
