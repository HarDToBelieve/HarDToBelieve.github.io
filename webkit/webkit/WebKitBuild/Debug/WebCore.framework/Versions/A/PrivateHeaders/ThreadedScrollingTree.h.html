30#include "ScrollingStateTree.h"
31#include "ScrollingTree.h"
32#include <wtf/Condition.h>
33#include <wtf/RefPtr.h>
34#include <wtf/RunLoop.h>

36namespace WebCore {
37
38class AsyncScrollingCoordinator;
39
40// The ThreadedScrollingTree class lives almost exclusively on the scrolling thread and manages the
41// hierarchy of scrollable regions on the page. It's also responsible for dispatching events
42// to the correct scrolling tree nodes or dispatching events back to the ScrollingCoordinator
43// object on the main thread if they can't be handled on the scrolling thread for various reasons.
44class ThreadedScrollingTree : public ScrollingTree {
45public:
46 virtual ~ThreadedScrollingTree();
47
48 WheelEventHandlingResult handleWheelEvent(const PlatformWheelEvent&, OptionSet<WheelEventProcessingSteps>) override;
49
50 bool handleWheelEventAfterMainThread(const PlatformWheelEvent&, ScrollingNodeID, std::optional<WheelScrollGestureState>);
51 void wheelEventWasProcessedByMainThread(const PlatformWheelEvent&, std::optional<WheelScrollGestureState>);
52
53 WEBCORE_EXPORT void willSendEventToMainThread(const PlatformWheelEvent&) final;
54 WEBCORE_EXPORT void waitForEventToBeProcessedByMainThread(const PlatformWheelEvent&) final;
55
56 void invalidate() override;
57
58 WEBCORE_EXPORT void displayDidRefresh(PlatformDisplayID);
59
60 void willStartRenderingUpdate();
61 void didCompleteRenderingUpdate();
62
63 Lock& treeLock() WTF_RETURNS_LOCK(m_treeLock) { return m_treeLock; }
64
65 bool scrollAnimatorEnabled() const { return m_scrollAnimatorEnabled; }
66
67protected:
68 explicit ThreadedScrollingTree(AsyncScrollingCoordinator&);
69
70 void scrollingTreeNodeDidScroll(ScrollingTreeScrollingNode&, ScrollingLayerPositionAction = ScrollingLayerPositionAction::Sync) override;
71#if PLATFORM(MAC)
72 void handleWheelEventPhase(ScrollingNodeID, PlatformWheelEventPhase) override;
73 void setActiveScrollSnapIndices(ScrollingNodeID, std::optional<unsigned> horizontalIndex, std::optional<unsigned> verticalIndex) override;
74#endif
75
76#if PLATFORM(COCOA)
77 void currentSnapPointIndicesDidChange(ScrollingNodeID, std::optional<unsigned> horizontal, std::optional<unsigned> vertical) override;
78#endif
79
80 void reportExposedUnfilledArea(MonotonicTime, unsigned unfilledArea) override;
81 void reportSynchronousScrollingReasonsChanged(MonotonicTime, OptionSet<SynchronousScrollingReason>) override;
82
83 RefPtr<AsyncScrollingCoordinator> m_scrollingCoordinator;
84
85private:
86 bool isThreadedScrollingTree() const override { return true; }
87 void propagateSynchronousScrollingReasons(const HashSet<ScrollingNodeID>&) WTF_REQUIRES_LOCK(m_treeLock) override;
88
89 void displayDidRefreshOnScrollingThread();
90 void waitForRenderingUpdateCompletionOrTimeout() WTF_REQUIRES_LOCK(m_treeLock);
91
92 bool canUpdateLayersOnScrollingThread() const WTF_REQUIRES_LOCK(m_treeLock);
93
94 void scheduleDelayedRenderingUpdateDetectionTimer(Seconds) WTF_REQUIRES_LOCK(m_treeLock);
95 void delayedRenderingUpdateDetectionTimerFired();
96
97 Seconds maxAllowableRenderingUpdateDurationForSynchronization();
98
99 enum class SynchronizationState : uint8_t {
100 Idle,
101 WaitingForRenderingUpdate,
102 InRenderingUpdate,
103 Desynchronized,
104 };
105
106 SynchronizationState m_state WTF_GUARDED_BY_LOCK(m_treeLock) { SynchronizationState::Idle };
107 Condition m_stateCondition;
108
109 bool m_receivedBeganEventFromMainThread WTF_GUARDED_BY_LOCK(m_treeLock) { false };
110 Condition m_waitingForBeganEventCondition;
111
112 // Dynamically allocated because it has to use the ScrollingThread's runloop.
113 std::unique_ptr<RunLoop::Timer<ThreadedScrollingTree>> m_delayedRenderingUpdateDetectionTimer WTF_GUARDED_BY_LOCK(m_treeLock);
114
115 const bool m_scrollAnimatorEnabled { false };
116 bool m_hasNodesWithSynchronousScrollingReasons WTF_GUARDED_BY_LOCK(m_treeLock) { false };
117};
118
119} // namespace WebCore
120
121SPECIALIZE_TYPE_TRAITS_SCROLLING_TREE(WebCore::ThreadedScrollingTree, isThreadedScrollingTree())
122
123#endif // ENABLE(ASYNC_SCROLLING) && ENABLE(SCROLLING_THREAD)
