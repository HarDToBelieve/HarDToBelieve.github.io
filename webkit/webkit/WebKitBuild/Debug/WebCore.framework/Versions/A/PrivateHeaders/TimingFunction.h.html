39class TimingFunction : public RefCounted<TimingFunction> {
40public:
41 virtual Ref<TimingFunction> clone() const = 0;
42
43 virtual ~TimingFunction() = default;
44
45 enum TimingFunctionType { LinearFunction, CubicBezierFunction, StepsFunction, SpringFunction };
46 TimingFunctionType type() const { return m_type; }
47
48 bool isLinearTimingFunction() const { return m_type == LinearFunction; }
49 bool isCubicBezierTimingFunction() const { return m_type == CubicBezierFunction; }
50 bool isStepsTimingFunction() const { return m_type == StepsFunction; }
51 bool isSpringTimingFunction() const { return m_type == SpringFunction; }
52
53 virtual bool operator==(const TimingFunction&) const = 0;
54 bool operator!=(const TimingFunction& other) const { return !(*this == other); }
55
56 static ExceptionOr<RefPtr<TimingFunction>> createFromCSSText(const String&);
57 static RefPtr<TimingFunction> createFromCSSValue(const CSSValue&);
58 double transformTime(double, double, bool before = false) const;
59 String cssText() const;
60
61protected:
62 explicit TimingFunction(TimingFunctionType type)
63 : m_type(type)
64 {
65 }
66
67private:
68 TimingFunctionType m_type;
69};

101class CubicBezierTimingFunction final : public TimingFunction {
102public:
103 enum TimingFunctionPreset { Ease, EaseIn, EaseOut, EaseInOut, Custom };
104
105 static Ref<CubicBezierTimingFunction> create(double x1, double y1, double x2, double y2)
106 {
107 return adoptRef(*new CubicBezierTimingFunction(Custom, x1, y1, x2, y2));
108 }
109
110 static Ref<CubicBezierTimingFunction> create()
111 {
112 return adoptRef(*new CubicBezierTimingFunction());
113 }
114
115 static Ref<CubicBezierTimingFunction> create(TimingFunctionPreset preset)
116 {
117 switch (preset) {
118 case Ease:
119 return adoptRef(*new CubicBezierTimingFunction);
120 case EaseIn:
121 return adoptRef(*new CubicBezierTimingFunction(EaseIn, 0.42, 0.0, 1.0, 1.0));
122 case EaseOut:
123 return adoptRef(*new CubicBezierTimingFunction(EaseOut, 0.0, 0.0, 0.58, 1.0));
124 case EaseInOut:
125 return adoptRef(*new CubicBezierTimingFunction(EaseInOut, 0.42, 0.0, 0.58, 1.0));
126 case Custom:
127 break;
128 }
129 ASSERT_NOT_REACHED();
130 return adoptRef(*new CubicBezierTimingFunction);
131 }
132
133 bool operator==(const TimingFunction& other) const final
134 {
135 if (!is<CubicBezierTimingFunction>(other))
136 return false;
137 auto& otherCubic = downcast<CubicBezierTimingFunction>(other);
138 if (m_timingFunctionPreset != otherCubic.m_timingFunctionPreset)
139 return false;
140 if (m_timingFunctionPreset != Custom)
141 return true;
142 return m_x1 == otherCubic.m_x1 && m_y1 == otherCubic.m_y1 && m_x2 == otherCubic.m_x2 && m_y2 == otherCubic.m_y2;
143 }
144
145 double x1() const { return m_x1; }
146 double y1() const { return m_y1; }
147 double x2() const { return m_x2; }
148 double y2() const { return m_y2; }
149
150 void setValues(double x1, double y1, double x2, double y2)
151 {
152 m_x1 = x1;
153 m_y1 = y1;
154 m_x2 = x2;
155 m_y2 = y2;
156 }
157
158 TimingFunctionPreset timingFunctionPreset() const { return m_timingFunctionPreset; }
159 void setTimingFunctionPreset(TimingFunctionPreset preset) { m_timingFunctionPreset = preset; }
160
161 static const CubicBezierTimingFunction& defaultTimingFunction()
162 {
163 static const CubicBezierTimingFunction& function = create().leakRef();
164 return function;
165 }
166
167 Ref<CubicBezierTimingFunction> createReversed() const
168 {
169 return create(1.0 - m_x2, 1.0 - m_y2, 1.0 - m_x1, 1.0 - m_y1);
170 }
171
172 bool isLinear() const
173 {
174 return (!m_x1 && !m_y1 && !m_x2 && !m_y2) || (m_x1 == 1.0 && m_y1 == 1.0 && m_x2 == 1.0 && m_y2 == 1.0) || (m_x1 == 0.0 && m_y1 == 0.0 && m_x2 == 1.0 && m_y2 == 1.0);
175 }
176
177private:
178 explicit CubicBezierTimingFunction(TimingFunctionPreset preset = Ease, double x1 = 0.25, double y1 = 0.1, double x2 = 0.25, double y2 = 1.0)
179 : TimingFunction(CubicBezierFunction)
180 , m_x1(x1)
181 , m_y1(y1)
182 , m_x2(x2)
183 , m_y2(y2)
184 , m_timingFunctionPreset(preset)
185 {
186 }
187
188 Ref<TimingFunction> clone() const final
189 {
190 return adoptRef(*new CubicBezierTimingFunction(m_timingFunctionPreset, m_x1, m_y1, m_x2, m_y2));
191 }
192
193 double m_x1;
194 double m_y1;
195 double m_x2;
196 double m_y2;
197 TimingFunctionPreset m_timingFunctionPreset;
198};

200class StepsTimingFunction final : public TimingFunction {
201public:
202 enum class StepPosition {
203 JumpStart,
204 JumpEnd,
205 JumpNone,
206 JumpBoth,
207 Start,
208 End,
209 };
210
211 static Ref<StepsTimingFunction> create(int steps, std::optional<StepPosition> stepPosition)
212 {
213 return adoptRef(*new StepsTimingFunction(steps, stepPosition));
214 }
215 static Ref<StepsTimingFunction> create()
216 {
217 return adoptRef(*new StepsTimingFunction(1, StepPosition::End));
218 }
219
220 bool operator==(const TimingFunction& other) const final
221 {
222 if (!is<StepsTimingFunction>(other))
223 return false;
224 auto& otherSteps = downcast<StepsTimingFunction>(other);
225 return m_steps == otherSteps.m_steps && m_stepPosition == otherSteps.m_stepPosition;
226 }
227
228 int numberOfSteps() const { return m_steps; }
229 void setNumberOfSteps(int steps) { m_steps = steps; }
230
231 std::optional<StepPosition> stepPosition() const { return m_stepPosition; }
232 void setStepPosition(std::optional<StepPosition> stepPosition) { m_stepPosition = stepPosition; }
233
234private:
235 StepsTimingFunction(int steps, std::optional<StepPosition> stepPosition)
236 : TimingFunction(StepsFunction)
237 , m_steps(steps)
238 , m_stepPosition(stepPosition)
239 {
240 }
241
242 Ref<TimingFunction> clone() const final
243 {
244 return adoptRef(*new StepsTimingFunction(m_steps, m_stepPosition));
245 }
246
247 int m_steps;
248 std::optional<StepPosition> m_stepPosition;
249};

251class SpringTimingFunction final : public TimingFunction {
252public:
253 static Ref<SpringTimingFunction> create(double mass, double stiffness, double damping, double initialVelocity)
254 {
255 return adoptRef(*new SpringTimingFunction(mass, stiffness, damping, initialVelocity));
256 }
257
258 static Ref<SpringTimingFunction> create()
259 {
260 // This create() function should only be used by the argument decoders, and it is expected that
261 // real values will be filled in using setValues().
262 return create(0, 0, 0, 0);
263 }
264
265 bool operator==(const TimingFunction& other) const final
266 {
267 if (!is<SpringTimingFunction>(other))
268 return false;
269 auto& otherSpring = downcast<SpringTimingFunction>(other);
270 return m_mass == otherSpring.m_mass && m_stiffness == otherSpring.m_stiffness && m_damping == otherSpring.m_damping && m_initialVelocity == otherSpring.m_initialVelocity;
271 }
272
273 double mass() const { return m_mass; }
274 double stiffness() const { return m_stiffness; }
275 double damping() const { return m_damping; }
276 double initialVelocity() const { return m_initialVelocity; }
277
278 void setValues(double mass, double stiffness, double damping, double initialVelocity)
279 {
280 m_mass = mass;
281 m_stiffness = stiffness;
282 m_damping = damping;
283 m_initialVelocity = initialVelocity;
284 }
285
286private:
287 explicit SpringTimingFunction(double mass, double stiffness, double damping, double initialVelocity)
288 : TimingFunction(SpringFunction)
289 , m_mass(mass)
290 , m_stiffness(stiffness)
291 , m_damping(damping)
292 , m_initialVelocity(initialVelocity)
293 {
294 }
295
296 Ref<TimingFunction> clone() const final
297 {
298 return adoptRef(*new SpringTimingFunction(m_mass, m_stiffness, m_damping, m_initialVelocity));
299 }
300
301 double m_mass;
302 double m_stiffness;
303 double m_damping;
304 double m_initialVelocity;
305};
