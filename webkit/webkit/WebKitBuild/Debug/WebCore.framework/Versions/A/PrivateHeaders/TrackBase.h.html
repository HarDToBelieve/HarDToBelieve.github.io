30#include <wtf/LoggerHelper.h>
31#include <wtf/WeakPtr.h>
32#include <wtf/text/AtomString.h>

34namespace WebCore {
35
36class Element;
37class HTMLMediaElement;
38class SourceBuffer;
39
40class TrackBase
41 : public RefCounted<TrackBase>
42#if !RELEASE_LOG_DISABLED
43 , private LoggerHelper
44#endif
45{
46public:
47 virtual ~TrackBase() = default;
48
49 enum Type { BaseTrack, TextTrack, AudioTrack, VideoTrack };
50 Type type() const { return m_type; }
51
52 virtual void setMediaElement(WeakPtr<HTMLMediaElement>);
53 WeakPtr<HTMLMediaElement> mediaElement() { return m_mediaElement; }
54 virtual Element* element();
55
56 virtual AtomString id() const { return m_id; }
57 virtual void setId(const AtomString& id) { m_id = id; }
58
59 AtomString label() const { return m_label; }
60 void setLabel(const AtomString& label) { m_label = label; }
61
62 AtomString validBCP47Language() const { return m_validBCP47Language; }
63 AtomString language() const { return m_language; }
64 virtual void setLanguage(const AtomString&);
65
66 virtual void clearClient() = 0;
67
68 virtual int uniqueId() const { return m_uniqueId; }
69
70#if ENABLE(MEDIA_SOURCE)
71 SourceBuffer* sourceBuffer() const { return m_sourceBuffer; }
72 void setSourceBuffer(SourceBuffer* buffer) { m_sourceBuffer = buffer; }
73#endif
74
75 virtual bool enabled() const = 0;
76
77#if !RELEASE_LOG_DISABLED
78 virtual void setLogger(const Logger&, const void*);
79 const Logger& logger() const final { ASSERT(m_logger); return *m_logger.get(); }
80 const void* logIdentifier() const final { return m_logIdentifier; }
81 WTFLogChannel& logChannel() const final;
82#endif
83
84protected:
85 TrackBase(Type, const AtomString& id, const AtomString& label, const AtomString& language);
86
87 WeakPtr<HTMLMediaElement> m_mediaElement { nullptr };
88
89#if ENABLE(MEDIA_SOURCE)
90 SourceBuffer* m_sourceBuffer { nullptr };
91#endif
92
93private:
94 Type m_type;
95 int m_uniqueId;
96 AtomString m_id;
97 AtomString m_label;
98 AtomString m_language;
99 AtomString m_validBCP47Language;
100#if !RELEASE_LOG_DISABLED
101 RefPtr<const Logger> m_logger;
102 const void* m_logIdentifier { nullptr };
103#endif
104};
105
106class MediaTrackBase : public TrackBase {
107public:
108 const AtomString& kind() const { return m_kind; }
109 virtual void setKind(const AtomString&);
110
111protected:
112 MediaTrackBase(Type, const AtomString& id, const AtomString& label, const AtomString& language);
113
114 void setKindInternal(const AtomString&);
115
116private:
117 virtual bool isValidKind(const AtomString&) const = 0;
118
119 AtomString m_kind;
120};
