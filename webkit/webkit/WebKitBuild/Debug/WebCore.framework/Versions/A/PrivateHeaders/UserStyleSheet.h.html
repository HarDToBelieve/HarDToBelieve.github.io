26#pragma once
27
28#include "PageIdentifier.h"
29#include "UserContentTypes.h"
30#include "UserStyleSheetTypes.h"
31#include <wtf/URL.h>
32#include <wtf/Vector.h>
33
34namespace WebCore {
35
36class UserStyleSheet {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 UserStyleSheet()
40 : m_injectedFrames(UserContentInjectedFrames::InjectInAllFrames)
41 , m_level(UserStyleUserLevel)
42 {
43 }
44
45 UserStyleSheet(const String& source, const URL& url, Vector<String>&& allowlist, Vector<String>&& blocklist, UserContentInjectedFrames injectedFrames, UserStyleLevel level, std::optional<PageIdentifier> pageID = std::nullopt)
46 : m_source(source)
47 , m_url(url)
48 , m_allowlist(WTFMove(allowlist))
49 , m_blocklist(WTFMove(blocklist))
50 , m_injectedFrames(injectedFrames)
51 , m_level(level)
52 , m_pageID(pageID)
53 {
54 }
55
56 const String& source() const { return m_source; }
57 const URL& url() const { return m_url; }
58 const Vector<String>& allowlist() const { return m_allowlist; }
59 const Vector<String>& blocklist() const { return m_blocklist; }
60 UserContentInjectedFrames injectedFrames() const { return m_injectedFrames; }
61 UserStyleLevel level() const { return m_level; }
62 std::optional<PageIdentifier> pageID() const { return m_pageID; }
63
64private:
65 String m_source;
66 URL m_url;
67 Vector<String> m_allowlist;
68 Vector<String> m_blocklist;
69 UserContentInjectedFrames m_injectedFrames;
70 UserStyleLevel m_level;
71 std::optional<PageIdentifier> m_pageID;
72};
73
74} // namespace WebCore
