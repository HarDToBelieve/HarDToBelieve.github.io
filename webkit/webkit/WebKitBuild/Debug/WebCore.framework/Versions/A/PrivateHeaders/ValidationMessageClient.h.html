26#pragma once
27
28#include <wtf/Forward.h>
29
30namespace WebCore {
31
32class Document;
33class Element;
34
35class ValidationMessageClient {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 virtual ~ValidationMessageClient() = default;
39
40 // Show validation message for the specified anchor element. An
41 // implementation of this function may hide the message automatically after
42 // some period.
43 virtual void showValidationMessage(const Element& anchor, const String& message) = 0;
44
45 // Hide validation message for the specified anchor if the message for the
46 // anchor is already visible.
47 virtual void hideValidationMessage(const Element& anchor) = 0;
48
49 // Hide any validation message currently displayed.
50 virtual void hideAnyValidationMessage() = 0;
51
52 // Returns true if the validation message for the specified anchor element
53 // is visible.
54 virtual bool isValidationMessageVisible(const Element& anchor) = 0;
55
56 virtual void updateValidationBubbleStateIfNeeded() = 0;
57
58 virtual void documentDetached(Document&) = 0;
59};
60
61} // namespace WebCore
