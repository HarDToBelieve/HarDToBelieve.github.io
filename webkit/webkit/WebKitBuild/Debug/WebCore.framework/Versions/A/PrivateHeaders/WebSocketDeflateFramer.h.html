38class WebSocketDeflateFramer;

41struct WebSocketFrame;

43class DeflateResultHolder {
44 WTF_MAKE_FAST_ALLOCATED;
45public:
46 explicit DeflateResultHolder(WebSocketDeflateFramer&);
47 ~DeflateResultHolder();
48
49 bool succeeded() const { return m_succeeded; }
50 String failureReason() const { return m_failureReason; }
51
52 void fail(const String& failureReason);
53
54private:
55 WebSocketDeflateFramer& m_framer;
56 bool m_succeeded { true };
57 String m_failureReason;
58};
59
60class InflateResultHolder {
61 WTF_MAKE_FAST_ALLOCATED;
62public:
63 explicit InflateResultHolder(WebSocketDeflateFramer&);
64 ~InflateResultHolder();
65
66 bool succeeded() const { return m_succeeded; }
67 String failureReason() const { return m_failureReason; }
68
69 void fail(const String& failureReason);
70
71private:
72 WebSocketDeflateFramer& m_framer;
73 bool m_succeeded { true };
74 String m_failureReason;
75};

77class WebSocketDeflateFramer {
78public:
79 std::unique_ptr<WebSocketExtensionProcessor> createExtensionProcessor();
80
81 bool enabled() const { return m_enabled; }
82
83 std::unique_ptr<DeflateResultHolder> deflate(WebSocketFrame&);
84 void resetDeflateContext();
85 std::unique_ptr<InflateResultHolder> inflate(WebSocketFrame&);
86 void resetInflateContext();
87
88 void didFail();
89
90 void enableDeflate(int windowBits, WebSocketDeflater::ContextTakeOverMode);
91
92private:
93 bool m_enabled { false };
94 std::unique_ptr<WebSocketDeflater> m_deflater;
95 std::unique_ptr<WebSocketInflater> m_inflater;
96};
