31#pragma once
32
33#include <wtf/Vector.h>
34
35struct z_stream_s;
36typedef z_stream_s z_stream;
37
38namespace WebCore {
39
40class WebSocketDeflater {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 enum ContextTakeOverMode {
44 DoNotTakeOverContext,
45 TakeOverContext
46 };
47
48 explicit WebSocketDeflater(int windowBits, ContextTakeOverMode = TakeOverContext);
49 ~WebSocketDeflater();
50
51 bool initialize();
52 bool addBytes(const uint8_t*, size_t);
53 bool finish();
54 const uint8_t* data() { return m_buffer.data(); }
55 size_t size() const { return m_buffer.size(); }
56 void reset();
57
58private:
59 int m_windowBits;
60 ContextTakeOverMode m_contextTakeOverMode;
61 Vector<uint8_t> m_buffer;
62 std::unique_ptr<z_stream> m_stream;
63};
64
65class WebSocketInflater {
66 WTF_MAKE_FAST_ALLOCATED;
67public:
68 explicit WebSocketInflater(int windowBits = 15);
69 ~WebSocketInflater();
70
71 bool initialize();
72 bool addBytes(const uint8_t*, size_t);
73 bool finish();
74 const uint8_t* data() { return m_buffer.data(); }
75 size_t size() const { return m_buffer.size(); }
76 void reset();
77
78private:
79 int m_windowBits;
80 Vector<uint8_t> m_buffer;
81 std::unique_ptr<z_stream> m_stream;
82};

84} // namespace WebCore
