42class WheelEventTestMonitor : public ThreadSafeRefCounted<WheelEventTestMonitor> {
43 WTF_MAKE_NONCOPYABLE(WheelEventTestMonitor); WTF_MAKE_FAST_ALLOCATED;
44public:
45 WheelEventTestMonitor(Page&);
46
47 WEBCORE_EXPORT void setTestCallbackAndStartMonitoring(bool expectWheelEndOrCancel, bool expectMomentumEnd, WTF::Function<void()>&&);
48 WEBCORE_EXPORT void clearAllTestDeferrals();
49
50 enum DeferReason {
51 HandlingWheelEvent = 1 << 0,
52 HandlingWheelEventOnMainThread = 1 << 1,
53 PostMainThreadWheelEventHandling = 1 << 2,
54 RubberbandInProgress = 1 << 3,
55 ScrollSnapInProgress = 1 << 4,
56 ScrollingThreadSyncNeeded = 1 << 5,
57 ContentScrollInProgress = 1 << 6,
58 RequestedScrollPosition = 1 << 7,
59 };
60 typedef const void* ScrollableAreaIdentifier;
61
62 WEBCORE_EXPORT void receivedWheelEvent(const PlatformWheelEvent&);
63 WEBCORE_EXPORT void deferForReason(ScrollableAreaIdentifier, DeferReason);
64 WEBCORE_EXPORT void removeDeferralForReason(ScrollableAreaIdentifier, DeferReason);
65
66 void checkShouldFireCallbacks();
67
68 using ScrollableAreaReasonMap = WTF::HashMap<ScrollableAreaIdentifier, OptionSet<DeferReason>>;
69
70private:
71 void scheduleCallbackCheck();
72
73 WTF::Function<void()> m_completionCallback;
74 Page& m_page;
75
76 Lock m_lock;
77 ScrollableAreaReasonMap m_deferCompletionReasons WTF_GUARDED_BY_LOCK(m_lock);
78 bool m_expectWheelEndOrCancel WTF_GUARDED_BY_LOCK(m_lock) { false };
79 bool m_receivedWheelEndOrCancel WTF_GUARDED_BY_LOCK(m_lock) { false };
80 bool m_expectMomentumEnd WTF_GUARDED_BY_LOCK(m_lock) { false };
81 bool m_receivedMomentumEnd WTF_GUARDED_BY_LOCK(m_lock) { false };
82 bool m_everHadDeferral WTF_GUARDED_BY_LOCK(m_lock) { false };
83};
84
85class WheelEventTestMonitorCompletionDeferrer {
86public:
87 WheelEventTestMonitorCompletionDeferrer(WheelEventTestMonitor* monitor, WheelEventTestMonitor::ScrollableAreaIdentifier identifier, WheelEventTestMonitor::DeferReason reason)
88 : m_monitor(monitor)
89 , m_identifier(identifier)
90 , m_reason(reason)
91 {
92 if (m_monitor)
93 m_monitor->deferForReason(m_identifier, m_reason);
94 }
95
96 WheelEventTestMonitorCompletionDeferrer(WheelEventTestMonitorCompletionDeferrer&& other)
97 : m_monitor(WTFMove(other.m_monitor))
98 , m_identifier(other.m_identifier)
99 , m_reason(other.m_reason)
100 {
101 }
102
103 ~WheelEventTestMonitorCompletionDeferrer()
104 {
105 if (m_monitor)
106 m_monitor->removeDeferralForReason(m_identifier, m_reason);
107 }
108
109private:
110 RefPtr<WheelEventTestMonitor> m_monitor;
111 WheelEventTestMonitor::ScrollableAreaIdentifier m_identifier;
112 WheelEventTestMonitor::DeferReason m_reason;
113};
