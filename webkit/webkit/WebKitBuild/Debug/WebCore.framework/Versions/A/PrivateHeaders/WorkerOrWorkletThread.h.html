28#include "WorkerRunLoop.h"
29#include <wtf/Forward.h>
30#include <wtf/Function.h>
31#include <wtf/Lock.h>
32#include <wtf/ThreadSafeRefCounted.h>
33#include <wtf/threads/BinarySemaphore.h>
34
35namespace WTF {
36class Thread;
37}
38
39namespace WebCore {
40
41class WorkerDebuggerProxy;
42class WorkerLoaderProxy;
43class WorkerRunLoop;
44
45class WorkerOrWorkletThread : public ThreadSafeRefCounted<WorkerOrWorkletThread> {
46public:
47 virtual ~WorkerOrWorkletThread();
48
49 WTF::Thread* thread() const { return m_thread.get(); }
50
51 virtual WorkerDebuggerProxy* workerDebuggerProxy() const = 0;
52 virtual WorkerLoaderProxy& workerLoaderProxy() = 0;
53
54 WorkerOrWorkletGlobalScope* globalScope() const { return m_globalScope.get(); }
55 WorkerRunLoop& runLoop() { return m_runLoop; }
56
57 void start(Function<void(const String&)>&& evaluateCallback = { });
58 void stop(Function<void()>&& terminatedCallback = { });
59
60 void startRunningDebuggerTasks();
61 void stopRunningDebuggerTasks();
62
63 void suspend();
64 void resume();
65
66 const String& identifier() const { return m_identifier; }
67
68 static HashSet<WorkerOrWorkletThread*>& workerOrWorkletThreads() WTF_REQUIRES_LOCK(workerOrWorkletThreadsLock());
69 static Lock& workerOrWorkletThreadsLock() WTF_RETURNS_LOCK(s_workerOrWorkletThreadsLock);
70 static void releaseFastMallocFreeMemoryInAllThreads();
71
72protected:
73 explicit WorkerOrWorkletThread(const String& identifier);
74 void workerOrWorkletThread();
75
76 // Executes the event loop for the worker thread. Derived classes can override to perform actions before/after entering the event loop.
77 virtual void runEventLoop();
78
79private:
80 virtual Ref<WTF::Thread> createThread() = 0;
81 virtual RefPtr<WorkerOrWorkletGlobalScope> createGlobalScope() = 0;
82 virtual void evaluateScriptIfNecessary(String&) { }
83 virtual bool shouldWaitForWebInspectorOnStartup() const { return false; }
84
85 static Lock s_workerOrWorkletThreadsLock;
86
87 String m_identifier;
88 Lock m_threadCreationAndGlobalScopeLock;
89 RefPtr<WorkerOrWorkletGlobalScope> m_globalScope;
90 RefPtr<WTF::Thread> m_thread;
91 WorkerRunLoop m_runLoop;
92 Function<void(const String&)> m_evaluateCallback;
93 Function<void()> m_stoppedCallback;
94 BinarySemaphore m_suspensionSemaphore;
95 bool m_isSuspended { false };
96 bool m_pausedForDebugger { false };
97};
98
99} // namespace WebCore
