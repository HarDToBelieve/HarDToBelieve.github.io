27#define RUNTIME_ROOT_H_
28
29#include <JavaScriptCore/Strong.h>
30#include <JavaScriptCore/Weak.h>
31#include <JavaScriptCore/WeakInlines.h>
32#include <wtf/Forward.h>
33#include <wtf/HashCountedSet.h>
34#include <wtf/HashSet.h>
35#include <wtf/Noncopyable.h>
36#include <wtf/RefCounted.h>
37
38namespace JSC {
39
40class Interpreter;
41class JSGlobalObject;
42
43namespace Bindings {
44
45class RootObject;
46class RuntimeObject;
47
48typedef HashCountedSet<JSObject*> ProtectCountSet;
49
50extern RootObject* findProtectingRootObject(JSObject*);
51extern RootObject* findRootObject(JSGlobalObject*);
52
53class RootObject : public RefCounted<RootObject>, private JSC::WeakHandleOwner {
54 friend class JavaJSObject;
55
56public:
57 WEBCORE_EXPORT virtual ~RootObject();
58
59 static Ref<RootObject> create(const void* nativeHandle, JSGlobalObject*);
60
61 bool isValid() { return m_isValid; }
62 void invalidate();
63
64 void gcProtect(JSObject*);
65 void gcUnprotect(JSObject*);
66 bool gcIsProtected(JSObject*);
67
68 const void* nativeHandle() const;
69 WEBCORE_EXPORT JSGlobalObject* globalObject() const;
70 void updateGlobalObject(JSGlobalObject*);
71
72 void addRuntimeObject(VM&, RuntimeObject*);
73 void removeRuntimeObject(RuntimeObject*);
74
75 struct InvalidationCallback {
76 virtual void operator()(RootObject*) = 0;
77 virtual ~InvalidationCallback();
78 };
79 void addInvalidationCallback(InvalidationCallback* callback) { m_invalidationCallbacks.add(callback); }
80
81private:
82 RootObject(const void* nativeHandle, JSGlobalObject*);
83
84 // WeakHandleOwner
85 void finalize(JSC::Handle<JSC::Unknown>, void* context) override;
86
87 bool m_isValid;
88
89 const void* m_nativeHandle;
90 Strong<JSGlobalObject> m_globalObject;
91
92 ProtectCountSet m_protectCountSet;
93 HashMap<RuntimeObject*, JSC::Weak<RuntimeObject>> m_runtimeObjects; // We use a map to implement a set.
94
95 HashSet<InvalidationCallback*> m_invalidationCallbacks;
96};
97
98} // namespace Bindings
99
100} // namespace JSC

102#endif // RUNTIME_ROOT_H_
