18//
19// This header file contains the declaration and definition of an "inlined
20// vector" which behaves in an equivalent fashion to a `std::vector`, except
21// that storage for small sequences of the vector are provided inline without
22// requiring any heap allocation.
23//
24// An `absl::InlinedVector<T, N>` specifies the default capacity `N` as one of
25// its template parameters. Instances where `size() <= N` hold contained
26// elements in inline space. Typically `N` is very small so that sequences that
27// are expected to be short do not require allocations.
28//
29// An `absl::InlinedVector` does not usually require a specific allocator. If
30// the inlined vector grows beyond its initial constraints, it will need to
31// allocate (as any normal `std::vector` would). This is usually performed with
32// the default allocator (defined as `std::allocator<T>`). Optionally, a custom
33// allocator type may be specified as `A` in `absl::InlinedVector<T, N, A>`.
34

59// -----------------------------------------------------------------------------
60// InlinedVector
61// -----------------------------------------------------------------------------
62//
63// An `absl::InlinedVector` is designed to be a drop-in replacement for
64// `std::vector` for use cases where the vector's size is sufficiently small
65// that it can be inlined. If the inlined vector does grow beyond its estimated
66// capacity, it will trigger an initial allocation on the heap, and will behave
67// as a `std::vector`. The API of the `absl::InlinedVector` within this file is
68// designed to cover the same API footprint as covered by `std::vector`.
69template <typename T, size_t N, typename A = std::allocator<T>>
70class InlinedVector {
71 static_assert(N > 0, "`absl::InlinedVector` requires an inlined capacity.");
72
73 using Storage = inlined_vector_internal::Storage<T, N, A>;
74
75 using AllocatorTraits = typename Storage::AllocatorTraits;
76 using RValueReference = typename Storage::RValueReference;
77 using MoveIterator = typename Storage::MoveIterator;
78 using IsMemcpyOk = typename Storage::IsMemcpyOk;
79
80 template <typename Iterator>
81 using IteratorValueAdapter =
82 typename Storage::template IteratorValueAdapter<Iterator>;
83 using CopyValueAdapter = typename Storage::CopyValueAdapter;
84 using DefaultValueAdapter = typename Storage::DefaultValueAdapter;
85
86 template <typename Iterator>
87 using EnableIfAtLeastForwardIterator = absl::enable_if_t<
88 inlined_vector_internal::IsAtLeastForwardIterator<Iterator>::value>;
89 template <typename Iterator>
90 using DisableIfAtLeastForwardIterator = absl::enable_if_t<
91 !inlined_vector_internal::IsAtLeastForwardIterator<Iterator>::value>;
92
93 public:
94 using allocator_type = typename Storage::allocator_type;
95 using value_type = typename Storage::value_type;
96 using pointer = typename Storage::pointer;
97 using const_pointer = typename Storage::const_pointer;
98 using size_type = typename Storage::size_type;
99 using difference_type = typename Storage::difference_type;
100 using reference = typename Storage::reference;
101 using const_reference = typename Storage::const_reference;
102 using iterator = typename Storage::iterator;
103 using const_iterator = typename Storage::const_iterator;
104 using reverse_iterator = typename Storage::reverse_iterator;
105 using const_reverse_iterator = typename Storage::const_reverse_iterator;

137 // Creates an inlined vector with elements constructed from the provided
138 // forward iterator range [`first`, `last`).
139 //

141 // this constructor with two integral arguments and a call to the above

152 // Creates an inlined vector with elements constructed from the provided input
153 // iterator range [`first`, `last`).
154 template <typename InputIterator,
155 DisableIfAtLeastForwardIterator<InputIterator>* = nullptr>
156 InlinedVector(InputIterator first, InputIterator last,
157 const allocator_type& alloc = allocator_type())

162 // Creates an inlined vector by copying the contents of `other` using
163 // `other`'s allocator.
164 InlinedVector(const InlinedVector& other)
165 : InlinedVector(other, *other.storage_.GetAllocPtr()) {}
166
167 // Creates an inlined vector by copying the contents of `other` using `alloc`.
168 InlinedVector(const InlinedVector& other, const allocator_type& alloc)
169 : storage_(alloc) {
170 if (other.empty()) {
171 // Empty; nothing to do.
172 } else if (IsMemcpyOk::value && !other.storage_.GetIsAllocated()) {
173 // Memcpy-able and do not need allocation.
174 storage_.MemcpyFrom(other.storage_);
175 } else {
176 storage_.InitFrom(other.storage_);
177 }
178 }
179
180 // Creates an inlined vector by moving in the contents of `other` without
181 // allocating. If `other` contains allocated memory, the newly-created inlined
182 // vector will take ownership of that memory. However, if `other` does not
183 // contain allocated memory, the newly-created inlined vector will perform
184 // element-wise move construction of the contents of `other`.
185 //
186 // NOTE: since no allocation is performed for the inlined vector in either
187 // case, the `noexcept(...)` specification depends on whether moving the
188 // underlying objects can throw. It is assumed assumed that...
189 // a) move constructors should only throw due to allocation failure.
190 // b) if `value_type`'s move constructor allocates, it uses the same
191 // allocation function as the inlined vector's allocator.
192 // Thus, the move constructor is non-throwing if the allocator is non-throwing
193 // or `value_type`'s move constructor is specified as `noexcept`.
194 InlinedVector(InlinedVector&& other) noexcept(
195 absl::allocator_is_nothrow<allocator_type>::value ||
196 std::is_nothrow_move_constructible<value_type>::value)
197 : storage_(*other.storage_.GetAllocPtr()) {
198 if (IsMemcpyOk::value) {
199 storage_.MemcpyFrom(other.storage_);
200
201 other.storage_.SetInlinedSize(0);
202 } else if (other.storage_.GetIsAllocated()) {
203 storage_.SetAllocatedData(other.storage_.GetAllocatedData(),
204 other.storage_.GetAllocatedCapacity());
205 storage_.SetAllocatedSize(other.storage_.GetSize());
206
207 other.storage_.SetInlinedSize(0);
208 } else {
209 IteratorValueAdapter<MoveIterator> other_values(
210 MoveIterator(other.storage_.GetInlinedData()));
211
212 inlined_vector_internal::ConstructElements(
213 storage_.GetAllocPtr(), storage_.GetInlinedData(), &other_values,
214 other.storage_.GetSize());
215
216 storage_.SetInlinedSize(other.storage_.GetSize());
217 }
218 }
219
220 // Creates an inlined vector by moving in the contents of `other` with a copy
221 // of `alloc`.
222 //
223 // NOTE: if `other`'s allocator is not equal to `alloc`, even if `other`
224 // contains allocated memory, this move constructor will still allocate. Since
225 // allocation is performed, this constructor can only be `noexcept` if the
226 // specified allocator is also `noexcept`.
227 InlinedVector(InlinedVector&& other, const allocator_type& alloc) noexcept(
228 absl::allocator_is_nothrow<allocator_type>::value)
229 : storage_(alloc) {
230 if (IsMemcpyOk::value) {
231 storage_.MemcpyFrom(other.storage_);
232
233 other.storage_.SetInlinedSize(0);
234 } else if ((*storage_.GetAllocPtr() == *other.storage_.GetAllocPtr()) &&
235 other.storage_.GetIsAllocated()) {
236 storage_.SetAllocatedData(other.storage_.GetAllocatedData(),
237 other.storage_.GetAllocatedCapacity());
238 storage_.SetAllocatedSize(other.storage_.GetSize());
239
240 other.storage_.SetInlinedSize(0);
241 } else {
242 storage_.Initialize(
243 IteratorValueAdapter<MoveIterator>(MoveIterator(other.data())),
244 other.size());
245 }
246 }

248 ~InlinedVector() {}

263
264 // `InlinedVector::max_size()`
265 //
266 // Returns the maximum number of elements the inlined vector can hold.
267 size_type max_size() const noexcept {
268 // One bit of the size storage is used to indicate whether the inlined
269 // vector contains allocated memory. As a result, the maximum size that the
270 // inlined vector can express is half of the max for `size_type`.
271 return (std::numeric_limits<size_type>::max)() / 2;
272 }
273
274 // `InlinedVector::capacity()`
275 //
276 // Returns the number of elements that could be stored in the inlined vector
277 // without requiring a reallocation.
278 //
279 // NOTE: for most inlined vectors, `capacity()` should be equal to the
280 // template parameter `N`. For inlined vectors which exceed this capacity,
281 // they will no longer be inlined and `capacity()` will equal the capactity of
282 // the allocated memory.
283 size_type capacity() const noexcept {
284 return storage_.GetIsAllocated() ? storage_.GetAllocatedCapacity()
285 : storage_.GetInlinedCapacity();
286 }

451 // `InlinedVector::operator=(...)`
452 //
453 // Replaces the elements of the inlined vector with copies of the elements of
454 // `list`.
455 InlinedVector& operator=(std::initializer_list<value_type> list) {
456 assign(list.begin(), list.end());
457
458 return *this;
459 }
460
461 // Overload of `InlinedVector::operator=(...)` that replaces the elements of
462 // the inlined vector with copies of the elements of `other`.
463 InlinedVector& operator=(const InlinedVector& other) {
464 if (ABSL_PREDICT_TRUE(this != std::addressof(other))) {
465 const_pointer other_data = other.data();
466 assign(other_data, other_data + other.size());
467 }
468
469 return *this;
470 }
471
472 // Overload of `InlinedVector::operator=(...)` that moves the elements of
473 // `other` into the inlined vector.
474 //
475 // NOTE: as a result of calling this overload, `other` is left in a valid but
476 // unspecified state.
477 InlinedVector& operator=(InlinedVector&& other) {
478 if (ABSL_PREDICT_TRUE(this != std::addressof(other))) {
479 if (IsMemcpyOk::value || other.storage_.GetIsAllocated()) {
480 inlined_vector_internal::DestroyElements(storage_.GetAllocPtr(), data(),
481 size());
482 storage_.DeallocateIfAllocated();
483 storage_.MemcpyFrom(other.storage_);
484
485 other.storage_.SetInlinedSize(0);
486 } else {
487 storage_.Assign(IteratorValueAdapter<MoveIterator>(
488 MoveIterator(other.storage_.GetInlinedData())),
489 other.size());
490 }
491 }
492
493 return *this;
494 }

509 // Overload of `InlinedVector::assign(...)` to replace the contents of the
510 // inlined vector with the range [`first`, `last`).
511 //
512 // NOTE: this overload is for iterators that are "forward" category or better.
513 template <typename ForwardIterator,
514 EnableIfAtLeastForwardIterator<ForwardIterator>* = nullptr>
515 void assign(ForwardIterator first, ForwardIterator last) {
516 storage_.Assign(IteratorValueAdapter<ForwardIterator>(first),
517 std::distance(first, last));
518 }
519
520 // Overload of `InlinedVector::assign(...)` to replace the contents of the
521 // inlined vector with the range [`first`, `last`).
522 //
523 // NOTE: this overload is for iterators that are "input" category.
524 template <typename InputIterator,
525 DisableIfAtLeastForwardIterator<InputIterator>* = nullptr>
526 void assign(InputIterator first, InputIterator last) {
527 size_type i = 0;
528 for (; i < size() && first != last; ++i, static_cast<void>(++first)) {
529 data()[i] = *first;
530 }
531
532 erase(data() + i, data() + size());
533 std::copy(first, last, std::back_inserter(*this));
534 }

557 // `InlinedVector::insert(...)`
558 //
559 // Inserts a copy of `v` at `pos`, returning an `iterator` to the newly
560 // inserted element.
561 iterator insert(const_iterator pos, const_reference v) {
562 return emplace(pos, v);
563 }
564
565 // Overload of `InlinedVector::insert(...)` that inserts `v` at `pos` using
566 // move semantics, returning an `iterator` to the newly inserted element.
567 iterator insert(const_iterator pos, RValueReference v) {
568 return emplace(pos, std::move(v));
569 }
570
571 // Overload of `InlinedVector::insert(...)` that inserts `n` contiguous copies
572 // of `v` starting at `pos`, returning an `iterator` pointing to the first of
573 // the newly inserted elements.
574 iterator insert(const_iterator pos, size_type n, const_reference v) {
575 ABSL_HARDENING_ASSERT(pos >= begin());
576 ABSL_HARDENING_ASSERT(pos <= end());
577
578 if (ABSL_PREDICT_TRUE(n != 0)) {
579 value_type dealias = v;
580 return storage_.Insert(pos, CopyValueAdapter(dealias), n);
581 } else {
582 return const_cast<iterator>(pos);
583 }
584 }
585
586 // Overload of `InlinedVector::insert(...)` that inserts copies of the
587 // elements of `list` starting at `pos`, returning an `iterator` pointing to
588 // the first of the newly inserted elements.
589 iterator insert(const_iterator pos, std::initializer_list<value_type> list) {
590 return insert(pos, list.begin(), list.end());
591 }
592
593 // Overload of `InlinedVector::insert(...)` that inserts the range [`first`,
594 // `last`) starting at `pos`, returning an `iterator` pointing to the first
595 // of the newly inserted elements.
596 //
597 // NOTE: this overload is for iterators that are "forward" category or better.
598 template <typename ForwardIterator,
599 EnableIfAtLeastForwardIterator<ForwardIterator>* = nullptr>
600 iterator insert(const_iterator pos, ForwardIterator first,
601 ForwardIterator last) {
602 ABSL_HARDENING_ASSERT(pos >= begin());
603 ABSL_HARDENING_ASSERT(pos <= end());
604
605 if (ABSL_PREDICT_TRUE(first != last)) {
606 return storage_.Insert(pos, IteratorValueAdapter<ForwardIterator>(first),
607 std::distance(first, last));
608 } else {
609 return const_cast<iterator>(pos);
610 }
611 }
612
613 // Overload of `InlinedVector::insert(...)` that inserts the range [`first`,
614 // `last`) starting at `pos`, returning an `iterator` pointing to the first
615 // of the newly inserted elements.
616 //
617 // NOTE: this overload is for iterators that are "input" category.
618 template <typename InputIterator,
619 DisableIfAtLeastForwardIterator<InputIterator>* = nullptr>
620 iterator insert(const_iterator pos, InputIterator first, InputIterator last) {
621 ABSL_HARDENING_ASSERT(pos >= begin());
622 ABSL_HARDENING_ASSERT(pos <= end());
623
624 size_type index = std::distance(cbegin(), pos);
625 for (size_type i = index; first != last; ++i, static_cast<void>(++first)) {
626 insert(data() + i, *first);
627 }
628
629 return iterator(data() + index);
630 }

632 // `InlinedVector::emplace(...)`
633 //
634 // Constructs and inserts an element using `args...` in the inlined vector at
635 // `pos`, returning an `iterator` pointing to the newly emplaced element.
636 template <typename... Args>
637 iterator emplace(const_iterator pos, Args&&... args) {
638 ABSL_HARDENING_ASSERT(pos >= begin());
639 ABSL_HARDENING_ASSERT(pos <= end());
640
641 value_type dealias(std::forward<Args>(args)...);
642 return storage_.Insert(pos,
643 IteratorValueAdapter<MoveIterator>(
644 MoveIterator(std::addressof(dealias))),
645 1);
646 }

660 void push_back(const_reference v) { static_cast<void>(emplace_back(v)); }
661
662 // Overload of `InlinedVector::push_back(...)` for inserting `v` at `end()`
663 // using move semantics.
664 void push_back(RValueReference v) {
665 static_cast<void>(emplace_back(std::move(v)));
666 }
667
668 // `InlinedVector::pop_back()`
669 //
670 // Destroys the element at `back()`, reducing the size by `1`.
671 void pop_back() noexcept {
672 ABSL_HARDENING_ASSERT(!empty());
673
674 AllocatorTraits::destroy(*storage_.GetAllocPtr(), data() + (size() - 1));
675 storage_.SubtractSize(1);
676 }
677
678 // `InlinedVector::erase(...)`
679 //
680 // Erases the element at `pos`, returning an `iterator` pointing to where the
681 // erased element was located.
682 //
683 // NOTE: may return `end()`, which is not dereferencable.
684 iterator erase(const_iterator pos) {
685 ABSL_HARDENING_ASSERT(pos >= begin());
686 ABSL_HARDENING_ASSERT(pos < end());
687
688 return storage_.Erase(pos, pos + 1);
689 }
690
691 // Overload of `InlinedVector::erase(...)` that erases every element in the
692 // range [`from`, `to`), returning an `iterator` pointing to where the first
693 // erased element was located.
694 //
695 // NOTE: may return `end()`, which is not dereferencable.
696 iterator erase(const_iterator from, const_iterator to) {
697 ABSL_HARDENING_ASSERT(from >= begin());
698 ABSL_HARDENING_ASSERT(from <= to);
699 ABSL_HARDENING_ASSERT(to <= end());
700
701 if (ABSL_PREDICT_TRUE(from != to)) {
702 return storage_.Erase(from, to);
703 } else {
704 return const_cast<iterator>(from);
705 }
706 }
707
708 // `InlinedVector::clear()`
709 //
710 // Destroys all elements in the inlined vector, setting the size to `0` and
711 // deallocating any held memory.
712 void clear() noexcept {
713 inlined_vector_internal::DestroyElements(storage_.GetAllocPtr(), data(),
714 size());
715 storage_.DeallocateIfAllocated();
716
717 storage_.SetInlinedSize(0);
718 }
719
720 // `InlinedVector::reserve(...)`
721 //
722 // Ensures that there is enough room for at least `n` elements.
723 void reserve(size_type n) { storage_.Reserve(n); }
724
725 // `InlinedVector::shrink_to_fit()`
726 //
727 // Reduces memory usage by freeing unused memory. After being called, calls to
728 // `capacity()` will be equal to `max(N, size())`.
729 //
730 // If `size() <= N` and the inlined vector contains allocated memory, the
731 // elements will all be moved to the inlined space and the allocated memory
732 // will be deallocated.
733 //
734 // If `size() > N` and `size() < capacity()`, the elements will be moved to a
735 // smaller allocation.
736 void shrink_to_fit() {
737 if (storage_.GetIsAllocated()) {
738 storage_.ShrinkToFit();
739 }
740 }
741
742 // `InlinedVector::swap(...)`
743 //
744 // Swaps the contents of the inlined vector with `other`.
745 void swap(InlinedVector& other) {
746 if (ABSL_PREDICT_TRUE(this != std::addressof(other))) {
747 storage_.Swap(std::addressof(other.storage_));
748 }
749 }

753 friend H AbslHashValue(H h, const absl::InlinedVector<TheT, TheN, TheA>& a);
754
755 Storage storage_;
