28/* BCOMPILER() - the compiler being used to build the project */
29#define BCOMPILER(BFEATURE) (defined BCOMPILER_##BFEATURE && BCOMPILER_##BFEATURE)
30
31/* BCOMPILER_HAS_CLANG_FEATURE() - whether the compiler supports a particular language or library feature. */
32/* http://clang.llvm.org/docs/LanguageExtensions.html#has-feature-and-has-extension */
33#ifdef __has_feature
34#define BCOMPILER_HAS_CLANG_FEATURE(x) __has_feature(x)
35#else
36#define BCOMPILER_HAS_CLANG_FEATURE(x) 0
37#endif
38
39#define BASAN_ENABLED BCOMPILER_HAS_CLANG_FEATURE(address_sanitizer)
40
41/* BCOMPILER_HAS_CLANG_DECLSPEC() - whether the compiler supports a Microsoft style __declspec attribute. */
42/* https://clang.llvm.org/docs/LanguageExtensions.html#has-declspec-attribute */
43#ifdef __has_declspec_attribute
44#define BCOMPILER_HAS_CLANG_DECLSPEC(x) __has_declspec_attribute(x)
45#else
46#define BCOMPILER_HAS_CLANG_DECLSPEC(x) 0
47#endif
48
49/* BCOMPILER(GCC_COMPATIBLE) - GNU Compiler Collection or compatibles */
50
51#if defined(__GNUC__)
52#define BCOMPILER_GCC_COMPATIBLE 1
53#endif
54
55/* BNO_RETURN */
56
57#if !defined(BNO_RETURN) && BCOMPILER(GCC_COMPATIBLE)
58#define BNO_RETURN __attribute((__noreturn__))
59#endif
60
61#if !defined(BNO_RETURN)
62#define BNO_RETURN
63#endif
64
65/* BFALLTHROUGH */
66
67#if !defined(BFALLTHROUGH) && defined(__cplusplus) && defined(__has_cpp_attribute)
68
69#if __has_cpp_attribute(fallthrough)
70#define BFALLTHROUGH [[fallthrough]]
71#elif __has_cpp_attribute(clang::fallthrough)
72#define BFALLTHROUGH [[clang::fallthrough]]
73#elif __has_cpp_attribute(gnu::fallthrough)
74#define BFALLTHROUGH [[gnu::fallthrough]]
75#endif
76
77#endif // !defined(BFALLTHROUGH) && defined(__cplusplus) && defined(__has_cpp_attribute)
78
79#if !defined(BFALLTHROUGH)
80#define BFALLTHROUGH
81#endif
82
83/* BUNUSED_TYPE_ALIAS */
84
85#if !defined(BUNUSED_TYPE_ALIAS) && BCOMPILER(GCC_COMPATIBLE)
86#define BUNUSED_TYPE_ALIAS __attribute__((unused))
87#endif
88
89#if !defined(BUNUSED_TYPE_ALIAS)
90#define BUNUSED_TYPE_ALIAS
