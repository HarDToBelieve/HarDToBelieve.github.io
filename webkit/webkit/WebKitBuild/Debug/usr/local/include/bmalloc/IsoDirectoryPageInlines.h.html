32template<typename Config>
33IsoDirectoryPage<Config>::IsoDirectoryPage(IsoHeapImpl<Config>& heap, unsigned index)
34 : payload(heap)
35 , m_index(index)
36{
37}
38
39template<typename Config>
40IsoDirectoryPage<Config>* IsoDirectoryPage<Config>::pageFor(IsoDirectory<Config, numPages>* payload)
41{
42 return reinterpret_cast<IsoDirectoryPage<Config>*>(
43 reinterpret_cast<char*>(payload) - BOFFSETOF(IsoDirectoryPage, payload));
44}
