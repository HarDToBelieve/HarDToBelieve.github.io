33namespace bmalloc {
34
35class IsoTLS;
36
37template<typename Entry>
38class IsoTLSEntryHolder {
39 MAKE_BMALLOCED;
40 IsoTLSEntryHolder(const IsoTLSEntryHolder&) = delete;
41 IsoTLSEntryHolder& operator=(const IsoTLSEntryHolder&) = delete;
42public:
43 template<typename... Args>
44 IsoTLSEntryHolder(Args&&... args)
45 : m_entry(std::forward<Args>(args)...)
46 {
47 IsoTLSLayout::get()->add(&m_entry);
48 RELEASE_BASSERT(m_entry.offset() != UINT_MAX);
49 }
50
51 inline const Entry& operator*() const { m_entry; }
52 inline Entry& operator*() { m_entry; }
53 inline const Entry* operator->() const { return &m_entry; }
54 inline Entry* operator->() { return &m_entry; }
55
56private:
57 Entry m_entry;
58};
59
60class BEXPORT IsoTLSEntry {
61 MAKE_BMALLOCED;
62 IsoTLSEntry(const IsoTLSEntry&) = delete;
63 IsoTLSEntry& operator=(const IsoTLSEntry&) = delete;
64public:
65 virtual ~IsoTLSEntry();
66
67 size_t offset() const { return m_offset; }
68 size_t alignment() const { return sizeof(void*); }
69 size_t size() const { return m_size; }
70 size_t extent() const { return m_offset + m_size; }
71
72 virtual void construct(void* entry) = 0;
73 virtual void move(void* src, void* dst) = 0;
74 virtual void destruct(void* entry) = 0;
75 virtual void scavenge(void* entry) = 0;
76
77 template<typename Func>
78 void walkUpToInclusive(IsoTLSEntry*, const Func&);
79
80protected:
81 IsoTLSEntry(size_t size);
82
83private:
84 friend class IsoTLS;
85 friend class IsoTLSLayout;
86
87 IsoTLSEntry* m_next { nullptr };
88
89 unsigned m_offset { UINT_MAX }; // Computed in constructor.
90 unsigned m_size;
91};
92
93template<typename EntryType>
94class DefaultIsoTLSEntry : public IsoTLSEntry {
95public:
96 ~DefaultIsoTLSEntry() = default;
97
98protected:
99 DefaultIsoTLSEntry();
100
101 // This clones src onto dst and then destructs src. Therefore, entry destructors cannot do
102 // scavenging.
103 void move(void* src, void* dst) override;
104
105 // Likewise, this is separate from scavenging. When the TLS is shutting down, we will be asked to
106 // scavenge and then we will be asked to destruct.
107 void destruct(void* entry) override;
108};
