26#ifndef LargeMap_h
27#define LargeMap_h
28
29#include "LargeRange.h"
30#include "Vector.h"
31#include <algorithm>
32
33namespace bmalloc {
34
35class LargeMap {
36public:
37 LargeRange* begin() { return m_free.begin(); }
38 LargeRange* end() { return m_free.end(); }
39
40 void add(const LargeRange&);
41 LargeRange remove(size_t alignment, size_t);
42 Vector<LargeRange>& ranges() { return m_free; }
43 void markAllAsEligibile();
44
45 size_t size() { return m_free.size(); }
46 LargeRange& at(size_t i) { return m_free[i]; }
47
48private:
49 Vector<LargeRange> m_free;
50};
51
52} // namespace bmalloc
53
54#endif // LargeMap_h
