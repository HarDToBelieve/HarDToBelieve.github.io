26#ifndef Object_h
27#define Object_h
28
29#include <cstddef>
30
31namespace bmalloc {
32
33class Chunk;
34class SmallLine;
35class SmallPage;
36
37class Object {
38public:
39 Object(void*);
40 Object(Chunk*, void*);
41 Object(Chunk* chunk, size_t offset)
42 : m_chunk(chunk)
43 , m_offset(offset)
44 {
45 }
46
47 Chunk* chunk() { return m_chunk; }
48 size_t offset() { return m_offset; }
49 char* address();
50
51 SmallLine* line();
52 SmallPage* page();
53
54 Object operator+(size_t);
55 Object operator-(size_t);
56 bool operator<=(const Object&);
57
58private:
59 Chunk* m_chunk;
60 size_t m_offset;
61};
62
63inline Object Object::operator+(size_t offset)
64{
65 return Object(m_chunk, m_offset + offset);
66}
67
68inline Object Object::operator-(size_t offset)
69{
70 return Object(m_chunk, m_offset - offset);
71}
72
73inline bool Object::operator<=(const Object& other)
74{
75 BASSERT(m_chunk == other.m_chunk);
76 return m_offset <= other.m_offset;
77}
78
79}; // namespace bmalloc
80
81#endif // Object_h
