26#ifndef SmallLine_h
27#define SmallLine_h
28
29#include "BAssert.h"
30#include "Mutex.h"
31#include "ObjectType.h"
32#include <mutex>
33
34namespace bmalloc {
35
36class SmallLine {
37public:
38 void ref(UniqueLockHolder&, unsigned char = 1);
39 bool deref(UniqueLockHolder&);
40 unsigned refCount(UniqueLockHolder&) { return m_refCount; }
41
42 char* begin();
43 char* end();
44
45private:
46 unsigned char m_refCount;
47
48static_assert(
49 smallLineSize / alignment <= std::numeric_limits<decltype(m_refCount)>::max(),
50 "maximum object count must fit in SmallLine::m_refCount");
51
52};
53
54inline void SmallLine::ref(UniqueLockHolder&, unsigned char refCount)
55{
56 BASSERT(!m_refCount);
57 m_refCount = refCount;
58}
59
60inline bool SmallLine::deref(UniqueLockHolder&)
61{
62 BASSERT(m_refCount);
63 --m_refCount;
64 return !m_refCount;
65}
66
67} // namespace bmalloc
68
69#endif // SmallLine_h
