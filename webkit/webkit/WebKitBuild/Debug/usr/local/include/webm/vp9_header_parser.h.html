8#ifndef LIBWEBM_COMMON_VP9_HEADER_PARSER_H_
9#define LIBWEBM_COMMON_VP9_HEADER_PARSER_H_
10
11#include <stddef.h>
12#include <stdint.h>
13
14namespace vp9_parser {
15
16const int kVp9FrameMarker = 2;
17const int kMinTileWidthB64 = 4;
18const int kMaxTileWidthB64 = 64;
19const int kRefFrames = 8;
20const int kRefsPerFrame = 3;
21const int kRefFrames_LOG2 = 3;
22const int kVpxCsBt601 = 1;
23const int kVpxCsSrgb = 7;
24const int kVpxCrStudioRange = 0;
25const int kVpxCrFullRange = 1;
26const int kMiSizeLog2 = 3;
27
28// Class to parse the header of a VP9 frame.
29class Vp9HeaderParser {
30 public:
31 Vp9HeaderParser()
32 : frame_(NULL),
33 frame_size_(0),
34 bit_offset_(0),
35 profile_(-1),
36 show_existing_frame_(0),
37 key_(0),
38 altref_(0),
39 error_resilient_mode_(0),
40 intra_only_(0),
41 reset_frame_context_(0),
42 bit_depth_(0),
43 color_space_(0),
44 color_range_(0),
45 subsampling_x_(0),
46 subsampling_y_(0),
47 refresh_frame_flags_(0),
48 width_(0),
49 height_(0),
50 row_tiles_(0),
51 column_tiles_(0),
52 frame_parallel_mode_(0) {}
53
54 // Parse the VP9 uncompressed header. The return values of the remaining
55 // functions are only valid on success.
56 bool ParseUncompressedHeader(const uint8_t* frame, size_t length);
57
58 size_t frame_size() const { return frame_size_; }
59 int profile() const { return profile_; }
60 int key() const { return key_; }
61 int altref() const { return altref_; }
62 int error_resilient_mode() const { return error_resilient_mode_; }
63 int bit_depth() const { return bit_depth_; }
64 int color_space() const { return color_space_; }
65 int width() const { return width_; }
66 int height() const { return height_; }
67 int refresh_frame_flags() const { return refresh_frame_flags_; }
68 int row_tiles() const { return row_tiles_; }
69 int column_tiles() const { return column_tiles_; }
70 int frame_parallel_mode() const { return frame_parallel_mode_; }
71
72 // WebKit Additions:
73 int color_range() const { return color_range_; }
74 int subsampling_x() const { return subsampling_x_; }
75 int subsampling_y() const { return subsampling_y_; }
76
77 private:
78 // Set the compressed VP9 frame.
79 bool SetFrame(const uint8_t* frame, size_t length);
80
81 // Returns the next bit of the frame.
82 int ReadBit();
83
84 // Returns the next |bits| of the frame.
85 int VpxReadLiteral(int bits);
86
87 // Returns true if the vp9 sync code is valid.
88 bool ValidateVp9SyncCode();
89
90 // Parses bit_depth_, color_space_, subsampling_x_, subsampling_y_, and
91 // color_range_.
92 void ParseColorSpace();
93
94 // Parses width and height of the frame.
95 void ParseFrameResolution();
96
97 // Parses frame_parallel_mode_. This function skips over some features.
98 void ParseFrameParallelMode();
99
100 // Parses row and column tiles. This function skips over some features.
101 void ParseTileInfo();
102 void SkipDeltaQ();
103 int AlignPowerOfTwo(int value, int n);
104
105 const uint8_t* frame_;
106 size_t frame_size_;
107 size_t bit_offset_;
108 int profile_;
109 int show_existing_frame_;
110 int key_;
111 int altref_;
112 int error_resilient_mode_;
113 int intra_only_;
114 int reset_frame_context_;
115 int bit_depth_;
116 int color_space_;
117 int color_range_;
118 int subsampling_x_;
119 int subsampling_y_;
120 int refresh_frame_flags_;
121 int width_;
122 int height_;
123 int row_tiles_;
124 int column_tiles_;
125 int frame_parallel_mode_;
126};
127
128} // namespace vp9_parser
129
130#endif // LIBWEBM_COMMON_VP9_HEADER_PARSER_H_
