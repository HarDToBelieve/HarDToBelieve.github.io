18namespace webrtc {
19
20// This interface exists to allow webrtc to be optionally built without
21// RtcEventLog support. A PeerConnectionFactory is constructed with an
22// RtcEventLogFactoryInterface, which may or may not be null.
23class RtcEventLogFactoryInterface {
24 public:
25 virtual ~RtcEventLogFactoryInterface() = default;
26
27 virtual std::unique_ptr<RtcEventLog> CreateRtcEventLog(
28 RtcEventLog::EncodingType encoding_type) = 0;
29};
30
31} // namespace webrtc
