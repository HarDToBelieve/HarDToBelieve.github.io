17namespace webrtc {
18
19// An observer for PeerConnectionInterface::SetRemoteDescription(). The
20// callback is invoked such that the state of the peer connection can be
21// examined to accurately reflect the effects of the SetRemoteDescription
22// operation.
23class SetRemoteDescriptionObserverInterface : public rtc::RefCountInterface {
24 public:
25 // On success, |error.ok()| is true.
26 virtual void OnSetRemoteDescriptionComplete(RTCError error) = 0;
27};

29} // namespace webrtc
