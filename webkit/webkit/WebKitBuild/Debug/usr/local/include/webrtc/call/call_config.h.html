10#ifndef CALL_CALL_CONFIG_H_
11#define CALL_CALL_CONFIG_H_

13#include "api/fec_controller.h"

23namespace webrtc {
24
25class AudioProcessing;
26class RtcEventLog;
27
28struct CallConfig {
29 // If |network_task_queue| is set to nullptr, Call will assume that network
30 // related callbacks will be made on the same TQ as the Call instance was
31 // constructed on.
32 explicit CallConfig(RtcEventLog* event_log,
33 TaskQueueBase* network_task_queue = nullptr);
34 CallConfig(const CallConfig&);
35 ~CallConfig();
36
37 // Bitrate config used until valid bitrate estimates are calculated. Also
38 // used to cap total bitrate used. This comes from the remote connection.
39 BitrateConstraints bitrate_config;
40
41 // AudioState which is possibly shared between multiple calls.
42 rtc::scoped_refptr<AudioState> audio_state;
43
44 // Audio Processing Module to be used in this call.
45 AudioProcessing* audio_processing = nullptr;
46
47 // RtcEventLog to use for this call. Required.
48 // Use webrtc::RtcEventLog::CreateNull() for a null implementation.
49 RtcEventLog* const event_log = nullptr;
50
51 // FecController to use for this call.
52 FecControllerFactoryInterface* fec_controller_factory = nullptr;
53
54 // Task Queue Factory to be used in this call. Required.
55 TaskQueueFactory* task_queue_factory = nullptr;
56
57 // NetworkStatePredictor to use for this call.
58 NetworkStatePredictorFactoryInterface* network_state_predictor_factory =
59 nullptr;
60
61 // Network controller factory to use for this call.
62 NetworkControllerFactoryInterface* network_controller_factory = nullptr;
63
64 // NetEq factory to use for this call.
65 NetEqFactory* neteq_factory = nullptr;
66
67 // Key-value mapping of internal configurations to apply,
68 // e.g. field trials.
69 const WebRtcKeyValueConfig* trials = nullptr;
70
71 TaskQueueBase* const network_task_queue_ = nullptr;
72};
73
74} // namespace webrtc
75
76#endif // CALL_CALL_CONFIG_H_
