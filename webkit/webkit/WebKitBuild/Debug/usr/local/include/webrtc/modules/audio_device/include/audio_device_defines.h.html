71// Helper class for storage of fundamental audio parameters such as sample rate,
72// number of channels, native buffer size etc.
73// Note that one audio frame can contain more than one channel sample and each
74// sample is assumed to be a 16-bit PCM sample. Hence, one audio frame in
75// stereo contains 2 * (16/8) = 4 bytes of data.
76class AudioParameters {
77 public:
78 // This implementation does only support 16-bit PCM samples.
79 static const size_t kBitsPerSample = 16;
80 AudioParameters()
81 : sample_rate_(0),
82 channels_(0),
83 frames_per_buffer_(0),
84 frames_per_10ms_buffer_(0) {}
85 AudioParameters(int sample_rate, size_t channels, size_t frames_per_buffer)
86 : sample_rate_(sample_rate),
87 channels_(channels),
88 frames_per_buffer_(frames_per_buffer),
89 frames_per_10ms_buffer_(static_cast<size_t>(sample_rate / 100)) {}
90 void reset(int sample_rate, size_t channels, size_t frames_per_buffer) {
91 sample_rate_ = sample_rate;
92 channels_ = channels;
93 frames_per_buffer_ = frames_per_buffer;
94 frames_per_10ms_buffer_ = static_cast<size_t>(sample_rate / 100);
95 }
96 size_t bits_per_sample() const { return kBitsPerSample; }
97 void reset(int sample_rate, size_t channels, double buffer_duration) {
98 reset(sample_rate, channels,
99 static_cast<size_t>(sample_rate * buffer_duration + 0.5));
100 }
101 void reset(int sample_rate, size_t channels) {
102 reset(sample_rate, channels, static_cast<size_t>(0));
103 }
104 int sample_rate() const { return sample_rate_; }
105 size_t channels() const { return channels_; }
106 size_t frames_per_buffer() const { return frames_per_buffer_; }
107 size_t frames_per_10ms_buffer() const { return frames_per_10ms_buffer_; }
108 size_t GetBytesPerFrame() const { return channels_ * kBitsPerSample / 8; }
109 size_t GetBytesPerBuffer() const {
110 return frames_per_buffer_ * GetBytesPerFrame();
111 }
112 // The WebRTC audio device buffer (ADB) only requires that the sample rate
113 // and number of channels are configured. Hence, to be "valid", only these
114 // two attributes must be set.
115 bool is_valid() const { return ((sample_rate_ > 0) && (channels_ > 0)); }
116 // Most platforms also require that a native buffer size is defined.
117 // An audio parameter instance is considered to be "complete" if it is both
118 // "valid" (can be used by the ADB) and also has a native frame size.
119 bool is_complete() const { return (is_valid() && (frames_per_buffer_ > 0)); }
120 size_t GetBytesPer10msBuffer() const {
121 return frames_per_10ms_buffer_ * GetBytesPerFrame();
122 }
123 double GetBufferSizeInMilliseconds() const {
124 if (sample_rate_ == 0)
125 return 0.0;
126 return frames_per_buffer_ / (sample_rate_ / 1000.0);
127 }
128 double GetBufferSizeInSeconds() const {
129 if (sample_rate_ == 0)
130 return 0.0;
131 return static_cast<double>(frames_per_buffer_) / (sample_rate_);
132 }
133 std::string ToString() const {
134 char ss_buf[1024];
135 rtc::SimpleStringBuilder ss(ss_buf);
136 ss << "AudioParameters: ";
137 ss << "sample_rate=" << sample_rate() << ", channels=" << channels();
138 ss << ", frames_per_buffer=" << frames_per_buffer();
139 ss << ", frames_per_10ms_buffer=" << frames_per_10ms_buffer();
140 ss << ", bytes_per_frame=" << GetBytesPerFrame();
141 ss << ", bytes_per_buffer=" << GetBytesPerBuffer();
142 ss << ", bytes_per_10ms_buffer=" << GetBytesPer10msBuffer();
143 ss << ", size_in_ms=" << GetBufferSizeInMilliseconds();
144 return ss.str();
145 }
146
147 private:
148 int sample_rate_;
149 size_t channels_;
150 size_t frames_per_buffer_;
151 size_t frames_per_10ms_buffer_;
152};
