20// Statistics for RTCP packet types.
21struct RtcpPacketTypeCounter {
22 RtcpPacketTypeCounter()
23 : first_packet_time_ms(-1),
24 nack_packets(0),
25 fir_packets(0),
26 pli_packets(0),
27 nack_requests(0),
28 unique_nack_requests(0) {}
29
30 void Add(const RtcpPacketTypeCounter& other) {
31 nack_packets += other.nack_packets;
32 fir_packets += other.fir_packets;
33 pli_packets += other.pli_packets;
34 nack_requests += other.nack_requests;
35 unique_nack_requests += other.unique_nack_requests;
36 if (other.first_packet_time_ms != -1 &&
37 (other.first_packet_time_ms < first_packet_time_ms ||
38 first_packet_time_ms == -1)) {
39 // Use oldest time.
40 first_packet_time_ms = other.first_packet_time_ms;
41 }
42 }
43
44 void Subtract(const RtcpPacketTypeCounter& other) {
45 nack_packets -= other.nack_packets;
46 fir_packets -= other.fir_packets;
47 pli_packets -= other.pli_packets;
48 nack_requests -= other.nack_requests;
49 unique_nack_requests -= other.unique_nack_requests;
50 if (other.first_packet_time_ms != -1 &&
51 (other.first_packet_time_ms > first_packet_time_ms ||
52 first_packet_time_ms == -1)) {
53 // Use youngest time.
54 first_packet_time_ms = other.first_packet_time_ms;
55 }
56 }
57
58 int64_t TimeSinceFirstPacketInMs(int64_t now_ms) const {
59 return (first_packet_time_ms == -1) ? -1 : (now_ms - first_packet_time_ms);
60 }
61
62 int UniqueNackRequestsInPercent() const {
63 if (nack_requests == 0) {
64 return 0;
65 }
66 return static_cast<int>((unique_nack_requests * 100.0f / nack_requests) +
67 0.5f);
68 }
69
70 int64_t first_packet_time_ms; // Time when first packet is sent/received.
71 uint32_t nack_packets; // Number of RTCP NACK packets.
72 uint32_t fir_packets; // Number of RTCP FIR packets.
73 uint32_t pli_packets; // Number of RTCP PLI packets.
74 uint32_t nack_requests; // Number of NACKed RTP packets.
75 uint32_t unique_nack_requests; // Number of unique NACKed RTP packets.
76};
