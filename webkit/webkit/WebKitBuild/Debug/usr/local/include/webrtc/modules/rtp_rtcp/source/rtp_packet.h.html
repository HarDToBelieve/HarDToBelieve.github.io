22namespace webrtc {

24class RtpPacket {
25 public:
26 using ExtensionType = RTPExtensionType;
27 using ExtensionManager = RtpHeaderExtensionMap;
28
29 // |extensions| required for SetExtension/ReserveExtension functions during
30 // packet creating and used if available in Parse function.
31 // Adding and getting extensions will fail until |extensions| is
32 // provided via constructor or IdentifyExtensions function.
33 // |*extensions| is only accessed during construction; the pointer is not
34 // stored.
35 RtpPacket();
36 explicit RtpPacket(const ExtensionManager* extensions);
37 RtpPacket(const RtpPacket&);
38 RtpPacket(const ExtensionManager* extensions, size_t capacity);
39 ~RtpPacket();
40
41 RtpPacket& operator=(const RtpPacket&) = default;
42
43 // Parse and copy given buffer into Packet.
44 // Does not require extension map to be registered (map is only required to
45 // read or allocate extensions in methods GetExtension, AllocateExtension,
46 // etc.)
47 bool Parse(const uint8_t* buffer, size_t size);
48 bool Parse(rtc::ArrayView<const uint8_t> packet);
49
50 // Parse and move given buffer into Packet.
51 bool Parse(rtc::CopyOnWriteBuffer packet);
52
53 // Maps extensions id to their types.
54 void IdentifyExtensions(const ExtensionManager& extensions);
55
56 // Header.
57 bool Marker() const { return marker_; }
58 uint8_t PayloadType() const { return payload_type_; }
59 uint16_t SequenceNumber() const { return sequence_number_; }
60 uint32_t Timestamp() const { return timestamp_; }
61 uint32_t Ssrc() const { return ssrc_; }
62 std::vector<uint32_t> Csrcs() const;
63
64 size_t headers_size() const { return payload_offset_; }
65
66 // Payload.
67 size_t payload_size() const { return payload_size_; }
68 size_t padding_size() const { return padding_size_; }
69 rtc::ArrayView<const uint8_t> payload() const {
70 return rtc::MakeArrayView(data() + payload_offset_, payload_size_);
71 }
72 rtc::CopyOnWriteBuffer PayloadBuffer() const {
73 return buffer_.Slice(payload_offset_, payload_size_);
74 }
75
76 // Buffer.
77 rtc::CopyOnWriteBuffer Buffer() const { return buffer_; }
78 size_t capacity() const { return buffer_.capacity(); }
79 size_t size() const {
80 return payload_offset_ + payload_size_ + padding_size_;
81 }
82 const uint8_t* data() const { return buffer_.cdata(); }
83 size_t FreeCapacity() const { return capacity() - size(); }
84 size_t MaxPayloadSize() const { return capacity() - headers_size(); }
85
86 // Reset fields and buffer.
87 void Clear();
88
89 // Header setters.
90 void CopyHeaderFrom(const RtpPacket& packet);
91 void SetMarker(bool marker_bit);
92 void SetPayloadType(uint8_t payload_type);
93 void SetSequenceNumber(uint16_t seq_no);
94 void SetTimestamp(uint32_t timestamp);
95 void SetSsrc(uint32_t ssrc);
96
97 // Fills with zeroes mutable extensions,
98 // which are modified after FEC protection is generated.
99 void ZeroMutableExtensions();
100
101 // Removes extension of given |type|, returns false is extension was not
102 // registered in packet's extension map or not present in the packet. Only
103 // extension that should be removed must be registered, other extensions may
104 // not be registered and will be preserved as is.
105 bool RemoveExtension(ExtensionType type);
106
107 // Writes csrc list. Assumes:
108 // a) There is enough room left in buffer.
109 // b) Extension headers, payload or padding data has not already been added.
110 void SetCsrcs(rtc::ArrayView<const uint32_t> csrcs);
111
112 // Header extensions.
113 template <typename Extension>
114 bool HasExtension() const;
115 bool HasExtension(ExtensionType type) const;
116
117 template <typename Extension, typename FirstValue, typename... Values>
118 bool GetExtension(FirstValue, Values...) const;
119
120 template <typename Extension>
121 absl::optional<typename Extension::value_type> GetExtension() const;
122
123 // Returns view of the raw extension or empty view on failure.
124 template <typename Extension>
125 rtc::ArrayView<const uint8_t> GetRawExtension() const;
126
127 template <typename Extension, typename... Values>
128 bool SetExtension(const Values&...);
129
130 template <typename Extension>
131 bool ReserveExtension();
132
133 // Find or allocate an extension |type|. Returns view of size |length|
134 // to write raw extension to or an empty view on failure.
135 rtc::ArrayView<uint8_t> AllocateExtension(ExtensionType type, size_t length);
136
137 // Find an extension |type|.
138 // Returns view of the raw extension or empty view on failure.
139 rtc::ArrayView<const uint8_t> FindExtension(ExtensionType type) const;
140
141 // Reserve size_bytes for payload. Returns nullptr on failure.
142 uint8_t* SetPayloadSize(size_t size_bytes);
143 // Same as SetPayloadSize but doesn't guarantee to keep current payload.
144 uint8_t* AllocatePayload(size_t size_bytes);
145
146 bool SetPadding(size_t padding_size);
147
148 // Returns debug string of RTP packet (without detailed extension info).
149 std::string ToString() const;
150
151 private:
152 struct ExtensionInfo {
153 explicit ExtensionInfo(uint8_t id) : ExtensionInfo(id, 0, 0) {}
154 ExtensionInfo(uint8_t id, uint8_t length, uint16_t offset)
155 : id(id), length(length), offset(offset) {}
156 uint8_t id;
157 uint8_t length;
158 uint16_t offset;
159 };
160
161 // Helper function for Parse. Fill header fields using data in given buffer,
162 // but does not touch packet own buffer, leaving packet in invalid state.
163 bool ParseBuffer(const uint8_t* buffer, size_t size);
164
165 // Returns pointer to extension info for a given id. Returns nullptr if not
166 // found.
167 const ExtensionInfo* FindExtensionInfo(int id) const;
168
169 // Returns reference to extension info for a given id. Creates a new entry
170 // with the specified id if not found.
171 ExtensionInfo& FindOrCreateExtensionInfo(int id);
172
173 // Allocates and returns place to store rtp header extension.
174 // Returns empty arrayview on failure.
175 rtc::ArrayView<uint8_t> AllocateRawExtension(int id, size_t length);
176
177 // Promotes existing one-byte header extensions to two-byte header extensions
178 // by rewriting the data and updates the corresponding extension offsets.
179 void PromoteToTwoByteHeaderExtension();
180
181 uint16_t SetExtensionLengthMaybeAddZeroPadding(size_t extensions_offset);
182
183 uint8_t* WriteAt(size_t offset) { return buffer_.MutableData() + offset; }
184 void WriteAt(size_t offset, uint8_t byte) {
185 buffer_.MutableData()[offset] = byte;
186 }
187 const uint8_t* ReadAt(size_t offset) const { return buffer_.data() + offset; }
188
189 // Header.
190 bool marker_;
191 uint8_t payload_type_;
192 uint8_t padding_size_;
193 uint16_t sequence_number_;
194 uint32_t timestamp_;
195 uint32_t ssrc_;
196 size_t payload_offset_; // Match header size with csrcs and extensions.
197 size_t payload_size_;
198
199 ExtensionManager extensions_;
200 std::vector<ExtensionInfo> extension_entries_;
201 size_t extensions_size_ = 0; // Unaligned.
202 rtc::CopyOnWriteBuffer buffer_;
203};
204
205template <typename Extension>
206bool RtpPacket::HasExtension() const {
207 return HasExtension(Extension::kId);
208}
209
210template <typename Extension, typename FirstValue, typename... Values>
211bool RtpPacket::GetExtension(FirstValue first, Values... values) const {
212 auto raw = FindExtension(Extension::kId);
213 if (raw.empty())
214 return false;
215 return Extension::Parse(raw, first, values...);
216}
217
218template <typename Extension>
219absl::optional<typename Extension::value_type> RtpPacket::GetExtension() const {
220 absl::optional<typename Extension::value_type> result;
221 auto raw = FindExtension(Extension::kId);
222 if (raw.empty() || !Extension::Parse(raw, &result.emplace()))
223 result = absl::nullopt;
224 return result;
225}
226
227template <typename Extension>
228rtc::ArrayView<const uint8_t> RtpPacket::GetRawExtension() const {
229 return FindExtension(Extension::kId);
230}
231
232template <typename Extension, typename... Values>
233bool RtpPacket::SetExtension(const Values&... values) {
234 const size_t value_size = Extension::ValueSize(values...);
235 auto buffer = AllocateExtension(Extension::kId, value_size);
236 if (buffer.empty())
237 return false;
238 return Extension::Write(buffer, values...);
239}
240
241template <typename Extension>
242bool RtpPacket::ReserveExtension() {
243 auto buffer = AllocateExtension(Extension::kId, Extension::kValueSizeBytes);
244 if (buffer.empty())
245 return false;
246 memset(buffer.data(), 0, Extension::kValueSizeBytes);
247 return true;
248}
