36// A bridge between a packet-oriented/transport-type interface on
37// the bottom and a StreamInterface on the top.
38class StreamInterfaceChannel : public rtc::StreamInterface {
39 public:
40 explicit StreamInterfaceChannel(IceTransportInternal* ice_transport);
41
42 // Push in a packet; this gets pulled out from Read().
43 bool OnPacketReceived(const char* data, size_t size);
44
45 // Implementations of StreamInterface
46 rtc::StreamState GetState() const override;
47 void Close() override;
48 rtc::StreamResult Read(void* buffer,
49 size_t buffer_len,
50 size_t* read,
51 int* error) override;
52 rtc::StreamResult Write(const void* data,
53 size_t data_len,
54 size_t* written,
55 int* error) override;
56
57 private:
58 RTC_NO_UNIQUE_ADDRESS webrtc::SequenceChecker sequence_checker_;
59 IceTransportInternal* const ice_transport_; // owned by DtlsTransport
60 rtc::StreamState state_ RTC_GUARDED_BY(sequence_checker_);
61 rtc::BufferQueue packets_ RTC_GUARDED_BY(sequence_checker_);
62
63 RTC_DISALLOW_COPY_AND_ASSIGN(StreamInterfaceChannel);
64};

66// This class provides a DTLS SSLStreamAdapter inside a TransportChannel-style
67// packet-based interface, wrapping an existing TransportChannel instance
68// (e.g a P2PTransportChannel)
69// Here's the way this works:
70//
71// DtlsTransport {
72// SSLStreamAdapter* dtls_ {
73// StreamInterfaceChannel downward_ {
74// IceTransportInternal* ice_transport_;
75// }
76// }
77// }
78//
79// - Data which comes into DtlsTransport from the underlying
80// ice_transport_ via OnReadPacket() is checked for whether it is DTLS
81// or not, and if it is, is passed to DtlsTransport::HandleDtlsPacket,
82// which pushes it into to downward_. dtls_ is listening for events on
83// downward_, so it immediately calls downward_->Read().
84//
85// - Data written to DtlsTransport is passed either to downward_ or directly
86// to ice_transport_, depending on whether DTLS is negotiated and whether
87// the flags include PF_SRTP_BYPASS
88//
89// - The SSLStreamAdapter writes to downward_->Write() which translates it
90// into packet writes on ice_transport_.
91//
92// This class is not thread safe; all methods must be called on the same thread
93// as the constructor.
94class DtlsTransport : public DtlsTransportInternal {
95 public:
96 // |ice_transport| is the ICE transport this DTLS transport is wrapping. It
97 // must outlive this DTLS transport.
98 //
99 // |crypto_options| are the options used for the DTLS handshake. This affects
100 // whether GCM crypto suites are negotiated.
101 //
102 // |event_log| is an optional RtcEventLog for logging state changes. It should
103 // outlive the DtlsTransport.
104 DtlsTransport(
105 IceTransportInternal* ice_transport,
106 const webrtc::CryptoOptions& crypto_options,
107 webrtc::RtcEventLog* event_log,
108 rtc::SSLProtocolVersion max_version = rtc::SSL_PROTOCOL_DTLS_12);
109
110 ~DtlsTransport() override;
111
112 DtlsTransportState dtls_state() const override;
113 const std::string& transport_name() const override;
114 int component() const override;
115
116 // DTLS is active if a local certificate was set. Otherwise this acts in a
117 // "passthrough" mode, sending packets directly through the underlying ICE
118 // transport.
119 // TODO(deadbeef): Remove this weirdness, and handle it in the upper layers.
120 bool IsDtlsActive() const override;
121
122 // SetLocalCertificate is what makes DTLS active. It must be called before
123 // SetRemoteFinterprint.
124 // TODO(deadbeef): Once DtlsTransport no longer has the concept of being
125 // "active" or not (acting as a passthrough if not active), just require this
126 // certificate on construction or "Start".
127 bool SetLocalCertificate(
128 const rtc::scoped_refptr<rtc::RTCCertificate>& certificate) override;
129 rtc::scoped_refptr<rtc::RTCCertificate> GetLocalCertificate() const override;
130
131 // SetRemoteFingerprint must be called after SetLocalCertificate, and any
132 // other methods like SetDtlsRole. It's what triggers the actual DTLS setup.
133 // TODO(deadbeef): Rename to "Start" like in ORTC?
134 bool SetRemoteFingerprint(const std::string& digest_alg,
135 const uint8_t* digest,
136 size_t digest_len) override;
137
138 // Called to send a packet (via DTLS, if turned on).
139 int SendPacket(const char* data,
140 size_t size,
141 const rtc::PacketOptions& options,
142 int flags) override;
143
144 bool GetOption(rtc::Socket::Option opt, int* value) override;
145
146 // Find out which TLS version was negotiated
147 bool GetSslVersionBytes(int* version) const override;
148 // Find out which DTLS-SRTP cipher was negotiated
149 bool GetSrtpCryptoSuite(int* cipher) override;
150
151 bool GetDtlsRole(rtc::SSLRole* role) const override;
152 bool SetDtlsRole(rtc::SSLRole role) override;
153
154 // Find out which DTLS cipher was negotiated
155 bool GetSslCipherSuite(int* cipher) override;
156
157 // Once DTLS has been established, this method retrieves the certificate
158 // chain in use by the remote peer, for use in external identity
159 // verification.
160 std::unique_ptr<rtc::SSLCertChain> GetRemoteSSLCertChain() const override;
161
162 // Once DTLS has established (i.e., this ice_transport is writable), this
163 // method extracts the keys negotiated during the DTLS handshake, for use in
164 // external encryption. DTLS-SRTP uses this to extract the needed SRTP keys.
165 // See the SSLStreamAdapter documentation for info on the specific parameters.
166 bool ExportKeyingMaterial(const std::string& label,
167 const uint8_t* context,
168 size_t context_len,
169 bool use_context,
170 uint8_t* result,
171 size_t result_len) override;
172
173 IceTransportInternal* ice_transport() override;
174
175 // For informational purposes. Tells if the DTLS handshake has finished.
176 // This may be true even if writable() is false, if the remote fingerprint
177 // has not yet been verified.
178 bool IsDtlsConnected();
179
180 bool receiving() const override;
181 bool writable() const override;
182
183 int GetError() override;
184
185 absl::optional<rtc::NetworkRoute> network_route() const override;
186
187 int SetOption(rtc::Socket::Option opt, int value) override;
188
189 std::string ToString() const {
190 const absl::string_view RECEIVING_ABBREV[2] = {"_", "R"};
191 const absl::string_view WRITABLE_ABBREV[2] = {"_", "W"};
192 rtc::StringBuilder sb;
193 sb << "DtlsTransport[" << transport_name() << "|" << component_ << "|"
194 << RECEIVING_ABBREV[receiving()] << WRITABLE_ABBREV[writable()] << "]";
195 return sb.Release();
196 }
197
198 private:
199 void ConnectToIceTransport();
200
201 void OnWritableState(rtc::PacketTransportInternal* transport);
202 void OnReadPacket(rtc::PacketTransportInternal* transport,
203 const char* data,
204 size_t size,
205 const int64_t& packet_time_us,
206 int flags);
207 void OnSentPacket(rtc::PacketTransportInternal* transport,
208 const rtc::SentPacket& sent_packet);
209 void OnReadyToSend(rtc::PacketTransportInternal* transport);
210 void OnReceivingState(rtc::PacketTransportInternal* transport);
211 void OnDtlsEvent(rtc::StreamInterface* stream_, int sig, int err);
212 void OnNetworkRouteChanged(absl::optional<rtc::NetworkRoute> network_route);
213 bool SetupDtls();
214 void MaybeStartDtls();
215 bool HandleDtlsPacket(const char* data, size_t size);
216 void OnDtlsHandshakeError(rtc::SSLHandshakeError error);
217 void ConfigureHandshakeTimeout();
218
219 void set_receiving(bool receiving);
220 void set_writable(bool writable);
221 // Sets the DTLS state, signaling if necessary.
222 void set_dtls_state(DtlsTransportState state);
223
224 webrtc::SequenceChecker thread_checker_;
225
226 const int component_;
227 DtlsTransportState dtls_state_ = DTLS_TRANSPORT_NEW;
228 // Underlying ice_transport, not owned by this class.
229 IceTransportInternal* const ice_transport_;
230 std::unique_ptr<rtc::SSLStreamAdapter> dtls_; // The DTLS stream
231 StreamInterfaceChannel*
232 downward_; // Wrapper for ice_transport_, owned by dtls_.
233 const std::vector<int> srtp_ciphers_; // SRTP ciphers to use with DTLS.
234 bool dtls_active_ = false;
235 rtc::scoped_refptr<rtc::RTCCertificate> local_certificate_;
236 absl::optional<rtc::SSLRole> dtls_role_;
237 const rtc::SSLProtocolVersion ssl_max_version_;
238 rtc::Buffer remote_fingerprint_value_;
239 std::string remote_fingerprint_algorithm_;
240
241 // Cached DTLS ClientHello packet that was received before we started the
242 // DTLS handshake. This could happen if the hello was received before the
243 // ice transport became writable, or before a remote fingerprint was received.
244 rtc::Buffer cached_client_hello_;
245
246 bool receiving_ = false;
247 bool writable_ = false;
248
249 webrtc::RtcEventLog* const event_log_;
250
251 RTC_DISALLOW_COPY_AND_ASSIGN(DtlsTransport);
252};
