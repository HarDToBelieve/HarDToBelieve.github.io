38// ChannelManager allows the MediaEngine to run on a separate thread, and takes
39// care of marshalling calls between threads. It also creates and keeps track of
40// voice and video channels; by doing so, it can temporarily pause all the
41// channels when a new audio or video device is chosen. The voice and video
42// channels are stored in separate vectors, to easily allow operations on just
43// voice or just video channels.
44// ChannelManager also allows the application to discover what devices it has
45// using device manager.
46class ChannelManager final {
47 public:
48 // Returns an initialized instance of ChannelManager.
49 // If media_engine is non-nullptr, then the returned ChannelManager instance
50 // will own that reference and media engine initialization
51 static std::unique_ptr<ChannelManager> Create(
52 std::unique_ptr<MediaEngineInterface> media_engine,
53 bool enable_rtx,
54 rtc::Thread* worker_thread,
55 rtc::Thread* network_thread);
56
57 ChannelManager() = delete;
58 ~ChannelManager();
59
60 rtc::Thread* worker_thread() const { return worker_thread_; }
61 rtc::Thread* network_thread() const { return network_thread_; }
62 MediaEngineInterface* media_engine() { return media_engine_.get(); }
63
64 // Retrieves the list of supported audio & video codec types.
65 // Can be called before starting the media engine.
66 void GetSupportedAudioSendCodecs(std::vector<AudioCodec>* codecs) const;
67 void GetSupportedAudioReceiveCodecs(std::vector<AudioCodec>* codecs) const;
68 void GetSupportedVideoSendCodecs(std::vector<VideoCodec>* codecs) const;
69 void GetSupportedVideoReceiveCodecs(std::vector<VideoCodec>* codecs) const;
70 RtpHeaderExtensions GetDefaultEnabledAudioRtpHeaderExtensions() const;
71 std::vector<webrtc::RtpHeaderExtensionCapability>
72 GetSupportedAudioRtpHeaderExtensions() const;
73 RtpHeaderExtensions GetDefaultEnabledVideoRtpHeaderExtensions() const;
74 std::vector<webrtc::RtpHeaderExtensionCapability>
75 GetSupportedVideoRtpHeaderExtensions() const;
76
77 // The operations below all occur on the worker thread.
78 // ChannelManager retains ownership of the created channels, so clients should
79 // call the appropriate Destroy*Channel method when done.
80
81 // Creates a voice channel, to be associated with the specified session.
82 VoiceChannel* CreateVoiceChannel(webrtc::Call* call,
83 const MediaConfig& media_config,
84 webrtc::RtpTransportInternal* rtp_transport,
85 rtc::Thread* signaling_thread,
86 const std::string& content_name,
87 bool srtp_required,
88 const webrtc::CryptoOptions& crypto_options,
89 rtc::UniqueRandomIdGenerator* ssrc_generator,
90 const AudioOptions& options);
91 // Destroys a voice channel created by CreateVoiceChannel.
92 void DestroyVoiceChannel(VoiceChannel* voice_channel);
93
94 // Creates a video channel, synced with the specified voice channel, and
95 // associated with the specified session.
96 // Version of the above that takes PacketTransportInternal.
97 VideoChannel* CreateVideoChannel(
98 webrtc::Call* call,
99 const MediaConfig& media_config,
100 webrtc::RtpTransportInternal* rtp_transport,
101 rtc::Thread* signaling_thread,
102 const std::string& content_name,
103 bool srtp_required,
104 const webrtc::CryptoOptions& crypto_options,
105 rtc::UniqueRandomIdGenerator* ssrc_generator,
106 const VideoOptions& options,
107 webrtc::VideoBitrateAllocatorFactory* video_bitrate_allocator_factory);
108 // Destroys a video channel created by CreateVideoChannel.
109 void DestroyVideoChannel(VideoChannel* video_channel);
110
111 // Starts AEC dump using existing file, with a specified maximum file size in
112 // bytes. When the limit is reached, logging will stop and the file will be
113 // closed. If max_size_bytes is set to <= 0, no limit will be used.
114 bool StartAecDump(webrtc::FileWrapper file, int64_t max_size_bytes);
115
116 // Stops recording AEC dump.
117 void StopAecDump();
118
119 private:
120 ChannelManager(std::unique_ptr<MediaEngineInterface> media_engine,
121 bool enable_rtx,
122 rtc::Thread* worker_thread,
123 rtc::Thread* network_thread);
124
125 const std::unique_ptr<MediaEngineInterface> media_engine_; // Nullable.
126 rtc::Thread* const worker_thread_;
127 rtc::Thread* const network_thread_;
128
129 // Vector contents are non-null.
130 std::vector<std::unique_ptr<VoiceChannel>> voice_channels_
131 RTC_GUARDED_BY(worker_thread_);
132 std::vector<std::unique_ptr<VideoChannel>> video_channels_
133 RTC_GUARDED_BY(worker_thread_);
134
135 const bool enable_rtx_;
136};
