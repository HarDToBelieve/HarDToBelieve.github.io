45// Options for an RtpSender contained with an media description/"m=" section.
46// Note: Spec-compliant Simulcast and legacy simulcast are mutually exclusive.
47struct SenderOptions {
48 std::string track_id;
49 std::vector<std::string> stream_ids;
50 // Use RIDs and Simulcast Layers to indicate spec-compliant Simulcast.
51 std::vector<RidDescription> rids;
52 SimulcastLayerList simulcast_layers;
53 // Use |num_sim_layers| to indicate legacy simulcast.
54 int num_sim_layers;
55};
56
57// Options for an individual media description/"m=" section.
58struct MediaDescriptionOptions {
59 MediaDescriptionOptions(MediaType type,
60 const std::string& mid,
61 webrtc::RtpTransceiverDirection direction,
62 bool stopped)
63 : type(type), mid(mid), direction(direction), stopped(stopped) {}
64
65 // TODO(deadbeef): When we don't support Plan B, there will only be one
66 // sender per media description and this can be simplified.
67 void AddAudioSender(const std::string& track_id,
68 const std::vector<std::string>& stream_ids);
69 void AddVideoSender(const std::string& track_id,
70 const std::vector<std::string>& stream_ids,
71 const std::vector<RidDescription>& rids,
72 const SimulcastLayerList& simulcast_layers,
73 int num_sim_layers);
74
75 MediaType type;
76 std::string mid;
77 webrtc::RtpTransceiverDirection direction;
78 bool stopped;
79 TransportOptions transport_options;
80 // Note: There's no equivalent "RtpReceiverOptions" because only send
81 // stream information goes in the local descriptions.
82 std::vector<SenderOptions> sender_options;
83 std::vector<webrtc::RtpCodecCapability> codec_preferences;
84 std::vector<webrtc::RtpHeaderExtensionCapability> header_extensions;
85
86 private:
87 // Doesn't DCHECK on |type|.
88 void AddSenderInternal(const std::string& track_id,
89 const std::vector<std::string>& stream_ids,
90 const std::vector<RidDescription>& rids,
91 const SimulcastLayerList& simulcast_layers,
92 int num_sim_layers);
93};

95// Provides a mechanism for describing how m= sections should be generated.
96// The m= section with index X will use media_description_options[X]. There
97// must be an option for each existing section if creating an answer, or a
98// subsequent offer.
99struct MediaSessionOptions {
100 MediaSessionOptions() {}
101
102 bool has_audio() const { return HasMediaDescription(MEDIA_TYPE_AUDIO); }
103 bool has_video() const { return HasMediaDescription(MEDIA_TYPE_VIDEO); }
104 bool has_data() const { return HasMediaDescription(MEDIA_TYPE_DATA); }
105
106 bool HasMediaDescription(MediaType type) const;
107
108 bool vad_enabled = true; // When disabled, removes all CN codecs from SDP.
109 bool rtcp_mux_enabled = true;
110 bool bundle_enabled = false;
111 bool offer_extmap_allow_mixed = false;
112 bool raw_packetization_for_video = false;
113 std::string rtcp_cname = kDefaultRtcpCname;
114 webrtc::CryptoOptions crypto_options;
115 // List of media description options in the same order that the media
116 // descriptions will be generated.
117 std::vector<MediaDescriptionOptions> media_description_options;
118 std::vector<IceParameters> pooled_ice_credentials;
119
120 // Use the draft-ietf-mmusic-sctp-sdp-03 obsolete syntax for SCTP
121 // datachannels.
122 // Default is true for backwards compatibility with clients that use
123 // this internal interface.
124 bool use_obsolete_sctp_sdp = true;
125};
