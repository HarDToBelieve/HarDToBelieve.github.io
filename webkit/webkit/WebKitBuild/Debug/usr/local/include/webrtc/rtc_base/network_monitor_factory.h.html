14namespace rtc {
15
16// Forward declaring this so it's not part of the API surface; it's only
17// expected to be used by Android/iOS SDK code.
18class NetworkMonitorInterface;
19
20/*
21 * NetworkMonitorFactory creates NetworkMonitors.
22 * Note that CreateNetworkMonitor is expected to be called on the network
23 * thread with the returned object only being used on that thread thereafter.
24 */
25class NetworkMonitorFactory {
26 public:
27 virtual NetworkMonitorInterface* CreateNetworkMonitor() = 0;
28
29 virtual ~NetworkMonitorFactory();
30
31 protected:
32 NetworkMonitorFactory();
33};

35} // namespace rtc
