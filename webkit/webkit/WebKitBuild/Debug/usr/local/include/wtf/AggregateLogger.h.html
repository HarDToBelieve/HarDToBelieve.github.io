26#pragma once
27
28#include <wtf/Algorithms.h>
29#include <wtf/HashSet.h>
30#include <wtf/Logger.h>
31#include <wtf/Ref.h>
32#include <wtf/RefPtr.h>
33
34namespace WTF {
35
36class AggregateLogger : public Logger {
37 WTF_MAKE_NONCOPYABLE(AggregateLogger);
38public:
39
40 static Ref<AggregateLogger> create(const void* owner)
41 {
42 return adoptRef(*new AggregateLogger(owner));
43 }
44
45 void addLogger(const Logger& logger)
46 {
47 m_loggers.add(&logger);
48 }
49
50 void removeLogger(const Logger& logger)
51 {
52 m_loggers.remove(&logger);
53 }
54
55 template<typename... Arguments>
56 inline void logAlways(WTFLogChannel& channel, UNUSED_FUNCTION const Arguments&... arguments) const
57 {
58#if RELEASE_LOG_DISABLED
59 // "Standard" WebCore logging goes to stderr, which is captured in layout test output and can generally be a problem
60 // on some systems, so don't allow it.
61 UNUSED_PARAM(channel);
62#else
63 log(channel, WTFLogLevel::Always, arguments...);
64#endif
65 }
66
67 template<typename... Arguments>
68 inline void error(WTFLogChannel& channel, const Arguments&... arguments) const
69 {
70 log(channel, WTFLogLevel::Error, arguments...);
71 }
72
73 template<typename... Arguments>
74 inline void warning(WTFLogChannel& channel, const Arguments&... arguments) const
75 {
76 log(channel, WTFLogLevel::Warning, arguments...);
77 }
78
79 template<typename... Arguments>
80 inline void info(WTFLogChannel& channel, const Arguments&... arguments) const
81 {
82 log(channel, WTFLogLevel::Info, arguments...);
83 }
84
85 template<typename... Arguments>
86 inline void debug(WTFLogChannel& channel, const Arguments&... arguments) const
87 {
88 log(channel, WTFLogLevel::Debug, arguments...);
89 }
90
91 inline bool willLog(const WTFLogChannel& channel, WTFLogLevel level) const
92 {
93 return allOf(m_loggers, [channel, level] (auto& logger) { return logger->willLog(channel, level); });
94 }
95
96private:
97 AggregateLogger(const void* owner)
98 : Logger(owner)
99 {
100 }
101
102 template<typename... Argument>
103 inline void log(WTFLogChannel& channel, WTFLogLevel level, const Argument&... arguments) const
104 {
105 if (!willLog(channel, level))
106 return;
107
108 Logger::log(channel, level, arguments...);
109
110 for (const auto& logger : m_loggers) {
111 for (Observer& observer : logger->observers())
112 observer.didLogMessage(channel, level, { ConsoleLogValue<Argument>::toValue(arguments)... });
113 }
114 }
115
116 HashSet<RefPtr<const Logger>> m_loggers;
117};
118
119} // namespace WTF
120
121using WTF::AggregateLogger;
