137/* CRASH() - Raises a fatal error resulting in program termination and triggering either the debugger or the crash reporter.
138
139 Use CRASH() in response to known, unrecoverable errors like out-of-memory.
140 Macro is enabled in both debug and release mode.
141 To test for unknown errors and verify assumptions, use ASSERT instead, to avoid impacting performance in release builds.
142
143 Signals are ignored by the crash reporter on OS X so we must do better.
144*/

232#define WTFBreakpointTrap() __debugbreak()
233#elif ASAN_ENABLED
234#define WTFBreakpointTrap() __builtin_trap()
235#elif CPU(X86_64) || CPU(X86)
236#define WTFBreakpointTrap() asm volatile ("int3")
237#elif CPU(ARM_THUMB2)
238#define WTFBreakpointTrap() asm volatile ("bkpt #0")
239#elif CPU(ARM64)
240#define WTFBreakpointTrap() asm volatile ("brk #0xc471")
241#else
242#define WTFBreakpointTrap() WTFCrash() // Not implemented.
243#endif
244
245#if COMPILER(MSVC)
246#define WTFBreakpointTrapUnderConstexprContext() __debugbreak()
247#else
248#define WTFBreakpointTrapUnderConstexprContext() __builtin_trap()
249#endif

251#ifndef CRASH
252
253#if defined(NDEBUG) && (OS(DARWIN) || PLATFORM(PLAYSTATION))
254// Crash with a SIGTRAP i.e EXC_BREAKPOINT.
255// We are not using __builtin_trap because it is only guaranteed to abort, but not necessarily
256// trigger a SIGTRAP. Instead, we use inline asm to ensure that we trigger the SIGTRAP.
257#define CRASH() do { \
258 WTFBreakpointTrap(); \
259 __builtin_unreachable(); \
260} while (0)
261#define CRASH_UNDER_CONSTEXPR_CONTEXT() do { \
262 WTFBreakpointTrapUnderConstexprContext(); \
263 __builtin_unreachable(); \
264} while (0)
265#elif !ENABLE(DEVELOPER_MODE) && !OS(DARWIN)
266#ifdef __cplusplus
267#define CRASH() std::abort()
268#define CRASH_UNDER_CONSTEXPR_CONTEXT() std::abort()
269#else
270#define CRASH() abort()
271#define CRASH_UNDER_CONSTEXPR_CONTEXT() abort()
272#endif // __cplusplus
273#else
274#define CRASH() WTFCrash()
275#define CRASH_UNDER_CONSTEXPR_CONTEXT() WTFCrash()
276#endif
277
278#endif // !defined(CRASH)

322#define ASSERT(assertion, ...) ((void)0)
323#define ASSERT_UNDER_CONSTEXPR_CONTEXT(assertion) ((void)0)
324#define ASSERT_AT(assertion, file, line, function) ((void)0)
325#define ASSERT_NOT_REACHED(...) ((void)0)
326#define ASSERT_NOT_IMPLEMENTED_YET() ((void)0)

330#define ASSERT_UNUSED(variable, assertion, ...) ((void)variable)

332#if ENABLE(SECURITY_ASSERTIONS)
333#define ASSERT_WITH_SECURITY_IMPLICATION(assertion) \
334 (!(assertion) ? \
335 (WTFReportAssertionFailure(__FILE__, __LINE__, WTF_PRETTY_FUNCTION, #assertion), \
336 CRASH_WITH_SECURITY_IMPLICATION()) : \
337 (void)0)
338
339#define ASSERT_WITH_SECURITY_IMPLICATION_DISABLED 0
340#else /* not ENABLE(SECURITY_ASSERTIONS) */
341#define ASSERT_WITH_SECURITY_IMPLICATION(assertion) ((void)0)
342#define ASSERT_WITH_SECURITY_IMPLICATION_DISABLED 1
343#endif /* ENABLE(SECURITY_ASSERTIONS) */
344
345#else /* ASSERT_ENABLED */
346
347#define ASSERT(assertion, ...) do { \
348 if (!(assertion)) { \
349 WTFReportAssertionFailure(__FILE__, __LINE__, WTF_PRETTY_FUNCTION, #assertion); \
350 CRASH_WITH_INFO(__VA_ARGS__); \
351 } \
352} while (0)
353
354#define ASSERT_UNDER_CONSTEXPR_CONTEXT(assertion) do { \
355 if (!(assertion)) { \
356 WTFReportAssertionFailure(__FILE__, __LINE__, WTF_PRETTY_FUNCTION, #assertion); \
357 CRASH_UNDER_CONSTEXPR_CONTEXT(); \
358 } \
359} while (0)
360
361#define ASSERT_AT(assertion, file, line, function) do { \
362 if (!(assertion)) { \
363 WTFReportAssertionFailure(file, line, function, #assertion); \
364 CRASH(); \
365 } \
366} while (0)
367
368#define ASSERT_NOT_REACHED(...) do { \
369 WTFReportAssertionFailure(__FILE__, __LINE__, WTF_PRETTY_FUNCTION, 0); \
370 CRASH_WITH_INFO(__VA_ARGS__); \
371} while (0)
372
373#define ASSERT_NOT_IMPLEMENTED_YET() do { \
374 WTFReportNotImplementedYet(__FILE__, __LINE__, WTF_PRETTY_FUNCTION); \
375 CRASH(); \
376} while (0)
377
378#define ASSERT_IMPLIES(condition, assertion) do { \
379 if ((condition) && !(assertion)) { \
380 WTFReportAssertionFailure(__FILE__, __LINE__, WTF_PRETTY_FUNCTION, #condition " => " #assertion); \
381 CRASH(); \
382 } \
383} while (0)
384
385#define ASSERT_UNUSED(variable, assertion, ...) ASSERT(assertion, __VA_ARGS__)

387#define NO_RETURN_DUE_TO_ASSERT NO_RETURN_DUE_TO_CRASH

389/* ASSERT_WITH_SECURITY_IMPLICATION

391 Failure of this assertion indicates a possible security vulnerability.
392 Class of vulnerabilities that it tests include bad casts, out of bounds
393 accesses, use-after-frees, etc. Please file a bug using the security
394 template - https://bugs.webkit.org/enter_bug.cgi?product=Security.
395
396*/

398 (!(assertion) ? \

401 (void)0)
