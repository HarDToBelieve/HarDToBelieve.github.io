31#ifdef __OBJC__
32#error Please use @autoreleasepool instead of AutodrainedPool.
33#endif
34
35#include <wtf/Noncopyable.h>
36
37namespace WTF {
38
39// This class allows non-Objective-C C++ code to create an autorelease pool.
40// It cannot be used in Objective-C++ code, won't be compiled; instead @autoreleasepool should be used.
41// It can be used in cross-platform code; will compile down to nothing for non-Cocoa platforms.
42
43class AutodrainedPool {
44 WTF_MAKE_FAST_ALLOCATED;
45 WTF_MAKE_NONCOPYABLE(AutodrainedPool);
46
47public:
48#if USE(FOUNDATION)
49 WTF_EXPORT_PRIVATE AutodrainedPool();
50 WTF_EXPORT_PRIVATE ~AutodrainedPool();
51#else
52 AutodrainedPool() { }
53 ~AutodrainedPool() { }
54#endif
55
56private:
57#if USE(FOUNDATION)
58 void* m_autoreleasePool;
59#endif
60};
61
62} // namespace WTF
63
64using WTF::AutodrainedPool;
