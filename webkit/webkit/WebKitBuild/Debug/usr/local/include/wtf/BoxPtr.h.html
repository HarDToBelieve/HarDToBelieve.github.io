27#pragma once
28
29#include <memory>
30#include <wtf/Box.h>
31
32namespace WTF {
33
34template<typename T> struct BoxPtrDeleter {
35 void operator()(T*) const = delete;
36};
37
38#define WTF_DEFINE_BOXPTR_DELETER(typeName, deleterFunction) \
39 template<> struct BoxPtrDeleter<typeName> \
40 { \
41 void operator() (typeName* ptr) const \
42 { \
43 deleterFunction(ptr); \
44 } \
45 };
46
47template<typename T> using BoxPtr = Box<std::unique_ptr<T, BoxPtrDeleter<T>>>;
48
49template<typename T> BoxPtr<T> adoptInBoxPtr(T* ptr)
50{
51 return BoxPtr<T>::create(ptr);
52}
53
54template<typename T> bool operator==(const BoxPtr<T>& lhs, const BoxPtr<T>& rhs)
55{
56 if (!lhs && !rhs)
57 return true;
58
59 if (!lhs || !rhs)
60 return false;
61
62 if (!lhs->get() && !rhs->get())
63 return true;
64
65 if (!lhs->get() || !rhs->get())
66 return false;
67
68 return *lhs == *rhs;
69}
70
71template<typename T> bool operator!=(const BoxPtr<T>& lhs, const BoxPtr<T>& rhs)
72{
73 return !(lhs == rhs);
74}
75
76} // namespace WTF
