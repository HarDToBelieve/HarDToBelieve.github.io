35
36 MonotonicTime cpuTime;

38 Seconds systemTime;
39
40 WTF_EXPORT_PRIVATE double percentageCPUUsageSince(const CPUTime&) const;
41
42 WTF_EXPORT_PRIVATE static std::optional<CPUTime> get();
43
44 // Returns the current CPU time of the current thread.
45 // Precision varies depending on platform but is usually as good or better
46 // than a millisecond.

48};

50} // namespace WTF
