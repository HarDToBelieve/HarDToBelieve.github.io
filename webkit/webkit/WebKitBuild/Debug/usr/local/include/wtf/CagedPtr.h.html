27
28#include <wtf/Gigacage.h>
29#include <wtf/MathExtras.h>
30#include <wtf/PtrTag.h>
31#include <wtf/RawPtrTraits.h>
32
33#include <climits>
34
35#if OS(DARWIN)
36#include <mach/vm_param.h>
37#endif
38
39namespace WTF {
40
41constexpr bool tagCagedPtr = true;
42
43template<Gigacage::Kind passedKind, typename T, bool shouldTag = false, typename PtrTraits = RawPtrTraits<T>>
44class CagedPtr {
45public:
46 static constexpr Gigacage::Kind kind = passedKind;
47 static constexpr unsigned numberOfPointerBits = sizeof(T*) * CHAR_BIT;
48 static constexpr unsigned maxNumberOfAllowedPACBits = numberOfPointerBits - OS_CONSTANT(EFFECTIVE_ADDRESS_WIDTH);
49 static constexpr uintptr_t nonPACBitsMask = (1ull << (numberOfPointerBits - maxNumberOfAllowedPACBits)) - 1;
50
51 CagedPtr() : CagedPtr(nullptr) { }
52 CagedPtr(std::nullptr_t)
53 : m_ptr(shouldTag ? tagArrayPtr<T>(nullptr, 0) : nullptr)
54 { }
55
56 CagedPtr(T* ptr, size_t size)
57 : m_ptr(shouldTag ? tagArrayPtr(ptr, size) : ptr)
58 { }
59
60 T* get(size_t size) const
61 {
62 ASSERT(m_ptr);
63 T* ptr = PtrTraits::unwrap(m_ptr);
64 T* cagedPtr = Gigacage::caged(kind, ptr);
65 T* untaggedPtr = shouldTag ? untagArrayPtr(mergePointers(ptr, cagedPtr), size) : cagedPtr;
66 return untaggedPtr;
67 }
68
69 T* getMayBeNull(size_t size) const
70 {
71 T* ptr = PtrTraits::unwrap(m_ptr);
72 if (!removeArrayPtrTag(ptr))
73 return nullptr;
74 T* cagedPtr = Gigacage::caged(kind, ptr);
75 T* untaggedPtr = shouldTag ? untagArrayPtr(mergePointers(ptr, cagedPtr), size) : cagedPtr;
76 return untaggedPtr;
77 }
78
79 T* getUnsafe() const
80 {
81 T* ptr = PtrTraits::unwrap(m_ptr);
82 ptr = shouldTag ? removeArrayPtrTag(ptr) : ptr;
83 return Gigacage::cagedMayBeNull(kind, ptr);
84 }
85
86 // We need the template here so that the type of U is deduced at usage time rather than class time. U should always be T.
87 template<typename U = T>
88 typename std::enable_if<!std::is_same<void, U>::value, T>::type&
89 /* T& */ at(size_t index, size_t size) const { return get(size)[index]; }
90
91 void recage(size_t oldSize, size_t newSize)
92 {
93 auto ptr = get(oldSize);
94 ASSERT(ptr == getUnsafe());
95 *this = CagedPtr(ptr, newSize);
96 }
97
98 CagedPtr(CagedPtr& other)
99 : m_ptr(other.m_ptr)
100 {
101 }
102
103 CagedPtr& operator=(const CagedPtr& ptr)
104 {
105 m_ptr = ptr.m_ptr;
106 return *this;
107 }
108
109 CagedPtr(CagedPtr&& other)
110 : m_ptr(PtrTraits::exchange(other.m_ptr, nullptr))
111 {
112 }
113
114 CagedPtr& operator=(CagedPtr&& ptr)
115 {
116 m_ptr = PtrTraits::exchange(ptr.m_ptr, nullptr);
117 return *this;
118 }
119
120 bool operator==(const CagedPtr& other) const
121 {
122 bool result = m_ptr == other.m_ptr;
123 ASSERT(result == (getUnsafe() == other.getUnsafe()));
124 return result;
125 }
126
127 bool operator!=(const CagedPtr& other) const
128 {
129 return !(*this == other);
130 }
131
132 explicit operator bool() const
133 {
134 return getUnsafe() != nullptr;
135 }
136
137 T* rawBits() const
138 {
139 return bitwise_cast<T*>(m_ptr);
140 }
141
142protected:
143 static inline T* mergePointers(T* sourcePtr, T* cagedPtr)
144 {
145#if CPU(ARM64E)
146 return reinterpret_cast<T*>((reinterpret_cast<uintptr_t>(sourcePtr) & ~nonPACBitsMask) | (reinterpret_cast<uintptr_t>(cagedPtr) & nonPACBitsMask));
147#else
148 UNUSED_PARAM(sourcePtr);
149 return cagedPtr;
150#endif
151 }
152
153 typename PtrTraits::StorageType m_ptr;
154};
155
156} // namespace WTF
