41/* Checked<T>
42 *
43 * This class provides a mechanism to perform overflow-safe integer arithmetic
44 * without having to manually ensure that you have all the required bounds checks
45 * directly in your code.
46 *
47 * There are two modes of operation:
48 * - The default is Checked<T, CrashOnOverflow>, and crashes at the point
49 * and overflow has occurred.
50 * - The alternative is Checked<T, RecordOverflow>, which uses an additional
51 * byte of storage to track whether an overflow has occurred, subsequent
52 * unchecked operations will crash if an overflow has occured
53 *
54 * It is possible to provide a custom overflow handler, in which case you need
55 * to support these functions:
56 * - void overflowed();
57 * This function is called when an operation has produced an overflow.
58 * - bool hasOverflowed();
59 * This function must return true if overflowed() has been called on an
60 * instance and false if it has not.
61 * - void clearOverflow();
62 * Used to reset overflow tracking when a value is being overwritten with
63 * a new value.
64 *
65 * Checked<T> works for all integer types, with the following caveats:
66 * - Mixing signedness of operands is only supported for types narrower than
67 * 64bits.
68 * - It does have a performance impact, so tight loops may want to be careful
69 * when using it.
70 *
71 */

80class AssertNoOverflow {
81public:

83 {
84 ASSERT_NOT_REACHED();

87 void clearOverflow() { }

98class CrashOnOverflow {

102 crash();

104
105 void clearOverflow() { }
106
107 static NO_RETURN_DUE_TO_CRASH void crash()

109 CRASH();

116class RecordOverflow {
117protected:
118 RecordOverflow()
119 : m_overflowed(false)
120 {
121 }
122
123 void clearOverflow()
124 {
125 m_overflowed = false;
126 }
127
128 static NO_RETURN_DUE_TO_CRASH void crash()
129 {
130 CRASH();
131 }
132
133public:
134 bool hasOverflowed() const { return m_overflowed; }
135 void overflowed() { m_overflowed = true; }
136
137private:
138 unsigned char m_overflowed;
139};

282template <typename LHS, typename RHS, typename ResultType> struct ArithmeticOperations<LHS, RHS, ResultType, true, true> {
283 // LHS and RHS are signed types
284
285 // Helper function
286 static inline bool signsMatch(LHS lhs, RHS rhs)
287 {
288 return (lhs ^ rhs) >= 0;
289 }
290
291 static inline bool add(LHS lhs, RHS rhs, ResultType& result) WARN_UNUSED_RETURN
292 {
293#if COMPILER(GCC_COMPATIBLE)
294 ResultType temp;
295 if (__builtin_add_overflow(lhs, rhs, &temp))
296 return false;
297 result = temp;
298 return true;
299#else
300 if (signsMatch(lhs, rhs)) {
301 if (lhs >= 0) {
302 if ((std::numeric_limits<ResultType>::max() - rhs) < lhs)
303 return false;
304 } else {
305 ResultType temp = lhs - std::numeric_limits<ResultType>::min();
306 if (rhs < -temp)
307 return false;
308 }
309 } // if the signs do not match this operation can't overflow
310 result = lhs + rhs;
311 return true;
312#endif
313 }
314
315 static inline bool sub(LHS lhs, RHS rhs, ResultType& result) WARN_UNUSED_RETURN
316 {
317#if COMPILER(GCC_COMPATIBLE)
318 ResultType temp;
319 if (__builtin_sub_overflow(lhs, rhs, &temp))
320 return false;
321 result = temp;
322 return true;
323#else
324 if (!signsMatch(lhs, rhs)) {
325 if (lhs >= 0) {
326 if (lhs > std::numeric_limits<ResultType>::max() + rhs)
327 return false;
328 } else {
329 if (lhs < std::numeric_limits<ResultType>::min() + rhs)
330 return false;
331 }
332 } // if the signs match this operation can't overflow
333 result = lhs - rhs;
334 return true;
335#endif
336 }
337
338 static inline bool multiply(LHS lhs, RHS rhs, ResultType& result) WARN_UNUSED_RETURN
339 {
340#if USE(MUL_OVERFLOW)
341 ResultType temp;
342 if (__builtin_mul_overflow(lhs, rhs, &temp))
343 return false;
344 result = temp;
345 return true;
346#else
347 if (signsMatch(lhs, rhs)) {
348 if (lhs >= 0) {
349 if (lhs && (std::numeric_limits<ResultType>::max() / lhs) < rhs)
350 return false;
351 } else {
352 if (static_cast<ResultType>(lhs) == std::numeric_limits<ResultType>::min() || static_cast<ResultType>(rhs) == std::numeric_limits<ResultType>::min())
353 return false;
354 if ((std::numeric_limits<ResultType>::max() / -lhs) < -rhs)
355 return false;
356 }
357 } else {
358 if (lhs < 0) {
359 if (rhs && lhs < (std::numeric_limits<ResultType>::min() / rhs))
360 return false;
361 } else {
362 if (lhs && rhs < (std::numeric_limits<ResultType>::min() / lhs))
363 return false;
364 }
365 }
366 result = lhs * rhs;
367 return true;
368#endif
369 }
370
371 static inline bool divide(LHS lhs, RHS rhs, ResultType& result) WARN_UNUSED_RETURN
372 {
373 if (!rhs)
374 return false;
375
376 result = lhs / rhs;
377 return true;
378 }
379
380 static inline bool equals(LHS lhs, RHS rhs) { return lhs == rhs; }
381
382};

384template <typename LHS, typename RHS, typename ResultType> struct ArithmeticOperations<LHS, RHS, ResultType, false, false> {
385 // LHS and RHS are unsigned types so bounds checks are nice and easy
386 static inline bool add(LHS lhs, RHS rhs, ResultType& result) WARN_UNUSED_RETURN
387 {
388#if COMPILER(GCC_COMPATIBLE)
389 ResultType temp;
390 if (__builtin_add_overflow(lhs, rhs, &temp))
391 return false;
392 result = temp;
393 return true;
394#else
395 ResultType temp = lhs + rhs;
396 if (temp < lhs)
397 return false;
398 result = temp;
399 return true;
400#endif
401 }
402
403 static inline bool sub(LHS lhs, RHS rhs, ResultType& result) WARN_UNUSED_RETURN
404 {
405#if COMPILER(GCC_COMPATIBLE)
406 ResultType temp;
407 if (__builtin_sub_overflow(lhs, rhs, &temp))
408 return false;
409 result = temp;
410 return true;
411#else
412 ResultType temp = lhs - rhs;
413 if (temp > lhs)
414 return false;
415 result = temp;
416 return true;
417#endif
418 }
419
420 static inline bool multiply(LHS lhs, RHS rhs, ResultType& result) WARN_UNUSED_RETURN
421 {
422#if USE(MUL_OVERFLOW)
423 ResultType temp;
424 if (__builtin_mul_overflow(lhs, rhs, &temp))
425 return false;
426 result = temp;
427 return true;
428#else
429 if (!lhs || !rhs) {
430 result = 0;
431 return true;
432 }
433 if (std::numeric_limits<ResultType>::max() / lhs < rhs)
434 return false;
435 result = lhs * rhs;
436 return true;
437#endif
438 }
439
440 static inline bool divide(LHS lhs, RHS rhs, ResultType& result) WARN_UNUSED_RETURN
441 {
442 if (!rhs)
443 return false;
444
445 result = lhs / rhs;
446 return true;
447 }
448
449 static inline bool equals(LHS lhs, RHS rhs) { return lhs == rhs; }

454 static inline bool add(int64_t lhs, int64_t rhs, ResultType& result)

479 result = temp;
480 return true;
481#else
482 int64_t temp = lhs - rhs;
483 if (temp < std::numeric_limits<ResultType>::min())
484 return false;
485 if (temp > std::numeric_limits<ResultType>::max())
486 return false;
487 result = static_cast<ResultType>(temp);
488 return true;
489#endif
490 }

492 static inline bool multiply(int64_t lhs, int64_t rhs, ResultType& result)

624template <typename T, class OverflowHandler> class Checked : public OverflowHandler {
625public:
626 template <typename _T, class _OverflowHandler> friend class Checked;
627 Checked()
628 : m_value(0)
629 {
630 }
631
632 Checked(ResultOverflowedTag)
633 : m_value(0)
634 {
635 this->overflowed();
636 }
637
638 Checked(const Checked& value)
639 {
640 if (value.hasOverflowed())
641 this->overflowed();
642 m_value = static_cast<T>(value.m_value);
643 }
644
645 template <typename U> Checked(U value)
646 {
647 if (!isInBounds<T>(value))
648 this->overflowed();
649 m_value = static_cast<T>(value);
650 }
651
652 template <typename V> Checked(const Checked<T, V>& rhs)
653 : m_value(rhs.m_value)
654 {
655 if (rhs.hasOverflowed())
656 this->overflowed();
657 }
658
659 template <typename U> Checked(const Checked<U, OverflowHandler>& rhs)
660 : OverflowHandler(rhs)
661 {
662 if (!isInBounds<T>(rhs.m_value))
663 this->overflowed();
664 m_value = static_cast<T>(rhs.m_value);
665 }
666
667 template <typename U, typename V> Checked(const Checked<U, V>& rhs)
668 {
669 if (rhs.hasOverflowed())
670 this->overflowed();
671 if (!isInBounds<T>(rhs.m_value))
672 this->overflowed();
673 m_value = static_cast<T>(rhs.m_value);
674 }
675
676 Checked& operator=(Checked rhs)
677 {
678 this->clearOverflow();
679 if (rhs.hasOverflowed())
680 this->overflowed();
681 m_value = static_cast<T>(rhs.m_value);
682 return *this;
683 }
684
685 template <typename U> Checked& operator=(U value)
686 {
687 return *this = Checked(value);
688 }
689
690 template <typename U, typename V> Checked& operator=(const Checked<U, V>& rhs)
691 {
692 return *this = Checked(rhs);
693 }
694
695 // prefix
696 Checked& operator++()
697 {
698 if (m_value == std::numeric_limits<T>::max())
699 this->overflowed();
700 m_value++;
701 return *this;
702 }
703
704 Checked& operator--()
705 {
706 if (m_value == std::numeric_limits<T>::min())
707 this->overflowed();
708 m_value--;
709 return *this;
710 }
711
712 // postfix operators
713 Checked operator++(int)
714 {
715 if (m_value == std::numeric_limits<T>::max())
716 this->overflowed();
717 return Checked(m_value++);
718 }
719
720 Checked operator--(int)
721 {
722 if (m_value == std::numeric_limits<T>::min())
723 this->overflowed();
724 return Checked(m_value--);
725 }
726
727 // Boolean operators
728 bool operator!() const
729 {
730 if (UNLIKELY(this->hasOverflowed()))
731 this->crash();
732 return !m_value;
733 }
734
735 explicit operator bool() const
736 {
737 if (UNLIKELY(this->hasOverflowed()))
738 this->crash();
739 return m_value;
740 }
741
742 operator T() const
743 {
744 if (UNLIKELY(this->hasOverflowed()))
745 this->crash();
746 return m_value;
747 }
748
749 // Value accessors. value() will crash if there's been an overflow.
750 template<typename U = T>
751 U value() const
752 {
753 if (UNLIKELY(this->hasOverflowed()))
754 this->crash();
755 return static_cast<U>(m_value);
756 }

758 // Mutating assignment
759 template <typename U> Checked& operator+=(U rhs)
760 {
761 if (!safeAdd<OverflowHandler>(m_value, rhs, m_value))
762 this->overflowed();
763 return *this;
764 }
765
766 template <typename U> Checked& operator-=(U rhs)

768 if (!safeSub<OverflowHandler>(m_value, rhs, m_value))
769 this->overflowed();
770 return *this;
771 }
772
773 template <typename U> Checked& operator*=(U rhs)
774 {
775 if (!safeMultiply<OverflowHandler>(m_value, rhs, m_value))

777 return *this;

779
780 template <typename U> Checked& operator/=(U rhs)
781 {
