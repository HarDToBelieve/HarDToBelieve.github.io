44/* COMPILER_HAS_CLANG_FEATURE() - whether the compiler supports a particular language or library feature. */

175/* ALWAYS_INLINE */
176
177/* In GCC functions marked with no_sanitize_address cannot call functions that are marked with always_inline and not marked with no_sanitize_address.
178 * Therefore we need to give up on the enforcement of ALWAYS_INLINE when bulding with ASAN. https://gcc.gnu.org/bugzilla/show_bug.cgi?id=67368 */
179#if !defined(ALWAYS_INLINE) && COMPILER(GCC_COMPATIBLE) && defined(NDEBUG) && !COMPILER(MINGW) && !(COMPILER(GCC) && ASAN_ENABLED)
180#define ALWAYS_INLINE inline __attribute__((__always_inline__))
181#endif
182

184#define ALWAYS_INLINE __forceinline
185#endif
186

188#define ALWAYS_INLINE inline
189#endif

436/* Can't use WTF_CONCAT() and STRINGIZE() because they are defined in
437 * StdLibExtras.h, which includes this file. */
438#define _COMPILER_CONCAT_I(a, b) a ## b
439#define _COMPILER_CONCAT(a, b) _COMPILER_CONCAT_I(a, b)

441#define _COMPILER_STRINGIZE(exp) #exp
442
443#define _COMPILER_WARNING_NAME(warning) "-W" warning
444
445#if COMPILER(GCC) || COMPILER(CLANG)
446#define IGNORE_WARNINGS_BEGIN_COND(cond, compiler, warning) \
447 _Pragma(_COMPILER_STRINGIZE(compiler diagnostic push)) \
448 _COMPILER_CONCAT(IGNORE_WARNINGS_BEGIN_IMPL_, cond)(compiler, warning)
449
450#define IGNORE_WARNINGS_BEGIN_IMPL_1(compiler, warning) \
451 _Pragma(_COMPILER_STRINGIZE(compiler diagnostic ignored warning))
452#define IGNORE_WARNINGS_BEGIN_IMPL_0(compiler, warning)
453#define IGNORE_WARNINGS_BEGIN_IMPL_(compiler, warning)
454
455
456#define IGNORE_WARNINGS_END_IMPL(compiler) _Pragma(_COMPILER_STRINGIZE(compiler diagnostic pop))
457
458#if defined(__has_warning)
459#define _IGNORE_WARNINGS_BEGIN_IMPL(compiler, warning) \
460 IGNORE_WARNINGS_BEGIN_COND(__has_warning(warning), compiler, warning)
461#else
462#define _IGNORE_WARNINGS_BEGIN_IMPL(compiler, warning) IGNORE_WARNINGS_BEGIN_COND(1, compiler, warning)
463#endif
464
465#define IGNORE_WARNINGS_BEGIN_IMPL(compiler, warning) \
466 _IGNORE_WARNINGS_BEGIN_IMPL(compiler, _COMPILER_WARNING_NAME(warning))

468#endif // COMPILER(GCC) || COMPILER(CLANG)
