35namespace WTF {
36
37// This is a concurrent hash-based set for pointers. It's optimized for:
38//
39// - High rate of contains() calls.
40// - High rate of add() calls that don't add anything new. add() calls that don't add anything (nop adds)
41// don't mutate the table at all.
42// - Not too many threads. I doubt this scales beyond ~4. Though, it may actually scale better than that
43// if the rate of nop adds is absurdly high.
44//
45// If we wanted this to scale better, the main change we'd have to make is how this table determines when
46// to resize. Right now it's a shared counter. We lock;xadd this counter. One easy way to make that
47// scalable is to require each thread that works with the ConcurrentPtrHashSet to register itself first.
48// Then each thread would have some data structure that has a counter. We could institute the policy that
49// each thread simply increments its own load counter, in its own data structure. Then, if any search to
50// resolve a collision does more than N iterations, we can compute a combined load by summing the load
51// counters of all of the thread data structures.
52//
53// ConcurrentPtrHashSet's main user, the GC, sees a 98% nop add rate in Speedometer. That's why this
54// focuses so much on cases where the table does not change.
55class ConcurrentPtrHashSet final {
56 WTF_MAKE_NONCOPYABLE(ConcurrentPtrHashSet);
57 WTF_MAKE_FAST_ALLOCATED;
58
59public:
60 WTF_EXPORT_PRIVATE ConcurrentPtrHashSet();
61 WTF_EXPORT_PRIVATE ~ConcurrentPtrHashSet();
62
63 template<typename T>
64 bool contains(T value) const
65 {
66 return containsImpl(cast(value));
67 }
68
69 template<typename T>
70 bool add(T value)
71 {
72 return addImpl(cast(value));
73 }
74
75 size_t size() const
76 {
77 Table* table = m_table.loadRelaxed();
78 if (table == &m_stubTable)
79 return sizeSlow();
80 return table->load.loadRelaxed();
81 }
82
83 // Only call when you know that no other thread can call add(). This frees up memory without changing
84 // the contents of the table.
85 WTF_EXPORT_PRIVATE void deleteOldTables();
86
87 // Only call when you know that no other thread can call add(). This frees up all memory except for
88 // the smallest possible hashtable.
89 WTF_EXPORT_PRIVATE void clear();
90
91private:
92 struct Table {
93 WTF_MAKE_STRUCT_FAST_ALLOCATED;
94
95 static std::unique_ptr<Table> create(unsigned size);
96 void initializeStub();
97
98 unsigned maxLoad() const { return size / 2; }
99
100 // This can be any value >= 1 because the stub's size is 0, ensuring that
101 // m_stubTable is always seen as "full". We choose 10 for no reason other
102 // than it gives some warm fuzzies since it is greater than 1.
103 static constexpr unsigned stubDefaultLoadValue = 10;
104
105 unsigned size; // This is immutable.
106 unsigned mask; // This is immutable.
107 Atomic<unsigned> load;
108 Atomic<void*> array[1];
109 };
110
111 static unsigned hash(void* ptr)
112 {
113 return PtrHash<void*>::hash(ptr);
114 }
115
116 void initialize();
117
118 template<typename T>
119 static void* cast(T value)
120 {
121 static_assert(sizeof(T) <= sizeof(void*), "type too big");
122 union {
123 void* ptr;
124 T value;
125 } u;
126 u.ptr = nullptr;
127 u.value = value;
128 return u.ptr;
129 }
130
131 bool containsImpl(void* ptr) const
132 {
133 Table* table = m_table.loadRelaxed();
134 if (table == &m_stubTable)
135 return containsImplSlow(ptr);
136
137 unsigned mask = table->mask;
138 unsigned startIndex = hash(ptr) & mask;
139 unsigned index = startIndex;
140 for (;;) {
141 void* entry = table->array[index].loadRelaxed();
142 if (!entry)
143 return false;
144 if (entry == ptr)
145 return true;
146 index = (index + 1) & mask;
147 RELEASE_ASSERT(index != startIndex);
148 }
149 }
150
151 // Returns true if a new entry was added.
152 bool addImpl(void* ptr)
153 {
154 Table* table = m_table.loadRelaxed();
155 unsigned mask = table->mask;
156 unsigned startIndex = hash(ptr) & mask;
157 unsigned index = startIndex;
158 for (;;) {
159 void* entry = table->array[index].loadRelaxed();
160 if (!entry)
161 return addSlow(table, mask, startIndex, index, ptr);
162 if (entry == ptr)
163 return false;
164 index = (index + 1) & mask;
165 RELEASE_ASSERT(index != startIndex);
166 }
167 }
168
169 WTF_EXPORT_PRIVATE bool addSlow(Table* table, unsigned mask, unsigned startIndex, unsigned index, void* ptr);
170 WTF_EXPORT_PRIVATE bool containsImplSlow(void* ptr) const;
171 WTF_EXPORT_PRIVATE size_t sizeSlow() const;
172
173 void resizeIfNecessary();
174 bool resizeAndAdd(void* ptr);
175
176 Vector<std::unique_ptr<Table>, 4> m_allTables;
177 Atomic<Table*> m_table; // This is never null.
178 Table m_stubTable;
179 mutable Lock m_lock; // We just use this to control resize races.
180};
181
182} // namespace WTF
