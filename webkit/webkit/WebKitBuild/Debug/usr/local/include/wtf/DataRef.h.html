21#pragma once
22
23#include <wtf/Ref.h>
24
25namespace WTF {
26
27template <typename T>
28class DataRef {
29public:
30 DataRef(Ref<T>&& data)
31 : m_data(WTFMove(data))
32 {
33 }
34
35 DataRef(const DataRef& other)
36 : m_data(other.m_data.copyRef())
37 {
38 }
39
40 DataRef& operator=(const DataRef& other)
41 {
42 m_data = other.m_data.copyRef();
43 return *this;
44 }
45
46 DataRef(DataRef&&) = default;
47 DataRef& operator=(DataRef&&) = default;
48
49 DataRef replace(DataRef&& other)
50 {
51 return m_data.replace(WTFMove(other.m_data));
52 }
53
54 operator const T&() const
55 {
56 return m_data;
57 }
58
59 const T* ptr() const
60 {
61 return m_data.ptr();
62 }
63
64 const T& get() const
65 {
66 return m_data;
67 }
68
69 const T& operator*() const
70 {
71 return m_data;
72 }
73
74 const T* operator->() const
75 {
76 return m_data.ptr();
77 }
78
79 T& access()
80 {
81 if (!m_data->hasOneRef())
82 m_data = m_data->copy();
83 return m_data;
84 }
85
86 bool operator==(const DataRef& other) const
87 {
88 return m_data.ptr() == other.m_data.ptr() || m_data.get() == other.m_data.get();
89 }
90
91 bool operator!=(const DataRef& other) const
92 {
93 return !(*this == other);
94 }
95
96private:
97 Ref<T> m_data;
98};
99
100} // namespace WTF
101
102using WTF::DataRef;
