33#define SLEEP_THREAD_FOR_DEBUGGER() \
34do { \
35 WTFReportError(__FILE__, __LINE__, WTF_PRETTY_FUNCTION, "Sleeping thread for debugger; attach to process (PID: %d) to unsleep the thread.", getCurrentProcessID()); \
36 do { \
37 sleep(1); \
38 if (WTFIsDebuggerAttached()) \
39 break; \
40 } while (1); \
41 WTFBreakpointTrap(); \
42} while (0)

44namespace WTF {
45
46template<typename StringType>
47const char* debugString(StringType string)
48{
49 return debugString(string, "");
50}
51
52template<typename... StringTypes>
53const char* debugString(StringTypes... strings)
54{
55 String result = tryMakeString(strings...);
56 if (!result)
57 CRASH();
58
59 auto cString = result.utf8();
60 const char* cStringData = cString.data();
61
62 callOnMainThread([cString = WTFMove(cString)] {
63 });
64
65 return cStringData;
66}

68} // namespace WTF
