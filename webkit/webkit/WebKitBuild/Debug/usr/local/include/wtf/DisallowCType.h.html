31// The behavior of many of the functions in the <ctype.h> header is dependent
32// on the current locale. But almost all uses of these functions are for
33// locale-independent, ASCII-specific purposes. In WebKit code we use our own
34// ASCII-specific functions instead. This header makes sure we get a compile-time
35// error if we use one of the <ctype.h> functions by accident.
36
37// this breaks compilation of <QFontDatabase>, at least, so turn it off for now
38// Also generates errors on wx on Windows, presumably because these functions
39// are used from wx headers. On GTK+ for Mac many GTK+ files include <libintl.h>
40// or <glib/gi18n-lib.h>, which in turn include <xlocale/_ctype.h> which uses
41// isacii().
42#if !(OS(DARWIN) && PLATFORM(GTK)) && !defined(_LIBCPP_VERSION) && defined(__GLIBC__)
43
