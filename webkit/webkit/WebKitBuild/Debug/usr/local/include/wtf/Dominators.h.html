35// This is a utility for finding the dominators of a graph. Dominators are almost universally used
36// for control flow graph analysis, so this code will refer to the graph's "nodes" as "blocks". In
37// that regard this code is kind of specialized for the various JSC compilers, but you could use it
38// for non-compiler things if you are OK with referring to your "nodes" as "blocks".
39
40template<typename Graph>
41class Dominators {
42 WTF_MAKE_FAST_ALLOCATED;
43public:
44 using List = typename Graph::List;
45
46 Dominators(Graph& graph, bool selfCheck = false)
47 : m_graph(graph)
48 , m_data(graph.template newMap<BlockData>())
49 {
50 LengauerTarjan lengauerTarjan(m_graph);
51 lengauerTarjan.compute();
52
53 // From here we want to build a spanning tree with both upward and downward links and we want
54 // to do a search over this tree to compute pre and post numbers that can be used for dominance
55 // tests.
56
57 for (unsigned blockIndex = m_graph.numNodes(); blockIndex--;) {
58 typename Graph::Node block = m_graph.node(blockIndex);
59 if (!block)
60 continue;
61
62 typename Graph::Node idomBlock = lengauerTarjan.immediateDominator(block);
63 m_data[block].idomParent = idomBlock;
64 if (idomBlock)
65 m_data[idomBlock].idomKids.append(block);
66 }
67
68 unsigned nextPreNumber = 0;
69 unsigned nextPostNumber = 0;
70
71 // Plain stack-based worklist because we are guaranteed to see each block exactly once anyway.
72 Vector<GraphNodeWithOrder<typename Graph::Node>> worklist;
73 worklist.append(GraphNodeWithOrder<typename Graph::Node>(m_graph.root(), GraphVisitOrder::Pre));
74 while (!worklist.isEmpty()) {
75 GraphNodeWithOrder<typename Graph::Node> item = worklist.takeLast();
76 switch (item.order) {
77 case GraphVisitOrder::Pre:
78 m_data[item.node].preNumber = nextPreNumber++;
79 worklist.append(GraphNodeWithOrder<typename Graph::Node>(item.node, GraphVisitOrder::Post));
80 for (typename Graph::Node kid : m_data[item.node].idomKids)
81 worklist.append(GraphNodeWithOrder<typename Graph::Node>(kid, GraphVisitOrder::Pre));
82 break;
83 case GraphVisitOrder::Post:
84 m_data[item.node].postNumber = nextPostNumber++;
85 break;
86 }
87 }
88
89 if (selfCheck) {
90 // Check our dominator calculation:
91 // 1) Check that our range-based ancestry test is the same as a naive ancestry test.
92 // 2) Check that our notion of who dominates whom is identical to a naive (not
93 // Lengauer-Tarjan) dominator calculation.
94
95 ValidationContext context(m_graph, *this);
96
97 for (unsigned fromBlockIndex = m_graph.numNodes(); fromBlockIndex--;) {
98 typename Graph::Node fromBlock = m_graph.node(fromBlockIndex);
99 if (!fromBlock || m_data[fromBlock].preNumber == UINT_MAX)
100 continue;
101 for (unsigned toBlockIndex = m_graph.numNodes(); toBlockIndex--;) {
102 typename Graph::Node toBlock = m_graph.node(toBlockIndex);
103 if (!toBlock || m_data[toBlock].preNumber == UINT_MAX)
104 continue;
105
106 if (dominates(fromBlock, toBlock) != naiveDominates(fromBlock, toBlock))
107 context.reportError(fromBlock, toBlock, "Range-based domination check is broken");
108 if (dominates(fromBlock, toBlock) != context.naiveDominators.dominates(fromBlock, toBlock))
109 context.reportError(fromBlock, toBlock, "Lengauer-Tarjan domination is broken");
110 }
111 }
112
113 context.handleErrors();
114 }
115 }

117 bool strictlyDominates(typename Graph::Node from, typename Graph::Node to) const
118 {
119 return m_data[to].preNumber > m_data[from].preNumber
120 && m_data[to].postNumber < m_data[from].postNumber;
121 }
122
123 bool dominates(typename Graph::Node from, typename Graph::Node to) const
124 {
125 return from == to || strictlyDominates(from, to);
126 }

296private:
297 // This implements Lengauer and Tarjan's "A Fast Algorithm for Finding Dominators in a Flowgraph"
298 // (TOPLAS 1979). It uses the "simple" implementation of LINK and EVAL, which yields an O(n log n)
299 // solution. The full paper is linked below; this code attempts to closely follow the algorithm as
300 // it is presented in the paper; in particular sections 3 and 4 as well as appendix B.
301 // https://www.cs.princeton.edu/courses/archive/fall03/cs528/handouts/a%20fast%20algorithm%20for%20finding.pdf
302 //
303 // This code is very subtle. The Lengauer-Tarjan algorithm is incredibly deep to begin with. The
304 // goal of this code is to follow the code in the paper, however our implementation must deviate
305 // from the paper when it comes to recursion. The authors had used recursion to implement DFS, and
306 // also to implement the "simple" EVAL. We convert both of those into worklist-based solutions.
307 // Finally, once the algorithm gives us immediate dominators, we implement dominance tests by
308 // walking the dominator tree and computing pre and post numbers. We then use the range inclusion
309 // check trick that was first discovered by Paul F. Dietz in 1982 in "Maintaining order in a linked
310 // list" (see http://dl.acm.org/citation.cfm?id=802184).
311
312 class LengauerTarjan {
313 WTF_MAKE_FAST_ALLOCATED;
314 public:
315 LengauerTarjan(Graph& graph)
316 : m_graph(graph)
317 , m_data(graph.template newMap<BlockData>())
318 {
319 for (unsigned blockIndex = m_graph.numNodes(); blockIndex--;) {
320 typename Graph::Node block = m_graph.node(blockIndex);
321 if (!block)
322 continue;
323 m_data[block].label = block;
324 }
325 }
326
327 void compute()
328 {
329 computeDepthFirstPreNumbering(); // Step 1.
330 computeSemiDominatorsAndImplicitImmediateDominators(); // Steps 2 and 3.
331 computeExplicitImmediateDominators(); // Step 4.
332 }
333
334 typename Graph::Node immediateDominator(typename Graph::Node block)
335 {
336 return m_data[block].dom;
337 }
338
339 private:
340 void computeDepthFirstPreNumbering()
341 {
342 // Use a block worklist that also tracks the index inside the successor list. This is
343 // necessary for ensuring that we don't attempt to visit a successor until the previous
344 // successors that we had visited are fully processed. This ends up being revealed in the
345 // output of this method because the first time we see an edge to a block, we set the
346 // block's parent. So, if we have:
347 //
348 // A -> B
349 // A -> C
350 // B -> C
351 //
352 // And we're processing A, then we want to ensure that if we see A->B first (and hence set
353 // B's prenumber before we set C's) then we also end up setting C's parent to B by virtue
354 // of not noticing A->C until we're done processing B.
355
356 ExtendedGraphNodeWorklist<typename Graph::Node, unsigned, typename Graph::Set> worklist;
357 worklist.push(m_graph.root(), 0);
358
359 while (GraphNodeWith<typename Graph::Node, unsigned> item = worklist.pop()) {
360 typename Graph::Node block = item.node;
361 unsigned successorIndex = item.data;
362
363 // We initially push with successorIndex = 0 regardless of whether or not we have any
364 // successors. This is so that we can assign our prenumber. Subsequently we get pushed
365 // with higher successorIndex values, but only if they are in range.
366 ASSERT(!successorIndex || successorIndex < m_graph.successors(block).size());
367
368 if (!successorIndex) {
369 m_data[block].semiNumber = m_blockByPreNumber.size();
370 m_blockByPreNumber.append(block);
371 }
372
373 if (successorIndex < m_graph.successors(block).size()) {
374 unsigned nextSuccessorIndex = successorIndex + 1;
375 if (nextSuccessorIndex < m_graph.successors(block).size())
376 worklist.forcePush(block, nextSuccessorIndex);
377
378 typename Graph::Node successorBlock = m_graph.successors(block)[successorIndex];
379 if (worklist.push(successorBlock, 0))
380 m_data[successorBlock].parent = block;
381 }
382 }
383 }
384
385 void computeSemiDominatorsAndImplicitImmediateDominators()
386 {
387 for (unsigned currentPreNumber = m_blockByPreNumber.size(); currentPreNumber-- > 1;) {
388 typename Graph::Node block = m_blockByPreNumber[currentPreNumber];
389 BlockData& blockData = m_data[block];
390
391 // Step 2:
392 for (typename Graph::Node predecessorBlock : m_graph.predecessors(block)) {
393 typename Graph::Node intermediateBlock = eval(predecessorBlock);
394 blockData.semiNumber = std::min(
395 m_data[intermediateBlock].semiNumber, blockData.semiNumber);
396 }
397 unsigned bucketPreNumber = blockData.semiNumber;
398 ASSERT(bucketPreNumber <= currentPreNumber);
399 m_data[m_blockByPreNumber[bucketPreNumber]].bucket.append(block);
400 link(blockData.parent, block);
401
402 // Step 3:
403 for (typename Graph::Node semiDominee : m_data[blockData.parent].bucket) {
404 typename Graph::Node possibleDominator = eval(semiDominee);
405 BlockData& semiDomineeData = m_data[semiDominee];
406 ASSERT(m_blockByPreNumber[semiDomineeData.semiNumber] == blockData.parent);
407 BlockData& possibleDominatorData = m_data[possibleDominator];
408 if (possibleDominatorData.semiNumber < semiDomineeData.semiNumber)
409 semiDomineeData.dom = possibleDominator;
410 else
411 semiDomineeData.dom = blockData.parent;
412 }
413 m_data[blockData.parent].bucket.clear();
414 }
415 }
416
417 void computeExplicitImmediateDominators()
418 {
419 for (unsigned currentPreNumber = 1; currentPreNumber < m_blockByPreNumber.size(); ++currentPreNumber) {
420 typename Graph::Node block = m_blockByPreNumber[currentPreNumber];
421 BlockData& blockData = m_data[block];
422
423 if (blockData.dom != m_blockByPreNumber[blockData.semiNumber])
424 blockData.dom = m_data[blockData.dom].dom;
425 }
426 }

428 void link(typename Graph::Node from, typename Graph::Node to)
429 {
430 m_data[to].ancestor = from;
431 }
432
433 typename Graph::Node eval(typename Graph::Node block)
434 {
435 if (!m_data[block].ancestor)
436 return block;
437
438 compress(block);
439 return m_data[block].label;
440 }
441
442 void compress(typename Graph::Node initialBlock)
443 {
444 // This was meant to be a recursive function, but we don't like recursion because we don't
445 // want to blow the stack. The original function will call compress() recursively on the
446 // ancestor of anything that has an ancestor. So, we populate our worklist with the
447 // recursive ancestors of initialBlock. Then we process the list starting from the block
448 // that is furthest up the ancestor chain.
449
450 typename Graph::Node ancestor = m_data[initialBlock].ancestor;
451 ASSERT(ancestor);
452 if (!m_data[ancestor].ancestor)
453 return;
454
455 Vector<typename Graph::Node, 16> stack;
456 for (typename Graph::Node block = initialBlock; block; block = m_data[block].ancestor)
457 stack.append(block);
458
459 // We only care about blocks that have an ancestor that has an ancestor. The last two
460 // elements in the stack won't satisfy this property.
461 ASSERT(stack.size() >= 2);
462 ASSERT(!m_data[stack[stack.size() - 1]].ancestor);
463 ASSERT(!m_data[m_data[stack[stack.size() - 2]].ancestor].ancestor);
464
465 for (unsigned i = stack.size() - 2; i--;) {
466 typename Graph::Node block = stack[i];
467 typename Graph::Node& labelOfBlock = m_data[block].label;
468 typename Graph::Node& ancestorOfBlock = m_data[block].ancestor;
469 ASSERT(ancestorOfBlock);
470 ASSERT(m_data[ancestorOfBlock].ancestor);
471
472 typename Graph::Node labelOfAncestorOfBlock = m_data[ancestorOfBlock].label;
473
474 if (m_data[labelOfAncestorOfBlock].semiNumber < m_data[labelOfBlock].semiNumber)
475 labelOfBlock = labelOfAncestorOfBlock;
476 ancestorOfBlock = m_data[ancestorOfBlock].ancestor;
477 }
478 }

507 class NaiveDominators {
508 WTF_MAKE_FAST_ALLOCATED;
509 public:
510 NaiveDominators(Graph& graph)
511 : m_graph(graph)
512 {
513 // This implements a naive dominator solver.
514
515 ASSERT(!graph.predecessors(graph.root()).size());
516
517 unsigned numBlocks = graph.numNodes();
518
519 // Allocate storage for the dense dominance matrix.
520 m_results.grow(numBlocks);
521 for (unsigned i = numBlocks; i--;)
522 m_results[i].resize(numBlocks);
523 m_scratch.resize(numBlocks);
524
525 // We know that the entry block is only dominated by itself.
526 m_results[0].clearAll();
527 m_results[0][0] = true;
528
529 // Find all of the valid blocks.
530 m_scratch.clearAll();
531 for (unsigned i = numBlocks; i--;) {
532 if (!graph.node(i))
533 continue;
534 m_scratch[i] = true;
535 }
536
537 // Mark all nodes as dominated by everything.
538 for (unsigned i = numBlocks; i-- > 1;) {
539 if (!graph.node(i) || !graph.predecessors(graph.node(i)).size())
540 m_results[i].clearAll();
541 else
542 m_results[i] = m_scratch;
543 }
544
545 // Iteratively eliminate nodes that are not dominator.
546 bool changed;
547 do {
548 changed = false;
549 // Prune dominators in all non entry blocks: forward scan.
550 for (unsigned i = 1; i < numBlocks; ++i)
551 changed |= pruneDominators(i);
552
553 if (!changed)
554 break;
555
556 // Prune dominators in all non entry blocks: backward scan.
557 changed = false;
558 for (unsigned i = numBlocks; i-- > 1;)
559 changed |= pruneDominators(i);
560 } while (changed);
561 }
562
563 bool dominates(unsigned from, unsigned to) const
564 {
565 return m_results[to][from];
566 }
567
568 bool dominates(typename Graph::Node from, typename Graph::Node to) const
569 {
570 return dominates(m_graph.index(from), m_graph.index(to));
571 }
572
573 void dump(PrintStream& out) const
574 {
575 for (unsigned blockIndex = 0; blockIndex < m_graph.numNodes(); ++blockIndex) {
576 typename Graph::Node block = m_graph.node(blockIndex);
577 if (!block)
578 continue;
579 out.print(" Block ", m_graph.dump(block), ":");
580 for (unsigned otherIndex = 0; otherIndex < m_graph.numNodes(); ++otherIndex) {
581 if (!dominates(m_graph.index(block), otherIndex))
582 continue;
583 out.print(" ", m_graph.dump(m_graph.node(otherIndex)));
584 }
585 out.print("\n");
586 }
587 }
588
589 private:
590 bool pruneDominators(unsigned idx)
591 {
592 typename Graph::Node block = m_graph.node(idx);
593
594 if (!block || !m_graph.predecessors(block).size())
595 return false;
596
597 // Find the intersection of dom(preds).
598 m_scratch = m_results[m_graph.index(m_graph.predecessors(block)[0])];
599 for (unsigned j = m_graph.predecessors(block).size(); j-- > 1;)
600 m_scratch &= m_results[m_graph.index(m_graph.predecessors(block)[j])];
601
602 // The block is also dominated by itself.
603 m_scratch[idx] = true;
604
605 return m_results[idx].setAndCheck(m_scratch);
606 }
607
608 Graph& m_graph;
609 Vector<FastBitVector> m_results; // For each block, the bitvector of blocks that dominate it.
610 FastBitVector m_scratch; // A temporary bitvector with bit for each block. We recycle this to save new/deletes.
611 };

632 void handleErrors()
633 {
634 if (errors.isEmpty())
635 return;
636
637 dataLog("DFG DOMINATOR VALIDATION FAILED:\n");
638 dataLog("\n");
639 dataLog("For block domination relationships:\n");
640 for (unsigned i = 0; i < errors.size(); ++i) {
641 dataLog(
642 " ", graph.dump(errors[i].from), " -> ", graph.dump(errors[i].to),
643 " (", errors[i].message, ")\n");
644 }
645 dataLog("\n");
646 dataLog("Control flow graph:\n");
647 for (unsigned blockIndex = 0; blockIndex < graph.numNodes(); ++blockIndex) {
648 typename Graph::Node block = graph.node(blockIndex);
649 if (!block)
650 continue;
651 dataLog(" Block ", graph.dump(graph.node(blockIndex)), ": successors = [");
652 CommaPrinter comma;
653 for (auto successor : graph.successors(block))
654 dataLog(comma, graph.dump(successor));
655 dataLog("], predecessors = [");
656 comma = CommaPrinter();
657 for (auto predecessor : graph.predecessors(block))
658 dataLog(comma, graph.dump(predecessor));
659 dataLog("]\n");
660 }
661 dataLog("\n");
662 dataLog("Lengauer-Tarjan Dominators:\n");
663 dataLog(dominators);
664 dataLog("\n");
665 dataLog("Naive Dominators:\n");
666 naiveDominators.dump(WTF::dataFile());
667 dataLog("\n");
668 dataLog("Graph at time of failure:\n");
669 dataLog(graph);
670 dataLog("\n");
671 dataLog("DFG DOMINATOR VALIDATION FAILIED!\n");
672 CRASH();
673 }

699 template<typename Functor>
700 void forAllBlocksInDominanceFrontierOfImpl(
701 typename Graph::Node from, const Functor& functor) const
702 {
703 // Paraphrasing from http://en.wikipedia.org/wiki/Dominator_(graph_theory):
704 // "The dominance frontier of a block 'from' is the set of all blocks 'to' such that
705 // 'from' dominates an immediate predecessor of 'to', but 'from' does not strictly
706 // dominate 'to'."
707 //
708 // A useful corner case to remember: a block may be in its own dominance frontier if it has
709 // a loop edge to itself, since it dominates itself and so it dominates its own immediate
710 // predecessor, and a block never strictly dominates itself.
711
712 forAllBlocksDominatedBy(
713 from,
714 [&] (typename Graph::Node block) {
715 for (typename Graph::Node to : m_graph.successors(block)) {
716 if (!strictlyDominates(from, to))
717 functor(to);
718 }
719 });
720 }
721
722 template<typename Functor>
723 void forAllBlocksInIteratedDominanceFrontierOfImpl(
724 const List& from, const Functor& functor) const
725 {
726 List worklist = from;
727 while (!worklist.isEmpty()) {
728 typename Graph::Node block = worklist.takeLast();
729 forAllBlocksInDominanceFrontierOfImpl(
730 block,
731 [&] (typename Graph::Node otherBlock) {
732 if (functor(otherBlock))
733 worklist.append(otherBlock);
734 });
735 }
736 }

742 : idomParent(nullptr)
743 , preNumber(UINT_MAX)

749 typename Graph::Node idomParent;
750
751 unsigned preNumber;
752 unsigned postNumber;
753 };

756 typename Graph::template Map<BlockData> m_data;
757};
