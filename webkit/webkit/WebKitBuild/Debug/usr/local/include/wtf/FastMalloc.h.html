30// There are several malloc-related macros to annotate class / struct. If these annotations are attached,
31// allocation is handled by bmalloc if bmalloc is available.
32//
33// TLDR; Here is a quick guidance.
34//
35// 1. If your class / struct is derived from a base class which uses WTF_MAKE_ISO_ALLOCATED / WTF_MAKE_ISO_ALLOCATED_EXPORT,
36// you must use WTF_MAKE_ISO_ALLOCATED / WTF_MAKE_ISO_ALLOCATED_EXPORT.
37// 2. If your class / struct is a DOM object, use WTF_MAKE_ISO_ALLOCATED.
38// 3. If your class / struct is particularly memory consuming and if you think tracking footprint of your class is helpful for memory-reduction work,
39// use WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER / WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER.
40// 4. Otherwise, use WTF_MAKE_FAST_ALLOCATED / WTF_MAKE_STRUCT_FAST_ALLOCATED.
41//
42// Let's explain the differences in detail.
43//
44// - WTF_MAKE_FAST_ALLOCATED
45// - WTF_MAKE_STRUCT_FAST_ALLOCATED
46// class / struct is allocated with FastMalloc (bmalloc if it is available). We encourage using FastMalloc for all the class / struct allocations if possible
47// to avoid using system malloc. If a class is not having WTF_MAKE_FAST_ALLOCATED, the class will be allocated with system malloc, which is less efficient
48// compared to FastMalloc in terms of performance and memory footprint. We would like to minimize the use of system malloc.
49// These annotations should be the default choice for allocations.
50// WTF_MAKE_FAST_ALLOCATED is for classes and WTF_MAKE_STRUCT_FAST_ALLOCATED is for structs. The difference between them is how we
51// use `public:` / `private:` access specifiers in the expanded macro.
52//
53// - WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(ClassName)
54// - WTF_MAKE_STRUCT_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(ClassName)
55// Normally, these are identical to WTF_MAKE_FAST_ALLOCATED.
56// When the MallocHeapBreakdown debugging feature is enabled, these macros act differently. bmalloc is switched
57// to using system malloc internally, and bmalloc creates a named MallocZone per this annotation. MallocZone allows us to track heap usage
58// per this annotation, offering quick way of breaking down memory usage.
59// To enable MallocHeapBreakdown, set ENABLE_MALLOC_HEAP_BREAKDOWN in WTF and BENABLE_MALLOC_HEAP_BREAKDOWN in bmalloc.
60// For the details of MallocHeapBreakdown, please look at ChangeLog of https://trac.webkit.org/changeset/253987/webkit.
61//
62// - WTF_MAKE_ISO_ALLOCATED(ClassName)
63// - WTF_MAKE_ISO_ALLOCATED_EXPORT(ClassName, exportMacro)
64// class / struct is allocated from bmalloc IsoHeap. IsoHeap assigns virtual address only for particular type,
65// so that this avoids use-after-free based type punning. We are adopting IsoHeap mainly for class / struct which is exposed to user JavaScript (e.g. DOM objects).
66// For example , all the derived classes of ScriptWrappable must be allocated in IsoHeap.
67// Unlike the other macros, you need to annotate each derived class with WTF_MAKE_ISO_ALLOCATED if your base class is annotated with WTF_MAKE_ISO_ALLOCATED.
68// When you annotate the class with WTF_MAKE_ISO_ALLOCATED(XXX), you need to add WTF_MAKE_ISO_ALLOCATED_IMPL(XXX) in cpp file side.
69// Because WTF_MAKE_ISO_ALLOCATED_IMPL defines functions in cpp side, you sometimes need to annotate these functions with export macros when your class is
70// used outside of the component defining your class (e.g. your class is in WebCore and it is also used in WebKit). In this case, you can use WTF_MAKE_ISO_ALLOCATED_EXPORT
71// to annotate these functions with appropriate export macros: e.g. WTF_MAKE_ISO_ALLOCATED_EXPORT(IDBTransaction, WEBCORE_EXPORT).

186template<typename T> inline bool TryMallocReturnValue::getValue(T*& data)
187{
188 data = static_cast<T*>(m_data);
189 m_data = nullptr;
190 return data;
191}

193// C++ STL allocator implementation. You can integrate fastMalloc into STL containers.
194// e.g. std::unordered_map<Key, Value, std::hash<Key>, std::equal_to<Key>, FastAllocator<std::pair<const Key, Value>>>.
195template<typename T>
196class FastAllocator {
197public:
198 using value_type = T;
199
200 FastAllocator() = default;
201
202 template<typename U> FastAllocator(const FastAllocator<U>&) { }
203
204 T* allocate(size_t count)
205 {
206 return static_cast<T*>(fastMalloc(sizeof(T) * count));
207 }
208
209 void deallocate(T* pointer, size_t)
210 {
211 fastFree(pointer);
212 }
213
214#if defined(__GLIBCXX__) && (!defined(_GLIBCXX_RELEASE) || _GLIBCXX_RELEASE < 6)
215 // This allocator also supports pre-C++11 STL allocator interface. This is a workaround for GCC < 6, which std::list
216 // does not support C++11 allocator. Note that _GLIBCXX_RELEASE is only defined after GCC 7 release. So currently
217 // this workaround is enabled in GCC 6 too.
218 // https://gcc.gnu.org/bugzilla/show_bug.cgi?id=55409
219
220 using pointer = value_type*;
221 using const_pointer = typename std::pointer_traits<pointer>::template rebind<value_type const>;
222 using void_pointer = typename std::pointer_traits<pointer>::template rebind<void>;
223 using const_void_pointer = typename std::pointer_traits<pointer>::template rebind<const void>;
224
225 using reference = T&;
226 using const_reference = const T&;
227
228 using difference_type = typename std::pointer_traits<pointer>::difference_type;
229 using size_type = std::make_unsigned_t<difference_type>;
230
231 template <class U> struct rebind {
232 using other = FastAllocator<U>;
233 };
234
235 value_type* allocate(std::size_t count, const_void_pointer)
236 {
237 return allocate(count);
238 }
239
240 template <class U, class ...Args>
241 void construct(U* p, Args&& ...args)
242 {
243 new (const_cast<void*>(static_cast<const void*>(p))) U(std::forward<Args>(args)...);
244 }
245
246 template <class U>
247 void destroy(U* p)
248 {
249 p->~U();
250 }
251
252 std::size_t max_size() const
253 {
254 return std::numeric_limits<size_type>::max();
255 }
256
257 FastAllocator<T> select_on_container_copy_construction() const
258 {
259 return *this;
260 }
261
262 using propagate_on_container_copy_assignment = std::false_type;
263 using propagate_on_container_move_assignment = std::false_type;
264 using propagate_on_container_swap = std::false_type;
265 using is_always_equal = std::is_empty<FastAllocator>;
266#endif // defined(__GLIBCXX__) && (!defined(_GLIBCXX_RELEASE) || _GLIBCXX_RELEASE < 6)
267};

270template<typename T, typename U> inline bool operator!=(const FastAllocator<T>&, const FastAllocator<U>&) { return false; }

272struct FastMalloc {
273 static void* malloc(size_t size) { return fastMalloc(size); }
274
275 static void* tryMalloc(size_t size)
276 {
277 auto result = tryFastMalloc(size);
278 void* realResult;
279 if (result.getValue(realResult))
280 return realResult;
281 return nullptr;
282 }
283
284 static void* zeroedMalloc(size_t size) { return fastZeroedMalloc(size); }
285
286 static void* tryZeroedMalloc(size_t size)
287 {
288 auto result = tryFastZeroedMalloc(size);
289 void* realResult;
290 if (result.getValue(realResult))
291 return realResult;
292 return nullptr;
293 }
294
295 static void* realloc(void* p, size_t size) { return fastRealloc(p, size); }
296
297 static void* tryRealloc(void* p, size_t size)
298 {
299 auto result = tryFastRealloc(p, size);
300 void* realResult;
301 if (result.getValue(realResult))
302 return realResult;
303 return nullptr;
304 }
305
306 static void free(void* p) { fastFree(p); }
307};

309template<typename T>
310struct FastFree {
311 static_assert(std::is_trivially_destructible<T>::value, "");
312
313 void operator()(T* pointer) const
314 {
315 fastFree(const_cast<typename std::remove_cv<T>::type*>(pointer));
316 }
317};
318
319template<typename T>
320struct FastFree<T[]> {
321 static_assert(std::is_trivially_destructible<T>::value, "");
322
323 void operator()(T* pointer) const
324 {
325 fastFree(const_cast<typename std::remove_cv<T>::type*>(pointer));
326 }
327};
