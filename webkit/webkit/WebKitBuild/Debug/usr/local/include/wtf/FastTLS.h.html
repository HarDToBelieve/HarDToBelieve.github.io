30#include <pthread.h>

32#include <wtf/Platform.h>

34namespace WTF {
35
36// __PTK_FRAMEWORK_JAVASCRIPTCORE_KEY0-1 is taken by bmalloc, so WTF's KEY0 maps to the
37// system's KEY2.
38#define WTF_FAST_TLS_KEY0 __PTK_FRAMEWORK_JAVASCRIPTCORE_KEY2
39#define WTF_FAST_TLS_KEY1 __PTK_FRAMEWORK_JAVASCRIPTCORE_KEY3
40#define WTF_FAST_TLS_KEY2 __PTK_FRAMEWORK_JAVASCRIPTCORE_KEY4
41
42// NOTE: We should manage our use of these keys here. If you want to use a key for something,
43// put a #define in here to give your key a symbolic name. This ensures that we don't
44// accidentally use the same key for more than one thing.
45
46#define WTF_THREAD_DATA_KEY WTF_FAST_TLS_KEY0
47#define WTF_WASM_CONTEXT_KEY WTF_FAST_TLS_KEY1
48#define WTF_TESTING_KEY WTF_WASM_CONTEXT_KEY // So far, this key is only used in places that don't do WebAssembly, so it's OK that they share the same key.
49#define WTF_GC_TLC_KEY WTF_FAST_TLS_KEY2
50
51#if ENABLE(FAST_TLS_JIT)
52inline unsigned fastTLSOffsetForKey(unsigned long slot)
53{
54 return slot * sizeof(void*);
55}
56#endif
57
58} // namespace WTF
59
60#if ENABLE(FAST_TLS_JIT)
61using WTF::fastTLSOffsetForKey;
62#endif
63
64#endif // HAVE(FAST_TLS)
