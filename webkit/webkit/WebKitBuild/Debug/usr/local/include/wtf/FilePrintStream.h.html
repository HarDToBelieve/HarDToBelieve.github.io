31namespace WTF {
32
33class FilePrintStream final : public PrintStream {
34public:
35 enum AdoptionMode {
36 Adopt,
37 Borrow
38 };
39
40 FilePrintStream(FILE*, AdoptionMode = Adopt);
41 WTF_EXPORT_PRIVATE ~FilePrintStream() final;
42
43 WTF_EXPORT_PRIVATE static std::unique_ptr<FilePrintStream> open(const char* filename, const char* mode);
44
45 FILE* file() { return m_file; }
46
47 void vprintf(const char* format, va_list) final WTF_ATTRIBUTE_PRINTF(2, 0);
48 void flush() final;
49
50private:
51 FILE* m_file;
52 AdoptionMode m_adoptionMode;
53};
54
55} // namespace WTF
