213class MappedFileData {
214 WTF_MAKE_FAST_ALLOCATED;
215public:
216 MappedFileData() { }
217 MappedFileData(MappedFileData&&);
218 WTF_EXPORT_PRIVATE MappedFileData(const String& filePath, MappedFileMode, bool& success);
219 WTF_EXPORT_PRIVATE MappedFileData(PlatformFileHandle, MappedFileMode, bool& success);
220 WTF_EXPORT_PRIVATE MappedFileData(PlatformFileHandle, FileOpenMode, MappedFileMode, bool& success);
221 WTF_EXPORT_PRIVATE ~MappedFileData();
222 MappedFileData& operator=(MappedFileData&&);
223
224 explicit operator bool() const { return !!m_fileData; }
225 const void* data() const { return m_fileData; }
226 unsigned size() const { return m_fileSize; }
227
228#if PLATFORM(COCOA)
229 void* leakHandle() { return std::exchange(m_fileData, nullptr); }
230#endif
231#if OS(WINDOWS)
232 HANDLE fileMapping() const { return m_fileMapping; }
233#endif
234
235private:
236 WTF_EXPORT_PRIVATE bool mapFileHandle(PlatformFileHandle, FileOpenMode, MappedFileMode);
237
238 void* m_fileData { nullptr };
239 unsigned m_fileSize { 0 };
240#if OS(WINDOWS)
241 HANDLE m_fileMapping { nullptr };
242#endif
243};
244
245inline MappedFileData::MappedFileData(PlatformFileHandle handle, MappedFileMode mapMode, bool& success)
246{
247 success = mapFileHandle(handle, FileOpenMode::Read, mapMode);
248}
249
250inline MappedFileData::MappedFileData(PlatformFileHandle handle, FileOpenMode openMode, MappedFileMode mapMode, bool& success)
251{
252 success = mapFileHandle(handle, openMode, mapMode);
253}
254
255inline MappedFileData::MappedFileData(MappedFileData&& other)
256 : m_fileData(std::exchange(other.m_fileData, nullptr))
257 , m_fileSize(std::exchange(other.m_fileSize, 0))
258#if OS(WINDOWS)
259 , m_fileMapping(std::exchange(other.m_fileMapping, nullptr))
260#endif
261{
262}
263
264inline MappedFileData& MappedFileData::operator=(MappedFileData&& other)
265{
266 m_fileData = std::exchange(other.m_fileData, nullptr);
267 m_fileSize = std::exchange(other.m_fileSize, 0);
268#if OS(WINDOWS)
269 m_fileMapping = std::exchange(other.m_fileMapping, nullptr);
270#endif
271 return *this;
272}
