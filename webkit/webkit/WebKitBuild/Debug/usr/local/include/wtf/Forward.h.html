162// Sometimes an inline method simply forwards to another one and does nothing else. If it were
163// just a forward declaration of that method then you would only need a forward declaration of
164// its return types and parameter types too, but because it's inline and it actually needs to
165// return / pass these types (even though it's just passing through whatever it called) you
166// now find yourself having to actually have a full declaration of these types. That might be
167// an include you'd rather avoid.
168//
169// No more. Enter template magic to lazily instantiate that method!
170//
171// This macro makes the method work as if you'd declared the return / parameter types as normal,
172// but forces lazy instantiation of the method at the call site, at which point the caller (not
173// the declaration) had better have a full declaration of the return / parameter types.
174//
175// Simply pass the forward-declared types to the macro, with an alias for each, and then define
176// your function as you otherwise would have but using the aliased name. Why the alias? So you
177// can be lazy on templated types! Sample usage:
178//
179// struct Foo; // No need to define Foo!
180// template<typename T>
181// struct A {
182// Foo declared(Bar); // Forward declarations of Foo and Bar are sufficient here.
183// // The below code would normally require a definition of Foo and Bar.
184// WTF_LAZY_INSTANTIATE(Foo=Foo, Bar=Bar) Foo forwarder(Bar b) { return declared(b); }
185// };
186#define WTF_LAZY_JOIN_UNLAZE(A, B) A##B
187#define WTF_LAZY_JOIN(A, B) WTF_LAZY_JOIN_UNLAZE(A, B)
188#define WTF_LAZY_ARGUMENT_NUMBER(_1, _2, _3, _4, _5, _6, _7, _8, N, ...) N
189#define WTF_LAZY_AUGMENT(...) unused, __VA_ARGS__
190#define WTF_LAZY_EXPAND(x) x
191#define WTF_LAZY_NUM_ARGS_(...) WTF_LAZY_EXPAND(WTF_LAZY_ARGUMENT_NUMBER(__VA_ARGS__, 7, 6, 5, 4, 3, 2, 1, 0))
192#define WTF_LAZY_NUM_ARGS(...) WTF_LAZY_NUM_ARGS_(WTF_LAZY_AUGMENT(__VA_ARGS__))
193#define WTF_LAZY_FOR_EACH_TERM(F, ...) \
194 WTF_LAZY_JOIN(WTF_LAZY_FOR_EACH_TERM_, WTF_LAZY_NUM_ARGS(__VA_ARGS__))(F, (__VA_ARGS__))
195#define WTF_LAZY_FIRST(_1, ...) _1
196#define WTF_LAZY_REST(_1, ...) (__VA_ARGS__)
197#define WTF_LAZY_REST_(_1, ...) __VA_ARGS__
198#define WTF_LAZY_CALL(F, ARG) F(ARG)
199#define WTF_LAZY_FOR_EACH_TERM_0(...)
200#define WTF_LAZY_FOR_EACH_TERM_1(F, ARGS) WTF_LAZY_CALL(F, WTF_LAZY_FIRST ARGS) WTF_LAZY_FOR_EACH_TERM_0(F, WTF_LAZY_REST ARGS)
201#define WTF_LAZY_FOR_EACH_TERM_2(F, ARGS) WTF_LAZY_CALL(F, WTF_LAZY_FIRST ARGS) WTF_LAZY_FOR_EACH_TERM_1(F, WTF_LAZY_REST ARGS)
202#define WTF_LAZY_FOR_EACH_TERM_3(F, ARGS) WTF_LAZY_CALL(F, WTF_LAZY_FIRST ARGS) WTF_LAZY_FOR_EACH_TERM_2(F, WTF_LAZY_REST ARGS)
203#define WTF_LAZY_FOR_EACH_TERM_4(F, ARGS) WTF_LAZY_CALL(F, WTF_LAZY_FIRST ARGS) WTF_LAZY_FOR_EACH_TERM_3(F, WTF_LAZY_REST ARGS)
204#define WTF_LAZY_FOR_EACH_TERM_5(F, ARGS) WTF_LAZY_CALL(F, WTF_LAZY_FIRST ARGS) WTF_LAZY_FOR_EACH_TERM_4(F, WTF_LAZY_REST ARGS)
205#define WTF_LAZY_FOR_EACH_TERM_6(F, ARGS) WTF_LAZY_CALL(F, WTF_LAZY_FIRST ARGS) WTF_LAZY_FOR_EACH_TERM_5(F, WTF_LAZY_REST ARGS)
206#define WTF_LAZY_FOR_EACH_TERM_7(F, ARGS) WTF_LAZY_CALL(F, WTF_LAZY_FIRST ARGS) WTF_LAZY_FOR_EACH_TERM_6(F, WTF_LAZY_REST ARGS)
