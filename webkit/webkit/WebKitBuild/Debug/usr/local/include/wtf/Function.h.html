34namespace Detail {

36template<typename Out, typename... In>
37class CallableWrapperBase {
38 WTF_MAKE_FAST_ALLOCATED;
39public:
40 virtual ~CallableWrapperBase() { }
41 virtual Out call(In...) = 0;
42};
43
44template<typename, typename, typename...> class CallableWrapper;
45
46template<typename CallableType, typename Out, typename... In>
47class CallableWrapper : public CallableWrapperBase<Out, In...> {
48public:
49 explicit CallableWrapper(CallableType&& callable)
50 : m_callable(WTFMove(callable)) { }
51 CallableWrapper(const CallableWrapper&) = delete;
52 CallableWrapper& operator=(const CallableWrapper&) = delete;
53 Out call(In... in) final { return m_callable(std::forward<In>(in)...); }
54private:
55 CallableType m_callable;
56};
57
58} // namespace Detail
59
60template<typename Out, typename... In> Function<Out(In...)> adopt(Detail::CallableWrapperBase<Out, In...>*);
61
62template <typename Out, typename... In>
63class Function<Out(In...)> {
64 WTF_MAKE_FAST_ALLOCATED;
65public:
66 using Impl = Detail::CallableWrapperBase<Out, In...>;
67
68 Function() = default;
69 Function(std::nullptr_t) { }
70
71 template<typename CallableType, class = typename std::enable_if<!(std::is_pointer<CallableType>::value && std::is_function<typename std::remove_pointer<CallableType>::type>::value) && std::is_rvalue_reference<CallableType&&>::value>::type>
72 Function(CallableType&& callable)
73 : m_callableWrapper(makeUnique<Detail::CallableWrapper<CallableType, Out, In...>>(std::forward<CallableType>(callable))) { }
74
75 template<typename FunctionType, class = typename std::enable_if<std::is_pointer<FunctionType>::value && std::is_function<typename std::remove_pointer<FunctionType>::type>::value>::type>
76 Function(FunctionType f)
77 : m_callableWrapper(makeUnique<Detail::CallableWrapper<FunctionType, Out, In...>>(std::forward<FunctionType>(f))) { }
78
79 Out operator()(In... in) const
80 {
81 ASSERT(m_callableWrapper);
82 return m_callableWrapper->call(std::forward<In>(in)...);
83 }
84
85 explicit operator bool() const { return !!m_callableWrapper; }
86
87 template<typename CallableType, class = typename std::enable_if<!(std::is_pointer<CallableType>::value && std::is_function<typename std::remove_pointer<CallableType>::type>::value) && std::is_rvalue_reference<CallableType&&>::value>::type>
88 Function& operator=(CallableType&& callable)
89 {
90 m_callableWrapper = makeUnique<Detail::CallableWrapper<CallableType, Out, In...>>(std::forward<CallableType>(callable));
91 return *this;
92 }
93
94 template<typename FunctionType, class = typename std::enable_if<std::is_pointer<FunctionType>::value && std::is_function<typename std::remove_pointer<FunctionType>::type>::value>::type>
95 Function& operator=(FunctionType f)
96 {
97 m_callableWrapper = makeUnique<Detail::CallableWrapper<FunctionType, Out, In...>>(std::forward<FunctionType>(f));
98 return *this;
99 }
100
101 Function& operator=(std::nullptr_t)
102 {
103 m_callableWrapper = nullptr;
104 return *this;
105 }
106
107 Impl* leak()
108 {
109 return m_callableWrapper.release();
110 }
111
112private:
113 enum AdoptTag { Adopt };
114 Function(Impl* impl, AdoptTag)
115 : m_callableWrapper(impl)
116 {
117 }
118
119 friend Function adopt<Out, In...>(Impl*);
120
121 std::unique_ptr<Impl> m_callableWrapper;
122};
123
124template<typename Out, typename... In> Function<Out(In...)> adopt(Detail::CallableWrapperBase<Out, In...>* impl)
125{
126 return Function<Out(In...)>(impl, Function<Out(In...)>::Adopt);
127}
