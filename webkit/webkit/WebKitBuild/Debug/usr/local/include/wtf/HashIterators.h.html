43 template<typename HashTableType, typename KeyType, typename MappedType> struct HashTableConstIteratorAdapter<HashTableType, KeyValuePair<KeyType, MappedType>> : public std::iterator<std::forward_iterator_tag, KeyValuePair<KeyType, MappedType>, std::ptrdiff_t, const KeyValuePair<KeyType, MappedType>*, const KeyValuePair<KeyType, MappedType>&> {
44 private:
45 typedef KeyValuePair<KeyType, MappedType> ValueType;
46 public:
47 typedef HashTableConstKeysIterator<HashTableType, KeyType, MappedType> Keys;
48 typedef HashTableConstValuesIterator<HashTableType, KeyType, MappedType> Values;
49
50 HashTableConstIteratorAdapter() {}
51 HashTableConstIteratorAdapter(const typename HashTableType::const_iterator& impl) : m_impl(impl) {}
52
53 const ValueType* get() const { return (const ValueType*)m_impl.get(); }
54 const ValueType& operator*() const { return *get(); }
55 const ValueType* operator->() const { return get(); }
56
57 HashTableConstIteratorAdapter& operator++() { ++m_impl; return *this; }
58 // postfix ++ intentionally omitted
59
60 Keys keys() { return Keys(*this); }
61 Values values() { return Values(*this); }
62
63 typename HashTableType::const_iterator m_impl;
64 };
