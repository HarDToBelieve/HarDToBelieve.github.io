41 struct KeyValuePairTraits : KeyValuePairHashTraits<KeyTraits, MappedTraits> {

44 {
45 return isHashTraitsEmptyValue<KeyTraits>(value.key);
46 }

67public:

93 HashMap(std::initializer_list<KeyValuePairType> initializerList)
94 {
95 reserveInitialCapacity(initializerList.size());
96 for (const auto& keyValuePair : initializerList)
97 add(keyValuePair.key, keyValuePair.value);
98 }

167 // An alternate version of find() that finds the object by hashing and comparing
168 // with some other type, to avoid the cost of type conversion. HashTranslator
169 // must have the following function members:
170 // static unsigned hash(const T&);
171 // static bool equal(const ValueType&, const T&);

190 template<typename K = KeyType> typename std::enable_if<IsSmartPtr<K>::value, MappedPeekType>::type inlineGet(typename GetPtrHelper<K>::PtrType) const;
191 template<typename K = KeyType> typename std::enable_if<IsSmartPtr<K>::value, MappedPeekType>::type get(typename GetPtrHelper<K>::PtrType) const;

224template<typename ValueTraits, typename HashFunctions>
225struct HashMapTranslator {
226 template<typename T> static unsigned hash(const T& key) { return HashFunctions::hash(key); }
227 template<typename T, typename U> static bool equal(const T& a, const U& b) { return HashFunctions::equal(a, b); }
228 template<typename T, typename U, typename V> static void translate(T& location, U&& key, V&& mapped)
229 {
230 ValueTraits::KeyTraits::assignToEmpty(location.key, std::forward<U>(key));
231 ValueTraits::ValueTraits::assignToEmpty(location.value, std::forward<V>(mapped));
232 }
233};
234
235template<typename ValueTraits, typename HashFunctions>
236struct HashMapEnsureTranslator {
237 template<typename T> static unsigned hash(const T& key) { return HashFunctions::hash(key); }
238 template<typename T, typename U> static bool equal(const T& a, const U& b) { return HashFunctions::equal(a, b); }
239 template<typename T, typename U, typename Functor> static void translate(T& location, U&& key, Functor&& functor)
240 {
241 ValueTraits::KeyTraits::assignToEmpty(location.key, std::forward<U>(key));
242 ValueTraits::ValueTraits::assignToEmpty(location.value, functor());
243 }
244};
245
246template<typename ValueTraits, typename Translator>
247struct HashMapTranslatorAdapter {
248 template<typename T> static unsigned hash(const T& key) { return Translator::hash(key); }
249 template<typename T, typename U> static bool equal(const T& a, const U& b) { return Translator::equal(a, b); }
250 template<typename T, typename U, typename V> static void translate(T& location, U&& key, V&& mapped, unsigned hashCode)
251 {
252 Translator::translate(location.key, key, hashCode);
253 location.value = std::forward<V>(mapped);
254 }
255};

364template<typename KeyArg, typename MappedArg, typename HashArg, typename KeyTraitsArg, typename MappedTraitsArg, typename TableTraitsArg>
365template<typename K, typename V>
366auto HashMap<KeyArg, MappedArg, HashArg, KeyTraitsArg, MappedTraitsArg, TableTraitsArg>::inlineSet(K&& key, V&& value) -> AddResult
367{
368 AddResult result = inlineAdd(std::forward<K>(key), std::forward<V>(value));
369 if (!result.isNewEntry) {
370 // The inlineAdd call above found an existing hash table entry; we need to set the mapped value.
371 result.iterator->value = std::forward<V>(value);
372 }
373 return result;
374}

557 iterator it = find(key);
558 if (it == end())
559 return MappedTraits::take(MappedTraits::emptyValue());
560 auto value = MappedTraits::take(WTFMove(it->value));
561 remove(it);
562 return value;
563}

571template<typename T, typename U, typename V, typename W, typename X, typename Y>
572inline bool HashMap<T, U, V, W, X, Y>::isValidKey(const KeyType& key)
573{
574 if (KeyTraits::isDeletedValue(key))
575 return false;
576
577 if (HashFunctions::safeToCompareToEmptyOrDeleted) {
578 if (key == KeyTraits::emptyValue())
579 return false;
580 } else {
581 if (isHashTraitsEmptyValue<KeyTraits>(key))
582 return false;
583 }
584
585 return true;
586}

588template<typename T, typename U, typename V, typename W, typename X, typename Y>
589bool operator==(const HashMap<T, U, V, W, X, Y>& a, const HashMap<T, U, V, W, X, Y>& b)
590{
591 if (a.size() != b.size())
592 return false;
593
594 typedef typename HashMap<T, U, V, W, X, Y>::const_iterator const_iterator;
595
596 const_iterator end = a.end();
597 const_iterator notFound = b.end();
598 for (const_iterator it = a.begin(); it != end; ++it) {
599 const_iterator bPos = b.find(it->key);
600 if (bPos == notFound || it->value != bPos->value)
601 return false;
602 }
603
604 return true;
605}
