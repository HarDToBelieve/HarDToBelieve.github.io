30template<typename ValueArg, typename HashArg, typename TraitsArg, typename TableTraitsArg>
31class HashSet final {
32 WTF_MAKE_FAST_ALLOCATED;
33private:
34 using HashFunctions = HashArg;
35 using ValueTraits = TraitsArg;
36 using TakeType = typename ValueTraits::TakeType;
37
38public:
39 using ValueType = typename ValueTraits::TraitType;
40
41private:
42 using HashTableType = typename TableTraitsArg::template TableType<ValueType, ValueType, IdentityExtractor, HashFunctions, ValueTraits, ValueTraits>;
43
44public:
45 // HashSet iterators have the following structure:
46 // const ValueType* get() const;
47 // const ValueType& operator*() const;
48 // const ValueType* operator->() const;
49 using iterator = HashTableConstIteratorAdapter<HashTableType, ValueType>;
50 using const_iterator = HashTableConstIteratorAdapter<HashTableType, ValueType>;
51
52 // HashSet AddResults have the following fields:
53 // IteratorType iterator;
54 // bool isNewEntry;
55 using AddResult = typename HashTableType::AddResult;
56
57 HashSet() = default;
58 HashSet(std::initializer_list<ValueArg> initializerList)
59 {
60 for (const auto& value : initializerList)
61 add(value);
62 }
63
64 void swap(HashSet&);
65
66 unsigned size() const;
67 unsigned capacity() const;
68 unsigned memoryUse() const;
69 bool isEmpty() const;
70
71 void reserveInitialCapacity(unsigned keyCount) { m_impl.reserveInitialCapacity(keyCount); }
72
73 iterator begin() const;
74 iterator end() const;
75
76 iterator random() const { return m_impl.random(); }
77
78 iterator find(const ValueType&) const;
79 bool contains(const ValueType&) const;
80
81 // An alternate version of find() that finds the object by hashing and comparing
82 // with some other type, to avoid the cost of type conversion. HashTranslator
83 // must have the following function members:
84 // static unsigned hash(const T&);
85 // static bool equal(const ValueType&, const T&);
86 template<typename HashTranslator, typename T> iterator find(const T&) const;
87 template<typename HashTranslator, typename T> bool contains(const T&) const;
88
89
90 // The return value includes both an iterator to the added value's location,
91 // and an isNewEntry bool that indicates if it is a new or existing entry in the set.
92 AddResult add(const ValueType&);
93 AddResult add(ValueType&&);
94 void add(std::initializer_list<std::reference_wrapper<const ValueType>>);
95
96 void addVoid(const ValueType&);
97 void addVoid(ValueType&&);
98
99 // An alternate version of add() that finds the object by hashing and comparing
100 // with some other type, to avoid the cost of type conversion if the object is already
101 // in the table. HashTranslator must have the following function members:
102 // static unsigned hash(const T&);
103 // static bool equal(const ValueType&, const T&);
104 // static translate(ValueType&, const T&, unsigned hashCode);
105 template<typename HashTranslator, typename T> AddResult add(const T&);
106
107 // An alternate version of translated add(), ensure() will still do translation
108 // by hashing and comparing with some other type, to avoid the cost of type
109 // conversion if the object is already in the table, but rather than a static
110 // translate() function, uses the passed in functor to perform lazy creation of
111 // the value only if it is not already there. HashTranslator must have the following
112 // function members:
113 // static unsigned hash(const T&);
114 // static bool equal(const ValueType&, const T&);
115 template<typename HashTranslator, typename T, typename Functor> AddResult ensure(T&&, Functor&&);
116
117 // Attempts to add a list of things to the set. Returns true if any of
118 // them are new to the set. Returns false if the set is unchanged.
119 template<typename IteratorType>
120 bool add(IteratorType begin, IteratorType end);
121 template<typename IteratorType>
122 bool remove(IteratorType begin, IteratorType end);
123
124 bool remove(const ValueType&);
125 bool remove(iterator);
126 template<typename Functor>
127 bool removeIf(const Functor&);
128 void clear();
129
130 TakeType take(const ValueType&);
131 TakeType take(iterator);
132 TakeType takeAny();
133
134 // Overloads for smart pointer values that take the raw pointer type as the parameter.
135 template<typename V = ValueType> typename std::enable_if<IsSmartPtr<V>::value, iterator>::type find(typename GetPtrHelper<V>::PtrType) const;
136 template<typename V = ValueType> typename std::enable_if<IsSmartPtr<V>::value, bool>::type contains(typename GetPtrHelper<V>::PtrType) const;
137 template<typename V = ValueType> typename std::enable_if<IsSmartPtr<V>::value, bool>::type remove(typename GetPtrHelper<V>::PtrType);
138 template<typename V = ValueType> typename std::enable_if<IsSmartPtr<V>::value, TakeType>::type take(typename GetPtrHelper<V>::PtrType);
139
140 static bool isValidValue(const ValueType&);
141
142 template<typename OtherCollection>
143 bool operator==(const OtherCollection&) const;
144
145 template<typename OtherCollection>
146 bool operator!=(const OtherCollection&) const;
147
148 void checkConsistency() const;
149
150private:
151 HashTableType m_impl;
152};

154struct IdentityExtractor {
155 template<typename T> static const T& extract(const T& t) { return t; }
156};

158template<typename ValueTraits, typename HashFunctions>
159struct HashSetTranslator {
160 template<typename T> static unsigned hash(const T& key) { return HashFunctions::hash(key); }
161 template<typename T, typename U> static bool equal(const T& a, const U& b) { return HashFunctions::equal(a, b); }
162 template<typename T, typename U, typename V> static void translate(T& location, U&&, V&& value)
163 {
164 ValueTraits::assignToEmpty(location, std::forward<V>(value));
165 }
166};
167
168template<typename Translator>
169struct HashSetTranslatorAdapter {
170 template<typename T> static unsigned hash(const T& key) { return Translator::hash(key); }
171 template<typename T, typename U> static bool equal(const T& a, const U& b) { return Translator::equal(a, b); }
172 template<typename T, typename U> static void translate(T& location, const U& key, const U&, unsigned hashCode)
173 {
174 Translator::translate(location, key, hashCode);
175 }
176};
177
178template<typename ValueTraits, typename Translator>
179struct HashSetEnsureTranslatorAdaptor {
180 template<typename T> static unsigned hash(const T& key) { return Translator::hash(key); }
181 template<typename T, typename U> static bool equal(const T& a, const U& b) { return Translator::equal(a, b); }
182 template<typename T, typename U, typename Functor> static void translate(T& location, U&&, Functor&& functor)
183 {
184 ValueTraits::assignToEmpty(location, functor());
185 }
186};

294template<typename T, typename U, typename V, typename W>
295template<typename IteratorType>
296inline bool HashSet<T, U, V, W>::add(IteratorType begin, IteratorType end)
297{
298 bool changed = false;
299 for (IteratorType iter = begin; iter != end; ++iter)
300 changed |= add(*iter).isNewEntry;
301 return changed;
302}

304template<typename T, typename U, typename V, typename W>
305template<typename IteratorType>
306inline bool HashSet<T, U, V, W>::remove(IteratorType begin, IteratorType end)
307{
308 bool changed = false;
309 for (IteratorType iter = begin; iter != end; ++iter)
310 changed |= remove(*iter);
311 return changed;
312}

314template<typename T, typename U, typename V, typename W>
315inline bool HashSet<T, U, V, W>::remove(iterator it)
316{
317 if (it.m_impl == m_impl.end())
318 return false;
319 m_impl.internalCheckTableConsistency();
320 m_impl.removeWithoutEntryConsistencyCheck(it.m_impl);
321 return true;
322}

343template<typename T, typename U, typename V, typename W>
344inline auto HashSet<T, U, V, W>::take(iterator it) -> TakeType
345{
346 if (it == end())
347 return ValueTraits::take(ValueTraits::emptyValue());
348
349 auto result = ValueTraits::take(WTFMove(const_cast<ValueType&>(*it)));
350 remove(it);
351 return result;
352}

394template<typename T, typename U, typename V, typename W>
395inline bool HashSet<T, U, V, W>::isValidValue(const ValueType& value)
396{
397 if (ValueTraits::isDeletedValue(value))
398 return false;
399
400 if (HashFunctions::safeToCompareToEmptyOrDeleted) {
401 if (value == ValueTraits::emptyValue())
402 return false;
403 } else {
404 if (isHashTraitsEmptyValue<ValueTraits>(value))
405 return false;
406 }
407
408 return true;
409}

411template<typename T, typename U, typename V, typename W>
412template<typename OtherCollection>
413inline bool HashSet<T, U, V, W>::operator==(const OtherCollection& otherCollection) const
414{
415 if (size() != otherCollection.size())
416 return false;
417 for (const auto& other : otherCollection) {
418 if (!contains(other))
419 return false;
420 }
421 return true;
422}

427{
