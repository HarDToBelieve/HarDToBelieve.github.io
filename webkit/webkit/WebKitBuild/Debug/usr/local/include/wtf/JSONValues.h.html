42// use a different inner namespace name so that JSON::Value is not ambigious.

52// FIXME: unify this JSON parser with JSONParse in JavaScriptCore.
53class WTF_EXPORT_PRIVATE Value : public RefCounted<Value> {
54public:
55 static constexpr int maxDepth = 1000;
56
57 virtual ~Value()
58 {
59 switch (m_type) {
60 case Type::Null:
61 case Type::Boolean:
62 case Type::Double:
63 case Type::Integer:
64 break;
65 case Type::String:
66 if (m_value.string)
67 m_value.string->deref();
68 break;
69 case Type::Object:
70 case Type::Array:
71 break;
72 }
73 }
74
75 static Ref<Value> null();
76 static Ref<Value> create(bool);
77 static Ref<Value> create(int);
78 static Ref<Value> create(double);
79 static Ref<Value> create(const String&);
80 template<class T>
81 static Ref<Value> create(T) = delete;
82
83 enum class Type {
84 Null = 0,
85 Boolean,
86 Double,
87 Integer,
88 String,
89 Object,
90 Array,
91 };
92
93 Type type() const { return m_type; }
94 bool isNull() const { return m_type == Type::Null; }
95
96 std::optional<bool> asBoolean() const;
97 std::optional<int> asInteger() const;
98 std::optional<double> asDouble() const;
99 String asString() const;
100 RefPtr<Value> asValue();
101 virtual RefPtr<Object> asObject();
102 virtual RefPtr<Array> asArray();
103
104 static RefPtr<Value> parseJSON(const String&);
105 static void escapeString(StringBuilder&, StringView);
106
107 String toJSONString() const;
108 virtual void writeJSON(StringBuilder& output) const;
109
110 virtual size_t memoryCost() const;
111
112 // FIXME: <http://webkit.org/b/179847> remove these functions when legacy InspectorObject symbols are no longer needed.
113 bool asDouble(double&) const;
114 bool asInteger(int&) const;
115 bool asString(String&) const;
116
117protected:
118 Value()
119 : m_type { Type::Null }
120 {
121 }
122
123 explicit Value(Type type)
124 : m_type(type)
125 {
126 }
127
128 explicit Value(bool value)
129 : m_type { Type::Boolean }
130 {
131 m_value.boolean = value;
132 }
133
134 explicit Value(int value)
135 : m_type { Type::Integer }
136 {
137 m_value.number = static_cast<double>(value);
138 }
139
140 explicit Value(double value)
141 : m_type(Type::Double)
142 {
143 m_value.number = value;
144 }
145
146 explicit Value(const String& value)
147 : m_type { Type::String }
148 {
149 m_value.string = value.impl();
150 if (m_value.string)
151 m_value.string->ref();
152 }
153
154private:
155 Type m_type { Type::Null };
156 union {
157 bool boolean;
158 double number;
159 StringImpl* string;
160 } m_value;
161};

398template<typename T>
399class ArrayOf final : public ArrayBase {
400private:
401 ArrayOf() { }
402
403 Array& castedArray()
404 {
405 COMPILE_ASSERT(sizeof(Array) == sizeof(ArrayOf<T>), cannot_cast);
406 return *static_cast<Array*>(static_cast<ArrayBase*>(this));
407 }
408
409public:
410
411 template <typename V = T>
412 std::enable_if_t<std::is_same_v<bool, V> || std::is_same_v<Value, V>> addItem(bool value)
413 {
414 castedArray().pushBoolean(value);
415 }
416
417 template <typename V = T>
418 std::enable_if_t<std::is_same_v<int, V> || std::is_same_v<Value, V>> addItem(int value)
419 {
420 castedArray().pushInteger(value);
421 }
422
423 template <typename V = T>
424 std::enable_if_t<std::is_same_v<double, V> || std::is_same_v<Value, V>> addItem(double value)
425 {
426 castedArray().pushDouble(value);
427 }
428
429 template <typename V = T>
430 std::enable_if_t<std::is_same_v<String, V> || std::is_same_v<Value, V>> addItem(const String& value)
431 {
432 castedArray().pushString(value);
433 }
434
435 template <typename V = T>
436 std::enable_if_t<std::is_base_of_v<Value, V> && !std::is_base_of_v<ObjectBase, V> && !std::is_base_of_v<ArrayBase, V>> addItem(Ref<Value>&& value)
437 {
438 castedArray().pushValue(WTFMove(value));
439 }
440
441 template <typename V = T>
442 std::enable_if_t<std::is_base_of_v<ObjectBase, V>> addItem(Ref<ObjectBase>&& value)
443 {
444 castedArray().pushObject(WTFMove(value));
445 }
446
447 template <typename V = T>
448 std::enable_if_t<std::is_base_of_v<ArrayBase, V>> addItem(Ref<ArrayBase>&& value)
449 {
450 castedArray().pushArray(WTFMove(value));
451 }
452
453 static Ref<ArrayOf<T>> create()
454 {
455 return adoptRef(*new ArrayOf<T>());
456 }
457
458 using ArrayBase::get;
459 using ArrayBase::begin;
460 using ArrayBase::end;
461};
