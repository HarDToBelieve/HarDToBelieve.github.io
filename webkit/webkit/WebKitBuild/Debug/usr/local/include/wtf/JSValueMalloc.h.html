37#define WTF_MAKE_JSVALUE_ALLOCATED \
38public: \
39 void* operator new(size_t, void* p) { return p; } \
40 void* operator new[](size_t, void* p) { return p; } \
41 \
42 void* operator new(size_t size) \
43 { \
44 return ::WTF::jsValueMalloc(size); \
45 } \
46 \
47 void operator delete(void* p) \
48 { \
49 ::WTF::jsValueFree(p); \
50 } \
51 \
52 void* operator new[](size_t size) \
53 { \
54 return ::WTF::jsValueMalloc(size); \
55 } \
56 \
57 void operator delete[](void* p) \
58 { \
59 ::WTF::jsValueFree(p); \
60 } \
61 void* operator new(size_t, NotNullTag, void* location) \
62 { \
63 ASSERT(location); \
64 return location; \
65 } \

67typedef int __thisIsHereToForceASemicolonAfterThisMacro UNUSED_TYPE_ALIAS

70struct JSValueMalloc {
71 static void* malloc(size_t size) { return jsValueMalloc(size); }
72 static void* tryMalloc(size_t size) { return tryJSValueMalloc(size); }
73 static void* realloc(void* p, size_t size) { return jsValueRealloc(p, size); }
74 static void free(void* p) { jsValueFree(p); }
75};
