36namespace WTF {
37
38enum class ShouldMinimizeLanguages { No, Yes };
39
40WTF_EXPORT_PRIVATE String defaultLanguage(ShouldMinimizeLanguages = ShouldMinimizeLanguages::Yes); // Thread-safe.
41WTF_EXPORT_PRIVATE Vector<String> userPreferredLanguages(ShouldMinimizeLanguages = ShouldMinimizeLanguages::Yes); // Thread-safe, returns BCP 47 language tags.
42WTF_EXPORT_PRIVATE Vector<String> userPreferredLanguagesOverride();
43WTF_EXPORT_PRIVATE void overrideUserPreferredLanguages(const Vector<String>&);
44WTF_EXPORT_PRIVATE size_t indexOfBestMatchingLanguageInList(const String& language, const Vector<String>& languageList, bool& exactMatch);
45
46// Called from platform specific code when the user's preferred language(s) change.
47WTF_EXPORT_PRIVATE void languageDidChange();
48
49// The observer function will be called when system language changes.
50typedef void (*LanguageChangeObserverFunction)(void* context);
51WTF_EXPORT_PRIVATE void addLanguageChangeObserver(void* context, LanguageChangeObserverFunction);
52WTF_EXPORT_PRIVATE void removeLanguageChangeObserver(void* context);
53WTF_EXPORT_PRIVATE String displayNameForLanguageLocale(const String&);
54
55Vector<String> platformUserPreferredLanguages(ShouldMinimizeLanguages = ShouldMinimizeLanguages::Yes);
