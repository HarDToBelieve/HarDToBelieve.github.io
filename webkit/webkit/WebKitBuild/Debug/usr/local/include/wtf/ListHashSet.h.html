32// ListHashSet: Just like HashSet, this class provides a Set
33// interface - a collection of unique objects with O(1) insertion,
34// removal and test for containership. However, it also has an
35// order - iterating it will always give back values in the order
36// in which they are added.
37
38// Unlike iteration of most WTF Hash data structures, iteration is
39// guaranteed safe against mutation of the ListHashSet, except for
40// removal of the item currently pointed to by a given iterator.
41
42template<typename Value, typename HashFunctions> class ListHashSet;
43
44template<typename ValueArg, typename HashArg> class ListHashSetIterator;
45template<typename ValueArg, typename HashArg> class ListHashSetConstIterator;
46
47template<typename ValueArg> struct ListHashSetNode;
48
49template<typename HashArg> struct ListHashSetNodeHashFunctions;
50template<typename HashArg> struct ListHashSetTranslator;
51

55#endif

59 typedef ListHashSetNode<ValueArg> Node;
60
61 typedef HashTraits<Node*> NodeTraits;
62 typedef ListHashSetNodeHashFunctions<HashArg> NodeHash;
63 typedef ListHashSetTranslator<HashArg> BaseTranslator;
64
65 typedef HashArg HashFunctions;
66
67public:
68 typedef ValueArg ValueType;
69
70 typedef ListHashSetIterator<ValueType, HashArg> iterator;
71 typedef ListHashSetConstIterator<ValueType, HashArg> const_iterator;
72 friend class ListHashSetConstIterator<ValueType, HashArg>;
73
74 typedef std::reverse_iterator<iterator> reverse_iterator;
75 typedef std::reverse_iterator<const_iterator> const_reverse_iterator;
76
77 typedef HashTableAddResult<iterator> AddResult;

121 // with some other type, to avoid the cost of type conversion.

153 template<typename V = ValueType> typename std::enable_if<IsSmartPtr<V>::value, const_iterator>::type find(typename GetPtrHelper<V>::PtrType) const;
154 template<typename V = ValueType> typename std::enable_if<IsSmartPtr<V>::value, bool>::type contains(typename GetPtrHelper<V>::PtrType) const;
155 template<typename V = ValueType> typename std::enable_if<IsSmartPtr<V>::value, AddResult>::type insertBefore(typename GetPtrHelper<V>::PtrType, const ValueType&);
156 template<typename V = ValueType> typename std::enable_if<IsSmartPtr<V>::value, AddResult>::type insertBefore(typename GetPtrHelper<V>::PtrType, ValueType&&);

159private:

246template<typename ValueArg, typename HashArg> class ListHashSetConstIterator {
247 WTF_MAKE_FAST_ALLOCATED;
248private:
249 typedef ListHashSet<ValueArg, HashArg> ListHashSetType;
250 typedef ListHashSetIterator<ValueArg, HashArg> iterator;
251 typedef ListHashSetConstIterator<ValueArg, HashArg> const_iterator;
252 typedef ListHashSetNode<ValueArg> Node;
253 typedef ValueArg ValueType;
254
255 friend class ListHashSet<ValueArg, HashArg>;
256 friend class ListHashSetIterator<ValueArg, HashArg>;
257
258 ListHashSetConstIterator(const ListHashSetType* set, Node* position)
259 : m_set(set)
260 , m_position(position)
261#if CHECK_HASHTABLE_ITERATORS
262 , m_weakSet(makeWeakPtr(set))
263 , m_weakPosition(makeWeakPtr(position))
264#endif
265 {
266 }
267
268public:
269 typedef ptrdiff_t difference_type;
270 typedef const ValueType value_type;
271 typedef const ValueType* pointer;
272 typedef const ValueType& reference;
273 typedef std::bidirectional_iterator_tag iterator_category;
274
275 ListHashSetConstIterator()
276 {
277 }
278
279 const ValueType* get() const
280 {
281#if CHECK_HASHTABLE_ITERATORS
282 ASSERT(m_weakPosition);
283#endif
284 return std::addressof(m_position->m_value);
285 }
286
287 const ValueType& operator*() const { return *get(); }
288 const ValueType* operator->() const { return get(); }
289
290 const_iterator& operator++()
291 {
292#if CHECK_HASHTABLE_ITERATORS
293 ASSERT(m_weakPosition);
294#endif
295 ASSERT(m_position);
296 m_position = m_position->m_next;
297#if CHECK_HASHTABLE_ITERATORS
298 m_weakPosition = makeWeakPtr(m_position);
299#endif
300 return *this;
301 }
302
303 // postfix ++ intentionally omitted
304
305 const_iterator& operator--()
306 {
307#if CHECK_HASHTABLE_ITERATORS
308 ASSERT(m_weakSet);
309 m_weakPosition.get();
310#endif
311 ASSERT(m_position != m_set->m_head);
312 if (!m_position)
313 m_position = m_set->m_tail;
314 else
315 m_position = m_position->m_prev;
316#if CHECK_HASHTABLE_ITERATORS
317 m_weakPosition = makeWeakPtr(m_position);
318#endif
319 return *this;
320 }
321
322 // postfix -- intentionally omitted
323
324 // Comparison.
325 bool operator==(const const_iterator& other) const
326 {
327 return m_position == other.m_position;
328 }
329 bool operator!=(const const_iterator& other) const
330 {
331 return m_position != other.m_position;
332 }
333
334private:
335 Node* node() { return m_position; }
336
337 const ListHashSetType* m_set { nullptr };
338 Node* m_position { nullptr };
339#if CHECK_HASHTABLE_ITERATORS
340 WeakPtr<const ListHashSetType> m_weakSet;
341 WeakPtr<Node> m_weakPosition;
342#endif
343};

718template<typename T, typename U>
719void ListHashSet<T, U>::unlink(Node* node)
720{
721 if (!node->m_prev) {
722 ASSERT(node == m_head);
723 m_head = node->m_next;
724 } else {
725 ASSERT(node != m_head);
726 node->m_prev->m_next = node->m_next;
727 }
728
729 if (!node->m_next) {
730 ASSERT(node == m_tail);
731 m_tail = node->m_prev;
732 } else {
733 ASSERT(node != m_tail);
734 node->m_next->m_prev = node->m_prev;
735 }
736}

738template<typename T, typename U>
739void ListHashSet<T, U>::unlinkAndDelete(Node* node)
740{
741 unlink(node);
742 delete node;
743}

745template<typename T, typename U>
746void ListHashSet<T, U>::appendNode(Node* node)
747{
748 node->m_prev = m_tail;
749 node->m_next = nullptr;
750
751 if (m_tail) {
752 ASSERT(m_head);
753 m_tail->m_next = node;
754 } else {
755 ASSERT(!m_head);
756 m_head = node;
757 }
758
759 m_tail = node;
760}

762template<typename T, typename U>
763void ListHashSet<T, U>::prependNode(Node* node)
764{
765 node->m_prev = nullptr;
766 node->m_next = m_head;
767
768 if (m_head)
769 m_head->m_prev = node;
770 else
771 m_tail = node;
772
773 m_head = node;
774}

776template<typename T, typename U>
777void ListHashSet<T, U>::insertNodeBefore(Node* beforeNode, Node* newNode)
778{
779 if (!beforeNode)
780 return appendNode(newNode);
781
782 newNode->m_next = beforeNode;
783 newNode->m_prev = beforeNode->m_prev;
784 if (beforeNode->m_prev)
785 beforeNode->m_prev->m_next = newNode;
786 beforeNode->m_prev = newNode;
787
788 if (!newNode->m_prev)
789 m_head = newNode;
790}

792template<typename T, typename U>
793void ListHashSet<T, U>::deleteAllNodes()
794{
795 if (!m_head)
796 return;
797
798 for (Node* node = m_head, *next = m_head->m_next; node; node = next, next = node ? node->m_next : nullptr)
799 delete node;
800}
