26#pragma once
27
28#include <wtf/PrintStream.h>
29#include <wtf/RecursiveLockAdapter.h>
30#include <wtf/WordLock.h>
31
32namespace WTF {
33
34// Makes every call to print() atomic.
35class LockedPrintStream final : public PrintStream {
36public:
37 LockedPrintStream(std::unique_ptr<PrintStream> target);
38 ~LockedPrintStream() final;
39
40 void vprintf(const char* format, va_list) final WTF_ATTRIBUTE_PRINTF(2, 0);
41 void flush() final;
42
43private:
44 PrintStream& begin() final;
45 void end() final;
46
47 // This needs to be a recursive lock because a printInternal or dump method could assert,
48 // and that assert might want to log. Better to let it. This needs to be a WordLock so that
49 // LockedPrintStream (i.e. cataLog) can be used to debug ParkingLot and Lock.
50 RecursiveLockAdapter<WordLock> m_lock;
51 std::unique_ptr<PrintStream> m_target;
52};
53
54} // namespace WTF
55
56using WTF::LockedPrintStream;
