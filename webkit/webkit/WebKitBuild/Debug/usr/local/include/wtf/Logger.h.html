39template<typename T>
40struct LogArgument {
41 template<typename U = T> static typename std::enable_if<std::is_same<U, bool>::value, String>::type toString(bool argument) { return argument ? "true"_s : "false"_s; }
42 template<typename U = T> static typename std::enable_if<std::is_same<U, int>::value, String>::type toString(int argument) { return String::number(argument); }
43 template<typename U = T> static typename std::enable_if<std::is_same<U, unsigned>::value, String>::type toString(unsigned argument) { return String::number(argument); }
44 template<typename U = T> static typename std::enable_if<std::is_same<U, unsigned long>::value, String>::type toString(unsigned long argument) { return String::number(argument); }
45 template<typename U = T> static typename std::enable_if<std::is_same<U, long>::value, String>::type toString(long argument) { return String::number(argument); }
46 template<typename U = T> static typename std::enable_if<std::is_same<U, unsigned long long>::value, String>::type toString(unsigned long long argument) { return String::number(argument); }
47 template<typename U = T> static typename std::enable_if<std::is_same<U, long long>::value, String>::type toString(long long argument) { return String::number(argument); }
48 template<typename U = T> static typename std::enable_if<std::is_enum<U>::value, String>::type toString(U argument) { return String::number(static_cast<typename std::underlying_type<U>::type>(argument)); }
49 template<typename U = T> static typename std::enable_if<std::is_same<U, float>::value, String>::type toString(float argument) { return String::number(argument); }
50 template<typename U = T> static typename std::enable_if<std::is_same<U, double>::value, String>::type toString(double argument) { return String::number(argument); }
51 template<typename U = T> static typename std::enable_if<std::is_same<typename std::remove_reference<U>::type, AtomString>::value, String>::type toString(const AtomString& argument) { return argument.string(); }
52 template<typename U = T> static typename std::enable_if<std::is_same<typename std::remove_reference<U>::type, String>::value, String>::type toString(String argument) { return argument; }
53 template<typename U = T> static typename std::enable_if<std::is_same<typename std::remove_reference<U>::type, StringBuilder*>::value, String>::type toString(StringBuilder* argument) { return argument->toString(); }
54 template<typename U = T> static typename std::enable_if<std::is_same<U, const char*>::value, String>::type toString(const char* argument) { return String(argument); }
55 template<size_t length> static String toString(const char (&argument)[length]) { return String(argument); }
56};

58struct JSONLogValue {
59 enum class Type { String, JSON };
60 Type type { Type::JSON };
61 String value;
62};

64template<class C>
65class HasToJSONString {
66 template <class T> static std::true_type testSignature(String (T::*)() const);
67
68 template <class T> static decltype(testSignature(&T::toJSONString)) test(std::nullptr_t);
69 template <class T> static std::false_type test(...);
70
71public:
72 static constexpr bool value = decltype(test<C>(nullptr))::value;
73};

75template<typename Argument, bool hasJSON = HasToJSONString<Argument>::value>
76struct ConsoleLogValueImpl;

78template<typename Argument>
79struct ConsoleLogValueImpl<Argument, true> {
80 static JSONLogValue toValue(const Argument& value)
81 {
82 return JSONLogValue { JSONLogValue::Type::JSON, value.toJSONString() };
83 }
84};

86template<typename Argument>
87struct ConsoleLogValueImpl<Argument, false> {

238 inline bool willLog(const WTFLogChannel& channel, WTFLogLevel level) const
239 {
240 if (!m_enabled)
241 return false;
242
243#if USE(SYSTEMD)
244 if (channel.state == WTFLogChannelState::Off)
245 return false;
246#endif
247
248 if (level <= WTFLogLevel::Error)
249 return true;
250
251 if (channel.state == WTFLogChannelState::Off || level > channel.level)
252 return false;
253
254 return true;
255 }

258 void setEnabled(const void* owner, bool enabled)
259 {
260 ASSERT(owner == m_owner);
261 if (owner == m_owner)
262 m_enabled = enabled;

286 static inline void addObserver(Observer& observer)
287 {
288 Locker locker { observerLock() };
289 observers().append(observer);
290 }
291 static inline void removeObserver(Observer& observer)
292 {
293 Locker locker { observerLock() };
294 observers().removeFirstMatching([&observer](auto anObserver) {
295 return &anObserver.get() == &observer;
296 });
297 }

307 template<typename... Argument>
308 static inline void log(WTFLogChannel& channel, WTFLogLevel level, const Argument&... arguments)
309 {
310 auto logMessage = makeString(LogArgument<Argument>::toString(arguments)...);
311
312#if RELEASE_LOG_DISABLED
313 WTFLog(&channel, "%s", logMessage.utf8().data());
314#endif
315#if USE(OS_LOG) && !RELEASE_LOG_DISABLED
316 os_log(channel.osLogChannel, "%{public}s", logMessage.utf8().data());
317#endif
318#if USE(JOURNALD) && !RELEASE_LOG_DISABLED
319 sd_journal_send("WEBKIT_SUBSYSTEM=%s", channel.subsystem, "WEBKIT_CHANNEL=%s", channel.name, "MESSAGE=%s", logMessage.utf8().data(), nullptr);
320#endif
321
322 if (channel.state == WTFLogChannelState::Off || level > channel.level)
323 return;
324
325 if (!observerLock().tryLock())
326 return;
327
328 Locker locker { AdoptLock, observerLock() };
329 for (Observer& observer : observers())
330 observer.didLogMessage(channel, level, { ConsoleLogValue<Argument>::toValue(arguments)... });
331 }
332
333 template<typename... Argument>
334 static inline void logVerbose(WTFLogChannel& channel, WTFLogLevel level, const char* file, const char* function, int line, const Argument&... arguments)
335 {
336 auto logMessage = makeString(LogArgument<Argument>::toString(arguments)...);
337
338#if RELEASE_LOG_DISABLED
339 WTFLogVerbose(file, line, function, &channel, "%s", logMessage.utf8().data());
340#endif
341#if USE(OS_LOG) && !RELEASE_LOG_DISABLED
342 os_log(channel.osLogChannel, "%{public}s", logMessage.utf8().data());
343 UNUSED_PARAM(file);
344 UNUSED_PARAM(line);
345 UNUSED_PARAM(function);
346#endif
347#if USE(JOURNALD) && !RELEASE_LOG_DISABLED
348 auto fileString = makeString("CODE_FILE=", file);
349 auto lineString = makeString("CODE_LINE=", line);
350 sd_journal_send_with_location(fileString.utf8().data(), lineString.utf8().data(), function, "WEBKIT_SUBSYSTEM=%s", channel.subsystem, "WEBKIT_CHANNEL=%s", channel.name, "MESSAGE=%s", logMessage.utf8().data(), nullptr);
351#endif
352
353 if (channel.state == WTFLogChannelState::Off || level > channel.level)
354 return;
355
356 if (!observerLock().tryLock())
357 return;
358
359 Locker locker { AdoptLock, observerLock() };
360 for (Observer& observer : observers())
361 observer.didLogMessage(channel, level, { ConsoleLogValue<Argument>::toValue(arguments)... });
362 }
363
364 WTF_EXPORT_PRIVATE static Vector<std::reference_wrapper<Observer>>& observers() WTF_REQUIRES_LOCK(observerLock());
365
366 static Lock& observerLock() WTF_RETURNS_LOCK(loggerObserverLock)
367 {
368 return loggerObserverLock;
369 }
370

372 bool m_enabled { true };
373 const void* m_owner;
374};
