28#include <wtf/GlobalVersion.h>
29#include <wtf/Noncopyable.h>
30#include <wtf/PrintStream.h>

32namespace WTF {
33
34class LoggingHashID {
35 WTF_MAKE_FAST_ALLOCATED;
36 WTF_MAKE_NONCOPYABLE(LoggingHashID);
37
38public:
39 LoggingHashID()
40 : m_version(newGlobalVersion())
41 {
42 }
43
44 void dump(PrintStream& out) const
45 {
46 out.print("_", m_version);
47 }
48private:
49 GlobalVersion m_version;
50};
