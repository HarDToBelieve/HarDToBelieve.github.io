40namespace WTF {

42#define ENABLE_META_ALLOCATOR_PROFILE 0
43
44class MetaAllocatorTracker {
45 WTF_MAKE_FAST_ALLOCATED;
46public:
47 void notify(MetaAllocatorHandle&);
48 void release(MetaAllocatorHandle&);
49
50 MetaAllocatorHandle* find(void* address)
51 {
52 MetaAllocatorHandle* handle = m_allocations.findGreatestLessThanOrEqual(address);
53 if (handle && handle->start().untaggedPtr() <= address && address < handle->end().untaggedPtr())
54 return handle;
55 return nullptr;
56 }
57
58 RedBlackTree<MetaAllocatorHandle, void*> m_allocations;
59};
60
61class MetaAllocator {
62 WTF_MAKE_NONCOPYABLE(MetaAllocator);
63
64public:
65 using FreeSpacePtr = MetaAllocatorPtr<FreeSpacePtrTag>;
66 using MemoryPtr = MetaAllocatorHandle::MemoryPtr;
67
68 WTF_EXPORT_PRIVATE MetaAllocator(Lock&, size_t allocationGranule, size_t pageSize = WTF::pageSize());
69
70 WTF_EXPORT_PRIVATE virtual ~MetaAllocator();
71
72 ALWAYS_INLINE RefPtr<MetaAllocatorHandle> allocate(size_t sizeInBytes)
73 {
74 Locker locker { m_lock };
75 return allocate(locker, sizeInBytes);
76 }
77 WTF_EXPORT_PRIVATE RefPtr<MetaAllocatorHandle> allocate(const LockHolder&, size_t sizeInBytes);
78
79 void trackAllocations(MetaAllocatorTracker* tracker)
80 {
81 m_tracker = tracker;
82 }
83
84 // Non-atomic methods for getting allocator statistics.
85 size_t bytesAllocated() { return m_bytesAllocated; }
86 size_t bytesReserved() { return m_bytesReserved; }
87 size_t bytesCommitted() { return m_bytesCommitted; }
88
89 // Atomic method for getting allocator statistics.
90 struct Statistics {
91 WTF_MAKE_STRUCT_FAST_ALLOCATED;
92 size_t bytesAllocated;
93 size_t bytesReserved;
94 size_t bytesCommitted;
95 };
96 Statistics currentStatistics()
97 {
98 Locker locker { m_lock };
99 return currentStatistics(locker);
100 }
101 WTF_EXPORT_PRIVATE Statistics currentStatistics(const LockHolder&);
102
103 // Add more free space to the allocator. Call this directly from
104 // the constructor if you wish to operate the allocator within a
105 // fixed pool.
106 WTF_EXPORT_PRIVATE void addFreshFreeSpace(void* start, size_t sizeInBytes);
107
108 // This is meant only for implementing tests. Never call this in release
109 // builds.
110 WTF_EXPORT_PRIVATE size_t debugFreeSpaceSize();
111
112 WTF_EXPORT_PRIVATE bool isInAllocatedMemory(const AbstractLocker&, void* address);
113
114#if ENABLE(META_ALLOCATOR_PROFILE)
115 void dumpProfile();
116#else
117 void dumpProfile() { }
118#endif
119
120protected:
121
122 // Allocate new virtual space, but don't commit. This may return more
123 // pages than we asked, in which case numPages is changed.
124 virtual FreeSpacePtr allocateNewSpace(size_t& numPages) = 0;
125
126 // Commit a page.
127 virtual void notifyNeedPage(void* page, size_t) = 0;
128
129 // Uncommit a page.
130 virtual void notifyPageIsFree(void* page, size_t) = 0;
131
132 // NOTE: none of the above methods are called during allocator
133 // destruction, in part because a MetaAllocator cannot die so long
134 // as there are Handles that refer to it.
135
136 // Release a MetaAllocatorHandle.
137 WTF_EXPORT_PRIVATE virtual void release(const LockHolder&, MetaAllocatorHandle&);
138private:
139
140 friend class MetaAllocatorHandle;
141
142 class FreeSpaceNode : public RedBlackTree<FreeSpaceNode, size_t>::Node {
143 public:
144 FreeSpaceNode() = default;
145
146 size_t sizeInBytes()
147 {
148 return m_end.untaggedPtr<size_t>() - m_start.untaggedPtr<size_t>();
149 }
150
151 size_t key()
152 {
153 return sizeInBytes();
154 }
155
156 FreeSpacePtr m_start;
157 FreeSpacePtr m_end;
158 };
159 typedef RedBlackTree<FreeSpaceNode, size_t> Tree;
160
161 // Remove free space from the allocator. This is effectively
162 // the allocate() function, except that it does not mark the
163 // returned space as being in-use.
164 FreeSpacePtr findAndRemoveFreeSpace(size_t sizeInBytes);
165
166 // This is called when memory from an allocation is freed.
167 void addFreeSpaceFromReleasedHandle(FreeSpacePtr start, size_t sizeInBytes);
168
169 // This is the low-level implementation of adding free space; it
170 // is called from both addFreeSpaceFromReleasedHandle and from
171 // addFreshFreeSpace.
172 void addFreeSpace(FreeSpacePtr start, size_t sizeInBytes);
173
174 // Management of used space.
175
176 void incrementPageOccupancy(void* address, size_t sizeInBytes);
177 void decrementPageOccupancy(void* address, size_t sizeInBytes);
178
179 // Utilities.
180
181 size_t roundUp(size_t sizeInBytes);
182
183 FreeSpaceNode* allocFreeSpaceNode();
184 WTF_EXPORT_PRIVATE void freeFreeSpaceNode(FreeSpaceNode*);
185
186 size_t m_allocationGranule;
187 size_t m_pageSize;
188 unsigned m_logAllocationGranule;
189 unsigned m_logPageSize;
190
191 Tree m_freeSpaceSizeMap;
192 HashMap<FreeSpacePtr, FreeSpaceNode*> m_freeSpaceStartAddressMap;
193 HashMap<FreeSpacePtr, FreeSpaceNode*> m_freeSpaceEndAddressMap;
194 HashMap<uintptr_t, size_t> m_pageOccupancyMap;
195
196 size_t m_bytesAllocated;
197 size_t m_bytesReserved;
198 size_t m_bytesCommitted;
199
200 Lock& m_lock;
201
202 MetaAllocatorTracker* m_tracker { nullptr };
203
204#ifndef NDEBUG
205 size_t m_mallocBalance;
206#endif
207
208#if ENABLE(META_ALLOCATOR_PROFILE)
209 unsigned m_numAllocations;
210 unsigned m_numFrees;
211#endif
212};
