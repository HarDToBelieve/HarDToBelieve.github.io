29#pragma once
30
31#include <wtf/Assertions.h>
32#include <wtf/MetaAllocatorPtr.h>
33#include <wtf/RedBlackTree.h>
34#include <wtf/ThreadSafeRefCounted.h>
35
36namespace WTF {
37
38class MetaAllocator;
39class PrintStream;
40
41DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(MetaAllocatorHandle);
42class MetaAllocatorHandle : public ThreadSafeRefCounted<MetaAllocatorHandle>, public RedBlackTree<MetaAllocatorHandle, void*>::Node {
43 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(MetaAllocatorHandle);
44
45public:
46 using MemoryPtr = MetaAllocatorPtr<HandleMemoryPtrTag>;
47
48 WTF_EXPORT_PRIVATE ~MetaAllocatorHandle();
49
50 MemoryPtr start() const
51 {
52 return m_start;
53 }
54
55 MemoryPtr end() const
56 {
57 return m_end;
58 }
59
60 uintptr_t startAsInteger() const
61 {
62 return m_start.untaggedPtr<uintptr_t>();
63 }
64
65 uintptr_t endAsInteger() const
66 {
67 return m_end.untaggedPtr<uintptr_t>();
68 }
69
70 size_t sizeInBytes() const
71 {
72 return m_end.untaggedPtr<size_t>() - m_start.untaggedPtr<size_t>();
73 }
74
75 bool containsIntegerAddress(uintptr_t address) const
76 {
77 return address >= startAsInteger() && address < endAsInteger();
78 }
79
80 bool contains(void* address) const
81 {
82 return containsIntegerAddress(reinterpret_cast<uintptr_t>(address));
83 }
84
85 WTF_EXPORT_PRIVATE void shrink(size_t newSizeInBytes);
86
87 MetaAllocator& allocator()
88 {
89 return m_allocator;
90 }
91
92 void* key()
93 {
94 return m_start.untaggedPtr();
95 }
96
97 WTF_EXPORT_PRIVATE void dump(PrintStream& out) const;
98
99private:
100 MetaAllocatorHandle(MetaAllocator&, MemoryPtr start, size_t sizeInBytes);
101
102 MetaAllocator& m_allocator;
103 MemoryPtr m_start;
104 MemoryPtr m_end;
105
106 friend class MetaAllocator;
107};
108
109}
