26#pragma once
27
28#include <wtf/HashTraits.h>
29#include <wtf/PtrTag.h>
30#include <wtf/StdLibExtras.h>
31
32namespace WTF {
33
34template<PtrTag tag>
35class MetaAllocatorPtr {
36public:
37 MetaAllocatorPtr() = default;
38 MetaAllocatorPtr(std::nullptr_t) { }
39
40 static MetaAllocatorPtr makeFromRawPointer(void* ptr) { return MetaAllocatorPtr(ptr); }
41
42 template<PtrTag otherTag>
43 explicit MetaAllocatorPtr(MetaAllocatorPtr<otherTag> other)
44 : m_ptr(other.template retaggedPtr<tag>())
45 { }
46
47 explicit MetaAllocatorPtr(uintptr_t ptrAsInt)
48 : MetaAllocatorPtr(reinterpret_cast<void*>(ptrAsInt))
49 { }
50
51 template<typename T = void*>
52 T untaggedPtr() const { return bitwise_cast<T>(untagCodePtr<tag>(m_ptr)); }
53
54 template<PtrTag newTag, typename T = void*>
55 T retaggedPtr() const { return bitwise_cast<T>(retagCodePtr<tag, newTag>(m_ptr)); }
56
57 // Disallow any casting operations (except for booleans).
58 template<typename T, typename = std::enable_if_t<!std::is_same<T, bool>::value>>
59 operator T() = delete;
60
61 explicit operator bool() const { return !!m_ptr; }
62 bool operator!() const { return !m_ptr; }
63
64 bool operator==(MetaAllocatorPtr other) const { return m_ptr == other.m_ptr; }
65 bool operator!=(MetaAllocatorPtr other) const { return m_ptr != other.m_ptr; }
66
67 MetaAllocatorPtr operator+(size_t sizeInBytes) const { return MetaAllocatorPtr(untaggedPtr<uint8_t*>() + sizeInBytes); }
68 MetaAllocatorPtr operator-(size_t sizeInBytes) const { return MetaAllocatorPtr(untaggedPtr<uint8_t*>() - sizeInBytes); }
69
70 MetaAllocatorPtr& operator+=(size_t sizeInBytes)
71 {
72 return *this = *this + sizeInBytes;
73 }
74
75 MetaAllocatorPtr& operator-=(size_t sizeInBytes)
76 {
77 return *this = *this - sizeInBytes;
78 }
79
80 enum EmptyValueTag { EmptyValue };
81 enum DeletedValueTag { DeletedValue };
82
83 MetaAllocatorPtr(EmptyValueTag)
84 : m_ptr(emptyValue())
85 { }
86
87 MetaAllocatorPtr(DeletedValueTag)
88 : m_ptr(deletedValue())
89 { }
90
91 bool isEmptyValue() const { return m_ptr == emptyValue(); }
92 bool isDeletedValue() const { return m_ptr == deletedValue(); }
93
94 unsigned hash() const { return PtrHash<void*>::hash(m_ptr); }
95
96private:
97 explicit MetaAllocatorPtr(void* ptr)
98 : m_ptr(tagCodePtr<tag>(ptr))
99 {
100 assertIsNotTagged(ptr);
101 }
102
103 static void* emptyValue() { return reinterpret_cast<void*>(1); }
104 static void* deletedValue() { return reinterpret_cast<void*>(2); }
105
106 void* m_ptr { nullptr };
107};
108
109template<PtrTag tag>
110struct MetaAllocatorPtrHash {
111 static unsigned hash(const MetaAllocatorPtr<tag>& ptr) { return ptr.hash(); }
112 static bool equal(const MetaAllocatorPtr<tag>& a, const MetaAllocatorPtr<tag>& b)
113 {
114 return a == b;
115 }
116 static constexpr bool safeToCompareToEmptyOrDeleted = true;
117};
118
119template<typename> struct DefaultHash;
120template<PtrTag tag> struct DefaultHash<MetaAllocatorPtr<tag>> : MetaAllocatorPtrHash<tag> { };
121
122template<PtrTag tag> struct HashTraits<MetaAllocatorPtr<tag>> : public CustomHashTraits<MetaAllocatorPtr<tag>> { };

124} // namespace WTF
