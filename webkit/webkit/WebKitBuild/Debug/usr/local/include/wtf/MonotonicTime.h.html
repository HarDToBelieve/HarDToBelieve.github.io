26#pragma once
27
28#include <wtf/ClockType.h>
29#include <wtf/Seconds.h>
30
31namespace WTF {
32
33class WallTime;
34class PrintStream;
35
36// The current time according to a monotonic clock. Monotonic clocks don't go backwards and
37// possibly don't count downtime. This uses floating point internally so that you can reason about
38// infinity and other things that arise in math. It's acceptable to use this to wrap NaN times,
39// negative times, and infinite times, so long as they are all relative to the same clock.
40class MonotonicTime final {
41 WTF_MAKE_FAST_ALLOCATED;
42public:
43 static constexpr ClockType clockType = ClockType::Monotonic;
44
45 // This is the epoch. So, x.secondsSinceEpoch() should be the same as x - MonotonicTime().
46 constexpr MonotonicTime() { }
47
48 // Call this if you know for sure that the double represents monotonic time according to the
49 // same time source as MonotonicTime. It must be in seconds.
50 static constexpr MonotonicTime fromRawSeconds(double value)
51 {
52 return MonotonicTime(value);
53 }
54
55#if OS(DARWIN)
56 WTF_EXPORT_PRIVATE static MonotonicTime fromMachAbsoluteTime(uint64_t);
57#endif
58
59 WTF_EXPORT_PRIVATE static MonotonicTime now();
60
61 static constexpr MonotonicTime infinity() { return fromRawSeconds(std::numeric_limits<double>::infinity()); }
62 static constexpr MonotonicTime nan() { return fromRawSeconds(std::numeric_limits<double>::quiet_NaN()); }
63
64 constexpr Seconds secondsSinceEpoch() const { return Seconds(m_value); }
65
66 MonotonicTime approximateMonotonicTime() const { return *this; }
67 WTF_EXPORT_PRIVATE WallTime approximateWallTime() const;
68
69 explicit constexpr operator bool() const { return !!m_value; }
70
71 constexpr MonotonicTime operator+(Seconds other) const
72 {
73 return fromRawSeconds(m_value + other.value());
74 }
75
76 constexpr MonotonicTime operator-(Seconds other) const
77 {
78 return fromRawSeconds(m_value - other.value());
79 }
80
81 Seconds operator%(Seconds other) const
82 {
83 return Seconds { fmod(m_value, other.value()) };
84 }
85
86 // Time is a scalar and scalars can be negated as this could arise from algebraic
87 // transformations. So, we allow it.
88 constexpr MonotonicTime operator-() const
89 {
90 return fromRawSeconds(-m_value);
91 }
92
93 MonotonicTime operator+=(Seconds other)
94 {
95 return *this = *this + other;
96 }
97
98 MonotonicTime operator-=(Seconds other)
99 {
100 return *this = *this - other;
101 }
102
103 constexpr Seconds operator-(MonotonicTime other) const
104 {
105 return Seconds(m_value - other.m_value);
106 }
107
108 constexpr bool operator==(MonotonicTime other) const
109 {
110 return m_value == other.m_value;
111 }
112
113 constexpr bool operator!=(MonotonicTime other) const
114 {
115 return m_value != other.m_value;
116 }
117
118 constexpr bool operator<(MonotonicTime other) const
119 {
120 return m_value < other.m_value;
121 }
122
123 constexpr bool operator>(MonotonicTime other) const
124 {
125 return m_value > other.m_value;
126 }
127
128 constexpr bool operator<=(MonotonicTime other) const
129 {
130 return m_value <= other.m_value;
131 }
132
133 constexpr bool operator>=(MonotonicTime other) const
134 {
135 return m_value >= other.m_value;
136 }
137
138 WTF_EXPORT_PRIVATE void dump(PrintStream&) const;
139
140 MonotonicTime isolatedCopy() const
141 {
142 return *this;
143 }
144
145 template<class Encoder>
146 void encode(Encoder& encoder) const
147 {
148 encoder << m_value;
149 }
150
151 template<class Decoder>
152 static std::optional<MonotonicTime> decode(Decoder& decoder)
153 {
154 std::optional<double> time;
155 decoder >> time;
156 if (!time)
157 return std::nullopt;
158 return MonotonicTime::fromRawSeconds(*time);
159 }
160
161 template<class Decoder>
162 static WARN_UNUSED_RETURN bool decode(Decoder& decoder, MonotonicTime& time)
163 {
164 double value;
165 if (!decoder.decode(value))
166 return false;
167
168 time = MonotonicTime::fromRawSeconds(value);
169 return true;
170 }
171
172 struct MarkableTraits;
173
174private:
175 constexpr MonotonicTime(double rawValue)
176 : m_value(rawValue)
177 {
178 }
179
180 double m_value { 0 };
181};
182
183struct MonotonicTime::MarkableTraits {
184 static bool isEmptyValue(MonotonicTime time)
185 {
186 return std::isnan(time.m_value);
187 }
188
189 static constexpr MonotonicTime emptyValue()
190 {
191 return MonotonicTime::nan();
192 }
193};
194
195} // namespace WTF

197namespace std {
198
199inline bool isnan(WTF::MonotonicTime time)
200{
201 return std::isnan(time.secondsSinceEpoch().value());
202}
203
204inline bool isinf(WTF::MonotonicTime time)
205{
206 return std::isinf(time.secondsSinceEpoch().value());
207}
208
209inline bool isfinite(WTF::MonotonicTime time)
210{
211 return std::isfinite(time.secondsSinceEpoch().value());
212}
213
214} // namespace std
