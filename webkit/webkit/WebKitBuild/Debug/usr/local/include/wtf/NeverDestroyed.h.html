34// NeverDestroyed is a smart-pointer-like class that ensures that the destructor
35// for the given object is never called, but doesn't use the heap to allocate it.
36// It's useful for static local variables, and can be used like so:
37//
38// MySharedGlobal& mySharedGlobal()
39// {
40// static NeverDestroyed<MySharedGlobal> myGlobal("Hello", 42);
41// return myGlobal;
42// }
43
44namespace WTF {
45
46struct AnyThreadsAccessTraits {
47 static void assertAccess()
48 {
49 }
50};
51
52struct MainThreadAccessTraits {
53 static void assertAccess()
54 {
55 ASSERT(isMainThread());
56 }
57};
58
59template<typename T, typename AccessTraits> class NeverDestroyed {
60 WTF_MAKE_NONCOPYABLE(NeverDestroyed);
61 WTF_FORBID_HEAP_ALLOCATION;
62public:
63
64 template<typename... Args> NeverDestroyed(Args&&... args)
65 {
66 AccessTraits::assertAccess();
67 MaybeRelax<T>(new (storagePointer()) T(std::forward<Args>(args)...));
68 }
69
70 NeverDestroyed(NeverDestroyed&& other)
71 {
72 AccessTraits::assertAccess();
73 MaybeRelax<T>(new (storagePointer()) T(WTFMove(*other.storagePointer())));
74 }
75
76 operator T&() { return *storagePointer(); }
77 T& get() { return *storagePointer(); }
78
79 T* operator->() { return storagePointer(); }
80
81 operator const T&() const { return *storagePointer(); }
82 const T& get() const { return *storagePointer(); }
83
84 const T* operator->() const { return storagePointer(); }
85
86private:
87 using PointerType = typename std::remove_const<T>::type*;
88
89 PointerType storagePointer() const
90 {
91 AccessTraits::assertAccess();
92 return const_cast<PointerType>(reinterpret_cast<const T*>(&m_storage));
93 }
94
95 template<typename PtrType, bool ShouldRelax = std::is_base_of<RefCountedBase, PtrType>::value> struct MaybeRelax {
96 explicit MaybeRelax(PtrType*) { }
97 };
98 template<typename PtrType> struct MaybeRelax<PtrType, true> {
99 explicit MaybeRelax(PtrType* ptr) { ptr->relaxAdoptionRequirement(); }
100 };
101
102 // FIXME: Investigate whether we should allocate a hunk of virtual memory
103 // and hand out chunks of it to NeverDestroyed instead, to reduce fragmentation.
104 typename std::aligned_storage<sizeof(T), std::alignment_of<T>::value>::type m_storage;
105};
106
107template<typename T, typename AccessTraits = AnyThreadsAccessTraits> NeverDestroyed<T, AccessTraits> makeNeverDestroyed(T&&);
108
109// FIXME: It's messy to have to repeat the whole class just to make this "lazy" version.
110// Should revisit clients to see if we really need this, and perhaps use templates to
111// share more of the code with the main NeverDestroyed above.
112template<typename T, typename AccessTraits> class LazyNeverDestroyed {
113 WTF_MAKE_NONCOPYABLE(LazyNeverDestroyed);
114 WTF_FORBID_HEAP_ALLOCATION;
115public:
116 LazyNeverDestroyed() = default;
117
118 template<typename... Args>
119 void construct(Args&&... args)
120 {
121 AccessTraits::assertAccess();
122 constructWithoutAccessCheck(std::forward<Args>(args)...);
123 }
124
125 template<typename... Args>
126 void constructWithoutAccessCheck(Args&&... args)
127 {
128 ASSERT(!m_isConstructed);
129#if ASSERT_ENABLED
130 m_isConstructed = true;
131#endif
132 MaybeRelax<T>(new (storagePointerWithoutAccessCheck()) T(std::forward<Args>(args)...));
133 }
134
135 operator T&() { return *storagePointer(); }
136 T& get() { return *storagePointer(); }
137
138 T* operator->() { return storagePointer(); }
139
140 operator const T&() const { return *storagePointer(); }
141 const T& get() const { return *storagePointer(); }
142
143 const T* operator->() const { return storagePointer(); }
144
145#if ASSERT_ENABLED
146 bool isConstructed() const { return m_isConstructed; }
147#endif
148
149private:
150 using PointerType = typename std::remove_const<T>::type*;
151
152 PointerType storagePointerWithoutAccessCheck() const
153 {
154 ASSERT(m_isConstructed);
155 return const_cast<PointerType>(reinterpret_cast<const T*>(&m_storage));
156 }
157
158 PointerType storagePointer() const
159 {
160 AccessTraits::assertAccess();
161 return storagePointerWithoutAccessCheck();
162 }
163
164 template<typename PtrType, bool ShouldRelax = std::is_base_of<RefCountedBase, PtrType>::value> struct MaybeRelax {
165 explicit MaybeRelax(PtrType*) { }
166 };
167 template<typename PtrType> struct MaybeRelax<PtrType, true> {
168 explicit MaybeRelax(PtrType* ptr) { ptr->relaxAdoptionRequirement(); }
169 };
170
171#if ASSERT_ENABLED
172 // LazyNeverDestroyed objects are always static, so this variable is initialized to false.
173 // It must not be initialized dynamically; that would not be thread safe.
174 bool m_isConstructed;
175#endif
176
177 // FIXME: Investigate whether we should allocate a hunk of virtual memory
178 // and hand out chunks of it to NeverDestroyed instead, to reduce fragmentation.
179 typename std::aligned_storage<sizeof(T), std::alignment_of<T>::value>::type m_storage;
180};
181
182template<typename T, typename AccessTraits> inline NeverDestroyed<T, AccessTraits> makeNeverDestroyed(T&& argument)
183{
184 return std::forward<T>(argument);
185}
186
187template<typename T>
188using MainThreadNeverDestroyed = NeverDestroyed<T, MainThreadAccessTraits>;
189
190template<typename T>
191using MainThreadLazyNeverDestroyed = LazyNeverDestroyed<T, MainThreadAccessTraits>;
