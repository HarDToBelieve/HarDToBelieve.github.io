42template<typename T, typename E> struct OptionSetValueChecker;
43
44template<typename T, typename E, E e, E... es>
45struct OptionSetValueChecker<T, EnumValues<E, e, es...>> {
46 static constexpr T allValidBits()
47 {
48 return static_cast<T>(e) | OptionSetValueChecker<T, EnumValues<E, es...>>::allValidBits();
49 }
50
51 static constexpr bool isValidOptionSetEnum(T t)
52 {
53 return (static_cast<T>(e) == t) ? true : OptionSetValueChecker<T, EnumValues<E, es...>>::isValidOptionSetEnum(t);
54 }
55};
56
57template<typename T, typename E>
58struct OptionSetValueChecker<T, EnumValues<E>> {
59 static constexpr T allValidBits()
60 {
61 return 0;
62 }
63
64 static constexpr bool isValidOptionSetEnum(T)
65 {
66 return false;
67 }
68};

71template<typename E, std::enable_if_t<std::is_enum<E>::value && IsTypeComplete<EnumTraits<E>>>* = nullptr>
72constexpr bool isValidOptionSetEnum(E e)
73{
74 return OptionSetValueChecker<std::underlying_type_t<E>, typename EnumTraits<E>::values>::isValidOptionSetEnum(static_cast<std::underlying_type_t<E>>(e));
75}
76
77template<typename E, std::enable_if_t<std::is_enum<E>::value && !IsTypeComplete<EnumTraits<E>>>* = nullptr>
78constexpr bool isValidOptionSetEnum(E e)
79{
80 // FIXME: Remove once all OptionSet<> enums have EnumTraits<> defined.
81 return hasOneBitSet(static_cast<typename OptionSet<E>::StorageType>(e));
82}

85template<typename E, std::enable_if_t<std::is_enum<E>::value && IsTypeComplete<EnumTraits<E>>>* = nullptr>
86constexpr typename OptionSet<E>::StorageType maskRawValue(typename OptionSet<E>::StorageType rawValue)
87{
88 auto allValidBitsValue = OptionSetValueChecker<std::underlying_type_t<E>, typename EnumTraits<E>::values>::allValidBits();
89 return rawValue & allValidBitsValue;
90}
91
92template<typename E, std::enable_if_t<std::is_enum<E>::value && !IsTypeComplete<EnumTraits<E>>>* = nullptr>
93constexpr typename OptionSet<E>::StorageType maskRawValue(typename OptionSet<E>::StorageType rawValue)
94{
95 // FIXME: Remove once all OptionSet<> enums have EnumTraits<> defined.
96 return rawValue;
97}

100// OptionSet is a class that represents a set of enumerators in a space-efficient manner. The enumerators
101// must be powers of two greater than 0. This class is useful as a replacement for passing a bitmask of
102// enumerators around.
103template<typename E> class OptionSet {
104 WTF_MAKE_FAST_ALLOCATED;
105 static_assert(std::is_enum<E>::value, "T is not an enum type");
106
107public:
108 using StorageType = std::make_unsigned_t<std::underlying_type_t<E>>;
109
110 template<typename StorageType> class Iterator {
111 WTF_MAKE_FAST_ALLOCATED;
112 public:
113 // Isolate the rightmost set bit.
114 E operator*() const { return static_cast<E>(m_value & -m_value); }
115
116 // Iterates from smallest to largest enum value by turning off the rightmost set bit.
117 Iterator& operator++()
118 {
119 m_value &= m_value - 1;
120 return *this;
121 }
122
123 Iterator& operator++(int) = delete;
124
125 bool operator==(const Iterator& other) const { return m_value == other.m_value; }
126 bool operator!=(const Iterator& other) const { return m_value != other.m_value; }
127
128 private:
129 Iterator(StorageType value) : m_value(value) { }
130 friend OptionSet;
131
132 StorageType m_value;
133 };
134 using iterator = Iterator<StorageType>;
135
136 static constexpr OptionSet fromRaw(StorageType rawValue)
137 {
138 return OptionSet(static_cast<E>(maskRawValue<E>(rawValue)), FromRawValue);
139 }
140
141 constexpr OptionSet() = default;
142
143 constexpr OptionSet(E e)
144 : m_storage(static_cast<StorageType>(e))
145 {
146 ASSERT(!m_storage || isValidOptionSetEnum(e));
147 }
148
149 constexpr OptionSet(std::initializer_list<E> initializerList)
150 {
151 for (auto& option : initializerList) {
152 ASSERT(isValidOptionSetEnum(option));
153 m_storage |= static_cast<StorageType>(option);
154 }
155 }
156
157 constexpr OptionSet(std::optional<E> optional)
158 : m_storage(optional ? static_cast<StorageType>(*optional) : 0)
159 {
160 }
161
162 constexpr StorageType toRaw() const { return m_storage; }
163
164 constexpr bool isEmpty() const { return !m_storage; }
165
166 constexpr iterator begin() const { return m_storage; }
167 constexpr iterator end() const { return 0; }
168
169 constexpr explicit operator bool() { return !isEmpty(); }
170
171 constexpr bool contains(E option) const
172 {
173 return containsAny(option);
174 }
175
176 constexpr bool containsAny(OptionSet optionSet) const
177 {
178 return !!(*this & optionSet);
179 }
180
181 constexpr bool containsAll(OptionSet optionSet) const
182 {
183 return (*this & optionSet) == optionSet;
184 }
185
186 constexpr void add(OptionSet optionSet)
187 {
188 m_storage |= optionSet.m_storage;
189 }
190
191 constexpr void remove(OptionSet optionSet)
192 {
193 m_storage &= ~optionSet.m_storage;
194 }
195
196 constexpr void set(OptionSet optionSet, bool value)
197 {
198 if (value)
199 add(optionSet);
200 else
201 remove(optionSet);
202 }
203
204 constexpr bool hasExactlyOneBitSet() const
205 {
206 return m_storage && !(m_storage & (m_storage - 1));
207 }
208
209 constexpr std::optional<E> toSingleValue() const
210 {
211 return hasExactlyOneBitSet() ? std::optional<E>(static_cast<E>(m_storage)) : std::nullopt;
212 }
213
214 constexpr friend bool operator==(OptionSet lhs, OptionSet rhs)
215 {
216 return lhs.m_storage == rhs.m_storage;
217 }
218
219 constexpr friend bool operator!=(OptionSet lhs, OptionSet rhs)
220 {
221 return lhs.m_storage != rhs.m_storage;
222 }
223
224 constexpr friend OptionSet operator|(OptionSet lhs, OptionSet rhs)
225 {
226 return fromRaw(lhs.m_storage | rhs.m_storage);
227 }
228
229 constexpr friend OptionSet operator&(OptionSet lhs, OptionSet rhs)
230 {
231 return fromRaw(lhs.m_storage & rhs.m_storage);
232 }
233
234 constexpr friend OptionSet operator-(OptionSet lhs, OptionSet rhs)
235 {
236 return fromRaw(lhs.m_storage & ~rhs.m_storage);
237 }
238
239 constexpr friend OptionSet operator^(OptionSet lhs, OptionSet rhs)
240 {
241 return fromRaw(lhs.m_storage ^ rhs.m_storage);
242 }
243
244private:
245 enum InitializationTag { FromRawValue };
246 constexpr OptionSet(E e, InitializationTag)
247 : m_storage(static_cast<StorageType>(e))
248 {
249 }
250 StorageType m_storage { 0 };
251};
252
253template<typename E>
254WARN_UNUSED_RETURN constexpr bool isValidOptionSet(OptionSet<E> optionSet)
255{
256 auto allValidBitsValue = OptionSetValueChecker<std::make_unsigned_t<std::underlying_type_t<E>>, typename EnumTraits<E>::values>::allValidBits();
257 return (optionSet.toRaw() | allValidBitsValue) == allValidBitsValue;
258}
