242#if WTF_DEFAULT_EVENT_LOOP
243#if USE(GLIB)
244/* Use GLib's event loop abstraction. Primarily GTK port uses it. */
245#define USE_GLIB_EVENT_LOOP 1
246#elif OS(WINDOWS)
247/* Use Windows message pump abstraction.
248 * Even if the port is AppleWin, we use the Windows message pump system for the event loop,
249 * so that USE(WINDOWS_EVENT_LOOP) && USE(CF) can be true.
250 * And PLATFORM(WIN) and PLATFORM(GTK) are exclusive. If the port is GTK,
251 * PLATFORM(WIN) should be false. And in that case, GLib's event loop is used.
252 */
253#define USE_WINDOWS_EVENT_LOOP 1
254#elif PLATFORM(COCOA)
255/* OS X and IOS. Use CoreFoundation & GCD abstraction. */
256#define USE_COCOA_EVENT_LOOP 1
257#else
258#define USE_GENERIC_EVENT_LOOP 1
259#endif
260#endif
