26#pragma once
27
28#include <cstdint>
29#include <utility>
30
31namespace WTF {
32
33template<typename T>
34struct RawPtrTraits {
35 template<typename U> using RebindTraits = RawPtrTraits<U>;
36
37 using StorageType = T*;
38
39 template<typename U>
40 static ALWAYS_INLINE T* exchange(StorageType& ptr, U&& newValue) { return std::exchange(ptr, newValue); }
41
42 static ALWAYS_INLINE void swap(StorageType& a, StorageType& b) { std::swap(a, b); }
43 static ALWAYS_INLINE T* unwrap(const StorageType& ptr) { return ptr; }
44
45 static StorageType hashTableDeletedValue() { return bitwise_cast<StorageType>(static_cast<uintptr_t>(-1)); }
46 static ALWAYS_INLINE bool isHashTableDeletedValue(const StorageType& ptr) { return ptr == hashTableDeletedValue(); }
47};
48
49} // namespace WTF
50
51using WTF::RawPtrTraits;
