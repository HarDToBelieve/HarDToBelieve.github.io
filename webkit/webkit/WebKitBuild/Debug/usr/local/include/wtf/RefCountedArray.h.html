33// This implements a reference counted array for POD** values, which is optimized for:
34// - An empty array only uses one word.
35// - A copy of the array only uses one word (i.e. assignment means aliasing).
36// - The vector can't grow beyond 2^32-1 elements.
37// - In all other regards this has similar space usage to a Vector.
38//
39// ** This could be modified to support non-POD values quite easily. It just
40// hasn't been, so far, because there has been no need. Moreover, even now,
41// it's used for things that aren't quite POD according to the official
42// defintion, such as JSC::Instruction.
43
44namespace WTF {
45
46DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(RefCountedArray);
47
48template<typename T, typename PtrTraits = RawPtrTraits<T>>
49class RefCountedArray {
50 enum CommonCopyConstructorTag { CommonCopyConstructor };
51
52public:
53 using iterator = T*;
54 using const_iterator = const T*;
55 using reverse_iterator = std::reverse_iterator<iterator>;
56 using const_reverse_iterator = std::reverse_iterator<const_iterator>;
57
58 RefCountedArray() = default;
59
60 RefCountedArray(const RefCountedArray& other)
61 : RefCountedArray(CommonCopyConstructor, other)
62 { }
63
64 template<typename OtherTraits>
65 RefCountedArray(const RefCountedArray<T, OtherTraits>& other)
66 : RefCountedArray(CommonCopyConstructor, other)
67 { }
68
69 RefCountedArray(RefCountedArray&& other)
70 : m_data(PtrTraits::exchange(other.m_data, nullptr))
71 { }
72
73 explicit RefCountedArray(size_t size)
74 {
75 if (!size) {
76 // NOTE: JSC's LowLevelInterpreter relies on this being nullptr when the size is zero.
77 PtrTraits::exchange(m_data, nullptr);
78 return;
79 }
80
81 m_data = allocateUninitializedData(size);
82 VectorTypeOperations<T>::initializeIfNonPOD(begin(), end());
83 }
84
85 template<typename OtherTraits = PtrTraits>
86 RefCountedArray<T, OtherTraits> clone() const
87 {
88 RefCountedArray<T, OtherTraits> result(size());
89 const T* data = this->data();
90 T* resultData = result.data();
91 for (unsigned i = size(); i--;)
92 resultData[i] = data[i];
93 return result;
94 }
95
96 template<size_t inlineCapacity, typename OverflowHandler>
97 explicit RefCountedArray(const Vector<T, inlineCapacity, OverflowHandler>& other)
98 {
99 if (other.isEmpty()) {
100 PtrTraits::exchange(m_data, nullptr);
101 return;
102 }
103
104 T* data = allocateUninitializedData(other.size());
105 m_data = data;
106 VectorTypeOperations<T>::uninitializedCopy(other.begin(), other.end(), data);
107 }
108
109 template<size_t inlineCapacity, typename OverflowHandler>
110 explicit RefCountedArray(Vector<T, inlineCapacity, OverflowHandler>&& other)
111 {
112 Vector<T, inlineCapacity, OverflowHandler> vector(WTFMove(other));
113 if (vector.isEmpty()) {
114 PtrTraits::exchange(m_data, nullptr);
115 return;
116 }
117
118 T* data = allocateUninitializedData(vector.size());
119 m_data = data;
120 for (unsigned index = 0; index < vector.size(); ++index)
121 new (data + index) T(WTFMove(vector[index]));
122 }
123
124 template<typename OtherTraits = PtrTraits>
125 RefCountedArray& operator=(const RefCountedArray<T, OtherTraits>& other)
126 {
127 return assign<OtherTraits>(other);
128 }
129
130 RefCountedArray& operator=(const RefCountedArray& other)
131 {
132 return assign<PtrTraits>(other);
133 }
134
135 template<size_t inlineCapacity, typename OverflowHandler>
136 RefCountedArray& operator=(const Vector<T, inlineCapacity, OverflowHandler>& other)
137 {
138 T* oldData = data();
139 if (other.isEmpty())
140 PtrTraits::exchange(m_data, nullptr);
141 else {
142 T* data = allocateUninitializedData(other.size());
143 m_data = data;
144 VectorTypeOperations<T>::uninitializedCopy(other.begin(), other.end(), data);
145 }
146 if (!oldData)
147 return *this;
148 unsigned refCount = Header::fromPayload(oldData)->refCount - 1;
149 if (refCount) {
150 Header::fromPayload(oldData)->refCount = refCount;
151 return *this;
152 }
153 VectorTypeOperations<T>::destruct(oldData, oldData + Header::fromPayload(oldData)->length);
154
155 RefCountedArrayMalloc::free(Header::fromPayload(oldData));
156 return *this;
157 }
158
159 template<size_t inlineCapacity, typename OverflowHandler>
160 RefCountedArray& operator=(Vector<T, inlineCapacity, OverflowHandler>&& other)
161 {
162 Vector<T, inlineCapacity, OverflowHandler> vector(WTFMove(other));
163 T* oldData = data();
164 if (vector.isEmpty())
165 PtrTraits::exchange(m_data, nullptr);
166 else {
167 T* data = allocateUninitializedData(vector.size());
168 m_data = data;
169 for (unsigned index = 0; index < vector.size(); ++index)
170 new (data + index) T(WTFMove(vector[index]));
171 }
172 if (!oldData)
173 return *this;
174 unsigned refCount = Header::fromPayload(oldData)->refCount - 1;
175 if (refCount) {
176 Header::fromPayload(oldData)->refCount = refCount;
177 return *this;
178 }
179 VectorTypeOperations<T>::destruct(oldData, oldData + Header::fromPayload(oldData)->length);
180
181 RefCountedArrayMalloc::free(Header::fromPayload(oldData));
182 return *this;
183 }
184
185 ~RefCountedArray()
186 {
187 if (!m_data)
188 return;
189 T* data = this->data();
190 unsigned refCount = Header::fromPayload(data)->refCount - 1;
191 if (refCount) {
192 Header::fromPayload(data)->refCount = refCount;
193 return;
194 }
195 VectorTypeOperations<T>::destruct(begin(), end());
196 RefCountedArrayMalloc::free(Header::fromPayload(data));
197 }
198
199 unsigned refCount() const
200 {
201 if (!m_data)
202 return 0;
203 return Header::fromPayload(data())->refCount;
204 }
205
206 size_t size() const
207 {
208 if (!m_data)
209 return 0;
210 return Header::fromPayload(data())->length;
211 }
212
213 bool isEmpty() const { return size() == 0; }
214
215 size_t byteSize() const { return size() * sizeof(T); }
216
217 T* data() { return PtrTraits::unwrap(m_data); }
218 iterator begin() { return data(); }
219 iterator end()
220 {
221 if (!m_data)
222 return 0;
223 T* data = this->data();
224 return data + Header::fromPayload(data)->length;
225 }
226
227 const T* data() const { return const_cast<RefCountedArray*>(this)->data(); }
228 const_iterator begin() const { return const_cast<RefCountedArray*>(this)->begin(); }
229 const_iterator end() const { return const_cast<RefCountedArray*>(this)->end(); }
230
231 reverse_iterator rbegin() { return reverse_iterator(end()); }
232 reverse_iterator rend() { return reverse_iterator(begin()); }
233 const_reverse_iterator rbegin() const { return const_reverse_iterator(end()); }
234 const_reverse_iterator rend() const { return const_reverse_iterator(begin()); }
235
236 T& at(size_t i)
237 {
238 ASSERT_WITH_SECURITY_IMPLICATION(i < size());
239 return begin()[i];
240 }
241
242 const T& at(size_t i) const
243 {
244 ASSERT_WITH_SECURITY_IMPLICATION(i < size());
245 return begin()[i];
246 }
247
248 T& operator[](size_t i) { return at(i); }
249 const T& operator[](size_t i) const { return at(i); }
250
251 T& first() { return (*this)[0]; }
252 const T& first() const { return (*this)[0]; }
253 T& last() { return (*this)[size() - 1]; }
254 const T& last() const { return (*this)[size() - 1]; }
255
256 void fill(const T& val)
257 {
258 std::fill(begin(), end(), val);
259 }
260
261 void swap(RefCountedArray& other)
262 {
263 PtrTraits::swap(m_data, other.m_data);
264 }
265
266 template<typename OtherTraits = PtrTraits>
267 bool operator==(const RefCountedArray<T, OtherTraits>& other) const
268 {
269 T* data = const_cast<T*>(this->data());
270 T* otherData = const_cast<T*>(other.data());
271 if (data == otherData)
272 return true;
273 if (!data || !otherData)
274 return false;
275 unsigned length = Header::fromPayload(data)->length;
276 if (length != Header::fromPayload(otherData)->length)
277 return false;
278 for (unsigned i = 0; i < length; ++i) {
279 if (data[i] != otherData[i])
280 return false;
281 }
282 return true;
283 }
284
285 bool operator==(const RefCountedArray& other) const { return this->operator==<PtrTraits>(other); }
286
287private:
288 static T* allocateUninitializedData(unsigned length)
289 {
290 T* data = (static_cast<Header*>(RefCountedArrayMalloc::malloc(Header::size() + sizeof(T) * length)))->payload();
291 Header::fromPayload(data)->refCount = 1;
292 Header::fromPayload(data)->length = length;
293 ASSERT(Header::fromPayload(data)->length == length);
294 return data;
295 }
296
297 template<typename OtherTraits = PtrTraits>
298 RefCountedArray& assign(const RefCountedArray<T, OtherTraits>& other)
299 {
300 T* oldData = data();
301 T* otherData = const_cast<T*>(other.data());
302 if (otherData)
303 Header::fromPayload(otherData)->refCount++;
304 m_data = otherData;
305
306 if (!oldData)
307 return *this;
308 unsigned refCount = Header::fromPayload(oldData)->refCount - 1;
309 if (refCount) {
310 Header::fromPayload(oldData)->refCount = refCount;
311 return *this;
312 }
313 VectorTypeOperations<T>::destruct(oldData, oldData + Header::fromPayload(oldData)->length);
314
315 RefCountedArrayMalloc::free(Header::fromPayload(oldData));
316 return *this;
317 }
318
319 struct Header {
320 unsigned refCount;
321 unsigned length;
322
323 static constexpr size_t size()
324 {
325 return (sizeof(Header) + 7) & ~7;
326 }
327
328 static ptrdiff_t offsetOfLength() { return OBJECT_OFFSETOF(Header, length); }
329
330 T* payload()
331 {
332 char* result = reinterpret_cast<char*>(this) + size();
333 ASSERT(!(bitwise_cast<uintptr_t>(result) & 7));
334 return reinterpret_cast_ptr<T*>(result);
335 }
336
337 static Header* fromPayload(T* payload)
338 {
339 return reinterpret_cast_ptr<Header*>(reinterpret_cast<char*>(payload) - size());
340 }
341
342 static const Header* fromPayload(const T* payload)
343 {
344 return fromPayload(const_cast<T*>(payload));
345 }
346 };
347
348 template<typename OtherTraits>
349 RefCountedArray(CommonCopyConstructorTag, const RefCountedArray<T, OtherTraits>& other)
350 : m_data(const_cast<T*>(other.data()))
351 {
352 if (m_data)
353 Header::fromPayload(data())->refCount++;
354 }
355
356 friend class JSC::LLIntOffsetsExtractor;
357 typename PtrTraits::StorageType m_data { nullptr };
358};

360} // namespace WTF
