170 template <typename TimerFiredClass>
171 class Timer : public TimerBase {
172 public:
173 typedef void (TimerFiredClass::*TimerFiredFunction)();
174
175 Timer(RunLoop& runLoop, TimerFiredClass* o, TimerFiredFunction f)
176 : TimerBase(runLoop)
177 , m_function(std::bind(f, o))
178 {
179 }
180
181 Timer(RunLoop& runLoop, Function<void ()>&& function)
182 : TimerBase(runLoop)
183 , m_function(WTFMove(function))
184 {
185 }
186
187 private:
188 void fired() override { m_function(); }
189
190 Function<void()> m_function;
191 };
