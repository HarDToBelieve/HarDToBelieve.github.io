26#pragma once
27
28#include <wtf/Assertions.h>
29#include <wtf/StdLibExtras.h>
30
31namespace WTF {
32
33template<typename T>
34class SinglyLinkedListWithTail {
35public:
36 bool isEmpty() const { return !m_first; }
37
38 template<typename SetNextFunc>
39 void append(SetNextFunc&& setNextFunc, T* node)
40 {
41 if (!m_first) {
42 RELEASE_ASSERT(!m_last);
43 m_first = node;
44 m_last = node;
45 return;
46 }
47
48 std::forward<SetNextFunc>(setNextFunc)(m_last, node);
49 m_last = node;
50 }
51
52 T* first() const { return m_first; }
53 T* last() const { return m_last; }
54
55private:
56 T* m_first { nullptr };
57 T* m_last { nullptr };
58};
59
60} // namespace WTF
61
62using WTF::SinglyLinkedListWithTail;
