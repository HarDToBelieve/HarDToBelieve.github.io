35DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(SmallSet);
36
37// Functionally, this class is very similar to Variant<Vector<T, SmallArraySize>, HashSet<T>>
38// It is optimized primarily for space, but is also quite fast
39// Its main limitation is that it has no way to remove elements once they have been added to it
40// Also, instead of being fully parameterized by a HashTrait parameter, it always uses -1 (all ones) as its empty value
41// Relatedly, it can only store objects of up to 64 bit size (but that particular limitation should be fairly easy to lift if needed)
42// Use it whenever you need to store an unbounded but probably small number of unsigned integers or pointers.
43template<typename T, typename Hash = PtrHashBase<T, false /* isSmartPtr */>, unsigned SmallArraySize = 8>
44class SmallSet {
45 WTF_MAKE_FAST_ALLOCATED;
46 WTF_MAKE_NONCOPYABLE(SmallSet);
47 static_assert(std::is_trivially_destructible<T>::value, "We currently don't support non-trivially destructible types.");
48 static_assert(!(SmallArraySize & (SmallArraySize - 1)), "Inline size must be a power of two.");
49 static_assert(sizeof(T*) <= SmallArraySize * sizeof(T), "This class has not been tested for m_inline.buffer larger than m_inline.smallStorage");
50
51public:
52 SmallSet()
53 : m_inline()
54 {
55 initialize();
56 }
57
58 // We take care to have SmallSet have partial move semantics allowable through
59 // memcpy. It's partial move semantics because our destructor should not be called
60 // on the SmallPtrObject in the old memory we were moved from (otherwise, we might free m_buffer twice)
61 // unless that old memory is reset to be isSmall(). See move constructor below.
62 // To maintain these semantics, we determine if we're small by checking our size
63 // and not our m_buffer pointer. And when we're small, we don't do operations on
64 // m_buffer, instead, we perform operations on m_smallStorage directly. The reason we want
65 // these semantics is that it's beneficial to have a Vector that contains SmallSet
66 // (or an object with SmallSet as a field) be allowed to use memcpy for its move operation.
67
68 SmallSet(SmallSet&& other)
69 {
70 memcpy(static_cast<void*>(this), static_cast<void*>(&other), sizeof(SmallSet));
71 other.initialize();
72 }
73
74 SmallSet& operator=(SmallSet&& other)
75 {
76 this->~SmallSet();
77 new (this) SmallSet(WTFMove(other));
78 return *this;
79 }
80
81 ~SmallSet()
82 {
83 if (!isSmall())
84 SmallSetMalloc::free(m_inline.buffer);
85 }
86
87 // We could easily include an iterator in this to fully match the HashSet interface, but currently none of our clients require it.
88 struct AddResult {
89 bool isNewEntry;
90 };
91
92 inline AddResult add(T value)
93 {
94 ASSERT(isValidEntry(value));
95
96 if (isSmall()) {
97 for (unsigned i = 0; i < m_size; i++) {
98 if (m_inline.smallStorage[i] == value)
99 return { false };
100 }
101
102 if (m_size < SmallArraySize) {
103 m_inline.smallStorage[m_size] = value;
104 ++m_size;
105 return { true };
106 }
107
108 grow(std::max(64u, SmallArraySize * 2));
109 // Fall through. We're no longer small :(
110 }
111
112 // If we're more than 3/4ths full we grow.
113 if (UNLIKELY(m_size * 4 >= m_capacity * 3)) {
114 grow(m_capacity * 2);
115 ASSERT(!(m_capacity & (m_capacity - 1)));
116 }
117
118 T* bucket = this->bucket(value);
119 if (*bucket != value) {
120 *bucket = value;
121 ++m_size;
122 return { true };
123 }
124 return { false };
125 }
126
127 inline bool contains(T value) const
128 {
129 ASSERT(isValidEntry(value));
130 if (isSmall()) {
131 // We only need to search up to m_size because we store things linearly inside m_smallStorage.
132 for (unsigned i = 0; i < m_size; i++) {
133 if (m_inline.smallStorage[i] == value)
134 return true;
135 }
136 return false;
137 }
138
139 T* bucket = this->bucket(value);
140 return *bucket == value;
141 }
142
143 class iterator {
144 WTF_MAKE_FAST_ALLOCATED;
145 public:
146 iterator& operator++()
147 {
148 m_index++;
149 ASSERT(m_index <= m_capacity);
150 while (m_index < m_capacity && m_buffer[m_index] == emptyValue())
151 m_index++;
152 return *this;
153 }
154
155 T operator*() const { ASSERT(m_index < m_capacity); return static_cast<T>(m_buffer[m_index]); }
156 bool operator==(const iterator& other) const { ASSERT(m_buffer == other.m_buffer); return m_index == other.m_index; }
157 bool operator!=(const iterator& other) const { ASSERT(m_buffer == other.m_buffer); return !(*this == other); }
158
159 private:
160 template<typename U, typename H, unsigned S> friend class WTF::SmallSet;
161 unsigned m_index;
162 unsigned m_capacity;
163 T* m_buffer;
164 };
165
166 iterator begin() const
167 {
168 iterator it;
169 it.m_index = std::numeric_limits<unsigned>::max();
170 it.m_capacity = m_capacity;
171 if (isSmall())
172 it.m_buffer = const_cast<T*>(m_inline.smallStorage);
173 else
174 it.m_buffer = m_inline.buffer;
175
176 ++it;
177
178 return it;
179 }
180
181 iterator end() const
182 {
183 iterator it;
184 it.m_index = m_capacity;
185 it.m_capacity = m_capacity;
186 if (isSmall())
187 it.m_buffer = const_cast<T*>(m_inline.smallStorage);
188 else
189 it.m_buffer = m_inline.buffer;
190
191 return it;
192 }
193
194 inline unsigned size() const { return m_size; }
195
196 inline bool isEmpty() const { return !size(); }
197
198 unsigned memoryUse() const
199 {
200 unsigned memory = sizeof(SmallSet);
201 if (!isSmall())
202 memory += m_capacity * sizeof(T);
203 return memory;
204 }
205
206private:
207 constexpr static T emptyValue()
208 {
209 if constexpr (std::is_pointer<T>::value)
210 return static_cast<T>(bitwise_cast<void*>(std::numeric_limits<uintptr_t>::max()));
211 return std::numeric_limits<T>::max();
212 }
213
214 bool isValidEntry(const T value) const
215 {
216 return value != emptyValue();
217 }
218
219 inline bool isSmall() const
220 {
221 return m_capacity == SmallArraySize;
222 }
223
224 inline void initialize()
225 {
226 m_size = 0;
227 m_capacity = SmallArraySize;
228 memset(static_cast<void*>(m_inline.smallStorage), -1, sizeof(T) * SmallArraySize);
229 ASSERT(isSmall());
230 }
231
232 inline void grow(unsigned size)
233 {
234 // We memset the new buffer with -1, so for consistency emptyValue() must return something which is all 1s.
235#if !defined(NDEBUG)
236 if constexpr (std::is_pointer<T>::value)
237 ASSERT(bitwise_cast<intptr_t>(emptyValue()) == -1ll);
238 else if constexpr (sizeof(T) == 8)
239 ASSERT(bitwise_cast<int64_t>(emptyValue()) == -1ll);
240 else if constexpr (sizeof(T) == 4)
241 ASSERT(bitwise_cast<int32_t>(emptyValue()) == -1);
242 else if constexpr (sizeof(T) == 2)
243 ASSERT(bitwise_cast<int16_t>(emptyValue()) == -1);
244 else if constexpr (sizeof(T) == 1)
245 ASSERT(bitwise_cast<int8_t>(emptyValue()) == -1);
246 else
247 RELEASE_ASSERT_NOT_REACHED();
248#endif
249
250 size_t allocationSize = sizeof(T) * size;
251 bool wasSmall = isSmall();
252 T* oldBuffer = wasSmall ? m_inline.smallStorage : m_inline.buffer;
253 unsigned oldCapacity = m_capacity;
254 T* newBuffer = static_cast<T*>(SmallSetMalloc::malloc(allocationSize));
255 memset(static_cast<void*>(newBuffer), -1, allocationSize);
256 m_capacity = size;
257
258 for (unsigned i = 0; i < oldCapacity; i++) {
259 if (oldBuffer[i] != emptyValue()) {
260 T* ptr = bucketInBuffer(newBuffer, static_cast<T>(oldBuffer[i]));
261 *ptr = oldBuffer[i];
262 }
263 }
264
265 if (!wasSmall)
266 SmallSetMalloc::free(oldBuffer);
267
268 m_inline.buffer = newBuffer;
269 }
270
271
272 inline T* bucket(T target) const
273 {
274 ASSERT(!isSmall());
275 return bucketInBuffer(m_inline.buffer, target);
276 }
277
278 inline T* bucketInBuffer(T* buffer, T target) const
279 {
280 ASSERT(!(m_capacity & (m_capacity - 1)));
281 unsigned bucket = Hash::hash(target) & (m_capacity - 1);
282 unsigned index = 0;
283 while (true) {
284 T* ptr = buffer + bucket;
285 if (*ptr == emptyValue())
286 return ptr;
287 if (*ptr == target)
288 return ptr;
289 index++;
290 bucket = (bucket + index) & (m_capacity - 1);
291 }
292 }
293
294 unsigned m_size;
295 unsigned m_capacity;
296 union U {
297 T* buffer;
298 T smallStorage[SmallArraySize];
299 U() { };
300 } m_inline;
301};
