33// SortedArrayMap is a map like HashMap, but it's read-only. It uses much less memory than HashMap.
34// It uses binary search instead of hashing, so can be outperformed by HashMap for large maps.
35// The array passed to the constructor has std::pair elements: keys first and values second.
36// The array and the SortedArrayMap should typically both be global constant expressions.
37
38class SortedArrayBase {
39protected:
40 // Some informal empirical tests indicate that arrays shorter than this are faster to
41 // search with linear search than with binary search. Even if we don't get this threshold
42 // exactly right, it's helpful for both performance and code size to use linear search at
43 // least for very small arrays, and important for performance to make sure that we use
44 // binary search for much larger ones.
45 static constexpr size_t binarySearchThreshold = 20;
46};
47
48template<typename ArrayType> class SortedArrayMap : public SortedArrayBase {
49public:
50 using ElementType = typename std::remove_extent_t<ArrayType>;
51 using ValueType = typename ElementType::second_type;
52
53 constexpr SortedArrayMap(const ArrayType&);
54
55 // FIXME: To match HashMap interface better, would be nice to get the default value from traits.
56 template<typename KeyArgument> ValueType get(const KeyArgument&, const ValueType& defaultValue = { }) const;
57
58 // FIXME: Should add a function like this to HashMap so the two kinds of maps are more interchangable.
59 template<typename KeyArgument> const ValueType* tryGet(const KeyArgument&) const;
60
61private:
62 const ArrayType& m_array;
63};

124template<ASCIISubset subset> constexpr bool isInSubset(char character)
125{
126 if (!(character && isASCII(character)))
127 return false;
128 switch (subset) {
129 case ASCIISubset::All:
130 return true;
131 case ASCIISubset::NoUppercaseLetters:
132 return !isASCIIUpper(character);
133 case ASCIISubset::NoUppercaseLettersOptimized:
134 return character == toASCIILowerUnchecked(character);
135 }
136}

138template<ASCIISubset subset> template<unsigned size> constexpr ComparableASCIISubsetLiteral<subset>::ComparableASCIISubsetLiteral(const char (&characters)[size])
139 : literal { ASCIILiteral::fromLiteralUnsafe(characters) }
140{
141 ASSERT_UNDER_CONSTEXPR_CONTEXT(allOfConstExpr(&characters[0], &characters[size - 1], [] (char character) {
142 return isInSubset<subset>(character);
143 }));
144 ASSERT_UNDER_CONSTEXPR_CONTEXT(!characters[size - 1]);
145}

155template<typename, typename = void> constexpr bool HasParseMember = false;
156template<typename T> constexpr bool HasParseMember<T, std::void_t<decltype(std::declval<T>().parse)>> = true;
157
158template<typename ArrayType> template<typename KeyArgument> inline auto SortedArrayMap<ArrayType>::tryGet(const KeyArgument& key) const -> const ValueType*
159{
160 using KeyType = typename ElementType::first_type;
161 auto parsedKey = [&key] {
162 if constexpr (HasParseMember<KeyType>)
163 return KeyType::parse(key);
164 else
165 return std::make_optional(key);
166 }();
167 if (!parsedKey)
168 return nullptr;
169 decltype(std::begin(m_array)) iterator;
170 if (std::size(m_array) < binarySearchThreshold) {
171 iterator = std::find_if(std::begin(m_array), std::end(m_array), [&parsedKey] (auto& pair) {
172 return pair.first == *parsedKey;
173 });
174 if (iterator == std::end(m_array))
175 return nullptr;
176 } else {
177 iterator = std::lower_bound(std::begin(m_array), std::end(m_array), *parsedKey, [] (auto& pair, auto& value) {
178 return pair.first < value;
179 });
180 if (iterator == std::end(m_array) || !(iterator->first == *parsedKey))
181 return nullptr;
182 }
183 return &iterator->second;
184}

186template<typename ArrayType> template<typename KeyArgument> inline auto SortedArrayMap<ArrayType>::get(const KeyArgument& key, const ValueType& defaultValue) const -> ValueType
187{
188 auto result = tryGet(key);
189 return result ? *result : defaultValue;
190}

198template<typename ArrayType> template<typename KeyArgument> inline bool SortedArraySet<ArrayType>::contains(const KeyArgument& key) const
199{
200 using KeyType = typename std::remove_extent_t<ArrayType>;
201 auto parsedKey = [&key] {
202 if constexpr (HasParseMember<KeyType>)
203 return KeyType::parse(key);
204 else
205 return std::make_optional(key);
206 }();
207 if (!parsedKey)
208 return false;
209 if (std::size(m_array) < binarySearchThreshold)
210 return std::find(std::begin(m_array), std::end(m_array), *parsedKey) != std::end(m_array);
211 auto iterator = std::lower_bound(std::begin(m_array), std::end(m_array), *parsedKey);
212 return iterator != std::end(m_array) && *iterator == *parsedKey;
213}

215constexpr int strcmpConstExpr(const char* a, const char* b)
216{
217 while (*a == *b && *a && *b) {
218 ++a;
219 ++b;
220 }
221 return *a == *b ? 0 : *a < *b ? -1 : 1;
222}

224template<typename CharacterType> inline bool lessThanASCIICaseFolding(const CharacterType* characters, unsigned length, const char* literalWithNoUppercase)
225{
226 for (unsigned i = 0; i < length; ++i) {
227 if (!literalWithNoUppercase[i])
228 return false;
229 if (auto character = toASCIILower(characters[i]); character != literalWithNoUppercase[i])
230 return character < literalWithNoUppercase[i];
231 }
232 return true;
233}

235inline bool lessThanASCIICaseFolding(StringView string, const char* literalWithNoUppercase)
236{
237 if (string.is8Bit())
238 return lessThanASCIICaseFolding(string.characters8(), string.length(), literalWithNoUppercase);
239 return lessThanASCIICaseFolding(string.characters16(), string.length(), literalWithNoUppercase);
240}

242template<typename CharacterType> inline bool lessThanASCIICaseFolding(const char* literalWithNoUppercase, const CharacterType* characters, unsigned length)
243{
244 for (unsigned i = 0; i < length; ++i) {
245 if (!literalWithNoUppercase[i])
246 return true;
247 if (auto character = toASCIILower(characters[i]); character != literalWithNoUppercase[i])
248 return literalWithNoUppercase[i] < character;
249 }
250 return false;
251}

253inline bool lessThanASCIICaseFolding(const char* literalWithNoUppercase, StringView string)
254{
255 if (string.is8Bit())
256 return lessThanASCIICaseFolding(literalWithNoUppercase, string.characters8(), string.length());
257 return lessThanASCIICaseFolding(literalWithNoUppercase, string.characters16(), string.length());
258}

330template<typename StorageInteger> template<unsigned size> constexpr StorageInteger PackedASCIILowerCodes<StorageInteger>::pack(const char (&string)[size])
331{
332 ASSERT_UNDER_CONSTEXPR_CONTEXT(size);
333 constexpr unsigned length = size - 1;
334 ASSERT_UNDER_CONSTEXPR_CONTEXT(!string[length]);
335 ASSERT_UNDER_CONSTEXPR_CONTEXT(length <= sizeof(StorageInteger));
336 StorageInteger result = 0;
337 for (unsigned index = 0; index < length; ++index) {
338 StorageInteger code = static_cast<uint8_t>(string[index]);
339 result |= code << ((sizeof(StorageInteger) - index - 1) * 8);
340 }
341 return result;
342}
343
344template<typename StorageInteger> auto PackedASCIILowerCodes<StorageInteger>::parse(StringView string) -> std::optional<PackedASCIILowerCodes>
345{
346 if (string.length() > sizeof(StorageInteger))
347 return std::nullopt;
348 StorageInteger result = 0;
349 for (unsigned index = 0; index < string.length(); ++index) {
350 UChar code = string[index];
351 if (!isASCII(code))
352 return std::nullopt;
353 result |= static_cast<StorageInteger>(toASCIILower(code)) << ((sizeof(StorageInteger) - index - 1) * 8);
354 }
355 return PackedASCIILowerCodes(result);
356}
