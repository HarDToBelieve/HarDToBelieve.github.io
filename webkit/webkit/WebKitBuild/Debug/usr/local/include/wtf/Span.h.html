70template <class _Tp, class _ElementType>
71struct __is_span_compatible_container<_Tp, _ElementType,
72 std::void_t<
73 // is not a specialization of Span
74 typename std::enable_if<!__is_span<_Tp>::value, std::nullptr_t>::type,
75 // is not a specialization of array
76 typename std::enable_if<!__is_std_array<_Tp>::value, std::nullptr_t>::type,
77 // sd::is_array_v<Container> is false,
78 typename std::enable_if<!std::is_array_v<_Tp>, std::nullptr_t>::type,
79 // std::data(cont) and std::size(cont) are well formed
80 decltype(std::data(std::declval<_Tp>())),
81 decltype(std::size(std::declval<_Tp>())),
82 // remove_pointer_t<decltype(data(cont))>(*)[] is convertible to ElementType(*)[]
83 typename std::enable_if<
84 std::is_convertible_v<std::remove_pointer_t<decltype(std::data(std::declval<_Tp &>()))>(*)[],
85 _ElementType(*)[]>,
86 std::nullptr_t>::type
87 >>
88 : public std::true_type {};

95 using element_type = _Tp;

112// [span.cons], span constructors, copy, assignment, and destructor
113 template <std::size_t _Sz = _Extent, std::enable_if_t<_Sz == 0, std::nullptr_t> = nullptr>

119 _WTF_LIBCPP_INLINE_VISIBILITY constexpr explicit Span(pointer __ptr, size_type __count) : __data{__ptr}

121 _WTF_LIBCPP_INLINE_VISIBILITY constexpr explicit Span(pointer __f, pointer __l) : __data{__f}
122 { (void)__l; _WTF_LIBCPP_ASSERT(_Extent == std::distance(__f, __l), "size mismatch in Span's constructor (ptr, ptr)"); }

201 template <std::size_t _Offset, std::size_t _Count = dynamic_extent>
202 _WTF_LIBCPP_INLINE_VISIBILITY
203 constexpr auto subspan() const noexcept
204 -> Span<element_type, _Count != dynamic_extent ? _Count : _Extent - _Offset>
205 {
206 static_assert(_Offset <= _Extent, "Offset out of range in Span::subspan()");
207 static_assert(_Count == dynamic_extent || _Count <= _Extent - _Offset, "Offset + count out of range in Span::subspan()");
208
209 using _ReturnType = Span<element_type, _Count != dynamic_extent ? _Count : _Extent - _Offset>;
210 return _ReturnType{data() + _Offset, _Count == dynamic_extent ? size() - _Offset : _Count};
211 }
212
213
214 _WTF_LIBCPP_INLINE_VISIBILITY
215 constexpr Span<element_type, dynamic_extent>
216 subspan(size_type __offset, size_type __count = dynamic_extent) const noexcept
217 {
218 _WTF_LIBCPP_ASSERT(__offset <= size(), "Offset out of range in Span::subspan(offset, count)");
219 _WTF_LIBCPP_ASSERT(__count <= size() || __count == dynamic_extent, "Count out of range in Span::subspan(offset, count)");
220 if (__count == dynamic_extent)
221 return {data() + __offset, size() - __offset};
222 _WTF_LIBCPP_ASSERT(__count <= size() - __offset, "Offset + count out of range in Span::subspan(offset, count)");
223 return {data() + __offset, __count};
224 }

227 _WTF_LIBCPP_INLINE_VISIBILITY constexpr size_type size_bytes() const noexcept { return _Extent * sizeof(element_type); }
228 _WTF_LIBCPP_INLINE_VISIBILITY constexpr bool empty() const noexcept { return _Extent == 0; }
229
230 _WTF_LIBCPP_INLINE_VISIBILITY constexpr reference operator[](size_type __idx) const noexcept
231 {
232 _WTF_LIBCPP_ASSERT(__idx < size(), "Span<T,N>[] index out of bounds");
233 return __data[__idx];
234 }

236 _WTF_LIBCPP_INLINE_VISIBILITY constexpr reference front() const noexcept
237 {
238 _WTF_LIBCPP_ASSERT(!empty(), "Span<T, N>::front() on empty Span");
239 return __data[0];
240 }
241
242 _WTF_LIBCPP_INLINE_VISIBILITY constexpr reference back() const noexcept
243 {
244 _WTF_LIBCPP_ASSERT(!empty(), "Span<T, N>::back() on empty Span");
245 return __data[size()-1];
246 }
247
248 _WTF_LIBCPP_INLINE_VISIBILITY constexpr pointer data() const noexcept { return __data; }
249
250// [span.iter], Span iterator support
251 _WTF_LIBCPP_INLINE_VISIBILITY constexpr iterator begin() const noexcept { return iterator(data()); }
252 _WTF_LIBCPP_INLINE_VISIBILITY constexpr iterator end() const noexcept { return iterator(data() + size()); }
253 _WTF_LIBCPP_INLINE_VISIBILITY constexpr reverse_iterator rbegin() const noexcept { return reverse_iterator(end()); }
254 _WTF_LIBCPP_INLINE_VISIBILITY constexpr reverse_iterator rend() const noexcept { return reverse_iterator(begin()); }

256private:
257 pointer __data;
