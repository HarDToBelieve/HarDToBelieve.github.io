27#pragma once
28
29#include <optional>
30#include <wtf/SystemFree.h>
31
32namespace WTF {
33
34class PrintStream;
35
36class StackTrace {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 WTF_EXPORT_PRIVATE static std::unique_ptr<StackTrace> captureStackTrace(int maxFrames, int framesToSkip = 0);
40
41 // Borrowed stack trace.
42 StackTrace(void** stack, int size, const char* prefix = "")
43 : m_size(size)
44 , m_capacity(0)
45 , m_borrowedStack(stack)
46 , m_prefix(prefix)
47 { }
48
49 int size() const { return m_size; }
50 void* const * stack() const
51 {
52 if (!m_capacity)
53 return m_borrowedStack;
54 return m_stack;
55 }
56
57 class DemangleEntry {
58 WTF_MAKE_FAST_ALLOCATED;
59 public:
60 friend class StackTrace;
61 const char* mangledName() const { return m_mangledName; }
62 const char* demangledName() const { return m_demangledName.get(); }
63
64 private:
65 DemangleEntry(const char* mangledName, const char* demangledName)
66 : m_mangledName(mangledName)
67 , m_demangledName(demangledName)
68 { }
69
70 const char* m_mangledName { nullptr };
71 std::unique_ptr<const char[], SystemFree<const char[]>> m_demangledName;
72 };
73
74 WTF_EXPORT_PRIVATE static std::optional<DemangleEntry> demangle(void*);
75
76 WTF_EXPORT_PRIVATE void dump(PrintStream&, const char* indentString = nullptr) const;
77
78private:
79 inline static size_t instanceSize(int capacity);
80
81 StackTrace()
82 : m_size(0)
83 { }
84
85 // We structure the top fields this way because the underlying stack capture
86 // facility will capture from the top of the stack, and we'll need to skip the
87 // top 2 frame which is of no interest. Setting up the fields layout this way
88 // allows us to capture the stack in place and minimize space wastage.
89 union {
90 struct {
91 int m_size;
92 int m_capacity;
93 };
94 struct {
95 void* m_skippedFrame0;
96 void* m_skippedFrame1;
97 };
98 };
99 union {
100 void** m_borrowedStack;
101 void* m_stack[1];
102 };
103
104 const char* m_prefix;
105};

107} // namespace WTF
