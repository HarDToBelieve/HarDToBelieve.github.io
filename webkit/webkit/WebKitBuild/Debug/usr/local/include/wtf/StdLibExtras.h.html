75 * sizeof(Type1) > sizeof(Type2) - cause the following warning on ARM with GCC:
76 * increases required alignment of target type.

78 * An implicit or an extra static_cast<void*> bypasses the warning.

130template<typename ToType, typename FromType>
131inline ToType bitwise_cast(FromType from)
132{
133 static_assert(sizeof(FromType) == sizeof(ToType), "bitwise_cast size of FromType and ToType must be equal!");
134#if COMPILER_SUPPORTS(BUILTIN_IS_TRIVIALLY_COPYABLE)
135 // Not all recent STL implementations support the std::is_trivially_copyable type trait. Work around this by only checking on toolchains which have the equivalent compiler intrinsic.
136 static_assert(__is_trivially_copyable(ToType), "bitwise_cast of non-trivially-copyable type!");
137 static_assert(__is_trivially_copyable(FromType), "bitwise_cast of non-trivially-copyable type!");
138#endif
139 typename std::remove_const<ToType>::type to { };
140 std::memcpy(static_cast<void*>(&to), static_cast<void*>(&from), sizeof(to));
141 return to;
142}

151// Returns a count of the number of bits set in 'bits'.
152inline size_t bitCount(unsigned bits)
153{
154 bits = bits - ((bits >> 1) & 0x55555555);
155 bits = (bits & 0x33333333) + ((bits >> 2) & 0x33333333);
156 return (((bits + (bits >> 4)) & 0xF0F0F0F) * 0x1010101) >> 24;
157}
158
159inline size_t bitCount(uint64_t bits)
160{
161 return bitCount(static_cast<unsigned>(bits)) + bitCount(static_cast<unsigned>(bits >> 32));
162}

172ALWAYS_INLINE constexpr size_t roundUpToMultipleOfImpl(size_t divisor, size_t x)
173{
174 size_t remainderMask = divisor - 1;
175 return (x + remainderMask) & ~remainderMask;
176}
177
178// Efficient implementation that takes advantage of powers of two.
179inline size_t roundUpToMultipleOf(size_t divisor, size_t x)
180{
181 ASSERT(divisor && !(divisor & (divisor - 1)));
182 return roundUpToMultipleOfImpl(divisor, x);
183}

185template<size_t divisor> constexpr size_t roundUpToMultipleOf(size_t x)
186{
187 static_assert(divisor && !(divisor & (divisor - 1)), "divisor must be a power of two!");
188 return roundUpToMultipleOfImpl(divisor, x);
189}

197enum BinarySearchMode {
198 KeyMustBePresentInArray,
199 KeyMightNotBePresentInArray,
200 ReturnAdjacentElementIfKeyIsNotPresent
201};
202
203template<typename ArrayElementType, typename KeyType, typename ArrayType, typename ExtractKey, BinarySearchMode mode>
204inline ArrayElementType* binarySearchImpl(ArrayType& array, size_t size, KeyType key, const ExtractKey& extractKey = ExtractKey())
205{
206 size_t offset = 0;
207 while (size > 1) {
208 size_t pos = (size - 1) >> 1;
209 auto val = extractKey(&array[offset + pos]);
210
211 if (val == key)
212 return &array[offset + pos];
213 // The item we are looking for is smaller than the item being check; reduce the value of 'size',
214 // chopping off the right hand half of the array.
215 if (key < val)
216 size = pos;
217 // Discard all values in the left hand half of the array, up to and including the item at pos.
218 else {
219 size -= (pos + 1);
220 offset += (pos + 1);
221 }
222
223 ASSERT(mode != KeyMustBePresentInArray || size);
224 }
225
226 if (mode == KeyMightNotBePresentInArray && !size)
227 return 0;
228
229 ArrayElementType* result = &array[offset];
230
231 if (mode == KeyMightNotBePresentInArray && key != extractKey(result))
232 return 0;
233
234 if (mode == KeyMustBePresentInArray) {
235 ASSERT(size == 1);
236 ASSERT(key == extractKey(result));
237 }
238
239 return result;
240}

242// If the element is not found, crash if asserts are enabled, and behave like approximateBinarySearch in release builds.
243template<typename ArrayElementType, typename KeyType, typename ArrayType, typename ExtractKey>
244inline ArrayElementType* binarySearch(ArrayType& array, size_t size, KeyType key, ExtractKey extractKey = ExtractKey())
245{
246 return binarySearchImpl<ArrayElementType, KeyType, ArrayType, ExtractKey, KeyMustBePresentInArray>(array, size, key, extractKey);
247}

249// Return zero if the element is not found.
250template<typename ArrayElementType, typename KeyType, typename ArrayType, typename ExtractKey>
251inline ArrayElementType* tryBinarySearch(ArrayType& array, size_t size, KeyType key, ExtractKey extractKey = ExtractKey())
252{
253 return binarySearchImpl<ArrayElementType, KeyType, ArrayType, ExtractKey, KeyMightNotBePresentInArray>(array, size, key, extractKey);
254}

319template<typename T>
320bool findBitInWord(T word, size_t& startOrResultIndex, size_t endIndex, bool value)
321{
322 static_assert(std::is_unsigned<T>::value, "Type used in findBitInWord must be unsigned");
323
324 constexpr size_t bitsInWord = sizeof(word) * 8;
325 ASSERT_UNUSED(bitsInWord, startOrResultIndex <= bitsInWord && endIndex <= bitsInWord);
326
327 size_t index = startOrResultIndex;
328 word >>= index;
329
330#if COMPILER(GCC_COMPATIBLE) && (CPU(X86_64) || CPU(ARM64))
331 // We should only use ctz() when we know that ctz() is implementated using
332 // a fast hardware instruction. Otherwise, this will actually result in
333 // worse performance.
334
335 word ^= (static_cast<T>(value) - 1);
336 index += ctz(word);
337 if (index < endIndex) {
338 startOrResultIndex = index;
339 return true;
340 }
341#else
342 while (index < endIndex) {
343 if ((word & 1) == static_cast<T>(value)) {
344 startOrResultIndex = index;
345 return true;
346 }
347 index++;
348 word >>= 1;
349 }
350#endif
351
352 startOrResultIndex = endIndex;
353 return false;
354}

428template<typename IteratorTypeLeft, typename IteratorTypeRight, typename IteratorTypeDst>
429IteratorTypeDst mergeDeduplicatedSorted(IteratorTypeLeft leftBegin, IteratorTypeLeft leftEnd, IteratorTypeRight rightBegin, IteratorTypeRight rightEnd, IteratorTypeDst dstBegin)
430{
431 IteratorTypeLeft leftIter = leftBegin;
432 IteratorTypeRight rightIter = rightBegin;
433 IteratorTypeDst dstIter = dstBegin;
434
435 if (leftIter < leftEnd && rightIter < rightEnd) {
436 for (;;) {
437 auto left = *leftIter;
438 auto right = *rightIter;
439 if (left < right) {
440 *dstIter++ = left;
441 leftIter++;
442 if (leftIter >= leftEnd)
443 break;
444 } else if (left == right) {
445 *dstIter++ = left;
446 leftIter++;
447 rightIter++;
448 if (leftIter >= leftEnd || rightIter >= rightEnd)
449 break;
450 } else {
451 *dstIter++ = right;
452 rightIter++;
453 if (rightIter >= rightEnd)
454 break;
455 }
456 }
457 }
458
459 while (leftIter < leftEnd)
460 *dstIter++ = *leftIter++;
461 while (rightIter < rightEnd)
462 *dstIter++ = *rightIter++;
463
464 return dstIter;
465}

530// FIXME: Use std::is_sorted instead of this and remove it, once we require C++20.
531template<typename Iterator, typename Predicate> constexpr bool isSortedConstExpr(Iterator first, Iterator last, Predicate predicate)
532{
533 if (first == last)
534 return true;
535 auto current = first;
536 auto previous = current;
537 while (++current != last) {
538 if (!predicate(*previous, *current))
539 return false;
540 previous = current;
541 }
542 return true;
543}

546template<typename Iterator> constexpr bool isSortedConstExpr(Iterator first, Iterator last)

548 return isSortedConstExpr(first, last, [] (auto& a, auto& b) { return a < b; });
549}
