25#pragma once
26
27#include <cmath>
28#include <wtf/MonotonicTime.h>
29#include <wtf/Ref.h>
30#include <wtf/RefCounted.h>
31
32namespace WTF {
33
34class Stopwatch : public RefCounted<Stopwatch> {
35public:
36 static Ref<Stopwatch> create()
37 {
38 return adoptRef(*new Stopwatch());
39 }
40
41 void reset();
42 void start();
43 void stop();
44
45 Seconds elapsedTime() const;
46 Seconds elapsedTimeSince(MonotonicTime) const;
47
48 bool isActive() const { return !std::isnan(m_lastStartTime); }
49private:
50 Stopwatch() { reset(); }
51
52 Seconds m_elapsedTime;
53 MonotonicTime m_lastStartTime;
54};
55
56inline void Stopwatch::reset()
57{
58 m_elapsedTime = 0_s;
59 m_lastStartTime = MonotonicTime::nan();
60}
61
62inline void Stopwatch::start()
63{
64 ASSERT_WITH_MESSAGE(std::isnan(m_lastStartTime), "Tried to start the stopwatch, but it is already running.");
65
66 m_lastStartTime = MonotonicTime::now();
67}
68
69inline void Stopwatch::stop()
70{
71 ASSERT_WITH_MESSAGE(!std::isnan(m_lastStartTime), "Tried to stop the stopwatch, but it is not running.");
72
73 m_elapsedTime += MonotonicTime::now() - m_lastStartTime;
74 m_lastStartTime = MonotonicTime::nan();
75}
76
77inline Seconds Stopwatch::elapsedTime() const
78{
79 if (!isActive())
80 return m_elapsedTime;
81
82 return m_elapsedTime + (MonotonicTime::now() - m_lastStartTime);
83}
84
85inline Seconds Stopwatch::elapsedTimeSince(MonotonicTime timeStamp) const
86{
87 if (!isActive())
88 return m_elapsedTime;
89
90 return m_elapsedTime + (timeStamp - m_lastStartTime);
91}
92
93} // namespace WTF
94
95using WTF::Stopwatch;
