26#pragma once
27
28#include <wtf/PrintStream.h>
29#include <wtf/text/CString.h>
30#include <wtf/text/WTFString.h>
31
32namespace WTF {
33
34class StringPrintStream final : public PrintStream {
35public:
36 WTF_EXPORT_PRIVATE StringPrintStream();
37 WTF_EXPORT_PRIVATE ~StringPrintStream() final;
38
39 WTF_EXPORT_PRIVATE void vprintf(const char* format, va_list) final WTF_ATTRIBUTE_PRINTF(2, 0);
40
41 size_t length() const { return m_next; }
42
43 WTF_EXPORT_PRIVATE CString toCString();
44 WTF_EXPORT_PRIVATE Expected<String, UTF8ConversionError> tryToString();
45 WTF_EXPORT_PRIVATE String toString();
46 WTF_EXPORT_PRIVATE String toStringWithLatin1Fallback();
47 WTF_EXPORT_PRIVATE void reset();
48
49private:
50 void increaseSize(size_t);
51
52 char* m_buffer;
53 size_t m_next;
54 size_t m_size;
55 char m_inlineBuffer[128];
56};
57
58// Stringify any type T that has a WTF::printInternal(PrintStream&, const T&)
59
60template<typename... Types>
61CString toCString(const Types&... values)
62{
63 StringPrintStream stream;
64 stream.print(values...);
65 return stream.toCString();
66}
67
68template<typename... Types>
69String toString(const Types&... values)
70{
71 StringPrintStream stream;
72 stream.print(values...);
73 return stream.toString();
74}
75
76} // namespace WTF
