26#pragma once
27
28#include <type_traits>
29
30namespace WTF {
31
32template<typename T>
33struct SystemFree {
34 static_assert(std::is_trivially_destructible<T>::value, "");
35
36 void operator()(T* pointer) const
37 {
38 free(const_cast<typename std::remove_cv<T>::type*>(pointer));
39 }
40};
41
42template<typename T>
43struct SystemFree<T[]> {
44 static_assert(std::is_trivially_destructible<T>::value, "");
45
46 void operator()(T* pointer) const
47 {
48 free(const_cast<typename std::remove_cv<T>::type*>(pointer));
49 }
50};
51
52} // namespace WTF
