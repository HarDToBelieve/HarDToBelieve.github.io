148inline void tracePoint(TracePointCode code, uint64_t data1 = 0, uint64_t data2 = 0, uint64_t data3 = 0, uint64_t data4 = 0)
149{
150#if HAVE(KDEBUG_H)
151 kdebug_trace(ARIADNEDBG_CODE(WEBKIT_COMPONENT, code), data1, data2, data3, data4);
152#else
153 UNUSED_PARAM(code);
154 UNUSED_PARAM(data1);
155 UNUSED_PARAM(data2);
156 UNUSED_PARAM(data3);
157 UNUSED_PARAM(data4);
158#endif
159}
160
161class TraceScope {
162 WTF_MAKE_FAST_ALLOCATED;
163public:
164
165 TraceScope(TracePointCode entryCode, TracePointCode exitCode, uint64_t data1 = 0, uint64_t data2 = 0, uint64_t data3 = 0, uint64_t data4 = 0)
166 : m_exitCode(exitCode)
167 {
168 tracePoint(entryCode, data1, data2, data3, data4);
169 }
170
171 ~TraceScope()
172 {
173 tracePoint(m_exitCode);
174 }
175
176private:
177 TracePointCode m_exitCode;
178};

189#import <os/signpost.h>

191WTF_EXTERN_C_BEGIN
192WTF_EXPORT_PRIVATE bool WTFSignpostsEnabled();
193WTF_EXPORT_PRIVATE os_log_t WTFSignpostLogHandle();
194WTF_EXTERN_C_END
195
196#define WTFEmitSignpost(pointer, name, ...) \
197 WTFEmitSignpostWithFunction(os_signpost_event_emit, (pointer), name, ##__VA_ARGS__)
198
199#define WTFBeginSignpost(pointer, name, ...) \
200 WTFEmitSignpostWithFunction(os_signpost_interval_begin, (pointer), name, ##__VA_ARGS__)
201
202#define WTFEndSignpost(pointer, name, ...) \
203 WTFEmitSignpostWithFunction(os_signpost_interval_end, (pointer), name, ##__VA_ARGS__)
204
205#define WTFEmitSignpostWithFunction(emitFunc, pointer, name, ...) \
206do { \
207 if (UNLIKELY(WTFSignpostsEnabled())) { \
208 os_log_t handle = WTFSignpostLogHandle(); \
209 os_signpost_id_t signpostID = os_signpost_id_make_with_pointer(handle, (pointer)); \
210 emitFunc(handle, signpostID, name, ##__VA_ARGS__); \
211 } \
212} while (0)
213
214#else
215
216#define WTFEmitSignpost(pointer, name, ...) do { } while (0)
217#define WTFBeginSignpost(pointer, name, ...) do { } while (0)
218#define WTFEndSignpost(pointer, name, ...) do { } while (0)
