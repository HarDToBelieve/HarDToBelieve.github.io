281 enum JoinableState : uint8_t {
282 // The default thread state. The thread can be joined on.
283 Joinable,
284
285 // Somebody waited on this thread to exit and this thread finally exited. This state is here because there can be a
286 // period of time between when the thread exits (which causes pthread_join to return and the remainder of waitOnThreadCompletion to run)
287 // and when threadDidExit is called. We need threadDidExit to take charge and delete the thread data since there's
288 // nobody else to pick up the slack in this case (since waitOnThreadCompletion has already returned).
289 Joined,
290
291 // The thread has been detached and can no longer be joined on. At this point, the thread must take care of cleaning up after itself.
292 Detached,
293 };
294
295 JoinableState joinableState() const { return m_joinableState; }
296 void didBecomeDetached() { m_joinableState = Detached; }
297 void didExit();
298 void didJoin() { m_joinableState = Joined; }
299 bool hasExited() const { return m_didExit; }

305 // For pthread, the Thread instance is ref'ed and held in thread-specific storage. It will be deref'ed by destructTLS at thread destruction time.
306 // It employs pthreads-specific 2-pass destruction to reliably remove Thread.
307
308#if !HAVE(FAST_TLS) && !OS(WINDOWS)
309 static WTF_EXPORT_PRIVATE ThreadSpecificKey s_key;
310 // One time initialization for this class as a whole.
311 // This method must be called before initializeTLS() and it is not thread-safe.
312 static void initializeTLSKey();
313#endif
314 // This thread-specific destructor is called 2 times when thread terminates:
315 // - first, when all the other thread-specific destructors are called, it simply remembers it was 'destroyed once'
316 // and (1) re-sets itself into the thread-specific slot or (2) constructs thread local value to call it again later.
317 // - second, after all thread-specific destructors were invoked, it gets called again - this time, we deref the
318 // Thread in the TLS, completing the cleanup.
319 static void destructTLS(void* data);
320
321 // Creates and puts an instance of Thread into thread-specific storage.
322 static Thread& initializeTLS(Ref<Thread>&&);

400inline Thread& Thread::current()
401{
402 // WRT WebCore:
403 // Thread::current() is used on main thread before it could possibly be used
404 // on secondary ones, so there is no need for synchronization here.
405 // WRT JavaScriptCore:
406 // Thread::initializeTLSKey() is initially called from initialize(), ensuring
407 // this is initially called in a std::call_once locked context.
408#if !HAVE(FAST_TLS) && !OS(WINDOWS)
409 if (UNLIKELY(Thread::s_key == InvalidThreadSpecificKey))
410 WTF::initialize();
411#endif
412 if (auto* thread = currentMayBeNull())
413 return *thread;
414 return initializeCurrentTLS();
415}
