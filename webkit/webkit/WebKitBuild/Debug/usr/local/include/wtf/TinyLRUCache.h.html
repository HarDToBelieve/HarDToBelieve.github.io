26#pragma once
27
28#include <wtf/NeverDestroyed.h>
29#include <wtf/Vector.h>
30
31namespace WTF {
32
33template<typename KeyType, typename ValueType>
34struct TinyLRUCachePolicy {
35 static bool isKeyNull(const KeyType&) { return false; }
36 static ValueType createValueForNullKey() { return { }; }
37 static ValueType createValueForKey(const KeyType&) { return { }; }
38 static KeyType createKeyForStorage(const KeyType& key) { return key; }
39};
40
41template<typename KeyType, typename ValueType, size_t capacity = 4, typename Policy = TinyLRUCachePolicy<KeyType, ValueType>>
42class TinyLRUCache {
43 WTF_MAKE_FAST_ALLOCATED;
44public:
45 const ValueType& get(const KeyType& key)
46 {
47 if (Policy::isKeyNull(key)) {
48 static NeverDestroyed<ValueType> valueForNull = Policy::createValueForNullKey();
49 return valueForNull;
50 }
51
52 for (size_t i = 0; i < m_cache.size(); ++i) {
53 if (m_cache[i].first != key)
54 continue;
55
56 if (i == m_cache.size() - 1)
57 return m_cache[i].second;
58
59 // If the entry is not the last one, move it to the end of the cache.
60 Entry entry = WTFMove(m_cache[i]);
61 m_cache.remove(i);
62 m_cache.append(WTFMove(entry));
63 return m_cache[m_cache.size() - 1].second;
64 }
65
66 // m_cache[0] is the LRU entry, so remove it.
67 if (m_cache.size() == capacity)
68 m_cache.remove(0);
69
70 m_cache.append(std::make_pair(Policy::createKeyForStorage(key), Policy::createValueForKey(key)));
71 return m_cache.last().second;
72 }
73
74private:
75 typedef std::pair<KeyType, ValueType> Entry;
76 typedef Vector<Entry, capacity> Cache;
77 Cache m_cache;
78};
79
80}
