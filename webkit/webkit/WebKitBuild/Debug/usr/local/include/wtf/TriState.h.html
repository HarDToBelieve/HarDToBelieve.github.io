28namespace WTF {
29
30enum class TriState : uint8_t {
31 False,
32 True,
33 Indeterminate
34};
35
36inline TriState triState(bool boolean)
37{
38 return static_cast<TriState>(boolean);
39}
40
41inline TriState invert(TriState triState)
42{
43 if (triState == TriState::True)
44 return TriState::False;
45 if (triState == TriState::False)
46 return TriState::True;
47 return TriState::Indeterminate;
48}
49
50}

53using WTF::triState;
54using WTF::invert;
