286template<typename Encoder> void URL::encode(Encoder& encoder) const
287{
288 encoder << m_string;
289}
290
291template<typename Decoder> bool URL::decode(Decoder& decoder, URL& url)
292{
293 auto optionalURL = decode(decoder);
294 if (!optionalURL)
295 return false;
296 url = WTFMove(*optionalURL);
297 return true;
298}
299
300template<typename Decoder> std::optional<URL> URL::decode(Decoder& decoder)
301{
302 std::optional<String> string;
303 decoder >> string;
304 if (!string)
305 return std::nullopt;
306 return URL(URL(), WTFMove(*string));
307}
