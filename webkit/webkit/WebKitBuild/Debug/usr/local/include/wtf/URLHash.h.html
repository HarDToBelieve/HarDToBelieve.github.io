26#ifndef URLHash_h
27#define URLHash_h
28
29#include <wtf/URL.h>
30#include <wtf/text/StringHash.h>
31#include <wtf/text/WTFString.h>
32
33namespace WTF {
34
35struct URLHash {
36 static unsigned hash(const URL& key)
37 {
38 return key.string().impl()->hash();
39 }
40
41 static bool equal(const URL& a, const URL& b)
42 {
43 return StringHash::equal(a.string(), b.string());
44 }
45
46 static constexpr bool safeToCompareToEmptyOrDeleted = false;
47};
48
49// URLHash is the default hash for String
50template<> struct DefaultHash<URL> : URLHash { };
51
52template<> struct HashTraits<URL> : SimpleClassHashTraits<URL> { };
53
54} // namespace WTF
55
56#endif // URLHash_h
