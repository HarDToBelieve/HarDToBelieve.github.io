28namespace WTF {
29
30// A UnsafePointer<> can be used to hold a pointer whose lifetime is not guaranteed
31// and where the dereferencing of that pointer is therefore unsafe. Once assigned
32// to an UnsafePoniter<>, the pointer itself cannot be extracted from the class, but
33// the class can still be used to test for pointer equality.
34template<typename T>
35class UnsafePointer {
36public:
37 typedef typename std::remove_pointer<T>::type ValueType;
38 typedef ValueType* PtrType;
39
40 UnsafePointer() : m_pointer(nullptr) { }
41 UnsafePointer(PtrType pointer) : m_pointer(pointer) { }
42
43 bool operator==(PtrType pointer) const { return pointer == m_pointer; }
44 bool operator!=(PtrType pointer) const { return pointer != m_pointer; }
45 operator bool() const { return m_pointer; }
46
47private:
48 PtrType m_pointer;
49};
50
51template<typename T>
52bool operator==(typename UnsafePointer<T>::PtrType barePointer, const UnsafePointer<T>& unsafePointer)
53{
54 return unsafePointer == barePointer;
55}
56
57template<typename T>
58bool operator!=(typename UnsafePointer<T>::PtrType barePointer, const UnsafePointer<T>& unsafePointer)
59{
60 return unsafePointer != barePointer;
61}
62
63} // namespace WTF
64
65using WTF::UnsafePointer;
