26#pragma once
27
28#include <wtf/FastMalloc.h>
29
30namespace WTF {
31
32template<typename T> struct ValueCheck {
33 typedef T TraitType;
34 static void checkConsistency(const T&) { }
35};
36
37#if ASSERT_ENABLED
38template<typename P> struct ValueCheck<P*> {
39 typedef P* TraitType;
40 static void checkConsistency(const P* p)
41 {
42 if (!p)
43 return;
44 ASSERT(fastMallocSize(p));
45 ValueCheck<P>::checkConsistency(*p);
46 }
47};
48#endif // ASSERT_ENABLED
49
50}
