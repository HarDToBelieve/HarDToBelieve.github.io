54namespace WTF {
55
56constexpr size_t ConfigAlignment = CeilingOnPageSize;
57constexpr size_t ConfigSizeToProtect = std::max(CeilingOnPageSize, 16 * KB);
58
59struct Config {
60 WTF_EXPORT_PRIVATE static void permanentlyFreeze();
61
62 struct AssertNotFrozenScope {
63 AssertNotFrozenScope();
64 ~AssertNotFrozenScope();
65 };
66
67 // All the fields in this struct should be chosen such that their
68 // initial value is 0 / null / falsy because Config is instantiated
69 // as a global singleton.
70
71 bool isPermanentlyFrozen;
72#if PLATFORM(COCOA)
73 bool disableForwardingVPrintfStdErrToOSLog;
74#endif
75
76#if USE(PTHREADS)
77 bool isUserSpecifiedThreadSuspendResumeSignalConfigured;
78 bool isThreadSuspendResumeSignalConfigured;
79 int sigThreadSuspendResume;
80#endif
81#if OS(UNIX)
82 SignalHandlers signalHandlers;
83#endif
84 PtrTagLookup* ptrTagLookupHead;
85
86 uint64_t spaceForExtensions[1];
87};
88
89#if ENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)
90
91constexpr size_t startSlotOfWTFConfig = Gigacage::reservedSlotsForGigacageConfig;
92constexpr size_t startOffsetOfWTFConfig = startSlotOfWTFConfig * sizeof(WebConfig::Slot);
93
94constexpr size_t offsetOfWTFConfigExtension = startOffsetOfWTFConfig + offsetof(WTF::Config, spaceForExtensions);
95
96constexpr size_t alignmentOfWTFConfig = std::alignment_of<WTF::Config>::value;
97
98static_assert(Gigacage::reservedBytesForGigacageConfig + sizeof(WTF::Config) <= ConfigSizeToProtect);
99static_assert(roundUpToMultipleOf<alignmentOfWTFConfig>(startOffsetOfWTFConfig) == startOffsetOfWTFConfig);
100
101WTF_EXPORT_PRIVATE void setPermissionsOfConfigPage();
102
103#define g_wtfConfig (*bitwise_cast<WTF::Config*>(&WebConfig::g_config[WTF::startSlotOfWTFConfig]))
104
105#else // not ENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)
106
107inline void setPermissionsOfConfigPage() { }
108
109extern "C" WTF_EXPORT_PRIVATE Config g_wtfConfig;
110
111#endif // ENABLE(UNIFIED_AND_FREEZABLE_CONFIG_RECORD)
112
113ALWAYS_INLINE Config::AssertNotFrozenScope::AssertNotFrozenScope()
114{
115 RELEASE_ASSERT(!g_wtfConfig.isPermanentlyFrozen);
116 compilerFence();
117};
118
119ALWAYS_INLINE Config::AssertNotFrozenScope::~AssertNotFrozenScope()
120{
121 compilerFence();
122 RELEASE_ASSERT(!g_wtfConfig.isPermanentlyFrozen);
123};
