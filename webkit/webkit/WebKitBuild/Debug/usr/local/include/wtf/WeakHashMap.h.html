34// Value will be deleted lazily upon rehash or amortized over time. For manual cleanup, call removeNullReferences().
35template<typename KeyType, typename ValueType, typename Counter = EmptyCounter>
36class WeakHashMap final {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39
40 using RefType = Ref<WeakPtrImpl<Counter>>;
41 using KeyTraits = HashTraits<KeyType>;
42 using ValueTraits = HashTraits<ValueType>;
43 using WeakHashImplMap = HashMap<RefType, ValueType>;
44
45 struct PeekKeyValuePairTraits : KeyValuePairHashTraits<HashTraits<KeyType&>, HashTraits<ValueType&>> {
46 static constexpr bool hasIsEmptyValueFunction = true;
47 static bool isEmptyValue(const typename KeyValuePairHashTraits<HashTraits<KeyType&>, HashTraits<ValueType&>>::TraitType& value)
48 {
49 return isHashTraitsEmptyValue<KeyTraits>(value.key);
50 }
51 };
52 using PeekKeyValuePairType = typename PeekKeyValuePairTraits::TraitType;
53
54 struct PeekType {
55 KeyType& key;
56 ValueType& value;
57 };
58
59 struct PeekPtrType {
60 PeekPtrType(const PeekType& peek)
61 : m_peek(peek)
62 { }
63
64 const PeekType& operator*() const { return m_peek; }
65 PeekType& operator*() { return m_peek; }
66
67 const PeekType* operator->() const { return &m_peek; }
68 PeekType* operator->() { return &m_peek; }
69
70 private:
71 PeekType m_peek;
72 };
73
74 template <typename MapType, typename IteratorType, typename IteratorPeekPtrType, typename IteratorPeekType>
75 class WeakHashMapIteratorBase : public std::iterator<std::forward_iterator_tag, IteratorPeekType, std::ptrdiff_t, IteratorPeekPtrType, IteratorPeekType> {
76 protected:
77 WeakHashMapIteratorBase(MapType& weakHashMap, IteratorType position)
78 : m_weakHashMap { weakHashMap }
79 , m_position { position }
80 , m_endPosition { weakHashMap.m_map.end() }
81 {
82 skipEmptyBuckets();
83 }
84
85 ~WeakHashMapIteratorBase()
86 {
87 if (m_emptyBucketCount > m_weakHashMap.m_map.size() / 8)
88 const_cast<WeakHashMap&>(m_weakHashMap).removeNullReferences();
89 else
90 m_weakHashMap.amortizedCleanupIfNeeded(m_advanceCount + m_emptyBucketCount);
91 }
92
93 ALWAYS_INLINE IteratorPeekType makePeek()
94 {
95 auto* entry = m_position.get();
96 auto* key = static_cast<KeyType*>(entry->key->template get<KeyType>());
97 return IteratorPeekType { *key, entry->value };
98 }
99
100 ALWAYS_INLINE IteratorPeekType makePeek() const
101 {
102 auto* entry = m_position.get();
103 auto* key = static_cast<KeyType*>(entry->key->template get<KeyType>());
104 return IteratorPeekType { *key, const_cast<ValueType&>(entry->value) };
105 }
106
107 void advance()
108 {
109 ASSERT(m_position != m_endPosition);
110 ++m_position;
111 ++m_advanceCount;
112 skipEmptyBuckets();
113 }
114
115 void skipEmptyBuckets()
116 {
117 while (m_position != m_endPosition && !m_position->key.get()) {
118 ++m_position;
119 ++m_emptyBucketCount;
120 }
121 }
122
123 const MapType& m_weakHashMap;
124 IteratorType m_position;
125 IteratorType m_endPosition;
126 unsigned m_advanceCount { 0 };
127 unsigned m_emptyBucketCount { 0 };
128 };
129
130 class WeakHashMapIterator : public WeakHashMapIteratorBase<WeakHashMap, typename WeakHashImplMap::iterator, PeekPtrType, PeekType> {
131 public:
132 using Base = WeakHashMapIteratorBase<WeakHashMap, typename WeakHashImplMap::iterator, PeekPtrType, PeekType>;
133
134 bool operator==(const WeakHashMapIterator& other) const { return Base::m_position == other.Base::m_position; }
135 bool operator!=(const WeakHashMapIterator& other) const { return Base::m_position != other.Base::m_position; }
136
137 PeekPtrType get() { return Base::makePeek(); }
138 PeekType operator*() { return Base::makePeek(); }
139 PeekPtrType operator->() { return Base::makePeek(); }
140
141 WeakHashMapIterator& operator++()
142 {
143 Base::advance();
144 return *this;
145 }
146
147 private:
148 WeakHashMapIterator(WeakHashMap& map, typename WeakHashImplMap::iterator position)
149 : Base { map, position }
150 { }
151
152 template <typename, typename, typename> friend class WeakHashMap;
153 };

160 bool operator!=(const WeakHashMapConstIterator& other) const { return Base::m_position != other.Base::m_position; }
161

163 const PeekType operator*() const { return Base::makePeek(); }

165

172 private:
173 WeakHashMapConstIterator(const WeakHashMap& map, typename WeakHashImplMap::const_iterator position)

178 };

180 struct AddResult {
181 AddResult() : isNewEntry(false) { }
182 AddResult(WeakHashMapIterator it, bool isNewEntry) : iterator(it), isNewEntry(isNewEntry) { }

192 iterator begin() { return WeakHashMapIterator(*this, m_map.begin()); }
193 iterator end() { return WeakHashMapIterator(*this, m_map.end()); }
194 const_iterator begin() const { return WeakHashMapConstIterator(*this, m_map.begin()); }

197 template <typename Functor>
198 AddResult ensure(const KeyType& key, Functor&& functor)
199 {
200 amortizedCleanupIfNeeded();
201 auto result = m_map.ensure(makeKeyImpl(key), functor);
202 return AddResult { WeakHashMapIterator(*this, result.iterator), result.isNewEntry };
203 }

205 template<typename T>
206 AddResult add(const KeyType& key, T&& value)
207 {
208 amortizedCleanupIfNeeded();
209 auto addResult = m_map.add(makeKeyImpl(key), std::forward<T>(value));
210 return AddResult { WeakHashMapIterator(*this, addResult.iterator), addResult.isNewEntry };
211 }

213 template<typename T, typename V>
214 void set(const T& key, V&& value)
215 {
216 amortizedCleanupIfNeeded();
217 m_map.set(makeKeyImpl(key), WTFMove(value));

220 iterator find(const KeyType& key)
221 {
222 amortizedCleanupIfNeeded();
223 auto* keyImpl = keyImplIfExists(key);
224 if (!keyImpl)
225 return end();
226 return WeakHashMapIterator(*this, m_map.find(*keyImpl));
227 }

229 const_iterator find(const KeyType& key) const
230 {
231 amortizedCleanupIfNeeded();
232 auto* keyImpl = keyImplIfExists(key);
233 if (!keyImpl)
234 return end();
235 return WeakHashMapConstIterator(*this, m_map.find(*keyImpl));
236 }

247 typename ValueTraits::TakeType take(const KeyType& key)

249 amortizedCleanupIfNeeded();

262 return m_map.get(*keyImpl);

280 template<typename Functor>
281 bool removeIf(Functor&& functor)
282 {
283 m_operationCountSinceLastCleanup = 0;
284 return m_map.removeIf([&](auto& entry) {
285 auto* key = static_cast<KeyType*>(entry.key->template get<KeyType>());
286 bool isReleasedWeakKey = !key;
287 if (isReleasedWeakKey)
288 return true;
289 PeekType peek { *key, entry.value };
290 return functor(peek);
291 });
292 }
293
294 void clear()
295 {
296 m_operationCountSinceLastCleanup = 0;
297 m_map.clear();
298 }
299
300 unsigned capacity() const { return m_map.capacity(); }
301
302 bool isEmptyIgnoringNullReferences() const
303 {
304 auto result = begin() == end();
305 if (UNLIKELY(result && m_map.size()))
306 const_cast<WeakHashMap&>(*this).clear();
307 return result;
308 }
309
310 bool hasNullReferences() const
311 {
312 unsigned count = 0;
313 auto result = WTF::anyOf(m_map, [&] (auto& iterator) {
314 ++count;
315 return !iterator.key.get();
316 });
317 if (result)
318 amortizedCleanupIfNeeded(count);
319 else
320 m_operationCountSinceLastCleanup = 0;
321 return result;
322 }
323
324 unsigned computeSize() const
325 {
326 const_cast<WeakHashMap&>(*this).removeNullReferences();
327 return m_map.size();
328 }
329
330 NEVER_INLINE bool removeNullReferences()
331 {
332 m_operationCountSinceLastCleanup = 0;
333 return m_map.removeIf([](auto& iterator) { return !iterator.key.get(); });
334 }

336#if ASSERT_ENABLED
337 void checkConsistency() const { m_map.checkConsistency(); }
338#else
339 void checkConsistency() const { }
340#endif

342private:
343 ALWAYS_INLINE void amortizedCleanupIfNeeded(unsigned operationsPerformed = 1) const
344 {
345 unsigned currentCount = m_operationCountSinceLastCleanup;
346 m_operationCountSinceLastCleanup = currentCount + operationsPerformed;
347 if (currentCount / 2 > m_map.size())
348 const_cast<WeakHashMap&>(*this).removeNullReferences();
349 }
350
351 template <typename T>
352 static RefType makeKeyImpl(const T& key)
353 {
354 return *makeWeakPtr<KeyType>(const_cast<T&>(key)).m_impl;
355 }
356
357 template <typename T>
358 static WeakPtrImpl<Counter>* keyImplIfExists(const T& key)
359 {
360 auto& weakPtrImpl = key.weakPtrFactory().m_impl;
361 if (!weakPtrImpl || !*weakPtrImpl)
362 return nullptr;
363 return weakPtrImpl.get();
364 }

366 WeakHashImplMap m_map;
