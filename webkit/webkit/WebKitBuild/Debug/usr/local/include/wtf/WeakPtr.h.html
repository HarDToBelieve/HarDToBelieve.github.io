37struct EmptyCounter {
38 static void increment() { }
39 static void decrement() { }
40};
41
42enum class EnableWeakPtrThreadingAssertions : bool { No, Yes };
43
44template<typename Counter = EmptyCounter> class WeakPtrImpl : public ThreadSafeRefCounted<WeakPtrImpl<Counter>> {
45 WTF_MAKE_NONCOPYABLE(WeakPtrImpl);
46 WTF_MAKE_FAST_ALLOCATED;
47public:
48 template<typename T> static Ref<WeakPtrImpl> create(T* ptr)
49 {
50 return adoptRef(*new WeakPtrImpl(ptr));
51 }
52
53 ~WeakPtrImpl()
54 {
55 Counter::decrement();
56 }
57
58 template<typename T> typename T::WeakValueType* get()
59 {
60 return static_cast<typename T::WeakValueType*>(m_ptr);
61 }
62
63 explicit operator bool() const { return m_ptr; }
64 void clear() { m_ptr = nullptr; }
65
66#if ASSERT_ENABLED
67 bool wasConstructedOnMainThread() const { return m_wasConstructedOnMainThread; }
68#endif
69
70private:
71 template<typename T> explicit WeakPtrImpl(T* ptr)
72 : m_ptr(static_cast<typename T::WeakValueType*>(ptr))
73#if ASSERT_ENABLED
74 , m_wasConstructedOnMainThread(isMainThread())
75#endif
76 {
77 Counter::increment();
78 }
79
80 void* m_ptr;
81#if ASSERT_ENABLED
82 bool m_wasConstructedOnMainThread;
83#endif
84};
85
86template<typename Counter> struct HashTraits<Ref<WeakPtrImpl<Counter>>> : RefHashTraits<WeakPtrImpl<Counter>> {
87 static constexpr bool hasIsReleasedWeakValueFunction = true;
88 static bool isReleasedWeakValue(const Ref<WeakPtrImpl<Counter>>& value)
89 {
90 return !value.isHashTableDeletedValue() && !value.isHashTableEmptyValue() && !value.get();
91 }
92};

94template<typename T, typename Counter> class WeakPtr {
95 WTF_MAKE_FAST_ALLOCATED;
96public:
97 WeakPtr() { }
98 WeakPtr(std::nullptr_t) { }
99 template<typename U> WeakPtr(const WeakPtr<U, Counter>&);
100 template<typename U> WeakPtr(WeakPtr<U, Counter>&&);
101
102 T* get() const
103 {
104 // FIXME: Our GC threads currently need to get opaque pointers from WeakPtrs and have to be special-cased.
105 ASSERT(!m_impl || !m_shouldEnableAssertions || Thread::mayBeGCThread() || m_impl->wasConstructedOnMainThread() == isMainThread());
106 return m_impl ? static_cast<T*>(m_impl->template get<T>()) : nullptr;
107 }
108
109 bool operator!() const { return !m_impl || !*m_impl; }
110 explicit operator bool() const { return m_impl && *m_impl; }
111
112 WeakPtr& operator=(std::nullptr_t) { m_impl = nullptr; return *this; }
113 template<typename U> WeakPtr& operator=(const WeakPtr<U, Counter>&);
114 template<typename U> WeakPtr& operator=(WeakPtr<U, Counter>&&);
115
116 T* operator->() const
117 {
118 ASSERT(!m_impl || !m_shouldEnableAssertions || m_impl->wasConstructedOnMainThread() == isMainThread());
119 return get();
120 }
121
122 T& operator*() const
123 {
124 ASSERT(!m_impl || !m_shouldEnableAssertions || m_impl->wasConstructedOnMainThread() == isMainThread());
125 return *get();
126 }
127
128 void clear() { m_impl = nullptr; }
129
130private:
131 template<typename, typename, typename> friend class WeakHashMap;
132 template<typename, typename> friend class WeakHashSet;
133 template<typename, typename> friend class WeakPtr;
134 template<typename, typename> friend class WeakPtrFactory;
135
136 explicit WeakPtr(Ref<WeakPtrImpl<Counter>>&& ref, EnableWeakPtrThreadingAssertions shouldEnableAssertions)
137 : m_impl(WTFMove(ref))
138#if ASSERT_ENABLED
139 , m_shouldEnableAssertions(shouldEnableAssertions == EnableWeakPtrThreadingAssertions::Yes)
140#endif
141 {
142 UNUSED_PARAM(shouldEnableAssertions);
143 }
144
145 RefPtr<WeakPtrImpl<Counter>> m_impl;
146#if ASSERT_ENABLED
147 bool m_shouldEnableAssertions { true };
148#endif
149};

151// Note: you probably want to inherit from CanMakeWeakPtr rather than use this directly.
152template<typename T, typename Counter = EmptyCounter> class WeakPtrFactory {
153 WTF_MAKE_NONCOPYABLE(WeakPtrFactory);
154 WTF_MAKE_FAST_ALLOCATED;
155public:
156 using CounterType = Counter;
157
158 WeakPtrFactory()
159#if ASSERT_ENABLED
160 : m_wasConstructedOnMainThread(isMainThread())
161#endif
162 {
163 }
164
165 ~WeakPtrFactory()
166 {
167 if (!m_impl)
168 return;
169 m_impl->clear();
170 }
171
172 void initializeIfNeeded(const T& object) const
173 {
174 if (m_impl)
175 return;
176
177 ASSERT(m_wasConstructedOnMainThread == isMainThread());
178 m_impl = WeakPtrImpl<Counter>::create(const_cast<T*>(&object));
179 }
180
181 template<typename U> WeakPtr<U, Counter> createWeakPtr(U& object, EnableWeakPtrThreadingAssertions enableWeakPtrThreadingAssertions = EnableWeakPtrThreadingAssertions::Yes) const
182 {
183 initializeIfNeeded(object);
184
185 ASSERT(&object == m_impl->template get<T>());
186 return WeakPtr<U, Counter>(makeRef(*m_impl), enableWeakPtrThreadingAssertions);
187 }
188
189 void revokeAll()
190 {
191 if (!m_impl)
192 return;
193
194 m_impl->clear();
195 m_impl = nullptr;
196 }
197
198 unsigned weakPtrCount() const { return m_impl ? m_impl->refCount() - 1 : 0; }
199
200#if ASSERT_ENABLED
201 bool isInitialized() const { return m_impl; }
202#endif
203
204private:
205 template<typename, typename> friend class WeakHashSet;
206 template<typename, typename, typename> friend class WeakHashMap;
207
208 mutable RefPtr<WeakPtrImpl<Counter>> m_impl;
209#if ASSERT_ENABLED
210 bool m_wasConstructedOnMainThread;
211#endif
212};

214// We use lazy initialization of the WeakPtrFactory by default to avoid unnecessary initialization. Eager
215// initialization is however useful if you plan to call makeWeakPtr() from other threads.
216enum class WeakPtrFactoryInitialization { Lazy, Eager };
217
218template<typename T, WeakPtrFactoryInitialization initializationMode = WeakPtrFactoryInitialization::Lazy, typename Counter = EmptyCounter> class CanMakeWeakPtr {
219public:
220 using WeakValueType = T;
221
222 const WeakPtrFactory<T, Counter>& weakPtrFactory() const { return m_weakPtrFactory; }
223 WeakPtrFactory<T, Counter>& weakPtrFactory() { return m_weakPtrFactory; }
224
225protected:
226 CanMakeWeakPtr()
227 {
228 if (initializationMode == WeakPtrFactoryInitialization::Eager)
229 initializeWeakPtrFactory();
230 }
231
232 void initializeWeakPtrFactory()
233 {
234 m_weakPtrFactory.initializeIfNeeded(static_cast<T&>(*this));
235 }
236
237private:
238 WeakPtrFactory<T, Counter> m_weakPtrFactory;
239};

241template<typename T, typename U, typename Counter> inline WeakPtrImpl<Counter>* weak_ptr_impl_cast(WeakPtrImpl<Counter>* impl)
242{
243 static_assert(std::is_same_v<typename T::WeakValueType, typename U::WeakValueType>, "Invalid weak pointer cast");
244 return impl;
245}

247template<typename T, typename Counter> template<typename U> inline WeakPtr<T, Counter>::WeakPtr(const WeakPtr<U, Counter>& o)
248 : m_impl(weak_ptr_impl_cast<T, U>(o.m_impl.get()))
249{
250}

252template<typename T, typename Counter> template<typename U> inline WeakPtr<T, Counter>::WeakPtr(WeakPtr<U, Counter>&& o)
253 : m_impl(adoptRef(weak_ptr_impl_cast<T, U>(o.m_impl.leakRef())))
254{
255}

257template<typename T, typename Counter> template<typename U> inline WeakPtr<T, Counter>& WeakPtr<T, Counter>::operator=(const WeakPtr<U, Counter>& o)
258{
259 m_impl = weak_ptr_impl_cast<T, U>(o.m_impl.get());
260 return *this;
261}

263template<typename T, typename Counter> template<typename U> inline WeakPtr<T, Counter>& WeakPtr<T, Counter>::operator=(WeakPtr<U, Counter>&& o)
264{
265 m_impl = adoptRef(weak_ptr_impl_cast<T, U>(o.m_impl.leakRef()));
266 return *this;
267}

269template<typename T> inline auto makeWeakPtr(T& object, EnableWeakPtrThreadingAssertions enableWeakPtrThreadingAssertions = EnableWeakPtrThreadingAssertions::Yes)
270{
271 return object.weakPtrFactory().template createWeakPtr<T>(object, enableWeakPtrThreadingAssertions);
272}
273
274template<typename T> inline auto makeWeakPtr(T* ptr, EnableWeakPtrThreadingAssertions enableWeakPtrThreadingAssertions = EnableWeakPtrThreadingAssertions::Yes) -> decltype(makeWeakPtr(*ptr))
275{
276 if (!ptr)
277 return { };
278 return makeWeakPtr(*ptr, enableWeakPtrThreadingAssertions);
279}
