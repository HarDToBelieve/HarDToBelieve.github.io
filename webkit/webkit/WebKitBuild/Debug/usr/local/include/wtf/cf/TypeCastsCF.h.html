31#ifndef CF_BRIDGED_TYPE
32#define CF_BRIDGED_TYPE(T)

35namespace WTF {
36
37template <typename> struct CFTypeTrait;
38
39#define WTF_DECLARE_CF_TYPE_TRAIT(ClassName) \
40template <> \
41struct WTF::CFTypeTrait<ClassName##Ref> { \
42 static inline CFTypeID typeID(void) { return ClassName##GetTypeID(); } \
43};
44
45WTF_DECLARE_CF_TYPE_TRAIT(CFArray);
46WTF_DECLARE_CF_TYPE_TRAIT(CFBoolean);
47WTF_DECLARE_CF_TYPE_TRAIT(CFData);
48WTF_DECLARE_CF_TYPE_TRAIT(CFDictionary);
49WTF_DECLARE_CF_TYPE_TRAIT(CFNumber);
50WTF_DECLARE_CF_TYPE_TRAIT(CFString);
51
52#define WTF_DECLARE_CF_MUTABLE_TYPE_TRAIT(ClassName, MutableClassName) \
53template <> \
54struct WTF::CFTypeTrait<MutableClassName##Ref> { \
55 static inline CFTypeID typeID(void) { return ClassName##GetTypeID(); } \
56};
57
58WTF_DECLARE_CF_MUTABLE_TYPE_TRAIT(CFArray, CFMutableArray);
59WTF_DECLARE_CF_MUTABLE_TYPE_TRAIT(CFData, CFMutableData);
60WTF_DECLARE_CF_MUTABLE_TYPE_TRAIT(CFDictionary, CFMutableDictionary);
61WTF_DECLARE_CF_MUTABLE_TYPE_TRAIT(CFString, CFMutableString);
62
63#undef WTF_DECLARE_CF_MUTABLE_TYPE_TRAIT

65// Use this instead of checked_cf_cast<> when actively checking CF types,
66// similar to dynamic_cast<> in C++. Be sure to include a nullptr check.
67template<typename T> T dynamic_cf_cast(CFTypeRef object)
68{
69 if (!object)
70 return nullptr;
71
72 if (CFGetTypeID(object) != CFTypeTrait<T>::typeID())
73 return nullptr;
74
75 return static_cast<T>(const_cast<CF_BRIDGED_TYPE(id) void*>(object));
76}
77
78// Use this instead of dynamic_cf_cast<> when a specifc CF type is required.
79template<typename T> T checked_cf_cast(CFTypeRef object)
80{
81 if (!object)
82 return nullptr;
83
84 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(CFGetTypeID(object) == CFTypeTrait<T>::typeID());
85
86 return static_cast<T>(const_cast<CF_BRIDGED_TYPE(id) void*>(object));
87}
