143#define SOFT_LINK(framework, functionName, resultType, parameterDeclarations, parameterNames) \
144 WTF_EXTERN_C_BEGIN \
145 resultType functionName parameterDeclarations; \
146 WTF_EXTERN_C_END \
147 static resultType init##functionName parameterDeclarations; \
148 static resultType (*softLink##functionName) parameterDeclarations = init##functionName; \
149 \
150 static resultType init##functionName parameterDeclarations \
151 { \
152 softLink##functionName = (resultType (*) parameterDeclarations) dlsym(framework##Library(), #functionName); \
153 RELEASE_ASSERT_WITH_MESSAGE(softLink##functionName, "%s", dlerror()); \
154 return softLink##functionName parameterNames; \
155 } \
156 \
157 inline __attribute__((__always_inline__)) resultType functionName parameterDeclarations \
158 { \
159 return softLink##functionName parameterNames; \
160 }

162#define SOFT_LINK_MAY_FAIL(framework, functionName, resultType, parameterDeclarations, parameterNames) \
163 WTF_EXTERN_C_BEGIN \
164 resultType functionName parameterDeclarations; \
165 WTF_EXTERN_C_END \
166 static resultType (*softLink##functionName) parameterDeclarations = 0; \
167 \
168 static bool init##functionName() \
169 { \
170 ASSERT(!softLink##functionName); \
171 softLink##functionName = (resultType (*) parameterDeclarations) dlsym(framework##Library(), #functionName); \
172 return !!softLink##functionName; \
173 } \
174 \
175 static bool canLoad##functionName() \
176 { \
177 static bool loaded = init##functionName(); \
178 return loaded; \
179 } \
180 \
181 inline __attribute__((__always_inline__)) __attribute__((visibility("hidden"))) resultType functionName parameterDeclarations \
182 { \
183 ASSERT(softLink##functionName); \
184 return softLink##functionName parameterNames; \
185 }

268 get##name = name##Function; \
269 return pointer##name; \
270 }
