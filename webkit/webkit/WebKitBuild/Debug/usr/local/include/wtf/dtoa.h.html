50inline double parseDouble(const LChar* string, size_t length, size_t& parsedLength)
51{
52 return double_conversion::StringToDoubleConverter::StringToDouble(reinterpret_cast<const char*>(string), length, &parsedLength);
53}
54
55inline double parseDouble(const UChar* string, size_t length, size_t& parsedLength)
56{
57 const size_t conversionBufferSize = 64;
58 if (length > conversionBufferSize)
59 return Internal::parseDoubleFromLongString(string, length, parsedLength);
60 LChar conversionBuffer[conversionBufferSize];
61 for (int i = 0; i < static_cast<int>(length); ++i)
62 conversionBuffer[i] = isASCII(string[i]) ? string[i] : 0;
63 return parseDouble(conversionBuffer, length, parsedLength);
64}
65
66inline double parseDouble(StringView string, size_t& parsedLength)
67{
68 if (string.is8Bit())
69 return parseDouble(string.characters8(), string.length(), parsedLength);
70 return parseDouble(string.characters16(), string.length(), parsedLength);
71}
