32namespace WTF {
33namespace Persistence {
34
35class Decoder {
36 WTF_MAKE_FAST_ALLOCATED;
37public:
38 WTF_EXPORT_PRIVATE Decoder(const uint8_t* buffer, size_t bufferSize);
39 WTF_EXPORT_PRIVATE ~Decoder();
40
41 size_t length() const { return m_bufferEnd - m_buffer; }
42 size_t currentOffset() const { return m_bufferPosition - m_buffer; }
43
44 WTF_EXPORT_PRIVATE WARN_UNUSED_RETURN bool rewind(size_t);
45
46 WTF_EXPORT_PRIVATE WARN_UNUSED_RETURN bool verifyChecksum();
47
48 WTF_EXPORT_PRIVATE WARN_UNUSED_RETURN bool decodeFixedLengthData(uint8_t*, size_t);
49
50 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<bool>&);
51 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<uint8_t>&);
52 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<uint16_t>&);
53 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<uint32_t>&);
54 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<uint64_t>&);
55 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<int16_t>&);
56 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<int32_t>&);
57 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<int64_t>&);
58 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<float>&);
59 WTF_EXPORT_PRIVATE Decoder& operator>>(std::optional<double>&);
60
61 template<typename T, std::enable_if_t<!std::is_arithmetic<typename std::remove_const<T>>::value && !std::is_enum<T>::value>* = nullptr>
62 Decoder& operator>>(std::optional<T>& result)
63 {
64 result = Coder<T>::decode(*this);
65 return *this;
66 }
67
68 template<typename E, std::enable_if_t<std::is_enum<E>::value>* = nullptr>
69 Decoder& operator>>(std::optional<E>& result)
70 {
71 static_assert(sizeof(E) <= 8, "Enum type T must not be larger than 64 bits!");
72 std::optional<uint64_t> value;
73 *this >> value;
74 if (!value)
75 return *this;
76 if (!isValidEnum<E>(*value))
77 return *this;
78 result = static_cast<E>(*value);
79 return *this;
80 }
81
82 template<typename T> WARN_UNUSED_RETURN
83 bool bufferIsLargeEnoughToContain(size_t numElements) const
84 {
85 static_assert(std::is_arithmetic<T>::value, "Type T must have a fixed, known encoded size!");
86 return numElements <= std::numeric_limits<size_t>::max() / sizeof(T) && bufferIsLargeEnoughToContain(numElements * sizeof(T));
87 }
88
89 WTF_EXPORT_PRIVATE WARN_UNUSED_RETURN const uint8_t* bufferPointerForDirectRead(size_t numBytes);
90
91 static constexpr bool isIPCDecoder = false;
92
93private:
94 WTF_EXPORT_PRIVATE WARN_UNUSED_RETURN bool bufferIsLargeEnoughToContain(size_t) const;
95 template<typename Type> Decoder& decodeNumber(std::optional<Type>&);
96
97 const uint8_t* m_buffer;
98 const uint8_t* m_bufferPosition;
99 const uint8_t* m_bufferEnd;
100
101 SHA1 m_sha1;
102};
