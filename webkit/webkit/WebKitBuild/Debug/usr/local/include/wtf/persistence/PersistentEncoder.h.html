33namespace WTF {
34namespace Persistence {
35
36class Encoder;
37class DataReference;
38
39class Encoder {
40 WTF_MAKE_FAST_ALLOCATED;
41public:
42 WTF_EXPORT_PRIVATE Encoder();
43 WTF_EXPORT_PRIVATE ~Encoder();
44
45 WTF_EXPORT_PRIVATE void encodeChecksum();
46 WTF_EXPORT_PRIVATE void encodeFixedLengthData(const uint8_t*, size_t);
47
48 template<typename T, std::enable_if_t<std::is_enum<T>::value>* = nullptr>
49 Encoder& operator<<(const T& t)
50 {
51 static_assert(sizeof(T) <= sizeof(uint64_t), "Enum type must not be larger than 64 bits.");
52 return *this << static_cast<uint64_t>(t);
53 }
54
55 template<typename T, std::enable_if_t<!std::is_enum<T>::value && !std::is_arithmetic<typename std::remove_const<T>>::value>* = nullptr>
56 Encoder& operator<<(const T& t)
57 {
58 Coder<T>::encode(*this, t);
59 return *this;
60 }
61
62 WTF_EXPORT_PRIVATE Encoder& operator<<(bool);
63 WTF_EXPORT_PRIVATE Encoder& operator<<(uint8_t);
64 WTF_EXPORT_PRIVATE Encoder& operator<<(uint16_t);
65 WTF_EXPORT_PRIVATE Encoder& operator<<(uint32_t);
66 WTF_EXPORT_PRIVATE Encoder& operator<<(uint64_t);
67 WTF_EXPORT_PRIVATE Encoder& operator<<(int16_t);
68 WTF_EXPORT_PRIVATE Encoder& operator<<(int32_t);
69 WTF_EXPORT_PRIVATE Encoder& operator<<(int64_t);
70 WTF_EXPORT_PRIVATE Encoder& operator<<(float);
71 WTF_EXPORT_PRIVATE Encoder& operator<<(double);
72
73 const uint8_t* buffer() const { return m_buffer.data(); }
74 size_t bufferSize() const { return m_buffer.size(); }
75
76 WTF_EXPORT_PRIVATE static void updateChecksumForData(SHA1&, const uint8_t*, size_t);
77 template <typename Type> static void updateChecksumForNumber(SHA1&, Type);
78
79 static constexpr bool isIPCEncoder = false;
80
81private:
82
83 template<typename Type> Encoder& encodeNumber(Type);
84
85 uint8_t* grow(size_t);
86
87 template <typename Type> struct Salt;
88
89 Vector<uint8_t, 4096> m_buffer;
90 SHA1 m_sha1;
91};
92
93template <> struct Encoder::Salt<bool> { static constexpr unsigned value = 3; };
94template <> struct Encoder::Salt<uint8_t> { static constexpr unsigned value = 5; };
95template <> struct Encoder::Salt<uint16_t> { static constexpr unsigned value = 7; };
96template <> struct Encoder::Salt<uint32_t> { static constexpr unsigned value = 11; };
97template <> struct Encoder::Salt<uint64_t> { static constexpr unsigned value = 13; };
98template <> struct Encoder::Salt<int32_t> { static constexpr unsigned value = 17; };
99template <> struct Encoder::Salt<int64_t> { static constexpr unsigned value = 19; };
100template <> struct Encoder::Salt<float> { static constexpr unsigned value = 23; };
101template <> struct Encoder::Salt<double> { static constexpr unsigned value = 29; };
102template <> struct Encoder::Salt<uint8_t*> { static constexpr unsigned value = 101; };
103template <> struct Encoder::Salt<int16_t> { static constexpr unsigned value = 103; };
104
105template <typename Type>
106void Encoder::updateChecksumForNumber(SHA1& sha1, Type value)
107{
108 auto typeSalt = Salt<Type>::value;
109 sha1.addBytes(reinterpret_cast<uint8_t*>(&typeSalt), sizeof(typeSalt));
110 sha1.addBytes(reinterpret_cast<uint8_t*>(&value), sizeof(value));
111}
