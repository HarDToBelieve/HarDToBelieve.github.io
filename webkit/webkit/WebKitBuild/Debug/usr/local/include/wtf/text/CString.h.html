36DECLARE_ALLOCATOR_WITH_HEAP_IDENTIFIER(CStringBuffer);
37
38// CStringBuffer is the ref-counted storage class for the characters in a CString.
39// The data is implicitly allocated 1 character longer than length(), as it is zero-terminated.
40class CStringBuffer final : public RefCounted<CStringBuffer> {
41 WTF_MAKE_FAST_ALLOCATED_WITH_HEAP_IDENTIFIER(CStringBuffer);
42public:
43 const char* data() { return mutableData(); }
44 size_t length() const { return m_length; }
45
46private:
47 friend class CString;
48
49 static Ref<CStringBuffer> createUninitialized(size_t length);
50
51 CStringBuffer(size_t length) : m_length(length) { }
52 char* mutableData() { return reinterpret_cast_ptr<char*>(this + 1); }
53
54 const size_t m_length;
55};
56
57// A container for a null-terminated char array supporting copy-on-write
58// assignment. The contained char array may be null.
59class CString final {
60 WTF_MAKE_FAST_ALLOCATED;
61public:
62 CString() { }
63 WTF_EXPORT_PRIVATE CString(const char*);
64 WTF_EXPORT_PRIVATE CString(const char*, size_t length);
65 CString(const uint8_t* data, size_t length) : CString(reinterpret_cast<const char*>(data), length) { }
66 CString(CStringBuffer* buffer) : m_buffer(buffer) { }
67 WTF_EXPORT_PRIVATE static CString newUninitialized(size_t length, char*& characterBuffer);
68 CString(HashTableDeletedValueType) : m_buffer(HashTableDeletedValue) { }
69
70 const char* data() const
71 {
72 return m_buffer ? m_buffer->data() : nullptr;
73 }
74
75 const uint8_t* dataAsUInt8Ptr() const { return reinterpret_cast<const uint8_t*>(data()); }
76
77 Span<const uint8_t> bytes() const
78 {
79 if (m_buffer)
80 return { reinterpret_cast<const uint8_t*>(m_buffer->data()), m_buffer->length() };
81 return { };
82 }
83
84 Span<const uint8_t> bytesInludingNullTerminator() const
85 {
86 if (m_buffer)
87 return { reinterpret_cast<const uint8_t*>(m_buffer->data()), m_buffer->length() + 1 };
88 return { };
89 }
90
91 WTF_EXPORT_PRIVATE char* mutableData();
92 size_t length() const
93 {
94 return m_buffer ? m_buffer->length() : 0;
95 }
96
97 bool isNull() const { return !m_buffer; }
98 bool isSafeToSendToAnotherThread() const;
99
100 CStringBuffer* buffer() const { return m_buffer.get(); }
101
102 bool isHashTableDeletedValue() const { return m_buffer.isHashTableDeletedValue(); }
103
104 WTF_EXPORT_PRIVATE unsigned hash() const;
105
106 // Useful if you want your CString to hold dynamic data.
107 WTF_EXPORT_PRIVATE void grow(size_t newLength);
108
109private:
110 void copyBufferIfNeeded();
111 void init(const char*, size_t length);
112 RefPtr<CStringBuffer> m_buffer;
113};

115WTF_EXPORT_PRIVATE bool operator==(const CString& a, const CString& b);
116inline bool operator!=(const CString& a, const CString& b) { return !(a == b); }
117WTF_EXPORT_PRIVATE bool operator==(const CString& a, const char* b);
118inline bool operator!=(const CString& a, const char* b) { return !(a == b); }
119WTF_EXPORT_PRIVATE bool operator<(const CString& a, const CString& b);

121struct CStringHash {
122 static unsigned hash(const CString& string) { return string.hash(); }
123 WTF_EXPORT_PRIVATE static bool equal(const CString& a, const CString& b);
124 static constexpr bool safeToCompareToEmptyOrDeleted = true;
125};
