26namespace WTF {

28enum PositiveOrNegativeNumber { PositiveNumber, NegativeNumber };
29
30template<typename> struct IntegerToStringConversionTrait;
31
32template<typename T, typename UnsignedIntegerType, PositiveOrNegativeNumber NumberType, typename AdditionalArgumentType>
33static typename IntegerToStringConversionTrait<T>::ReturnType numberToStringImpl(UnsignedIntegerType number, AdditionalArgumentType additionalArgument)
34{
35 LChar buf[sizeof(UnsignedIntegerType) * 3 + 1];
36 LChar* end = std::end(buf);
37 LChar* p = end;
38
39 do {
40 *--p = static_cast<LChar>((number % 10) + '0');
41 number /= 10;
42 } while (number);
43
44 if (NumberType == NegativeNumber)
45 *--p = '-';
46
47 return IntegerToStringConversionTrait<T>::flush(p, static_cast<unsigned>(end - p), additionalArgument);
48}
49
50template<typename T, typename SignedIntegerType>
51inline typename IntegerToStringConversionTrait<T>::ReturnType numberToStringSigned(SignedIntegerType number, typename IntegerToStringConversionTrait<T>::AdditionalArgumentType* additionalArgument = nullptr)
52{
53 if (number < 0)
54 return numberToStringImpl<T, typename std::make_unsigned_t<SignedIntegerType>, NegativeNumber>(-static_cast<typename std::make_unsigned_t<SignedIntegerType>>(number), additionalArgument);
55 return numberToStringImpl<T, typename std::make_unsigned_t<SignedIntegerType>, PositiveNumber>(number, additionalArgument);
56}
57
58template<typename T, typename UnsignedIntegerType>
59inline typename IntegerToStringConversionTrait<T>::ReturnType numberToStringUnsigned(UnsignedIntegerType number, typename IntegerToStringConversionTrait<T>::AdditionalArgumentType* additionalArgument = nullptr)
60{
61 return numberToStringImpl<T, UnsignedIntegerType, PositiveNumber>(number, additionalArgument);
62}

64template<typename CharacterType, typename UnsignedIntegerType, PositiveOrNegativeNumber NumberType>
65static void writeIntegerToBufferImpl(UnsignedIntegerType number, CharacterType* destination)
66{
67 static_assert(!std::is_same_v<bool, std::remove_cv_t<UnsignedIntegerType>>, "'bool' not supported");
68 LChar buf[sizeof(UnsignedIntegerType) * 3 + 1];
69 LChar* end = std::end(buf);
70 LChar* p = end;
71
72 do {
73 *--p = static_cast<LChar>((number % 10) + '0');
74 number /= 10;
75 } while (number);
76
77 if (NumberType == NegativeNumber)
78 *--p = '-';
79
80 while (p < end)
81 *destination++ = static_cast<CharacterType>(*p++);
82}
83
84template<typename CharacterType, typename IntegerType>
85inline void writeIntegerToBuffer(IntegerType integer, CharacterType* destination)
86{
87 static_assert(std::is_integral_v<IntegerType>);
88 if constexpr (std::is_same_v<IntegerType, bool>)
89 return writeIntegerToBufferImpl<CharacterType, uint8_t, PositiveNumber>(integer ? 1 : 0, destination);
90 else if constexpr (std::is_signed_v<IntegerType>) {
91 if (integer < 0)
92 return writeIntegerToBufferImpl<CharacterType, typename std::make_unsigned_t<IntegerType>, NegativeNumber>(-integer, destination);
93 return writeIntegerToBufferImpl<CharacterType, typename std::make_unsigned_t<IntegerType>, PositiveNumber>(integer, destination);
94 } else
95 return writeIntegerToBufferImpl<CharacterType, IntegerType, PositiveNumber>(integer, destination);
96}

98template<typename UnsignedIntegerType, PositiveOrNegativeNumber NumberType>
99constexpr unsigned lengthOfIntegerAsStringImpl(UnsignedIntegerType number)
100{
101 unsigned length = 0;
102
103 do {
104 ++length;
105 number /= 10;
106 } while (number);
107
108 if (NumberType == NegativeNumber)
109 ++length;
110
111 return length;
112}
113
114template<typename IntegerType>
115constexpr unsigned lengthOfIntegerAsString(IntegerType integer)
116{
117 static_assert(std::is_integral_v<IntegerType>);
118 if constexpr (std::is_same_v<IntegerType, bool>) {
119 UNUSED_PARAM(integer);
120 return 1;
121 }
122 else if constexpr (std::is_signed_v<IntegerType>) {
123 if (integer < 0)
124 return lengthOfIntegerAsStringImpl<typename std::make_unsigned_t<IntegerType>, NegativeNumber>(-integer);
125 return lengthOfIntegerAsStringImpl<typename std::make_unsigned_t<IntegerType>, PositiveNumber>(integer);
126 } else
127 return lengthOfIntegerAsStringImpl<IntegerType, PositiveNumber>(integer);
128}
