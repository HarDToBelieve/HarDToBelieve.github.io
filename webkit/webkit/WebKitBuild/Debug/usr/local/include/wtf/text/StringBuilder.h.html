134inline StringBuilder::StringBuilder(OverflowHandler policy)
135 : m_shouldCrashOnOverflow { policy == OverflowHandler::CrashOnOverflow }
136{
137}

139inline void StringBuilder::clear()
140{
141 m_string = { };
142 m_buffer = nullptr;
143 m_length = 0;
144 // We intentionally do not change m_shouldCrashOnOverflow.
145}
146
147inline void StringBuilder::swap(StringBuilder& other)
148{
149 m_string.swap(other.m_string);
150 m_buffer.swap(other.m_buffer);
151 std::swap(m_length, other.m_length);
152 std::swap(m_shouldCrashOnOverflow, other.m_shouldCrashOnOverflow);
153}

155inline StringBuilder::operator StringView() const
156{
157 if (is8Bit())
158 return { characters<LChar>(), length() };
159 return { characters<UChar>(), length() };
160}
161
162inline void StringBuilder::append(UChar character)
163{
164 if (m_buffer && m_length < m_buffer->length() && m_string.isNull()) {
165 if (!m_buffer->is8Bit()) {
166 const_cast<UChar*>(m_buffer->characters<UChar>())[m_length++] = character;
167 return;
168 }
169 if (isLatin1(character)) {
170 const_cast<LChar*>(m_buffer->characters<LChar>())[m_length++] = static_cast<LChar>(character);
171 return;
172 }
173 }
174 appendCharacters(&character, 1);
175}
176
177inline void StringBuilder::append(LChar character)
178{
179 if (m_buffer && m_length < m_buffer->length() && m_string.isNull()) {
180 if (m_buffer->is8Bit())
181 const_cast<LChar*>(m_buffer->characters<LChar>())[m_length++] = character;
182 else
183 const_cast<UChar*>(m_buffer->characters<UChar>())[m_length++] = character;
184 return;
185 }
186 appendCharacters(&character, 1);
187}

189inline void StringBuilder::append(const String& string)
190{
191 // If we're appending to an empty string, and there is not a buffer (reserveCapacity has not been called)
192 // then just retain the string.
193 if (!m_length && !m_buffer) {
194 m_string = string;
195 m_length = string.length();
196 return;
197 }
198
199 append(StringView { string });
200}
201
202inline void StringBuilder::append(const StringBuilder& other)
203{
204 // If we're appending to an empty string, and there is not a buffer (reserveCapacity has not been called)
205 // then just retain the string.
206 if (!m_length && !m_buffer && !other.m_string.isNull()) {
207 // Use the length function here so we crash on overflow without explicit overflow checks.
208 m_string = other.m_string;
209 m_length = other.length();
210 return;
211 }
212
213 append(StringView { other });
214}

216inline void StringBuilder::append(StringView string)
217{
218 if (string.is8Bit())
219 appendCharacters(string.characters8(), string.length());
220 else
221 appendCharacters(string.characters16(), string.length());
222}

224inline void StringBuilder::appendSubstring(const String& string, unsigned offset, unsigned length)
225{
226 append(StringView { string }.substring(offset, length));
227}

229inline void StringBuilder::append(const char* characters)
230{
231 append(StringView { characters });
232}

234inline void StringBuilder::appendCharacter(UChar32 c)
235{
236 if (U_IS_BMP(c)) {
237 append(static_cast<UChar>(c));
238 return;
239 }
240 append(U16_LEAD(c));
241 append(U16_TRAIL(c));
242}

244inline String StringBuilder::toString()
245{
246 if (m_string.isNull()) {
247 shrinkToFit();
248 reifyString();
249 }
250 return m_string;
251}

253inline const String& StringBuilder::toStringPreserveCapacity() const
254{
255 if (m_string.isNull())
256 reifyString();
257 return m_string;
258}

260inline AtomString StringBuilder::toAtomString() const
261{
262 if (isEmpty())
263 return emptyAtom();
264
265 // If the buffer is sufficiently over-allocated, make a new AtomString from a copy so its buffer is not so large.
266 if (shouldShrinkToFit())
267 return StringView { *this }.toAtomString();
268
269 if (!m_string.isNull())
270 return m_string;
271
272 // Use the length function here so we crash on overflow without explicit overflow checks.
273 ASSERT(m_buffer);
274 return { m_buffer.get(), 0, length() };
275}

279 RELEASE_ASSERT(!hasOverflowed());

288inline UChar StringBuilder::operator[](unsigned i) const
289{
290 RELEASE_ASSERT_WITH_SECURITY_IMPLICATION(i < length());
291 return is8Bit() ? characters<LChar>()[i] : characters<UChar>()[i];
292}

294inline bool StringBuilder::is8Bit() const
295{
296 return m_buffer ? m_buffer->is8Bit() : m_string.is8Bit();
297}

299template<typename CharacterType> inline const CharacterType* StringBuilder::characters() const
300{
301 if (!m_length)
302 return nullptr;
303 if (!m_string.isNull())
304 return m_string.characters<CharacterType>();
305 return m_buffer->characters<CharacterType>();
306}

308template<typename... StringTypeAdapters> void StringBuilder::appendFromAdapters(StringTypeAdapters... adapters)
309{
310 auto requiredLength = saturatedSum<uint32_t>(m_length, adapters.length()...);
311 if (is8Bit() && are8Bit(adapters...)) {
312 auto destination = extendBufferForAppendingLChar(requiredLength);
313 if (!destination)
314 return;
315 stringTypeAdapterAccumulator(destination, adapters...);
316 } else {
317 auto destination = extendBufferForAppendingWithUpconvert(requiredLength);
318 if (!destination)
319 return;
320 stringTypeAdapterAccumulator(destination, adapters...);
321 }
322}
323
324template<typename... StringTypes> void StringBuilder::append(StringTypes... strings)
325{
326 appendFromAdapters(StringTypeAdapter<StringTypes>(strings)...);
327}
328
329template<typename CharacterType> bool equal(const StringBuilder& builder, const CharacterType* buffer, unsigned length)
330{
331 return builder == StringView { buffer, length };
332}
