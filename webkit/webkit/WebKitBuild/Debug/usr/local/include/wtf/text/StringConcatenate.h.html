78template<> class StringTypeAdapter<const LChar*, void> {
79public:
80 StringTypeAdapter(const LChar* characters)
81 : m_characters { characters }
82 , m_length { computeLength(characters) }
83 {
84 }
85
86 unsigned length() const { return m_length; }
87 bool is8Bit() const { return true; }
88 template<typename CharacterType> void writeTo(CharacterType* destination) const { StringImpl::copyCharacters(destination, m_characters, m_length); }
89
90private:
91 static unsigned computeLength(const LChar* characters)
92 {
93 size_t length = std::strlen(reinterpret_cast<const char*>(characters));
94 RELEASE_ASSERT(length <= String::MaxLength);
95 return static_cast<unsigned>(length);
96 }
97
98 const LChar* m_characters;
99 unsigned m_length;
100};

102template<> class StringTypeAdapter<const UChar*, void> {
103public:
104 StringTypeAdapter(const UChar* characters)
105 : m_characters { characters }
106 , m_length { computeLength(characters) }
107 {
108 }
109
110 unsigned length() const { return m_length; }
111 bool is8Bit() const { return !m_length; }
112 void writeTo(LChar*) const { ASSERT(!m_length); }
113 void writeTo(UChar* destination) const { StringImpl::copyCharacters(destination, m_characters, m_length); }
114
115private:
116 static unsigned computeLength(const UChar* characters)
117 {
118 size_t length = 0;
119 while (characters[length])
120 ++length;
121 RELEASE_ASSERT(length <= String::MaxLength);
122 return static_cast<unsigned>(length);
123 }
124
125 const UChar* m_characters;
126 unsigned m_length;
127};

243template<typename... StringTypes> class StringTypeAdapter<std::tuple<StringTypes...>, void> {
244public:
245 StringTypeAdapter(std::tuple<StringTypes...> tuple)
246 : m_tuple { tuple }
247 , m_length { std::apply(computeLength, tuple) }
248 , m_is8Bit { std::apply(computeIs8Bit, tuple) }
249 {
250 }
251
252 unsigned length() const { return m_length; }
253 bool is8Bit() const { return m_is8Bit; }
254 template<typename CharacterType> void writeTo(CharacterType* destination) const
255 {
256 std::apply([&](StringTypes... strings) {
257 unsigned offset = 0;
258 (..., (
259 StringTypeAdapter<StringTypes>(strings).writeTo(destination + (offset * sizeof(CharacterType))),
260 offset += StringTypeAdapter<StringTypes>(strings).length()
261 ));
262 }, m_tuple);
263 }
264
265private:
266 static unsigned computeLength(StringTypes... strings)
267 {
268 return (... + StringTypeAdapter<StringTypes>(strings).length());
269 }
270
271 static bool computeIs8Bit(StringTypes... strings)
272 {
273 return (... && StringTypeAdapter<StringTypes>(strings).is8Bit());
274 }
275
276 std::tuple<StringTypes...> m_tuple;
277 unsigned m_length;
278 bool m_is8Bit;
279};

281template<typename UnderlyingElementType> struct PaddingSpecification {
282 LChar character;
283 unsigned length;
284 UnderlyingElementType underlyingElement;
285};
286
287template<typename UnderlyingElementType> PaddingSpecification<UnderlyingElementType> pad(char character, unsigned length, UnderlyingElementType element)
288{
289 return { static_cast<LChar>(character), length, element };
290}

292template<typename UnderlyingElementType> class StringTypeAdapter<PaddingSpecification<UnderlyingElementType>> {
293public:
294 StringTypeAdapter(const PaddingSpecification<UnderlyingElementType>& padding)
295 : m_padding { padding }
296 , m_underlyingAdapter { m_padding.underlyingElement }
297 {
298 }
299
300 unsigned length() const { return std::max(m_padding.length, m_underlyingAdapter.length()); }
301 bool is8Bit() const { return m_underlyingAdapter.is8Bit(); }
302 template<typename CharacterType> void writeTo(CharacterType* destination) const
303 {
304 unsigned underlyingLength = m_underlyingAdapter.length();
305 unsigned count = 0;
306 if (underlyingLength < m_padding.length) {
307 count = m_padding.length - underlyingLength;
308 for (unsigned i = 0; i < count; ++i)
309 destination[i] = m_padding.character;
310 }
311 m_underlyingAdapter.writeTo(destination + count);
312 }
313
314private:
315 const PaddingSpecification<UnderlyingElementType>& m_padding;
316 StringTypeAdapter<UnderlyingElementType> m_underlyingAdapter;
317};

319template<unsigned N>
320struct Indentation {
321 unsigned operator++() { return ++value; }
322 unsigned operator++(int) { return value++; }
323 unsigned operator--() { return --value; }
324 unsigned operator--(int) { return value--; }

326 unsigned value { 0 };

404template<typename Adapter>
405inline bool are8Bit(Adapter adapter)
406{
407 return adapter.is8Bit();
408}

410template<typename Adapter, typename... Adapters>
411inline bool are8Bit(Adapter adapter, Adapters ...adapters)
412{
413 return adapter.is8Bit() && are8Bit(adapters...);
414}

416template<typename ResultType, typename Adapter>
417inline void stringTypeAdapterAccumulator(ResultType* result, Adapter adapter)
418{
419 adapter.writeTo(result);
420}
421
422template<typename ResultType, typename Adapter, typename... Adapters>
423inline void stringTypeAdapterAccumulator(ResultType* result, Adapter adapter, Adapters ...adapters)
424{
425 adapter.writeTo(result);
426 stringTypeAdapterAccumulator(result + adapter.length(), adapters...);
427}

429template<typename StringTypeAdapter, typename... StringTypeAdapters>
430String tryMakeStringFromAdapters(StringTypeAdapter adapter, StringTypeAdapters ...adapters)
431{
432 static_assert(String::MaxLength == std::numeric_limits<int32_t>::max(), "");
433 auto sum = checkedSum<int32_t>(adapter.length(), adapters.length()...);
434 if (sum.hasOverflowed())
435 return String();
436
437 unsigned length = sum;
438 ASSERT(length <= String::MaxLength);
439 if (are8Bit(adapter, adapters...)) {
440 LChar* buffer;
441 RefPtr<StringImpl> resultImpl = StringImpl::tryCreateUninitialized(length, buffer);
442 if (!resultImpl)
443 return String();
444
445 stringTypeAdapterAccumulator(buffer, adapter, adapters...);
446
447 return resultImpl;
448 }
449
450 UChar* buffer;
451 RefPtr<StringImpl> resultImpl = StringImpl::tryCreateUninitialized(length, buffer);
452 if (!resultImpl)
453 return String();
454
455 stringTypeAdapterAccumulator(buffer, adapter, adapters...);
456
457 return resultImpl;
458}

460template<typename... StringTypes>
461String tryMakeString(StringTypes ...strings)
462{
463 return tryMakeStringFromAdapters(StringTypeAdapter<StringTypes>(strings)...);
464}
465
466template<typename... StringTypes>
467String makeString(StringTypes... strings)
468{
469 String result = tryMakeString(strings...);
470 if (!result)
471 CRASH();
472 return result;
473}
