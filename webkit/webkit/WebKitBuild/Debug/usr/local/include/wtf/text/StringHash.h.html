43 // The hash() functions on StringHash and ASCIICaseInsensitiveHash do not support
44 // null strings. get(), contains(), and add() on HashMap<String,..., StringHash>
45 // cause a null-pointer dereference when passed null strings.
46
47 // FIXME: We should really figure out a way to put the computeHash function that's
48 // currently a member function of StringImpl into this file so we can be a little
49 // closer to having all the nearly-identical hash functions in one place.
50
51 struct StringHash {
52 static unsigned hash(StringImpl* key) { return key->hash(); }
53 static inline bool equal(const StringImpl* a, const StringImpl* b)
54 {
55 return WTF::equal(*a, *b);
56 }
57
58 static unsigned hash(const RefPtr<StringImpl>& key) { return key->hash(); }
59 static unsigned hash(const PackedPtr<StringImpl>& key) { return key->hash(); }
60 static bool equal(const RefPtr<StringImpl>& a, const RefPtr<StringImpl>& b)
61 {
62 return equal(a.get(), b.get());
63 }
64 static bool equal(const RefPtr<StringImpl>& a, const StringImpl* b)
65 {
66 return equal(a.get(), b);
67 }
68 static bool equal(const StringImpl* a, const RefPtr<StringImpl>& b)
69 {
70 return equal(a, b.get());
71 }
72
73 static bool equal(const PackedPtr<StringImpl>& a, const PackedPtr<StringImpl>& b)
74 {
75 return equal(a.get(), b.get());
76 }
77 static bool equal(const PackedPtr<StringImpl>& a, const StringImpl* b)
78 {
79 return equal(a.get(), b);
80 }
81 static bool equal(const StringImpl* a, const PackedPtr<StringImpl>& b)
82 {
83 return equal(a, b.get());
84 }
85
86 static unsigned hash(const String& key) { return key.impl()->hash(); }
87 static bool equal(const String& a, const String& b)
88 {
89 return equal(a.impl(), b.impl());
90 }
91
92 static constexpr bool safeToCompareToEmptyOrDeleted = false;
93 static constexpr bool hasHashInValue = true;
94 };

96 struct ASCIICaseInsensitiveHash {
97 template<typename T>
98 struct FoldCase {
99 static inline UChar convert(T character)
100 {
101 return toASCIILower(character);
102 }
103 };
104
105 static unsigned hash(const UChar* data, unsigned length)
106 {
107 return StringHasher::computeHashAndMaskTop8Bits<UChar, FoldCase<UChar>>(data, length);
108 }
109
110 static unsigned hash(StringImpl& string)
111 {
112 if (string.is8Bit())
113 return hash(string.characters8(), string.length());
114 return hash(string.characters16(), string.length());
115 }
116 static unsigned hash(StringImpl* string)
117 {
118 ASSERT(string);
119 return hash(*string);
120 }
121
122 static unsigned hash(const LChar* data, unsigned length)
123 {
124 return StringHasher::computeHashAndMaskTop8Bits<LChar, FoldCase<LChar>>(data, length);
125 }
126
127 static inline unsigned hash(const char* data, unsigned length)
128 {
129 return hash(reinterpret_cast<const LChar*>(data), length);
130 }
131
132 static inline bool equal(const StringImpl& a, const StringImpl& b)
133 {
134 return equalIgnoringASCIICase(a, b);
135 }
136 static inline bool equal(const StringImpl* a, const StringImpl* b)
137 {
138 ASSERT(a);
139 ASSERT(b);
140 return equal(*a, *b);
141 }
142
143 static unsigned hash(const RefPtr<StringImpl>& key)
144 {
145 return hash(key.get());
146 }
147
148 static bool equal(const RefPtr<StringImpl>& a, const RefPtr<StringImpl>& b)
149 {
150 return equal(a.get(), b.get());
151 }
152
153 static unsigned hash(const PackedPtr<StringImpl>& key)
154 {
155 return hash(key.get());
156 }
157
158 static bool equal(const PackedPtr<StringImpl>& a, const PackedPtr<StringImpl>& b)
159 {
160 return equal(a.get(), b.get());
161 }
162
163 static unsigned hash(const String& key)
164 {
165 return hash(key.impl());
166 }
167 static unsigned hash(const AtomString& key)
168 {
169 return hash(key.impl());
170 }
171 static bool equal(const String& a, const String& b)
172 {
173 return equal(a.impl(), b.impl());
174 }
175 static bool equal(const AtomString& a, const AtomString& b)
176 {
177 // FIXME: Is the "a == b" here a helpful optimization?
178 // It makes all cases where the strings are not identical slightly slower.
179 return a == b || equal(a.impl(), b.impl());
180 }
181
182 static constexpr bool safeToCompareToEmptyOrDeleted = false;
183 };

185 // This hash can be used in cases where the key is a hash of a string, but we don't
186 // want to store the string. It's not really specific to string hashing, but all our
187 // current uses of it are for strings.
188 struct AlreadyHashed : IntHash<unsigned> {
189 static unsigned hash(unsigned key) { return key; }
190
191 // To use a hash value as a key for a hash table, we need to eliminate the
192 // "deleted" value, which is negative one. That could be done by changing
193 // the string hash function to never generate negative one, but this works
194 // and is still relatively efficient.
195 static unsigned avoidDeletedValue(unsigned hash)
196 {
197 ASSERT(hash);
198 unsigned newHash = hash | (!(hash + 1) << 31);
199 ASSERT(newHash);
200 ASSERT(newHash != 0xFFFFFFFF);
201 return newHash;
202 }
203 };
