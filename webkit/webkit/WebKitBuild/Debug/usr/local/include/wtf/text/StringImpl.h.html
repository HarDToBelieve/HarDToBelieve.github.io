343 class StaticStringImpl : private StringImplShape {
344 WTF_MAKE_NONCOPYABLE(StaticStringImpl);
345 public:
346 // Used to construct static strings, which have an special refCount that can never hit zero.
347 // This means that the static string will never be destroyed, which is important because
348 // static strings will be shared across threads & ref-counted in a non-threadsafe manner.
349 //
350 // In order to make StaticStringImpl thread safe, we also need to ensure that the rest of
351 // the fields are never mutated by threads. We have this guarantee because:
352 //
353 // 1. m_length is only set on construction and never mutated thereafter.
354 //
355 // 2. m_data8 and m_data16 are only set on construction and never mutated thereafter.
356 // We also know that a StringImpl never changes from 8 bit to 16 bit because there
357 // is no way to set/clear the s_hashFlag8BitBuffer flag other than at construction.
358 //
359 // 3. m_hashAndFlags will not be mutated by different threads because:
360 //
361 // a. StaticStringImpl's constructor sets the s_hashFlagDidReportCost flag to ensure
362 // that StringImpl::cost() returns early.
363 // This means StaticStringImpl costs are not counted. But since there should only
364 // be a finite set of StaticStringImpls, their cost can be aggregated into a single
365 // system cost if needed.
366 // b. setIsAtom() is never called on a StaticStringImpl.
367 // setIsAtom() asserts !isStatic().
368 // c. setHash() is never called on a StaticStringImpl.
369 // StaticStringImpl's constructor sets the hash on construction.
370 // StringImpl::hash() only sets a new hash iff !hasHash().
371 // Additionally, StringImpl::setHash() asserts hasHash() and !isStatic().
372
373 template<unsigned characterCount> constexpr StaticStringImpl(const char (&characters)[characterCount], StringKind = StringNormal);
374 template<unsigned characterCount> constexpr StaticStringImpl(const char16_t (&characters)[characterCount], StringKind = StringNormal);
375 operator StringImpl&();
376 };

722template<typename CharacterType1, typename CharacterType2> inline int codePointCompare(const CharacterType1* characters1, unsigned length1, const CharacterType2* characters2, unsigned length2)
723{
724 unsigned commonLength = std::min(length1, length2);
725
726 unsigned position = 0;
727 while (position < commonLength && *characters1 == *characters2) {
728 ++characters1;
729 ++characters2;
730 ++position;
731 }
732
733 if (position < commonLength)
734 return (characters1[0] > characters2[0]) ? 1 : -1;
735
736 if (length1 == length2)
737 return 0;
738 return (length1 > length2) ? 1 : -1;
739}

741inline int codePointCompare(const StringImpl* string1, const StringImpl* string2)
742{
743 // FIXME: Should null strings compare as less than empty strings rather than equal to them?
744 if (!string1)
745 return (string2 && string2->length()) ? -1 : 0;
746 if (!string2)
747 return string1->length() ? 1 : 0;
748
749 bool string1Is8Bit = string1->is8Bit();
750 bool string2Is8Bit = string2->is8Bit();
751 if (string1Is8Bit) {
752 if (string2Is8Bit)
753 return codePointCompare(string1->characters8(), string1->length(), string2->characters8(), string2->length());
754 return codePointCompare(string1->characters8(), string1->length(), string2->characters16(), string2->length());
755 }
756 if (string2Is8Bit)
757 return codePointCompare(string1->characters16(), string1->length(), string2->characters8(), string2->length());
758 return codePointCompare(string1->characters16(), string1->length(), string2->characters16(), string2->length());
759}

960ALWAYS_INLINE Ref<StringImpl> StringImpl::createSubstringSharingImpl(StringImpl& rep, unsigned offset, unsigned length)
961{
962 ASSERT(length <= rep.length());
963
964 if (!length)
965 return *empty();
966
967 // Coyping the thing would save more memory sometimes, largely due to the size of pointer.
968 size_t substringSize = allocationSize<StringImpl*>(1);
969 if (rep.is8Bit()) {
970 if (substringSize >= allocationSize<LChar>(length))
971 return create(rep.m_data8 + offset, length);
972 } else {
973 if (substringSize >= allocationSize<UChar>(length))
974 return create(rep.m_data16 + offset, length);
975 }
976
977 auto* ownerRep = ((rep.bufferOwnership() == BufferSubstring) ? rep.substringBuffer() : &rep);
978
979 // We allocate a buffer that contains both the StringImpl struct as well as the pointer to the owner string.
980 auto* stringImpl = static_cast<StringImpl*>(StringImplMalloc::malloc(substringSize));
981 if (rep.is8Bit())
982 return adoptRef(*new (NotNull, stringImpl) StringImpl(rep.m_data8 + offset, length, *ownerRep));

1031inline size_t StringImpl::cost() const
1032{
1033 // For substrings, return the cost of the base string.
1034 if (bufferOwnership() == BufferSubstring)
1035 return substringBuffer()->cost();
1036
1037 // Note: we must not alter the m_hashAndFlags field in instances of StaticStringImpl.
1038 // We ensure this by pre-setting the s_hashFlagDidReportCost bit in all instances of
1039 // StaticStringImpl. As a result, StaticStringImpl instances will always return a cost of
1040 // 0 here and avoid modifying m_hashAndFlags.
1041 if (m_hashAndFlags & s_hashFlagDidReportCost)
1042 return 0;
1043
1044 m_hashAndFlags |= s_hashFlagDidReportCost;
1045 size_t result = m_length;
1046 if (!is8Bit())
1047 result <<= 1;
1048 return result;
1049}

1052{
1053 if (isStatic())
1054 return 0;
1055
1056 if (bufferOwnership() == BufferSubstring)
1057 return divideRoundedUp(substringBuffer()->costDuringGC(), refCount());
1058
1059 size_t result = m_length;
1060 if (!is8Bit())
1061 result <<= 1;
1062 return divideRoundedUp(result, refCount());

1075inline void StringImpl::setHash(unsigned hash) const
1076{
1077 // The high bits of 'hash' are always empty, but we prefer to store our flags
1078 // in the low bits because it makes them slightly more efficient to access.
1079 // So, we shift left and right when setting and getting our hash code.
1080
1081 ASSERT(!hasHash());
1082 ASSERT(!isStatic());
1083 // Multiple clients assume that StringHasher is the canonical string hash function.
1084 ASSERT(hash == (is8Bit() ? StringHasher::computeHashAndMaskTop8Bits(m_data8, m_length) : StringHasher::computeHashAndMaskTop8Bits(m_data16, m_length)));
1085 ASSERT(!(hash & (s_flagMask << (8 * sizeof(hash) - s_flagCount)))); // Verify that enough high bits are empty.
1086
1087 hash <<= s_flagCount;
1088 ASSERT(!(hash & m_hashAndFlags)); // Verify that enough low bits are empty after shift.
1089 ASSERT(hash); // Verify that 0 is a valid sentinel hash value.
1090
1091 m_hashAndFlags |= hash; // Store hash with flags in low bits.
1092}

1275template<typename CharacterType, typename Predicate> ALWAYS_INLINE Ref<StringImpl> StringImpl::removeCharactersImpl(const CharacterType* characters, const Predicate& findMatch)
1276{
1277 auto* from = characters;
1278 auto* fromEnd = from + m_length;
1279
1280 // Assume the common case will not remove any characters
1281 while (from != fromEnd && !findMatch(*from))
1282 ++from;
1283 if (from == fromEnd)
1284 return *this;
1285
1286 StringBuffer<CharacterType> data(m_length);
1287 auto* to = data.characters();
1288 unsigned outc = from - characters;
1289
1290 if (outc)
1291 copyCharacters(to, characters, outc);
1292
1293 do {
1294 while (from != fromEnd && findMatch(*from))
1295 ++from;
1296 while (from != fromEnd && !findMatch(*from))
1297 to[outc++] = *from++;
1298 } while (from != fromEnd);
1299
1300 data.shrink(outc);
1301
1302 return adopt(WTFMove(data));
1303}
