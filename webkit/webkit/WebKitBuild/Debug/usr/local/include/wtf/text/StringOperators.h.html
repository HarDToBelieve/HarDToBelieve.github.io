28template<typename StringType1, typename StringType2> class StringAppend {
29public:
30 StringAppend(StringType1 string1, StringType2 string2)
31 : m_string1 { string1 }
32 , m_string2 { string2 }
33 {
34 }
35
36 operator String() const
37 {
38 String result = tryMakeString(m_string1, m_string2);
39 if (!result)
40 CRASH();
41 return result;
42 }
43
44 operator AtomString() const
45 {
46 return operator String();
47 }
48
49 bool is8Bit()
50 {
51 StringTypeAdapter<StringType1> adapter1(m_string1);
52 StringTypeAdapter<StringType2> adapter2(m_string2);
53 return adapter1.is8Bit() && adapter2.is8Bit();
54 }
55
56 void writeTo(LChar* destination)
57 {
58 ASSERT(is8Bit());
59 StringTypeAdapter<StringType1> adapter1(m_string1);
60 StringTypeAdapter<StringType2> adapter2(m_string2);
61 adapter1.writeTo(destination);
62 adapter2.writeTo(destination + adapter1.length());
63 }
64
65 void writeTo(UChar* destination)
66 {
67 StringTypeAdapter<StringType1> adapter1(m_string1);
68 StringTypeAdapter<StringType2> adapter2(m_string2);
69 adapter1.writeTo(destination);
70 adapter2.writeTo(destination + adapter1.length());
71 }
72
73 unsigned length()
74 {
75 StringTypeAdapter<StringType1> adapter1(m_string1);
76 StringTypeAdapter<StringType2> adapter2(m_string2);
77 return adapter1.length() + adapter2.length();
78 }
79
80private:
81 StringType1 m_string1;
82 StringType2 m_string2;
83};
84
85template<typename StringType1, typename StringType2>
86class StringTypeAdapter<StringAppend<StringType1, StringType2>> {
87public:
88 StringTypeAdapter<StringAppend<StringType1, StringType2>>(StringAppend<StringType1, StringType2>& buffer)
89 : m_buffer { buffer }
90 {
91 }
92
93 unsigned length() const { return m_buffer.length(); }
94 bool is8Bit() const { return m_buffer.is8Bit(); }
95 template<typename CharacterType> void writeTo(CharacterType* destination) const { m_buffer.writeTo(destination); }
96
97private:
98 StringAppend<StringType1, StringType2>& m_buffer;
99};
