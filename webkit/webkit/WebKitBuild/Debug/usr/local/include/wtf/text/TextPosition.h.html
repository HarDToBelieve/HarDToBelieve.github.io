25#pragma once
26
27#include <wtf/text/OrdinalNumber.h>
28
29namespace WTF {
30
31// TextPosition structure specifies coordinates within an text resource. It is used mostly
32// for saving script source position.
33class TextPosition {
34 WTF_MAKE_FAST_ALLOCATED;
35public:
36 TextPosition(OrdinalNumber line, OrdinalNumber column)
37 : m_line(line)
38 , m_column(column)
39 {
40 }
41
42 TextPosition() { }
43 bool operator==(const TextPosition& other) const { return m_line == other.m_line && m_column == other.m_column; }
44 bool operator!=(const TextPosition& other) const { return !(*this == other); }
45
46 // A value with line value less than a minimum; used as an impossible position.
47 static TextPosition belowRangePosition() { return TextPosition(OrdinalNumber::beforeFirst(), OrdinalNumber::beforeFirst()); }
48
49 OrdinalNumber m_line;
50 OrdinalNumber m_column;
51};
52
53}
