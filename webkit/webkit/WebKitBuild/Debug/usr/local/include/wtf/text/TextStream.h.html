36class TextStream {
37 WTF_MAKE_FAST_ALLOCATED;
38public:
39 struct FormatNumberRespectingIntegers {
40 WTF_MAKE_STRUCT_FAST_ALLOCATED;
41 FormatNumberRespectingIntegers(double number)
42 : value(number) { }
43
44 double value;

47 enum class Formatting : uint8_t {
48 SVGStyleRect = 1 << 0, // "at (0,0) size 10x10"
49 NumberRespectingIntegers = 1 << 1,
50 LayoutUnitsAsIntegers = 1 << 2,
51 };
52
53 enum class LineMode { SingleLine, MultipleLine };
54 TextStream(LineMode lineMode = LineMode::MultipleLine, OptionSet<Formatting> formattingFlags = { })
55 : m_formattingFlags(formattingFlags)
56 , m_multiLineMode(lineMode == LineMode::MultipleLine)
57 {
58 }

88 template<typename T>
89 void dumpProperty(const String& name, const T& value)
90 {
91 TextStream& ts = *this;
92 ts.startGroup();
93 ts << name << " " << value;
94 ts.endGroup();
95 }

116 struct Repeat {
117 WTF_MAKE_STRUCT_FAST_ALLOCATED;
118 Repeat(unsigned inWidth, char inCharacter)
119 : width(inWidth), character(inCharacter)
120 { }
121 unsigned width { 0 };
122 char character { ' ' };
123 };
124
125 TextStream& operator<<(const Repeat& repeated)
126 {
127 for (unsigned i = 0; i < repeated.width; ++i)
128 m_text.append(repeated.character);
129
130 return *this;
131 }

133 class IndentScope {
134 public:
135 IndentScope(TextStream& ts, int amount = 1)
136 : m_stream(ts)
137 , m_amount(amount)
138 {
139 m_stream.increaseIndent(m_amount);
140 }
141 ~IndentScope()
142 {
143 m_stream.decreaseIndent(m_amount);
144 }
145
146 private:
147 TextStream& m_stream;
148 int m_amount;
149 };
150
151 class GroupScope {
152 public:
153 GroupScope(TextStream& ts)
154 : m_stream(ts)
155 {
156 m_stream.startGroup();
157 }
158 ~GroupScope()
159 {
160 m_stream.endGroup();
161 }
162
163 private:
164 TextStream& m_stream;
165 };
166
167private:
168 StringBuilder m_text;
169 int m_indent { 0 };
170 OptionSet<Formatting> m_formattingFlags;
171 bool m_multiLineMode { true };
172};

174inline TextStream& indent(TextStream& ts)
175{
176 ts.writeIndent();
177 return ts;

180template<typename T>
181struct ValueOrNull {
182 explicit ValueOrNull(T* inValue)
183 : value(inValue)
184 { }
185 T* value;
186};
187
188template<typename T>
189TextStream& operator<<(TextStream& ts, ValueOrNull<T> item)
190{
191 if (item.value)
192 ts << *item.value;
193 else
194 ts << "null";
195 return ts;
196}
197
198template<typename Item>
199TextStream& operator<<(TextStream& ts, const std::optional<Item>& item)
200{
201 if (item)
202 return ts << item.value();
203
204 return ts << "nullopt";
205}
206
207template<typename T, typename Traits>
208TextStream& operator<<(TextStream& ts, const Markable<T, Traits>& item)
209{
210 if (item)
211 return ts << item.value();
212
213 return ts << "unset";
214}

216template<typename ItemType, size_t inlineCapacity>
217TextStream& operator<<(TextStream& ts, const Vector<ItemType, inlineCapacity>& vector)
218{
219 ts << "[";
220
221 unsigned size = vector.size();
222 for (unsigned i = 0; i < size; ++i) {
223 ts << vector[i];
224 if (i < size - 1)
225 ts << ", ";
226 }
227
228 return ts << "]";
229}
230
231template<typename T>
232TextStream& operator<<(TextStream& ts, const WeakPtr<T>& item)
233{
234 if (item)
235 return ts << *item;
236
237 return ts << "null";
238}

240template<typename KeyArg, typename MappedArg, typename HashArg, typename KeyTraitsArg, typename MappedTraitsArg>
241TextStream& operator<<(TextStream& ts, const HashMap<KeyArg, MappedArg, HashArg, KeyTraitsArg, MappedTraitsArg>& map)
242{
243 ts << "{";
244
245 bool first = true;
246 for (const auto& keyValuePair : map) {
247 ts << keyValuePair.key << ": " << keyValuePair.value;
248 if (!first)
249 ts << ", ";
250 first = false;
251 }
252
253 return ts << "}";
254}
255
256template<typename ValueArg, typename HashArg, typename TraitsArg>
257TextStream& operator<<(TextStream& ts, const HashSet<ValueArg, HashArg, TraitsArg>& set)
258{
259 ts << "[";
260
261 bool first = true;
262 for (const auto& item : set) {
263 ts << item;
264 if (!first)
265 ts << ", ";
266 first = false;
267 }
268
269 return ts << "]";
270}
271
272template<typename Option>
273TextStream& operator<<(TextStream& ts, const OptionSet<Option>& options)
274{
275 ts << "[";
276 bool needComma = false;
277 for (auto option : options) {
278 if (needComma)
279 ts << ", ";
280 needComma = true;
281 ts << option;
282 }
283 return ts << "]";
284}
