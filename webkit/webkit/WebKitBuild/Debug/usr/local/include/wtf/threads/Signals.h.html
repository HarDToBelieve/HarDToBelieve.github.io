42// Note that SIGUSR1 is used in Pthread-based ports except for Darwin to suspend and resume threads.
43enum class Signal {
44 // Usr will always chain to any non-default handler install before us. Since there is no way to know
45 // if a signal was intended exclusively for us.
46 Usr,
47
48 // These signals will only chain if we don't have a handler that can process them. If there is nothing
49 // to chain to we restore the default handler and crash.
50#if !OS(DARWIN)
51 Abort,
52#endif
53 FloatingPoint,
54 Breakpoint, // Be VERY careful with this, installing a handler for this will break lldb/gdb.
55 IllegalInstruction,
56 AccessFault, // For posix this is both SIGSEGV and SIGBUS
57 NumberOfSignals = AccessFault + 2, // AccessFault is really two signals.
58 Unknown = NumberOfSignals
59};
60
61inline std::tuple<int, std::optional<int>> toSystemSignal(Signal signal)
62{
63 switch (signal) {
64 case Signal::AccessFault: return std::make_tuple(SIGSEGV, SIGBUS);
65 case Signal::IllegalInstruction: return std::make_tuple(SIGILL, std::nullopt);
66 case Signal::Usr: return std::make_tuple(SIGILL, std::nullopt);
67 case Signal::FloatingPoint: return std::make_tuple(SIGFPE, std::nullopt);
68 case Signal::Breakpoint: return std::make_tuple(SIGTRAP, std::nullopt);
69#if !OS(DARWIN)
70 case Signal::Abort: return std::make_tuple(SIGABRT, std::nullopt);
71#endif
72 default: break;
73 }
74 RELEASE_ASSERT_NOT_REACHED();
75}
76
77inline Signal fromSystemSignal(int signal)
78{
79 switch (signal) {
80 case SIGSEGV: return Signal::AccessFault;
81 case SIGBUS: return Signal::AccessFault;
82 case SIGFPE: return Signal::FloatingPoint;
83 case SIGTRAP: return Signal::Breakpoint;
84 case SIGILL: return Signal::IllegalInstruction;
85 case SIGUSR2: return Signal::Usr;
86#if !OS(DARWIN)
87 case SIGABRT: return Signal::Abort;
88#endif
89 default: return Signal::Unknown;
90 }
91}

93enum class SignalAction {
94 Handled,
95 NotHandled,
96 ForceDefault
97};

99struct SigInfo {
100 void* faultingAddress { 0 };
101};
102
103using SignalHandler = Function<SignalAction(Signal, SigInfo&, PlatformRegisters&)>;
104using SignalHandlerMemory = std::aligned_storage<sizeof(SignalHandler), std::alignment_of<SignalHandler>::value>::type;
105
106struct SignalHandlers {
107 static void initialize();
108
109 void add(Signal, SignalHandler&&);
110 template<typename Func>
111 void forEachHandler(Signal, const Func&) const;
112
113 static constexpr size_t numberOfSignals = static_cast<size_t>(Signal::NumberOfSignals);
114 static constexpr size_t maxNumberOfHandlers = 4;
115
116 static_assert(numberOfSignals < std::numeric_limits<uint8_t>::max());
117
118#if HAVE(MACH_EXCEPTIONS)
119 mach_port_t exceptionPort;
120 exception_mask_t addedExceptions;
121 bool useMach;
122#else
123 static constexpr bool useMach = false;
124#endif
125 uint8_t numberOfHandlers[numberOfSignals];
126 SignalHandlerMemory handlers[numberOfSignals][maxNumberOfHandlers];
127 struct sigaction oldActions[numberOfSignals];
128};
129
130// Call this method whenever you want to add a signal handler. This function needs to be called
131// before g_wtfConfig is frozen. After the g_wtfConfig is frozen, no additional signal handlers may
132// be installed. Any attempt to do so will trigger a crash.
133// Note: Your signal handler will be called every time the handler for the desired signal is called.
134// Thus it is your responsibility to discern if the signal fired was yours.
135// These functions are a one way street i.e. once installed, a signal handler cannot be uninstalled
136// and once commited they can't be turned off.
137WTF_EXPORT_PRIVATE void addSignalHandler(Signal, SignalHandler&&);
138WTF_EXPORT_PRIVATE void activateSignalHandlersFor(Signal);
