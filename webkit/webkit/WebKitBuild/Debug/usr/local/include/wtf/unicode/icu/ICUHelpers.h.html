35constexpr bool needsToGrowToProduceBuffer(UErrorCode);
36constexpr bool needsToGrowToProduceCString(UErrorCode);
37
38// Use this to call a function from ICU that has the following properties:
39// - Takes a buffer pointer and capacity.
40// - Returns the length of the buffer needed.
41// - Takes a UErrorCode* as its last argument, returning the status, including U_BUFFER_OVERFLOW_ERROR.
42// Pass the arguments, but pass a Vector in place of the buffer pointer and capacity, and don't pass a UErrorCode*.
43// This will call the function, once or twice as needed, resizing the buffer as needed.
44//
45// Example:
46//
47// Vector<UChar, 32> buffer;
48// auto status = callBufferProducingFunction(ucal_getDefaultTimeZone, buffer);
49//
50template<typename FunctionType, typename ...ArgumentTypes> UErrorCode callBufferProducingFunction(const FunctionType&, ArgumentTypes&&...);
51
52// Implementations of the functions declared above.
53
54constexpr bool needsToGrowToProduceBuffer(UErrorCode errorCode)
55{
56 return errorCode == U_BUFFER_OVERFLOW_ERROR;
57}
58
59constexpr bool needsToGrowToProduceCString(UErrorCode errorCode)
60{
61 return needsToGrowToProduceBuffer(errorCode) || errorCode == U_STRING_NOT_TERMINATED_WARNING;
62}

64namespace CallBufferProducingFunction {
65
66template<typename CharacterType, size_t inlineCapacity, typename ...ArgumentTypes> auto& findVector(Vector<CharacterType, inlineCapacity>& buffer, ArgumentTypes&&...)
67{
68 return buffer;
69}
70
71template<typename FirstArgumentType, typename ...ArgumentTypes> auto& findVector(FirstArgumentType&&, ArgumentTypes&&... arguments)
72{
73 return findVector(std::forward<ArgumentTypes>(arguments)...);
74}
75
76constexpr std::tuple<> argumentTuple() { return { }; }
77
78template<typename FirstArgumentType, typename ...OtherArgumentTypes> auto argumentTuple(FirstArgumentType&&, OtherArgumentTypes&&...);
79
80template<typename CharacterType, size_t inlineCapacity, typename ...OtherArgumentTypes> auto argumentTuple(Vector<CharacterType, inlineCapacity>& buffer, OtherArgumentTypes&&... otherArguments)
81{
82 return tuple_cat(std::make_tuple(buffer.data(), buffer.size()), argumentTuple(std::forward<OtherArgumentTypes>(otherArguments)...));
83}
84
85template<typename FirstArgumentType, typename ...OtherArgumentTypes> auto argumentTuple(FirstArgumentType&& firstArgument, OtherArgumentTypes&&... otherArguments)
86{
87 // This technique of building a tuple and passing it twice does not work well for complex types, so assert this is a relatively simple one.
88 static_assert(std::is_trivial_v<std::remove_reference_t<FirstArgumentType>>);
89 return tuple_cat(std::make_tuple(firstArgument), argumentTuple(std::forward<OtherArgumentTypes>(otherArguments)...));
90}
91
92}
93
94template<typename FunctionType, typename ...ArgumentTypes> UErrorCode callBufferProducingFunction(const FunctionType& function, ArgumentTypes&&... arguments)
95{
96 auto& buffer = CallBufferProducingFunction::findVector(std::forward<ArgumentTypes>(arguments)...);
97 buffer.grow(buffer.capacity());
98 auto status = U_ZERO_ERROR;
99 auto resultLength = apply(function, CallBufferProducingFunction::argumentTuple(std::forward<ArgumentTypes>(arguments)..., &status));
100 if (U_SUCCESS(status))
101 buffer.shrink(resultLength);
102 else if (needsToGrowToProduceBuffer(status)) {
103 status = U_ZERO_ERROR;
104 buffer.grow(resultLength);
105 apply(function, CallBufferProducingFunction::argumentTuple(std::forward<ArgumentTypes>(arguments)..., &status));
106 ASSERT(U_SUCCESS(status));
107 }
108 return status;
109}

111template<auto deleteFunction>
112struct ICUDeleter {
113 void operator()(typename FunctionTraits<decltype(deleteFunction)>::template ArgumentType<0> pointer)
114 {
115 if (pointer)
116 deleteFunction(pointer);
117 }
118};
